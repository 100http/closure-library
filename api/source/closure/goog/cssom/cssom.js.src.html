77/**
78 * Returns the CSSRules from a styleSheet.
79 * Worth noting here is that IE and FF differ in terms of what they will return.
80 * Firefox will return styleSheet.cssRules, which includes ImportRules and
81 * anything which implements the CSSRules interface. IE returns simply a list of
82 * CSSRules.
83 * @param {CSSStyleSheet} styleSheet The CSSStyleSheet.
84 * @throws {Error} If we cannot access the rules on a stylesheet object - this
85 * can happen if a stylesheet object's rules are accessed before the rules
86 * have been downloaded and parsed and are "ready".
87 * @return {CSSRuleList} An array of CSSRules or null.
88 */
89goog.cssom.getCssRulesFromStyleSheet = function(styleSheet) {
90 var cssRuleList = null;
91 try {
92 // Select cssRules unless it isn't present. For pre-IE9 IE, use the rules
93 // collection instead.
94 // It's important to be consistent in using only the W3C or IE apis on
95 // IE9+ where both are present to ensure that there is no indexing
96 // mismatches - the collections are subtly different in what the include or
97 // exclude which can lead to one collection being longer than the other
98 // depending on the page's construction.
99 cssRuleList = styleSheet.cssRules /* W3C */ || styleSheet.rules /* IE */;
100 } catch (e) {
101 // This can happen if we try to access the CSSOM before it's "ready".
102 if (e.code == 15) {
103 // Firefox throws an NS_ERROR_DOM_INVALID_ACCESS_ERR error if a stylesheet
104 // is read before it has been fully parsed. Let the caller know which
105 // stylesheet failed.
106 e.styleSheet = styleSheet;
107 throw e;
108 }
109 }
110 return cssRuleList;
111};

114/**
115 * Gets all CSSStyleSheet objects starting from some CSSStyleSheet. Note that we
116 * want to return the sheets in the order of the cascade, therefore if we
117 * encounter an import, we will splice that CSSStyleSheet object in front of
118 * the CSSStyleSheet that contains it in the returned array of CSSStyleSheets.
119 * @param {(CSSStyleSheet|StyleSheetList)=} opt_styleSheet A CSSStyleSheet.
120 * @param {boolean=} opt_includeDisabled If true, includes disabled stylesheets,
121 * defaults to false.
122 * @return {!Array<CSSStyleSheet>} A list of CSSStyleSheet objects.
123 */
124goog.cssom.getAllCssStyleSheets = function(opt_styleSheet,
125 opt_includeDisabled) {
126 var styleSheetsOutput = [];
127 var styleSheet = opt_styleSheet || document.styleSheets;
128 var includeDisabled = goog.isDef(opt_includeDisabled) ? opt_includeDisabled :
129 false;
130
131 // Imports need to go first.
132 if (styleSheet.imports && styleSheet.imports.length) {
133 for (var i = 0, n = styleSheet.imports.length; i < n; i++) {
134 goog.array.extend(styleSheetsOutput,
135 goog.cssom.getAllCssStyleSheets(styleSheet.imports[i]));
136 }
137
138 } else if (styleSheet.length) {
139 // In case we get a StyleSheetList object.
140 // http://dev.w3.org/csswg/cssom/#the-stylesheetlist
141 for (var i = 0, n = styleSheet.length; i < n; i++) {
142 goog.array.extend(styleSheetsOutput,
143 goog.cssom.getAllCssStyleSheets(styleSheet[i]));
144 }
145 } else {
146 // We need to walk through rules in browsers which implement .cssRules
147 // to see if there are styleSheets buried in there.
148 // If we have a CSSStyleSheet within CssRules.
149 var cssRuleList = goog.cssom.getCssRulesFromStyleSheet(
150 /** @type {!CSSStyleSheet} */ (styleSheet));
151 if (cssRuleList && cssRuleList.length) {
152 // Chrome does not evaluate cssRuleList[i] to undefined when i >=n;
153 // so we use a (i < n) check instead of cssRuleList[i] in the loop below
154 // and in other places where we iterate over a rules list.
155 // See issue # 5917 in Chromium.
156 for (var i = 0, n = cssRuleList.length, cssRule; i < n; i++) {
157 cssRule = cssRuleList[i];
158 // There are more stylesheets to get on this object..
159 if (cssRule.styleSheet) {
160 goog.array.extend(styleSheetsOutput,
161 goog.cssom.getAllCssStyleSheets(cssRule.styleSheet));
162 }
163 }
164 }
165 }
166
167 // This is a CSSStyleSheet. (IE uses .rules, W3c and Opera cssRules.)
168 if ((styleSheet.type || styleSheet.rules || styleSheet.cssRules) &&
169 (!styleSheet.disabled || includeDisabled)) {
170 styleSheetsOutput.push(styleSheet);
171 }
172
173 return styleSheetsOutput;
174};

177/**
178 * Gets the cssText from a CSSRule object cross-browserly.
179 * @param {CSSRule} cssRule A CSSRule.
180 * @return {string} cssText The text for the rule, including the selector.
181 */
182goog.cssom.getCssTextFromCssRule = function(cssRule) {
183 var cssText = '';
184
185 if (cssRule.cssText) {
186 // W3C.
187 cssText = cssRule.cssText;
188 } else if (cssRule.style && cssRule.style.cssText && cssRule.selectorText) {
189 // IE: The spacing here is intended to make the result consistent with
190 // FF and Webkit.
191 // We also remove the special properties that we may have added in
192 // getAllCssStyleRules since IE includes those in the cssText.
193 var styleCssText = cssRule.style.cssText.
194 replace(/\s*-closure-parent-stylesheet:\s*\[object\];?\s*/gi, '').
195 replace(/\s*-closure-rule-index:\s*[\d]+;?\s*/gi, '');
196 var thisCssText = cssRule.selectorText + ' { ' + styleCssText + ' }';
197 cssText = thisCssText;
198 }
199
200 return cssText;
201};

204/**
205 * Get the index of the CSSRule in it's CSSStyleSheet.
206 * @param {CSSRule} cssRule A CSSRule.
207 * @param {CSSStyleSheet=} opt_parentStyleSheet A reference to the stylesheet
208 * object this cssRule belongs to.
209 * @throws {Error} When we cannot get the parentStyleSheet.
210 * @return {number} The index of the CSSRule, or -1.
211 */
212goog.cssom.getCssRuleIndexInParentStyleSheet = function(cssRule,
213 opt_parentStyleSheet) {
214 // Look for our special style.ruleIndex property from getAllCss.
215 if (cssRule.style && cssRule.style['-closure-rule-index']) {
216 return cssRule.style['-closure-rule-index'];
217 }
218
219 var parentStyleSheet = opt_parentStyleSheet ||
220 goog.cssom.getParentStyleSheet(cssRule);
221
222 if (!parentStyleSheet) {
223 // We could call getAllCssStyleRules() here to get our special indexes on
224 // the style object, but that seems like it could be wasteful.
225 throw Error('Cannot find a parentStyleSheet.');
226 }
227
228 var cssRuleList = goog.cssom.getCssRulesFromStyleSheet(parentStyleSheet);
229 if (cssRuleList && cssRuleList.length) {
230 for (var i = 0, n = cssRuleList.length, thisCssRule; i < n; i++) {
231 thisCssRule = cssRuleList[i];
232 if (thisCssRule == cssRule) {
233 return i;
234 }
235 }
236 }
237 return -1;
238};

255/**
256 * Replace a cssRule with some cssText for a new rule.
257 * If the cssRule object is not one of objects returned by
258 * getAllCssStyleRules, then you'll need to provide both the styleSheet and
259 * possibly the index, since we can't infer them from the standard cssRule
260 * object in IE. We do some trickery in getAllCssStyleRules to hack this in.
261 * @param {CSSRule} cssRule A CSSRule.
262 * @param {string} cssText The text for the new CSSRule.
263 * @param {CSSStyleSheet=} opt_parentStyleSheet A reference to the stylesheet
264 * object this cssRule belongs to.
265 * @param {number=} opt_index The index of the cssRule in its parentStylesheet.
266 * @throws {Error} If we cannot find a parentStyleSheet.
267 * @throws {Error} If we cannot find a css rule index.
268 */
269goog.cssom.replaceCssRule = function(cssRule, cssText, opt_parentStyleSheet,
270 opt_index) {
271 var parentStyleSheet = opt_parentStyleSheet ||
272 goog.cssom.getParentStyleSheet(cssRule);
273 if (parentStyleSheet) {
274 var index = opt_index >= 0 ? opt_index :
275 goog.cssom.getCssRuleIndexInParentStyleSheet(cssRule, parentStyleSheet);
276 if (index >= 0) {
277 goog.cssom.removeCssRule(parentStyleSheet, index);
278 goog.cssom.addCssRule(parentStyleSheet, cssText, index);
279 } else {
280 throw Error('Cannot proceed without the index of the cssRule.');
281 }
282 } else {
283 throw Error('Cannot proceed without the parentStyleSheet.');
284 }
285};

288/**
289 * Cross browser function to add a CSSRule into a CSSStyleSheet, optionally
290 * at a given index.
291 * @param {CSSStyleSheet} cssStyleSheet The CSSRule's parentStyleSheet.
292 * @param {string} cssText The text for the new CSSRule.
293 * @param {number=} opt_index The index of the cssRule in its parentStylesheet.
294 * @throws {Error} If the css rule text appears to be ill-formatted.
295 * TODO(bowdidge): Inserting at index 0 fails on Firefox 2 and 3 with an
296 * exception warning "Node cannot be inserted at the specified point in
297 * the hierarchy."
298 */
299goog.cssom.addCssRule = function(cssStyleSheet, cssText, opt_index) {
300 var index = opt_index;
301 if (index < 0 || index == undefined) {
302 // If no index specified, insert at the end of the current list
303 // of rules.
304 var rules = goog.cssom.getCssRulesFromStyleSheet(cssStyleSheet);
305 index = rules.length;
306 }
307 if (cssStyleSheet.insertRule) {
308 // W3C (including IE9+).
309 cssStyleSheet.insertRule(cssText, index);
310
311 } else {
312 // IE, pre 9: We have to parse the cssRule text to get the selector
313 // separated from the style text.
314 // aka Everything that isn't a colon, followed by a colon, then
315 // the rest is the style part.
316 var matches = /^([^\{]+)\{([^\{]+)\}/.exec(cssText);
317 if (matches.length == 3) {
318 var selector = matches[1];
319 var style = matches[2];
320 cssStyleSheet.addRule(selector, style, index);
321 } else {
322 throw Error('Your CSSRule appears to be ill-formatted.');
323 }
324 }
325};

345/**
346 * Appends a DOM node to HEAD containing the css text that's passed in.
347 * @param {string} cssText CSS to add to the end of the document.
348 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper user for
349 * document interactions.
350 * @return {!Element} The newly created STYLE element.
351 */
352goog.cssom.addCssText = function(cssText, opt_domHelper) {
353 var document = opt_domHelper ? opt_domHelper.getDocument() :
354 goog.dom.getDocument();
355 var cssNode = document.createElement(goog.dom.TagName.STYLE);
356 cssNode.type = 'text/css';
357 var head = document.getElementsByTagName(goog.dom.TagName.HEAD)[0];
358 head.appendChild(cssNode);
359 if (cssNode.styleSheet) {
360 // IE.
361 cssNode.styleSheet.cssText = cssText;
362 } else {
363 // W3C.
364 var cssTextNode = document.createTextNode(cssText);
365 cssNode.appendChild(cssTextNode);
366 }
367 return cssNode;
368};

371/**
372 * Cross browser method to get the filename from the StyleSheet's href.
373 * Explorer only returns the filename in the href, while other agents return
374 * the full path.
375 * @param {!StyleSheet} styleSheet Any valid StyleSheet object with an href.
376 * @throws {Error} When there's no href property found.
377 * @return {?string} filename The filename, or null if not an external
378 * styleSheet.
379 */
380goog.cssom.getFileNameFromStyleSheet = function(styleSheet) {
381 var href = styleSheet.href;
382
383 // Another IE/FF difference. IE returns an empty string, while FF and others
384 // return null for CSSStyleSheets not from an external file.
385 if (!href) {
386 return null;
387 }
388
389 // We need the regexp to ensure we get the filename minus any query params.
390 var matches = /([^\/\?]+)[^\/]*$/.exec(href);
391 var filename = matches[1];
392 return filename;
393};

396/**
397 * Recursively gets all CSS text or rules.
398 * @param {CSSStyleSheet|StyleSheetList} styleSheet The CSSStyleSheet.
399 * @param {boolean} isTextOutput If true, output is cssText, otherwise cssRules.
400 * @return {string|!Array<CSSRule>} cssText or cssRules.
401 * @private
402 */
403goog.cssom.getAllCss_ = function(styleSheet, isTextOutput) {
404 var cssOut = [];
405 var styleSheets = goog.cssom.getAllCssStyleSheets(styleSheet);
406
407 for (var i = 0; styleSheet = styleSheets[i]; i++) {
408 var cssRuleList = goog.cssom.getCssRulesFromStyleSheet(styleSheet);
409
410 if (cssRuleList && cssRuleList.length) {
411
412 // We're going to track cssRule index if we want rule output.
413 if (!isTextOutput) {
414 var ruleIndex = 0;
415 }
416
417 for (var j = 0, n = cssRuleList.length, cssRule; j < n; j++) {
418 cssRule = cssRuleList[j];
419 // Gets cssText output, ignoring CSSImportRules.
420 if (isTextOutput && !cssRule.href) {
421 var res = goog.cssom.getCssTextFromCssRule(cssRule);
422 cssOut.push(res);
423
424 } else if (!cssRule.href) {
425 // Gets cssRules output, ignoring CSSImportRules.
426 if (cssRule.style) {
427 // This is a fun little hack to get parentStyleSheet into the rule
428 // object for IE since it failed to implement rule.parentStyleSheet.
429 // We can later read this property when doing things like hunting
430 // for indexes in order to delete a given CSSRule.
431 // Unfortunately we have to use the style object to store these
432 // pieces of info since the rule object is read-only.
433 if (!cssRule.parentStyleSheet) {
434 cssRule.style['-closure-parent-stylesheet'] = styleSheet;
435 }
436
437 // This is a hack to help with possible removal of the rule later,
438 // where we just append the rule's index in its parentStyleSheet
439 // onto the style object as a property.
440 // Unfortunately we have to use the style object to store these
441 // pieces of info since the rule object is read-only.
442 cssRule.style['-closure-rule-index'] = ruleIndex;
443 }
444 cssOut.push(cssRule);
445 }
446
447 if (!isTextOutput) {
448 ruleIndex++;
449 }
450 }
451 }
452 }
453 return isTextOutput ? cssOut.join(' ') : cssOut;
454};
