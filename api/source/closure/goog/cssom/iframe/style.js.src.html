16/**
17 * @fileoverview Provides utility routines for copying modified
18 * {@code CSSRule} objects from the parent document into iframes so that any
19 * content in the iframe will be styled as if it was inline in the parent
20 * document.
21 *
22 * <p>
23 * For example, you might have this CSS rule:
24 *
25 * #content .highlighted { background-color: yellow; }
26 *
27 * And this DOM structure:
28 *
29 * <div id="content">
30 * <iframe />
31 * </div>
32 *
33 * Then inside the iframe you have:
34 *
35 * <body>
36 * <div class="highlighted">
37 * </body>
38 *
39 * If you copied the CSS rule directly into the iframe, it wouldn't match the
40 * .highlighted div. So we rewrite the original stylesheets based on the
41 * context where the iframe is going to be inserted. In this case the CSS
42 * selector would be rewritten to:
43 *
44 * body .highlighted { background-color: yellow; }
45 * </p>
46 *
47 */
48

105/**
106 * Class representing a CSS rule set. A rule set is something like this:
107 * h1, h2 { font-family: Arial; color: red; }
108 * @constructor
109 * @private
110 */
111goog.cssom.iframe.style.CssRuleSet_ = function() {
112 /**
113 * Text of the declarations inside the rule set.
114 * For example: 'font-family: Arial; color: red;'
115 * @type {string}
116 */
117 this.declarationText = '';
118
119 /**
120 * Array of CssSelector objects, one for each selector.
121 * Example: [h1, h2]
122 * @type {Array<goog.cssom.iframe.style.CssSelector_>}
123 */
124 this.selectors = [];
125};
126

128/**
129 * Initializes the rule set from a {@code CSSRule}.
130 *
131 * @param {CSSRule} cssRule The {@code CSSRule} to initialize from.
132 * @return {boolean} True if initialization succeeded. We only support
133 * {@code CSSStyleRule} and {@code CSSFontFaceRule} objects.
134 */
135goog.cssom.iframe.style.CssRuleSet_.prototype.initializeFromCssRule =
136 function(cssRule) {
137 var ruleStyle = cssRule.style; // Cache object for performance.
138 if (!ruleStyle) {
139 return false;
140 }
141 var selector;
142 var declarations = '';
143 if (ruleStyle &&
144 (selector = cssRule.selectorText) &&
145 (declarations = ruleStyle.cssText)) {
146 // IE get confused about cssText context if a stylesheet uses the
147 // mid-pass hack, and it ends up with an open comment (/*) but no
148 // closing comment. This will effectively comment out large parts
149 // of generated stylesheets later. This errs on the safe side by
150 // always tacking on an empty comment to force comments to be closed
151 // We used to check for a troublesome open comment using a regular
152 // expression, but it's faster not to check and always do this.
153 if (goog.userAgent.IE) {
154 declarations += '/* */';
155 }
156 } else if (cssRule.cssText) {
157 var cssSelectorMatch = /([^\{]+)\{/;
158 var endTagMatch = /\}[^\}]*$/g;
159 // cssRule.cssText contains both selector and declarations:
160 // parse them out.
161 selector = cssSelectorMatch.exec(cssRule.cssText)[1];
162 // Remove selector, {, and trailing }.
163 declarations = cssRule.cssText.replace(cssSelectorMatch, '').replace(
164 endTagMatch, '');
165 }
166 if (selector) {
167 this.setSelectorsFromString(selector);
168 this.declarationText = declarations;
169 return true;
170 }
171 return false;
172};

175/**
176 * Parses a selectors string (which may contain multiple comma-delimited
177 * selectors) and loads the results into this.selectors.
178 * @param {string} selectorsString String containing selectors.
179 */
180goog.cssom.iframe.style.CssRuleSet_.prototype.setSelectorsFromString =
181 function(selectorsString) {
182 this.selectors = [];
183 var selectors = selectorsString.split(/,\s*/gm);
184 for (var i = 0; i < selectors.length; i++) {
185 var selector = selectors[i];
186 if (selector.length > 0) {
187 this.selectors.push(new goog.cssom.iframe.style.CssSelector_(selector));
188 }
189 }
190};

206/**
207 * Set the declaration text with properties from a given object.
208 * @param {Object} sourceObject Object whose properties and values should
209 * be used to generate the declaration text.
210 * @param {boolean=} opt_important Whether !important should be added to each
211 * declaration.
212 */
213goog.cssom.iframe.style.CssRuleSet_.prototype.setDeclarationTextFromObject =
214 function(sourceObject, opt_important) {
215 var stringParts = [];
216 // TODO(user): for ... in is costly in IE6 (extra garbage collection).
217 for (var prop in sourceObject) {
218 var value = sourceObject[prop];
219 if (value) {
220 stringParts.push(prop,
221 ':',
222 value, (opt_important ? ' !important' : ''),
223 ';');
224 }
225 }
226 this.declarationText = stringParts.join('');
227};

230/**
231 * Serializes this CssRuleSet_ into an array as a series of strings.
232 * The array can then be join()-ed to get a string representation
233 * of this ruleset.
234 * @param {Array<string>} array The array to which to append strings.
235 */
236goog.cssom.iframe.style.CssRuleSet_.prototype.writeToArray = function(array) {
237 var selectorCount = this.selectors.length;
238 var matchesAnchorTag = false;
239 for (var i = 0; i < selectorCount; i++) {
240 var selectorParts = this.selectors[i].parts;
241 var partCount = selectorParts.length;
242 for (var j = 0; j < partCount; j++) {
243 array.push(selectorParts[j].inputString_,
244 goog.cssom.iframe.style.SELECTOR_PART_DELIMITER_);
245 }
246 if (i < (selectorCount - 1)) {
247 array.push(goog.cssom.iframe.style.SELECTOR_DELIMITER_);
248 }
249 if (goog.userAgent.GECKO &&
250 !goog.userAgent.isVersionOrHigher('1.9a')) {
251 // In Gecko pre-1.9 (Firefox 2 and lower) we need to add !important
252 // to rulesets that match "A" tags, otherwise Gecko's built-in
253 // stylesheet will take precedence when designMode is on.
254 matchesAnchorTag = matchesAnchorTag ||
255 goog.cssom.iframe.style.selectorPartAnchorRegex_.test(
256 selectorParts[partCount - 1].inputString_);
257 }
258 }
259 var declarationText = this.declarationText;
260 if (matchesAnchorTag) {
261 declarationText = goog.cssom.iframe.style.makeColorRuleImportant_(
262 declarationText);
263 }
264 array.push(goog.cssom.iframe.style.DECLARATION_START_DELIMITER_,
265 declarationText,
266 goog.cssom.iframe.style.DECLARATION_END_DELIMITER_);
267};

322/**
323 * Parses a selector string into individual parts.
324 * @param {string} selectorString A string containing a CSS selector.
325 * @private
326 */
327goog.cssom.iframe.style.CssSelector_.prototype.setPartsFromString_ =
328 function(selectorString) {
329 var parts = [];
330 var selectorPartStrings = selectorString.split(/\s+/gm);
331 for (var i = 0; i < selectorPartStrings.length; i++) {
332 if (!selectorPartStrings[i]) {
333 continue; // Skip empty strings.
334 }
335 var part = new goog.cssom.iframe.style.CssSelectorPart_(
336 selectorPartStrings[i]);
337 parts.push(part);
338 }
339 this.parts = parts;

343/**
344 * Tests to see what part of a DOM element hierarchy would be matched by
345 * this selector, and returns the indexes of the matching element and matching
346 * selector part.
347 * <p>
348 * For example, given this hierarchy:
349 * document > html > body > div.content > div.sidebar > p
350 * and this CSS selector:
351 * body div.sidebar h1
352 * This would return {elementIndex: 4, selectorPartIndex: 1},
353 * indicating that the element at index 4 matched
354 * the css selector at index 1.
355 * </p>
356 * @param {goog.cssom.iframe.style.NodeAncestry_} elementAncestry Object
357 * representing an element and its ancestors.
358 * @return {Object} Object with the properties elementIndex and
359 * selectorPartIndex, or null if there was no match.
360 */
361goog.cssom.iframe.style.CssSelector_.prototype.matchElementAncestry =
362 function(elementAncestry) {
363
364 var ancestryUid = elementAncestry.uid;
365 if (this.ancestryMatchCache_[ancestryUid]) {
366 return this.ancestryMatchCache_[ancestryUid];
367 }
368
369 // Walk through the selector parts and see how far down the element hierarchy
370 // we can go while matching the selector parts.
371 var elementIndex = 0;
372 var match = null;
373 var selectorPart = null;
374 var lastSelectorPart = null;
375 var ancestorNodes = elementAncestry.nodes;
376 var ancestorNodeCount = ancestorNodes.length;
377
378 for (var i = 0; i <= this.parts.length; i++) {
379 selectorPart = this.parts[i];
380 while (elementIndex < ancestorNodeCount) {
381 var currentElementInfo = ancestorNodes[elementIndex];
382 if (selectorPart &&
383 selectorPart.testElement(currentElementInfo)) {
384 match = {
385 elementIndex: elementIndex,
386 selectorPartIndex: i
387 };
388 elementIndex++;
389 break;
390 } else if (lastSelectorPart &&
391 lastSelectorPart.testElement(currentElementInfo)) {
392 match = {
393 elementIndex: elementIndex,
394 selectorPartIndex: i - 1
395 };
396 }
397 elementIndex++;
398 }
399 lastSelectorPart = selectorPart;
400 }
401 this.ancestryMatchCache_[ancestryUid] = match;
402 return match;
403};

407/**
408 * Represents one part of a CSS Selector. For example in the selector
409 * 'body #foo .bar', body, #foo, and .bar would be considered selector parts.
410 * In the official CSS spec these are called "simple selectors".
411 * @param {string} selectorPartString A string containing the selector part
412 * in css format.
413 * @constructor
414 * @private
415 */
416goog.cssom.iframe.style.CssSelectorPart_ = function(selectorPartString) {
417 // Only one CssSelectorPart instance should exist for a given string.
418 var cacheEntry = goog.cssom.iframe.style.CssSelectorPart_.instances_[
419 selectorPartString];
420 if (cacheEntry) {
421 return cacheEntry;
422 }
423
424 // Optimization to avoid the more-expensive lookahead.
425 var identifiers;
426 if (selectorPartString.match(/[#\.]/)) {
427 // Lookahead regexp, won't work on IE 5.0.
428 identifiers = selectorPartString.split(/(?=[#\.])/);
429 } else {
430 identifiers = [selectorPartString];
431 }
432 var properties = {};
433 for (var i = 0; i < identifiers.length; i++) {
434 var identifier = identifiers[i];
435 if (identifier.charAt(0) == '.') {
436 properties.className = identifier.substring(1, identifier.length);
437 } else if (identifier.charAt(0) == '#') {
438 properties.id = identifier.substring(1, identifier.length);
439 } else {
440 properties.tagName = identifier.toUpperCase();
441 }
442 }
443 this.inputString_ = selectorPartString;
444 this.matchProperties_ = properties;
445 this.testedElements_ = {};
446 goog.cssom.iframe.style.CssSelectorPart_.instances_[selectorPartString] =
447 this;
448};

451/**
452 * Cache of existing CssSelectorPart_ instances.
453 * @type {Object}
454 * @private
455 */
456goog.cssom.iframe.style.CssSelectorPart_.instances_ = {};

459/**
460 * Test whether an element matches this selector part, considered in isolation.
461 * @param {Object} elementInfo Element properties to test.
462 * @return {boolean} Whether the element matched.
463 */
464goog.cssom.iframe.style.CssSelectorPart_.prototype.testElement =
465 function(elementInfo) {
466
467 var elementUid = elementInfo.uid;
468 var cachedMatch = this.testedElements_[elementUid];
469 if (typeof cachedMatch != 'undefined') {
470 return cachedMatch;
471 }
472
473 var matchProperties = this.matchProperties_;
474 var testTag = matchProperties.tagName;
475 var testClass = matchProperties.className;
476 var testId = matchProperties.id;
477
478 var matched = true;
479 if (testTag && testTag != '*' && testTag != elementInfo.nodeName) {
480 matched = false;
481 } else if (testId && testId != elementInfo.id) {
482 matched = false;
483 } else if (testClass &&
484 !elementInfo.classNames[testClass]) {
485 matched = false;
486 }
487
488 this.testedElements_[elementUid] = matched;
489 return matched;
490};

494/**
495 * Represents an element and all its parent/ancestor nodes.
496 * This class exists as an optimization so we run tests on an element
497 * hierarchy multiple times without walking the dom each time.
498 * @param {Element} el The DOM element whose ancestry should be stored.
499 * @constructor
500 * @private
501 */
502goog.cssom.iframe.style.NodeAncestry_ = function(el) {
503 var node = el;
504 var nodeUid = goog.getUid(node);
505
506 // Return an existing object from the cache if one exits for this node.
507 var ancestry = goog.cssom.iframe.style.NodeAncestry_.instances_[nodeUid];
508 if (ancestry) {
509 return ancestry;
510 }
511
512 var nodes = [];
513 do {
514 var nodeInfo = {
515 id: node.id,
516 nodeName: node.nodeName
517 };
518 nodeInfo.uid = goog.getUid(nodeInfo);
519 var className = node.className;
520 var classNamesLookup = {};
521 if (className) {
522 var classNames = goog.dom.classlist.get(goog.asserts.assertElement(node));
523 for (var i = 0; i < classNames.length; i++) {
524 classNamesLookup[classNames[i]] = 1;
525 }
526 }
527 nodeInfo.classNames = classNamesLookup;
528 nodes.unshift(nodeInfo);
529 } while (node = node.parentNode);
530
531 /**
532 * Array of nodes in order of hierarchy from the top of the document
533 * to the node passed to the constructor
534 * @type {Array<Node>}
535 */
536 this.nodes = nodes;
537
538 this.uid = goog.getUid(this);
539 goog.cssom.iframe.style.NodeAncestry_.instances_[nodeUid] = this;
540};

560/**
561 * Inspects a document and returns all active rule sets
562 * @param {Document} doc The document from which to read CSS rules.
563 * @return {!Array<goog.cssom.iframe.style.CssRuleSet_>} An array of CssRuleSet
564 * objects representing all the active rule sets in the document.
565 * @private
566 */
567goog.cssom.iframe.style.getRuleSetsFromDocument_ = function(doc) {
568 var ruleSets = [];
569 var styleSheets = goog.cssom.getAllCssStyleSheets(doc.styleSheets);
570 for (var i = 0, styleSheet; styleSheet = styleSheets[i]; i++) {
571 var domRuleSets = goog.cssom.getCssRulesFromStyleSheet(styleSheet);
572 if (domRuleSets && domRuleSets.length) {
573 for (var j = 0, n = domRuleSets.length; j < n; j++) {
574 var ruleSet = new goog.cssom.iframe.style.CssRuleSet_();
575 if (ruleSet.initializeFromCssRule(domRuleSets[j])) {
576 ruleSets.push(ruleSet);
577 }
578 }
579 }
580 }
581 return ruleSets;
582};

696/**
697 * Reads the current css rules from element's document, and returns them
698 * rewriting selectors so that any rules that formerly applied to element will
699 * be applied to doc.body. This makes it possible to replace a block in a page
700 * with an iframe and preserve the css styling of the contents.
701 *
702 * @param {Element} element The element for which context should be calculated.
703 * @param {boolean=} opt_forceRuleSetCacheUpdate Flag to force the internal
704 * cache of rulesets to refresh itself before we read the same.
705 * @param {boolean=} opt_copyBackgroundContext Flag indicating that if the
706 * {@code element} has a transparent background, background rules
707 * from the nearest ancestor element(s) that have background-color
708 * and/or background-image set should be copied.
709 * @return {string} String containing all CSS rules present in the original
710 * document, with modified selectors.
711 * @see goog.cssom.iframe.style.getBackgroundContext.
712 */
713goog.cssom.iframe.style.getElementContext = function(
714 element,
715 opt_forceRuleSetCacheUpdate,
716 opt_copyBackgroundContext) {
717 var sourceDocument = element.ownerDocument;
718 if (opt_forceRuleSetCacheUpdate) {
719 goog.cssom.iframe.style.ruleSetCache_.loadRuleSetsForDocument(
720 sourceDocument);
721 }
722 var ruleSets = goog.cssom.iframe.style.ruleSetCache_.
723 getRuleSetsForDocument(sourceDocument);
724
725 var elementAncestry = new goog.cssom.iframe.style.NodeAncestry_(element);
726 var bodySelectorPart = new goog.cssom.iframe.style.CssSelectorPart_('body');
727
728 for (var i = 0; i < ruleSets.length; i++) {
729 var ruleSet = ruleSets[i];
730 var selectors = ruleSet.selectors;
731 // Cache selectors.length since we may be adding rules in the loop.
732 var ruleCount = selectors.length;
733 for (var j = 0; j < ruleCount; j++) {
734 var selector = selectors[j];
735 // Test whether all or part of this selector would match
736 // this element or one of its ancestors
737 var match = selector.matchElementAncestry(elementAncestry);
738 if (match) {
739 var ruleIndex = match.selectorPartIndex;
740 var selectorParts = selector.parts;
741 var lastSelectorPartIndex = selectorParts.length - 1;
742 var selectorCopy;
743 if (match.elementIndex == elementAncestry.nodes.length - 1 ||
744 ruleIndex < lastSelectorPartIndex) {
745 // Either the first part(s) of the selector matched this element,
746 // or the first part(s) of the selector matched a parent element
747 // and there are more parts of the selector that could target
748 // children of this element.
749 // So we inject a new selector, replacing the part that matched this
750 // element with 'body' so it will continue to match.
751 var selectorPartsCopy = selectorParts.concat();
752 selectorPartsCopy.splice(0,
753 ruleIndex + 1,
754 bodySelectorPart);
755 selectorCopy = new goog.cssom.iframe.style.CssSelector_();
756 selectorCopy.parts = selectorPartsCopy;
757 selectors.push(selectorCopy);
758 } else if (ruleIndex > 0 && ruleIndex == lastSelectorPartIndex) {
759 // The rule didn't match this element, but the entire rule did
760 // match an ancestor element. In this case we want to copy
761 // just the last part of the rule, to give it a chance to be applied
762 // to additional matching elements inside this element.
763 // Example DOM structure: body > div.funky > ul > li#editme
764 // Example CSS selector: .funky ul
765 // New CSS selector: body ul
766 selectorCopy = new goog.cssom.iframe.style.CssSelector_();
767 selectorCopy.parts = [
768 bodySelectorPart,
769 selectorParts[lastSelectorPartIndex]
770 ];
771 selectors.push(selectorCopy);
772 }
773 }
774 }
775 }
776
777 // Insert a new ruleset, setting the current inheritable styles of this
778 // element as the defaults for everything under in the frame.
779 var defaultPropertiesRuleSet = new goog.cssom.iframe.style.CssRuleSet_();
780 var computedStyle = goog.cssom.iframe.style.getComputedStyleObject_(element);
781
782 // Copy inheritable styles so they are applied to everything under HTML.
783 var htmlSelector = new goog.cssom.iframe.style.CssSelector_();
784 htmlSelector.parts = [new goog.cssom.iframe.style.CssSelectorPart_('html')];
785 defaultPropertiesRuleSet.selectors = [htmlSelector];
786 var defaultProperties = {};
787 for (var i = 0, prop;
788 prop = goog.cssom.iframe.style.inheritedProperties_[i];
789 i++) {
790 defaultProperties[prop] = computedStyle[goog.string.toCamelCase(prop)];
791 }
792 defaultPropertiesRuleSet.setDeclarationTextFromObject(defaultProperties);
793 ruleSets.push(defaultPropertiesRuleSet);
794
795 var bodyRuleSet = new goog.cssom.iframe.style.CssRuleSet_();
796 var bodySelector = new goog.cssom.iframe.style.CssSelector_();
797 bodySelector.parts = [new goog.cssom.iframe.style.CssSelectorPart_('body')];
798 // Core set of sane property values for BODY, to prevent copied
799 // styles from completely breaking the display.
800 var bodyProperties = {
801 position: 'relative',
802 top: '0',
803 left: '0',
804 right: 'auto', // Override any existing right value so 'left' works.
805 display: 'block',
806 visibility: 'visible'
807 };
808 // Text formatting property values, to keep text nodes directly under BODY
809 // looking right.
810 for (i = 0; prop = goog.cssom.iframe.style.textProperties_[i]; i++) {
811 bodyProperties[prop] = computedStyle[goog.string.toCamelCase(prop)];
812 }
813 if (opt_copyBackgroundContext &&
814 goog.cssom.iframe.style.isTransparentValue_(
815 computedStyle['backgroundColor'])) {
816 // opt_useAncestorBackgroundRules means that, if the original element
817 // has a transparent backgorund, background properties rules should be
818 // added to explicitly make the body have the same background appearance
819 // as in the original element, even if its positioned somewhere else
820 // in the DOM.
821 var bgProperties =
822 goog.cssom.iframe.style.getBackgroundContext(element);
823 bodyProperties['background-color'] = bgProperties['backgroundColor'];
824 var elementBgImage = computedStyle['backgroundImage'];
825 if (!elementBgImage || elementBgImage == 'none') {
826 bodyProperties['background-image'] = bgProperties['backgroundImage'];
827 bodyProperties['background-repeat'] = bgProperties['backgroundRepeat'];
828 bodyProperties['background-position'] =
829 bgProperties['backgroundPosition'];
830 }
831 }
832
833 bodyRuleSet.setDeclarationTextFromObject(bodyProperties, true);
834 bodyRuleSet.selectors = [bodySelector];
835 ruleSets.push(bodyRuleSet);
836
837 // Write outputTextParts to doc.
838 var ruleSetStrings = [];
839 ruleCount = ruleSets.length;
840 for (i = 0; i < ruleCount; i++) {
841 ruleSets[i].writeToArray(ruleSetStrings);
842 }
843 return ruleSetStrings.join('');
844};

906/**
907 * Generates a set of CSS properties that can be used to make another
908 * element's background look like the background of a given element.
909 * This is useful when you want to copy the CSS context of an element,
910 * but the element's background is transparent. In the original context
911 * you would see the ancestor's backround color/image showing through,
912 * but in the new context there might be a something different underneath.
913 * Note that this assumes the element you're copying context from has a
914 * fairly standard positioning/layout - it assumes that when the element
915 * has a transparent background what you're going to see through it is its
916 * ancestors.
917 * @param {Element} element The element from which to copy background styles.
918 * @return {!Object} Object containing background* properties.
919 */
920goog.cssom.iframe.style.getBackgroundContext = function(element) {
921 var propertyValues = {
922 'backgroundImage': 'none'
923 };
924 var ancestor = element;
925 var currentIframeWindow;
926 // Walk up the DOM tree to find the ancestor nodes whose backgrounds
927 // may be visible underneath this element. Background-image and
928 // background-color don't have to come from the same node, but as soon
929 // an element with background-color is found there's no need to continue
930 // because backgrounds farther up the chain won't be visible.
931 // (This implementation is not sophisticated enough to handle opacity,
932 // or multple layered partially-transparent background images.)
933 while ((ancestor = /** @type {!Element} */ (ancestor.parentNode)) &&
934 ancestor.nodeType == goog.dom.NodeType.ELEMENT) {
935 var computedStyle =
936 goog.cssom.iframe.style.getComputedStyleObject_(ancestor);
937 // Copy background color if a non-transparent value is found.
938 var backgroundColorValue = computedStyle['backgroundColor'];
939 if (!goog.cssom.iframe.style.isTransparentValue_(backgroundColorValue)) {
940 propertyValues['backgroundColor'] = backgroundColorValue;
941 }
942 // If a background image value is found, copy background-image,
943 // background-repeat, and background-position.
944 if (computedStyle['backgroundImage'] &&
945 computedStyle['backgroundImage'] != 'none') {
946 propertyValues['backgroundImage'] = computedStyle['backgroundImage'];
947 propertyValues['backgroundRepeat'] = computedStyle['backgroundRepeat'];
948 // Calculate the offset between the original element and the element
949 // providing the background image, so the background position can be
950 // adjusted.
951 var relativePosition;
952 if (currentIframeWindow) {
953 relativePosition = goog.style.getFramedPageOffset(
954 element, currentIframeWindow);
955 var frameElement = currentIframeWindow.frameElement;
956 var iframeRelativePosition = goog.style.getRelativePosition(
957 /** @type {!Element} */ (frameElement), ancestor);
958 var iframeBorders = goog.style.getBorderBox(frameElement);
959 relativePosition.x += iframeRelativePosition.x + iframeBorders.left;
960 relativePosition.y += iframeRelativePosition.y + iframeBorders.top;
961 } else {
962 relativePosition = goog.style.getRelativePosition(element, ancestor);
963 }
964 var backgroundXYValues = goog.cssom.iframe.style.getBackgroundXYValues_(
965 computedStyle);
966 // Parse background-repeat-* values in the form "10px", and adjust them.
967 for (var i = 0; i < 2; i++) {
968 var positionValue = backgroundXYValues[i];
969 var coordinate = i == 0 ? 'X' : 'Y';
970 var positionProperty = 'backgroundPosition' + coordinate;
971 // relative position to its ancestor.
972 var positionValueParts =
973 goog.cssom.iframe.style.valueWithUnitsRegEx_.exec(positionValue);
974 if (positionValueParts) {
975 var value = parseInt(
976 positionValueParts[1] + positionValueParts[2], 10);
977 var units = positionValueParts[3];
978 // This only attempts to handle pixel values for now (plus
979 // '0anything', which is equivalent to 0px).
980 // TODO(user) Convert non-pixel values to pixels when possible.
981 if (value == 0 || units == 'px') {
982 value -= (coordinate == 'X' ?
983 relativePosition.x : relativePosition.y);
984 }
985 positionValue = value + units;
986 }
987 propertyValues[positionProperty] = positionValue;
988 }
989 propertyValues['backgroundPosition'] =
990 propertyValues['backgroundPositionX'] + ' ' +
991 propertyValues['backgroundPositionY'];
992 }
993 if (propertyValues['backgroundColor']) {
994 break;
995 }
996 if (ancestor.tagName == goog.dom.TagName.HTML) {
997 try {
998 currentIframeWindow = goog.dom.getWindow(
999 /** @type {Document} */ (ancestor.parentNode));
1000 // This could theoretically throw a security exception if the parent
1001 // iframe is in a different domain.
1002 ancestor = currentIframeWindow.frameElement;
1003 if (!ancestor) {
1004 // Loop has reached the top level window.
1005 break;
1006 }
1007 } catch (e) {
1008 // We don't have permission to go up to the parent window, stop here.
1009 break;
1010 }
1011 }
1012 }
1013 return propertyValues;
1014};
