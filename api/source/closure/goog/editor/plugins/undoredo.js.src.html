41/**
42 * Encapsulates undo/redo logic using a custom undo stack (i.e. not browser
43 * built-in). Browser built-in undo stacks are too flaky (e.g. IE's gets
44 * clobbered on DOM modifications). Also, this allows interleaving non-editing
45 * commands into the undo stack via the UndoRedoManager.
46 *
47 * @param {goog.editor.plugins.UndoRedoManager=} opt_manager An undo redo
48 * manager to be used by this plugin. If none is provided one is created.
49 * @constructor
50 * @extends {goog.editor.Plugin}
51 */
52goog.editor.plugins.UndoRedo = function(opt_manager) {
53 goog.editor.Plugin.call(this);
54
55 this.setUndoRedoManager(opt_manager ||
56 new goog.editor.plugins.UndoRedoManager());
57
58 // Map of goog.editor.Field hashcode to goog.events.EventHandler
59 this.eventHandlers_ = {};
60
61 this.currentStates_ = {};
62
63 /**
64 * @type {?string}
65 * @private
66 */
67 this.initialFieldChange_ = null;
68
69 /**
70 * A copy of {@code goog.editor.plugins.UndoRedo.restoreState} bound to this,
71 * used by undo-redo state objects to restore the state of an editable field.
72 * @type {Function}
73 * @see goog.editor.plugins.UndoRedo#restoreState
74 * @private
75 */
76 this.boundRestoreState_ = goog.bind(this.restoreState, this);
77};
78goog.inherits(goog.editor.plugins.UndoRedo, goog.editor.Plugin);

229/** @override */
230goog.editor.plugins.UndoRedo.prototype.enable = function(fieldObject) {
231 if (this.isEnabled(fieldObject)) {
232 return;
233 }
234
235 // Don't want pending delayed changes from when undo-redo was disabled
236 // firing after undo-redo is enabled since they might cause undo-redo stack
237 // updates.
238 fieldObject.clearDelayedChange();
239
240 var eventHandler = new goog.events.EventHandler(this);
241
242 // TODO(user): From ojan during a code review:
243 // The beforechange handler is meant to be there so you can grab the cursor
244 // position *before* the change is made as that's where you want the cursor to
245 // be after an undo.
246 //
247 // It kinda looks like updateCurrentState_ doesn't do that correctly right
248 // now, but it really should be fixed to do so. The cursor position stored in
249 // the state should be the cursor position before any changes are made, not
250 // the cursor position when the change finishes.
251 //
252 // It also seems like the if check below is just a bad one. We should do this
253 // for browsers that use mutation events as well even though the beforechange
254 // happens too late...maybe not. I don't know about this.
255 if (!goog.editor.BrowserFeature.USE_MUTATION_EVENTS) {
256 // We don't listen to beforechange in mutation-event browsers because
257 // there we fire beforechange, then syncronously file change. The point
258 // of before change is to capture before the user has changed anything.
259 eventHandler.listen(fieldObject,
260 goog.editor.Field.EventType.BEFORECHANGE, this.handleBeforeChange_);
261 }
262 eventHandler.listen(fieldObject,
263 goog.editor.Field.EventType.DELAYEDCHANGE, this.handleDelayedChange_);
264 eventHandler.listen(fieldObject, goog.editor.Field.EventType.BLUR,
265 this.handleBlur_);
266
267 this.eventHandlers_[fieldObject.getHashCode()] = eventHandler;
268
269 // We want to capture the initial state of a Trogedit field before any
270 // editing has happened. This is necessary so that we can undo the first
271 // change to a field, even if we don't handle beforeChange.
272 this.updateCurrentState_(fieldObject);
273};

276/** @override */
277goog.editor.plugins.UndoRedo.prototype.disable = function(fieldObject) {
278 // Process any pending changes so we don't lose any undo-redo states that we
279 // want prior to disabling undo-redo.
280 fieldObject.clearDelayedChange();
281
282 var eventHandler = this.eventHandlers_[fieldObject.getHashCode()];
283 if (eventHandler) {
284 eventHandler.dispose();
285 delete this.eventHandlers_[fieldObject.getHashCode()];
286 }
287
288 // We delete the current state of the field on disable. When we re-enable
289 // the state will be re-fetched. In most cases the content will be the same,
290 // but this allows us to pick up changes while not editable. That way, when
291 // undoing after starting an editable session, you can always undo to the
292 // state you started in. Given this sequence of events:
293 // Make editable
294 // Type 'anakin'
295 // Make not editable
296 // Set HTML to be 'padme'
297 // Make editable
298 // Type 'dark side'
299 // Undo
300 // Without re-snapshoting current state on enable, the undo would go from
301 // 'dark-side' -> 'anakin', rather than 'dark-side' -> 'padme'. You couldn't
302 // undo the field to the state that existed immediately after it was made
303 // editable for the second time.
304 if (this.currentStates_[fieldObject.getHashCode()]) {
305 delete this.currentStates_[fieldObject.getHashCode()];
306 }
307};

381/**
382 * Restores the state of the editable field.
383 * @param {goog.editor.plugins.UndoRedo.UndoState_} state The state initiating
384 * the restore.
385 * @param {string} content The content to restore.
386 * @param {goog.editor.plugins.UndoRedo.CursorPosition_?} cursorPosition
387 * The cursor position within the content.
388 */
389goog.editor.plugins.UndoRedo.prototype.restoreState = function(
390 state, content, cursorPosition) {
391 // Fire any pending changes to get the current field state up to date and
392 // then stop listening to changes while doing the undo/redo.
393 var fieldObj = this.getFieldObjectForHash(state.fieldHashCode);
394 if (!fieldObj) {
395 return;
396 }
397
398 // Fires any pending changes, and stops the change events. Still want to
399 // dispatch before change, as a change is being made and the change event
400 // will be manually dispatched below after the new content has been restored
401 // (also restarting change events).
402 fieldObj.stopChangeEvents(true, true);
403
404 // To prevent the situation where we stop change events and then an exception
405 // happens before we can restart change events, the following code must be in
406 // a try-finally block.
407 try {
408 fieldObj.dispatchBeforeChange();
409
410 // Restore the state
411 fieldObj.execCommand(goog.editor.Command.CLEAR_LOREM, true);
412
413 // We specifically set the raw innerHTML of the field here as that's what
414 // we get from the field when we save an undo/redo state. There's
415 // no need to clean/unclean the contents in either direction.
416 goog.editor.node.replaceInnerHtml(fieldObj.getElement(), content);
417
418 if (cursorPosition) {
419 cursorPosition.select();
420 }
421
422 var previousFieldObject = this.getCurrentFieldObject();
423 fieldObj.focus();
424
425 // Apps that integrate their undo-redo with Trogedit may be
426 // in a state where there is no previous field object (no field focused at
427 // the time of undo), so check for existence first.
428 if (previousFieldObject &&
429 previousFieldObject.getHashCode() != state.fieldHashCode) {
430 previousFieldObject.execCommand(goog.editor.Command.UPDATE_LOREM);
431 }
432
433 // We need to update currentState_ to reflect the change.
434 this.currentStates_[state.fieldHashCode].setUndoState(
435 content, cursorPosition);
436 } catch (e) {
437 goog.log.error(this.logger, 'Error while restoring undo state', e);
438 } finally {
439 // Clear the delayed change event, set flag so we know not to act on it.
440 this.inProgressUndo_ = state;
441 // Notify the editor that we've changed (fire autosave).
442 // Note that this starts up change events again, so we don't have to
443 // manually do so even though we stopped change events above.
444 fieldObj.dispatchChange();
445 fieldObj.dispatchSelectionChangeEvent();
446 }
447};

450/**
451 * @override
452 */
453goog.editor.plugins.UndoRedo.prototype.handleKeyboardShortcut = function(e, key,
454 isModifierPressed) {
455 if (isModifierPressed) {
456 var command;
457 if (key == 'z') {
458 command = e.shiftKey ? goog.editor.plugins.UndoRedo.COMMAND.REDO :
459 goog.editor.plugins.UndoRedo.COMMAND.UNDO;
460 } else if (key == 'y') {
461 command = goog.editor.plugins.UndoRedo.COMMAND.REDO;
462 }
463
464 if (command) {
465 // In the case where Trogedit shares its undo redo stack with another
466 // application it's possible that an undo or redo will not be for an
467 // goog.editor.Field. In this case we don't want to go through the
468 // goog.editor.Field execCommand flow which stops and restarts events on
469 // the current field. Only Trogedit UndoState's have a fieldHashCode so
470 // use that to distinguish between Trogedit and other states.
471 var state = command == goog.editor.plugins.UndoRedo.COMMAND.UNDO ?
472 this.undoManager_.undoPeek() : this.undoManager_.redoPeek();
473 if (state && state.fieldHashCode) {
474 this.getCurrentFieldObject().execCommand(command);
475 } else {
476 this.execCommand(command);
477 }
478
479 return true;
480 }
481 }
482
483 return false;
484};

499/**
500 * Refreshes the current state of the editable field as maintained by undo-redo,
501 * without adding any undo-redo states to the stack.

504goog.editor.plugins.UndoRedo.prototype.refreshCurrentState = function(

515/**
516 * Before the field changes, we want to save the state.
517 * @param {goog.events.Event} e The event.
518 * @private
519 */
520goog.editor.plugins.UndoRedo.prototype.handleBeforeChange_ = function(e) {
521 if (this.inProgressUndo_) {
522 // We are in between a previous undo and its delayed change event.
523 // Continuing here clobbers the redo stack.
524 // This does mean that if you are trying to undo/redo really quickly, it
525 // will be gated by the speed of delayed change events.
526 return;
527 }
528
529 var fieldObj = /** @type {goog.editor.Field} */ (e.target);
530 var fieldHashCode = fieldObj.getHashCode();
531
532 if (this.initialFieldChange_ != fieldHashCode) {
533 this.initialFieldChange_ = fieldHashCode;
534 this.updateCurrentState_(fieldObj);
535 }
536};
537
538
539/**
540 * After some idle time, we want to save the state.
541 * @param {goog.events.Event} e The event.
542 * @private
543 */
544goog.editor.plugins.UndoRedo.prototype.handleDelayedChange_ = function(e) {
545 // This was undo making a change, don't add it BACK into the history
546 if (this.inProgressUndo_) {
547 // Must clear this.inProgressUndo_ before dispatching event because the
548 // dispatch can cause another, queued undo that should be allowed to go
549 // through.
550 var state = this.inProgressUndo_;
551 this.inProgressUndo_ = null;
552 state.dispatchEvent(goog.editor.plugins.UndoRedoState.ACTION_COMPLETED);
553 return;
554 }
555
556 this.updateCurrentState_(/** @type {goog.editor.Field} */ (e.target));
557};

590/**
591 * Helper method for saving state.
592 * @param {goog.editor.Field} fieldObj The field object.
593 * @private
594 */
595goog.editor.plugins.UndoRedo.prototype.updateCurrentState_ = function(
596 fieldObj) {
597 var fieldHashCode = fieldObj.getHashCode();
598 // We specifically grab the raw innerHTML of the field here as that's what
599 // we would set on the field in the case of an undo/redo operation. There's
600 // no need to clean/unclean the contents in either direction. In the case of
601 // lorem ipsum being used, we want to capture the effective state (empty, no
602 // cursor position) rather than capturing the lorem html.
603 var content, cursorPos;
604 if (fieldObj.queryCommandValue(goog.editor.Command.USING_LOREM)) {
605 content = '';
606 cursorPos = null;
607 } else {
608 content = fieldObj.getElement().innerHTML;
609 cursorPos = this.getCursorPosition_(fieldObj);
610 }
611
612 var currentState = this.currentStates_[fieldHashCode];
613 if (currentState) {
614 // Don't create states if the content hasn't changed (spurious
615 // delayed change). This can happen when lorem is cleared, for example.
616 if (currentState.undoContent_ == content) {
617 return;
618 } else if (content == '' || currentState.undoContent_ == '') {
619 // If lorem ipsum is on we say the contents are the empty string. However,
620 // for an empty text shape with focus, the empty contents might not be
621 // the same, depending on plugins. We want these two empty states to be
622 // considered identical because to the user they are indistinguishable,
623 // so we use fieldObj.getInjectableContents to map between them.
624 // We cannot use getInjectableContents when first creating the undo
625 // content for a field with lorem, because on enable when this is first
626 // called we can't guarantee plugin registration order, so the
627 // injectableContents at that time might not match the final
628 // injectableContents.
629 var emptyContents = fieldObj.getInjectableContents('', {});
630 if (content == emptyContents && currentState.undoContent_ == '' ||
631 currentState.undoContent_ == emptyContents && content == '') {
632 return;
633 }
634 }
635
636 currentState.setRedoState(content, cursorPos);
637 this.undoManager_.addState(currentState);
638 }
639
640 this.currentStates_[fieldHashCode] =
641 new goog.editor.plugins.UndoRedo.UndoState_(fieldHashCode, content,
642 cursorPos, this.boundRestoreState_);
643};

813/**
814 * The standards compliant version keeps a list of childNode offsets.
815 * @param {goog.dom.AbstractRange?} range The range to save.
816 * @private
817 */
818goog.editor.plugins.UndoRedo.CursorPosition_.prototype.initW3C_ = function(
819 range) {
820 this.isValid_ = false;
821
822 // TODO: Check if the range is in the field before trying to save it
823 // for FF 3 contentEditable.
824 if (!range) {
825 return;
826 }
827
828 var anchorNode = range.getAnchorNode();
829 var focusNode = range.getFocusNode();
830 if (!anchorNode || !focusNode) {
831 return;
832 }
833
834 var anchorOffset = range.getAnchorOffset();
835 var anchor = new goog.dom.NodeOffset(anchorNode, this.field_.getElement());
836
837 var focusOffset = range.getFocusOffset();
838 var focus = new goog.dom.NodeOffset(focusNode, this.field_.getElement());
839
840 // Test range direction.
841 if (range.isReversed()) {
842 this.startOffset_ = focus;
843 this.startChildOffset_ = focusOffset;
844 this.endOffset_ = anchor;
845 this.endChildOffset_ = anchorOffset;
846 } else {
847 this.startOffset_ = anchor;
848 this.startChildOffset_ = anchorOffset;
849 this.endOffset_ = focus;
850 this.endChildOffset_ = focusOffset;
851 }
852
853 this.isValid_ = true;
854};

857/**
858 * In IE, we just keep track of the text offset (number of characters).
859 * @param {goog.dom.AbstractRange?} range The range to save.
860 * @private
861 */
862goog.editor.plugins.UndoRedo.CursorPosition_.prototype.initIE_ = function(
863 range) {
864 this.isValid_ = false;
865
866 if (!range) {
867 return;
868 }
869
870 var ieRange = range.getTextRange(0).getBrowserRangeObject();
871
872 if (!goog.dom.contains(this.field_.getElement(), ieRange.parentElement())) {
873 return;
874 }
875
876 // Create a range that encompasses the contentEditable region to serve
877 // as a reference to form ranges below.
878 var contentEditableRange =
879 this.field_.getEditableDomHelper().getDocument().body.createTextRange();
880 contentEditableRange.moveToElementText(this.field_.getElement());
881
882 // startMarker is a range from the start of the contentEditable node to the
883 // start of the current selection.
884 var startMarker = ieRange.duplicate();
885 startMarker.collapse(true);
886 startMarker.setEndPoint('StartToStart', contentEditableRange);
887 this.startOffset_ =
888 goog.editor.plugins.UndoRedo.CursorPosition_.computeEndOffsetIE_(
889 startMarker);
890
891 // endMarker is a range from the start of teh contentEditable node to the
892 // end of the current selection.
893 var endMarker = ieRange.duplicate();
894 endMarker.setEndPoint('StartToStart', contentEditableRange);
895 this.endOffset_ =
896 goog.editor.plugins.UndoRedo.CursorPosition_.computeEndOffsetIE_(
897 endMarker);

899 this.isValid_ = true;

939/**
940 * Get the range that encompases the the cursor position relative to a given
941 * base node.
942 * @param {Element} baseNode The node to get the cursor position relative to.
943 * @return {Range|TextRange|null} The browser range for this position.
944 * @private
945 */
946goog.editor.plugins.UndoRedo.CursorPosition_.prototype.getRange_ =
947 function(baseNode) {
948 if (goog.editor.BrowserFeature.HAS_W3C_RANGES) {
949 var startNode = this.startOffset_.findTargetNode(baseNode);
950 var endNode = this.endOffset_.findTargetNode(baseNode);
951 if (!startNode || !endNode) {
952 return null;
953 }
954
955 // Create range.
956 return /** @type {Range} */ (
957 goog.dom.Range.createFromNodes(startNode, this.startChildOffset_,
958 endNode, this.endChildOffset_).getBrowserRangeObject());
959 }
960
961 // Create a collapsed selection at the start of the contentEditable region,
962 // which the offsets were calculated relative to before. Note that we force
963 // a text range here so we can use moveToElementText.
964 var sel = baseNode.ownerDocument.body.createTextRange();
965 sel.moveToElementText(baseNode);
966 sel.collapse(true);
967 sel.moveEnd('character', this.endOffset_);
968 sel.moveStart('character', this.startOffset_);
969 return sel;
970};

973/**
974 * Compute the number of characters to the end of the range in IE.
975 * @param {TextRange} range The range to compute an offset for.
976 * @return {number} The number of characters to the end of the range.
977 * @private
978 */
979goog.editor.plugins.UndoRedo.CursorPosition_.computeEndOffsetIE_ =
980 function(range) {
981 var testRange = range.duplicate();
982
983 // The number of offset characters is a little off depending on
984 // what type of block elements happen to be between the start of the
985 // textedit and the cursor position. We fudge the offset until the
986 // two ranges match.
987 var text = range.text;
988 var guess = text.length;
989
990 testRange.collapse(true);
991 testRange.moveEnd('character', guess);
992
993 // Adjust the range until the end points match. This doesn't quite
994 // work if we're at the end of the field so we give up after a few
995 // iterations.
996 var diff;
997 var numTries = 10;
998 while (diff = testRange.compareEndPoints('EndToEnd', range)) {
999 guess -= diff;
1000 testRange.moveEnd('character', -diff);
1001 --numTries;
1002 if (0 == numTries) {
1003 break;
1004 }
1005 }
1006 // When we set innerHTML, blank lines become a single space, causing
1007 // the cursor position to be off by one. So we accommodate for blank
1008 // lines.
1009 var offset = 0;
1010 var pos = text.indexOf('\n\r');
1011 while (pos != -1) {
1012 ++offset;
1013 pos = text.indexOf('\n\r', pos + 1);
1014 }
1015 return guess + offset;
1016};
