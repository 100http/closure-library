185/**
186 * Tells the plugin to stop leaking the page's url via the referrer header when
187 * the link text link is clicked. When the user clicks on a link, the
188 * browser makes a request for the link url, passing the url of the current page
189 * in the request headers. If the user wants the current url to be kept secret
190 * (e.g. an unpublished document), the owner of the url that was clicked will
191 * see the secret url in the request headers, and it will no longer be a secret.
192 * Calling this method will not send a referrer header in the request, just as
193 * if the user had opened a blank window and typed the url in themselves.
194 */
195goog.editor.plugins.LinkBubble.prototype.stopReferrerLeaks = function() {
196 // TODO(user): Right now only 2 plugins have this API to stop
197 // referrer leaks. If more plugins need to do this, come up with a way to
198 // enable the functionality in all plugins at once. Same thing for
199 // setBlockOpeningUnsafeSchemes and associated functionality.
200 this.stopReferrerLeaks_ = true;
201};

266/** @override */
267goog.editor.plugins.LinkBubble.prototype.getBubbleTargetFromSelection =
268 function(selectedElement) {
269 var bubbleTarget = goog.dom.getAncestorByTagNameAndClass(selectedElement,
270 goog.dom.TagName.A);
271
272 if (!bubbleTarget) {
273 // See if the selection is touching the right side of a link, and if so,
274 // show a bubble for that link. The check for "touching" is very brittle,
275 // and currently only guarantees that it will pop up a bubble at the
276 // position the cursor is placed at after the link dialog is closed.
277 // NOTE(robbyw): This assumes this method is always called with
278 // selected element = range.getContainerElement(). Right now this is true,
279 // but attempts to re-use this method for other purposes could cause issues.
280 // TODO(robbyw): Refactor this method to also take a range, and use that.
281 var range = this.getFieldObject().getRange();
282 if (range && range.isCollapsed() && range.getStartOffset() == 0) {
283 var startNode = range.getStartNode();
284 var previous = startNode.previousSibling;
285 if (previous && previous.tagName == goog.dom.TagName.A) {
286 bubbleTarget = previous;
287 }
288 }
289 }
290
291 return /** @type {Element} */ (bubbleTarget);
292};

340/** @override */
341goog.editor.plugins.LinkBubble.prototype.createBubbleContents = function(
342 bubbleContainer) {
343 var linkObj = this.getLinkToTextObj_();
344
345 // Create linkTextSpan, show plain text for e-mail address or truncate the
346 // text to <= 48 characters so that property bubbles don't grow too wide and
347 // create a link if URL. Only linkify valid links.
348 // TODO(robbyw): Repalce this color with a CSS class.
349 var color = linkObj.valid ? 'black' : 'red';
350 var shouldOpenUrl = this.shouldOpenUrl(linkObj.linkText);
351 var linkTextSpan;
352 if (goog.editor.Link.isLikelyEmailAddress(linkObj.linkText) ||
353 !linkObj.valid || !shouldOpenUrl) {
354 linkTextSpan = this.dom_.createDom(goog.dom.TagName.SPAN,
355 {
356 id: goog.editor.plugins.LinkBubble.LINK_TEXT_ID_,
357 style: 'color:' + color
358 }, this.dom_.createTextNode(linkObj.linkText));
359 } else {
360 var testMsgSpan = this.dom_.createDom(goog.dom.TagName.SPAN,
361 {id: goog.editor.plugins.LinkBubble.TEST_LINK_SPAN_ID_},
362 this.getTestLinkMessage());
363 linkTextSpan = this.dom_.createDom(goog.dom.TagName.SPAN,
364 {
365 id: goog.editor.plugins.LinkBubble.LINK_TEXT_ID_,
366 style: 'color:' + color
367 }, '');
368 var linkText = goog.string.truncateMiddle(linkObj.linkText, 48);
369 // Actually creates a pseudo-link that can't be right-clicked to open in a
370 // new tab, because that would avoid the logic to stop referrer leaks.
371 this.createLink(goog.editor.plugins.LinkBubble.TEST_LINK_ID_,
372 this.dom_.createTextNode(linkText).data,
373 this.testLink,
374 linkTextSpan);
375 }
376
377 var changeLinkSpan = this.createLinkOption(
378 goog.editor.plugins.LinkBubble.CHANGE_LINK_SPAN_ID_);
379 this.createLink(goog.editor.plugins.LinkBubble.CHANGE_LINK_ID_,
380 goog.editor.plugins.LinkBubble.MSG_LINK_BUBBLE_CHANGE,
381 this.showLinkDialog_, changeLinkSpan);
382
383 // This function is called multiple times - we have to reset the array.
384 this.actionSpans_ = [];
385 for (var i = 0; i < this.extraActions_.length; i++) {
386 var action = this.extraActions_[i];
387 var actionSpan = this.createLinkOption(action.spanId_);
388 this.actionSpans_.push(actionSpan);
389 this.createLink(action.linkId_, action.message_,
390 function() {
391 action.actionFn_(this.getTargetUrl());
392 },
393 actionSpan);
394 }
395
396 var removeLinkSpan = this.createLinkOption(
397 goog.editor.plugins.LinkBubble.DELETE_LINK_SPAN_ID_);
398 this.createLink(goog.editor.plugins.LinkBubble.DELETE_LINK_ID_,
399 goog.editor.plugins.LinkBubble.MSG_LINK_BUBBLE_REMOVE,
400 this.deleteLink_, removeLinkSpan);
401
402 this.onShow();
403
404 var bubbleContents = this.dom_.createDom(goog.dom.TagName.DIV,
405 {id: goog.editor.plugins.LinkBubble.LINK_DIV_ID_},
406 testMsgSpan || '', linkTextSpan, changeLinkSpan);
407
408 for (i = 0; i < this.actionSpans_.length; i++) {
409 bubbleContents.appendChild(this.actionSpans_[i]);
410 }
411 bubbleContents.appendChild(removeLinkSpan);
412
413 goog.dom.appendChild(bubbleContainer, bubbleContents);
414};

417/**
418 * Tests the link by opening it in a new tab/window. Should be used as the
419 * click event handler for the test pseudo-link.
420 * @param {!Event=} opt_event If passed in, the event will be stopped.
421 * @protected
422 */
423goog.editor.plugins.LinkBubble.prototype.testLink = function(opt_event) {
424 goog.window.open(this.getTestLinkAction_(),
425 {
426 'target': '_blank',
427 'noreferrer': this.stopReferrerLeaks_
428 }, this.getFieldObject().getAppWindow());
429 if (opt_event) {
430 opt_event.stopPropagation();
431 opt_event.preventDefault();
432 }

446/**
447 * Gets the text to display for a link, based on the type of link
448 * @return {!Object} Returns an object of the form:
449 * {linkText: displayTextForLinkTarget, valid: ifTheLinkIsValid}.
450 * @private
451 */
452goog.editor.plugins.LinkBubble.prototype.getLinkToTextObj_ = function() {
453 var isError;
454 var targetUrl = this.getTargetUrl();
455
456 if (this.isInvalidUrl(targetUrl)) {
457
458 targetUrl = goog.editor.plugins.LinkBubble.MSG_INVALID_URL_LINK_BUBBLE;
459 isError = true;
460 } else if (goog.editor.Link.isMailto(targetUrl)) {
461 targetUrl = targetUrl.substring(7); // 7 == "mailto:".length
462 }
463
464 return {linkText: targetUrl, valid: !isError};
465};

473goog.editor.plugins.LinkBubble.prototype.showLinkDialog_ = function(e) {
474 // Needed when this occurs due to an ENTER key event, else the newly created
475 // dialog manages to have its OK button pressed, causing it to disappear.
476 e.preventDefault();
477
478 this.getFieldObject().execCommand(goog.editor.Command.MODAL_LINK_EDITOR,
479 new goog.editor.Link(
480 /** @type {HTMLAnchorElement} */ (this.getTargetElement()),
481 false));
482 this.closeBubble();

486/**
487 * Deletes the link associated with the bubble
488 * @private
489 */
490goog.editor.plugins.LinkBubble.prototype.deleteLink_ = function() {
491 this.getFieldObject().dispatchBeforeChange();
492
493 var link = this.getTargetElement();
494 var child = link.lastChild;
495 goog.dom.flattenElement(link);
496 goog.editor.range.placeCursorNextTo(child, false);
497
498 this.closeBubble();
499
500 this.getFieldObject().dispatchChange();
501 this.getFieldObject().focus();
502};

505/**
506 * Sets the proper state for the action links.
507 * @protected
508 * @override
509 */
510goog.editor.plugins.LinkBubble.prototype.onShow = function() {
511 var linkDiv = this.dom_.getElement(
512 goog.editor.plugins.LinkBubble.LINK_DIV_ID_);
513 if (linkDiv) {
514 var testLinkSpan = this.dom_.getElement(
515 goog.editor.plugins.LinkBubble.TEST_LINK_SPAN_ID_);
516 if (testLinkSpan) {
517 var url = this.getTargetUrl();
518 goog.style.setElementShown(testLinkSpan, !goog.editor.Link.isMailto(url));
519 }
520
521 for (var i = 0; i < this.extraActions_.length; i++) {
522 var action = this.extraActions_[i];
523 var actionSpan = this.dom_.getElement(action.spanId_);
524 if (actionSpan) {
525 goog.style.setElementShown(actionSpan, action.toShowFn_(
526 this.getTargetUrl()));
527 }
528 }
529 }

556/**
557 * Determines whether or not a url has a scheme which is safe to open.
558 * Schemes like javascript are unsafe due to the possibility of XSS.
559 * @param {string} url A url.
560 * @return {boolean} Whether the url has a safe scheme.
561 * @private
562 */
563goog.editor.plugins.LinkBubble.prototype.isSafeSchemeToOpen_ =
564 function(url) {
565 var scheme = goog.uri.utils.getScheme(url) || 'http';
566 return goog.array.contains(this.safeToOpenSchemes_, scheme.toLowerCase());
567};
