41/**
42 * Plugin to handle enter keys. This subclass normalizes all browsers to use
43 * the given block tag on enter.
44 * @param {goog.dom.TagName} tag The type of tag to add on enter.
45 * @constructor
46 * @extends {goog.editor.plugins.EnterHandler}
47 */
48goog.editor.plugins.TagOnEnterHandler = function(tag) {
49 this.tag = tag;
50
51 goog.editor.plugins.EnterHandler.call(this);
52};
53goog.inherits(goog.editor.plugins.TagOnEnterHandler,
54 goog.editor.plugins.EnterHandler);

121/** @override */
122goog.editor.plugins.TagOnEnterHandler.prototype.handleDeleteGecko = function(
123 e) {
124 var range = this.getFieldObject().getRange();
125 var container = goog.editor.style.getContainer(
126 range && range.getContainerElement());
127 if (this.getFieldObject().getElement().lastChild == container &&
128 goog.editor.plugins.EnterHandler.isBrElem(container)) {
129 // Don't delete if it's the last node in the field and just has a BR.
130 e.preventDefault();
131 // TODO(user): I think we probably don't need to stopPropagation here
132 e.stopPropagation();
133 } else {
134 // Go ahead with deletion.
135 // Prevent an existing BR immediately following the selection being deleted
136 // from being removed in the keyup stage (as opposed to a BR added by FF
137 // after deletion, which we do remove).
138 this.markBrToNotBeRemoved_(range, false);
139 // Manually delete the selection if it's at a BR.
140 this.deleteBrGecko(e);
141 }

147 e) {
148 if (goog.userAgent.GECKO) {
149 if (e.keyCode == goog.events.KeyCodes.DELETE) {
150 this.removeBrIfNecessary_(false);

183/**
184 * Ensures the current node is wrapped in the tag.
185 * @param {Node} node The node to ensure gets wrapped.
186 * @param {Element} container Element containing the selection.
187 * @return {Element} Element containing the selection, after the wrapping.
188 * @private
189 */
190goog.editor.plugins.TagOnEnterHandler.prototype.ensureNodeIsWrappedW3c_ =
191 function(node, container) {
192 if (container == this.getFieldObject().getElement()) {
193 // If the first block-level ancestor of cursor is the field,
194 // don't split the tree. Find all the text from the cursor
195 // to both block-level elements surrounding it (if they exist)
196 // and split the text into two elements.
197 // This is the IE contentEditable behavior.
198
199 // The easy way to do this is to wrap all the text in an element
200 // and then split the element as if the user had hit enter
201 // in the paragraph
202
203 // However, simply wrapping the text into an element creates problems
204 // if the text was already wrapped using some other element such as an
205 // anchor. For example, wrapping the text of
206 // <a href="">Text</a>
207 // would produce
208 // <a href=""><p>Text</p></a>
209 // which is not what we want. What we really want is
210 // <p><a href="">Text</a></p>
211 // So we need to search for an ancestor of position.node to be wrapped.
212 // We do this by iterating up the hierarchy of postiion.node until we've
213 // reached the node that's just under the container.
214 var isChildOfFn = function(child) {
215 return container == child.parentNode; };
216 var nodeToWrap = goog.dom.getAncestor(node, isChildOfFn, true);
217 container = goog.editor.plugins.TagOnEnterHandler.wrapInContainerW3c_(
218 this.tag, {node: nodeToWrap, offset: 0}, container);
219 }
220 return container;
221};

239/** @override */
240goog.editor.plugins.TagOnEnterHandler.prototype.
241 handleEnterAtCursorGeckoInternal = function(e, wasCollapsed, range) {
242 // We use this because there are a few cases where FF default
243 // implementation doesn't follow IE's:
244 // -Inserts BRs into empty elements instead of NBSP which has nasty
245 // side effects w/ making/deleting selections
246 // -Hitting enter when your cursor is in the field itself. IE will
247 // create two elements. FF just inserts a BR.
248 // -Hitting enter inside an empty list-item doesn't create a block
249 // tag. It just splits the list and puts your cursor in the middle.
250 var li = null;
251 if (wasCollapsed) {
252 // Only break out of lists for collapsed selections.
253 li = goog.dom.getAncestorByTagNameAndClass(
254 range && range.getContainerElement(), goog.dom.TagName.LI);
255 }
256 var isEmptyLi = (li &&
257 li.innerHTML.match(
258 goog.editor.plugins.TagOnEnterHandler.emptyLiRegExp_));
259 var elementAfterCursor = isEmptyLi ?
260 this.breakOutOfEmptyListItemGecko_(li) :
261 this.handleRegularEnterGecko_();
262
263 // Move the cursor in front of "nodeAfterCursor", and make sure it
264 // is visible
265 this.scrollCursorIntoViewGecko_(elementAfterCursor);
266
267 // Fix for http://b/1991234 :
268 if (goog.editor.plugins.EnterHandler.isBrElem(elementAfterCursor)) {
269 // The first element in the new line is a line with just a BR and maybe some
270 // whitespace.
271 // Calling normalize() is needed because there might be empty text nodes
272 // before BR and empty text nodes cause the cursor position bug in Firefox.
273 // See http://b/5220858
274 elementAfterCursor.normalize();
275 var br = elementAfterCursor.getElementsByTagName(goog.dom.TagName.BR)[0];
276 if (br.previousSibling &&
277 br.previousSibling.nodeType == goog.dom.NodeType.TEXT) {
278 // If there is some whitespace before the BR, don't put the selection on
279 // the BR, put it in the text node that's there, otherwise when you type
280 // it will create adjacent text nodes.
281 elementAfterCursor = br.previousSibling;
282 }
283 }
284
285 goog.editor.range.selectNodeStart(elementAfterCursor);
286
287 e.preventDefault();
288 // TODO(user): I think we probably don't need to stopPropagation here
289 e.stopPropagation();
290};

293/**
294 * If The cursor is in an empty LI then break out of the list like in IE
295 * @param {Node} li LI to break out of.
296 * @return {!Element} Element to put the cursor after.
297 * @private
298 */
299goog.editor.plugins.TagOnEnterHandler.prototype.breakOutOfEmptyListItemGecko_ =
300 function(li) {
301 // Do this as follows:
302 // 1. <ul>...<li>&nbsp;</li>...</ul>
303 // 2. <ul id='foo1'>...<li id='foo2'>&nbsp;</li>...</ul>
304 // 3. <ul id='foo1'>...</ul><p id='foo3'>&nbsp;</p><ul id='foo2'>...</ul>
305 // 4. <ul>...</ul><p>&nbsp;</p><ul>...</ul>
306 //
307 // There are a couple caveats to the above. If the UL is contained in
308 // a list, then the new node inserted is an LI, not a P.
309 // For an OL, it's all the same, except the tagname of course.
310 // Finally, it's possible that with the LI at the beginning or the end
311 // of the list that we'll end up with an empty list. So we special case
312 // those cases.
313
314 var listNode = li.parentNode;
315 var grandparent = listNode.parentNode;
316 var inSubList = grandparent.tagName == goog.dom.TagName.OL ||
317 grandparent.tagName == goog.dom.TagName.UL;
318
319 // TODO(robbyw): Should we apply the list or list item styles to the new node?
320 var newNode = goog.dom.getDomHelper(li).createElement(
321 inSubList ? goog.dom.TagName.LI : this.tag);
322
323 if (!li.previousSibling) {
324 goog.dom.insertSiblingBefore(newNode, listNode);
325 } else {
326 if (li.nextSibling) {
327 var listClone = listNode.cloneNode(false);
328 while (li.nextSibling) {
329 listClone.appendChild(li.nextSibling);
330 }
331 goog.dom.insertSiblingAfter(listClone, listNode);
332 }
333 goog.dom.insertSiblingAfter(newNode, listNode);
334 }
335 if (goog.editor.node.isEmpty(listNode)) {
336 goog.dom.removeNode(listNode);
337 }
338 goog.dom.removeNode(li);
339 newNode.innerHTML = '&nbsp;';
340
341 return newNode;
342};

345/**
346 * Wrap the text indicated by "position" in an HTML container of type
347 * "nodeName".
348 * @param {string} nodeName Type of container, e.g. "p" (paragraph).
349 * @param {Object} position The W3C cursor position object
350 * (from getCursorPositionW3c).
351 * @param {Node} container The field containing position.
352 * @return {!Element} The container element that holds the contents from
353 * position.
354 * @private
355 */
356goog.editor.plugins.TagOnEnterHandler.wrapInContainerW3c_ = function(nodeName,
357 position, container) {
358 var start = position.node;
359 while (start.previousSibling &&
360 !goog.editor.style.isContainer(start.previousSibling)) {
361 start = start.previousSibling;
362 }
363
364 var end = position.node;
365 while (end.nextSibling &&
366 !goog.editor.style.isContainer(end.nextSibling)) {
367 end = end.nextSibling;
368 }
369
370 var para = container.ownerDocument.createElement(nodeName);
371 while (start != end) {
372 var newStart = start.nextSibling;
373 goog.dom.appendChild(para, start);
374 start = newStart;
375 }
376 var nextSibling = end.nextSibling;
377 goog.dom.appendChild(para, end);
378 container.insertBefore(para, nextSibling);
379
380 return para;
381};

384/**
385 * When we delete an element, FF inserts a BR. We want to strip that
386 * BR after the fact, but in the case where your cursor is at a character
387 * right before a BR and you delete that character, we don't want to
388 * strip it. So we detect this case on keydown and mark the BR as not needing
389 * removal.
390 * @param {goog.dom.AbstractRange} range The closure range object.
391 * @param {boolean} isBackspace Whether this is handling the backspace key.
392 * @private
393 */
394goog.editor.plugins.TagOnEnterHandler.prototype.markBrToNotBeRemoved_ =
395 function(range, isBackspace) {
396 var focusNode = range.getFocusNode();
397 var focusOffset = range.getFocusOffset();
398 var newEndOffset = isBackspace ? focusOffset : focusOffset + 1;
399
400 if (goog.editor.node.getLength(focusNode) == newEndOffset) {
401 var sibling = focusNode.nextSibling;
402 if (sibling && sibling.tagName == goog.dom.TagName.BR) {
403 this.brToKeep_ = sibling;
404 }
405 }
406};

409/**
410 * If we hit delete/backspace to merge elements, FF inserts a BR.
411 * We want to strip that BR. In markBrToNotBeRemoved, we detect if
412 * there was already a BR there before the delete/backspace so that
413 * we don't accidentally remove a user-inserted BR.
414 * @param {boolean} isBackSpace Whether this is handling the backspace key.
415 * @private
416 */
417goog.editor.plugins.TagOnEnterHandler.prototype.removeBrIfNecessary_ = function(
418 isBackSpace) {
419 var range = this.getFieldObject().getRange();
420 var focusNode = range.getFocusNode();
421 var focusOffset = range.getFocusOffset();
422
423 var sibling;
424 if (isBackSpace && focusNode.data == '') {
425 // nasty hack. sometimes firefox will backspace a paragraph and put
426 // the cursor before the BR. when it does this, the focusNode is
427 // an empty textnode.
428 sibling = focusNode.nextSibling;
429 } else if (isBackSpace && focusOffset == 0) {
430 var node = focusNode;
431 while (node && !node.previousSibling &&
432 node.parentNode != this.getFieldObject().getElement()) {
433 node = node.parentNode;
434 }
435 sibling = node.previousSibling;
436 } else if (focusNode.length == focusOffset) {
437 sibling = focusNode.nextSibling;
438 }
439
440 if (!sibling || sibling.tagName != goog.dom.TagName.BR ||
441 this.brToKeep_ == sibling) {
442 return;
443 }
444
445 goog.dom.removeNode(sibling);
446 if (focusNode.nodeType == goog.dom.NodeType.TEXT) {
447 // Sometimes firefox inserts extra whitespace. Do our best to deal.
448 // This is buggy though.
449 focusNode.data =
450 goog.editor.plugins.TagOnEnterHandler.trimTabsAndLineBreaks_(
451 focusNode.data);
452 // When we strip whitespace, make sure that our cursor is still at
453 // the end of the textnode.
454 goog.dom.Range.createCaret(focusNode,
455 Math.min(focusOffset, focusNode.length)).select();
456 }
457};

472/**
473 * Called in response to a normal enter keystroke. It has the action of
474 * splitting elements.
475 * @return {Element} The node that the cursor should be before.
476 * @private
477 */
478goog.editor.plugins.TagOnEnterHandler.prototype.handleRegularEnterGecko_ =
479 function() {
480 var range = this.getFieldObject().getRange();
481 var container =
482 goog.editor.style.getContainer(range.getContainerElement());
483 var newNode;
484 if (goog.editor.plugins.EnterHandler.isBrElem(container)) {
485 if (container.tagName == goog.dom.TagName.BODY) {
486 // If the field contains only a single BR, this code ensures we don't
487 // try to clone the body tag.
488 container = this.ensureNodeIsWrappedW3c_(
489 container.getElementsByTagName(goog.dom.TagName.BR)[0],
490 container);
491 }
492
493 newNode = container.cloneNode(true);
494 goog.dom.insertSiblingAfter(newNode, container);
495 } else {
496 if (!container.firstChild) {
497 container.innerHTML = '&nbsp;';
498 }
499
500 var position = goog.editor.range.getDeepEndPoint(range, true);
501 container = this.ensureNodeIsWrappedW3c_(position.node, container);
502
503 newNode = goog.editor.plugins.TagOnEnterHandler.splitDomAndAppend_(
504 position.node, position.offset, container);
505
506 // If the left half and right half of the splitted node are anchors then
507 // that means the user pressed enter while the caret was inside
508 // an anchor tag and split it. The left half is the first anchor
509 // found while traversing the right branch of container. The right half
510 // is the first anchor found while traversing the left branch of newNode.
511 var leftAnchor =
512 goog.editor.plugins.TagOnEnterHandler.findAnchorInTraversal_(
513 container);
514 var rightAnchor =
515 goog.editor.plugins.TagOnEnterHandler.findAnchorInTraversal_(
516 newNode, true);
517 if (leftAnchor && rightAnchor &&
518 leftAnchor.tagName == goog.dom.TagName.A &&
519 rightAnchor.tagName == goog.dom.TagName.A) {
520 // If the original anchor (left anchor) is now empty, that means
521 // the user pressed [Enter] at the beginning of the anchor,
522 // in which case we we
523 // want to replace that anchor with its child nodes
524 // Otherwise, we take the second half of the splitted text and break
525 // it out of the anchor.
526 var anchorToRemove = goog.editor.node.isEmpty(leftAnchor, false) ?
527 leftAnchor : rightAnchor;
528 goog.dom.flattenElement(/** @type {!Element} */ (anchorToRemove));
529 }
530 }
531 return /** @type {!Element} */ (newNode);
532};

535/**
536 * Scroll the cursor into view, resulting from splitting the paragraph/adding
537 * a br. It behaves differently than scrollIntoView
538 * @param {Element} element The element immediately following the cursor. Will
539 * be used to determine how to scroll in order to make the cursor visible.
540 * CANNOT be a BR, as they do not have offsetHeight/offsetTop.
541 * @private
542 */
543goog.editor.plugins.TagOnEnterHandler.prototype.scrollCursorIntoViewGecko_ =
544 function(element) {
545 if (!this.getFieldObject().isFixedHeight()) {
546 return; // Only need to scroll fixed height fields.
547 }
548
549 var field = this.getFieldObject().getElement();
550
551 // Get the y position of the element we want to scroll to
552 var elementY = goog.style.getPageOffsetTop(element);
553
554 // Determine the height of that element, since we want the bottom of the
555 // element to be in view.
556 var bottomOfNode = elementY + element.offsetHeight;
557
558 var dom = this.getFieldDomHelper();
559 var win = this.getFieldDomHelper().getWindow();
560 var scrollY = dom.getDocumentScroll().y;
561 var viewportHeight = goog.dom.getViewportSize(win).height;
562
563 // If the botom of the element is outside the viewport, move it into view
564 if (bottomOfNode > viewportHeight + scrollY) {
565 // In standards mode, use the html element and not the body
566 if (field.tagName == goog.dom.TagName.BODY &&
567 goog.editor.node.isStandardsMode(field)) {
568 field = field.parentNode;
569 }
570 field.scrollTop = bottomOfNode - viewportHeight;
571 }
572};

575/**
576 * Splits the DOM tree around the given node and returns the node
577 * containing the second half of the tree. The first half of the tree
578 * is modified, but not removed from the DOM.
579 * @param {Node} positionNode Node to split at.
580 * @param {number} positionOffset Offset into positionNode to split at. If
581 * positionNode is a text node, this offset is an offset in to the text
582 * content of that node. Otherwise, positionOffset is an offset in to
583 * the childNodes array. All elements with child index of positionOffset
584 * or greater will be moved to the second half. If positionNode is an
585 * empty element, the dom will be split at that element, with positionNode
586 * ending up in the second half. positionOffset must be 0 in this case.
587 * @param {Node=} opt_root Node at which to stop splitting the dom (the root
588 * is also split).
589 * @return {!Node} The node containing the second half of the tree.
590 * @private
591 */
592goog.editor.plugins.TagOnEnterHandler.splitDom_ = function(
593 positionNode, positionOffset, opt_root) {
594 if (!opt_root) opt_root = positionNode.ownerDocument.body;
595
596 // Split the node.
597 var textSplit = positionNode.nodeType == goog.dom.NodeType.TEXT;
598 var secondHalfOfSplitNode;
599 if (textSplit) {
600 if (goog.userAgent.IE &&
601 positionOffset == positionNode.nodeValue.length) {
602 // Since splitText fails in IE at the end of a node, we split it manually.
603 secondHalfOfSplitNode = goog.dom.getDomHelper(positionNode).
604 createTextNode('');
605 goog.dom.insertSiblingAfter(secondHalfOfSplitNode, positionNode);
606 } else {
607 secondHalfOfSplitNode = positionNode.splitText(positionOffset);
608 }
609 } else {
610 // Here we ensure positionNode is the last node in the first half of the
611 // resulting tree.
612 if (positionOffset) {
613 // Use offset as an index in to childNodes.
614 positionNode = positionNode.childNodes[positionOffset - 1];
615 } else {
616 // In this case, positionNode would be the last node in the first half
617 // of the tree, but we actually want to move it to the second half.
618 // Therefore we set secondHalfOfSplitNode to the same node.
619 positionNode = secondHalfOfSplitNode = positionNode.firstChild ||
620 positionNode;
621 }
622 }
623
624 // Create second half of the tree.
625 var secondHalf = goog.editor.node.splitDomTreeAt(
626 positionNode, secondHalfOfSplitNode, opt_root);
627
628 if (textSplit) {
629 // Join secondHalfOfSplitNode and its right text siblings together and
630 // then replace leading NonNbspWhiteSpace with a Nbsp. If
631 // secondHalfOfSplitNode has a right sibling that isn't a text node,
632 // then we can leave secondHalfOfSplitNode empty.
633 secondHalfOfSplitNode =
634 goog.editor.plugins.TagOnEnterHandler.joinTextNodes_(
635 secondHalfOfSplitNode, true);
636 goog.editor.plugins.TagOnEnterHandler.replaceWhiteSpaceWithNbsp_(
637 secondHalfOfSplitNode, true, !!secondHalfOfSplitNode.nextSibling);
638
639 // Join positionNode and its left text siblings together and then replace
640 // trailing NonNbspWhiteSpace with a Nbsp.
641 var firstHalf = goog.editor.plugins.TagOnEnterHandler.joinTextNodes_(
642 positionNode, false);
643 goog.editor.plugins.TagOnEnterHandler.replaceWhiteSpaceWithNbsp_(
644 firstHalf, false, false);
645 }
646
647 return secondHalf;
648};

651/**
652 * Splits the DOM tree around the given node and returns the node containing
653 * second half of the tree, which is appended after the old node. The first
654 * half of the tree is modified, but not removed from the DOM.
655 * @param {Node} positionNode Node to split at.
656 * @param {number} positionOffset Offset into positionNode to split at. If
657 * positionNode is a text node, this offset is an offset in to the text
658 * content of that node. Otherwise, positionOffset is an offset in to
659 * the childNodes array. All elements with child index of positionOffset
660 * or greater will be moved to the second half. If positionNode is an
661 * empty element, the dom will be split at that element, with positionNode
662 * ending up in the second half. positionOffset must be 0 in this case.
663 * @param {Node} node Node to split.
664 * @return {!Node} The node containing the second half of the tree.
665 * @private
666 */
667goog.editor.plugins.TagOnEnterHandler.splitDomAndAppend_ = function(
668 positionNode, positionOffset, node) {

676/**
677 * Joins node and its adjacent text nodes together.
678 * @param {Node} node The node to start joining.
679 * @param {boolean} moveForward Determines whether to join left siblings (false)
680 * or right siblings (true).
681 * @return {Node} The joined text node.
682 * @private
683 */
684goog.editor.plugins.TagOnEnterHandler.joinTextNodes_ = function(node,
685 moveForward) {
686 if (node && node.nodeName == '#text') {
687 var nextNodeFn = moveForward ? 'nextSibling' : 'previousSibling';
688 var prevNodeFn = moveForward ? 'previousSibling' : 'nextSibling';
689 var nodeValues = [node.nodeValue];
690 while (node[nextNodeFn] &&
691 node[nextNodeFn].nodeType == goog.dom.NodeType.TEXT) {
692 node = node[nextNodeFn];
693 nodeValues.push(node.nodeValue);
694 goog.dom.removeNode(node[prevNodeFn]);
695 }
696 if (!moveForward) {
697 nodeValues.reverse();
698 }
699 node.nodeValue = nodeValues.join('');
700 }
701 return node;
