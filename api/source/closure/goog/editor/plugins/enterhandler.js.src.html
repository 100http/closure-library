42/**
43 * Plugin to handle enter keys. This does all the crazy to normalize (as much as
44 * is reasonable) what happens when you hit enter. This also handles the
45 * special casing of hitting enter in a blockquote.
46 *
47 * In IE, Webkit, and Opera, the resulting HTML uses one DIV tag per line. In
48 * Firefox, the resulting HTML uses BR tags at the end of each line.
49 *
50 * @constructor
51 * @extends {goog.editor.Plugin}
52 */
53goog.editor.plugins.EnterHandler = function() {
54 goog.editor.Plugin.call(this);
55};
56goog.inherits(goog.editor.plugins.EnterHandler, goog.editor.Plugin);

122goog.editor.plugins.EnterHandler.prototype.handleBackspaceInternal = function(e,

124 var field = this.getFieldObject().getElement();
125 var container = range && range.getStartNode();

127 if (field.firstChild == container && goog.editor.node.isEmpty(container)) {

130 e.stopPropagation();

135/**
136 * Fix paragraphs to be the correct type of node.
137 * @param {goog.events.Event} e The <enter> key event.
138 * @param {boolean} split Whether we already split up a blockquote by
139 * manually inserting elements.
140 * @protected
141 */
142goog.editor.plugins.EnterHandler.prototype.processParagraphTagsInternal =
143 function(e, split) {
144 // Force IE to turn the node we are leaving into a DIV. If we do turn
145 // it into a DIV, the node IE creates in response to ENTER will also be
146 // a DIV. If we don't, it will be a P. We handle that case
147 // in handleKeyUpIE_
148 if (goog.userAgent.IE || goog.userAgent.OPERA) {
149 this.ensureBlockIeOpera(goog.dom.TagName.DIV);
150 } else if (!split && goog.userAgent.WEBKIT) {
151 // WebKit duplicates a blockquote when the user hits enter. Let's cancel
152 // this and insert a BR instead, to make it more consistent with the other
153 // browsers.
154 var range = this.getFieldObject().getRange();
155 if (!range || !goog.editor.plugins.EnterHandler.isDirectlyInBlockquote(
156 range.getContainerElement())) {
157 return;
158 }
159
160 var dh = this.getFieldDomHelper();
161 var br = dh.createElement(goog.dom.TagName.BR);
162 range.insertNode(br, true);
163
164 // If the BR is at the end of a block element, Safari still thinks there is
165 // only one line instead of two, so we need to add another BR in that case.
166 if (goog.editor.node.isBlockTag(br.parentNode) &&
167 !goog.editor.node.skipEmptyTextNodes(br.nextSibling)) {
168 goog.dom.insertSiblingBefore(
169 dh.createElement(goog.dom.TagName.BR), br);
170 }
171
172 goog.editor.range.placeCursorNextTo(br, false);
173 e.preventDefault();
174 }
175};

178/**
179 * Determines whether the lowest containing block node is a blockquote.
180 * @param {Node} n The node.
181 * @return {boolean} Whether the deepest block ancestor of n is a blockquote.
182 */
183goog.editor.plugins.EnterHandler.isDirectlyInBlockquote = function(n) {
184 for (var current = n; current; current = current.parentNode) {
185 if (goog.editor.node.isBlockTag(current)) {
186 return /** @type {!Element} */ (current).tagName ==
187 goog.dom.TagName.BLOCKQUOTE;
188 }
189 }
190
191 return false;

205/**
206 * Deletes the element at the cursor if it is a BR node, and if it does, calls
207 * e.preventDefault to stop the browser from deleting. Only necessary in Gecko
208 * as a workaround for mozilla bug 205350 where deleting a BR that is followed
209 * by a block element doesn't work (the BR gets immediately replaced). We also
210 * need to account for an ill-formed cursor which occurs from us trying to
211 * stop the browser from deleting.
212 *
213 * @param {goog.events.Event} e The DELETE keypress event.
214 * @protected
215 */
216goog.editor.plugins.EnterHandler.prototype.deleteBrGecko = function(e) {
217 var range = this.getFieldObject().getRange();
218 if (range.isCollapsed()) {
219 var container = range.getEndNode();
220 if (container.nodeType == goog.dom.NodeType.ELEMENT) {
221 var nextNode = container.childNodes[range.getEndOffset()];
222 if (nextNode && nextNode.tagName == goog.dom.TagName.BR) {
223 // We want to retrieve the first non-whitespace previous sibling
224 // as we could have added an empty text node below and want to
225 // properly handle deleting a sequence of BR's.
226 var previousSibling = goog.editor.node.getPreviousSibling(nextNode);
227 var nextSibling = nextNode.nextSibling;
228
229 container.removeChild(nextNode);
230 e.preventDefault();
231
232 // When we delete a BR followed by a block level element, the cursor
233 // has a line-height which spans the height of the block level element.
234 // e.g. If we delete a BR followed by a UL, the resulting HTML will
235 // appear to the end user like:-
236 //
237 // | * one
238 // | * two
239 // | * three
240 //
241 // There are a couple of cases that we have to account for in order to
242 // properly conform to what the user expects when DELETE is pressed.
243 //
244 // 1. If the BR has a previous sibling and the previous sibling is
245 // not a block level element or a BR, we place the cursor at the
246 // end of that.
247 // 2. If the BR doesn't have a previous sibling or the previous sibling
248 // is a block level element or a BR, we place the cursor at the
249 // beginning of the leftmost leaf of its next sibling.
250 if (nextSibling && goog.editor.node.isBlockTag(nextSibling)) {
251 if (previousSibling &&
252 !(previousSibling.tagName == goog.dom.TagName.BR ||
253 goog.editor.node.isBlockTag(previousSibling))) {
254 goog.dom.Range.createCaret(
255 previousSibling,
256 goog.editor.node.getLength(previousSibling)).select();
257 } else {
258 var leftMostLeaf = goog.editor.node.getLeftMostLeaf(nextSibling);
259 goog.dom.Range.createCaret(leftMostLeaf, 0).select();
260 }
261 }
262 }
263 }
264 }
265};

268/** @override */
269goog.editor.plugins.EnterHandler.prototype.handleKeyPress = function(e) {
270 // If a dialog doesn't have selectable field, Gecko grabs the event and
271 // performs actions in editor window. This solves that problem and allows
272 // the event to be passed on to proper handlers.
273 if (goog.userAgent.GECKO && this.getFieldObject().inModalMode()) {
274 return false;
275 }
276
277 // Firefox will allow the first node in an iframe to be deleted
278 // on a backspace. Disallow it if the node is empty.
279 if (e.keyCode == goog.events.KeyCodes.BACKSPACE) {
280 this.handleBackspaceInternal(e, this.getFieldObject().getRange());
281
282 } else if (e.keyCode == goog.events.KeyCodes.ENTER) {
283 if (goog.userAgent.GECKO) {
284 if (!e.shiftKey) {
285 // Behave similarly to IE's content editable return carriage:
286 // If the shift key is down or specified by the application, insert a
287 // BR, otherwise split paragraphs
288 this.handleEnterGecko_(e);
289 }
290 } else {
291 // In Gecko-based browsers, this is handled in the handleEnterGecko_
292 // method.
293 this.getFieldObject().dispatchBeforeChange();
294 var cursorPosition = this.deleteCursorSelection_();
295
296 var split = !!this.getFieldObject().execCommand(
297 goog.editor.plugins.Blockquote.SPLIT_COMMAND, cursorPosition);
298 if (split) {
299 // TODO(user): I think we probably don't need to stopPropagation here
300 e.preventDefault();
301 e.stopPropagation();
302 }
303
304 this.releasePositionObject_(cursorPosition);
305
306 if (goog.userAgent.WEBKIT) {
307 this.handleEnterWebkitInternal(e);
308 }
309
310 this.processParagraphTagsInternal(e, split);
311 this.getFieldObject().dispatchChange();
312 }
313
314 } else if (goog.userAgent.GECKO && e.keyCode == goog.events.KeyCodes.DELETE) {
315 this.handleDeleteGecko(e);
316 }
317
318 return false;
319};

348/**
349 * Handles an enter keypress event on fields in Gecko.
350 * @param {goog.events.BrowserEvent} e The key event.
351 * @private
352 */
353goog.editor.plugins.EnterHandler.prototype.handleEnterGecko_ = function(e) {
354 // Retrieve whether the selection is collapsed before we delete it.
355 var range = this.getFieldObject().getRange();
356 var wasCollapsed = !range || range.isCollapsed();
357 var cursorPosition = this.deleteCursorSelection_();
358
359 var handled = this.getFieldObject().execCommand(
360 goog.editor.plugins.Blockquote.SPLIT_COMMAND, cursorPosition);
361 if (handled) {
362 // TODO(user): I think we probably don't need to stopPropagation here
363 e.preventDefault();
364 e.stopPropagation();
365 }
366
367 this.releasePositionObject_(cursorPosition);
368 if (!handled) {
369 this.handleEnterAtCursorGeckoInternal(e, wasCollapsed, range);
370 }
371};

425/**
426 * Ensures all text in IE and Opera to be in the given tag in order to control
427 * Enter spacing. Call this when Enter is pressed if desired.
428 *
429 * We want to make sure the user is always inside of a block (or other nodes
430 * listed in goog.editor.plugins.EnterHandler.IGNORE_ENSURE_BLOCK_NODES_). We
431 * listen to keypress to force nodes that the user is leaving to turn into
432 * blocks, but we also need to listen to keyup to force nodes that the user is
433 * entering to turn into blocks.
434 * Example: html is: "<h2>foo[cursor]</h2>", and the user hits enter. We
435 * don't want to format the h2, but we do want to format the P that is
436 * created on enter. The P node is not available until keyup.
437 * @param {goog.dom.TagName} tag The tag name to convert to.
438 * @param {boolean=} opt_keyUp Whether the function is being called on key up.
439 * When called on key up, the cursor is in the newly created node, so the
440 * semantics for when to change it to a block are different. Specifically,
441 * if the resulting node contains only a BR, it is converted to <tag>.
442 * @protected
443 */
444goog.editor.plugins.EnterHandler.prototype.ensureBlockIeOpera = function(tag,
445 opt_keyUp) {
446 var range = this.getFieldObject().getRange();
447 var container = range.getContainer();
448 var field = this.getFieldObject().getElement();
449
450 var paragraph;
451 while (container && container != field) {
452 // We don't need to ensure a block if we are already in the same block, or
453 // in another block level node that we don't want to change the format of
454 // (unless we're handling keyUp and that block node just contains a BR).
455 var nodeName = container.nodeName;
456 // Due to @bug 2455389, the call to isBrElem needs to be inlined in the if
457 // instead of done before and saved in a variable, so that it can be
458 // short-circuited and avoid a weird IE edge case.
459 if (nodeName == tag ||
460 (goog.editor.plugins.EnterHandler.
461 DO_NOT_ENSURE_BLOCK_NODES_[nodeName] && !(opt_keyUp &&
462 goog.editor.plugins.EnterHandler.isBrElem(container)))) {
463 // Opera can create a <p> inside of a <div> in some situations,
464 // such as when breaking out of a list that is contained in a <div>.
465 if (goog.userAgent.OPERA && paragraph) {
466 if (nodeName == tag &&
467 paragraph == container.lastChild &&
468 goog.editor.node.isEmpty(paragraph)) {
469 goog.dom.insertSiblingAfter(paragraph, container);
470 goog.dom.Range.createFromNodeContents(paragraph).select();
471 }
472 break;
473 }
474 return;
475 }
476 if (goog.userAgent.OPERA && opt_keyUp && nodeName == goog.dom.TagName.P &&
477 nodeName != tag) {
478 paragraph = container;
479 }
480
481 container = container.parentNode;
482 }
483
484
485 if (goog.userAgent.IE && !goog.userAgent.isVersionOrHigher(9)) {
486 // IE (before IE9) has a bug where if the cursor is directly before a block
487 // node (e.g., the content is "foo[cursor]<blockquote>bar</blockquote>"),
488 // the FormatBlock command actually formats the "bar" instead of the "foo".
489 // This is just wrong. To work-around this, we want to move the
490 // selection back one character, and then restore it to its prior position.
491 // NOTE: We use the following "range math" to detect this situation because
492 // using Closure ranges here triggers a bug in IE that causes a crash.
493 // parent2 != parent3 ensures moving the cursor forward one character
494 // crosses at least 1 element boundary, and therefore tests if the cursor is
495 // at such a boundary. The second check, parent3 != range.parentElement()
496 // weeds out some cases where the elements are siblings instead of cousins.
497 var needsHelp = false;
498 range = range.getBrowserRangeObject();
499 var range2 = range.duplicate();
500 range2.moveEnd('character', 1);
501 // In whitebox mode, when the cursor is at the end of the field, trying to
502 // move the end of the range will do nothing, and hence the range's text
503 // will be empty. In this case, the cursor clearly isn't sitting just
504 // before a block node, since it isn't before anything.
505 if (range2.text.length) {
506 var parent2 = range2.parentElement();
507
508 var range3 = range2.duplicate();
509 range3.collapse(false);
510 var parent3 = range3.parentElement();
511
512 if ((needsHelp = parent2 != parent3 &&
513 parent3 != range.parentElement())) {
514 range.move('character', -1);
515 range.select();
516 }
517 }
518 }
519
520 this.getFieldObject().getEditableDomHelper().getDocument().execCommand(
521 'FormatBlock', false, '<' + tag + '>');
522
523 if (needsHelp) {
524 range.move('character', 1);
525 range.select();
526 }
527};

576/**
577 * Delete the selection at the current cursor position, then returns the node
578 * at the current position.
579 * @return {!goog.editor.range.Point} The current cursor position. Note that
580 * unlike simulateEnterIE_, this should not be removed from the DOM.
581 * @private
582 */
583goog.editor.plugins.EnterHandler.prototype.deleteCursorSelectionW3C_ =
584 function() {
585 var range = this.getFieldObject().getRange();
586
587 // Delete the current selection if it's is non-collapsed.
588 // Although this is redundant in FF, it's necessary for Safari
589 if (range && !range.isCollapsed()) {
590 var shouldDelete = true;
591 // Opera selects the <br> in an empty block if there is no text node
592 // preceding it. To preserve inline formatting when pressing [enter] inside
593 // an empty block, don't delete the selection if it only selects a <br> at
594 // the end of the block.
595 // TODO(user): Move this into goog.dom.Range. It should detect this state
596 // when creating a range from the window selection and fix it in the created
597 // range.
598 if (goog.userAgent.OPERA) {
599 var startNode = range.getStartNode();
600 var startOffset = range.getStartOffset();
601 if (startNode == range.getEndNode() &&
602 // This weeds out cases where startNode is a text node.
603 startNode.lastChild &&
604 /** @type {!Element} */ (startNode.lastChild).tagName ==
605 goog.dom.TagName.BR &&
606 // If this check is true, then endOffset is implied to be
607 // startOffset + 1, because the selection is not collapsed and
608 // it starts and ends within the same element.
609 startOffset == startNode.childNodes.length - 1) {
610 shouldDelete = false;
611 }
612 }
613 if (shouldDelete) {
614 goog.editor.plugins.EnterHandler.deleteW3cRange_(range);
615 }
616 }
617
618 return goog.editor.range.getDeepEndPoint(range, true);
619};

622/**
623 * Deletes the contents of the selection from the DOM.
624 * @param {goog.dom.AbstractRange} range The range to remove contents from.
625 * @return {goog.dom.AbstractRange} The resulting range. Used for testing.
626 * @private
627 */
628goog.editor.plugins.EnterHandler.deleteW3cRange_ = function(range) {
629 if (range && !range.isCollapsed()) {
630 var reselect = true;
631 var baseNode = range.getContainerElement();
632 var nodeOffset = new goog.dom.NodeOffset(range.getStartNode(), baseNode);
633 var rangeOffset = range.getStartOffset();
634
635 // Whether the selection crosses no container boundaries.
636 var isInOneContainer =
637 goog.editor.plugins.EnterHandler.isInOneContainerW3c_(range);
638
639 // Whether the selection ends in a container it doesn't fully select.
640 var isPartialEnd = !isInOneContainer &&
641 goog.editor.plugins.EnterHandler.isPartialEndW3c_(range);
642
643 // Remove The range contents, and ensure the correct content stays selected.
644 range.removeContents();
645 var node = nodeOffset.findTargetNode(baseNode);
646 if (node) {
647 range = goog.dom.Range.createCaret(node, rangeOffset);
648 } else {
649 // This occurs when the node that would have been referenced has now been
650 // deleted and there are no other nodes in the baseNode. Thus need to
651 // set the caret to the end of the base node.
652 range =
653 goog.dom.Range.createCaret(baseNode, baseNode.childNodes.length);
654 reselect = false;
655 }
656 range.select();
657
658 // If we just deleted everything from the container, add an nbsp
659 // to the container, and leave the cursor inside of it
660 if (isInOneContainer) {
661 var container = goog.editor.style.getContainer(range.getStartNode());
662 if (goog.editor.node.isEmpty(container, true)) {
663 var html = '&nbsp;';
664 if (goog.userAgent.OPERA &&
665 container.tagName == goog.dom.TagName.LI) {
666 // Don't break Opera's native break-out-of-lists behavior.
667 html = '<br>';
668 }
669 goog.editor.node.replaceInnerHtml(container, html);
670 goog.editor.range.selectNodeStart(container.firstChild);
671 reselect = false;
672 }
673 }
674
675 if (isPartialEnd) {
676 /*
677 This code handles the following, where | is the cursor:
678 <div>a|b</div><div>c|d</div>
679 After removeContents, the remaining HTML is
680 <div>a</div><div>d</div>
681 which means the line break between the two divs remains. This block
682 moves children of the second div in to the first div to get the correct
683 result:
684 <div>ad</div>
685
686 TODO(robbyw): Should we wrap the second div's contents in a span if they
687 have inline style?
688 */
689 var rangeStart = goog.editor.style.getContainer(range.getStartNode());
690 var redundantContainer = goog.editor.node.getNextSibling(rangeStart);
691 if (rangeStart && redundantContainer) {
692 goog.dom.append(rangeStart, redundantContainer.childNodes);
693 goog.dom.removeNode(redundantContainer);
694 }
695 }
696
697 if (reselect) {
698 // The contents of the original range are gone, so restore the cursor
699 // position at the start of where the range once was.
700 range = goog.dom.Range.createCaret(nodeOffset.findTargetNode(baseNode),
701 rangeOffset);
702 range.select();
703 }
704 }
705
706 return range;
707};

710/**
711 * Checks whether the whole range is in a single block-level element.
712 * @param {goog.dom.AbstractRange} range The range to check.
713 * @return {boolean} Whether the whole range is in a single block-level element.
714 * @private
715 */
716goog.editor.plugins.EnterHandler.isInOneContainerW3c_ = function(range) {
717 // Find the block element containing the start of the selection.
718 var startContainer = range.getStartNode();
719 if (goog.editor.style.isContainer(startContainer)) {
720 startContainer = startContainer.childNodes[range.getStartOffset()] ||
721 startContainer;
722 }
723 startContainer = goog.editor.style.getContainer(startContainer);
724
725 // Find the block element containing the end of the selection.
726 var endContainer = range.getEndNode();
727 if (goog.editor.style.isContainer(endContainer)) {
728 endContainer = endContainer.childNodes[range.getEndOffset()] ||
729 endContainer;
730 }
731 endContainer = goog.editor.style.getContainer(endContainer);
732
733 // Compare the two.
734 return startContainer == endContainer;

738/**
739 * Checks whether the end of the range is not at the end of a block-level
740 * element.
741 * @param {goog.dom.AbstractRange} range The range to check.
742 * @return {boolean} Whether the end of the range is not at the end of a
743 * block-level element.
744 * @private
745 */
746goog.editor.plugins.EnterHandler.isPartialEndW3c_ = function(range) {
747 var endContainer = range.getEndNode();
748 var endOffset = range.getEndOffset();
749 var node = endContainer;
750 if (goog.editor.style.isContainer(node)) {
751 var child = node.childNodes[endOffset];
752 // Child is null when end offset is >= length, which indicates the entire
753 // container is selected. Otherwise, we also know the entire container
754 // is selected if the selection ends at a new container.
755 if (!child ||
756 child.nodeType == goog.dom.NodeType.ELEMENT &&
757 goog.editor.style.isContainer(child)) {
758 return false;
759 }
760 }
761
762 var container = goog.editor.style.getContainer(node);
763 while (container != node) {
764 if (goog.editor.node.getNextSibling(node)) {
765 return true;
766 }
767 node = node.parentNode;
768 }
769
770 return endOffset != goog.editor.node.getLength(endContainer);
