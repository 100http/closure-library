173/**
174 * Execute a user-initiated command.
175 * @param {string} command Command to execute.
176 * @param {...*} var_args For color commands, this
177 * should be the hex color (with the #). For FORMAT_BLOCK, this should be
178 * the goog.editor.plugins.BasicTextFormatter.BLOCK_COMMAND.
179 * It will be unused for other commands.
180 * @return {Object|undefined} The result of the command.
181 * @override
182 */
183goog.editor.plugins.BasicTextFormatter.prototype.execCommandInternal = function(
184 command, var_args) {
185 var preserveDir, styleWithCss, needsFormatBlockDiv, hasDummySelection;
186 var result;
187 var opt_arg = arguments[1];
188
189 switch (command) {
190 case goog.editor.plugins.BasicTextFormatter.COMMAND.BACKGROUND_COLOR:
191 // Don't bother for no color selected, color picker is resetting itself.
192 if (!goog.isNull(opt_arg)) {
193 if (goog.editor.BrowserFeature.EATS_EMPTY_BACKGROUND_COLOR) {
194 this.applyBgColorManually_(opt_arg);
195 } else if (goog.userAgent.OPERA) {
196 // backColor will color the block level element instead of
197 // the selected span of text in Opera.
198 this.execCommandHelper_('hiliteColor', opt_arg);
199 } else {
200 this.execCommandHelper_(command, opt_arg);
201 }
202 }
203 break;
204
205 case goog.editor.plugins.BasicTextFormatter.COMMAND.CREATE_LINK:
206 result = this.createLink_(arguments[1], arguments[2], arguments[3]);
207 break;
208
209 case goog.editor.plugins.BasicTextFormatter.COMMAND.LINK:
210 result = this.toggleLink_(opt_arg);
211 break;
212
213 case goog.editor.plugins.BasicTextFormatter.COMMAND.JUSTIFY_CENTER:
214 case goog.editor.plugins.BasicTextFormatter.COMMAND.JUSTIFY_FULL:
215 case goog.editor.plugins.BasicTextFormatter.COMMAND.JUSTIFY_RIGHT:
216 case goog.editor.plugins.BasicTextFormatter.COMMAND.JUSTIFY_LEFT:
217 this.justify_(command);
218 break;
219
220 default:
221 if (goog.userAgent.IE &&
222 command ==
223 goog.editor.plugins.BasicTextFormatter.COMMAND.FORMAT_BLOCK &&
224 opt_arg) {
225 // IE requires that the argument be in the form of an opening
226 // tag, like <h1>, including angle brackets. WebKit will accept
227 // the arguemnt with or without brackets, and Firefox pre-3 supports
228 // only a fixed subset of tags with brackets, and prefers without.
229 // So we only add them IE only.
230 opt_arg = '<' + opt_arg + '>';
231 }
232
233 if (command ==
234 goog.editor.plugins.BasicTextFormatter.COMMAND.FONT_COLOR &&
235 goog.isNull(opt_arg)) {
236 // If we don't have a color, then FONT_COLOR is a no-op.
237 break;
238 }
239
240 switch (command) {
241 case goog.editor.plugins.BasicTextFormatter.COMMAND.INDENT:
242 case goog.editor.plugins.BasicTextFormatter.COMMAND.OUTDENT:
243 if (goog.editor.BrowserFeature.HAS_STYLE_WITH_CSS) {
244 if (goog.userAgent.GECKO) {
245 styleWithCss = true;
246 }
247 if (goog.userAgent.OPERA) {
248 if (command ==
249 goog.editor.plugins.BasicTextFormatter.COMMAND.OUTDENT) {
250 // styleWithCSS actually sets negative margins on <blockquote>
251 // to outdent them. If the command is enabled without
252 // styleWithCSS flipped on, then the caret is in a blockquote so
253 // styleWithCSS must not be used. But if the command is not
254 // enabled, styleWithCSS should be used so that elements such as
255 // a <div> with a margin-left style can still be outdented.
256 // (Opera bug: CORE-21118)
257 styleWithCss =
258 !this.getDocument_().queryCommandEnabled('outdent');
259 } else {
260 // Always use styleWithCSS for indenting. Otherwise, Opera will
261 // make separate <blockquote>s around *each* indented line,
262 // which adds big default <blockquote> margins between each
263 // indented line.
264 styleWithCss = true;
265 }
266 }
267 }
268 // Fall through.
269
270 case goog.editor.plugins.BasicTextFormatter.COMMAND.ORDERED_LIST:
271 case goog.editor.plugins.BasicTextFormatter.COMMAND.UNORDERED_LIST:
272 if (goog.editor.BrowserFeature.LEAVES_P_WHEN_REMOVING_LISTS &&
273 this.queryCommandStateInternal_(this.getDocument_(),
274 command)) {
275 // IE leaves behind P tags when unapplying lists.
276 // If we're not in P-mode, then we want divs
277 // So, unlistify, then convert the Ps into divs.
278 needsFormatBlockDiv = this.getFieldObject().queryCommandValue(
279 goog.editor.Command.DEFAULT_TAG) != goog.dom.TagName.P;
280 } else if (!goog.editor.BrowserFeature.CAN_LISTIFY_BR) {
281 // IE doesn't convert BRed line breaks into separate list items.
282 // So convert the BRs to divs, then do the listify.
283 this.convertBreaksToDivs_();
284 }
285
286 // This fix only works in Gecko.
287 if (goog.userAgent.GECKO &&
288 goog.editor.BrowserFeature.FORGETS_FORMATTING_WHEN_LISTIFYING &&
289 !this.queryCommandValue(command)) {
290 hasDummySelection |= this.beforeInsertListGecko_();
291 }
292 // Fall through to preserveDir block
293
294 case goog.editor.plugins.BasicTextFormatter.COMMAND.FORMAT_BLOCK:
295 // Both FF & IE may lose directionality info. Save/restore it.
296 // TODO(user): Does Safari also need this?
297 // TODO (gmark, jparent): This isn't ideal because it uses a string
298 // literal, so if the plugin name changes, it would break. We need a
299 // better solution. See also other places in code that use
300 // this.getPluginByClassId('Bidi').
301 preserveDir = !!this.getFieldObject().getPluginByClassId('Bidi');
302 break;
303
304 case goog.editor.plugins.BasicTextFormatter.COMMAND.SUBSCRIPT:
305 case goog.editor.plugins.BasicTextFormatter.COMMAND.SUPERSCRIPT:
306 if (goog.editor.BrowserFeature.NESTS_SUBSCRIPT_SUPERSCRIPT) {
307 // This browser nests subscript and superscript when both are
308 // applied, instead of canceling out the first when applying the
309 // second.
310 this.applySubscriptSuperscriptWorkarounds_(command);
311 }
312 break;
313
314 case goog.editor.plugins.BasicTextFormatter.COMMAND.UNDERLINE:
315 case goog.editor.plugins.BasicTextFormatter.COMMAND.BOLD:
316 case goog.editor.plugins.BasicTextFormatter.COMMAND.ITALIC:
317 // If we are applying the formatting, then we want to have
318 // styleWithCSS false so that we generate html tags (like <b>). If we
319 // are unformatting something, we want to have styleWithCSS true so
320 // that we can unformat both html tags and inline styling.
321 // TODO(user): What about WebKit and Opera?
322 styleWithCss = goog.userAgent.GECKO &&
323 goog.editor.BrowserFeature.HAS_STYLE_WITH_CSS &&
324 this.queryCommandValue(command);
325 break;
326
327 case goog.editor.plugins.BasicTextFormatter.COMMAND.FONT_COLOR:
328 case goog.editor.plugins.BasicTextFormatter.COMMAND.FONT_FACE:
329 // It is very expensive in FF (order of magnitude difference) to use
330 // font tags instead of styled spans. Whenever possible,
331 // force FF to use spans.
332 // Font size is very expensive too, but FF always uses font tags,
333 // regardless of which styleWithCSS value you use.
334 styleWithCss = goog.editor.BrowserFeature.HAS_STYLE_WITH_CSS &&
335 goog.userAgent.GECKO;
336 }
337
338 /**
339 * Cases where we just use the default execCommand (in addition
340 * to the above fall-throughs)
341 * goog.editor.plugins.BasicTextFormatter.COMMAND.STRIKE_THROUGH:
342 * goog.editor.plugins.BasicTextFormatter.COMMAND.HORIZONTAL_RULE:
343 * goog.editor.plugins.BasicTextFormatter.COMMAND.SUBSCRIPT:
344 * goog.editor.plugins.BasicTextFormatter.COMMAND.SUPERSCRIPT:
345 * goog.editor.plugins.BasicTextFormatter.COMMAND.UNDERLINE:
346 * goog.editor.plugins.BasicTextFormatter.COMMAND.BOLD:
347 * goog.editor.plugins.BasicTextFormatter.COMMAND.ITALIC:
348 * goog.editor.plugins.BasicTextFormatter.COMMAND.FONT_SIZE:
349 * goog.editor.plugins.BasicTextFormatter.COMMAND.FONT_FACE:
350 */
351 this.execCommandHelper_(command, opt_arg, preserveDir, !!styleWithCss);
352
353 if (hasDummySelection) {
354 this.getDocument_().execCommand('Delete', false, true);
355 }
356
357 if (needsFormatBlockDiv) {
358 this.getDocument_().execCommand('FormatBlock', false, '<div>');
359 }
360 }
361 // FF loses focus, so we have to set the focus back to the document or the
362 // user can't type after selecting from menu. In IE, focus is set correctly
363 // and resetting it here messes it up.
364 if (goog.userAgent.GECKO && !this.getFieldObject().inModalMode()) {
365 this.focusField_();
366 }
367 return result;
368};

380/**
381 * Gets the command value.
382 * @param {string} command The command value to get.
383 * @return {string|boolean|null} The current value of the command in the given
384 * selection. NOTE: This return type list is not documented in MSDN or MDC
385 * and has been constructed from experience. Please update it
386 * if necessary.
387 * @override
388 */
389goog.editor.plugins.BasicTextFormatter.prototype.queryCommandValue = function(
390 command) {
391 var styleWithCss;
392 switch (command) {
393 case goog.editor.plugins.BasicTextFormatter.COMMAND.LINK:
394 return this.isNodeInState_(goog.dom.TagName.A);
395
396 case goog.editor.plugins.BasicTextFormatter.COMMAND.JUSTIFY_CENTER:
397 case goog.editor.plugins.BasicTextFormatter.COMMAND.JUSTIFY_FULL:
398 case goog.editor.plugins.BasicTextFormatter.COMMAND.JUSTIFY_RIGHT:
399 case goog.editor.plugins.BasicTextFormatter.COMMAND.JUSTIFY_LEFT:
400 return this.isJustification_(command);
401
402 case goog.editor.plugins.BasicTextFormatter.COMMAND.FORMAT_BLOCK:
403 // TODO(nicksantos): See if we can use queryCommandValue here.
404 return goog.editor.plugins.BasicTextFormatter.getSelectionBlockState_(
405 this.getFieldObject().getRange());
406
407 case goog.editor.plugins.BasicTextFormatter.COMMAND.INDENT:
408 case goog.editor.plugins.BasicTextFormatter.COMMAND.OUTDENT:
409 case goog.editor.plugins.BasicTextFormatter.COMMAND.HORIZONTAL_RULE:
410 // TODO: See if there are reasonable results to return for
411 // these commands.
412 return false;
413
414 case goog.editor.plugins.BasicTextFormatter.COMMAND.FONT_SIZE:
415 case goog.editor.plugins.BasicTextFormatter.COMMAND.FONT_FACE:
416 case goog.editor.plugins.BasicTextFormatter.COMMAND.FONT_COLOR:
417 case goog.editor.plugins.BasicTextFormatter.COMMAND.BACKGROUND_COLOR:
418 // We use queryCommandValue here since we don't just want to know if a
419 // color/fontface/fontsize is applied, we want to know WHICH one it is.
420 return this.queryCommandValueInternal_(this.getDocument_(), command,
421 goog.editor.BrowserFeature.HAS_STYLE_WITH_CSS &&
422 goog.userAgent.GECKO);
423
424 case goog.editor.plugins.BasicTextFormatter.COMMAND.UNDERLINE:
425 case goog.editor.plugins.BasicTextFormatter.COMMAND.BOLD:
426 case goog.editor.plugins.BasicTextFormatter.COMMAND.ITALIC:
427 styleWithCss = goog.editor.BrowserFeature.HAS_STYLE_WITH_CSS &&
428 goog.userAgent.GECKO;
429
430 default:
431 /**
432 * goog.editor.plugins.BasicTextFormatter.COMMAND.STRIKE_THROUGH
433 * goog.editor.plugins.BasicTextFormatter.COMMAND.SUBSCRIPT
434 * goog.editor.plugins.BasicTextFormatter.COMMAND.SUPERSCRIPT
435 * goog.editor.plugins.BasicTextFormatter.COMMAND.UNDERLINE
436 * goog.editor.plugins.BasicTextFormatter.COMMAND.BOLD
437 * goog.editor.plugins.BasicTextFormatter.COMMAND.ITALIC
438 * goog.editor.plugins.BasicTextFormatter.COMMAND.ORDERED_LIST
439 * goog.editor.plugins.BasicTextFormatter.COMMAND.UNORDERED_LIST
440 */
441 // This only works for commands that use the default execCommand
442 return this.queryCommandStateInternal_(this.getDocument_(), command,
443 styleWithCss);
444 }

448/**
449 * @override
450 */
451goog.editor.plugins.BasicTextFormatter.prototype.prepareContentsHtml =
452 function(html) {
453 // If the browser collapses empty nodes and the field has only a script
454 // tag in it, then it will collapse this node. Which will mean the user
455 // can't click into it to edit it.
456 if (goog.editor.BrowserFeature.COLLAPSES_EMPTY_NODES &&
457 html.match(/^\s*<script/i)) {
458 html = '&nbsp;' + html;
459 }
460
461 if (goog.editor.BrowserFeature.CONVERT_TO_B_AND_I_TAGS) {
462 // Some browsers (FF) can't undo strong/em in some cases, but can undo b/i!
463 html = html.replace(/<(\/?)strong([^\w])/gi, '<$1b$2');
464 html = html.replace(/<(\/?)em([^\w])/gi, '<$1i$2');
465 }
466
467 return html;
468};

500/**
501 * @override
502 */
503goog.editor.plugins.BasicTextFormatter.prototype.cleanContentsHtml =
504 function(html) {
505 if (goog.editor.BrowserFeature.MOVES_STYLE_TO_HEAD) {
506 // Safari creates a new <head> element for <style> tags, so prepend their
507 // contents to the output.
508 var heads = this.getFieldObject().getEditableDomHelper().
509 getElementsByTagNameAndClass(goog.dom.TagName.HEAD);
510 var stylesHtmlArr = [];
511
512 // i starts at 1 so we don't copy in the original, legitimate <head>.
513 var numHeads = heads.length;
514 for (var i = 1; i < numHeads; ++i) {
515 var styles = heads[i].getElementsByTagName(goog.dom.TagName.STYLE);
516 var numStyles = styles.length;
517 for (var j = 0; j < numStyles; ++j) {
518 stylesHtmlArr.push(styles[j].outerHTML);
519 }
520 }
521 return stylesHtmlArr.join('') + html;
522 }
523
524 return html;

567/**
568 * Regular expression to match BRs in HTML. Saves the BRs' attributes in $1 for
569 * use with replace(). In non-IE browsers, does not match BRs adjacent to an
570 * opening or closing DIV or P tag, since nonrendered BR elements can occur at
571 * the end of block level containers in those browsers' editors.
572 * @type {RegExp}
573 * @private
574 */
575goog.editor.plugins.BasicTextFormatter.BR_REGEXP_ =
576 goog.userAgent.IE ? /<br([^\/>]*)\/?>/gi :
577 /<br([^\/>]*)\/?>(?!<\/(div|p)>)/gi;
578
579
580/**
581 * Convert BRs in the selection to divs.
582 * This is only intended to be used in IE and Opera.
583 * @return {boolean} Whether any BR's were converted.
584 * @private
585 */
586goog.editor.plugins.BasicTextFormatter.prototype.convertBreaksToDivs_ =
587 function() {
588 if (!goog.userAgent.IE && !goog.userAgent.OPERA) {
589 // This function is only supported on IE and Opera.
590 return false;
591 }
592 var range = this.getRange_();
593 var parent = range.getContainerElement();
594 var doc = this.getDocument_();
595
596 goog.editor.plugins.BasicTextFormatter.BR_REGEXP_.lastIndex = 0;
597 // Only mess with the HTML/selection if it contains a BR.
598 if (goog.editor.plugins.BasicTextFormatter.BR_REGEXP_.test(
599 parent.innerHTML)) {
600 // Insert temporary markers to remember the selection.
601 var savedRange = range.saveUsingCarets();
602
603 if (parent.tagName == goog.dom.TagName.P) {
604 // Can't append paragraphs to paragraph tags. Throws an exception in IE.
605 goog.editor.plugins.BasicTextFormatter.convertParagraphToDiv_(
606 parent, true);
607 } else {
608 // Used to do:
609 // IE: <div>foo<br>bar</div> --> <div>foo<p id="temp_br">bar</div>
610 // Opera: <div>foo<br>bar</div> --> <div>foo<p class="temp_br">bar</div>
611 // To fix bug 1939883, now does for both:
612 // <div>foo<br>bar</div> --> <div>foo<p trtempbr="temp_br">bar</div>
613 // TODO(user): Confirm if there's any way to skip this
614 // intermediate step of converting br's to p's before converting those to
615 // div's. The reason may be hidden in CLs 5332866 and 8530601.
616 var attribute = 'trtempbr';
617 var value = 'temp_br';
618 var newHtml = parent.innerHTML.replace(
619 goog.editor.plugins.BasicTextFormatter.BR_REGEXP_,
620 '<p$1 ' + attribute + '="' + value + '">');
621 goog.editor.node.replaceInnerHtml(parent, newHtml);
622
623 var paragraphs =
624 goog.array.toArray(parent.getElementsByTagName(goog.dom.TagName.P));
625 goog.iter.forEach(paragraphs, function(paragraph) {
626 if (paragraph.getAttribute(attribute) == value) {
627 paragraph.removeAttribute(attribute);
628 if (goog.string.isBreakingWhitespace(
629 goog.dom.getTextContent(paragraph))) {
630 // Prevent the empty blocks from collapsing.
631 // A <BR> is preferable because it doesn't result in any text being
632 // added to the "blank" line. In IE, however, it is possible to
633 // place the caret after the <br>, which effectively creates a
634 // visible line break. Because of this, we have to resort to using a
635 // &nbsp; in IE.
636 var child = goog.userAgent.IE ?
637 doc.createTextNode(goog.string.Unicode.NBSP) :
638 doc.createElement(goog.dom.TagName.BR);
639 paragraph.appendChild(child);
640 }
641 goog.editor.plugins.BasicTextFormatter.convertParagraphToDiv_(
642 paragraph);
643 }
644 });
645 }
646
647 // Select the previously selected text so we only listify
648 // the selected portion and maintain the user's selection.
649 savedRange.restore();
650 return true;
651 }
652
653 return false;
654};

657/**
658 * Convert the given paragraph to being a div. This clobbers the
659 * passed-in node!
660 * This is only intended to be used in IE and Opera.
661 * @param {Node} paragraph Paragragh to convert to a div.
662 * @param {boolean=} opt_convertBrs If true, also convert BRs to divs.
663 * @private
664 */
665goog.editor.plugins.BasicTextFormatter.convertParagraphToDiv_ =
666 function(paragraph, opt_convertBrs) {
667 if (!goog.userAgent.IE && !goog.userAgent.OPERA) {
668 // This function is only supported on IE and Opera.
669 return;
670 }
671 var outerHTML = paragraph.outerHTML.replace(/<(\/?)p/gi, '<$1div');

983/**

985 * a new link. Otherwise, enclose the selection in a link.
986 * @param {goog.dom.AbstractRange} range The closure range object for the
987 * current selection.
988 * @param {string} url The url to link to.
989 * @param {string=} opt_target Target for the link.
990 * @return {goog.editor.Link?} The newly created link, or null if the link
991 * couldn't be created.
992 * @private
993 */

995 url, opt_target) {
996 var anchor = null;
997 var anchors = [];
998 var parent = range && range.getContainerElement();
999 // We do not yet support creating links around images. Instead of throwing
1000 // lots of js errors, just fail silently.
1001 // TODO(user): Add support for linking images.

1003 return null;
1004 }
1005 // If range is not present, the editable field doesn't have focus, abort
1006 // creating a link.
1007 if (!range) {
1008 return null;
1009 }
1010
1011 if (range.isCollapsed()) {
1012 var textRange = range.getTextRange(0).getBrowserRangeObject();
1013 if (goog.editor.BrowserFeature.HAS_W3C_RANGES) {
1014 anchor = this.getFieldDomHelper().createElement(goog.dom.TagName.A);

1020 anchor.removeAttribute('id');
1021 }
1022 } else {

1035 };

1038 goog.dom.TagName.A), setHrefAndLink);

1050 }

1054 /** @type {HTMLAnchorElement} */ (anchor), url, opt_target, anchors);
