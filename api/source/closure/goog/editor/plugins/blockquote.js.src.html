36/**
37 * Plugin to handle splitting block quotes. This plugin does nothing on its
38 * own and should be used in conjunction with EnterHandler or one of its
39 * subclasses.
40 * @param {boolean} requiresClassNameToSplit Whether to split only blockquotes
41 * that have the given classname.
42 * @param {string=} opt_className The classname to apply to generated
43 * blockquotes. Defaults to 'tr_bq'.
44 * @constructor
45 * @extends {goog.editor.Plugin}
46 * @final
47 */
48goog.editor.plugins.Blockquote = function(requiresClassNameToSplit,
49 opt_className) {
50 goog.editor.Plugin.call(this);
51
52 /**
53 * Whether we only split blockquotes that have {@link classname}, or whether
54 * all blockquote tags should be split on enter.
55 * @type {boolean}
56 * @private
57 */
58 this.requiresClassNameToSplit_ = requiresClassNameToSplit;
59
60 /**
61 * Classname to put on blockquotes that are generated via the toolbar for
62 * blockquote, so that we can internally distinguish these from blockquotes
63 * that are used for indentation. This classname can be over-ridden by
64 * clients for styling or other purposes.
65 * @type {string}
66 * @private
67 */
68 this.className_ = opt_className || goog.getCssName('tr_bq');
69};
70goog.inherits(goog.editor.plugins.Blockquote, goog.editor.Plugin);

110/**
111 * Checks if a node is a blockquote which can be split. A splittable blockquote
112 * meets the following criteria:
113 * <ol>
114 * <li>Node is a blockquote element</li>
115 * <li>Node has the blockquote classname if the classname is required to
116 * split</li>
117 * </ol>
118 *
119 * @param {Node} node DOM node in question.
120 * @return {boolean} Whether the node is a splittable blockquote.
121 */
122goog.editor.plugins.Blockquote.prototype.isSplittableBlockquote =
123 function(node) {
124 if (node.tagName != goog.dom.TagName.BLOCKQUOTE) {
125 return false;
126 }
127
128 if (!this.requiresClassNameToSplit_) {
129 return true;
130 }
131
132 return goog.dom.classlist.contains(/** @type {!Element} */ (node),
133 this.className_);
134};

138 * Checks if a node is a blockquote element which has been setup.
139 * @param {Node} node DOM node to check.
140 * @return {boolean} Whether the node is a blockquote with the required class

143goog.editor.plugins.Blockquote.prototype.isSetupBlockquote =

145 return node.tagName == goog.dom.TagName.BLOCKQUOTE &&
146 goog.dom.classlist.contains(/** @type {!Element} */ (node),

173/**
174 * Helper routine which walks up the tree to find the topmost
175 * ancestor with only a single child. The ancestor node or the original
176 * node (if no ancestor was found) is then removed from the DOM.
177 *
178 * @param {Node} node The node whose ancestors have to be searched.
179 * @param {Node} root The root node to stop the search at.
180 * @private
181 */
182goog.editor.plugins.Blockquote.findAndRemoveSingleChildAncestor_ = function(
183 node, root) {
184 var predicateFunc = function(parentNode) {
185 return parentNode != root && parentNode.childNodes.length == 1;
186 };
187 var ancestor = goog.editor.node.findHighestMatchingAncestor(node,
188 predicateFunc);
189 if (!ancestor) {
190 ancestor = node;
191 }
192 goog.dom.removeNode(ancestor);
193};

198 * @param {Array<Node>} nodes Nodes to be checked.

201goog.editor.plugins.Blockquote.removeAllWhiteSpaceNodes_ = function(nodes) {

217/**
218 * Splits a quoted region if any. To be called on a key press event. When this
219 * function returns true, the event that caused it to be called should be
220 * canceled.
221 * @param {string} command The command to execute.
222 * @param {...*} var_args Single additional argument representing the current
223 * cursor position. If BrowserFeature.HAS_W3C_RANGES it is an object with a
224 * {@code node} key and an {@code offset} key. In other cases (legacy IE)
225 * it is a single node.
226 * @return {boolean|undefined} Boolean true when the quoted region has been
227 * split, false or undefined otherwise.
228 * @override
229 */
230goog.editor.plugins.Blockquote.prototype.execCommandInternal = function(
231 command, var_args) {
232 var pos = arguments[1];
233 if (command == goog.editor.plugins.Blockquote.SPLIT_COMMAND && pos &&
234 (this.className_ || !this.requiresClassNameToSplit_)) {
235 return goog.editor.BrowserFeature.HAS_W3C_RANGES ?
236 this.splitQuotedBlockW3C_(pos) :
237 this.splitQuotedBlockIE_(/** @type {Node} */ (pos));
238 }
239};

242/**
243 * Version of splitQuotedBlock_ that uses W3C ranges.
244 * @param {Object} anchorPos The current cursor position.
245 * @return {boolean} Whether the blockquote was split.
246 * @private
247 */
248goog.editor.plugins.Blockquote.prototype.splitQuotedBlockW3C_ =
249 function(anchorPos) {
250 var cursorNode = anchorPos.node;
251 var quoteNode = goog.editor.node.findTopMostEditableAncestor(
252 cursorNode.parentNode, goog.bind(this.isSplittableBlockquote, this));
253
254 var secondHalf, textNodeToRemove;
255 var insertTextNode = false;
256 // There are two special conditions that we account for here.
257 //
258 // 1. Whenever the cursor is after (one<BR>|) or just before a BR element
259 // (one|<BR>) and the user presses enter, the second quoted block starts
260 // with a BR which appears to the user as an extra newline. This stems
261 // from the fact that we create two text nodes as our split boundaries
262 // and the BR becomes a part of the second half because of this.
263 //
264 // 2. When the cursor is at the end of a text node with no siblings and
265 // the user presses enter, the second blockquote might contain a
266 // empty subtree that ends in a 0 length text node. We account for that
267 // as a post-splitting operation.
268 if (quoteNode) {
269
270 // selection is in a line that has text in it
271 if (cursorNode.nodeType == goog.dom.NodeType.TEXT) {
272 if (anchorPos.offset == cursorNode.length) {
273 var siblingNode = cursorNode.nextSibling;
274
275 // This accounts for the condition where the cursor appears at the
276 // end of a text node and right before the BR eg: one|<BR>. We ensure
277 // that we split on the BR in that case.
278 if (siblingNode && siblingNode.tagName == goog.dom.TagName.BR) {
279 cursorNode = siblingNode;
280 // This might be null but splitDomTreeAt accounts for the null case.
281 secondHalf = siblingNode.nextSibling;
282 } else {
283 textNodeToRemove = cursorNode.splitText(anchorPos.offset);
284 secondHalf = textNodeToRemove;
285 }
286 } else {
287 secondHalf = cursorNode.splitText(anchorPos.offset);
288 }
289 } else if (cursorNode.tagName == goog.dom.TagName.BR) {
290 // This might be null but splitDomTreeAt accounts for the null case.
291 secondHalf = cursorNode.nextSibling;
292 } else {
293 // The selection is in a line that is empty, with more than 1 level
294 // of quote.
295 insertTextNode = true;
296 }
297 } else {
298 // Check if current node is a quote node.
299 // This will happen if user clicks in an empty line in the quote,
300 // when there is 1 level of quote.
301 if (this.isSetupBlockquote(cursorNode)) {
302 quoteNode = cursorNode;
303 insertTextNode = true;
304 }
305 }
306
307 if (insertTextNode) {
308 // Create two empty text nodes to split between.
309 cursorNode = this.insertEmptyTextNodeBeforeRange_();
310 secondHalf = this.insertEmptyTextNodeBeforeRange_();
311 }
312
313 if (!quoteNode) {
314 return false;
315 }
316
317 secondHalf = goog.editor.node.splitDomTreeAt(cursorNode, secondHalf,
318 quoteNode);
319 goog.dom.insertSiblingAfter(secondHalf, quoteNode);
320
321 // Set the insertion point.
322 var dh = this.getFieldDomHelper();
323 var tagToInsert =
324 this.getFieldObject().queryCommandValue(
325 goog.editor.Command.DEFAULT_TAG) ||
326 goog.dom.TagName.DIV;
327 var container = dh.createElement(/** @type {string} */ (tagToInsert));
328 container.innerHTML = '&nbsp;'; // Prevent the div from collapsing.
329 quoteNode.parentNode.insertBefore(container, secondHalf);
330 dh.getWindow().getSelection().collapse(container, 0);
331
332 // We need to account for the condition where the second blockquote
333 // might contain an empty DOM tree. This arises from trying to split
334 // at the end of an empty text node. We resolve this by walking up the tree
335 // till we either reach the blockquote or till we hit a node with more
336 // than one child. The resulting node is then removed from the DOM.
337 if (textNodeToRemove) {
338 goog.editor.plugins.Blockquote.findAndRemoveSingleChildAncestor_(
339 textNodeToRemove, secondHalf);
340 }
341
342 goog.editor.plugins.Blockquote.removeAllWhiteSpaceNodes_(
343 [quoteNode, secondHalf]);
344 return true;
345};

348/**
349 * Inserts an empty text node before the field's range.

353goog.editor.plugins.Blockquote.prototype.insertEmptyTextNodeBeforeRange_ =

355 var range = this.getFieldObject().getRange();
356 var node = this.getFieldDomHelper().createTextNode('');
357 range.insertNode(node, true);

362/**
363 * IE version of splitQuotedBlock_.
364 * @param {Node} splitNode The current cursor position.
365 * @return {boolean} Whether the blockquote was split.
366 * @private
367 */
368goog.editor.plugins.Blockquote.prototype.splitQuotedBlockIE_ =
369 function(splitNode) {
370 var dh = this.getFieldDomHelper();
371 var quoteNode = goog.editor.node.findTopMostEditableAncestor(
372 splitNode.parentNode, goog.bind(this.isSplittableBlockquote, this));
373
374 if (!quoteNode) {
375 return false;
376 }
377
378 var clone = splitNode.cloneNode(false);
379
380 // Whenever the cursor is just before a BR element (one|<BR>) and the user
381 // presses enter, the second quoted block starts with a BR which appears
382 // to the user as an extra newline. This stems from the fact that the
383 // dummy span that we create (splitNode) occurs before the BR and we split
384 // on that.
385 if (splitNode.nextSibling &&
386 splitNode.nextSibling.tagName == goog.dom.TagName.BR) {
387 splitNode = splitNode.nextSibling;
388 }
389 var secondHalf = goog.editor.node.splitDomTreeAt(splitNode, clone, quoteNode);
390 goog.dom.insertSiblingAfter(secondHalf, quoteNode);
391
392 // Set insertion point.
393 var tagToInsert =
394 this.getFieldObject().queryCommandValue(
395 goog.editor.Command.DEFAULT_TAG) ||
396 goog.dom.TagName.DIV;
397 var div = dh.createElement(/** @type {string} */ (tagToInsert));
398 quoteNode.parentNode.insertBefore(div, secondHalf);
399
400 // The div needs non-whitespace contents in order for the insertion point
401 // to get correctly inserted.
402 div.innerHTML = '&nbsp;';
403
404 // Moving the range 1 char isn't enough when you have markup.
405 // This moves the range to the end of the nbsp.
406 var range = dh.getDocument().selection.createRange();
407 range.moveToElementText(splitNode);
408 range.move('character', 2);
409 range.select();
410
411 // Remove the no-longer-necessary nbsp.
412 div.innerHTML = '';
413
414 // Clear the original selection.
415 range.pasteHTML('');
416
417 // We need to remove clone from the DOM but just removing clone alone will
418 // not suffice. Let's assume we have the following DOM structure and the
419 // cursor is placed after the first numbered list item "one".
420 //
421 // <blockquote class="gmail-quote">
422 // <div><div>a</div><ol><li>one|</li></ol></div>
423 // <div>b</div>
424 // </blockquote>
425 //
426 // After pressing enter, we have the following structure.
427 //
428 // <blockquote class="gmail-quote">
429 // <div><div>a</div><ol><li>one|</li></ol></div>
430 // </blockquote>
431 // <div>&nbsp;</div>
432 // <blockquote class="gmail-quote">
433 // <div><ol><li><span id=""></span></li></ol></div>
434 // <div>b</div>
435 // </blockquote>
436 //
437 // The clone is contained in a subtree which should be removed. This stems
438 // from the fact that we invoke splitDomTreeAt with the dummy span
439 // as the starting splitting point and this results in the empty subtree
440 // <div><ol><li><span id=""></span></li></ol></div>.
441 //
442 // We resolve this by walking up the tree till we either reach the
443 // blockquote or till we hit a node with more than one child. The resulting
444 // node is then removed from the DOM.
445 goog.editor.plugins.Blockquote.findAndRemoveSingleChildAncestor_(
446 clone, secondHalf);
447
448 goog.editor.plugins.Blockquote.removeAllWhiteSpaceNodes_(
449 [quoteNode, secondHalf]);
450 return true;
451};
