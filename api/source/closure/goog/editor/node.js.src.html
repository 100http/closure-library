200/**
201 * Internal helper for lastChild/firstChild that skips nodes that are entirely
202 * whitespace or comments.
203 * @param {Node} parent The reference node.
204 * @param {boolean} isReversed Whether children should be traversed forward
205 * or backward.
206 * @return {Node} The first/last child of sibling that is important according
207 * to goog.editor.node.isImportant, or null if no such node exists.
208 * @private

210goog.editor.node.getChildHelper_ = function(parent, isReversed) {
211 return (!parent || parent.nodeType != goog.dom.NodeType.ELEMENT) ? null :
212 /** @type {Node} */ (goog.editor.node.getFirstValue_(goog.iter.filter(

214 /** @type {!Element} */ (parent), isReversed),
215 goog.editor.node.isImportant)));

236/**
237 * Determine if a node should be returned by the iterator functions.
238 * @param {Node} node An object implementing the DOM1 Node interface.
239 * @return {boolean} Whether the node is an element, or a text node that
240 * is not all whitespace.

242goog.editor.node.isImportant = function(node) {
243 // Return true if the node is not either a TextNode or an ElementNode.
244 return node.nodeType == goog.dom.NodeType.ELEMENT ||

252 * @param {Node} textNode A node implementing the CharacterData interface (i.e.,

262/**
263 * Returns true if the node contains only whitespace and is not and does not
264 * contain any images, iframes or embed tags.
265 * @param {Node} node The node to check.
266 * @param {boolean=} opt_prohibitSingleNbsp By default, this function treats a
267 * single nbsp as empty. Set this to true to treat this case as non-empty.
268 * @return {boolean} Whether the node contains only whitespace.
269 */
270goog.editor.node.isEmpty = function(node, opt_prohibitSingleNbsp) {
271 var nodeData = goog.dom.getRawTextContent(node);
272
273 if (node.getElementsByTagName) {
274 for (var tag in goog.editor.node.NON_EMPTY_TAGS_) {
275 if (node.tagName == tag || node.getElementsByTagName(tag).length > 0) {
276 return false;
277 }
278 }
279 }
280 return (!opt_prohibitSingleNbsp && nodeData == goog.string.Unicode.NBSP) ||
281 goog.string.isBreakingWhitespace(nodeData);
282};

297/**
298 * Search child nodes using a predicate function and return the first node that
299 * satisfies the condition.
300 * @param {Node} parent The parent node to search.
301 * @param {function(Node):boolean} hasProperty A function that takes a child
302 * node as a parameter and returns true if it meets the criteria.
303 * @return {?number} The index of the node found, or null if no node is found.
304 */
305goog.editor.node.findInChildren = function(parent, hasProperty) {
306 for (var i = 0, len = parent.childNodes.length; i < len; i++) {
307 if (hasProperty(parent.childNodes[i])) {
308 return i;
309 }
310 }
311 return null;
312};

315/**
316 * Search ancestor nodes using a predicate function and returns the topmost
317 * ancestor in the chain of consecutive ancestors that satisfies the condition.
318 *
319 * @param {Node} node The node whose ancestors have to be searched.
320 * @param {function(Node): boolean} hasProperty A function that takes a parent
321 * node as a parameter and returns true if it meets the criteria.
322 * @return {Node} The topmost ancestor or null if no ancestor satisfies the
323 * predicate function.
324 */
325goog.editor.node.findHighestMatchingAncestor = function(node, hasProperty) {
326 var parent = node.parentNode;
327 var ancestor = null;
328 while (parent && hasProperty(parent)) {
329 ancestor = parent;
330 parent = parent.parentNode;
331 }
332 return ancestor;
333};

384/**
385 * Finds the top-most DOM node inside an editable field that is an ancestor
386 * (or self) of a given DOM node and meets the specified criteria.
387 * @param {Node} node The DOM node where the search starts.
388 * @param {function(Node) : boolean} criteria A function that takes a DOM node
389 * as a parameter and returns a boolean to indicate whether the node meets
390 * the criteria or not.
391 * @return {Node} The DOM node if found, or null.
392 */
393goog.editor.node.findTopMostEditableAncestor = function(node, criteria) {
394 var targetNode = null;
395 while (node && !goog.editor.node.isEditableContainer(node)) {
396 if (criteria(node)) {
397 targetNode = node;
398 }
399 node = node.parentNode;
400 }
401 return targetNode;
402};

405/**
406 * Splits off a subtree.
407 * @param {!Node} currentNode The starting splitting point.
408 * @param {Node=} opt_secondHalf The initial leftmost leaf the new subtree.
409 * If null, siblings after currentNode will be placed in the subtree, but
410 * no additional node will be.
411 * @param {Node=} opt_root The top of the tree where splitting stops at.
412 * @return {!Node} The new subtree.
413 */
414goog.editor.node.splitDomTreeAt = function(currentNode,
415 opt_secondHalf, opt_root) {
416 var parent;
417 while (currentNode != opt_root && (parent = currentNode.parentNode)) {
418 opt_secondHalf = goog.editor.node.getSecondHalfOfNode_(parent, currentNode,
419 opt_secondHalf);
420 currentNode = parent;
421 }
422 return /** @type {!Node} */(opt_secondHalf);
423};

426/**
427 * Creates a clone of node, moving all children after startNode to it.
428 * When firstChild is not null or undefined, it is also appended to the clone
429 * as the first child.
430 * @param {!Node} node The node to clone.
431 * @param {!Node} startNode All siblings after this node will be moved to the
432 * clone.
433 * @param {Node|undefined} firstChild The first child of the new cloned element.
434 * @return {!Node} The cloned node that now contains the children after
435 * startNode.
436 * @private
437 */
438goog.editor.node.getSecondHalfOfNode_ = function(node, startNode, firstChild) {
439 var secondHalf = /** @type {!Node} */(node.cloneNode(false));
440 while (startNode.nextSibling) {
441 goog.dom.appendChild(secondHalf, startNode.nextSibling);
442 }
443 if (firstChild) {
444 secondHalf.insertBefore(firstChild, secondHalf.firstChild);
445 }
446 return secondHalf;
447};

462/**
463 * Replaces the innerHTML of a node.
464 *
465 * IE has serious problems if you try to set innerHTML of an editable node with
466 * any selection. Early versions of IE tear up the old internal tree storage, to
467 * help avoid ref-counting loops. But this sometimes leaves the selection object
468 * in a bad state and leads to segfaults.
469 *
470 * Removing the nodes first prevents IE from tearing them up. This is not
471 * strictly necessary in nodes that do not have the selection. You should always
472 * use this function when setting innerHTML inside of a field.
473 *
474 * @param {Node} node A node.
475 * @param {string} html The innerHTML to set on the node.
476 */
477goog.editor.node.replaceInnerHtml = function(node, html) {
478 // Only do this IE. On gecko, we use element change events, and don't
479 // want to trigger spurious events.
480 if (goog.userAgent.IE) {
481 goog.dom.removeChildren(node);
482 }
483 node.innerHTML = html;
484};
