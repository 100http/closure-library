16 * @fileoverview Class to encapsulate an editable field that blends in with
17 * the style of the page. The field can be fixed height, grow with its
18 * contents, or have a min height after which it grows to its contents.
19 * This is a goog.editor.Field, but with blending and sizing capabilities,
20 * and avoids using an iframe whenever possible.
21 *
22 * @author nicksantos@google.com (Nick Santos)
23 * @see ../demos/editor/seamlessfield.html
24 */

175/**
176 * Sizes the iframe to its body's height.
177 * @private
178 */
179goog.editor.SeamlessField.prototype.sizeIframeToBodyHeightGecko_ = function() {
180 if (this.acquireSizeIframeLockGecko_()) {
181 var resized = false;
182 var ifr = this.getEditableIframe();
183 if (ifr) {
184 var fieldHeight = this.getIframeBodyHeightGecko_();
185
186 if (this.minHeight_) {
187 fieldHeight = Math.max(fieldHeight, this.minHeight_);
188 }
189 if (parseInt(goog.style.getStyle(ifr, 'height'), 10) != fieldHeight) {
190 ifr.style.height = fieldHeight + 'px';
191 resized = true;
192 }
193 }
194 this.releaseSizeIframeLockGecko_();
195 if (resized) {
196 this.dispatchEvent(goog.editor.Field.EventType.IFRAME_RESIZED);
197 }
198 }
199};

202/**
203 * @return {number} The height of the editable iframe's body.
204 * @private
205 */
206goog.editor.SeamlessField.prototype.getIframeBodyHeightGecko_ = function() {
207 var ifr = this.getEditableIframe();
208 var body = ifr.contentDocument.body;
209 var htmlElement = body.parentNode;
210
211
212 // If the iframe's height is 0, then the offsetHeight/scrollHeight of the
213 // HTML element in the iframe can be totally wack (i.e. too large
214 // by 50-500px). Also, in standard's mode the clientHeight is 0.
215 if (parseInt(goog.style.getStyle(ifr, 'height'), 10) === 0) {
216 goog.style.setStyle(ifr, 'height', 1 + 'px');
217 }
218
219 var fieldHeight;
220 if (goog.editor.node.isStandardsMode(body)) {
221
222 // If in standards-mode,
223 // grab the HTML element as it will contain all the field's
224 // contents. The body's height, for example, will not include that of
225 // floated images at the bottom in standards mode.
226 // Note that this value include all scrollbars *except* for scrollbars
227 // on the HTML element itself.
228 fieldHeight = htmlElement.offsetHeight;
229 } else {
230 // In quirks-mode, the body-element always seems
231 // to size to the containing window. The html-element however,
232 // sizes to the content, and can thus end up with a value smaller
233 // than its child body-element if the content is shrinking.
234 // We want to make the iframe shrink too when the content shrinks,
235 // so rather than size the iframe to the body-element, size it to
236 // the html-element.
237 fieldHeight = htmlElement.scrollHeight;
238
239 // If there is a horizontal scroll, add in the thickness of the
240 // scrollbar.
241 if (htmlElement.clientHeight != htmlElement.offsetHeight) {
242 fieldHeight += goog.editor.SeamlessField.getScrollbarWidth_();
243 }
244 }
245
246 return fieldHeight;
247};

262/**
263 * Sizes the iframe to its container div's width. The width of the div
264 * is controlled by its containing context, not by its contents.
265 * if it extends outside of it's contents, then it gets a horizontal scroll.
266 * @private
267 */
268goog.editor.SeamlessField.prototype.sizeIframeToWrapperGecko_ = function() {
269 if (this.acquireSizeIframeLockGecko_()) {
270 var ifr = this.getEditableIframe();
271 var field = this.getElement();
272 var resized = false;
273 if (ifr && field) {
274 var fieldPaddingBox;
275 var widthDiv = ifr.parentNode;
276
277 var width = widthDiv.offsetWidth;
278 if (parseInt(goog.style.getStyle(ifr, 'width'), 10) != width) {
279 fieldPaddingBox = goog.style.getPaddingBox(field);
280 ifr.style.width = width + 'px';
281 field.style.width =
282 width - fieldPaddingBox.left - fieldPaddingBox.right + 'px';
283 resized = true;
284 }
285
286 var height = widthDiv.offsetHeight;
287 if (this.isFixedHeight() &&
288 parseInt(goog.style.getStyle(ifr, 'height'), 10) != height) {
289 if (!fieldPaddingBox) {
290 fieldPaddingBox = goog.style.getPaddingBox(field);
291 }
292 ifr.style.height = height + 'px';
293 field.style.height =
294 height - fieldPaddingBox.top - fieldPaddingBox.bottom + 'px';
295 resized = true;
296 }
297
298 }
299 this.releaseSizeIframeLockGecko_();
300 if (resized) {
301 this.dispatchEvent(goog.editor.Field.EventType.IFRAME_RESIZED);
302 }
303 }
304};

307/**
308 * Perform all the sizing immediately.
309 */
310goog.editor.SeamlessField.prototype.doFieldSizingGecko = function() {
311 // Because doFieldSizingGecko can be called after a setTimeout
312 // it is possible that the field has been destroyed before this call
313 // to do the sizing is executed. Check for field existence and do nothing
314 // if it has already been destroyed.
315 if (this.getElement()) {
316 // The order of operations is important here. Sizing the iframe to the
317 // wrapper could cause the width to change, which could change the line
318 // wrapping, which could change the body height. So we need to do that
319 // first, then size the iframe to fit the body height.
320 this.sizeIframeToWrapperGecko_();
321 if (!this.isFixedHeight()) {
322 this.sizeIframeToBodyHeightGecko_();
323 }
324 }
325};

414/**
415 * Applies CSS from the wrapper-div to the field iframe.
416 */
417goog.editor.SeamlessField.prototype.inheritBlendedCSS = function() {
418 // No-op if the field isn't using an iframe.
419 if (!this.usesIframe()) {
420 return;
421 }
422 var field = this.getElement();
423 var head = goog.dom.getDomHelper(field).getElementsByTagNameAndClass(
424 goog.dom.TagName.HEAD)[0];
425 if (head) {
426 // We created this <head>, and we know the only thing we put in there
427 // is a <style> block. So it's safe to blow away all the children
428 // as part of rewriting the styles.
429 goog.dom.removeChildren(head);
430 }
431
432 // Force a cache-clearing in CssUtil - this function was called because
433 // we're applying the 'blend' for the first time, or because we
434 // *need* to recompute the blend.
435 var newCSS = this.getIframeableCss(true);
436 goog.style.installStyles(newCSS, field);
437};

494/** @override */
495goog.editor.SeamlessField.prototype.dispatchBlur = function() {
496 if (this.isEventStopped(goog.editor.Field.EventType.BLUR)) {
497 return;
498 }
499
500 goog.editor.SeamlessField.superClass_.dispatchBlur.call(this);
501
502 // Clear the selection and restore the current range back after collapsing
503 // it. The ideal solution would have been to just leave the range intact; but
504 // when there are multiple fields present on the page, its important that
505 // the selection isn't retained when we switch between the fields. We also
506 // have to make sure that the cursor position is retained when we tab in and
507 // out of a field and our approach addresses both these issues.
508 // Another point to note is that we do it on a setTimeout to allow for
509 // DOM modifications on blur. Otherwise, something like setLoremIpsum will
510 // leave a blinking cursor in the field even though it's blurred.
511 if (!goog.editor.BrowserFeature.HAS_CONTENT_EDITABLE &&
512 !goog.editor.BrowserFeature.CLEARS_SELECTION_WHEN_FOCUS_LEAVES) {
513 var win = this.getEditableDomHelper().getWindow();
514 var dragging = false;
515 goog.events.unlistenByKey(this.listenForDragOverEventKey_);
516 this.listenForDragOverEventKey_ = goog.events.listenOnce(
517 win.document.body, 'dragover',
518 function() {
519 dragging = true;
520 });
521 goog.global.setTimeout(goog.bind(function() {
522 // Do not clear the selection if we're only dragging text.
523 // This addresses a bug on FF1.5/linux where dragging fires a blur,
524 // but clearing the selection confuses Firefox's drag-and-drop
525 // implementation. For more info, see http://b/1061064
526 if (!dragging) {
527 if (this.editableDomHelper) {
528 var rng = this.getRange();
529
530 // If there are multiple fields on a page, we need to make sure that
531 // the selection isn't retained when we switch between fields. We
532 // could have collapsed the range but there is a bug in GECKO where
533 // the selection stays highlighted even though its backing range is
534 // collapsed (http://b/1390115). To get around this, we clear the
535 // selection and restore the collapsed range back in. Restoring the
536 // range is important so that the cursor stays intact when we tab out
537 // and into a field (See http://b/1790301 for additional details on
538 // this).
539 var iframeWindow = this.editableDomHelper.getWindow();
540 goog.dom.Range.clearSelection(iframeWindow);
541
542 if (rng) {
543 rng.collapse(true);
544 rng.select();
545 }
546 }
547 }
548 }, this), 0);
549 }
550};

599/** @override */
600goog.editor.SeamlessField.prototype.handleFieldLoad = function() {
601 if (this.usesIframe()) {
602 // If the CSS inheriting code screws up (e.g. makes fonts too large) and
603 // the field is sized off in goog.editor.Field.makeIframeField, then we need
604 // to size it correctly, but it needs to be visible for the browser
605 // to have fully rendered it. We need to put this on a timeout to give
606 // the browser time to render.
607 var self = this;

622/** @override */
623goog.editor.SeamlessField.prototype.attachIframe = function(iframe) {
624 this.autoDetectFixedHeight_();
625 var field = this.getOriginalElement();
626 var dh = goog.dom.getDomHelper(field);
627
628 // Grab the width/height values of the field before modifying any CSS
629 // as some of the modifications affect its size (e.g. innerHTML='')
630 // Here, we set the size of the field to fixed so there's not too much
631 // jiggling when we set the innerHTML of the field.
632 var oldWidth = field.style.width;
633 var oldHeight = field.style.height;
634 goog.style.setStyle(field, 'visibility', 'hidden');
635
636 // If there is a floated element at the bottom of the field,
637 // then it needs a clearing div at the end to cause the clientHeight
638 // to contain the entire field.
639 // Also, with css re-writing, the margins of the first/last
640 // paragraph don't seem to get included in the clientHeight. Specifically,
641 // the extra divs below force the field's clientHeight to include the
642 // margins on the first and last elements contained within it.
643 var startDiv = dh.createDom(goog.dom.TagName.DIV,
644 {'style': 'height:0;clear:both', 'innerHTML': '&nbsp;'});
645 var endDiv = startDiv.cloneNode(true);
646 field.insertBefore(startDiv, field.firstChild);
647 goog.dom.appendChild(field, endDiv);
648
649 var contentBox = goog.style.getContentBoxSize(field);
650 var width = contentBox.width;
651 var height = contentBox.height;
652
653 var html = '';
654 if (this.isFixedHeight()) {
655 html = '&nbsp;';
656
657 goog.style.setStyle(field, 'position', 'relative');
658 goog.style.setStyle(field, 'overflow', 'visible');
659
660 goog.style.setStyle(iframe, 'position', 'absolute');
661 goog.style.setStyle(iframe, 'top', '0');
662 goog.style.setStyle(iframe, 'left', '0');
663 }
664 goog.style.setSize(field, width, height);
665
666 // In strict mode, browsers put blank space at the bottom and right
667 // if a field when it has an iframe child, to fill up the remaining line
668 // height. So make the line height = 0.
669 if (goog.editor.node.isStandardsMode(field)) {
670 this.originalFieldLineHeight_ = field.style.lineHeight;
671 goog.style.setStyle(field, 'lineHeight', '0');
672 }
673
674 goog.editor.node.replaceInnerHtml(field, html);
675 // Set the initial size
676 goog.style.setSize(iframe, width, height);
677 goog.style.setSize(field, oldWidth, oldHeight);
678 goog.style.setStyle(field, 'visibility', '');
679 goog.dom.appendChild(field, iframe);
680
681 // Only write if its not IE HTTPS in which case we're waiting for load.
682 if (!this.shouldLoadAsynchronously()) {
683 var doc = iframe.contentWindow.document;
684 if (goog.editor.node.isStandardsMode(iframe.ownerDocument)) {
685 doc.open();
686 var emptyHtml = goog.html.uncheckedconversions
687 .safeHtmlFromStringKnownToSatisfyTypeContract(
688 goog.string.Const.from('HTML from constant string'),
689 '<!DOCTYPE HTML><html></html>');
690 goog.dom.safe.documentWrite(doc, emptyHtml);
691 doc.close();
692 }
693 }
694};

713/** @override */
714goog.editor.SeamlessField.prototype.writeIframeContent = function(
715 iframe, innerHtml, extraStyles) {
716 // For seamless iframes, hide the iframe while we're laying it out to
717 // prevent the flicker.
718 goog.style.setStyle(iframe, 'visibility', 'hidden');
719 var formatInfo = this.getFieldFormatInfo(extraStyles);
