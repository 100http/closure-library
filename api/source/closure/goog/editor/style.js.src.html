15/**
16 * @fileoverview Utilties for working with the styles of DOM nodes, and
17 * related to rich text editing.
18 *
19 * Many of these are not general enough to go into goog.style, and use
20 * constructs (like "isContainer") that only really make sense inside
21 * of an HTML editor.
22 *
23 * The API has been optimized for iterating over large, irregular DOM
24 * structures (with lots of text nodes), and so the API tends to be a bit
25 * more permissive than the goog.style API should be. For example,
26 * goog.style.getComputedStyle will throw an exception if you give it a
27 * text node.
28 *
29 * @author nicksantos@google.com (Nick Santos)
30 */

45/**
46 * Gets the computed or cascaded style.
47 *
48 * This is different than goog.style.getStyle_ because it returns null
49 * for text nodes (instead of throwing an exception), and never reads
50 * inline style. These two functions may need to be reconciled.
51 *
52 * @param {!Node} node Node to get style of.
53 * @param {string} stylePropertyName Property to get (must be camelCase,
54 * not css-style).
55 * @return {?string} Style value, or null if this is not an element node.
56 * @private
57 */
58goog.editor.style.getComputedOrCascadedStyle_ = function(
59 node, stylePropertyName) {
60 if (node.nodeType != goog.dom.NodeType.ELEMENT) {
61 // Only element nodes have style.
62 return null;
63 }
64 return goog.userAgent.IE ?
65 goog.style.getCascadedStyle(/** @type {!Element} */ (node),
66 stylePropertyName) :
67 goog.style.getComputedStyle(/** @type {!Element} */ (node),
68 stylePropertyName);
69};

72/**
73 * Checks whether the given element inherits display: block.
74 * @param {!Node} node The Node to check.
75 * @return {boolean} Whether the element inherits CSS display: block.
76 */
77goog.editor.style.isDisplayBlock = function(node) {
78 return goog.editor.style.getComputedOrCascadedStyle_(
79 node, 'display') == 'block';
80};

83/**
84 * Returns true if the element is a container of other non-inline HTML
85 * Note that span, strong and em tags, being inline can only contain
86 * other inline elements and are thus, not containers. Containers are elements
87 * that should not be broken up when wrapping selections with a node of an
88 * inline block styling.
89 * @param {Node} element The element to check.
90 * @return {boolean} Whether the element is a container.
91 */
92goog.editor.style.isContainer = function(element) {
93 var nodeName = element && element.nodeName;
94 return !!(element &&
95 (goog.editor.style.isDisplayBlock(element) ||
96 nodeName == goog.dom.TagName.TD ||
97 nodeName == goog.dom.TagName.TABLE ||
98 nodeName == goog.dom.TagName.LI));
99};

102/**
103 * Return the first ancestor of this node that is a container, inclusive.
104 * @see isContainer
105 * @param {Node} node Node to find the container of.
106 * @return {Element} The element which contains node.
107 */
108goog.editor.style.getContainer = function(node) {
109 // We assume that every node must have a container.
110 return /** @type {Element} */ (
111 goog.dom.getAncestor(node, goog.editor.style.isContainer, true));
112};

130goog.editor.style.cancelMouseDownHelper_ = function(e) {

139/**
140 * Makes the given element unselectable, as well as all of its children, except
141 * for text areas, text, file and url inputs.
142 * @param {Element} element The element to make unselectable.
143 * @param {goog.events.EventHandler} eventHandler An EventHandler to register
144 * the event with. Assumes when the node is destroyed, the eventHandler's
145 * listeners are destroyed as well.
146 */
147goog.editor.style.makeUnselectable = function(element, eventHandler) {
148 if (goog.editor.BrowserFeature.HAS_UNSELECTABLE_STYLE) {
149 // The mousing down on a node should not blur the focused node.
150 // This is consistent with how IE works.
151 // TODO: Consider using just the mousedown handler and not the css property.
152 eventHandler.listen(element, goog.events.EventType.MOUSEDOWN,
153 goog.editor.style.cancelMouseDownHelper_, true);
154 }
155
156 goog.style.setUnselectable(element, true);
157
158 // Make inputs and text areas selectable.
159 var inputs = element.getElementsByTagName(goog.dom.TagName.INPUT);
160 for (var i = 0, len = inputs.length; i < len; i++) {
161 var input = inputs[i];
162 if (input.type in goog.editor.style.SELECTABLE_INPUT_TYPES_) {
163 goog.editor.style.makeSelectable(input);
164 }
165 }
166 goog.array.forEach(element.getElementsByTagName(goog.dom.TagName.TEXTAREA),
167 goog.editor.style.makeSelectable);
168};

171/**
172 * Make the given element selectable.
173 *
174 * For IE this simply turns off the "unselectable" property.
175 *
176 * Under FF no descendent of an unselectable node can be selectable:
177 *
178 * https://bugzilla.mozilla.org/show_bug.cgi?id=203291
179 *
180 * So we make each ancestor of node selectable, while trying to preserve the
181 * unselectability of other nodes along that path
182 *
183 * This may cause certain text nodes which should be unselectable, to become
184 * selectable. For example:
185 *
186 * <div id=div1 style="-moz-user-select: none">
187 * Text1
188 * <span id=span1>Text2</span>
189 * </div>
190 *
191 * If we call makeSelectable on span1, then it will cause "Text1" to become
192 * selectable, since it had to make div1 selectable in order for span1 to be
193 * selectable.
194 *
195 * If "Text1" were enclosed within a <p> or <span>, then this problem would
196 * not arise. Text nodes do not have styles, so its style can't be set to
197 * unselectable.
198 *
199 * @param {Element} element The element to make selectable.
200 */
201goog.editor.style.makeSelectable = function(element) {
202 goog.style.setUnselectable(element, false);
203 if (goog.editor.BrowserFeature.HAS_UNSELECTABLE_STYLE) {
204 // Go up ancestor chain, searching for nodes that are unselectable.
205 // If such a node exists, mark it as selectable but mark its other children
206 // as unselectable so the minimum set of nodes is changed.
207 var child = element;
208 var current = /** @type {Element} */ (element.parentNode);
209 while (current && current.tagName != goog.dom.TagName.HTML) {
210 if (goog.style.isUnselectable(current)) {
211 goog.style.setUnselectable(current, false, true);
212
213 for (var i = 0, len = current.childNodes.length; i < len; i++) {
214 var node = current.childNodes[i];
215 if (node != child && node.nodeType == goog.dom.NodeType.ELEMENT) {
216 goog.style.setUnselectable(current.childNodes[i], true);
217 }
218 }
219 }
220
221 child = current;
222 current = /** @type {Element} */ (current.parentNode);
223 }
224 }
