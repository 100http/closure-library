21goog.require('goog.string');
22goog.require('goog.userAgent');

25/**
26 * Detects the version of Windows or Mac OS that is running.
27 *
28 * @private
29 * @return {string} The platform version.
30 */
31goog.userAgent.platform.determineVersion_ = function() {
32 var re;
33 if (goog.userAgent.WINDOWS) {
34 re = /Windows NT ([0-9.]+)/;
35 var match = re.exec(goog.userAgent.getUserAgentString());
36 if (match) {
37 return match[1];
38 } else {
39 return '0';
40 }
41 } else if (goog.userAgent.MAC) {
42 re = /10[_.][0-9_.]+/;
43 var match = re.exec(goog.userAgent.getUserAgentString());
44 // Note: some old versions of Camino do not report an OSX version.
45 // Default to 10.
46 return match ? match[0].replace(/_/g, '.') : '10';
47 } else if (goog.userAgent.ANDROID) {
48 re = /Android\s+([^\);]+)(\)|;)/;
49 var match = re.exec(goog.userAgent.getUserAgentString());
50 return match ? match[1] : '';
51 } else if (goog.userAgent.IPHONE || goog.userAgent.IPAD) {
52 re = /(?:iPhone|CPU)\s+OS\s+(\S+)/;
53 var match = re.exec(goog.userAgent.getUserAgentString());
54 // Report the version as x.y.z and not x_y_z
55 return match ? match[1].replace(/_/g, '.') : '';
56 }
57
58 return '';
59};
60

62/**
63 * The version of the platform. We only determine the version for Windows and
64 * Mac, since it doesn't make much sense on Linux. For Windows, we only look at
65 * the NT version. Non-NT-based versions (e.g. 95, 98, etc.) are given version
66 * 0.0
67 * @type {string}
68 */
69goog.userAgent.platform.VERSION = goog.userAgent.platform.determineVersion_();
70
71
72/**
73 * Whether the user agent platform version is higher or the same as the given
74 * version.
75 *
76 * @param {string|number} version The version to check.
77 * @return {boolean} Whether the user agent platform version is higher or the
78 * same as the given version.
79 */
80goog.userAgent.platform.isVersion = function(version) {
81 return goog.string.compareVersions(
82 goog.userAgent.platform.VERSION, version) >= 0;
83};
