36/**
37 * Rerun the initialization code to set all of the goog.userAgent constants.
38 * @suppress {accessControls}
39 */
40goog.userAgentTestUtil.reinitializeUserAgent = function() {
41 // Unfortunately we can't isolate the useragent setting in a function
42 // we can call, because things rely on it compiling to nothing when
43 // one of the ASSUME flags is set, and the compiler isn't smart enough
44 // to do that when the setting is done inside a function that's inlined.
45 goog.userAgent.OPERA = goog.labs.userAgent.browser.isOpera();
46 goog.userAgent.IE = goog.labs.userAgent.browser.isIE();
47 goog.userAgent.GECKO = goog.labs.userAgent.engine.isGecko();
48 goog.userAgent.WEBKIT = goog.labs.userAgent.engine.isWebKit();
49 goog.userAgent.MOBILE = goog.userAgent.isMobile_();
50 goog.userAgent.SAFARI = goog.userAgent.WEBKIT;
51
52 // Platform in goog.userAgent.
53 goog.userAgent.PLATFORM = goog.userAgent.determinePlatform_();
54
55 goog.userAgent.MAC = goog.labs.userAgent.platform.isMacintosh();
56 goog.userAgent.WINDOWS = goog.labs.userAgent.platform.isWindows();
57 goog.userAgent.LINUX = goog.userAgent.isLegacyLinux_();
58 goog.userAgent.X11 = goog.userAgent.isX11_();
59 goog.userAgent.ANDROID = goog.labs.userAgent.platform.isAndroid();
60 goog.userAgent.IPAD = goog.labs.userAgent.platform.isIpad();
61 goog.userAgent.IPHONE = goog.labs.userAgent.platform.isIphone();
62 goog.userAgent.VERSION = goog.userAgent.determineVersion_();
63
64 // Platform in goog.userAgent.platform.
65 goog.userAgent.platform.VERSION = goog.userAgent.platform.determineVersion_();
66
67 // Update goog.userAgent.product
68 goog.userAgent.product.ANDROID =
69 goog.labs.userAgent.browser.isAndroidBrowser();
70 goog.userAgent.product.CHROME =
71 goog.labs.userAgent.browser.isChrome();
72 goog.userAgent.product.FIREFOX =
73 goog.labs.userAgent.browser.isFirefox();
74 goog.userAgent.product.IE =
75 goog.labs.userAgent.browser.isIE();
76 goog.userAgent.product.IPAD = goog.labs.userAgent.platform.isIpad();
77 goog.userAgent.product.IPHONE = goog.userAgent.product.isIphoneOrIpod_();
78 goog.userAgent.product.OPERA = goog.labs.userAgent.browser.isOpera();
79 goog.userAgent.product.SAFARI = goog.userAgent.product.isSafariDesktop_();
80
81 // Still uses its own implementation.
82 goog.userAgent.product.VERSION = goog.userAgent.product.determineVersion_();
83
84 // goog.userAgent.keyboard
85 goog.userAgent.keyboard.MAC_KEYBOARD =
86 goog.userAgent.keyboard.determineMacKeyboard_();
87};

102/**
103 * Return whether a given user agent has been detected.
104 * @param {string} agent Value in UserAgents.
105 * @return {boolean} Whether the user agent has been detected.
106 */
107goog.userAgentTestUtil.getUserAgentDetected = function(agent) {
108 switch (agent) {
109 case goog.userAgentTestUtil.UserAgents.GECKO:
110 return goog.userAgent.GECKO;
111 case goog.userAgentTestUtil.UserAgents.IE:
112 return goog.userAgent.IE;
113 case goog.userAgentTestUtil.UserAgents.OPERA:
114 return goog.userAgent.OPERA;
115 case goog.userAgentTestUtil.UserAgents.WEBKIT:
116 return goog.userAgent.WEBKIT;
117 }
118
119 throw Error('Unrecognized user agent');
120};
