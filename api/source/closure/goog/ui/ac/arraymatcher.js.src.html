27/**
28 * Basic class for matching words in an array
29 * @constructor
30 * @param {Array<?>} rows Dictionary of items to match. Can be objects if they
31 * have a toString method that returns the value to match against.
32 * @param {boolean=} opt_noSimilar if true, do not do similarity matches for the
33 * input token against the dictionary.
34 */
35goog.ui.ac.ArrayMatcher = function(rows, opt_noSimilar) {
36 this.rows_ = rows || [];
37 this.useSimilar_ = !opt_noSimilar;
38};

46 this.rows_ = rows || [];

50/**
51 * Function used to pass matches to the autocomplete
52 * @param {string} token Token to match.
53 * @param {number} maxMatches Max number of matches to return.
54 * @param {Function} matchHandler callback to execute after matching.
55 * @param {string=} opt_fullString The full string from the input box.
56 */
57goog.ui.ac.ArrayMatcher.prototype.requestMatchingRows =
58 function(token, maxMatches, matchHandler, opt_fullString) {
59
60 var matches = this.useSimilar_ ?
61 goog.ui.ac.ArrayMatcher.getMatchesForRows(token, maxMatches, this.rows_) :
62 this.getPrefixMatches(token, maxMatches);
63
64 matchHandler(token, matches);
65};

68/**
69 * Matches the token against the specified rows, first looking for prefix
70 * matches and if that fails, then looking for similar matches.
71 *
72 * @param {string} token Token to match.
73 * @param {number} maxMatches Max number of matches to return.
74 * @param {!Array<?>} rows Rows to search for matches. Can be objects if they
75 * have a toString method that returns the value to match against.
76 * @return {!Array<?>} Rows that match.
77 */
78goog.ui.ac.ArrayMatcher.getMatchesForRows =
79 function(token, maxMatches, rows) {
80 var matches =
81 goog.ui.ac.ArrayMatcher.getPrefixMatchesForRows(token, maxMatches, rows);
82
83 if (matches.length == 0) {
84 matches = goog.ui.ac.ArrayMatcher.getSimilarMatchesForRows(token,
85 maxMatches, rows);
86 }
87 return matches;
88};

104/**
105 * Matches the token against the start of words in the row.
106 * @param {string} token Token to match.
107 * @param {number} maxMatches Max number of matches to return.
108 * @param {!Array<?>} rows Rows to search for matches. Can be objects if they have
109 * a toString method that returns the value to match against.
110 * @return {!Array<?>} Rows that match.
111 */
112goog.ui.ac.ArrayMatcher.getPrefixMatchesForRows =
113 function(token, maxMatches, rows) {
114 var matches = [];
115
116 if (token != '') {
117 var escapedToken = goog.string.regExpEscape(token);
118 var matcher = new RegExp('(^|\\W+)' + escapedToken, 'i');
119
120 for (var i = 0; i < rows.length && matches.length < maxMatches; i++) {
121 var row = rows[i];
122 if (String(row).match(matcher)) {
123 matches.push(row);
124 }
125 }
126 }
127 return matches;
128};

131/**
132 * Matches the token against similar rows, by calculating "distance" between the
133 * terms.
134 * @param {string} token Token to match.
135 * @param {number} maxMatches Max number of matches to return.
136 * @return {!Array<?>} The best maxMatches rows.
137 */
138goog.ui.ac.ArrayMatcher.prototype.getSimilarRows = function(token, maxMatches) {
139 return goog.ui.ac.ArrayMatcher.getSimilarMatchesForRows(token, maxMatches,
140 this.rows_);
141};

144/**
145 * Matches the token against similar rows, by calculating "distance" between the
146 * terms.
147 * @param {string} token Token to match.
148 * @param {number} maxMatches Max number of matches to return.
149 * @param {!Array<?>} rows Rows to search for matches. Can be objects
150 * if they have a toString method that returns the value to
151 * match against.
152 * @return {!Array<?>} The best maxMatches rows.
153 */
154goog.ui.ac.ArrayMatcher.getSimilarMatchesForRows =
155 function(token, maxMatches, rows) {
156 var results = [];
157
158 for (var index = 0; index < rows.length; index++) {
159 var row = rows[index];
160 var str = token.toLowerCase();
161 var txt = String(row).toLowerCase();
162 var score = 0;
163
164 if (txt.indexOf(str) != -1) {
165 score = parseInt((txt.indexOf(str) / 4).toString(), 10);
166
167 } else {
168 var arr = str.split('');
169
170 var lastPos = -1;
171 var penalty = 10;
172
173 for (var i = 0, c; c = arr[i]; i++) {
174 var pos = txt.indexOf(c);
175
176 if (pos > lastPos) {
177 var diff = pos - lastPos - 1;
178
179 if (diff > penalty - 5) {
180 diff = penalty - 5;
181 }
182
183 score += diff;
184
185 lastPos = pos;
186 } else {
187 score += penalty;
188 penalty += 5;
189 }
190 }
191 }
192
193 if (score < str.length * 6) {
194 results.push({
195 str: row,
196 score: score,
197 index: index
198 });
199 }
200 }
201
202 results.sort(function(a, b) {
203 var diff = a.score - b.score;
204 if (diff != 0) {
205 return diff;
206 }
207 return a.index - b.index;
208 });
209
210 var matches = [];
211 for (var i = 0; i < maxMatches && i < results.length; i++) {
212 matches.push(results[i].str);
213 }
214
215 return matches;
216};
