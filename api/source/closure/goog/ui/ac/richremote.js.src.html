15/**
16 * @fileoverview Factory class to create a rich autocomplete that will match
17 * from an array of data provided via ajax. The server returns a complex data
18 * structure that is used with client-side javascript functions to render the
19 * results.
20 *
21 * The server sends a list of the form:
22 * [["type1", {...}, {...}, ...], ["type2", {...}, {...}, ...], ...]
23 * The first element of each sublist is a string designating the type of the
24 * hashes in the sublist, each of which represents one match. The type string
25 * must be the name of a function(item) which converts the hash into a rich
26 * row that contains both a render(node, token) and a select(target) method.
27 * The render method is called by the renderer when rendering the rich row,
28 * and the select method is called by the RichInputHandler when the rich row is
29 * selected.
30 *
31 * @see ../../demos/autocompleterichremote.html
32 */
33
34goog.provide('goog.ui.ac.RichRemote');

36goog.require('goog.ui.ac.AutoComplete');
37goog.require('goog.ui.ac.Remote');
38goog.require('goog.ui.ac.Renderer');
39goog.require('goog.ui.ac.RichInputHandler');
40goog.require('goog.ui.ac.RichRemoteArrayMatcher');

44/**
45 * Factory class to create a rich autocomplete widget that autocompletes an
46 * inputbox or textarea from data provided via ajax. The server returns a
47 * complex data structure that is used with client-side javascript functions to
48 * render the results.
49 *
50 * This class makes use of goog.html.legacyconversions and provides no
51 * HTML-type-safe alternative. As such, it is not compatible with
52 * code that sets goog.html.legacyconversions.ALLOW_LEGACY_CONVERSIONS to
53 * false.
54 *
55 * @param {string} url The Uri which generates the auto complete matches.
56 * @param {Element} input Input element or text area.
57 * @param {boolean=} opt_multi Whether to allow multiple entries; defaults
58 * to false.
59 * @param {boolean=} opt_useSimilar Whether to use similar matches; e.g.
60 * "gost" => "ghost".
61 * @constructor
62 * @extends {goog.ui.ac.Remote}
63 */
64goog.ui.ac.RichRemote = function(url, input, opt_multi, opt_useSimilar) {
65 // Create a custom renderer that renders rich rows. The renderer calls
66 // row.render(node, token) for each row.
67 var customRenderer = {};
68 customRenderer.renderRow = function(row, token, node) {
69 return row.data.render(node, token);
70 };
71
72 /**
73 * A standard renderer that uses a custom row renderer to display the
74 * rich rows generated by this autocomplete widget.
75 * @type {goog.ui.ac.Renderer}
76 * @private
77 */
78 var renderer = new goog.ui.ac.Renderer(null, customRenderer);
79 this.renderer_ = renderer;
80
81 /**
82 * A remote matcher that parses rich results returned by the server.
83 * @type {goog.ui.ac.RichRemoteArrayMatcher}
84 * @private
85 */
86 var matcher = new goog.ui.ac.RichRemoteArrayMatcher(url,
87 !opt_useSimilar);
88 this.matcher_ = matcher;
89
90 /**
91 * An input handler that calls select on a row when it is selected.
92 * @type {goog.ui.ac.RichInputHandler}
93 * @private
94 */
95 var inputhandler = new goog.ui.ac.RichInputHandler(null, null,
96 !!opt_multi, 300);
97
98 // Create the widget and connect it to the input handler.
99 goog.ui.ac.AutoComplete.call(this, matcher, renderer, inputhandler);
100 inputhandler.attachAutoComplete(this);
101 inputhandler.attachInputs(input);
102};
103goog.inherits(goog.ui.ac.RichRemote, goog.ui.ac.Remote);
