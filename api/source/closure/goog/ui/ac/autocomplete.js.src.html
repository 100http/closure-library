32/**
33 * This is the central manager class for an AutoComplete instance. The matcher
34 * can specify disabled rows that should not be hilited or selected by
35 * implementing <code>isRowDisabled(row):boolean</code> for each autocomplete
36 * row. No row will be considered disabled if this method is not implemented.
37 *
38 * @param {Object} matcher A data source and row matcher, implements
39 * <code>requestMatchingRows(token, maxMatches, matchCallback)</code>.
40 * @param {goog.events.EventTarget} renderer An object that implements
41 * <code>
42 * isVisible():boolean<br>
43 * renderRows(rows:Array, token:string, target:Element);<br>
44 * hiliteId(row-id:number);<br>
45 * dismiss();<br>
46 * dispose():
47 * </code>.
48 * @param {Object} selectionHandler An object that implements
49 * <code>
50 * selectRow(row);<br>
51 * update(opt_force);
52 * </code>.
53 *
54 * @constructor
55 * @extends {goog.events.EventTarget}
56 */

58 goog.events.EventTarget.call(this);

372/**
373 * Generic event handler that handles any events this object is listening to.
374 * @param {goog.events.Event} e Event Object.
375 */
376goog.ui.ac.AutoComplete.prototype.handleEvent = function(e) {
377 var matcher = /** @type {?goog.ui.ac.AutoComplete.Matcher} */ (this.matcher_);
378
379 if (e.target == this.renderer_) {
380 switch (e.type) {
381 case goog.ui.ac.AutoComplete.EventType.HILITE:
382 this.hiliteId(/** @type {number} */ (e.row));
383 break;
384
385 case goog.ui.ac.AutoComplete.EventType.SELECT:
386 var rowDisabled = false;
387
388 // e.row can be either a valid row id or empty.
389 if (goog.isNumber(e.row)) {
390 var rowId = e.row;
391 var index = this.getIndexOfId(rowId);
392 var row = this.rows_[index];
393
394 // Make sure the row selected is not a disabled row.
395 rowDisabled = !!row && matcher.isRowDisabled &&
396 matcher.isRowDisabled(row);
397 if (row && !rowDisabled && this.hiliteId_ != rowId) {
398 // Event target row not currently highlighted - fix the mismatch.
399 this.hiliteId(rowId);
400 }
401 }
402 if (!rowDisabled) {
403 // Note that rowDisabled can be false even if e.row does not
404 // contain a valid row ID; at least one client depends on us
405 // proceeding anyway.
406 this.selectHilited();
407 }
408 break;
409
410 case goog.ui.ac.AutoComplete.EventType.CANCEL_DISMISS:
411 this.cancelDelayedDismiss();
412 break;
413
414 case goog.ui.ac.AutoComplete.EventType.DISMISS:
415 this.dismissOnDelay();
416 break;
417 }
418 }
419};

538/**
539 * Moves the hilite to the next non-disabled row.
540 * Calls renderer.hiliteId() when there's something to do.
541 * @return {boolean} Returns true on a successful hilite.
542 */
543goog.ui.ac.AutoComplete.prototype.hiliteNext = function() {
544 var lastId = this.firstRowId_ + this.rows_.length - 1;
545 var toHilite = this.hiliteId_;
546 // Hilite the next row, skipping any disabled rows.
547 for (var i = 0; i < this.rows_.length; i++) {
548 // Increment to the next row.
549 if (toHilite >= this.firstRowId_ && toHilite < lastId) {
550 toHilite++;
551 } else if (toHilite == -1) {
552 toHilite = this.firstRowId_;
553 } else if (this.allowFreeSelect_ && toHilite == lastId) {
554 this.hiliteId(-1);
555 return false;
556 } else if (this.wrap_ && toHilite == lastId) {
557 toHilite = this.firstRowId_;
558 } else {
559 return false;
560 }
561
562 if (this.hiliteId(toHilite)) {
563 return true;
564 }
565 }
566 return false;
567};

570/**
571 * Moves the hilite to the previous non-disabled row. Calls
572 * renderer.hiliteId() when there's something to do.
573 * @return {boolean} Returns true on a successful hilite.
574 */
575goog.ui.ac.AutoComplete.prototype.hilitePrev = function() {
576 var lastId = this.firstRowId_ + this.rows_.length - 1;
577 var toHilite = this.hiliteId_;
578 // Hilite the previous row, skipping any disabled rows.
579 for (var i = 0; i < this.rows_.length; i++) {
580 // Decrement to the previous row.
581 if (toHilite > this.firstRowId_) {
582 toHilite--;
583 } else if (this.allowFreeSelect_ && toHilite == this.firstRowId_) {
584 this.hiliteId(-1);
585 return false;
586 } else if (this.wrap_ && (toHilite == -1 || toHilite == this.firstRowId_)) {
587 toHilite = lastId;
588 } else {
589 return false;
590 }
591
592 if (this.hiliteId(toHilite)) {
593 return true;
594 }
595 }
596 return false;
597};

600/**
601 * Hilites the id if it's valid and the row is not disabled, otherwise does
602 * nothing.
603 * @param {number} id A row id (not index).
604 * @return {boolean} Whether the id was hilited. Returns false if the row is
605 * disabled.
606 */
607goog.ui.ac.AutoComplete.prototype.hiliteId = function(id) {
608 var index = this.getIndexOfId(id);
609 var row = this.rows_[index];
610 var rowDisabled = !!row && this.matcher_.isRowDisabled &&
611 this.matcher_.isRowDisabled(row);
612 if (!rowDisabled) {
613 this.hiliteId_ = id;
614 this.renderer_.hiliteId(id);
615 return index != -1;
616 }
617 return false;
618};

632/**
633 * If there are any current matches, this passes the hilited row data to
634 * <code>selectionHandler.selectRow()</code>
635 * @return {boolean} Whether there are any current matches.
636 */
637goog.ui.ac.AutoComplete.prototype.selectHilited = function() {
638 var index = this.getIndexOfId(this.hiliteId_);
639 if (index != -1) {
640 var selectedRow = this.rows_[index];
641 var suppressUpdate = this.selectionHandler_.selectRow(selectedRow);
642 if (this.triggerSuggestionsOnUpdate_) {
643 this.token_ = null;
644 this.dismissOnDelay();
645 } else {
646 this.dismiss();
647 }
648 if (!suppressUpdate) {
649 this.dispatchEvent({
650 type: goog.ui.ac.AutoComplete.EventType.UPDATE,
651 row: selectedRow,
652 index: index
653 });
654 if (this.triggerSuggestionsOnUpdate_) {
655 this.selectionHandler_.update(true);
656 }
657 }
658 return true;
659 } else {
660 this.dismiss();
661 this.dispatchEvent(
662 {
663 type: goog.ui.ac.AutoComplete.EventType.UPDATE,
664 row: null,
665 index: null
666 });
667 return false;
668 }
669};

751/**
752 * Callback passed to Matcher when requesting matches for a token.
753 * This might be called synchronously, or asynchronously, or both, for
754 * any implementation of a Matcher.
755 * If the Matcher calls this back, with the same token this AutoComplete
756 * has set currently, then this will package the matching rows in object
757 * of the form
758 * <pre>
759 * {
760 * id: an integer ID unique to this result set and AutoComplete instance,
761 * data: the raw row data from Matcher
762 * }
763 * </pre>
764 *
765 * @param {string} matchedToken Token that corresponds with the rows.
766 * @param {!Array<?>} rows Set of data that match the given token.
767 * @param {(boolean|goog.ui.ac.RenderOptions)=} opt_options If true,
768 * keeps the currently hilited (by index) element hilited. If false not.
769 * Otherwise a RenderOptions object.
770 * @private
771 */
772goog.ui.ac.AutoComplete.prototype.matchListener_ =
773 function(matchedToken, rows, opt_options) {
774 if (this.token_ != matchedToken) {
775 // Matcher's response token doesn't match current token.
776 // This is probably an async response that came in after
777 // the token was changed, so don't do anything.
778 return;
779 }
780
781 this.renderRows(rows, opt_options);
782};

785/**
786 * Renders the rows and adds highlighting.
787 * @param {!Array<?>} rows Set of data that match the given token.
788 * @param {(boolean|goog.ui.ac.RenderOptions)=} opt_options If true,
789 * keeps the currently hilited (by index) element hilited. If false not.
790 * Otherwise a RenderOptions object.
791 */
792goog.ui.ac.AutoComplete.prototype.renderRows = function(rows, opt_options) {
793 // The optional argument should be a RenderOptions object. It can be a
794 // boolean for backwards compatibility, defaulting to false.
795 var optionsObj = goog.typeOf(opt_options) == 'object' && opt_options;
796
797 var preserveHilited =
798 optionsObj ? optionsObj.getPreserveHilited() : opt_options;
799 var indexToHilite = preserveHilited ? this.getIndexOfId(this.hiliteId_) : -1;
800
801 // Current token matches the matcher's response token.
802 this.firstRowId_ += this.rows_.length;
803 this.rows_ = rows;
804 var rendRows = [];
805 for (var i = 0; i < rows.length; ++i) {
806 rendRows.push({
807 id: this.getIdOfIndex_(i),
808 data: rows[i]
809 });
810 }
811
812 var anchor = null;
813 if (this.target_) {
814 anchor = this.inputToAnchorMap_[goog.getUid(this.target_)] || this.target_;
815 }
816 this.renderer_.setAnchorElement(anchor);
817 this.renderer_.renderRows(rendRows, this.token_, this.target_);
818
819 var autoHilite = this.autoHilite_;
820 if (optionsObj && optionsObj.getAutoHilite() !== undefined) {
821 autoHilite = optionsObj.getAutoHilite();
822 }
823 this.hiliteId_ = -1;
824 if ((autoHilite || indexToHilite >= 0) &&
825 rendRows.length != 0 &&
826 this.token_) {
827 if (indexToHilite >= 0) {
828 this.hiliteId(this.getIdOfIndex_(indexToHilite));
829 } else {
830 // Hilite the first non-disabled row.
831 this.hiliteNext();
832 }
833 }
834 this.dispatchEvent(goog.ui.ac.AutoComplete.EventType.SUGGESTIONS_UPDATE);
835};
