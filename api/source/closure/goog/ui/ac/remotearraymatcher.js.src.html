31/**
32 * An array matcher that requests matches via ajax.
33 * @param {string} url The Uri which generates the auto complete matches. The
34 * search term is passed to the server as the 'token' query param.
35 * @param {boolean=} opt_noSimilar If true, request that the server does not do
36 * similarity matches for the input token against the dictionary.
37 * The value is sent to the server as the 'use_similar' query param which is
38 * either "1" (opt_noSimilar==false) or "0" (opt_noSimilar==true).
39 * @param {goog.net.XmlHttpFactory=} opt_xmlHttpFactory Specify the
40 * XmlHttpFactory used to retrieve the matches.
41 * @constructor
42 * @extends {goog.Disposable}
43 */
44goog.ui.ac.RemoteArrayMatcher =
45 function(url, opt_noSimilar, opt_xmlHttpFactory) {
46 goog.Disposable.call(this);
47
48 /**
49 * The base URL for the ajax call. The token and max_matches are added as
50 * query params.
51 * @type {string}
52 * @private
53 */
54 this.url_ = url;
55
56 /**
57 * Whether similar matches should be found as well. This is sent as a hint
58 * to the server only.
59 * @type {boolean}
60 * @private
61 */
62 this.useSimilar_ = !opt_noSimilar;
63
64 /**
65 * The XhrIo object used for making remote requests. When a new request
66 * is made, the current one is aborted and the new one sent.
67 * @type {goog.net.XhrIo}
68 * @private
69 */
70 this.xhr_ = new goog.net.XhrIo(opt_xmlHttpFactory);
71};
72goog.inherits(goog.ui.ac.RemoteArrayMatcher, goog.Disposable);

146/**
147 * Builds a complete GET-style URL, given the base URI and autocomplete related
148 * parameter values.
149 * <b>Override this to build any customized lookup URLs.</b>
150 * <b>Can be used to change request method and any post content as well.</b>
151 * @param {string} uri The base URI of the request target.
152 * @param {string} token Current token in autocomplete.
153 * @param {number} maxMatches Maximum number of matches required.
154 * @param {boolean} useSimilar A hint to the server.
155 * @param {string=} opt_fullString Complete text in the input element.
156 * @return {?string} The complete url. Return null if no request should be sent.
157 * @protected
158 */
159goog.ui.ac.RemoteArrayMatcher.prototype.buildUrl = function(uri,
160 token, maxMatches, useSimilar, opt_fullString) {
161 var url = new goog.Uri(uri);
162 url.setParameterValue('token', token);
163 url.setParameterValue('max_matches', String(maxMatches));
164 url.setParameterValue('use_similar', String(Number(useSimilar)));
165 return url.toString();
166};

186/**
187 * Parses and retrieves the array of suggestions from XHR response.
188 * <b>Override this if the response is not a simple JSON array.</b>
189 * @param {string} responseText The XHR response text.
190 * @return {Array<string>} The array of suggestions.
191 * @protected
192 */
193goog.ui.ac.RemoteArrayMatcher.prototype.parseResponseText = function(
194 responseText) {
195
196 var matches = [];
197 // If there is no response text, unsafeParse will throw a syntax error.
198 if (responseText) {
199 /** @preserveTry */
200 try {
201 matches = goog.json.unsafeParse(responseText);
202 } catch (exception) {
203 }
204 }
205 return /** @type {Array<string>} */ (matches);
206};

209/**
210 * Handles the XHR response.
211 * @param {string} token The XHR autocomplete token.
212 * @param {Function} matchHandler The AutoComplete match handler.
213 * @param {goog.events.Event} event The XHR success event.
214 */
215goog.ui.ac.RemoteArrayMatcher.prototype.xhrCallback = function(token,
216 matchHandler, event) {
217 var text = event.target.getResponseText();
218 matchHandler(token, this.parseResponseText(text));
219};

222/**
223 * Retrieve a set of matching rows from the server via ajax.
224 * @param {string} token The text that should be matched; passed to the server
225 * as the 'token' query param.
226 * @param {number} maxMatches The maximum number of matches requested from the
227 * server; passed as the 'max_matches' query param. The server is
228 * responsible for limiting the number of matches that are returned.
229 * @param {Function} matchHandler Callback to execute on the result after
230 * matching.
231 * @param {string=} opt_fullString The full string from the input box.
232 */
233goog.ui.ac.RemoteArrayMatcher.prototype.requestMatchingRows =
234 function(token, maxMatches, matchHandler, opt_fullString) {
235
236 if (!this.shouldRequestMatches(this.url_, token, maxMatches, this.useSimilar_,
237 opt_fullString)) {
238 return;
239 }
240 // Set the query params on the URL.
241 var url = this.buildUrl(this.url_, token, maxMatches, this.useSimilar_,
242 opt_fullString);
243 if (!url) {
244 // Do nothing if there is no URL.
245 return;
246 }
247
248 // The callback evals the server response and calls the match handler on
249 // the array of matches.
250 var callback = goog.bind(this.xhrCallback, this, token, matchHandler);
251
252 // Abort the current request and issue the new one; prevent requests from
253 // being queued up by the browser with a slow server
254 if (this.xhr_.isActive()) {
255 this.xhr_.abort();
256 }
257 // This ensures if previous XHR is aborted or ends with error, the
258 // corresponding success-callbacks are cleared.
259 if (this.lastListenerKey_) {
260 goog.events.unlistenByKey(this.lastListenerKey_);
261 }
262 // Listen once ensures successful callback gets cleared by itself.
263 this.lastListenerKey_ = goog.events.listenOnce(this.xhr_,
264 goog.net.EventType.SUCCESS, callback);
265 this.xhr_.send(url, this.method_, this.content_, this.headers_);
266};
