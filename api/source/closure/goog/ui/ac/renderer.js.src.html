49/**
50 * Class for rendering the results of an auto-complete in a drop down list.
51 *
52 * @constructor
53 * @param {Element=} opt_parentNode optional reference to the parent element
54 * that will hold the autocomplete elements. goog.dom.getDocument().body
55 * will be used if this is null.
56 * @param {?({renderRow}|{render})=} opt_customRenderer Custom full renderer to
57 * render each row. Should be something with a renderRow or render method.
58 * @param {boolean=} opt_rightAlign Determines if the autocomplete will always
59 * be right aligned. False by default.
60 * @param {boolean=} opt_useStandardHighlighting Determines if standard
61 * highlighting should be applied to each row of data. Standard highlighting
62 * bolds every matching substring for a given token in each row. True by
63 * default.
64 * @extends {goog.events.EventTarget}

476/**
477 * Toggle the ARIA markup to add popup semantics when the target is shown and
478 * to remove them when it is hidden.
479 * @param {boolean} isShown Whether the menu is being shown.
480 * @private
481 */
482goog.ui.ac.Renderer.prototype.toggleAriaMarkup_ = function(isShown) {
483 if (!this.target_) {
484 return;
485 }
486
487 goog.a11y.aria.setState(this.target_,
488 goog.a11y.aria.State.HASPOPUP,
489 isShown);
490 goog.a11y.aria.setState(goog.asserts.assert(this.element_),
491 goog.a11y.aria.State.EXPANDED,
492 isShown);
493 goog.a11y.aria.setState(this.target_,
494 goog.a11y.aria.State.EXPANDED,
495 isShown);
496 if (isShown) {
497 goog.a11y.aria.setState(this.target_,
498 goog.a11y.aria.State.OWNS,
499 this.element_.id);
500 } else {
501 goog.a11y.aria.removeState(this.target_,
502 goog.a11y.aria.State.OWNS);
503 goog.a11y.aria.setActiveDescendant(this.target_,
504 null);
505 }

517/**
518 * Sets the 'active' class of the nth item.
519 * @param {number} index Index of the item to highlight.
520 */
521goog.ui.ac.Renderer.prototype.hiliteRow = function(index) {
522 var row = index >= 0 && index < this.rows_.length ?
523 this.rows_[index] : undefined;
524 var rowDiv = index >= 0 && index < this.rowDivs_.length ?
525 this.rowDivs_[index] : undefined;
526
527 var evtObj = /** @lends {goog.events.Event.prototype} */ ({
528 type: goog.ui.ac.AutoComplete.EventType.ROW_HILITE,
529 rowNode: rowDiv,
530 row: row ? row.data : null
531 });
532 if (this.dispatchEvent(evtObj)) {
533 this.hiliteNone();
534 this.hilitedRow_ = index;
535 if (rowDiv) {
536 goog.dom.classlist.addAll(rowDiv, [this.activeClassName,
537 this.legacyActiveClassName_]);
538 if (this.target_) {
539 goog.a11y.aria.setActiveDescendant(this.target_, rowDiv);
540 }
541 goog.style.scrollIntoContainerView(rowDiv, this.element_);
542 }
543 }
544};
545
546
547/**
548 * Removes the 'active' class from the currently selected row.
549 */
550goog.ui.ac.Renderer.prototype.hiliteNone = function() {
551 if (this.hilitedRow_ >= 0) {
552 goog.dom.classlist.removeAll(
553 goog.asserts.assert(this.rowDivs_[this.hilitedRow_]),
554 [this.activeClassName, this.legacyActiveClassName_]);
555 }
556};
557
558
559/**
560 * Sets the 'active' class of the item with a given id.
561 * @param {number} id Id of the row to hilight. If id is -1 then no rows get
562 * hilited.
563 */
564goog.ui.ac.Renderer.prototype.hiliteId = function(id) {
565 if (id == -1) {
566 this.hiliteRow(-1);
567 } else {
568 for (var i = 0; i < this.rows_.length; i++) {
569 if (this.rows_[i].id == id) {
570 this.hiliteRow(i);
571 return;
572 }
573 }
574 }
575};

592/**
593 * If the main HTML element hasn't been made yet, creates it and appends it
594 * to the parent.
595 * @private
596 */
597goog.ui.ac.Renderer.prototype.maybeCreateElement_ = function() {
598 if (!this.element_) {
599 // Make element and add it to the parent
600 var el = this.dom_.createDom(goog.dom.TagName.DIV, {style: 'display:none'});
601 if (this.showScrollbarsIfTooLarge_) {
602 // Make sure that the dropdown will get scrollbars if it isn't large
603 // enough to show all rows.
604 el.style.overflowY = 'auto';
605 }
606 this.element_ = el;
607 this.setMenuClasses_(el);
608 goog.a11y.aria.setRole(el, goog.a11y.aria.Role.LISTBOX);
609
610 el.id = goog.ui.IdGenerator.getInstance().getNextUniqueId();
611
612 this.dom_.appendChild(this.parent_, el);
613
614 // Add this object as an event handler
615 goog.events.listen(el, goog.events.EventType.CLICK,
616 this.handleClick_, false, this);
617 goog.events.listen(el, goog.events.EventType.MOUSEDOWN,
618 this.handleMouseDown_, false, this);
619 goog.events.listen(el, goog.events.EventType.MOUSEOVER,
620 this.handleMouseOver_, false, this);
621 }
622};

625/**
626 * Redraw (or draw if this is the first call) the rendered auto-complete drop
627 * down.
628 */
629goog.ui.ac.Renderer.prototype.redraw = function() {
630 // Create the element if it doesn't yet exist
631 this.maybeCreateElement_();
632
633 // For top aligned with target (= bottom aligned element),
634 // we need to hide and then add elements while hidden to prevent
635 // visible repositioning
636 if (this.topAlign_) {
637 this.element_.style.visibility = 'hidden';
638 }
639
640 if (this.widthProvider_) {
641 var width = this.widthProvider_.clientWidth + 'px';
642 this.element_.style.minWidth = width;
643 }
644
645 // Remove the current child nodes
646 this.rowDivs_.length = 0;
647 this.dom_.removeChildren(this.element_);
648
649 // Generate the new rows (use forEach so we can change rows_ from an
650 // array to a different datastructure if required)
651 if (this.customRenderer_ && this.customRenderer_.render) {
652 this.customRenderer_.render(this, this.element_, this.rows_, this.token_);
653 } else {
654 var curRow = null;
655 goog.array.forEach(this.rows_, function(row) {
656 row = this.renderRowHtml(row, this.token_);
657 if (this.topAlign_) {
658 // Aligned with top of target = best match at bottom
659 this.element_.insertBefore(row, curRow);
660 } else {
661 this.dom_.appendChild(this.element_, row);
662 }
663 curRow = row;
664 }, this);
665 }
666
667 // Don't show empty result sets
668 if (this.rows_.length == 0) {
669 this.dismiss();
670 return;
671 } else {
672 this.show();
673 }
674
675 this.reposition();
676
677 // Make the autocompleter unselectable, so that it
678 // doesn't steal focus from the input field when clicked.
679 goog.style.setUnselectable(this.element_, true);
680};

698/**
699 * Repositions the auto complete popup relative to the location node, if it
700 * exists and the auto position has been set.
701 */
702goog.ui.ac.Renderer.prototype.reposition = function() {
703 if (this.target_ && this.reposition_) {
704 var anchorElement = this.anchorElement_ || this.target_;
705 var anchorCorner = this.getAnchorCorner();
706
707 var overflowMode = goog.positioning.Overflow.ADJUST_X_EXCEPT_OFFSCREEN;
708 if (this.showScrollbarsIfTooLarge_) {
709 // positionAtAnchor will set the height of this.element_ when it runs
710 // (because of RESIZE_HEIGHT), and it will never increase it relative to
711 // its current value when it runs again. But if the user scrolls their
712 // page, then we might actually want a bigger height when the dropdown is
713 // displayed next time. So we clear the height before calling
714 // positionAtAnchor, so it is free to set the height as large as it
715 // chooses.
716 this.element_.style.height = '';
717 overflowMode |= goog.positioning.Overflow.RESIZE_HEIGHT;
718 }
719
720 goog.positioning.positionAtAnchor(
721 anchorElement, anchorCorner,
722 this.element_, goog.positioning.flipCornerVertical(anchorCorner),
723 null, null, overflowMode);
724
725 if (this.topAlign_) {
726 // This flickers, but is better than the alternative of positioning
727 // in the wrong place and then moving.
728 this.element_.style.visibility = 'visible';
729 }
730 }
731};

808 * this.highlightAllTokens_ value.
809 *
810 * @param {Node} node Node to match.
811 * @param {string|Array<string>} tokenOrArray Token to match or array of tokens
812 * to match. By default, only the first match will be highlighted. If
813 * highlightAllTokens is set, then all tokens appearing at the start of a
814 * word, in whatever order and however many times, will be highlighted.
815 * @private
816 */
817goog.ui.ac.Renderer.prototype.startHiliteMatchingText_ =
818 function(node, tokenOrArray) {
819 this.wasHighlightedAtLeastOnce_ = false;
820 this.hiliteMatchingText_(node, tokenOrArray);

824/**
825 * @param {Node} node Node to match.
826 * @param {string|Array<string>} tokenOrArray Token to match or array of tokens
827 * to match.
828 * @private
829 */
830goog.ui.ac.Renderer.prototype.hiliteMatchingText_ =
831 function(node, tokenOrArray) {
832 if (!this.highlightAllTokens_ && this.wasHighlightedAtLeastOnce_) {
833 return;
834 }
835
836 if (node.nodeType == goog.dom.NodeType.TEXT) {
837 var rest = null;
838 if (goog.isArray(tokenOrArray) &&
839 tokenOrArray.length > 1 &&
840 !this.highlightAllTokens_) {
841 rest = goog.array.slice(tokenOrArray, 1);
842 }
843
844 var token = this.getTokenRegExp_(tokenOrArray);
845 if (token.length == 0) return;
846
847 var text = node.nodeValue;
848
849 // Create a regular expression to match a token at the beginning of a line
850 // or preceeded by non-alpha-numeric characters. Note: token could have |
851 // operators in it, so we need to parenthesise it before adding \b to it.
852 // or preceeded by non-alpha-numeric characters
853 //
854 // NOTE(user): When using word matches, this used to have
855 // a (^|\\W+) clause where it now has \\b but it caused various
856 // browsers to hang on really long strings. The (^|\\W+) matcher was also
857 // unnecessary, because \b already checks that the character before the
858 // is a non-word character, and ^ matches the start of the line or following
859 // a line terminator character, which is also \W. The regexp also used to
860 // have a capturing match before the \\b, which would capture the
861 // non-highlighted content, but that caused the regexp matching to run much
862 // slower than the current version.
863 var re = this.matchWordBoundary_ ?
864 new RegExp('\\b(?:' + token + ')', 'gi') :
865 new RegExp(token, 'gi');
866 var textNodes = [];
867 var lastIndex = 0;
868
869 // Find all matches
870 // Note: text.split(re) has inconsistencies between IE and FF, so
871 // manually recreated the logic
872 var match = re.exec(text);
873 var numMatches = 0;
874 while (match) {
875 numMatches++;
876 textNodes.push(text.substring(lastIndex, match.index));
877 textNodes.push(text.substring(match.index, re.lastIndex));
878 lastIndex = re.lastIndex;
879 match = re.exec(text);
880 }
881 textNodes.push(text.substring(lastIndex));
882
883 // Replace the tokens with bolded text. Each pair of textNodes
884 // (starting at index idx) includes a node of text before the bolded
885 // token, and a node (at idx + 1) consisting of what should be
886 // enclosed in bold tags.
887 if (textNodes.length > 1) {
888 var maxNumToBold = !this.highlightAllTokens_ ? 1 : numMatches;
889 for (var i = 0; i < maxNumToBold; i++) {
890 var idx = 2 * i;
891
892 node.nodeValue = textNodes[idx];
893 var boldTag = this.dom_.createElement(goog.dom.TagName.B);
894 boldTag.className = this.highlightedClassName;
895 this.dom_.appendChild(boldTag,
896 this.dom_.createTextNode(textNodes[idx + 1]));
897 boldTag = node.parentNode.insertBefore(boldTag, node.nextSibling);
898 node.parentNode.insertBefore(this.dom_.createTextNode(''),
899 boldTag.nextSibling);
900 node = boldTag.nextSibling;
901 }
902
903 // Append the remaining text nodes to the end.
904 var remainingTextNodes = goog.array.slice(textNodes, maxNumToBold * 2);
905 node.nodeValue = remainingTextNodes.join('');
906
907 this.wasHighlightedAtLeastOnce_ = true;
908 } else if (rest) {
909 this.hiliteMatchingText_(node, rest);
910 }
911 } else {
912 var child = node.firstChild;
913 while (child) {
914 var nextChild = child.nextSibling;
915 this.hiliteMatchingText_(child, tokenOrArray);
916 child = nextChild;
917 }
918 }
919};

922/**
923 * Transforms a token into a string ready to be put into the regular expression
924 * in hiliteMatchingText_.
925 * @param {string|Array<string>} tokenOrArray The token or array to get the
926 * regex string from.
927 * @return {string} The regex-ready token.
928 * @private
929 */
930goog.ui.ac.Renderer.prototype.getTokenRegExp_ = function(tokenOrArray) {
931 var token = '';
932
933 if (!tokenOrArray) {
934 return token;
935 }
936
937 if (goog.isArray(tokenOrArray)) {
938 // Remove invalid tokens from the array, which may leave us with nothing.
939 tokenOrArray = goog.array.filter(tokenOrArray, function(str) {
940 return !goog.string.isEmptyOrWhitespace(goog.string.makeSafe(str));
941 });
942 }
943
944 // If highlighting all tokens, join them with '|' so the regular expression
945 // will match on any of them.
946 if (this.highlightAllTokens_) {
947 if (goog.isArray(tokenOrArray)) {
948 var tokenArray = goog.array.map(tokenOrArray, goog.string.regExpEscape);
949 token = tokenArray.join('|');
950 } else {
951 // Remove excess whitespace from the string so bars will separate valid
952 // tokens in the regular expression.
953 token = goog.string.collapseWhitespace(tokenOrArray);
954
955 token = goog.string.regExpEscape(token);
956 token = token.replace(/ /g, '|');
957 }
958 } else {
959 // Not highlighting all matching tokens. If tokenOrArray is a string, use
960 // that as the token. If it is an array, use the first element in the
961 // array.
962 // TODO(user): why is this this way?. We should match against all
963 // tokens in the array, but only accept the first match.
964 if (goog.isArray(tokenOrArray)) {
965 token = tokenOrArray.length > 0 ?
966 goog.string.regExpEscape(tokenOrArray[0]) : '';
967 } else {
968 // For the single-match string token, we refuse to match anything if
969 // the string begins with a non-word character, as matches by definition
970 // can only occur at the start of a word. (This also handles the
971 // goog.string.isEmptyOrWhitespace(goog.string.makeSafe(tokenOrArray))
972 // case.)
973 if (!/^\W/.test(tokenOrArray)) {
974 token = goog.string.regExpEscape(tokenOrArray);
975 }
976 }
977 }
978
979 return token;
980};
