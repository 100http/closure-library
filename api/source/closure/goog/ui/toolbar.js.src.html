15/**
16 * @fileoverview A toolbar class that hosts {@link goog.ui.Control}s such as
17 * buttons and menus, along with toolbar-specific renderers of those controls.
18 *
19 * @author attila@google.com (Attila Bodis)
20 * @see ../demos/toolbar.html
21 */
22
23goog.provide('goog.ui.Toolbar');
24
25goog.require('goog.ui.Container');
26goog.require('goog.ui.ToolbarRenderer');
27
28
29
30/**
31 * A toolbar class, implemented as a {@link goog.ui.Container} that defaults to
32 * having a horizontal orientation and {@link goog.ui.ToolbarRenderer} as its
33 * renderer.
34 * @param {goog.ui.ToolbarRenderer=} opt_renderer Renderer used to render or
35 * decorate the toolbar; defaults to {@link goog.ui.ToolbarRenderer}.
36 * @param {?goog.ui.Container.Orientation=} opt_orientation Toolbar orientation;
37 * defaults to {@code HORIZONTAL}.
38 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper.
39 * @constructor
40 * @extends {goog.ui.Container}
41 */
42goog.ui.Toolbar = function(opt_renderer, opt_orientation, opt_domHelper) {
43 goog.ui.Container.call(this, opt_orientation, opt_renderer ||
44 goog.ui.ToolbarRenderer.getInstance(), opt_domHelper);
45};
46goog.inherits(goog.ui.Toolbar, goog.ui.Container);
47
48
49/** @override */
50goog.ui.Toolbar.prototype.handleFocus = function(e) {
51 goog.ui.Toolbar.base(this, 'handleFocus', e);
52 // Highlight the first highlightable item on focus via the keyboard for ARIA
53 // spec compliance. Do not highlight the item if the mouse button is pressed,
54 // since this method is also called from handleMouseDown when a toolbar button
55 // is clicked.
56 if (!this.isMouseButtonPressed()) {
57 this.highlightFirst();
58 }
59};
