36/**
37 * A color menu button control. Extends {@link goog.ui.MenuButton} by adding
38 * an API for getting and setting the currently selected color from a menu of
39 * color palettes.
40 *
41 * @param {goog.ui.ControlContent} content Text caption or existing DOM
42 * structure to display as the button's caption.
43 * @param {goog.ui.Menu=} opt_menu Menu to render under the button when clicked;
44 * should contain at least one {@link goog.ui.ColorPalette} if present.
45 * @param {goog.ui.MenuButtonRenderer=} opt_renderer Button renderer;
46 * defaults to {@link goog.ui.ColorMenuButtonRenderer}.
47 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper, used for
48 * document interaction.
49 * @constructor
50 * @extends {goog.ui.MenuButton}
51 */
52goog.ui.ColorMenuButton = function(content, opt_menu, opt_renderer,
53 opt_domHelper) {
54 goog.ui.MenuButton.call(this, content, opt_menu, opt_renderer ||
55 goog.ui.ColorMenuButtonRenderer.getInstance(), opt_domHelper);
56};
57goog.inherits(goog.ui.ColorMenuButton, goog.ui.MenuButton);

93/**
94 * Value for the "no color" menu item object in the color menu (if present).
95 * The {@link goog.ui.ColorMenuButton#handleMenuAction} method interprets
96 * ACTION events dispatched by an item with this value as meaning "clear the
97 * selected color."
98 * @type {string}
99 */
100goog.ui.ColorMenuButton.NO_COLOR = 'none';
101
102
103/**
104 * Factory method that creates and returns a new {@link goog.ui.Menu} instance
105 * containing default color palettes.
106 * @param {Array<goog.ui.Control>=} opt_extraItems Optional extra menu items to
107 * add before the color palettes.
108 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper, used for
109 * document interaction.
110 * @return {!goog.ui.Menu} Color menu.
111 */
112goog.ui.ColorMenuButton.newColorMenu = function(opt_extraItems, opt_domHelper) {
113 var menu = new goog.ui.Menu(opt_domHelper);
114
115 if (opt_extraItems) {
116 goog.array.forEach(opt_extraItems, function(item) {
117 menu.addChild(item, true);
118 });
119 }
120
121 goog.object.forEach(goog.ui.ColorMenuButton.PALETTES, function(colors) {
122 var palette = new goog.ui.ColorPalette(colors, null, opt_domHelper);
123 palette.setSize(8);
124 menu.addChild(palette, true);
125 });
126
127 return menu;
128};

131/**
132 * Returns the currently selected color (null if none).
133 * @return {string} The selected color.
134 */
135goog.ui.ColorMenuButton.prototype.getSelectedColor = function() {
136 return /** @type {string} */ (this.getValue());

142 * null or not any of the available color choices.

145goog.ui.ColorMenuButton.prototype.setSelectedColor = function(color) {

150/**
151 * Sets the value associated with the color menu button. Overrides
152 * {@link goog.ui.Button#setValue} by interpreting the value as a color
153 * spec string.
154 * @param {*} value New button value; should be a color spec string.
155 * @override
156 */
157goog.ui.ColorMenuButton.prototype.setValue = function(value) {
158 var color = /** @type {?string} */ (value);
159 for (var i = 0, item; item = this.getItemAt(i); i++) {
160 if (typeof item.setSelectedColor == 'function') {
161 // This menu item looks like a color palette.
162 item.setSelectedColor(color);
163 }
164 }
165 goog.ui.ColorMenuButton.superClass_.setValue.call(this, color);
166};

169/**
170 * Handles {@link goog.ui.Component.EventType.ACTION} events dispatched by
171 * the menu item clicked by the user. Updates the button, calls the superclass
172 * implementation to hide the menu, stops the propagation of the event, and
173 * dispatches an ACTION event on behalf of the button itself. Overrides
174 * {@link goog.ui.MenuButton#handleMenuAction}.
175 * @param {goog.events.Event} e Action event to handle.
176 * @override
177 */
178goog.ui.ColorMenuButton.prototype.handleMenuAction = function(e) {
179 if (typeof e.target.getSelectedColor == 'function') {
180 // User clicked something that looks like a color palette.
181 this.setValue(e.target.getSelectedColor());
182 } else if (e.target.getValue() == goog.ui.ColorMenuButton.NO_COLOR) {
183 // User clicked the special "no color" menu item.
184 this.setValue(null);
185 }
186 goog.ui.ColorMenuButton.superClass_.handleMenuAction.call(this, e);
187 e.stopPropagation();
188 this.dispatchEvent(goog.ui.Component.EventType.ACTION);
189};

192/**
193 * Opens or closes the menu. Overrides {@link goog.ui.MenuButton#setOpen} by
194 * generating a default color menu on the fly if needed.
195 * @param {boolean} open Whether to open or close the menu.
196 * @param {goog.events.Event=} opt_e Mousedown event that caused the menu to
197 * be opened.
198 * @override
199 */
200goog.ui.ColorMenuButton.prototype.setOpen = function(open, opt_e) {
201 if (open && this.getItemCount() == 0) {
202 this.setMenu(
203 goog.ui.ColorMenuButton.newColorMenu(null, this.getDomHelper()));
204 this.setValue(/** @type {?string} */ (this.getValue()));
205 }
206 goog.ui.ColorMenuButton.superClass_.setOpen.call(this, open, opt_e);
207};
