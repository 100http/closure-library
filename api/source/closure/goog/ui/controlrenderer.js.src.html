40/**
41 * Default renderer for {@link goog.ui.Control}s. Can be used as-is, but
42 * subclasses of Control will probably want to use renderers specifically
43 * tailored for them by extending this class. Controls that use renderers
44 * delegate one or more of the following API methods to the renderer:
45 * <ul>
46 * <li>{@code createDom} - renders the DOM for the component
47 * <li>{@code canDecorate} - determines whether an element can be decorated
48 * by the component
49 * <li>{@code decorate} - decorates an existing element with the component
50 * <li>{@code setState} - updates the appearance of the component based on
51 * its state
52 * <li>{@code getContent} - returns the component's content
53 * <li>{@code setContent} - sets the component's content
54 * </ul>
55 * Controls are stateful; renderers, on the other hand, should be stateless and
56 * reusable.
57 * @constructor
58 */
59goog.ui.ControlRenderer = function() {
60};
61goog.addSingletonGetter(goog.ui.ControlRenderer);
62goog.tagUnsealableClass(goog.ui.ControlRenderer);

65/**
66 * Constructs a new renderer and sets the CSS class that the renderer will use
67 * as the base CSS class to apply to all elements rendered by that renderer.
68 * An example to use this function using a color palette:
69 *
70 * <pre>
71 * var myCustomRenderer = goog.ui.ControlRenderer.getCustomRenderer(
72 * goog.ui.PaletteRenderer, 'my-special-palette');
73 * var newColorPalette = new goog.ui.ColorPalette(
74 * colors, myCustomRenderer, opt_domHelper);
75 * </pre>
76 *
77 * Your CSS can look like this now:
78 * <pre>
79 * .my-special-palette { }
80 * .my-special-palette-table { }
81 * .my-special-palette-cell { }
82 * etc.
83 * </pre>
84 *
85 * <em>instead</em> of
86 * <pre>
87 * .CSS_MY_SPECIAL_PALETTE .goog-palette { }
88 * .CSS_MY_SPECIAL_PALETTE .goog-palette-table { }
89 * .CSS_MY_SPECIAL_PALETTE .goog-palette-cell { }
90 * etc.
91 * </pre>
92 *
93 * You would want to use this functionality when you want an instance of a
94 * component to have specific styles different than the other components of the
95 * same type in your application. This avoids using descendant selectors to
96 * apply the specific styles to this component.
97 *
98 * @param {Function} ctor The constructor of the renderer you are trying to
99 * create.
100 * @param {string} cssClassName The name of the CSS class for this renderer.
101 * @return {goog.ui.ControlRenderer} An instance of the desired renderer with
102 * its getCssClass() method overridden to return the supplied custom CSS
103 * class name.
104 */
105goog.ui.ControlRenderer.getCustomRenderer = function(ctor, cssClassName) {
106 var renderer = new ctor();
107
108 /**
109 * Returns the CSS class to be applied to the root element of components
110 * rendered using this renderer.
111 * @return {string} Renderer-specific CSS class.
112 */
113 renderer.getCssClass = function() {
114 return cssClassName;
115 };
116
117 return renderer;
118};

222/**
223 * Updates the control's DOM by adding or removing the specified class name
224 * to/from its root element. May add additional combined classes as needed in
225 * IE6 and lower. Because of this, subclasses should use this method when
226 * modifying class names on the control's root element.
227 * @param {goog.ui.Control|Element} control Control instance (or root element)
228 * to be updated.
229 * @param {string} className CSS class name to add or remove.
230 * @param {boolean} enable Whether to add or remove the class name.
231 */
232goog.ui.ControlRenderer.prototype.enableClassName = function(control,
233 className, enable) {
234 var element = /** @type {Element} */ (
235 control.getElement ? control.getElement() : control);
236 if (element) {
237 var classNames = [className];
238
239 // For IE6, we need to enable any combined classes involving this class
240 // as well.
241 // TODO(user): Remove this as IE6 is no longer in use.
242 if (goog.userAgent.IE && !goog.userAgent.isVersionOrHigher('7')) {
243 classNames = this.getAppliedCombinedClassNames_(
244 goog.dom.classlist.get(element), className);
245 classNames.push(className);
246 }
247
248 goog.dom.classlist.enableAll(element, classNames, enable);
249 }
250};

279/**
280 * Default implementation of {@code decorate} for {@link goog.ui.Control}s.
281 * Initializes the control's ID, content, and state based on the ID of the
282 * element, its child nodes, and its CSS classes, respectively. Returns the
283 * element.
284 * @param {goog.ui.Control} control Control instance to decorate the element.
285 * @param {Element} element Element to decorate.
286 * @return {Element} Decorated element.
287 */
288goog.ui.ControlRenderer.prototype.decorate = function(control, element) {
289 // Set the control's ID to the decorated element's DOM ID, if any.
290 if (element.id) {
291 control.setId(element.id);
292 }
293
294 // Set the control's content to the decorated element's content.
295 var contentElem = this.getContentElement(element);
296 if (contentElem && contentElem.firstChild) {
297 control.setContentInternal(contentElem.firstChild.nextSibling ?
298 goog.array.clone(contentElem.childNodes) : contentElem.firstChild);
299 } else {
300 control.setContentInternal(null);
301 }
302
303 // Initialize the control's state based on the decorated element's CSS class.
304 // This implementation is optimized to minimize object allocations, string
305 // comparisons, and DOM access.
306 var state = 0x00;
307 var rendererClassName = this.getCssClass();
308 var structuralClassName = this.getStructuralCssClass();
309 var hasRendererClassName = false;
310 var hasStructuralClassName = false;
311 var hasCombinedClassName = false;
312 var classNames = goog.array.toArray(goog.dom.classlist.get(element));
313 goog.array.forEach(classNames, function(className) {
314 if (!hasRendererClassName && className == rendererClassName) {
315 hasRendererClassName = true;
316 if (structuralClassName == rendererClassName) {
317 hasStructuralClassName = true;
318 }
319 } else if (!hasStructuralClassName && className == structuralClassName) {
320 hasStructuralClassName = true;
321 } else {
322 state |= this.getStateFromClass(className);
323 }
324 if (this.getStateFromClass(className) == goog.ui.Component.State.DISABLED) {
325 goog.asserts.assertElement(contentElem);
326 if (goog.dom.isFocusableTabIndex(contentElem)) {
327 goog.dom.setFocusableTabIndex(contentElem, false);
328 }
329 }
330 }, this);
331 control.setStateInternal(state);
332
333 // Make sure the element has the renderer's CSS classes applied, as well as
334 // any extra class names set on the control.
335 if (!hasRendererClassName) {
336 classNames.push(rendererClassName);
337 if (structuralClassName == rendererClassName) {
338 hasStructuralClassName = true;
339 }
340 }
341 if (!hasStructuralClassName) {
342 classNames.push(structuralClassName);
343 }
344 var extraClassNames = control.getExtraClassNames();
345 if (extraClassNames) {
346 classNames.push.apply(classNames, extraClassNames);
347 }
348
349 // For IE6, rewrite all classes on the decorated element if any combined
350 // classes apply.
351 if (goog.userAgent.IE && !goog.userAgent.isVersionOrHigher('7')) {
352 var combinedClasses = this.getAppliedCombinedClassNames_(
353 classNames);
354 if (combinedClasses.length > 0) {
355 classNames.push.apply(classNames, combinedClasses);
356 hasCombinedClassName = true;
357 }
358 }
359
360 // Only write to the DOM if new class names had to be added to the element.
361 if (!hasRendererClassName || !hasStructuralClassName ||
362 extraClassNames || hasCombinedClassName) {
363 goog.dom.classlist.set(element, classNames.join(' '));
364 }
365
366 return element;
367};

370/**
371 * Initializes the control's DOM by configuring properties that can only be set
372 * after the DOM has entered the document. This implementation sets up BiDi
373 * and keyboard focus. Called from {@link goog.ui.Control#enterDocument}.
374 * @param {goog.ui.Control} control Control whose DOM is to be initialized
375 * as it enters the document.
376 */
377goog.ui.ControlRenderer.prototype.initializeDom = function(control) {
378 // Initialize render direction (BiDi). We optimize the left-to-right render
379 // direction by assuming that elements are left-to-right by default, and only
380 // updating their styling if they are explicitly set to right-to-left.
381 if (control.isRightToLeft()) {
382 this.setRightToLeft(control.getElement(), true);
383 }
384
385 // Initialize keyboard focusability (tab index). We assume that components
386 // aren't focusable by default (i.e have no tab index), and only touch the
387 // DOM if the component is focusable, enabled, and visible, and therefore
388 // needs a tab index.
389 if (control.isEnabled()) {
390 this.setFocusable(control, control.isVisible());
391 }
392};

415/**
416 * Sets the element's ARIA attributes, including distinguishing between
417 * universally supported ARIA properties and ARIA states that are only
418 * supported by certain ARIA roles. Only attributes which are initialized to be
419 * true will be set.
420 * @param {!goog.ui.Control} control Control whose ARIA state will be updated.
421 * @param {!Element} element Element whose ARIA state is to be updated.
422 */
423goog.ui.ControlRenderer.prototype.setAriaStates = function(control, element) {
424 goog.asserts.assert(control);
425 goog.asserts.assert(element);
426
427 var ariaLabel = control.getAriaLabel();
428 if (goog.isDefAndNotNull(ariaLabel)) {
429 this.setAriaLabel(element, ariaLabel);
430 }
431
432 if (!control.isVisible()) {
433 goog.a11y.aria.setState(
434 element, goog.a11y.aria.State.HIDDEN, !control.isVisible());
435 }
436 if (!control.isEnabled()) {
437 this.updateAriaState(
438 element, goog.ui.Component.State.DISABLED, !control.isEnabled());
439 }
440 if (control.isSupportedState(goog.ui.Component.State.SELECTED)) {
441 this.updateAriaState(
442 element, goog.ui.Component.State.SELECTED, control.isSelected());
443 }
444 if (control.isSupportedState(goog.ui.Component.State.CHECKED)) {
445 this.updateAriaState(
446 element, goog.ui.Component.State.CHECKED, control.isChecked());
447 }
448 if (control.isSupportedState(goog.ui.Component.State.OPENED)) {
449 this.updateAriaState(
450 element, goog.ui.Component.State.OPENED, control.isOpen());
451 }
452};

494/**
495 * Returns true if the control's key event target supports keyboard focus
496 * (based on its {@code tabIndex} attribute), false otherwise.
497 * @param {goog.ui.Control} control Control whose key event target is to be
498 * checked.
499 * @return {boolean} Whether the control's key event target is focusable.
500 */
501goog.ui.ControlRenderer.prototype.isFocusable = function(control) {
502 var keyTarget;
503 if (control.isSupportedState(goog.ui.Component.State.FOCUSED) &&
504 (keyTarget = control.getKeyEventTarget())) {
505 return goog.dom.isFocusableTabIndex(keyTarget);
506 }
507 return false;

511/**
512 * Updates the control's key event target to make it focusable or non-focusable
513 * via its {@code tabIndex} attribute. Does nothing if the control doesn't
514 * support the {@code FOCUSED} state, or if it has no key event target.
515 * @param {goog.ui.Control} control Control whose key event target is to be
516 * updated.
517 * @param {boolean} focusable Whether to enable keyboard focus support on the
518 * control's key event target.
519 */
520goog.ui.ControlRenderer.prototype.setFocusable = function(control, focusable) {
521 var keyTarget;
522 if (control.isSupportedState(goog.ui.Component.State.FOCUSED) &&
523 (keyTarget = control.getKeyEventTarget())) {
524 if (!focusable && control.isFocused()) {
525 // Blur before hiding. Note that IE calls onblur handlers asynchronously.
526 try {
527 keyTarget.blur();
528 } catch (e) {
529 // TODO(user|user): Find out why this fails on IE.
530 }
531 // The blur event dispatched by the key event target element when blur()
532 // was called on it should have been handled by the control's handleBlur()
533 // method, so at this point the control should no longer be focused.
534 // However, blur events are unreliable on IE and FF3, so if at this point
535 // the control is still focused, we trigger its handleBlur() method
536 // programmatically.
537 if (control.isFocused()) {
538 control.handleBlur(null);
539 }
540 }
541 // Don't overwrite existing tab index values unless needed.
542 if (goog.dom.isFocusableTabIndex(keyTarget) != focusable) {
543 goog.dom.setFocusableTabIndex(keyTarget, focusable);
544 }
545 }
546};

582/**
583 * Updates the element's ARIA (accessibility) attributes , including
584 * distinguishing between universally supported ARIA properties and ARIA states
585 * that are only supported by certain ARIA roles.
586 * @param {Element} element Element whose ARIA state is to be updated.
587 * @param {goog.ui.Component.State} state Component state being enabled or
588 * disabled.
589 * @param {boolean} enable Whether the state is being enabled or disabled.
590 * @protected
591 */
592goog.ui.ControlRenderer.prototype.updateAriaState = function(element, state,
593 enable) {
594 // Ensure the ARIA attribute map exists.
595 if (!goog.ui.ControlRenderer.ariaAttributeMap_) {
596 goog.ui.ControlRenderer.ariaAttributeMap_ = goog.object.create(
597 goog.ui.Component.State.DISABLED, goog.a11y.aria.State.DISABLED,
598 goog.ui.Component.State.SELECTED, goog.a11y.aria.State.SELECTED,
599 goog.ui.Component.State.CHECKED, goog.a11y.aria.State.CHECKED,
600 goog.ui.Component.State.OPENED, goog.a11y.aria.State.EXPANDED);
601 }
602 goog.asserts.assert(element,
603 'The element passed as a first parameter cannot be null.');
604 var ariaAttr = goog.ui.ControlRenderer.getAriaStateForAriaRole_(
605 element, goog.ui.ControlRenderer.ariaAttributeMap_[state]);
606 if (ariaAttr) {
607 goog.a11y.aria.setState(element, ariaAttr, enable);
608 }
609};

612/**
613 * Returns the appropriate ARIA attribute based on ARIA role if the ARIA
614 * attribute is an ARIA state.
615 * @param {!Element} element The element from which to get the ARIA role for
616 * matching ARIA state.
617 * @param {goog.a11y.aria.State} attr The ARIA attribute to check to see if it
618 * can be applied to the given ARIA role.
619 * @return {goog.a11y.aria.State} An ARIA attribute that can be applied to the
620 * given ARIA role.
621 * @private
622 */
623goog.ui.ControlRenderer.getAriaStateForAriaRole_ = function(element, attr) {
624 var role = goog.a11y.aria.getRole(element);
625 if (!role) {
626 return attr;
627 }
628 role = /** @type {goog.a11y.aria.Role} */ (role);
629 var matchAttr = goog.ui.ControlRenderer.TOGGLE_ARIA_STATE_MAP_[role] || attr;
630 return goog.ui.ControlRenderer.isAriaState_(attr) ? matchAttr : attr;
631};

646/**
647 * Takes a control's root element, and sets its content to the given text
648 * caption or DOM structure. The default implementation replaces the children
649 * of the given element. Renderers that create more complex DOM structures
650 * must override this method accordingly.
651 * @param {Element} element The control's root element.
652 * @param {goog.ui.ControlContent} content Text caption or DOM structure to be
653 * set as the control's content. The DOM nodes will not be cloned, they
654 * will only moved under the content element of the control.
655 */
656goog.ui.ControlRenderer.prototype.setContent = function(element, content) {
657 var contentElem = this.getContentElement(element);
658 if (contentElem) {
659 goog.dom.removeChildren(contentElem);
660 if (content) {
661 if (goog.isString(content)) {
662 goog.dom.setTextContent(contentElem, content);
663 } else {
664 var childHandler = function(child) {
665 if (child) {
666 var doc = goog.dom.getOwnerDocument(contentElem);
667 contentElem.appendChild(goog.isString(child) ?
668 doc.createTextNode(child) : child);
669 }
670 };
671 if (goog.isArray(content)) {
672 // Array of nodes.
673 goog.array.forEach(content, childHandler);
674 } else if (goog.isArrayLike(content) && !('nodeType' in content)) {
675 // NodeList. The second condition filters out TextNode which also has
676 // length attribute but is not array like. The nodes have to be cloned
677 // because childHandler removes them from the list during iteration.
678 goog.array.forEach(
679 goog.array.clone(/** @type {!NodeList} */(content)),
680 childHandler);
681 } else {
682 // Node or string.
683 childHandler(content);
684 }
685 }
686 }
687 }

750/**
751 * Returns all CSS class names applicable to the given control, based on its
752 * state. The return value is an array of strings containing
753 * <ol>
754 * <li>the renderer-specific CSS class returned by {@link #getCssClass},
755 * followed by
756 * <li>the structural CSS class returned by {@link getStructuralCssClass} (if
757 * different from the renderer-specific CSS class), followed by
758 * <li>any state-specific classes returned by {@link #getClassNamesForState},
759 * followed by
760 * <li>any extra classes returned by the control's {@code getExtraClassNames}
761 * method and
762 * <li>for IE6 and lower, additional combined classes from
763 * {@link getAppliedCombinedClassNames_}.
764 * </ol>
765 * Since all controls have at least one renderer-specific CSS class name, this
766 * method is guaranteed to return an array of at least one element.
767 * @param {goog.ui.Control} control Control whose CSS classes are to be
768 * returned.
769 * @return {!Array<string>} Array of CSS class names applicable to the control.
770 * @protected
771 */
772goog.ui.ControlRenderer.prototype.getClassNames = function(control) {
773 var cssClass = this.getCssClass();
774
775 // Start with the renderer-specific class name.
776 var classNames = [cssClass];
777
778 // Add structural class name, if different.
779 var structuralCssClass = this.getStructuralCssClass();
780 if (structuralCssClass != cssClass) {
781 classNames.push(structuralCssClass);
782 }
783
784 // Add state-specific class names, if any.
785 var classNamesForState = this.getClassNamesForState(control.getState());
786 classNames.push.apply(classNames, classNamesForState);
787
788 // Add extra class names, if any.
789 var extraClassNames = control.getExtraClassNames();
790 if (extraClassNames) {
791 classNames.push.apply(classNames, extraClassNames);
792 }
793
794 // Add composite classes for IE6 support
795 if (goog.userAgent.IE && !goog.userAgent.isVersionOrHigher('7')) {
796 classNames.push.apply(classNames,
797 this.getAppliedCombinedClassNames_(classNames));
798 }
799
800 return classNames;
801};

804/**
805 * Returns an array of all the combined class names that should be applied based
806 * on the given list of classes. Checks the result of
807 * {@link getIe6ClassCombinations} for any combinations that have all
808 * members contained in classes. If a combination matches, the members are
809 * joined with an underscore (in order), and added to the return array.
810 *
811 * If opt_includedClass is provided, return only the combined classes that have
812 * all members contained in classes AND include opt_includedClass as well.
813 * opt_includedClass is added to classes as well.
814 * @param {goog.array.ArrayLike<string>} classes Array-like thing of classes to
815 * return matching combined classes for.
816 * @param {?string=} opt_includedClass If provided, get only the combined
817 * classes that include this one.
818 * @return {!Array<string>} Array of combined class names that should be
819 * applied.
820 * @private
821 */
822goog.ui.ControlRenderer.prototype.getAppliedCombinedClassNames_ = function(
823 classes, opt_includedClass) {
824 var toAdd = [];
825 if (opt_includedClass) {
826 classes = classes.concat([opt_includedClass]);
827 }
828 goog.array.forEach(this.getIe6ClassCombinations(), function(combo) {
829 if (goog.array.every(combo, goog.partial(goog.array.contains, classes)) &&
830 (!opt_includedClass || goog.array.contains(combo, opt_includedClass))) {
831 toAdd.push(combo.join('_'));
832 }
833 });
834 return toAdd;
835};

838/**
839 * Takes a bit mask of {@link goog.ui.Component.State}s, and returns an array
840 * of the appropriate class names representing the given state, suitable to be
841 * applied to the root element of a component rendered using this renderer, or
842 * null if no state-specific classes need to be applied. This default
843 * implementation uses the renderer's {@link getClassForState} method to
844 * generate each state-specific class.
845 * @param {number} state Bit mask of component states.
846 * @return {!Array<string>} Array of CSS class names representing the given
847 * state.
848 * @protected
849 */
850goog.ui.ControlRenderer.prototype.getClassNamesForState = function(state) {
851 var classNames = [];
852 while (state) {
853 // For each enabled state, push the corresponding CSS class name onto
854 // the classNames array.
855 var mask = state & -state; // Least significant bit
856 classNames.push(this.getClassForState(
857 /** @type {goog.ui.Component.State} */ (mask)));
858 state &= ~mask;
859 }
860 return classNames;
861};
