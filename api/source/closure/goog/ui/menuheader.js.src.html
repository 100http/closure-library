21goog.provide('goog.ui.MenuHeader');
22
23goog.require('goog.ui.Component');
24goog.require('goog.ui.Control');
25goog.require('goog.ui.MenuHeaderRenderer');
26goog.require('goog.ui.registry');
27
28
29
30/**
31 * Class representing a menu header.
32 * @param {goog.ui.ControlContent} content Text caption or DOM structure to
33 * display as the content of the item (use to add icons or styling to
34 * menus).
35 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper used for
36 * document interactions.
37 * @param {goog.ui.MenuHeaderRenderer=} opt_renderer Optional renderer.
38 * @constructor
39 * @extends {goog.ui.Control}
40 */
41goog.ui.MenuHeader = function(content, opt_domHelper, opt_renderer) {
42 goog.ui.Control.call(this, content, opt_renderer ||
43 goog.ui.MenuHeaderRenderer.getInstance(), opt_domHelper);
44
45 this.setSupportedState(goog.ui.Component.State.DISABLED, false);
46 this.setSupportedState(goog.ui.Component.State.HOVER, false);
47 this.setSupportedState(goog.ui.Component.State.ACTIVE, false);
48 this.setSupportedState(goog.ui.Component.State.FOCUSED, false);
49
50 // Headers are always considered disabled.
51 this.setStateInternal(goog.ui.Component.State.DISABLED);
52};
53goog.inherits(goog.ui.MenuHeader, goog.ui.Control);
54
55
56// Register a decorator factory function for goog.ui.MenuHeaders.
57goog.ui.registry.setDecoratorByClassName(
58 goog.ui.MenuHeaderRenderer.CSS_CLASS,
59 function() {
60 // MenuHeader defaults to using MenuHeaderRenderer.
61 return new goog.ui.MenuHeader(null);
62 });
