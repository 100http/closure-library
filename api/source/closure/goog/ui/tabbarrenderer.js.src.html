31/**
32 * Default renderer for {@link goog.ui.TabBar}s, based on the {@code TabPane}
33 * code. The tab bar's DOM structure is determined by its orientation and
34 * location relative to tab contents. For example, a horizontal tab bar
35 * located above tab contents looks like this:
36 * <pre>
37 * <div class="goog-tab-bar goog-tab-bar-horizontal goog-tab-bar-top">
38 * ...(tabs here)...
39 * </div>
40 * </pre>
41 * @constructor
42 * @extends {goog.ui.ContainerRenderer}
43 */
44goog.ui.TabBarRenderer = function() {
45 goog.ui.ContainerRenderer.call(this, goog.a11y.aria.Role.TAB_LIST);
46};
47goog.inherits(goog.ui.TabBarRenderer, goog.ui.ContainerRenderer);
48goog.addSingletonGetter(goog.ui.TabBarRenderer);
49goog.tagUnsealableClass(goog.ui.TabBarRenderer);

71/**
72 * Sets the tab bar's state based on the given CSS class name, encountered
73 * during decoration. Overrides the superclass implementation by recognizing
74 * class names representing tab bar orientation and location.
75 * @param {goog.ui.Container} tabBar Tab bar to configure.
76 * @param {string} className CSS class name.
77 * @param {string} baseClass Base class name used as the root of state-specific
78 * class names (typically the renderer's own class name).
79 * @protected
80 * @override
81 */
82goog.ui.TabBarRenderer.prototype.setStateFromClassName = function(tabBar,
83 className, baseClass) {
84 // Create the class-to-location lookup table on first access.
85 if (!this.locationByClass_) {
86 this.createLocationByClassMap_();
87 }
88
89 // If the class name corresponds to a location, update the tab bar's location;
90 // otherwise let the superclass handle it.
91 var location = this.locationByClass_[className];
92 if (location) {
93 tabBar.setLocation(location);
94 } else {
95 goog.ui.TabBarRenderer.superClass_.setStateFromClassName.call(this, tabBar,
96 className, baseClass);
97 }
98};

101/**
102 * Returns all CSS class names applicable to the tab bar, based on its state.
103 * Overrides the superclass implementation by appending the location-specific
104 * class name to the list.
105 * @param {goog.ui.Container} tabBar Tab bar whose CSS classes are to be
106 * returned.
107 * @return {!Array<string>} Array of CSS class names applicable to the tab bar.
108 * @override
109 */
110goog.ui.TabBarRenderer.prototype.getClassNames = function(tabBar) {
111 var classNames = goog.ui.TabBarRenderer.superClass_.getClassNames.call(this,
112 tabBar);
113
114 // Create the location-to-class lookup table on first access.
115 if (!this.classByLocation_) {
116 this.createClassByLocationMap_();
117 }
118
119 // Apped the class name corresponding to the tab bar's location to the list.
120 classNames.push(this.classByLocation_[tabBar.getLocation()]);
121 return classNames;
122};

125/**
126 * Creates the location-to-class lookup table.
127 * @private
128 */
129goog.ui.TabBarRenderer.prototype.createClassByLocationMap_ = function() {
130 var baseClass = this.getCssClass();
131
132 /**
133 * Map of locations to location-specific structural class names,
134 * precomputed and cached on first use to minimize object allocations
135 * and string concatenation.
136 * @type {Object}
137 * @private
138 */
139 this.classByLocation_ = goog.object.create(
140 goog.ui.TabBar.Location.TOP, goog.getCssName(baseClass, 'top'),
141 goog.ui.TabBar.Location.BOTTOM, goog.getCssName(baseClass, 'bottom'),
142 goog.ui.TabBar.Location.START, goog.getCssName(baseClass, 'start'),
143 goog.ui.TabBar.Location.END, goog.getCssName(baseClass, 'end'));
144};
145
146
147/**
148 * Creates the class-to-location lookup table, used during decoration.
149 * @private
150 */
151goog.ui.TabBarRenderer.prototype.createLocationByClassMap_ = function() {
152 // We need the classByLocation_ map so we can transpose it.
153 if (!this.classByLocation_) {
154 this.createClassByLocationMap_();
155 }
156
157 /**
158 * Map of location-specific structural class names to locations, used during
159 * element decoration. Precomputed and cached on first use to minimize object
160 * allocations and string concatenation.
161 * @type {Object}
162 * @private
163 */
164 this.locationByClass_ = goog.object.transpose(this.classByLocation_);
165};
