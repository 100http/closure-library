54/**
55 * Overrides the superclass implementation by wrapping the caption text or DOM
56 * structure in a color indicator element. Creates the following DOM structure:
57 * <div class="goog-inline-block goog-menu-button-caption">
58 * <div class="goog-color-menu-button-indicator">
59 * Contents...
60 * </div>
61 * </div>
62 * The 'goog-color-menu-button-indicator' style should be defined to have a
63 * bottom border of nonzero width and a default color that blends into its
64 * background.
65 * @param {goog.ui.ControlContent} content Text caption or DOM structure.
66 * @param {goog.dom.DomHelper} dom DOM helper, used for document interaction.
67 * @return {Element} Caption element.
68 * @override
69 */
70goog.ui.ColorMenuButtonRenderer.prototype.createCaption = function(content,
71 dom) {
72 return goog.ui.ColorMenuButtonRenderer.superClass_.createCaption.call(this,
73 goog.ui.ColorMenuButtonRenderer.wrapCaption(content, dom), dom);
74};

77/**
78 * Wrap a caption in a div with the color-menu-button-indicator CSS class.
79 * @param {goog.ui.ControlContent} content Text caption or DOM structure.
80 * @param {goog.dom.DomHelper} dom DOM helper, used for document interaction.
81 * @return {!Element} Caption element.
82 */
83goog.ui.ColorMenuButtonRenderer.wrapCaption = function(content, dom) {
84 return dom.createDom(goog.dom.TagName.DIV,
85 goog.getCssName(goog.ui.ColorMenuButtonRenderer.CSS_CLASS, 'indicator'),
86 content);
87};

90/**
91 * Takes a color menu button control's root element and a value object
92 * (which is assumed to be a color), and updates the button's DOM to reflect
93 * the new color. Overrides {@link goog.ui.ButtonRenderer#setValue}.
94 * @param {Element} element The button control's root element (if rendered).
95 * @param {*} value New value; assumed to be a color spec string.
96 * @override
97 */
98goog.ui.ColorMenuButtonRenderer.prototype.setValue = function(element, value) {
99 if (element) {
100 goog.ui.ColorMenuButtonRenderer.setCaptionValue(
101 this.getContentElement(element), value);
102 }
103};

105
106/**
107 * Takes a control's content element and a value object (which is assumed
108 * to be a color), and updates its DOM to reflect the new color.
109 * @param {Element} caption A content element of a control.
110 * @param {*} value New value; assumed to be a color spec string.
111 */
112goog.ui.ColorMenuButtonRenderer.setCaptionValue = function(caption, value) {
113 // Assume that the caption's first child is the indicator.
114 if (caption && caption.firstChild) {
115 // Normalize the value to a hex color spec or null (otherwise setting
116 // borderBottomColor will cause a JS error on IE).
117 var hexColor;
118
119 var strValue = /** @type {string} */ (value);
120 hexColor = strValue && goog.color.isValidColor(strValue) ?
121 goog.color.parse(strValue).hex :
122 null;
123
124 // Stupid IE6/7 doesn't do transparent borders.
125 // TODO(attila): Add user-agent version check when IE8 comes out...
126 caption.firstChild.style.borderBottomColor = hexColor ||
127 (goog.userAgent.IE ? '' : 'transparent');
128 }
129};

132/**
133 * Initializes the button's DOM when it enters the document. Overrides the
134 * superclass implementation by making sure the button's color indicator is
135 * initialized.
136 * @param {goog.ui.Control} button goog.ui.ColorMenuButton whose DOM is to be
137 * initialized as it enters the document.
138 * @override
139 */
140goog.ui.ColorMenuButtonRenderer.prototype.initializeDom = function(button) {
141 var buttonElement = button.getElement();
142 goog.asserts.assert(buttonElement);
143 this.setValue(buttonElement, button.getValue());
144 goog.dom.classlist.add(buttonElement,
145 goog.ui.ColorMenuButtonRenderer.CSS_CLASS);
146 goog.ui.ColorMenuButtonRenderer.superClass_.initializeDom.call(this,
147 button);
148};
