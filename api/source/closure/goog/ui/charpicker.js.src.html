54/**
55 * Character Picker Class. This widget can be used to pick any Unicode
56 * character by traversing a category-subcategory structure or by inputing its
57 * hex value.
58 *
59 * See charpicker.html demo for example usage.
60 * @param {goog.i18n.CharPickerData} charPickerData Category names and charlist.
61 * @param {!goog.i18n.uChar.NameFetcher} charNameFetcher Object which fetches
62 * the names of the characters that are shown in the widget. These names
63 * may be stored locally or come from an external source.
64 * @param {Array<string>=} opt_recents List of characters to be displayed in
65 * resently selected characters area.
66 * @param {number=} opt_initCategory Sequence number of initial category.
67 * @param {number=} opt_initSubcategory Sequence number of initial subcategory.
68 * @param {number=} opt_rowCount Number of rows in the grid.
69 * @param {number=} opt_columnCount Number of columns in the grid.
70 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper.
71 * @constructor
72 * @extends {goog.ui.Component}
73 * @final
74 */
75goog.ui.CharPicker = function(charPickerData, charNameFetcher, opt_recents,
76 opt_initCategory, opt_initSubcategory,
77 opt_rowCount, opt_columnCount, opt_domHelper) {
78 goog.ui.Component.call(this, opt_domHelper);
79
80 /**
81 * Object used to retrieve character names.
82 * @type {!goog.i18n.uChar.NameFetcher}
83 * @private
84 */
85 this.charNameFetcher_ = charNameFetcher;
86
87 /**
88 * Object containing character lists and category names.
89 * @type {goog.i18n.CharPickerData}
90 * @private
91 */
92 this.data_ = charPickerData;
93
94 /**
95 * The category number to be used on widget init.
96 * @type {number}
97 * @private
98 */
99 this.initCategory_ = opt_initCategory || 0;
100
101 /**
102 * The subcategory number to be used on widget init.
103 * @type {number}
104 * @private
105 */
106 this.initSubcategory_ = opt_initSubcategory || 0;
107
108 /**
109 * Number of columns in the grid.
110 * @type {number}
111 * @private
112 */
113 this.columnCount_ = opt_columnCount || 10;
114
115 /**
116 * Number of entries to be added to the grid.
117 * @type {number}
118 * @private
119 */
120 this.gridsize_ = (opt_rowCount || 10) * this.columnCount_;
121
122 /**
123 * Number of the recently selected characters displayed.
124 * @type {number}
125 * @private

127 this.recentwidth_ = this.columnCount_ + 1;
128
129 /**

135

139 * @private

141 this.eventHandler_ = new goog.events.EventHandler(this);

520/** @override */
521goog.ui.CharPicker.prototype.enterDocument = function() {
522 goog.ui.CharPicker.superClass_.enterDocument.call(this);
523 var inputkh = new goog.events.InputHandler(this.input_.getElement());
524 this.keyHandler_ = new goog.events.KeyHandler(this.input_.getElement());
525
526 // Stop the propagation of ACTION events at menu and submenu buttons.
527 // If stopped at capture phase, the button will not be set to normal state.
528 // If not stopped, the user widget will receive the event, which is
529 // undesired. User widget should receive an event only on the character
530 // click.
531 this.eventHandler_.
532 listen(
533 this.menubutton_,
534 goog.ui.Component.EventType.ACTION,
535 goog.events.Event.stopPropagation).
536 listen(
537 this.submenubutton_,
538 goog.ui.Component.EventType.ACTION,
539 goog.events.Event.stopPropagation).
540 listen(
541 this,
542 goog.ui.Component.EventType.ACTION,
543 this.handleSelectedItem_,
544 true).
545 listen(
546 inputkh,
547 goog.events.InputHandler.EventType.INPUT,
548 this.handleInput_).
549 listen(
550 this.keyHandler_,
551 goog.events.KeyHandler.EventType.KEY,
552 this.handleEnter_).
553 listen(
554 this.recentgrid_,
555 goog.ui.Component.EventType.FOCUS,
556 this.handleFocus_).
557 listen(
558 this.grid_,
559 goog.ui.Component.EventType.FOCUS,
560 this.handleFocus_);
561
562 goog.events.listen(this.okbutton_.getElement(),
563 goog.events.EventType.MOUSEDOWN, this.handleOkClick_, true, this);
564
565 goog.events.listen(this.stickwrap_, goog.events.EventType.SCROLL,
566 this.handleScroll_, true, this);
567};

570/**
571 * Handles the button focus by updating the aria label with the character name
572 * so it becomes possible to get spoken feedback while tabbing through the
573 * visible symbols.
574 * @param {goog.events.Event} e The focus event.
575 * @private
576 */
577goog.ui.CharPicker.prototype.handleFocus_ = function(e) {
578 var button = e.target;
579 var element = button.getElement();
580 var ch = this.getChar_(element);
581
582 // Clear the aria label to avoid speaking the old value in case the button
583 // element has no char attribute or the character name cannot be retrieved.
584 goog.a11y.aria.setState(element, goog.a11y.aria.State.LABEL, '');
585
586 if (ch) {
587 // This is working with screen readers because the call to getName is
588 // synchronous once the values have been prefetched by the RemoteNameFetcher
589 // and because it is always synchronous when using the LocalNameFetcher.
590 // Also, the special character itself is not used as the label because some
591 // screen readers, notably ChromeVox, are not able to speak them.
592 // TODO(user): Consider changing the NameFetcher API to provide a
593 // method that lets the caller retrieve multiple character names at once
594 // so that this asynchronous gymnastic can be avoided.
595 this.charNameFetcher_.getName(ch, function(charName) {
596 if (charName) {
597 goog.a11y.aria.setState(
598 element, goog.a11y.aria.State.LABEL, charName);
599 }
600 });
601 }
602};

605/**
606 * On scroll, updates the grid with characters correct to the scroll position.
607 * @param {goog.events.Event} e Scroll event to handle.
608 * @private
609 */
610goog.ui.CharPicker.prototype.handleScroll_ = function(e) {
611 var height = e.target.scrollHeight;
612 var top = e.target.scrollTop;
613 var itempos = Math.ceil(top * this.items.length / (this.columnCount_ *
614 height)) * this.columnCount_;
615 if (this.itempos != itempos) {
616 this.itempos = itempos;
617 this.modifyGridWithItems_(this.grid_, this.items, itempos);
618 }
619 e.stopPropagation();
620};

623/**
624 * On a menu click, sets correct character set in the grid; on a grid click
625 * accept the character as the selected one and adds to recent selection, if not
626 * already present.
627 * @param {goog.events.Event} e Event for the click on menus or grid.
628 * @private
629 */
630goog.ui.CharPicker.prototype.handleSelectedItem_ = function(e) {
631 var parent = /** @type {goog.ui.Component} */ (e.target).getParent();
632 if (parent == this.menu_) {
633 this.menu_.setVisible(false);
634 this.setSelectedCategory_(e.target.getValue());
635 } else if (parent == this.submenu_) {
636 this.submenu_.setVisible(false);
637 this.setSelectedSubcategory_(e.target.getValue());
638 } else if (parent == this.grid_) {
639 var button = e.target.getElement();
640 this.selectedChar_ = this.getChar_(button);
641 this.updateRecents_(this.selectedChar_);
642 } else if (parent == this.recentgrid_) {
643 this.selectedChar_ = this.getChar_(e.target.getElement());
644 }
645};

648/**
649 * When user types the characters displays the preview. Enables the OK button,
650 * if the character is valid.
651 * @param {goog.events.Event} e Event for typing in input field.
652 * @private
653 */
654goog.ui.CharPicker.prototype.handleInput_ = function(e) {
655 var ch = this.getInputChar();
656 if (ch) {
657 goog.dom.setTextContent(this.zoomEl_, ch);
658 goog.dom.setTextContent(this.unicodeEl_, goog.i18n.uChar.toHexString(ch));
659 goog.dom.setTextContent(this.charNameEl_, '');
660 var coord =
661 new goog.ui.Tooltip.ElementTooltipPosition(this.input_.getElement());
662 this.hc_.setPosition(coord);
663 this.hc_.triggerForElement(this.input_.getElement());
664 this.okbutton_.setEnabled(true);
665 } else {
666 this.hc_.cancelTrigger();
667 this.hc_.setVisible(false);
668 this.okbutton_.setEnabled(false);
669 }
670};

673/**
674 * On OK click accepts the character and updates the recent char list.
675 * @param {goog.events.Event=} opt_event Event for click on OK button.
676 * @return {boolean} Indicates whether to propagate event.
677 * @private
678 */
679goog.ui.CharPicker.prototype.handleOkClick_ = function(opt_event) {
680 var ch = this.getInputChar();
681 if (ch && ch.charCodeAt(0)) {
682 this.selectedChar_ = ch;
683 this.updateRecents_(ch);
684 return true;
685 }
686 return false;
687};

690/**

787 * @param {Array<string>} items Characters to be added to the grid.
788 * @private
789 */
790goog.ui.CharPicker.prototype.updateGrid_ = function(grid, items) {
791 if (grid == this.grid_) {
792 /**
793 * @desc The message used when there are invisible characters like space
794 * or format control characters.
795 */
796 var MSG_PLEASE_HOVER =
797 goog.getMsg('Please hover over each cell for the character name.');
798
799 goog.dom.setTextContent(this.notice_.getElement(),
800 this.charNameFetcher_.isNameAvailable(
801 items[0]) ? MSG_PLEASE_HOVER : '');
802 this.items = items;
803 if (this.stickwrap_.offsetHeight > 0) {
804 this.stick_.style.height =
805 this.stickwrap_.offsetHeight * items.length / this.gridsize_ + 'px';
806 } else {
807 // This is the last ditch effort if height is not avaialble.
808 // Maximum of 3em is assumed to the the cell height. Extra space after
809 // last character in the grid is OK.
810 this.stick_.style.height = 3 * this.columnCount_ * items.length /
811 this.gridsize_ + 'em';
812 }
813 this.stickwrap_.scrollTop = 0;
814 }
815
816 this.modifyGridWithItems_(grid, items, 0);
