52/**
53 * A menu button control. Extends {@link goog.ui.Button} by composing a button
54 * with a dropdown arrow and a popup menu.
55 *
56 * @param {goog.ui.ControlContent=} opt_content Text caption or existing DOM
57 * structure to display as the button's caption (if any).
58 * @param {goog.ui.Menu=} opt_menu Menu to render under the button when clicked.
59 * @param {goog.ui.ButtonRenderer=} opt_renderer Renderer used to render or
60 * decorate the menu button; defaults to {@link goog.ui.MenuButtonRenderer}.
61 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper, used for
62 * document interaction.
63 * @param {!goog.ui.MenuRenderer=} opt_menuRenderer Renderer used to render or
64 * decorate the menu; defaults to {@link goog.ui.MenuRenderer}.
65 * @constructor
66 * @extends {goog.ui.Button}
67 */
68goog.ui.MenuButton = function(opt_content, opt_menu, opt_renderer,
69 opt_domHelper, opt_menuRenderer) {
70 goog.ui.Button.call(this, opt_content, opt_renderer ||
71 goog.ui.MenuButtonRenderer.getInstance(), opt_domHelper);
72
73 // Menu buttons support the OPENED state.
74 this.setSupportedState(goog.ui.Component.State.OPENED, true);
75
76 /**
77 * The menu position on this button.
78 * @type {!goog.positioning.AnchoredPosition}
79 * @private
80 */
81 this.menuPosition_ = new goog.positioning.MenuAnchoredPosition(
82 null, goog.positioning.Corner.BOTTOM_START);
83
84 if (opt_menu) {
85 this.setMenu(opt_menu);
86 }
87 this.menuMargin_ = null;
88 this.timer_ = new goog.Timer(500); // 0.5 sec
89
90 // Phones running iOS prior to version 4.2.
91 if ((goog.userAgent.product.IPHONE || goog.userAgent.product.IPAD) &&
92 // Check the webkit version against the version for iOS 4.2.1.
93 !goog.userAgent.isVersionOrHigher('533.17.9')) {
94 // @bug 4322060 This is required so that the menu works correctly on
95 // iOS prior to version 4.2. Otherwise, the blur action closes the menu
96 // before the menu button click can be processed.
97 this.setFocusablePopupMenu(true);
98 }
99
100 /** @private {!goog.ui.MenuRenderer} */
101 this.menuRenderer_ = opt_menuRenderer || goog.ui.MenuRenderer.getInstance();
102};
103goog.inherits(goog.ui.MenuButton, goog.ui.Button);
104goog.tagUnsealableClass(goog.ui.MenuButton);

327/** @override */
328goog.ui.MenuButton.prototype.handleKeyEventInternal = function(e) {
329 // Handle SPACE on keyup and all other keys on keypress.
330 if (e.keyCode == goog.events.KeyCodes.SPACE) {
331 // Prevent page scrolling in Chrome.
332 e.preventDefault();
333 if (e.type != goog.events.EventType.KEYUP) {
334 // Ignore events because KeyCodes.SPACE is handled further down.
335 return true;
336 }
337 } else if (e.type != goog.events.KeyHandler.EventType.KEY) {
338 return false;
339 }
340
341 if (this.menu_ && this.menu_.isVisible()) {
342 // Menu is open.
343 var isEnterOrSpace = e.keyCode == goog.events.KeyCodes.ENTER ||
344 e.keyCode == goog.events.KeyCodes.SPACE;
345 var handledByMenu = this.menu_.handleKeyEvent(e);
346 if (e.keyCode == goog.events.KeyCodes.ESC || isEnterOrSpace) {
347 // Dismiss the menu.
348 this.setOpen(false);
349 return true;
350 }
351 return handledByMenu;
352 }
353
354 if (e.keyCode == goog.events.KeyCodes.DOWN ||
355 e.keyCode == goog.events.KeyCodes.UP ||
356 e.keyCode == goog.events.KeyCodes.SPACE ||
357 e.keyCode == goog.events.KeyCodes.ENTER) {
358 // Menu is closed, and the user hit the down/up/space/enter key; open menu.
359 this.setOpen(true, e);
360 return true;
361 }
362
363 // Key event wasn't handled by the component.
364 return false;
365};

422/**
423 * Replaces the menu attached to the button with the argument, and returns the
424 * previous menu (if any).
425 * @param {goog.ui.Menu?} menu New menu to be attached to the menu button (null
426 * to remove the menu).
427 * @return {goog.ui.Menu|undefined} Previous menu (undefined if none).
428 */
429goog.ui.MenuButton.prototype.setMenu = function(menu) {
430 var oldMenu = this.menu_;
431
432 // Do nothing unless the new menu is different from the current one.
433 if (menu != oldMenu) {
434 if (oldMenu) {
435 this.setOpen(false);
436 if (this.isInDocument()) {
437 this.attachMenuEventListeners_(oldMenu, false);
438 }
439 delete this.menu_;
440 }
441 if (this.isInDocument()) {
442 goog.a11y.aria.setState(this.getElementStrict(),
443 goog.a11y.aria.State.HASPOPUP, !!menu);
444 }
445 if (menu) {
446 this.menu_ = menu;
447 menu.setParent(this);
448 menu.setVisible(false);
449 menu.setAllowAutoFocus(this.isFocusablePopupMenu());
450 if (this.isInDocument()) {
451 this.attachMenuEventListeners_(menu, true);
452 }
453 }
454 }
455
456 return oldMenu;
457};

645/**
646 * Sets whether the menu should scroll when it's too big to fix vertically on
647 * the screen. The css of the menu element should have overflow set to auto.
648 * Note: Adding or removing items while the menu is open will not work correctly
649 * if scrollOnOverflow is on.
650 * @param {boolean} scrollOnOverflow Whether the menu should scroll when too big
651 * to fit on the screen. If false, adjust logic will be used to try and

655 if (this.menuPosition_.setLastResortOverflow) {
656 var overflowX = goog.positioning.Overflow.ADJUST_X;
657 var overflowY = scrollOnOverflow ?
658 goog.positioning.Overflow.RESIZE_HEIGHT :
659 goog.positioning.Overflow.ADJUST_Y;
660 this.menuPosition_.setLastResortOverflow(overflowX | overflowY);
661 }
662};

671 !!(this.menuPosition_.getLastResortOverflow() &
672 goog.positioning.Overflow.RESIZE_HEIGHT);

736 */
737goog.ui.MenuButton.prototype.setOpen = function(open, opt_e) {
738 goog.ui.MenuButton.superClass_.setOpen.call(this, open);
739 if (this.menu_ && this.hasState(goog.ui.Component.State.OPENED) == open) {
740 if (open) {
741 if (!this.menu_.isInDocument()) {
742 if (this.renderMenuAsSibling_) {
743 // When we render the menu in the same parent as this button, we
744 // prefer to add it immediately after the button. This way, the screen
745 // readers will go to the menu on the very next element after the
746 // button is read.
747 var nextElementSibling =
748 goog.dom.getNextElementSibling(this.getElement());
749 if (nextElementSibling) {
750 this.menu_.renderBefore(nextElementSibling);
751 } else {
752 this.menu_.render(/** @type {Element} */ (
753 this.getElement().parentNode));
754 }
755 } else {
756 this.menu_.render();
757 }
758 }
759 this.viewportBox_ =
760 goog.style.getVisibleRectForElement(this.getElement());
761 this.buttonRect_ = goog.style.getBounds(this.getElement());
762 this.positionMenu();
763
764 // As per aria spec, highlight the first element in the menu when
765 // keyboarding up or down. Thus, the first menu item will be announced
766 // for screen reader users. If selectFirstOnEnterOrSpace is set, do this
767 // for enter or space as well.
768 var isEnterOrSpace = !!opt_e &&
769 (opt_e.keyCode == goog.events.KeyCodes.ENTER ||
770 opt_e.keyCode == goog.events.KeyCodes.SPACE);
771 var isUpOrDown = !!opt_e &&
772 (opt_e.keyCode == goog.events.KeyCodes.DOWN ||
773 opt_e.keyCode == goog.events.KeyCodes.UP);
774 var focus = isUpOrDown ||
775 (isEnterOrSpace && this.selectFirstOnEnterOrSpace_);
776 if (focus) {
777 this.menu_.highlightFirst();
778 } else {
779 this.menu_.setHighlightedIndex(-1);
780 }
781 } else {
782 this.setActive(false);
783 this.menu_.setMouseButtonPressed(false);
784
785 var element = this.getElement();
786 // Clear any remaining a11y state.
787 if (element) {
788 goog.a11y.aria.setState(element,
789 goog.a11y.aria.State.ACTIVEDESCENDANT,
790 '');
791 goog.a11y.aria.setState(element,
792 goog.a11y.aria.State.OWNS,
793 '');
794 }
795
796 // Clear any sizes that might have been stored.
797 if (goog.isDefAndNotNull(this.originalSize_)) {
798 this.originalSize_ = undefined;
799 var elem = this.menu_.getElement();
800 if (elem) {
801 goog.style.setSize(elem, '', '');
802 }
803 }
804 }
805 this.menu_.setVisible(open, false, opt_e);
806 // In Pivot Tables the menu button somehow gets disposed of during the
807 // setVisible call, causing attachPopupListeners_ to fail.
808 // TODO(user): Debug what happens.
809 if (!this.isDisposed()) {
810 this.attachPopupListeners_(open);
811 }
812 }
813 if (this.menu_ && this.menu_.getElement()) {
814 // Remove the aria-hidden state on the menu element so that it won't be
815 // hidden to screen readers if it's inside a dialog (see b/17610491).
816 goog.a11y.aria.removeState(
817 this.menu_.getElementStrict(), goog.a11y.aria.State.HIDDEN);
818 }
819};

839 return;

844 this.menuPosition_.element = positionElement;

987/**
988 * Updates the aria-activedescendant attribute to the given target element.
989 * @param {!Element} targetEl The target element.
990 * @private
991 */
992goog.ui.MenuButton.prototype.setAriaActiveDescendant_ = function(targetEl) {
993 var element = this.getElement();
994 goog.asserts.assert(element, 'The menu button DOM element cannot be null.');
995
996 // If target element has an activedescendant, then set this control's
997 // activedescendant to that, otherwise set it to the target element. This is
998 // a workaround for some screen readers which do not handle
999 // aria-activedescendant redirection properly.
1000 var targetActiveDescendant = goog.a11y.aria.getActiveDescendant(targetEl);
1001 var activeDescendant = targetActiveDescendant || targetEl;
1002
1003 if (!activeDescendant.id) {
1004 // Create an id if there isn't one already.
1005 var idGenerator = goog.ui.IdGenerator.getInstance();
1006 activeDescendant.id = idGenerator.getNextUniqueId();
1007 }
1008
1009 goog.a11y.aria.setActiveDescendant(element, activeDescendant);
1010 goog.a11y.aria.setState(
1011 element, goog.a11y.aria.State.OWNS, activeDescendant.id);
1012};

1015/**
1016 * Attaches or detaches event listeners depending on whether the popup menu
1017 * is being shown or hidden. Starts listening for document mousedown events
1018 * and for menu blur events when the menu is shown, and stops listening for
1019 * these events when it is hidden. Called from {@link #setOpen}.
1020 * @param {boolean} attach Whether to attach or detach event listeners.
1021 * @private
1022 */
1023goog.ui.MenuButton.prototype.attachPopupListeners_ = function(attach) {
1024 var handler = this.getHandler();
1025 var method = attach ? handler.listen : handler.unlisten;
1026
1027 // Listen for document mousedown events in the capture phase, because
1028 // the target may stop propagation of the event in the bubble phase.
1029 method.call(handler, this.getDomHelper().getDocument(),
1030 goog.events.EventType.MOUSEDOWN, this.handleDocumentMouseDown, true);
1031
1032 // Only listen for blur events dispatched by the menu if it is focusable.
1033 if (this.isFocusablePopupMenu()) {
1034 method.call(handler, /** @type {!goog.events.EventTarget} */ (this.menu_),
1035 goog.ui.Component.EventType.BLUR, this.handleMenuBlur);
1036 }
1037
1038 method.call(handler, this.timer_, goog.Timer.TICK, this.onTick_);
1039 if (attach) {
1040 this.timer_.start();
1041 } else {
1042 this.timer_.stop();
1043 }
1044};
