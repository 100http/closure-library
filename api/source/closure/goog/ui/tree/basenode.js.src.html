45/**

180/**
181 * Adds roles and states.
182 * @protected
183 */
184goog.ui.tree.BaseNode.prototype.initAccessibility = function() {
185 var el = this.getElement();
186 if (el) {
187 // Set an id for the label
188 var label = this.getLabelElement();
189 if (label && !label.id) {
190 label.id = this.getId() + '.label';
191 }
192
193 goog.a11y.aria.setRole(el, 'treeitem');
194 goog.a11y.aria.setState(el, 'selected', false);
195 goog.a11y.aria.setState(el, 'expanded', false);
196 goog.a11y.aria.setState(el, 'level', this.getDepth());
197 if (label) {
198 goog.a11y.aria.setState(el, 'labelledby', label.id);
199 }
200
201 var img = this.getIconElement();
202 if (img) {
203 goog.a11y.aria.setRole(img, 'presentation');
204 }
205 var ei = this.getExpandIconElement();
206 if (ei) {
207 goog.a11y.aria.setRole(ei, 'presentation');
208 }
209
210 var ce = this.getChildrenElement();
211 if (ce) {
212 goog.a11y.aria.setRole(ce, 'group');
213
214 // In case the children will be created lazily.
215 if (ce.hasChildNodes()) {
216 // do setsize for each child
217 var count = this.getChildCount();
218 for (var i = 1; i <= count; i++) {
219 var child = this.getChildAt(i - 1).getElement();
220 goog.asserts.assert(child, 'The child element cannot be null');
221 goog.a11y.aria.setState(child, 'setsize', count);
222 goog.a11y.aria.setState(child, 'posinset', i);
223 }
224 }
225 }
226 }

252/**
253 * The method assumes that the child doesn't have parent node yet.
254 * The {@code opt_render} argument is not used. If the parent node is expanded,
255 * the child node's state will be the same as the parent's. Otherwise the
256 * child's DOM tree won't be created.
257 * @override
258 */
259goog.ui.tree.BaseNode.prototype.addChildAt = function(child, index,
260 opt_render) {
261 goog.asserts.assert(!child.getParent());
262 goog.asserts.assertInstanceof(child, goog.ui.tree.BaseNode);
263 var prevNode = this.getChildAt(index - 1);
264 var nextNode = this.getChildAt(index);
265
266 goog.ui.tree.BaseNode.superClass_.addChildAt.call(this, child, index);
267
268 child.previousSibling_ = prevNode;
269 child.nextSibling_ = nextNode;
270
271 if (prevNode) {
272 prevNode.nextSibling_ = child;
273 } else {
274 this.firstChild_ = child;
275 }
276 if (nextNode) {
277 nextNode.previousSibling_ = child;
278 } else {
279 this.lastChild_ = child;
280 }
281
282 var tree = this.getTree();
283 if (tree) {
284 child.setTreeInternal(tree);
285 }
286
287 child.setDepth_(this.getDepth() + 1);
288
289 if (this.getElement()) {
290 this.updateExpandIcon();
291 if (this.getExpanded()) {
292 var el = this.getChildrenElement();
293 if (!child.getElement()) {
294 child.createDom();
295 }
296 var childElement = child.getElement();
297 var nextElement = nextNode && nextNode.getElement();
298 el.insertBefore(childElement, nextElement);
299
300 if (this.isInDocument()) {
301 child.enterDocument();
302 }
303
304 if (!nextNode) {
305 if (prevNode) {
306 prevNode.updateExpandIcon();
307 } else {
308 goog.style.setElementShown(el, true);
309 this.setExpanded(this.getExpanded());
310 }
311 }
312 }
313 }

317/**
318 * Adds a node as a child to the current node.
319 * @param {goog.ui.tree.BaseNode} child The child to add.
320 * @param {goog.ui.tree.BaseNode=} opt_before If specified, the new child is
321 * added as a child before this one. If not specified, it's appended to the
322 * end.
323 * @return {!goog.ui.tree.BaseNode} The added child.
324 */
325goog.ui.tree.BaseNode.prototype.add = function(child, opt_before) {
326 goog.asserts.assert(!opt_before || opt_before.getParent() == this,
327 'Can only add nodes before siblings');
328 if (child.getParent()) {
329 child.getParent().removeChild(child);
330 }
331 this.addChildAt(child,
332 opt_before ? this.indexOfChild(opt_before) : this.getChildCount());
333 return child;
334};

337/**
338 * Removes a child. The caller is responsible for disposing the node.
339 * @param {goog.ui.Component|string} childNode The child to remove. Must be a
340 * {@link goog.ui.tree.BaseNode}.
341 * @param {boolean=} opt_unrender Unused. The child will always be unrendered.
342 * @return {!goog.ui.tree.BaseNode} The child that was removed.
343 * @override
344 */
345goog.ui.tree.BaseNode.prototype.removeChild =
346 function(childNode, opt_unrender) {
347 // In reality, this only accepts BaseNodes.
348 var child = /** @type {goog.ui.tree.BaseNode} */ (childNode);
349
350 // if we remove selected or tree with the selected we should select this
351 var tree = this.getTree();
352 var selectedNode = tree ? tree.getSelectedItem() : null;
353 if (selectedNode == child || child.contains(selectedNode)) {
354 if (tree.hasFocus()) {
355 this.select();
356 goog.Timer.callOnce(this.onTimeoutSelect_, 10, this);
357 } else {
358 this.select();
359 }
360 }
361
362 goog.ui.tree.BaseNode.superClass_.removeChild.call(this, child);
363
364 if (this.lastChild_ == child) {
365 this.lastChild_ = child.previousSibling_;
366 }
367 if (this.firstChild_ == child) {
368 this.firstChild_ = child.nextSibling_;
369 }
370 if (child.previousSibling_) {
371 child.previousSibling_.nextSibling_ = child.nextSibling_;
372 }
373 if (child.nextSibling_) {
374 child.nextSibling_.previousSibling_ = child.previousSibling_;
375 }
376
377 var wasLast = child.isLastSibling();
378
379 child.tree = null;
380 child.depth_ = -1;
381
382 if (tree) {
383 // Tell the tree control that this node is now removed.
384 tree.removeNode(this);
385
386 if (this.isInDocument()) {
387 var el = this.getChildrenElement();
388
389 if (child.isInDocument()) {
390 var childEl = child.getElement();
391 el.removeChild(childEl);
392
393 child.exitDocument();
394 }
395
396 if (wasLast) {
397 var newLast = this.getLastChild();
398 if (newLast) {
399 newLast.updateExpandIcon();
400 }
401 }
402 if (!this.hasChildren()) {
403 el.style.display = 'none';
404 this.updateExpandIcon();
405 this.updateIcon_();
406 }
407 }
408 }
409
410 return child;
411};

450/**
451 * Computes the depth of the node in the tree.
452 * Called only by getDepth, when the depth hasn't already been cached.
453 * @return {number} The non-negative depth of this node (the root is zero).
454 * @private
455 */
456goog.ui.tree.BaseNode.prototype.computeDepth_ = function() {

466/**
467 * Changes the depth of a node (and all its descendants).
468 * @param {number} depth The new nesting depth; must be non-negative.
469 * @private
470 */
471goog.ui.tree.BaseNode.prototype.setDepth_ = function(depth) {
472 if (depth != this.depth_) {
473 this.depth_ = depth;
474 var row = this.getRowElement();
475 if (row) {
476 var indent = this.getPixelIndent_() + 'px';
477 if (this.isRightToLeft()) {
478 row.style.paddingRight = indent;
479 } else {
480 row.style.paddingLeft = indent;
481 }
482 }
483 this.forEachChild(function(child) {
484 child.setDepth_(depth + 1);
485 });
486 }
487};

647/**
648 * Sets the node to be expanded.
649 * @param {boolean} expanded Whether to expand or close the node.
650 */
651goog.ui.tree.BaseNode.prototype.setExpanded = function(expanded) {
652 var isStateChange = expanded != this.expanded_;
653 if (isStateChange) {
654 // Only fire events if the expanded state has actually changed.
655 var prevented = !this.dispatchEvent(
656 expanded ? goog.ui.tree.BaseNode.EventType.BEFORE_EXPAND :
657 goog.ui.tree.BaseNode.EventType.BEFORE_COLLAPSE);
658 if (prevented) return;
659 }
660 var ce;
661 this.expanded_ = expanded;
662 var tree = this.getTree();
663 var el = this.getElement();
664
665 if (this.hasChildren()) {
666 if (!expanded && tree && this.contains(tree.getSelectedItem())) {
667 this.select();
668 }
669
670 if (el) {
671 ce = this.getChildrenElement();
672 if (ce) {
673 goog.style.setElementShown(ce, expanded);
674
675 // Make sure we have the HTML for the children here.
676 if (expanded && this.isInDocument() && !ce.hasChildNodes()) {
677 var children = [];
678 this.forEachChild(function(child) {
679 children.push(child.toSafeHtml());
680 });
681 goog.dom.safe.setInnerHtml(ce, goog.html.SafeHtml.concat(children));
682 this.forEachChild(function(child) {
683 child.enterDocument();
684 });
685 }
686 }
687 this.updateExpandIcon();
688 }
689 } else {
690 ce = this.getChildrenElement();
691 if (ce) {
692 goog.style.setElementShown(ce, false);
693 }
694 }
695 if (el) {
696 this.updateIcon_();
697 goog.a11y.aria.setState(el, 'expanded', expanded);
698 }
699
700 if (isStateChange) {
701 this.dispatchEvent(expanded ? goog.ui.tree.BaseNode.EventType.EXPAND :
702 goog.ui.tree.BaseNode.EventType.COLLAPSE);
703 }
704};

804/**
805 * Creates HTML for the node.
806 * @return {!goog.html.SafeHtml}
807 * @protected
808 */
809goog.ui.tree.BaseNode.prototype.toSafeHtml = function() {
810 var tree = this.getTree();
811 var hideLines = !tree.getShowLines() ||
812 tree == this.getParent() && !tree.getShowRootLines();
813
814 var childClass = hideLines ? this.config_.cssChildrenNoLines :
815 this.config_.cssChildren;
816
817 var nonEmptyAndExpanded = this.getExpanded() && this.hasChildren();
818
819 var attributes = {
820 'class': childClass,
821 'style': this.getLineStyle()
822 };
823
824 var content = [];
825 if (nonEmptyAndExpanded) {
826 // children
827 this.forEachChild(function(child) {
828 content.push(child.toSafeHtml());
829 });
830 }
831
832 var children = goog.html.SafeHtml.create('div', attributes, content);
833
834 return goog.html.SafeHtml.create('div',
835 {'class': this.config_.cssItem, 'id': this.getId()},
836 [this.getRowSafeHtml(), children]);
837};

981/**
982 * @return {string} The class names of the icon used for expanding the node.
983 * @protected
984 */
985goog.ui.tree.BaseNode.prototype.getExpandIconClass = function() {
986 var tree = this.getTree();
987 var hideLines = !tree.getShowLines() ||
988 tree == this.getParent() && !tree.getShowRootLines();
989
990 var config = this.config_;
991 var sb = new goog.string.StringBuffer();
992 sb.append(config.cssTreeIcon, ' ', config.cssExpandTreeIcon, ' ');
993
994 if (this.hasChildren()) {
995 var bits = 0;
996 /*
997 Bitmap used to determine which icon to use
998 1 Plus
999 2 Minus
1000 4 T Line
1001 8 L Line
1002 */
1003
1004 if (tree.getShowExpandIcons() && this.isUserCollapsible_) {
1005 if (this.getExpanded()) {
1006 bits = 2;
1007 } else {
1008 bits = 1;
1009 }
1010 }
1011
1012 if (!hideLines) {
1013 if (this.isLastSibling()) {
1014 bits += 4;
1015 } else {
1016 bits += 8;
1017 }
1018 }
1019
1020 switch (bits) {
1021 case 1:
1022 sb.append(config.cssExpandTreeIconPlus);
1023 break;
1024 case 2:
1025 sb.append(config.cssExpandTreeIconMinus);
1026 break;
1027 case 4:
1028 sb.append(config.cssExpandTreeIconL);
1029 break;
1030 case 5:
1031 sb.append(config.cssExpandTreeIconLPlus);
1032 break;
1033 case 6:
1034 sb.append(config.cssExpandTreeIconLMinus);
1035 break;
1036 case 8:
1037 sb.append(config.cssExpandTreeIconT);
1038 break;
1039 case 9:
1040 sb.append(config.cssExpandTreeIconTPlus);
1041 break;
1042 case 10:
1043 sb.append(config.cssExpandTreeIconTMinus);
1044 break;
1045 default: // 0
1046 sb.append(config.cssExpandTreeIconBlank);
1047 }
1048 } else {
1049 if (hideLines) {
1050 sb.append(config.cssExpandTreeIconBlank);
1051 } else if (this.isLastSibling()) {
1052 sb.append(config.cssExpandTreeIconL);
1053 } else {
1054 sb.append(config.cssExpandTreeIconT);

1373/**
1374 * Handles a key down event.
1375 * @param {!goog.events.BrowserEvent} e The browser event.
1376 * @return {boolean} The handled value.
1377 * @protected
1378 */
1379goog.ui.tree.BaseNode.prototype.onKeyDown = function(e) {
1380 var handled = true;
1381 switch (e.keyCode) {
1382 case goog.events.KeyCodes.RIGHT:
1383 if (e.altKey) {
1384 break;
1385 }
1386 if (this.hasChildren()) {
1387 if (!this.getExpanded()) {
1388 this.setExpanded(true);
1389 } else {
1390 this.getFirstChild().select();
1391 }
1392 }
1393 break;
1394
1395 case goog.events.KeyCodes.LEFT:
1396 if (e.altKey) {
1397 break;
1398 }
1399 if (this.hasChildren() && this.getExpanded() && this.isUserCollapsible_) {
1400 this.setExpanded(false);
1401 } else {

1403 var tree = this.getTree();
1404 // don't go to root if hidden
1405 if (parent && (tree.getShowRootNode() || parent != tree)) {
1406 parent.select();
1407 }
1408 }

1411 case goog.events.KeyCodes.DOWN:
1412 var nextNode = this.getNextShownNode();
1413 if (nextNode) {
1414 nextNode.select();

1417
1418 case goog.events.KeyCodes.UP:

1420 if (previousNode) {
1421 previousNode.select();
1422 }
1423 break;

1425 default:
1426 handled = false;
1427 }

1430 e.preventDefault();
1431 var tree = this.getTree();
1432 if (tree) {

1434 tree.clearTypeAhead();
