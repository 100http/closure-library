40/**
41 * TabPane widget. All children already inside the tab pane container element
42 * will be be converted to tabs. Each tab is represented by a goog.ui.TabPane.
43 * TabPage object. Further pages can be constructed either from an existing
44 * container or created from scratch.
45 *
46 * @param {Element} el Container element to create the tab pane out of.
47 * @param {goog.ui.TabPane.TabLocation=} opt_tabLocation Location of the tabs
48 * in relation to the content container. Default is top.
49 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper.
50 * @param {boolean=} opt_useMouseDown Whether to use MOUSEDOWN instead of CLICK
51 * for tab changes.
52 * @extends {goog.events.EventTarget}
53 * @constructor
54 * @see ../demos/tabpane.html
55 * @deprecated Use goog.ui.TabBar instead.
56 */
57goog.ui.TabPane = function(el, opt_tabLocation, opt_domHelper,
58 opt_useMouseDown) {
59 goog.events.EventTarget.call(this);
60
61 /**
62 * DomHelper used to interact with the document, allowing components to be
63 * created in a different window. This property is considered protected;
64 * subclasses of Component may refer to it directly.
65 * @type {goog.dom.DomHelper}
66 * @protected
67 * @suppress {underscore|visibility}
68 */
69 this.dom_ = opt_domHelper || goog.dom.getDomHelper();
70
71 /**
72 * Tab pane element.
73 * @type {Element}
74 * @private
75 */
76 this.el_ = el;
77
78 /**
79 * Collection of tab panes.
80 * @type {Array<goog.ui.TabPane.TabPage>}
81 * @private
82 */
83 this.pages_ = [];
84
85 /**
86 * Location of the tabs with respect to the content box.
87 * @type {goog.ui.TabPane.TabLocation}
88 * @private
89 */
90 this.tabLocation_ =
91 opt_tabLocation ? opt_tabLocation : goog.ui.TabPane.TabLocation.TOP;
92
93 /**
94 * Whether to use MOUSEDOWN instead of CLICK for tab change events. This
95 * fixes some focus problems on Safari/Chrome.
96 * @type {boolean}
97 * @private
98 */
99 this.useMouseDown_ = !!opt_useMouseDown;
100
101 this.create_();
102};

154/**
155 * Creates HTML nodes for tab pane.
156 *
157 * @private
158 */
159goog.ui.TabPane.prototype.create_ = function() {
160 this.el_.className = goog.getCssName('goog-tabpane');
161
162 var nodes = this.getChildNodes_();
163
164 // Create tab strip
165 this.elButtonBar_ = this.dom_.createDom(goog.dom.TagName.UL,
166 {'className': goog.getCssName('goog-tabpane-tabs'), 'tabIndex': '0'});
167
168 // Create content area
169 this.elContent_ = this.dom_.createDom(goog.dom.TagName.DIV,
170 goog.getCssName('goog-tabpane-cont'));
171 this.el_.appendChild(this.elContent_);
172
173 var element = goog.asserts.assertElement(this.el_);
174
175 switch (this.tabLocation_) {
176 case goog.ui.TabPane.TabLocation.TOP:
177 element.insertBefore(this.elButtonBar_, this.elContent_);
178 element.insertBefore(this.createClear_(), this.elContent_);
179 goog.dom.classlist.add(element, goog.getCssName('goog-tabpane-top'));
180 break;
181 case goog.ui.TabPane.TabLocation.BOTTOM:
182 element.appendChild(this.elButtonBar_);
183 element.appendChild(this.createClear_());
184 goog.dom.classlist.add(element, goog.getCssName('goog-tabpane-bottom'));
185 break;
186 case goog.ui.TabPane.TabLocation.LEFT:
187 element.insertBefore(this.elButtonBar_, this.elContent_);
188 goog.dom.classlist.add(element, goog.getCssName('goog-tabpane-left'));
189 break;
190 case goog.ui.TabPane.TabLocation.RIGHT:
191 element.insertBefore(this.elButtonBar_, this.elContent_);
192 goog.dom.classlist.add(element, goog.getCssName('goog-tabpane-right'));
193 break;
194 default:
195 throw Error('Invalid tab location');
196 }
197
198 // Listen for click and keydown events on header
199 this.elButtonBar_.tabIndex = 0;
200 goog.events.listen(this.elButtonBar_,
201 this.useMouseDown_ ?
202 goog.events.EventType.MOUSEDOWN :
203 goog.events.EventType.CLICK,
204 this.onHeaderClick_, false, this);
205 goog.events.listen(this.elButtonBar_, goog.events.EventType.KEYDOWN,
206 this.onHeaderKeyDown_, false, this);
207
208 this.createPages_(nodes);
209};

274/**
275 * Adds a page to the tab pane.
276 *
277 * @param {goog.ui.TabPane.TabPage} page Tab page to add.
278 * @param {number=} opt_index Zero based index to insert tab at. Inserted at the
279 * end if not specified.
280 */
281goog.ui.TabPane.prototype.addPage = function(page, opt_index) {
282 // If page is already in another tab pane it's removed from that one before it
283 // can be added to this one.
284 if (page.parent_ && page.parent_ != this &&
285 page.parent_ instanceof goog.ui.TabPane) {
286 page.parent_.removePage(page);
287 }
288
289 // Insert page at specified position
290 var index = this.pages_.length;
291 if (goog.isDef(opt_index) && opt_index != index) {
292 index = opt_index;
293 this.pages_.splice(index, 0, page);
294 this.elButtonBar_.insertBefore(page.elTitle_,
295 this.elButtonBar_.childNodes[index]);
296 }
297
298 // Append page to end
299 else {
300 this.pages_.push(page);
301 this.elButtonBar_.appendChild(page.elTitle_);
302 }
303
304 page.setParent_(this, index);
305
306 // Select first page and fire change event
307 if (!this.selected_) {
308 this.selected_ = page;
309 this.dispatchEvent(new goog.ui.TabPaneEvent(goog.ui.TabPane.Events.CHANGE,
310 this, this.selected_));
311 }
312
313 // Move page content to the tab pane and update visibility.
314 this.elContent_.appendChild(page.elContent_);
315 page.setVisible_(page == this.selected_);
316
317 // Update index for following pages
318 for (var pg, i = index + 1; pg = this.pages_[i]; i++) {
319 pg.index_ = i;
320 }
321};

324/**
325 * Removes the specified page from the tab pane.
326 *
327 * @param {goog.ui.TabPane.TabPage|number} page Reference to tab page or zero
328 * based index.
329 */
330goog.ui.TabPane.prototype.removePage = function(page) {
331 if (goog.isNumber(page)) {
332 page = this.pages_[page];
333 }
334 this.pages_.splice(page.index_, 1);
335 page.setParent_(null);
336
337 goog.dom.removeNode(page.elTitle_);
338 goog.dom.removeNode(page.elContent_);
339
340 for (var pg, i = 0; pg = this.pages_[i]; i++) {
341 pg.setParent_(this, i);
342 }
343};

421/**
422 * Click event handler for header element, handles clicks on tabs.
423 *
424 * @param {goog.events.BrowserEvent} event Click event.
425 * @private
426 */
427goog.ui.TabPane.prototype.onHeaderClick_ = function(event) {
428 var el = event.target;
429
430 // Determine index if a tab (li element) was clicked.
431 while (el != this.elButtonBar_) {
432 if (el.tagName == goog.dom.TagName.LI) {
433 var i;
434 // {} prevents compiler warning
435 for (i = 0; el = el.previousSibling; i++) {}
436 this.setSelectedIndex(i);
437 break;
438 }
439 el = el.parentNode;
440 }
441 event.preventDefault();
442};

445/**
446 * KeyDown event handler for header element. Arrow keys moves between pages.
447 * Home and end selects the first/last page.
448 *
449 * @param {goog.events.BrowserEvent} event KeyDown event.
450 * @private
451 */
452goog.ui.TabPane.prototype.onHeaderKeyDown_ = function(event) {
453 if (event.altKey || event.metaKey || event.ctrlKey) {
454 return;
455 }
456
457 switch (event.keyCode) {
458 case goog.events.KeyCodes.LEFT:
459 var index = this.selected_.getIndex() - 1;
460 this.setSelectedIndex(index < 0 ? this.pages_.length - 1 : index);
461 break;
462 case goog.events.KeyCodes.RIGHT:
463 var index = this.selected_.getIndex() + 1;
464 this.setSelectedIndex(index >= this.pages_.length ? 0 : index);
465 break;
466 case goog.events.KeyCodes.HOME:
467 this.setSelectedIndex(0);
468 break;
469 case goog.events.KeyCodes.END:
470 this.setSelectedIndex(this.pages_.length - 1);
471 break;
472 }
473};

477/**
478 * Object representing an individual tab pane.
479 *
480 * @param {Element=} opt_el Container element to create the pane out of.
481 * @param {(Element|string)=} opt_title Pane title or element to use as the
482 * title. If not specified the first element in the container is used as
483 * the title.
484 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper
485 * The first parameter can be omitted.
486 * @constructor
487 */
488goog.ui.TabPane.TabPage = function(opt_el, opt_title, opt_domHelper) {
489 var title, el;
490 if (goog.isString(opt_el) && !goog.isDef(opt_title)) {
491 title = opt_el;
492 } else if (opt_title) {
493 title = opt_title;
494 el = opt_el;
495 } else if (opt_el) {
496 var child = goog.dom.getFirstElementChild(opt_el);
497 if (child) {
498 title = goog.dom.getTextContent(child);
499 child.parentNode.removeChild(child);
500 }
501 el = opt_el;
502 }
