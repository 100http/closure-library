15/**
16 * @fileoverview Implementation of a basic slider control.
17 *
18 * Models a control that allows to select a sub-range within a given
19 * range of values using two thumbs. The underlying range is modeled
20 * as a range model, where the min thumb points to value of the
21 * rangemodel, and the max thumb points to value + extent of the range
22 * model.
23 *
24 * The currently selected range is exposed through methods
25 * getValue() and getExtent().
26 *
27 * The reason for modelling the basic slider state as value + extent is
28 * to be able to capture both, a two-thumb slider to select a range, and
29 * a single-thumb slider to just select a value (in the latter case, extent
30 * is always zero). We provide subclasses (twothumbslider.js and slider.js)
31 * that model those special cases of this control.
32 *
33 * All rendering logic is left out, so that the subclasses can define
34 * their own rendering. To do so, the subclasses overwrite:
35 * - createDom
36 * - decorateInternal
37 * - getCssClass
38 *
39 * @author arv@google.com (Erik Arvidsson)
40 */
41

115/**
116 * Event types used to listen for dragging events. Note that extent drag events
117 * are also sent for single-thumb sliders, since the one thumb controls both
118 * value and extent together; in this case, they can simply be ignored.
119 * @enum {string}
120 */
121goog.ui.SliderBase.EventType = {
122 /** User started dragging the value thumb */
123 DRAG_VALUE_START: goog.events.getUniqueId('dragvaluestart'),
124 /** User is done dragging the value thumb */
125 DRAG_VALUE_END: goog.events.getUniqueId('dragvalueend'),
126 /** User started dragging the extent thumb */
127 DRAG_EXTENT_START: goog.events.getUniqueId('dragextentstart'),
128 /** User is done dragging the extent thumb */
129 DRAG_EXTENT_END: goog.events.getUniqueId('dragextentend'),
130 // Note that the following two events are sent twice, once for the value
131 // dragger, and once of the extent dragger. If you need to differentiate
132 // between the two, or if your code relies on receiving a single event per
133 // START/END event, it should listen to one of the VALUE/EXTENT-specific
134 // events.
135 /** User started dragging a thumb */
136 DRAG_START: goog.events.getUniqueId('dragstart'),
137 /** User is done dragging a thumb */
138 DRAG_END: goog.events.getUniqueId('dragend')

452/**
453 * Attaches/Detaches the event handlers on the slider.
454 * @param {boolean} enable Whether to attach or detach the event handlers.
455 * @private
456 */
457goog.ui.SliderBase.prototype.enableEventHandlers_ = function(enable) {
458 if (enable) {
459 this.getHandler().
460 listen(this.valueDragger_, goog.fx.Dragger.EventType.BEFOREDRAG,
461 this.handleBeforeDrag_).
462 listen(this.extentDragger_, goog.fx.Dragger.EventType.BEFOREDRAG,
463 this.handleBeforeDrag_).
464 listen(this.valueDragger_,
465 [goog.fx.Dragger.EventType.START, goog.fx.Dragger.EventType.END],
466 this.handleThumbDragStartEnd_).
467 listen(this.extentDragger_,
468 [goog.fx.Dragger.EventType.START, goog.fx.Dragger.EventType.END],
469 this.handleThumbDragStartEnd_).
470 listen(this.keyHandler_, goog.events.KeyHandler.EventType.KEY,
471 this.handleKeyDown_).
472 listen(this.getElement(), goog.events.EventType.CLICK,
473 this.handleMouseDownAndClick_).
474 listen(this.getElement(), goog.events.EventType.MOUSEDOWN,
475 this.handleMouseDownAndClick_);
476 if (this.isHandleMouseWheel()) {
477 this.enableMouseWheelHandling_(true);
478 }
479 } else {
480 this.getHandler().
481 unlisten(this.valueDragger_, goog.fx.Dragger.EventType.BEFOREDRAG,
482 this.handleBeforeDrag_).
483 unlisten(this.extentDragger_, goog.fx.Dragger.EventType.BEFOREDRAG,
484 this.handleBeforeDrag_).
485 unlisten(this.valueDragger_,
486 [goog.fx.Dragger.EventType.START, goog.fx.Dragger.EventType.END],
487 this.handleThumbDragStartEnd_).
488 unlisten(this.extentDragger_,
489 [goog.fx.Dragger.EventType.START, goog.fx.Dragger.EventType.END],
490 this.handleThumbDragStartEnd_).
491 unlisten(this.keyHandler_, goog.events.KeyHandler.EventType.KEY,
492 this.handleKeyDown_).
493 unlisten(this.getElement(), goog.events.EventType.CLICK,
494 this.handleMouseDownAndClick_).
495 unlisten(this.getElement(), goog.events.EventType.MOUSEDOWN,
496 this.handleMouseDownAndClick_);
497 if (this.isHandleMouseWheel()) {
498 this.enableMouseWheelHandling_(false);
499 }
500 }

512/**
513 * Handler for the before drag event. We use the event properties to determine
514 * the new value.
515 * @param {goog.fx.DragEvent} e The drag event used to drag the thumb.
516 * @private
517 */
518goog.ui.SliderBase.prototype.handleBeforeDrag_ = function(e) {
519 var thumbToDrag = e.dragger == this.valueDragger_ ?
520 this.valueThumb : this.extentThumb;
521 var value;
522 if (this.orientation_ == goog.ui.SliderBase.Orientation.VERTICAL) {
523 var availHeight = this.getElement().clientHeight - thumbToDrag.offsetHeight;
524 value = (availHeight - e.top) / availHeight *
525 (this.getMaximum() - this.getMinimum()) + this.getMinimum();
526 } else {
527 var availWidth = this.getElement().clientWidth - thumbToDrag.offsetWidth;
528 value = (e.left / availWidth) * (this.getMaximum() - this.getMinimum()) +
529 this.getMinimum();
530 }
531 // Bind the value within valid range before calling setThumbPosition_.
532 // This is necessary because setThumbPosition_ is a no-op for values outside
533 // of the legal range. For drag operations, we want the handle to snap to the
534 // last valid value instead of remaining at the previous position.
535 if (e.dragger == this.valueDragger_) {
536 value = Math.min(Math.max(value, this.getMinimum()),
537 this.getValue() + this.getExtent());
538 } else {
539 value = Math.min(Math.max(value, this.getValue()), this.getMaximum());
540 }
541 this.setThumbPosition_(thumbToDrag, value);
542};

545/**
546 * Handler for the start/end drag event on the thumbs. Adds/removes
547 * the "-dragging" CSS classes on the slider and thumb.
548 * @param {goog.fx.DragEvent} e The drag event used to drag the thumb.
549 * @private
550 */
551goog.ui.SliderBase.prototype.handleThumbDragStartEnd_ = function(e) {
552 var isDragStart = e.type == goog.fx.Dragger.EventType.START;
553 goog.dom.classlist.enable(goog.asserts.assertElement(this.getElement()),
554 goog.ui.SliderBase.SLIDER_DRAGGING_CSS_CLASS_, isDragStart);
555 goog.dom.classlist.enable(goog.asserts.assertElement(e.target.handle),
556 goog.ui.SliderBase.THUMB_DRAGGING_CSS_CLASS_, isDragStart);
557 var isValueDragger = e.dragger == this.valueDragger_;
558 if (isDragStart) {
559 this.dispatchEvent(goog.ui.SliderBase.EventType.DRAG_START);
560 this.dispatchEvent(isValueDragger ?
561 goog.ui.SliderBase.EventType.DRAG_VALUE_START :
562 goog.ui.SliderBase.EventType.DRAG_EXTENT_START);
563 } else {
564 this.dispatchEvent(goog.ui.SliderBase.EventType.DRAG_END);
565 this.dispatchEvent(isValueDragger ?
566 goog.ui.SliderBase.EventType.DRAG_VALUE_END :
567 goog.ui.SliderBase.EventType.DRAG_EXTENT_END);
568 }
569};

572/**
573 * Event handler for the key down event. This is used to update the value
574 * based on the key pressed.
575 * @param {goog.events.KeyEvent} e The keyboard event object.
576 * @private
577 */
578goog.ui.SliderBase.prototype.handleKeyDown_ = function(e) {
579 var handled = true;
580 switch (e.keyCode) {
581 case goog.events.KeyCodes.HOME:
582 this.animatedSetValue(this.getMinimum());
583 break;
584 case goog.events.KeyCodes.END:
585 this.animatedSetValue(this.getMaximum());
586 break;
587 case goog.events.KeyCodes.PAGE_UP:
588 this.moveThumbs(this.getBlockIncrement());
589 break;
590 case goog.events.KeyCodes.PAGE_DOWN:
591 this.moveThumbs(-this.getBlockIncrement());
592 break;
593 case goog.events.KeyCodes.LEFT:
594 var sign = this.flipForRtl_ && this.isRightToLeft() ? 1 : -1;
595 this.moveThumbs(e.shiftKey ?
596 sign * this.getBlockIncrement() : sign * this.getUnitIncrement());
597 break;
598 case goog.events.KeyCodes.DOWN:
599 this.moveThumbs(e.shiftKey ?
600 -this.getBlockIncrement() : -this.getUnitIncrement());
601 break;
602 case goog.events.KeyCodes.RIGHT:
603 var sign = this.flipForRtl_ && this.isRightToLeft() ? -1 : 1;
604 this.moveThumbs(e.shiftKey ?
605 sign * this.getBlockIncrement() : sign * this.getUnitIncrement());
606 break;
607 case goog.events.KeyCodes.UP:
608 this.moveThumbs(e.shiftKey ?
609 this.getBlockIncrement() : this.getUnitIncrement());
610 break;
611
612 default:
613 handled = false;
614 }
615
616 if (handled) {
617 e.preventDefault();
618 }
619};

622/**
623 * Handler for the mouse down event and click event.
624 * @param {goog.events.Event} e The mouse event object.
625 * @private
626 */
627goog.ui.SliderBase.prototype.handleMouseDownAndClick_ = function(e) {
628 if (this.getElement().focus) {
629 this.getElement().focus();
630 }
631
632 // Known Element.
633 var target = /** @type {Element} */ (e.target);
634
635 if (!goog.dom.contains(this.valueThumb, target) &&
636 !goog.dom.contains(this.extentThumb, target)) {
637 var isClick = e.type == goog.events.EventType.CLICK;
638 if (isClick && goog.now() < this.mouseDownTime_ + this.MOUSE_DOWN_DELAY_) {
639 // Ignore a click event that comes a short moment after a mousedown
640 // event. This happens for desktop. For devices with both a touch
641 // screen and a mouse pad we do not get a mousedown event from the mouse
642 // pad and do get a click event.
643 return;
644 }
645 if (!isClick) {
646 this.mouseDownTime_ = goog.now();
647 }
648
649 if (this.moveToPointEnabled_) {
650 // just set the value directly based on the position of the click
651 this.animatedSetValue(this.getValueFromMousePosition(e));
652 } else {
653 // start a timer that incrementally moves the handle
654 this.startBlockIncrementing_(e);
655 }
656 }
657};

673/**
674 * Starts the animation that causes the thumb to increment/decrement by the
675 * block increment when the user presses down on the background.
676 * @param {goog.events.Event} e The mouse event object.
677 * @private
678 */
679goog.ui.SliderBase.prototype.startBlockIncrementing_ = function(e) {
680 this.storeMousePos_(e);
681 this.thumbToMove_ = this.getClosestThumb_(this.getValueFromMousePosition(e));
682 if (this.orientation_ == goog.ui.SliderBase.Orientation.VERTICAL) {
683 this.incrementing_ = this.lastMousePosition_ < this.thumbToMove_.offsetTop;
684 } else {
685 this.incrementing_ = this.lastMousePosition_ >
686 this.getOffsetStart_(this.thumbToMove_) +
687 this.thumbToMove_.offsetWidth;
688 }
689
690 var doc = goog.dom.getOwnerDocument(this.getElement());
691 this.getHandler().
692 listen(doc, goog.events.EventType.MOUSEUP,
693 this.stopBlockIncrementing_, true).
694 listen(this.getElement(), goog.events.EventType.MOUSEMOVE,
695 this.storeMousePos_);
696
697 if (!this.incTimer_) {
698 this.incTimer_ = new goog.Timer(
699 goog.ui.SliderBase.MOUSE_DOWN_INCREMENT_INTERVAL_);
700 this.getHandler().listen(this.incTimer_, goog.Timer.TICK,
701 this.handleTimerTick_);
702 }
703 this.handleTimerTick_();
704 this.incTimer_.start();
705};

708/**
709 * Handler for the tick event dispatched by the timer used to update the value
710 * in a block increment. This is also called directly from
711 * startBlockIncrementing_.
712 * @private
713 */
714goog.ui.SliderBase.prototype.handleTimerTick_ = function() {
715 var value;
716 if (this.orientation_ == goog.ui.SliderBase.Orientation.VERTICAL) {
717 var mouseY = this.lastMousePosition_;
718 var thumbY = this.thumbToMove_.offsetTop;
719 if (this.incrementing_) {
720 if (mouseY < thumbY) {
721 value = this.getThumbPosition_(this.thumbToMove_) +
722 this.getBlockIncrement();
723 }
724 } else {
725 var thumbH = this.thumbToMove_.offsetHeight;
726 if (mouseY > thumbY + thumbH) {
727 value = this.getThumbPosition_(this.thumbToMove_) -
728 this.getBlockIncrement();
729 }
730 }
731 } else {
732 var mouseX = this.lastMousePosition_;
733 var thumbX = this.getOffsetStart_(this.thumbToMove_);
734 if (this.incrementing_) {
735 var thumbW = this.thumbToMove_.offsetWidth;
736 if (mouseX > thumbX + thumbW) {
737 value = this.getThumbPosition_(this.thumbToMove_) +
738 this.getBlockIncrement();
739 }
740 } else {
741 if (mouseX < thumbX) {
742 value = this.getThumbPosition_(this.thumbToMove_) -
743 this.getBlockIncrement();
744 }
745 }
746 }
747
748 if (goog.isDef(value)) { // not all code paths sets the value variable
749 this.setThumbPosition_(this.thumbToMove_, value);
750 }
751};

853/**
854 * Moves the thumbs by the specified delta as follows
855 * - as long as both thumbs stay within [min,max], both thumbs are moved
856 * - once a thumb reaches or exceeds min (or max, respectively), it stays
857 * - at min (or max, respectively).
858 * In case both thumbs have reached min (or max), no change event will fire.
859 * If the specified delta is smaller than the step size, it will be rounded
860 * to the step size.
861 * @param {number} delta The delta by which to move the selected range.
862 */
863goog.ui.SliderBase.prototype.moveThumbs = function(delta) {
864 // Assume that a small delta is supposed to be at least a step.
865 if (Math.abs(delta) < this.getStep()) {
866 delta = goog.math.sign(delta) * this.getStep();
867 }
868 var newMinPos = this.getThumbPosition_(this.valueThumb) + delta;
869 var newMaxPos = this.getThumbPosition_(this.extentThumb) + delta;
870 // correct min / max positions to be within bounds
871 newMinPos = goog.math.clamp(
872 newMinPos, this.getMinimum(), this.getMaximum() - this.minExtent_);
873 newMaxPos = goog.math.clamp(
874 newMaxPos, this.getMinimum() + this.minExtent_, this.getMaximum());
875 // Set value and extent atomically
876 this.setValueAndExtent(newMinPos, newMaxPos - newMinPos);
877};

880/**
881 * Sets the position of the given thumb. The set is ignored and no CHANGE event
882 * fires if it violates the constraint minimum <= value (valueThumb position) <=
883 * value + extent (extentThumb position) <= maximum.
884 *
885 * Note: To keep things simple, the setThumbPosition_ function does not have the
886 * side-effect of "correcting" value or extent to fit the above constraint as it
887 * is the case in the underlying range model. Instead, we simply ignore the
888 * call. Callers must make these adjustements explicitly if they wish.
889 * @param {Element} thumb The thumb whose position to set.
890 * @param {number} position The position to move the thumb to.
891 * @private
892 */
893goog.ui.SliderBase.prototype.setThumbPosition_ = function(thumb, position) {
894 // Round first so that all computations and checks are consistent.
895 var roundedPosition = this.rangeModel.roundToStepWithMin(position);
896 var value = thumb == this.valueThumb ? roundedPosition :
897 this.rangeModel.getValue();
898 var end = thumb == this.extentThumb ? roundedPosition :
899 this.rangeModel.getValue() + this.rangeModel.getExtent();
900 if (value >= this.getMinimum() && end >= value + this.minExtent_ &&
901 this.getMaximum() >= end) {
902 this.setValueAndExtent(value, end - value);
903 }
904};

907/**
908 * Sets the value and extent of the underlying range model. We enforce that
909 * getMinimum() <= value <= getMaximum() - extent and
910 * getMinExtent <= extent <= getMaximum() - getValue()
911 * If this is not satisfied for the given extent, the call is ignored and no
912 * CHANGE event fires. This is a utility method to allow setting the thumbs
913 * simultaneously and ensuring that only one event fires.
914 * @param {number} value The value to which to set the value.
915 * @param {number} extent The value to which to set the extent.
916 */
917goog.ui.SliderBase.prototype.setValueAndExtent = function(value, extent) {
918 if (this.getMinimum() <= value &&
919 value <= this.getMaximum() - extent &&
920 this.minExtent_ <= extent &&
921 extent <= this.getMaximum() - value) {
922
923 if (value == this.getValue() && extent == this.getExtent()) {
924 return;
925 }
926 // because the underlying range model applies adjustements of value
927 // and extent to fit within bounds, we need to reset the extent
928 // first so these adjustements don't kick in.
929 this.rangeModel.setMute(true);
930 this.rangeModel.setExtent(0);
931 this.rangeModel.setValue(value);
932 this.rangeModel.setExtent(extent);
933 this.rangeModel.setMute(false);
934 this.handleRangeModelChange(null);
935 }
936};

1053/**
1054 * Calculates the start position (offset) and size of the range highlight, e.g.
1055 * for a horizontal slider, this will return [left, width] for the highlight.
1056 * @param {number} firstThumbPos The position of the first thumb along the
1057 * slider axis.
1058 * @param {number} secondThumbPos The position of the second thumb along the
1059 * slider axis, must be >= firstThumbPos.
1060 * @param {number} thumbSize The size of the thumb, along the slider axis.
1061 * @return {{offset: number, size: number}} The positioning parameters for the
1062 * range highlight.
1063 * @private
1064 */
1065goog.ui.SliderBase.prototype.calculateRangeHighlightPositioning_ = function(
1066 firstThumbPos, secondThumbPos, thumbSize) {
1067 // Highlight is inset by half the thumb size, from the edges of the thumb.
1068 var highlightInset = Math.ceil(thumbSize / 2);
1069 var size = secondThumbPos - firstThumbPos + thumbSize - 2 * highlightInset;
1070 // Don't return negative size since it causes an error. IE sometimes attempts
1071 // to position the thumbs while slider size is 0, resulting in size < 0 here.
1072 return {
1073 offset: firstThumbPos + highlightInset,
1074 size: Math.max(size, 0)
1075 };

1079/**
1080 * Returns the position to move the handle to for a given value
1081 * @param {number} val The value to get the coordinate for.
1082 * @return {!goog.math.Coordinate} Coordinate with either x or y set.
1083 */
1084goog.ui.SliderBase.prototype.getThumbCoordinateForValue = function(val) {
1085 var coord = new goog.math.Coordinate;
1086 if (this.valueThumb) {
1087 var min = this.getMinimum();
1088 var max = this.getMaximum();
1089
1090 // This check ensures the ratio never take NaN value, which is possible when
1091 // the slider min & max are same numbers (i.e. 1).
1092 var ratio = (val == min && min == max) ? 0 : (val - min) / (max - min);
1093
1094 if (this.orientation_ == goog.ui.SliderBase.Orientation.VERTICAL) {
1095 var thumbHeight = this.valueThumb.offsetHeight;
1096 var h = this.getElement().clientHeight - thumbHeight;
1097 var bottom = Math.round(ratio * h);
1098 coord.x = this.getOffsetStart_(this.valueThumb); // Keep x the same.
1099 coord.y = h - bottom;
1100 } else {
1101 var w = this.getElement().clientWidth - this.valueThumb.offsetWidth;
1102 var left = Math.round(ratio * w);
1103 coord.x = left;
1104 coord.y = this.valueThumb.offsetTop; // Keep y the same.
1105 }
1106 }
1107 return coord;
1108};

1111/**
1112 * Sets the value and starts animating the handle towards that position.
1113 * @param {number} v Value to set and animate to.
1114 */
1115goog.ui.SliderBase.prototype.animatedSetValue = function(v) {
1116 // the value might be out of bounds
1117 v = goog.math.clamp(v, this.getMinimum(), this.getMaximum());
1118
1119 if (this.isAnimating_) {
1120 this.currentAnimation_.stop(true);
1121 }
1122 var animations = new goog.fx.AnimationParallelQueue();
1123 var end;
1124
1125 var thumb = this.getClosestThumb_(v);
1126 var previousValue = this.getValue();
1127 var previousExtent = this.getExtent();
1128 var previousThumbValue = this.getThumbPosition_(thumb);
1129 var previousCoord = this.getThumbCoordinateForValue(previousThumbValue);
1130 var stepSize = this.getStep();
1131
1132 // If the delta is less than a single step, increase it to a step, else the
1133 // range model will reduce it to zero.
1134 if (Math.abs(v - previousThumbValue) < stepSize) {
1135 var delta = v > previousThumbValue ? stepSize : -stepSize;
1136 v = previousThumbValue + delta;
1137
1138 // The resulting value may be out of bounds, sanitize.
1139 v = goog.math.clamp(v, this.getMinimum(), this.getMaximum());
1140 }
1141
1142 this.setThumbPosition_(thumb, v);
1143 var coord = this.getThumbCoordinateForValue(this.getThumbPosition_(thumb));
1144
1145 if (this.orientation_ == goog.ui.SliderBase.Orientation.VERTICAL) {
1146 end = [this.getOffsetStart_(thumb), coord.y];
1147 } else {
1148 end = [coord.x, thumb.offsetTop];
1149 }
1150
1151 var slide = new goog.fx.dom.Slide(thumb,
1152 [previousCoord.x, previousCoord.y],
1153 end,
1154 goog.ui.SliderBase.ANIMATION_INTERVAL_);
1155 slide.enableRightPositioningForRtl(this.flipForRtl_);
1156 animations.add(slide);
1157 if (this.rangeHighlight) {
1158 this.addRangeHighlightAnimations_(thumb, previousValue, previousExtent,
1159 coord, animations);
1160 }
1161
1162 // Create additional animations to play if a factory has been set.
1163 if (this.additionalAnimations_) {
1164 var additionalAnimations = this.additionalAnimations_.createAnimations(
1165 previousValue, v, goog.ui.SliderBase.ANIMATION_INTERVAL_);
1166 goog.array.forEach(additionalAnimations, function(animation) {
1167 animations.add(animation);
1168 });
1169 }
1170
1171 this.currentAnimation_ = animations;
1172 this.getHandler().listen(animations, goog.fx.Transition.EventType.END,
1173 this.endAnimation_);
1174
1175 this.isAnimating_ = true;
1176 animations.play(false);
1177};

1205/**
1206 * Adds animations for the range highlight element to the animation queue.
1207 *
1208 * @param {Element} thumb The thumb that's moving, must be
1209 * either valueThumb or extentThumb.
1210 * @param {number} previousValue The previous value of the slider.
1211 * @param {number} previousExtent The previous extent of the
1212 * slider.
1213 * @param {goog.math.Coordinate} newCoord The new pixel coordinate of the
1214 * thumb that's moving.
1215 * @param {goog.fx.AnimationParallelQueue} animations The animation queue.
1216 * @private
1217 */
1218goog.ui.SliderBase.prototype.addRangeHighlightAnimations_ = function(thumb,
1219 previousValue, previousExtent, newCoord, animations) {
1220 var previousMinCoord = this.getThumbCoordinateForValue(previousValue);
1221 var previousMaxCoord = this.getThumbCoordinateForValue(
1222 previousValue + previousExtent);
1223 var minCoord = previousMinCoord;
1224 var maxCoord = previousMaxCoord;
1225 if (thumb == this.valueThumb) {
1226 minCoord = newCoord;
1227 } else {
1228 maxCoord = newCoord;
1229 }
1230
1231 if (this.orientation_ == goog.ui.SliderBase.Orientation.VERTICAL) {
1232 var previousHighlightPositioning = this.calculateRangeHighlightPositioning_(
1233 previousMaxCoord.y, previousMinCoord.y, this.valueThumb.offsetHeight);
1234 var highlightPositioning = this.calculateRangeHighlightPositioning_(
1235 maxCoord.y, minCoord.y, this.valueThumb.offsetHeight);
1236 var slide = new goog.fx.dom.Slide(this.rangeHighlight,
1237 [this.getOffsetStart_(this.rangeHighlight),
1238 previousHighlightPositioning.offset],
1239 [this.getOffsetStart_(this.rangeHighlight),
1240 highlightPositioning.offset],
1241 goog.ui.SliderBase.ANIMATION_INTERVAL_);
1242 var resizeHeight = new goog.fx.dom.ResizeHeight(this.rangeHighlight,
1243 previousHighlightPositioning.size, highlightPositioning.size,
1244 goog.ui.SliderBase.ANIMATION_INTERVAL_);
1245 slide.enableRightPositioningForRtl(this.flipForRtl_);
1246 resizeHeight.enableRightPositioningForRtl(this.flipForRtl_);
1247 animations.add(slide);
1248 animations.add(resizeHeight);
1249 } else {
1250 var previousHighlightPositioning = this.calculateRangeHighlightPositioning_(
1251 previousMinCoord.x, previousMaxCoord.x, this.valueThumb.offsetWidth);
1252 var highlightPositioning = this.calculateRangeHighlightPositioning_(
1253 minCoord.x, maxCoord.x, this.valueThumb.offsetWidth);
1254 var slide = new goog.fx.dom.Slide(this.rangeHighlight,
1255 [previousHighlightPositioning.offset, this.rangeHighlight.offsetTop],
1256 [highlightPositioning.offset, this.rangeHighlight.offsetTop],
1257 goog.ui.SliderBase.ANIMATION_INTERVAL_);
1258 var resizeWidth = new goog.fx.dom.ResizeWidth(this.rangeHighlight,
1259 previousHighlightPositioning.size, highlightPositioning.size,
1260 goog.ui.SliderBase.ANIMATION_INTERVAL_);
1261 slide.enableRightPositioningForRtl(this.flipForRtl_);
1262 resizeWidth.enableRightPositioningForRtl(this.flipForRtl_);
1263 animations.add(slide);
1264 animations.add(resizeWidth);
1265 }
1266};
