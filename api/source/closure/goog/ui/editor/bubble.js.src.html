52/**
53 * Property bubble UI element.
54 * @param {Element} parent The parent element for this bubble.
55 * @param {number} zIndex The z index to draw the bubble at.
56 * @constructor

59goog.ui.editor.Bubble = function(parent, zIndex) {
60 goog.ui.editor.Bubble.base(this, 'constructor');
61
62 /**
63 * Dom helper for the document the bubble should be shown in.
64 * @type {!goog.dom.DomHelper}
65 * @private
66 */
67 this.dom_ = goog.dom.getDomHelper(parent);
68
69 /**
70 * Event handler for this bubble.
71 * @type {goog.events.EventHandler<!goog.ui.editor.Bubble>}
72 * @private
73 */
74 this.eventHandler_ = new goog.events.EventHandler(this);
75
76 /**
77 * Object that monitors the application window for size changes.
78 * @type {goog.dom.ViewportSizeMonitor}
79 * @private
80 */
81 this.viewPortSizeMonitor_ = new goog.dom.ViewportSizeMonitor(
82 this.dom_.getWindow());

84 /**
85 * Maps panel ids to panels.
86 * @type {Object<goog.ui.editor.Bubble.Panel_>}
87 * @private
88 */
89 this.panels_ = {};

91 /**
92 * Container element for the entire bubble. This may contain elements related
93 * to look and feel or styling of the bubble.
94 * @type {Element}
95 * @private
96 */
97 this.bubbleContainer_ =
98 this.dom_.createDom(goog.dom.TagName.DIV,
99 {'className': goog.ui.editor.Bubble.BUBBLE_CLASSNAME});
100
101 goog.style.setElementShown(this.bubbleContainer_, false);
102 goog.dom.appendChild(parent, this.bubbleContainer_);
103 goog.style.setStyle(this.bubbleContainer_, 'zIndex', zIndex);
104
105 /**
106 * Container element for the bubble panels - this should be some inner element
107 * within (or equal to) bubbleContainer.
108 * @type {Element}
109 * @private
110 */
111 this.bubbleContents_ = this.createBubbleDom(this.dom_, this.bubbleContainer_);
112
113 /**
114 * Element showing the close box.
115 * @type {!Element}
116 * @private
117 */
118 this.closeBox_ = this.dom_.createDom(goog.dom.TagName.DIV, {
119 'className': goog.getCssName('tr_bubble_closebox'),
120 'innerHTML': '&nbsp;'
121 });
122 this.bubbleContents_.appendChild(this.closeBox_);
123
124 // We make bubbles unselectable so that clicking on them does not steal focus
125 // or move the cursor away from the element the bubble is attached to.
126 goog.editor.style.makeUnselectable(this.bubbleContainer_, this.eventHandler_);
127
128 /**
129 * Popup that controls showing and hiding the bubble at the appropriate
130 * position.
131 * @type {goog.ui.PopupBase}
132 * @private
133 */
134 this.popup_ = new goog.ui.PopupBase(this.bubbleContainer_);
135};

243/**
244 * Adds a panel to the bubble.
245 * @param {string} type The type of bubble panel this is. Should usually be
246 * the same as the tagName of the targetElement. This ensures multiple
247 * bubble panels don't appear for the same element.
248 * @param {string} title The title of the panel.
249 * @param {Element} targetElement The target element of the bubble.
250 * @param {function(Element): void} contentFn Function that when called with
251 * a container element, will add relevant panel content to it.
252 * @param {boolean=} opt_preferTopPosition Whether to prefer placing the bubble
253 * above the element instead of below it. Defaults to preferring below.
254 * If any panel prefers the top position, the top position is used.
255 * @return {string} The id of the panel.
256 */
257goog.ui.editor.Bubble.prototype.addPanel = function(type, title, targetElement,
258 contentFn, opt_preferTopPosition) {
259 var id = goog.string.createUniqueString();
260 var panel = new goog.ui.editor.Bubble.Panel_(this.dom_, id, type, title,
261 targetElement, !opt_preferTopPosition);
262 this.panels_[id] = panel;
263
264 // Insert the panel in string order of type. Technically we could use binary
265 // search here but n is really small (probably 0 - 2) so it's not worth it.
266 // The last child of bubbleContents_ is the close box so we take care not
267 // to treat it as a panel element, and we also ensure it stays as the last
268 // element. The intention here is not to create any artificial order, but
269 // just to ensure that it is always consistent.
270 var nextElement;
271 for (var i = 0, len = this.bubbleContents_.childNodes.length - 1; i < len;
272 i++) {
273 var otherChild = this.bubbleContents_.childNodes[i];
274 var otherPanel = this.panels_[otherChild.id];
275 if (otherPanel.type > type) {
276 nextElement = otherChild;
277 break;
278 }
279 }
280 goog.dom.insertSiblingBefore(panel.element,
281 nextElement || this.bubbleContents_.lastChild);
282
283 contentFn(panel.getContentElement());
284 goog.editor.style.makeUnselectable(panel.element, this.eventHandler_);
285
286 var numPanels = goog.object.getCount(this.panels_);
287 if (numPanels == 1) {
288 this.openBubble_();
289 } else if (numPanels == 2) {
290 goog.dom.classlist.add(
291 goog.asserts.assert(this.bubbleContainer_),
292 goog.getCssName('tr_multi_bubble'));
293 }
294 this.reposition();
295
296 return id;
297};

300/**
301 * Removes the panel with the given id.
302 * @param {string} id The id of the panel.
303 */
304goog.ui.editor.Bubble.prototype.removePanel = function(id) {
305 var panel = this.panels_[id];
306 goog.dom.removeNode(panel.element);
307 delete this.panels_[id];
308
309 var numPanels = goog.object.getCount(this.panels_);
310 if (numPanels <= 1) {
311 goog.dom.classlist.remove(
312 goog.asserts.assert(this.bubbleContainer_),
313 goog.getCssName('tr_multi_bubble'));
314 }
315
316 if (numPanels == 0) {
317 this.closeBubble_();
318 } else {
319 this.reposition();
320 }
321};

351/**
352 * Handles the popup's hide event by removing all panels and dispatching a
353 * HIDE event.
354 * @protected
355 */
356goog.ui.editor.Bubble.prototype.handlePopupHide = function() {
357 // Remove the panel elements.
358 for (var panelId in this.panels_) {
359 goog.dom.removeNode(this.panels_[panelId].element);
360 }
361
362 // Update the state to reflect no panels.
363 this.panels_ = {};
364 goog.dom.classlist.remove(
365 goog.asserts.assert(this.bubbleContainer_),
366 goog.getCssName('tr_multi_bubble'));
367
368 this.eventHandler_.removeAll();
369 this.dispatchEvent(goog.ui.Component.EventType.HIDE);
370};

411/**
412 * Positions and displays this bubble below its targetElement. Assumes that
413 * the bubbleContainer is already contained in the document object it applies
414 * to.
415 */
416goog.ui.editor.Bubble.prototype.reposition = function() {
417 var targetElement = null;
418 var preferBottomPosition = true;
419 for (var panelId in this.panels_) {
420 var panel = this.panels_[panelId];
421 // We don't care which targetElement we get, so we just take the last one.
422 targetElement = panel.targetElement;
423 preferBottomPosition = preferBottomPosition && panel.preferBottomPosition;
424 }
425 var status = goog.positioning.OverflowStatus.FAILED;
426
427 // Fix for bug when bubbleContainer and targetElement have
428 // opposite directionality, the bubble should anchor to the END of
429 // the targetElement instead of START.
430 var reverseLayout = (goog.style.isRightToLeft(this.bubbleContainer_) !=
431 goog.style.isRightToLeft(targetElement));
432
433 // Try to put the bubble at the bottom of the target unless the plugin has
434 // requested otherwise.
435 if (preferBottomPosition) {
436 status = this.positionAtAnchor_(reverseLayout ?
437 goog.positioning.Corner.BOTTOM_END :
438 goog.positioning.Corner.BOTTOM_START,
439 goog.positioning.Corner.TOP_START,
440 goog.positioning.Overflow.ADJUST_X | goog.positioning.Overflow.FAIL_Y);
441 }
442
443 if (status & goog.positioning.OverflowStatus.FAILED) {
444 // Try to put it at the top of the target if there is not enough
445 // space at the bottom.
446 status = this.positionAtAnchor_(reverseLayout ?
447 goog.positioning.Corner.TOP_END : goog.positioning.Corner.TOP_START,
448 goog.positioning.Corner.BOTTOM_START,
449 goog.positioning.Overflow.ADJUST_X | goog.positioning.Overflow.FAIL_Y);
450 }
451
452 if (status & goog.positioning.OverflowStatus.FAILED) {
453 // Put it at the bottom again with adjustment if there is no
454 // enough space at the top.
455 status = this.positionAtAnchor_(reverseLayout ?
456 goog.positioning.Corner.BOTTOM_END :
457 goog.positioning.Corner.BOTTOM_START,
458 goog.positioning.Corner.TOP_START,
459 goog.positioning.Overflow.ADJUST_X |
460 goog.positioning.Overflow.ADJUST_Y);
461 if (status & goog.positioning.OverflowStatus.FAILED) {
462 goog.log.warning(this.logger,
463 'reposition(): positionAtAnchor() failed with ' + status);
464 }
465 }
466};

469/**
470 * A helper for reposition() - positions the bubble in regards to the position
471 * of the elements the bubble is attached to.
472 * @param {goog.positioning.Corner} targetCorner The corner of
473 * the target element.
474 * @param {goog.positioning.Corner} bubbleCorner The corner of the bubble.
475 * @param {number} overflow Overflow handling mode bitmap,
476 * {@see goog.positioning.Overflow}.
477 * @return {number} Status bitmap, {@see goog.positioning.OverflowStatus}.
478 * @private
479 */
480goog.ui.editor.Bubble.prototype.positionAtAnchor_ = function(
481 targetCorner, bubbleCorner, overflow) {
482 var targetElement = null;
483 for (var panelId in this.panels_) {
484 // For now, we use the outermost element. This assumes the multiple
485 // elements this panel is showing for contain each other - in the event
486 // that is not generally the case this may need to be updated to pick
487 // the lowest or highest element depending on targetCorner.
488 var candidate = this.panels_[panelId].targetElement;
489 if (!targetElement || goog.dom.contains(candidate, targetElement)) {
490 targetElement = this.panels_[panelId].targetElement;
491 }
492 }
493 return goog.positioning.positionAtAnchor(
494 targetElement, targetCorner, this.bubbleContainer_,
495 bubbleCorner, null, this.getMarginBox(), overflow,
496 null, this.getViewportBox());
497};

528
529 /**
