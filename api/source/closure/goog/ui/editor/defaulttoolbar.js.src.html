140/**
141 * Initializes the given font menu button by adding default fonts to the menu.
142 * If goog.ui.editor.DefaultToolbar.setLocale was called to specify a locale
143 * for which locale-specific default fonts exist, those are added before
144 * common fonts.
145 * @param {!goog.ui.Select} button Font menu button.
146 */
147goog.ui.editor.DefaultToolbar.addDefaultFonts = function(button) {
148 // Normalize locale to lowercase, with a hyphen (see bug 1036165).
149 var locale =
150 goog.ui.editor.DefaultToolbar.locale_.replace(/_/, '-').toLowerCase();
151 // Add locale-specific default fonts, if any.
152 var fontlist = [];
153
154 if (locale in goog.ui.editor.DefaultToolbar.I18N_FONTS_) {
155 fontlist = goog.ui.editor.DefaultToolbar.I18N_FONTS_[locale];
156 }
157 if (fontlist.length) {
158 goog.ui.editor.ToolbarFactory.addFonts(button, fontlist);
159 }
160 // Add locale-independent default fonts.
161 goog.ui.editor.ToolbarFactory.addFonts(button,
162 goog.ui.editor.DefaultToolbar.FONTS_);
163};

274/**
275 * Creates a {@link goog.ui.Toolbar} containing a default set of editor
276 * toolbar buttons, and renders it into the given parent element.
277 * @param {!Element} elem Toolbar parent element.
278 * @param {boolean=} opt_isRightToLeft Whether the editor chrome is
279 * right-to-left; defaults to the directionality of the toolbar parent
280 * element.
281 * @return {!goog.ui.Toolbar} Default editor toolbar, rendered into the given
282 * parent element.
283 * @see goog.ui.editor.DefaultToolbar.DEFAULT_BUTTONS
284 */
285goog.ui.editor.DefaultToolbar.makeDefaultToolbar = function(elem,
286 opt_isRightToLeft) {
287 var isRightToLeft = opt_isRightToLeft || goog.style.isRightToLeft(elem);
288 var buttons = isRightToLeft ?
289 goog.ui.editor.DefaultToolbar.DEFAULT_BUTTONS_RTL :
290 goog.ui.editor.DefaultToolbar.DEFAULT_BUTTONS;
291 return goog.ui.editor.DefaultToolbar.makeToolbar(buttons, elem,
292 opt_isRightToLeft);
293};

296/**
297 * Creates a {@link goog.ui.Toolbar} containing the specified set of
298 * toolbar buttons, and renders it into the given parent element. Each
299 * item in the {@code items} array must either be a
300 * {@link goog.editor.Command} (to create a built-in button) or a subclass
301 * of {@link goog.ui.Control} (to create a custom control).
302 * @param {!Array<string|goog.ui.Control>} items Toolbar items; each must
303 * be a {@link goog.editor.Command} or a {@link goog.ui.Control}.
304 * @param {!Element} elem Toolbar parent element.
305 * @param {boolean=} opt_isRightToLeft Whether the editor chrome is
306 * right-to-left; defaults to the directionality of the toolbar parent
307 * element.
308 * @return {!goog.ui.Toolbar} Editor toolbar, rendered into the given parent
309 * element.
310 */
311goog.ui.editor.DefaultToolbar.makeToolbar = function(items, elem,
312 opt_isRightToLeft) {
313 var domHelper = goog.dom.getDomHelper(elem);
314 var controls = [];
315
316 for (var i = 0, button; button = items[i]; i++) {
317 if (goog.isString(button)) {
318 button = goog.ui.editor.DefaultToolbar.makeBuiltInToolbarButton(button,
319 domHelper);
320 }
321 if (button) {
322 controls.push(button);
323 }
324 }
325
326 return goog.ui.editor.ToolbarFactory.makeToolbar(controls, elem,
327 opt_isRightToLeft);
328};

331/**
332 * Creates an instance of a subclass of {@link goog.ui.Button} for the given
333 * {@link goog.editor.Command}, or null if no built-in button exists for the
334 * command. Note that this function is only intended to create built-in
335 * buttons; please don't try to hack it!
336 * @param {string} command Editor command ID.
337 * @param {goog.dom.DomHelper=} opt_domHelper DOM helper, used for DOM
338 * creation; defaults to the current document if unspecified.
339 * @return {goog.ui.Button} Toolbar button (null if no built-in button exists
340 * for the command).
341 */
342goog.ui.editor.DefaultToolbar.makeBuiltInToolbarButton = function(command,
343 opt_domHelper) {
344 var button;
345 var descriptor = goog.ui.editor.DefaultToolbar.buttons_[command];
346 if (descriptor) {
347 // Default the factory method to makeToggleButton, since most built-in
348 // toolbar buttons are toggle buttons. See also
349 // goog.ui.editor.DefaultToolbar.button_list_.
350 var factory = descriptor.factory ||
351 goog.ui.editor.ToolbarFactory.makeToggleButton;
352 var id = descriptor.command;
353 var tooltip = descriptor.tooltip;
354 var caption = descriptor.caption;
355 var classNames = descriptor.classes;
356 // Default the DOM helper to the one for the current document.
357 var domHelper = opt_domHelper || goog.dom.getDomHelper();
358 // Instantiate the button based on the descriptor.
359 button = factory(id, tooltip, caption, classNames, null, domHelper);
360 // If this button's state should be queried when updating the toolbar,
361 // set the button object's queryable property to true.
362 if (descriptor.queryable) {
363 button.queryable = true;
364 }
365 }
366 return button;
367};

413/**
414 * Creates a toolbar button with the given ID, tooltip, and caption. Applies
415 * any custom CSS class names to the button's caption element. This button
416 * is designed to be used as the RTL button.
417 * @param {string} id Button ID; must equal a {@link goog.editor.Command} for
418 * built-in buttons, anything else for custom buttons.
419 * @param {string} tooltip Tooltip to be shown on hover.
420 * @param {goog.ui.ControlContent} caption Button caption.
421 * @param {string=} opt_classNames CSS class name(s) to apply to the caption
422 * element.
423 * @param {goog.ui.ButtonRenderer=} opt_renderer Button renderer; defaults to
424 * {@link goog.ui.ToolbarButtonRenderer} if unspecified.
425 * @param {goog.dom.DomHelper=} opt_domHelper DOM helper, used for DOM
426 * creation; defaults to the current document if unspecified.
427 * @return {!goog.ui.Button} A toolbar button.
428 * @private
429 */
430goog.ui.editor.DefaultToolbar.rtlButtonFactory_ = function(id, tooltip,
431 caption, opt_classNames, opt_renderer, opt_domHelper) {
432 var button = goog.ui.editor.ToolbarFactory.makeToggleButton(id, tooltip,
433 caption, opt_classNames, opt_renderer, opt_domHelper);
434 button.updateFromValue = function(value) {
435 // Enable/disable right-to-left text editing mode in the toolbar.
436 var isRtl = !!value;
437 // Enable/disable a marker class on the toolbar's root element; the rest is
438 // done using CSS scoping in editortoolbar.css. This changes
439 // direction-senitive toolbar icons (like indent/outdent)
440 goog.dom.classlist.enable(
441 goog.asserts.assert(button.getParent().getElement()),
442 goog.getCssName('tr-rtl-mode'), isRtl);
443 button.setChecked(isRtl);
444 };
445 return button;
446};

484 * @param {goog.ui.ControlContent} caption Button caption.
485 * @param {string=} opt_classNames CSS class name(s) to apply to the caption
486 * element.
487 * @param {goog.ui.MenuButtonRenderer=} opt_renderer Button renderer; defaults
488 * to {@link goog.ui.ToolbarMenuButtonRenderer} if unspecified.
489 * @param {goog.dom.DomHelper=} opt_domHelper DOM helper, used for DOM
490 * creation; defaults to the current document if unspecified.
491 * @return {!goog.ui.Button} A toolbar button.
492 * @private
493 */
494goog.ui.editor.DefaultToolbar.fontFaceFactory_ = function(id, tooltip,
495 caption, opt_classNames, opt_renderer, opt_domHelper) {
496 var button = goog.ui.editor.ToolbarFactory.makeSelectButton(id, tooltip,
497 caption, opt_classNames, opt_renderer, opt_domHelper);
498 goog.ui.editor.DefaultToolbar.addDefaultFonts(button);
499 button.setDefaultCaption(goog.ui.editor.DefaultToolbar.MSG_FONT_NORMAL);
500 // Font options don't have keyboard accelerators.
501 goog.dom.classlist.add(
502 goog.asserts.assert(button.getMenu().getContentElement()),
503 goog.getCssName('goog-menu-noaccel'));
504
505 // How to update this button's state.
506 button.updateFromValue = function(value) {
507 // Normalize value to null or a non-empty string (sometimes we get
508 // the empty string, sometimes we get false...), extract the substring
509 // up to the first comma to get the primary font name, and normalize
510 // to lowercase. This allows us to map a font spec like "Arial,
511 // Helvetica, sans-serif" to a font menu item.
512 // TODO (attila): Try to make this more robust.
513 var item = null;
514 if (value && value.length > 0) {
515 item = /** @type {goog.ui.MenuItem} */ (button.getMenu().getChild(
516 goog.ui.editor.ToolbarFactory.getPrimaryFont(value)));
517 }
518 var selectedItem = button.getSelectedItem();
519 if (item != selectedItem) {
520 button.setSelectedItem(item);
521 }
522 };
523 return button;
524};

531 * @param {string} id Button ID; must equal a {@link goog.editor.Command} for

533 * @param {string} tooltip Tooltip to be shown on hover.
534 * @param {goog.ui.ControlContent} caption Button caption.
535 * @param {string=} opt_classNames CSS class name(s) to apply to the caption
536 * element.
537 * @param {goog.ui.MenuButtonRenderer=} opt_renderer Button renderer; defaults
538 * to {@link goog.ui.ToolbarMebuButtonRenderer} if unspecified.
539 * @param {goog.dom.DomHelper=} opt_domHelper DOM helper, used for DOM
540 * creation; defaults to the current document if unspecified.
541 * @return {!goog.ui.Button} A toolbar button.
542 * @private
543 */
544goog.ui.editor.DefaultToolbar.fontSizeFactory_ = function(id, tooltip,
545 caption, opt_classNames, opt_renderer, opt_domHelper) {
546 var button = goog.ui.editor.ToolbarFactory.makeSelectButton(id, tooltip,
547 caption, opt_classNames, opt_renderer, opt_domHelper);
548 goog.ui.editor.DefaultToolbar.addDefaultFontSizes(button);
549 button.setDefaultCaption(goog.ui.editor.DefaultToolbar.MSG_FONT_SIZE_NORMAL);
550 // Font size options don't have keyboard accelerators.
551 goog.dom.classlist.add(
552 goog.asserts.assert(button.getMenu().getContentElement()),
553 goog.getCssName('goog-menu-noaccel'));
554 // How to update this button's state.
555 button.updateFromValue = function(value) {
556 // Webkit pre-534.7 returns a string like '32px' instead of the equivalent
557 // integer, so normalize that first.
558 // NOTE(user): Gecko returns "6" so can't just normalize all
559 // strings, only ones ending in "px".
560 if (goog.isString(value) &&
561 goog.style.getLengthUnits(value) == 'px') {
562 value = goog.ui.editor.ToolbarFactory.getLegacySizeFromPx(
563 parseInt(value, 10));
564 }
565 // Normalize value to null or a positive integer (sometimes we get
566 // the empty string, sometimes we get false, or -1 if the above
567 // normalization didn't match to a particular 0-7 size)
568 value = value > 0 ? value : null;
569 if (value != button.getValue()) {
570 button.setValue(value);
571 }
572 };
573 return button;
574};

577/**
578 * Function to update the state of a color menu button.
579 * @param {goog.ui.ToolbarColorMenuButton} button The button to which the
580 * color menu is attached.
581 * @param {number} color Color value to update to.
582 * @private
583 */
584goog.ui.editor.DefaultToolbar.colorUpdateFromValue_ = function(button, color) {
585 var value = color;
586 /** @preserveTry */
587 try {
588 if (goog.userAgent.IE) {
589 // IE returns a number that, converted to hex, is a BGR color.
590 // Convert from decimal to BGR to RGB.
591 var hex = '000000' + value.toString(16);
592 var bgr = hex.substr(hex.length - 6, 6);
593 value = '#' + bgr.substring(4, 6) + bgr.substring(2, 4) +
594 bgr.substring(0, 2);
595 }
596 if (value != button.getValue()) {
597 button.setValue(/** @type {string} */ (value));
598 }
599 } catch (ex) {
600 // TODO(attila): Find out when/why this happens.
601 }
602};
