16/**
17 * @fileoverview provides a reusable FlickrSet photo UI component given a public
18 * FlickrSetModel.
19 *
20 * goog.ui.media.FlickrSet is actually a {@link goog.ui.ControlRenderer}, a
21 * stateless class - that could/should be used as a Singleton with the static
22 * method {@code goog.ui.media.FlickrSet.getInstance} -, that knows how to
23 * render Flickr sets. It is designed to be used with a {@link goog.ui.Control},
24 * which will actually control the media renderer and provide the
25 * {@link goog.ui.Component} base. This design guarantees that all different
26 * types of medias will behave alike but will look different.
27 *
28 * goog.ui.media.FlickrSet expects a {@code goog.ui.media.FlickrSetModel} on
29 * {@code goog.ui.Control.getModel} as data models, and renders a flash object
30 * that will show the contents of that set.
31 *
32 * Example of usage:
33 *
34 * <pre>
35 * var flickrSet = goog.ui.media.FlickrSetModel.newInstance(flickrSetUrl);
36 * goog.ui.media.FlickrSet.newControl(flickrSet).render();
37 * </pre>
38 *
39 * FlickrSet medias currently support the following states:
40 *
41 * <ul>
42 * <li> {@link goog.ui.Component.State.DISABLED}: shows 'flash not available'
43 * <li> {@link goog.ui.Component.State.HOVER}: mouse cursor is over the video
44 * <li> {@link goog.ui.Component.State.SELECTED}: flash video is shown
45 * </ul>
46 *
47 * Which can be accessed by
48 * <pre>
49 * video.setEnabled(true);
50 * video.setHighlighted(true);
51 * video.setSelected(true);
52 * </pre>
53 *
54 *
55 * @supported IE6, FF2+, Safari. Requires flash to actually work.
56 *
57 * TODO(user): Support non flash users. Maybe show a link to the Flick set,
58 * or fetch the data and rendering it using javascript (instead of a broken
59 * 'You need to install flash' message).
60 */

74/**
75 * Subclasses a goog.ui.media.MediaRenderer to provide a FlickrSet specific
76 * media renderer.
77 *
78 * This class knows how to parse FlickrSet URLs, and render the DOM structure
79 * of flickr set players. This class is meant to be used as a singleton static
80 * stateless class, that takes {@code goog.ui.media.Media} instances and renders
81 * it. It expects {@code goog.ui.media.Media.getModel} to return a well formed,
82 * previously constructed, set id {@see goog.ui.media.FlickrSet.parseUrl},
83 * which is the data model this renderer will use to construct the DOM
84 * structure. {@see goog.ui.media.FlickrSet.newControl} for a example of
85 * constructing a control with this renderer.
86 *
87 * This design is patterned after
88 * http://go/closure_control_subclassing
89 *
90 * It uses {@link goog.ui.media.FlashObject} to embed the flash object.
91 *
92 * @constructor
93 * @extends {goog.ui.media.MediaRenderer}
94 * @final
95 */
96goog.ui.media.FlickrSet = function() {
97 goog.ui.media.MediaRenderer.call(this);
98};
99goog.inherits(goog.ui.media.FlickrSet, goog.ui.media.MediaRenderer);
100goog.addSingletonGetter(goog.ui.media.FlickrSet);

123/**
124 * A static convenient method to construct a goog.ui.media.Media control out of
125 * a FlickrSet URL. It extracts the set id information on the URL, sets it
126 * as the data model goog.ui.media.FlickrSet renderer uses, sets the states
127 * supported by the renderer, and returns a Control that binds everything
128 * together. This is what you should be using for constructing FlickrSet videos,
129 * except if you need more fine control over the configuration.
130 *
131 * @param {goog.ui.media.FlickrSetModel} dataModel The Flickr Set data model.
132 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper, used for
133 * document interaction.
134 * @return {!goog.ui.media.Media} A Control binded to the FlickrSet renderer.
135 * @throws exception in case {@code flickrSetUrl} is an invalid flickr set URL.
136 * TODO(user): use {@link goog.ui.media.MediaModel} once it is checked in.
137 */
138goog.ui.media.FlickrSet.newControl = function(dataModel, opt_domHelper) {
139 var control = new goog.ui.media.Media(
140 dataModel, goog.ui.media.FlickrSet.getInstance(), opt_domHelper);
141 control.setSelected(true);
142 return control;
143};

158/**
159 * Creates the initial DOM structure of the flickr set, which is basically a
160 * the flash object pointing to a flickr set player.
161 *
162 * @param {goog.ui.Control} c The media control.
163 * @return {!Element} The DOM structure that represents this control.
164 * @override
165 */
166goog.ui.media.FlickrSet.prototype.createDom = function(c) {
167 var control = /** @type {goog.ui.media.Media} */ (c);
168 var div = goog.ui.media.FlickrSet.superClass_.createDom.call(this, control);
169
170 var model =
171 /** @type {goog.ui.media.FlickrSetModel} */ (control.getDataModel());
172
173 // TODO(user): find out what is the policy about hosting this SWF. figure out
174 // if it works over https.
175 var flash = new goog.ui.media.FlashObject(
176 model.getPlayer().getTrustedResourceUrl(),
177 control.getDomHelper());
178 flash.addFlashVars(model.getPlayer().getVars());
179 flash.render(div);
180
181 return div;
182};

197/**
198 * The {@code goog.ui.media.FlickrAlbum} media data model. It stores a required
199 * {@code userId} and {@code setId} fields, sets the flickr Set URL, and
200 * allows a few optional parameters.
201 *
202 * @param {string} userId The flickr userId associated with this set.
203 * @param {string} setId The flickr setId associated with this set.
204 * @param {string=} opt_caption An optional caption of the flickr set.
205 * @param {string=} opt_description An optional description of the flickr set.
206 * @constructor
207 * @extends {goog.ui.media.MediaModel}
208 * @final
209 */
210goog.ui.media.FlickrSetModel = function(userId,
211 setId,
212 opt_caption,
213 opt_description) {
214 goog.ui.media.MediaModel.call(
215 this,
216 goog.ui.media.FlickrSetModel.buildUrl(userId, setId),
217 opt_caption,
218 opt_description,
219 goog.ui.media.MediaModel.MimeType.FLASH);
220
221 /**
222 * The Flickr user id.
223 * @type {string}
224 * @private
225 */
226 this.userId_ = userId;
227
228 /**
229 * The Flickr set id.
230 * @type {string}
231 * @private
232 */
233 this.setId_ = setId;
234
235 var flashVars = {
236 'offsite': 'true',
237 'lang': 'en',
238 'page_show_url': '/photos/' + userId + '/sets/' + setId + '/show/',
239 'page_show_back_url': '/photos/' + userId + '/sets/' + setId,
240 'set_id': setId
241 };
242
243 var player = new goog.ui.media.MediaModel.Player(
244 goog.ui.media.FlickrSet.flashUrl_, flashVars);
245
246 this.setPlayer(player);
247};
248goog.inherits(goog.ui.media.FlickrSetModel, goog.ui.media.MediaModel);

251/**
252 * Regular expression used to extract the username and set id out of the flickr
253 * URLs.
254 *
255 * Copied from http://go/markdownlite.js and {@link FlickrExtractor.xml}.
256 *
257 * @type {RegExp}
258 * @private
259 * @const
260 */
261goog.ui.media.FlickrSetModel.MATCHER_ =
262 /(?:http:\/\/)?(?:www\.)?flickr\.com\/(?:photos\/([\d\w@\-]+)\/sets\/(\d+))\/?/i;
263
264
265/**
266 * Takes a {@code flickrSetUrl} and extracts the flickr username and set id.
267 *
268 * @param {string} flickrSetUrl A Flickr set URL.
269 * @param {string=} opt_caption An optional caption of the flickr set.
270 * @param {string=} opt_description An optional description of the flickr set.
271 * @return {!goog.ui.media.FlickrSetModel} The data model that represents the
272 * Flickr set.
273 * @throws exception in case the parsing fails
274 */
275goog.ui.media.FlickrSetModel.newInstance = function(flickrSetUrl,
276 opt_caption,
277 opt_description) {
278 if (goog.ui.media.FlickrSetModel.MATCHER_.test(flickrSetUrl)) {
279 var data = goog.ui.media.FlickrSetModel.MATCHER_.exec(flickrSetUrl);
280 return new goog.ui.media.FlickrSetModel(
281 data[1], data[2], opt_caption, opt_description);
282 }
283 throw Error('failed to parse flickr url: ' + flickrSetUrl);
284};

287/**
288 * Takes a flickr username and set id and returns an URL.
289 *
290 * @param {string} userId The owner of the set.
291 * @param {string} setId The set id.
292 * @return {string} The URL of the set.
293 */
294goog.ui.media.FlickrSetModel.buildUrl = function(userId, setId) {
295 return 'http://flickr.com/photos/' + userId + '/sets/' + setId;
