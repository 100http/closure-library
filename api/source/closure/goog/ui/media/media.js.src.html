15/**
16 * @fileoverview Provides the base goog.ui.Control and goog.ui.ControlRenderer
17 * for media types, as well as a media model consistent with the Yahoo Media RSS
18 * specification {@link http://search.yahoo.com/mrss/}.
19 *
20 * The goog.ui.media.* package is basically a set of goog.ui.ControlRenderers
21 * subclasses (such as goog.ui.media.Youtube, goog.ui.media.Picasa, etc) that
22 * should all work with the same goog.ui.Control (goog.ui.media.Media) logic.
23 *
24 * This design guarantees that all different types of medias will behave alike
25 * (in a base level) but will look different.
26 *
27 * In MVC terms, {@link goog.ui.media.Media} is the Controller,
28 * {@link goog.ui.media.MediaRenderer} + CSS definitions are the View and
29 * {@code goog.ui.media.MediaModel} is the data Model. Typically,
30 * MediaRenderer will be subclassed to provide media specific renderers.
31 * MediaRenderer subclasses are also responsible for defining the data model.
32 *
33 * This design is strongly patterned after:
34 * http://go/closure_control_subclassing
35 *
36 * goog.ui.media.MediaRenderer handles the basic common ways to display media,
37 * such as displaying tooltips, frames, minimize/maximize buttons, play buttons,
38 * etc. Its subclasses are responsible for rendering media specific DOM
39 * structures, like youtube flash players, picasa albums, etc.
40 *
41 * goog.ui.media.Media handles the Control of Medias, by listening to events
42 * and firing the appropriate actions. It knows about the existence of captions,
43 * minimize/maximize buttons, and takes all the actions needed to change states,
44 * including delegating the UI actions to MediaRenderers.
45 *
46 * Although MediaRenderer is a base class designed to be subclassed, it can
47 * be used by itself:
48 *
49 * <pre>
50 * var renderer = new goog.ui.media.MediaRenderer();
51 * var control = new goog.ui.media.Media('hello world', renderer);
52 * var control.render(goog.dom.getElement('mediaHolder'));
53 * </pre>
54 *
55 * It requires a few CSS rules to be defined, which you should use to control
56 * how the component is displayed. {@link goog.ui.ControlRenderer}s is very CSS
57 * intensive, which separates the UI structure (the HTML DOM elements, which is
58 * created by the {@code goog.ui.media.MediaRenderer}) from the UI view (which
59 * nodes are visible, which aren't, where they are positioned. These are defined
60 * on the CSS rules for each state). A few examples of CSS selectors that needs
61 * to be defined are:
62 *
63 * <ul>
64 * <li>.goog-ui-media
65 * <li>.goog-ui-media-hover
66 * <li>.goog-ui-media-selected
67 * </ul>
68 *
69 * If you want to have different custom renderers CSS namespaces (eg. you may
70 * want to show a small thumbnail, or you may want to hide the caption, etc),
71 * you can do so by using:
72 *
73 * <pre>
74 * var renderer = goog.ui.ControlRenderer.getCustomRenderer(
75 * goog.ui.media.MediaRenderer, 'my-custom-namespace');
76 * var media = new goog.ui.media.Media('', renderer);
77 * media.render(goog.dom.getElement('parent'));
78 * </pre>
79 *
80 * Which will allow you to set your own .my-custom-namespace-hover,
81 * .my-custom-namespace-selected CSS selectors.
82 *
83 * NOTE(user): it seems like an overkill to subclass goog.ui.Control instead of
84 * using a factory, but we wanted to make sure we had more control over the
85 * events for future media implementations. Since we intent to use it in many
86 * different places, it makes sense to have a more flexible design that lets us
87 * control the inner workings of goog.ui.Control.
88 *
89 * TODO(user): implement, as needed, the Media specific state changes UI, such
90 * as minimize/maximize buttons, expand/close buttons, etc.
91 *
92 */

106/**
107 * Provides the control mechanism of media types.
108 *
109 * @param {goog.ui.media.MediaModel} dataModel The data model to be used by the
110 * renderer.
111 * @param {goog.ui.ControlRenderer=} opt_renderer Renderer used to render or
112 * decorate the component; defaults to {@link goog.ui.ControlRenderer}.
113 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper, used for
114 * document interaction.
115 * @constructor
116 * @extends {goog.ui.Control}
117 * @final
118 */
119goog.ui.media.Media = function(dataModel, opt_renderer, opt_domHelper) {
120 goog.ui.Control.call(this, null, opt_renderer, opt_domHelper);
121
122 // Sets up the data model.
123 this.setDataModel(dataModel);
124 this.setSupportedState(goog.ui.Component.State.OPENED, true);
125 this.setSupportedState(goog.ui.Component.State.SELECTED, true);
126 // TODO(user): had to do this to for mouseDownHandler not to
127 // e.preventDefault(), because it was not allowing the event to reach the
128 // flash player. figure out a better way to not e.preventDefault().
129 this.setAllowTextSelection(true);
130
131 // Media items don't use RTL styles, so avoid accessing computed styles to
132 // figure out if the control is RTL.
133 this.setRightToLeft(false);
134};
135goog.inherits(goog.ui.media.Media, goog.ui.Control);

193/**
194 * Builds the common DOM structure of medias. Builds an outer div, and appends
195 * a child div with the {@code goog.ui.Control.getContent} content. Marks the
196 * caption with a {@code this.getClassClass()} + '-caption' css flag, so that
197 * specific renderers can hide/show the caption as desired.
198 *
199 * @param {goog.ui.Control} control The control instance.
200 * @return {!Element} The DOM structure that represents control.
201 * @override
202 */
203goog.ui.media.MediaRenderer.prototype.createDom = function(control) {
204 goog.asserts.assertInstanceof(control, goog.ui.media.Media);
205 var domHelper = control.getDomHelper();
206 var div = domHelper.createElement(goog.dom.TagName.DIV);
207 div.className = this.getClassNames(control).join(' ');
208
209 var dataModel = control.getDataModel();
210
211 // Only creates DOMs if the data is available.
212 if (dataModel.getCaption()) {
213 var caption = domHelper.createElement(goog.dom.TagName.DIV);
214 caption.className = goog.getCssName(this.getCssClass(), 'caption');
215 caption.appendChild(domHelper.createDom(
216 goog.dom.TagName.P, goog.getCssName(this.getCssClass(), 'caption-text'),
217 dataModel.getCaption()));
218 domHelper.appendChild(div, caption);
219 }
220
221 if (dataModel.getDescription()) {
222 var description = domHelper.createElement(goog.dom.TagName.DIV);
223 description.className = goog.getCssName(this.getCssClass(), 'description');
224 description.appendChild(domHelper.createDom(
225 goog.dom.TagName.P,
226 goog.getCssName(this.getCssClass(), 'description-text'),
227 dataModel.getDescription()));
228 domHelper.appendChild(div, description);
229 }
230
231 // Creates thumbnails of the media.
232 var thumbnails = dataModel.getThumbnails() || [];
233 for (var index = 0; index < thumbnails.length; index++) {
234 var thumbnail = thumbnails[index];
235 var thumbnailElement = domHelper.createElement(goog.dom.TagName.IMG);
236 thumbnailElement.src = thumbnail.getUrl();
237 thumbnailElement.className = this.getThumbnailCssName(index);
238
239 // Check that the size is defined and that the size's height and width
240 // are defined. Undefined height and width is deprecated but still
241 // seems to exist in some cases.
242 var size = thumbnail.getSize();
243
244 if (size && goog.isDefAndNotNull(size.height) &&
245 goog.isDefAndNotNull(size.width)) {
246 goog.style.setSize(thumbnailElement, size);
247 }
248 domHelper.appendChild(div, thumbnailElement);
249 }
250
251 if (dataModel.getPlayer()) {
252 // if medias have players, allow UI for a play button.
253 var playButton = domHelper.createElement(goog.dom.TagName.DIV);
254 playButton.className = goog.getCssName(this.getCssClass(), 'playbutton');
255 domHelper.appendChild(div, playButton);
256 }
257
258 control.setElementInternal(div);
259
260 this.setState(
261 control,
262 /** @type {goog.ui.Component.State} */ (control.getState()),
263 true);
264
265 return div;
266};

269/**
270 * Returns a renamable CSS class name for a numbered thumbnail. The default
271 * implementation generates the class names goog-ui-media-thumbnail0,
272 * goog-ui-media-thumbnail1, and the generic goog-ui-media-thumbnailn.
273 * Subclasses can override this method when their media requires additional
274 * specific class names (Applications are supposed to know how many thumbnails
275 * media will have).
276 *
277 * @param {number} index The thumbnail index.
278 * @return {string} CSS class name.
279 * @protected
280 */
281goog.ui.media.MediaRenderer.prototype.getThumbnailCssName = function(index) {
282 switch (index) {
283 case 0: return goog.getCssName(this.getCssClass(), 'thumbnail0');
284 case 1: return goog.getCssName(this.getCssClass(), 'thumbnail1');
285 case 2: return goog.getCssName(this.getCssClass(), 'thumbnail2');
286 case 3: return goog.getCssName(this.getCssClass(), 'thumbnail3');
287 case 4: return goog.getCssName(this.getCssClass(), 'thumbnail4');
288 default: return goog.getCssName(this.getCssClass(), 'thumbnailn');
289 }
