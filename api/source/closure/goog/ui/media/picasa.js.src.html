16/**
17 * @fileoverview provides a reusable picasa album UI component given a public
18 * picasa album URL.
19 *
20 * TODO(user): implement the javascript viewer, for users without flash. Get it
21 * from the Gmail Picasa gadget.
22 *
23 * goog.ui.media.PicasaAlbum is actually a {@link goog.ui.ControlRenderer}, a
24 * stateless class - that could/should be used as a Singleton with the static
25 * method {@code goog.ui.media.PicasaAlbum.getInstance} -, that knows how to
26 * render picasa albums. It is designed to be used with a
27 * {@link goog.ui.Control}, which will actually control the media renderer and
28 * provide the {@link goog.ui.Component} base. This design guarantees that all
29 * different types of medias will behave alike but will look different.
30 *
31 * goog.ui.media.PicasaAlbum expects {@code goog.ui.media.PicasaAlbumModel}s on
32 * {@code goog.ui.Control.getModel} as data models, and render a flash object
33 * that will show a slideshow with the contents of that album URL.
34 *
35 * Example of usage:
36 *
37 * <pre>
38 * var album = goog.ui.media.PicasaAlbumModel.newInstance(
39 * 'http://picasaweb.google.com/username/SanFranciscoCalifornia');
40 * goog.ui.media.PicasaAlbum.newControl(album).render();
41 * </pre>
42 *
43 * picasa medias currently support the following states:
44 *
45 * <ul>
46 * <li> {@link goog.ui.Component.State.DISABLED}: shows 'flash not available'
47 * <li> {@link goog.ui.Component.State.HOVER}: mouse cursor is over the album
48 * <li> {@link goog.ui.Component.State.SELECTED}: flash album is shown
49 * </ul>
50 *
51 * Which can be accessed by
52 *
53 * <pre>
54 * picasa.setEnabled(true);
55 * picasa.setHighlighted(true);
56 * picasa.setSelected(true);
57 * </pre>
58 *
59 *
60 * @supported IE6, FF2+, Safari. Requires flash to actually work.
61 *
62 * TODO(user): test on other browsers
63 */

77/**
78 * Subclasses a goog.ui.media.MediaRenderer to provide a Picasa specific media
79 * renderer.
80 *
81 * This class knows how to parse picasa URLs, and render the DOM structure
82 * of picasa album players and previews. This class is meant to be used as a
83 * singleton static stateless class, that takes {@code goog.ui.media.Media}
84 * instances and renders it. It expects {@code goog.ui.media.Media.getModel} to
85 * return a well formed, previously constructed, object with a user and album
86 * fields {@see goog.ui.media.PicasaAlbum.parseUrl}, which is the data model
87 * this renderer will use to construct the DOM structure.
88 * {@see goog.ui.media.PicasaAlbum.newControl} for a example of constructing a
89 * control with this renderer.
90 *
91 * goog.ui.media.PicasaAlbum currently displays a picasa-made flash slideshow
92 * with the photos, but could possibly display a handwritten js photo viewer,
93 * in case flash is not available.
94 *
95 * This design is patterned after http://go/closure_control_subclassing
96 *
97 * It uses {@link goog.ui.media.FlashObject} to embed the flash object.
98 *
99 * @constructor
100 * @extends {goog.ui.media.MediaRenderer}
101 * @final
102 */
103goog.ui.media.PicasaAlbum = function() {
104 goog.ui.media.MediaRenderer.call(this);
105};
106goog.inherits(goog.ui.media.PicasaAlbum, goog.ui.media.MediaRenderer);
107goog.addSingletonGetter(goog.ui.media.PicasaAlbum);

119/**
120 * A static convenient method to construct a goog.ui.media.Media control out of
121 * a picasa data model. It sets it as the data model goog.ui.media.PicasaAlbum
122 * renderer uses, sets the states supported by the renderer, and returns a
123 * Control that binds everything together. This is what you should be using for
124 * constructing Picasa albums, except if you need finer control over the
125 * configuration.
126 *
127 * @param {goog.ui.media.PicasaAlbumModel} dataModel A picasa album data model.
128 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper, used for
129 * document interaction.
130 * @return {!goog.ui.media.Media} A Control instance binded to the Picasa
131 * renderer.
132 */
133goog.ui.media.PicasaAlbum.newControl = function(dataModel, opt_domHelper) {
134 var control = new goog.ui.media.Media(
135 dataModel,
136 goog.ui.media.PicasaAlbum.getInstance(),
137 opt_domHelper);
138 control.setSelected(true);
139 return control;
140};

143/**
144 * Creates the initial DOM structure of the picasa album, which is basically a
145 * the flash object pointing to a flash picasa album player.
146 *
147 * @param {goog.ui.Control} c The media control.
148 * @return {!Element} The DOM structure that represents the control.
149 * @override
150 */
151goog.ui.media.PicasaAlbum.prototype.createDom = function(c) {
152 var control = /** @type {goog.ui.media.Media} */ (c);
153 var div = goog.ui.media.PicasaAlbum.superClass_.createDom.call(this, control);
154
155 var picasaAlbum =
156 /** @type {goog.ui.media.PicasaAlbumModel} */ (control.getDataModel());
157 var authParam =
158 picasaAlbum.getAuthKey() ? ('&authkey=' + picasaAlbum.getAuthKey()) : '';
159 var flash = new goog.ui.media.FlashObject(
160 picasaAlbum.getPlayer().getTrustedResourceUrl(),
161 control.getDomHelper());
162 flash.addFlashVars(picasaAlbum.getPlayer().getVars());
163 flash.render(div);
164
165 return div;
166};

181/**
182 * The {@code goog.ui.media.PicasaAlbum} media data model. It stores a required
183 * {@code userId} and {@code albumId} fields, sets the picasa album URL, and
184 * allows a few optional parameters.
185 *
186 * @param {string} userId The picasa userId associated with this album.
187 * @param {string} albumId The picasa albumId associated with this album.
188 * @param {string=} opt_authKey An optional authentication key, used on private
189 * albums.
190 * @param {string=} opt_caption An optional caption of the picasa album.
191 * @param {string=} opt_description An optional description of the picasa album.
192 * @param {boolean=} opt_autoplay Whether to autoplay the slideshow.
193 * @constructor
194 * @extends {goog.ui.media.MediaModel}
195 * @final
196 */
197goog.ui.media.PicasaAlbumModel = function(userId,
198 albumId,
199 opt_authKey,
200 opt_caption,
201 opt_description,
202 opt_autoplay) {
203 goog.ui.media.MediaModel.call(
204 this,
205 goog.ui.media.PicasaAlbumModel.buildUrl(userId, albumId),
206 opt_caption,
207 opt_description,
208 goog.ui.media.MediaModel.MimeType.FLASH);
209
210 /**
211 * The Picasa user id.
212 * @type {string}
213 * @private
214 */
215 this.userId_ = userId;
216
217 /**
218 * The Picasa album id.
219 * @type {string}
220 * @private
221 */
222 this.albumId_ = albumId;
223
224 /**
225 * The Picasa authentication key, used on private albums.
226 * @type {?string}
227 * @private
228 */
229 this.authKey_ = opt_authKey || null;
230
231 var authParam = opt_authKey ? ('&authkey=' + opt_authKey) : '';
232
233 var flashVars = {
234 'host': 'picasaweb.google.com',
235 'RGB': '0x000000',
236 'feed': 'http://picasaweb.google.com/data/feed/api/user/' +
237 userId + '/album/' + albumId + '?kind=photo&alt=rss' + authParam
238 };
239 flashVars[opt_autoplay ? 'autoplay' : 'noautoplay'] = '1';
240
241 var flashUrl = goog.html.TrustedResourceUrl.fromConstant(
242 goog.string.Const.from(
243 'http://picasaweb.google.com/s/c/bin/slideshow.swf'));
244 var player = new goog.ui.media.MediaModel.Player(flashUrl, flashVars);
245
246 this.setPlayer(player);
247};
248goog.inherits(goog.ui.media.PicasaAlbumModel, goog.ui.media.MediaModel);

251/**
252 * Regular expression used to extract the picasa username and albumid out of
253 * picasa URLs.
254 *
255 * Copied from http://go/markdownlite.js,
256 * and {@link PicasaWebExtractor.xml}.
257 *
258 * @type {RegExp}
259 * @private
260 * @const
261 */
262goog.ui.media.PicasaAlbumModel.MATCHER_ =
263 /https?:\/\/(?:www\.)?picasaweb\.(?:google\.)?com\/([\d\w\.]+)\/([\d\w_\-\.]+)(?:\?[\w\d\-_=&amp;;\.]*&?authKey=([\w\d\-_=;\.]+))?(?:#([\d]+)?)?/im;

266/**
267 * Gets a {@code picasaUrl} and extracts the user and album id.
268 *
269 * @param {string} picasaUrl A picasa album URL.
270 * @param {string=} opt_caption An optional caption of the picasa album.
271 * @param {string=} opt_description An optional description of the picasa album.
272 * @param {boolean=} opt_autoplay Whether to autoplay the slideshow.
273 * @return {!goog.ui.media.PicasaAlbumModel} The picasa album data model that
274 * represents the picasa URL.
275 * @throws exception in case the parsing fails
276 */
277goog.ui.media.PicasaAlbumModel.newInstance = function(picasaUrl,
278 opt_caption,
279 opt_description,
280 opt_autoplay) {
281 if (goog.ui.media.PicasaAlbumModel.MATCHER_.test(picasaUrl)) {
282 var data = goog.ui.media.PicasaAlbumModel.MATCHER_.exec(picasaUrl);
283 return new goog.ui.media.PicasaAlbumModel(
284 data[1], data[2], data[3], opt_caption, opt_description, opt_autoplay);
285 }
286 throw Error('failed to parse user and album from picasa url: ' + picasaUrl);
287};

290/**
291 * The opposite of {@code newInstance}: takes an {@code userId} and an
292 * {@code albumId} and builds a URL.
293 *
294 * @param {string} userId The user that owns the album.
295 * @param {string} albumId The album id.
296 * @return {string} The URL of the album.
297 */
298goog.ui.media.PicasaAlbumModel.buildUrl = function(userId, albumId) {
299 return 'http://picasaweb.google.com/' + userId + '/' + albumId;
300};
