15/**
16 * @fileoverview provides a reusable youtube UI component given a youtube data
17 * model.
18 *
19 * goog.ui.media.Youtube is actually a {@link goog.ui.ControlRenderer}, a
20 * stateless class - that could/should be used as a Singleton with the static
21 * method {@code goog.ui.media.Youtube.getInstance} -, that knows how to render
22 * youtube videos. It is designed to be used with a {@link goog.ui.Control},
23 * which will actually control the media renderer and provide the
24 * {@link goog.ui.Component} base. This design guarantees that all different
25 * types of medias will behave alike but will look different.
26 *
27 * goog.ui.media.Youtube expects {@code goog.ui.media.YoutubeModel} on
28 * {@code goog.ui.Control.getModel} as data models, and render a flash object
29 * that will play that URL.
30 *
31 * Example of usage:
32 *
33 * <pre>
34 * var video = goog.ui.media.YoutubeModel.newInstance(
35 * 'http://www.youtube.com/watch?v=ddl5f44spwQ');
36 * goog.ui.media.Youtube.newControl(video).render();
37 * </pre>
38 *
39 * youtube medias currently support the following states:
40 *
41 * <ul>
42 * <li> {@link goog.ui.Component.State.DISABLED}: shows 'flash not available'
43 * <li> {@link goog.ui.Component.State.HOVER}: mouse cursor is over the video
44 * <li> {@link !goog.ui.Component.State.SELECTED}: a static thumbnail is shown
45 * <li> {@link goog.ui.Component.State.SELECTED}: video is playing
46 * </ul>
47 *
48 * Which can be accessed by
49 * <pre>
50 * youtube.setEnabled(true);
51 * youtube.setHighlighted(true);
52 * youtube.setSelected(true);
53 * </pre>
54 *
55 * This package also provides a few static auxiliary methods, such as:
56 *
57 * <pre>
58 * var videoId = goog.ui.media.Youtube.parseUrl(
59 * 'http://www.youtube.com/watch?v=ddl5f44spwQ');
60 * </pre>
61 *
62 *
63 * @supported IE6, FF2+, Safari. Requires flash to actually work.
64 *
65 * TODO(user): test on other browsers
66 */

82/**
83 * Subclasses a goog.ui.media.MediaRenderer to provide a Youtube specific media
84 * renderer.
85 *
86 * This class knows how to parse youtube urls, and render the DOM structure
87 * of youtube video players and previews. This class is meant to be used as a
88 * singleton static stateless class, that takes {@code goog.ui.media.Media}
89 * instances and renders it. It expects {@code goog.ui.media.Media.getModel} to
90 * return a well formed, previously constructed, youtube video id, which is the
91 * data model this renderer will use to construct the DOM structure.
92 * {@see goog.ui.media.Youtube.newControl} for a example of constructing a
93 * control with this renderer.
94 *
95 * goog.ui.media.Youtube currently supports all {@link goog.ui.Component.State}.
96 * It will change its DOM structure between SELECTED and !SELECTED, and rely on
97 * CSS definitions on the others. On !SELECTED, the renderer will render a
98 * youtube static <img>, with a thumbnail of the video. On SELECTED, the
99 * renderer will append to the DOM a flash object, that contains the youtube
100 * video.
101 *
102 * This design is patterned after http://go/closure_control_subclassing
103 *
104 * It uses {@link goog.ui.media.FlashObject} to embed the flash object.
105 *
106 * @constructor
107 * @extends {goog.ui.media.MediaRenderer}
108 * @final
109 */
110goog.ui.media.Youtube = function() {
111 goog.ui.media.MediaRenderer.call(this);
112};
113goog.inherits(goog.ui.media.Youtube, goog.ui.media.MediaRenderer);
114goog.addSingletonGetter(goog.ui.media.Youtube);

117/**
118 * A static convenient method to construct a goog.ui.media.Media control out of
119 * a youtube model. It sets it as the data model goog.ui.media.Youtube renderer
120 * uses, sets the states supported by the renderer, and returns a Control that
121 * binds everything together. This is what you should be using for constructing
122 * Youtube videos, except if you need finer control over the configuration.
123 *
124 * @param {goog.ui.media.YoutubeModel} youtubeModel The youtube data model.
125 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper, used for
126 * document interaction.
127 * @return {!goog.ui.media.Media} A Control binded to the youtube renderer.
128 */
129goog.ui.media.Youtube.newControl = function(youtubeModel, opt_domHelper) {
130 var control = new goog.ui.media.Media(
131 youtubeModel,
132 goog.ui.media.Youtube.getInstance(),
133 opt_domHelper);
134 control.setStateInternal(goog.ui.Component.State.ACTIVE);
135 return control;
136};

147/**
148 * Changes the state of a {@code control}. Currently only changes the DOM
149 * structure when the youtube movie is SELECTED (by default fired by a MOUSEUP
150 * on the thumbnail), which means we have to embed the youtube flash video and
151 * play it.
152 *
153 * @param {goog.ui.Control} c The media control.
154 * @param {goog.ui.Component.State} state The state to be set or cleared.
155 * @param {boolean} enable Whether the state is enabled or disabled.
156 * @override
157 */
158goog.ui.media.Youtube.prototype.setState = function(c, state, enable) {
159 var control = /** @type {goog.ui.media.Media} */ (c);
160 goog.ui.media.Youtube.superClass_.setState.call(this, control, state, enable);
161
162 // control.createDom has to be called before any state is set.
163 // Use control.setStateInternal if you need to set states
164 if (!control.getElement()) {
165 throw Error(goog.ui.Component.Error.STATE_INVALID);
166 }
167
168 var domHelper = control.getDomHelper();
169 var dataModel =
170 /** @type {goog.ui.media.YoutubeModel} */ (control.getDataModel());
171
172 if (!!(state & goog.ui.Component.State.SELECTED) && enable) {
173 var flashEls = domHelper.getElementsByTagNameAndClass(
174 goog.dom.TagName.DIV,
175 goog.ui.media.FlashObject.CSS_CLASS,
176 control.getElement());
177 if (flashEls.length > 0) {
178 return;
179 }
180 var youtubeFlash = new goog.ui.media.FlashObject(
181 dataModel.getPlayer().getTrustedResourceUrl(),
182 domHelper);
183 control.addChild(youtubeFlash, true);
184 }
185};

201/**
202 * The {@code goog.ui.media.Youtube} media data model. It stores a required
203 * {@code videoId} field, sets the youtube URL, and allows a few optional
204 * parameters.
205 *
206 * @param {string} videoId The youtube video id.
207 * @param {string=} opt_caption An optional caption of the youtube video.
208 * @param {string=} opt_description An optional description of the youtube
209 * video.
210 * @constructor
211 * @extends {goog.ui.media.MediaModel}
212 * @final
213 */
214goog.ui.media.YoutubeModel = function(videoId, opt_caption, opt_description) {
215 goog.ui.media.MediaModel.call(
216 this,
217 goog.ui.media.YoutubeModel.buildUrl(videoId),
218 opt_caption,
219 opt_description,
220 goog.ui.media.MediaModel.MimeType.FLASH);
221
222 /**
223 * The Youtube video id.
224 * @type {string}
225 * @private
226 */
227 this.videoId_ = videoId;
228
229 this.setThumbnails([new goog.ui.media.MediaModel.Thumbnail(
230 goog.ui.media.YoutubeModel.getThumbnailUrl(videoId))]);
231
232 this.setPlayer(new goog.ui.media.MediaModel.Player(
233 goog.ui.media.YoutubeModel.getFlashUrl(videoId, true)));
234};
235goog.inherits(goog.ui.media.YoutubeModel, goog.ui.media.MediaModel);

238/**
239 * A youtube regular expression matcher. It matches the VIDEOID of URLs like
240 * http://www.youtube.com/watch?v=VIDEOID. Based on:
241 * googledata/contentonebox/opencob/specs/common/YTPublicExtractorCard.xml
242 * @type {RegExp}
243 * @private
244 * @const
245 */
246// Be careful about the placement of the dashes in the character classes. Eg,
247// use "[\\w=-]" instead of "[\\w-=]" if you mean to include the dash as a
248// character and not create a character range like "[a-f]".
249goog.ui.media.YoutubeModel.MATCHER_ = new RegExp(
250 // Lead in.
251 'https?://(?:[a-zA-Z]{1,3}\\.)?' +
252 // Watch URL prefix. This should handle new URLs of the form:
253 // http://www.youtube.com/watch#!v=jqxENMKaeCU&feature=related
254 // where the parameters appear after "#!" instead of "?".
255 '(?:youtube\\.com/watch|youtu\\.be/watch)' +
256 // Get the video id:
257 // The video ID is a parameter v=[videoid] either right after the "?"
258 // or after some other parameters.
259 '(?:\\?(?:[\\w=-]+&(?:amp;)?)*v=([\\w-]+)' +
260 '(?:&(?:amp;)?[\\w=-]+)*)?' +
261 // Get any extra arguments in the URL's hash part.
262 '(?:#[!]?(?:' +
263 // Video ID from the v=[videoid] parameter, optionally surrounded by other
264 // & separated parameters.
265 '(?:(?:[\\w=-]+&(?:amp;)?)*(?:v=([\\w-]+))' +
266 '(?:&(?:amp;)?[\\w=-]+)*)' +
267 '|' +
268 // Continue supporting "?" for the video ID
269 // and "#" for other hash parameters.
270 '(?:[\\w=&-]+)' +
271 '))?' +
272 // Should terminate with a non-word, non-dash (-) character.
273 '[^\\w-]?', 'i');

276/**
277 * A auxiliary static method that parses a youtube URL, extracting the ID of the
278 * video, and builds a YoutubeModel.
279 *
280 * @param {string} youtubeUrl A youtube URL.
281 * @param {string=} opt_caption An optional caption of the youtube video.
282 * @param {string=} opt_description An optional description of the youtube
283 * video.
284 * @return {!goog.ui.media.YoutubeModel} The data model that represents the
285 * youtube URL.
286 * @see goog.ui.media.YoutubeModel.getVideoId()
287 * @throws Error in case the parsing fails.
288 */
289goog.ui.media.YoutubeModel.newInstance = function(youtubeUrl,
290 opt_caption,
291 opt_description) {
292 var extract = goog.ui.media.YoutubeModel.MATCHER_.exec(youtubeUrl);
293 if (extract) {
294 var videoId = extract[1] || extract[2];
295 return new goog.ui.media.YoutubeModel(
296 videoId, opt_caption, opt_description);
297 }
298
299 throw Error('failed to parse video id from youtube url: ' + youtubeUrl);
300};

303/**
304 * The opposite of {@code goog.ui.media.Youtube.newInstance}: it takes a videoId
305 * and returns a youtube URL.
306 *
307 * @param {string} videoId The youtube video ID.
308 * @return {string} The youtube URL.
309 */
310goog.ui.media.YoutubeModel.buildUrl = function(videoId) {
311 return 'http://www.youtube.com/watch?v=' + goog.string.urlEncode(videoId);

333/**
334 * A static auxiliary method that builds URL of the flash movie to be embedded,
335 * out of the youtube video id.
336 *
337 * @param {string} videoId The youtube video ID.
338 * @param {boolean=} opt_autoplay Whether the flash movie should start playing
339 * as soon as it is shown, or if it should show a 'play' button.
340 * @return {string} The flash URL to be embedded on the page.
341 */
342goog.ui.media.YoutubeModel.getFlashUrl = function(videoId, opt_autoplay) {
343 var autoplay = opt_autoplay ? '&autoplay=1' : '';
344 // YouTube video ids are extracted from youtube URLs, which are user
345 // generated input. the video id is later used to embed a flash object,
346 // which is generated through HTML construction. We goog.string.urlEncode
347 // the video id to make sure the URL is safe to be embedded.
348 return 'http://www.youtube.com/v/' + goog.string.urlEncode(videoId) +
349 '&hl=en&fs=1' + autoplay;
350};
