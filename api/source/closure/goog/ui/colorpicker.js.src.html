15/**
16 * @fileoverview A color picker component. A color picker can compose several
17 * instances of goog.ui.ColorPalette.
18 *
19 * NOTE: The ColorPicker is in a state of transition towards the common
20 * component/control/container interface we are developing. If the API changes
21 * we will do our best to update your code. The end result will be that a
22 * color picker will compose multiple color palettes. In the simple case this
23 * will be one grid, but may consistute 3 distinct grids, a custom color picker
24 * or even a color wheel.
25 *
26 */

36/**
37 * Create a new, empty color picker.
38 *
39 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper.
40 * @param {goog.ui.ColorPalette=} opt_colorPalette Optional color palette to
41 * use for this color picker.
42 * @extends {goog.ui.Component}
43 * @constructor
44 * @final
45 */
46goog.ui.ColorPicker = function(opt_domHelper, opt_colorPalette) {
47 goog.ui.Component.call(this, opt_domHelper);
48
49 /**
50 * The color palette used inside the color picker.
51 * @type {goog.ui.ColorPalette?}
52 * @private
53 */
54 this.colorPalette_ = opt_colorPalette || null;
55
56 this.getHandler().listen(
57 this, goog.ui.Component.EventType.ACTION, this.onColorPaletteAction_);
58};

286/**
287 * Create a color palette for the color picker.
288 * @param {Array<string>} colors Array of colors.
289 * @private
290 */
291goog.ui.ColorPicker.prototype.createColorPalette_ = function(colors) {
292 // TODO(user): The color picker should eventually just contain a number of
293 // palettes and manage the interactions between them. This will go away then.
294 var cp = new goog.ui.ColorPalette(colors, null, this.getDomHelper());
295 cp.setSize(goog.ui.ColorPicker.DEFAULT_NUM_COLS);
296 cp.setSupportedState(goog.ui.Component.State.FOCUSED, this.focusable_);
297 // TODO(user): Use addChild(cp, true) and remove calls to render.
298 this.addChild(cp);
299 this.colorPalette_ = cp;
300 if (this.isInDocument()) {
301 this.colorPalette_.render(this.getElement());
302 }
303};
