35/**

37 * overrides {@link goog.ui.style.app.ButtonRenderer#createButton} to insert a
38 * dropdown element into the content element after the specified content.

40 * @extends {goog.ui.style.app.ButtonRenderer}

91/**
92 * Returns the ARIA role to be applied to menu buttons, which
93 * have a menu attached to them.
94 * @return {goog.a11y.aria.Role} ARIA role.
95 * @override
96 */
97goog.ui.style.app.MenuButtonRenderer.prototype.getAriaRole = function() {
98 // If we apply the 'button' ARIA role to the menu button, the
99 // screen reader keeps referring to menus as buttons, which
100 // might be misleading for the users. Hence the ARIA role
101 // 'menu' is assigned.
102 return goog.a11y.aria.Role.MENU;
103};

122/**
123 * Takes an element, decorates it with the menu button control, and returns
124 * the element. Overrides {@link goog.ui.style.app.ButtonRenderer#decorate} by
125 * looking for a child element that can be decorated by a menu, and if it
126 * finds one, decorates it and attaches it to the menu button.
127 * @param {goog.ui.Control} control goog.ui.MenuButton to decorate the element.
128 * @param {Element} element Element to decorate.
129 * @return {Element} Decorated element.
130 * @override
131 */
132goog.ui.style.app.MenuButtonRenderer.prototype.decorate =
133 function(control, element) {
134 var button = /** @type {goog.ui.MenuButton} */ (control);
135 // TODO(attila): Add more robust support for subclasses of goog.ui.Menu.
136 var menuElem = goog.dom.getElementsByTagNameAndClass(
137 '*', goog.ui.MenuRenderer.CSS_CLASS, element)[0];
138 if (menuElem) {
139 // Move the menu element directly under the body (but hide it first to
140 // prevent flicker; see bug 1089244).
141 goog.style.setElementShown(menuElem, false);
142 goog.dom.appendChild(goog.dom.getOwnerDocument(menuElem).body, menuElem);
143
144 // Decorate the menu and attach it to the button.
145 var menu = new goog.ui.Menu();
146 menu.decorate(menuElem);
147 button.setMenu(menu);
148 }
149
150 // Let the superclass do the rest.
151 return goog.ui.style.app.MenuButtonRenderer.superClass_.decorate.call(this,
152 button, element);
153};

156/**
157 * Takes a text caption or existing DOM structure, and returns the content and
158 * a dropdown arrow element wrapped in a pseudo-rounded-corner box. Creates
159 * the following DOM structure:
160 * <div class="goog-inline-block goog-button-outer-box">
161 * <div class="goog-inline-block goog-button-inner-box">
162 * <div class="goog-button-pos">
163 * <div class="goog-button-top-shadow">&nbsp;</div>
164 * <div class="goog-button-content">
165 * Contents...
166 * <div class="goog-menu-button-dropdown"> </div>
167 * </div>
168 * </div>
169 * </div>
170 * </div>
171 * @param {goog.ui.ControlContent} content Text caption or DOM structure to wrap
172 * in a box.
173 * @param {goog.dom.DomHelper} dom DOM helper, used for document interaction.
174 * @return {Element} Pseudo-rounded-corner box containing the content.
175 * @override
176 */
177goog.ui.style.app.MenuButtonRenderer.prototype.createButton = function(content,
178 dom) {
179 var contentWithDropdown = this.createContentWithDropdown(content, dom);
180 return goog.ui.style.app.MenuButtonRenderer.superClass_.createButton.call(
181 this, contentWithDropdown, dom);
182};

194/**
195 * Inserts dropdown element as last child of existing content.
196 * @param {goog.ui.ControlContent} content Text caption or DOM structure.
197 * @param {goog.dom.DomHelper} dom DOM helper, used for document ineraction.
198 * @return {Array<Node>} DOM structure to be set as the button's content.
199 */
200goog.ui.style.app.MenuButtonRenderer.prototype.createContentWithDropdown =
201 function(content, dom) {
202 var caption = dom.createDom(goog.dom.TagName.DIV, null, content,
203 this.createDropdown(dom));
204 return goog.array.toArray(caption.childNodes);
205};

208/**
209 * Returns an appropriately-styled DIV containing a dropdown arrow.
210 * Creates the following DOM structure:
211 * <div class="goog-menu-button-dropdown"> </div>
212 * @param {goog.dom.DomHelper} dom DOM helper, used for document interaction.
213 * @return {Element} Dropdown element.
214 */
215goog.ui.style.app.MenuButtonRenderer.prototype.createDropdown = function(dom) {
216 return dom.createDom(goog.dom.TagName.DIV,
217 goog.getCssName(this.getCssClass(), 'dropdown'));
218};
