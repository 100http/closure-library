35/**
36 * A table sorter allows for sorting of a table by column. This component can
37 * be used to decorate an already existing TABLE element with sorting
38 * features.
39 *
40 * The TABLE should use a THEAD containing TH elements for the table column
41 * headers.
42 *
43 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper, used for
44 * document interaction.
45 * @constructor
46 * @extends {goog.ui.Component}
47 */
48goog.ui.TableSorter = function(opt_domHelper) {
49 goog.ui.Component.call(this, opt_domHelper);

51 /**
52 * The current sort header of the table, or null if none.
53 * @type {HTMLTableCellElement}
54 * @private
55 */
56 this.header_ = null;
57
58 /**
59 * Whether the last sort was in reverse.
60 * @type {boolean}
61 * @private
62 */
63 this.reversed_ = false;

65 /**
66 * The default sorting function.
67 * @type {function(*, *) : number}
68 * @private

189/**
190 * Sort the table contents by the values in the given column.
191 * @param {goog.events.BrowserEvent} e The click event.
192 * @private
193 */
194goog.ui.TableSorter.prototype.sort_ = function(e) {
195 // Determine what column was clicked.
196 // TODO(robbyw): If this table cell contains another table, this could break.
197 var target = /** @type {Node} */ (e.target);
198 var th = goog.dom.getAncestorByTagNameAndClass(target,
199 goog.dom.TagName.TH);
200
201 // If the user clicks on the same column, sort it in reverse of what it is
202 // now. Otherwise, sort forward.
203 var reverse = th == this.header_ ? !this.reversed_ : false;
204
205 // Perform the sort.
206 if (this.dispatchEvent(goog.ui.TableSorter.EventType.BEFORESORT)) {
207 if (this.sort(th.cellIndex, reverse)) {
208 this.dispatchEvent(goog.ui.TableSorter.EventType.SORT);
209 }
210 }
211};

214/**
215 * Sort the table contents by the values in the given column.
216 * @param {number} column The column to sort by.
217 * @param {boolean=} opt_reverse Whether to sort in reverse.
218 * @return {boolean} Whether the sort was executed.
219 */
220goog.ui.TableSorter.prototype.sort = function(column, opt_reverse) {
221 var sortFunction = this.getSortFunction(column);
222 if (sortFunction === goog.ui.TableSorter.noSort) {
223 return false;
224 }
225
226 // Remove old header classes.
227 if (this.header_) {
228 goog.dom.classlist.remove(this.header_, this.reversed_ ?
229 goog.getCssName('goog-tablesorter-sorted-reverse') :
230 goog.getCssName('goog-tablesorter-sorted'));
231 }
232
233 // If the user clicks on the same column, sort it in reverse of what it is
234 // now. Otherwise, sort forward.
235 this.reversed_ = !!opt_reverse;
236 var multiplier = this.reversed_ ? -1 : 1;
237 var cmpFn = function(a, b) {
238 return multiplier * sortFunction(a[0], b[0]) || a[1] - b[1];
239 };
240
241 // Sort all tBodies
242 var table = this.getElement();
243 goog.array.forEach(table.tBodies, function(tBody) {
244 // Collect all of the rows into an array.
245 var values = goog.array.map(tBody.rows, function(row, rowIndex) {
246 return [goog.dom.getTextContent(row.cells[column]), rowIndex, row];
247 });
248
249 goog.array.sort(values, cmpFn);
250
251 // Remove the tBody temporarily since this speeds up the sort on some
252 // browsers.
253 var nextSibling = tBody.nextSibling;
254 table.removeChild(tBody);
255
256 // Sort the rows, using the resulting array.
257 goog.array.forEach(values, function(row) {
258 tBody.appendChild(row[2]);
259 });
260
261 // Reinstate the tBody.
262 table.insertBefore(tBody, nextSibling);
263 });
264
265 // Mark this as the last sorted column.
266 this.header_ = table.tHead.rows[this.sortableHeaderRowIndex_].cells[column];
267
268 // Update the header class.
269 goog.dom.classlist.add(this.header_, this.reversed_ ?
270 goog.getCssName('goog-tablesorter-sorted-reverse') :
271 goog.getCssName('goog-tablesorter-sorted'));
272
273 return true;
274};

286/**
287 * A numeric sort function. NaN values (or values that do not parse as float
288 * numbers) compare equal to each other and greater to any other number.
289 * @param {*} a First sort value.
290 * @param {*} b Second sort value.
291 * @return {number} Negative if a < b, 0 if a = b, and positive if a > b.
292 */
293goog.ui.TableSorter.numericSort = function(a, b) {
294 a = parseFloat(a);
295 b = parseFloat(b);
296 // foo == foo is false if and only if foo is NaN.
297 if (a == a) {
298 return b == b ? a - b : -1;
299 } else {
300 return b == b ? 1 : 0;
301 }
302};
