41/**
42 * Component for handling keyboard shortcuts. A shortcut is registered and bound
43 * to a specific identifier. Once the shortcut is triggered an event is fired
44 * with the identifier for the shortcut. This allows keyboard shortcuts to be
45 * customized without modifying the code that listens for them.
46 *
47 * Supports keyboard shortcuts triggered by a single key, a stroke stroke (key
48 * plus at least one modifier) and a sequence of keys or strokes.
49 *
50 * @param {goog.events.EventTarget|EventTarget} keyTarget Event target that the
51 * key event listener is attached to, typically the applications root
52 * container.
53 * @constructor
54 * @extends {goog.events.EventTarget}
55 */
56goog.ui.KeyboardShortcutHandler = function(keyTarget) {
57 goog.events.EventTarget.call(this);
58
59 /**
60 * Registered keyboard shortcuts tree. Stored as a map with the keyCode and
61 * modifier(s) as the key and either a list of further strokes or the shortcut
62 * task identifier as the value.
63 * @type {!goog.ui.KeyboardShortcutHandler.SequenceTree_}
64 * @see #makeStroke_
65 * @private
66 */
67 this.shortcuts_ = {};
68
69 /**
70 * The currently active shortcut sequence tree, which represents the position
71 * in the complete shortcuts_ tree reached by recent key strokes.
72 * @type {!goog.ui.KeyboardShortcutHandler.SequenceTree_}
73 * @private
74 */
75 this.currentTree_ = this.shortcuts_;
76
77 /**
78 * The time (in ms, epoch time) of the last keystroke which made progress in
79 * the shortcut sequence tree (i.e. the time that currentTree_ was last set).
80 * Used for timing out stroke sequences.
81 * @type {number}
82 * @private
83 */
84 this.lastStrokeTime_ = 0;
85
86 /**
87 * List of numeric key codes for keys that are safe to always regarded as
88 * shortcuts, even if entered in a textarea or input field.
89 * @type {Object}
90 * @private
91 */
92 this.globalKeys_ = goog.object.createSet(
93 goog.ui.KeyboardShortcutHandler.DEFAULT_GLOBAL_KEYS_);
94
95 /**
96 * List of input types that should only accept ENTER as a shortcut.
97 * @type {Object}
98 * @private
99 */
100 this.textInputs_ = goog.object.createSet(
101 goog.ui.KeyboardShortcutHandler.DEFAULT_TEXT_INPUTS_);
102
103 /**
104 * Whether to always prevent the default action if a shortcut event is fired.
105 * @type {boolean}
106 * @private
107 */
108 this.alwaysPreventDefault_ = true;
109
110 /**
111 * Whether to always stop propagation if a shortcut event is fired.
112 * @type {boolean}
113 * @private
114 */
115 this.alwaysStopPropagation_ = false;
116
117 /**
118 * Whether to treat all shortcuts as if they had been passed
119 * to setGlobalKeys().
120 * @type {boolean}
121 * @private
122 */
123 this.allShortcutsAreGlobal_ = false;
124
125 /**
126 * Whether to treat shortcuts with modifiers as if they had been passed
127 * to setGlobalKeys(). Ignored if allShortcutsAreGlobal_ is true. Applies
128 * only to form elements (not content-editable).
129 * @type {boolean}
130 * @private
131 */
132 this.modifierShortcutsAreGlobal_ = true;
133
134 /**
135 * Whether to treat space key as a shortcut when the focused element is a
136 * checkbox, radiobutton or button.
137 * @type {boolean}
138 * @private
139 */
140 this.allowSpaceKeyOnButtons_ = false;
141
142 /**
143 * Tracks the currently pressed shortcut key, for Firefox.
144 * @type {?number}
145 * @private
146 */
147 this.activeShortcutKeyForGecko_ = null;
148
149 this.initializeKeyListener(keyTarget);
150};
151goog.inherits(goog.ui.KeyboardShortcutHandler, goog.events.EventTarget);
152goog.tagUnsealableClass(goog.ui.KeyboardShortcutHandler);

157 * A node in a keyboard shortcut sequence tree. A node is either:
158 * 1. A terminal node with a non-nullable shortcut string which is the
159 * identifier for the shortcut triggered by traversing the tree to that node.
160 * 2. An internal node with a null shortcut string and a
161 * {@code goog.ui.KeyboardShortcutHandler.SequenceTree_} representing the
162 * continued stroke sequences from this node.

167 * @struct
168 * @private

171 /** @const {?string} The shorcut action identifier, for terminal nodes. */
172 this.shortcut = opt_shortcut || null;
173
174 /** @const {goog.ui.KeyboardShortcutHandler.SequenceTree_} */
175 this.next = opt_shortcut ? null : {};
176};

331/**
332 * Static method for getting the key code for a given key.
333 * @param {string} name Name of key.
334 * @return {number} The key code.
335 */
336goog.ui.KeyboardShortcutHandler.getKeyCode = function(name) {
337 // Build reverse lookup object the first time this method is called.
338 if (!goog.ui.KeyboardShortcutHandler.nameToKeyCodeCache_) {
339 var map = {};
340 for (var key in goog.events.KeyNames) {
341 // Explicitly convert the stringified map keys to numbers and normalize.
342 map[goog.events.KeyNames[key]] =
343 goog.events.KeyCodes.normalizeKeyCode(parseInt(key, 10));
344 }
345 goog.ui.KeyboardShortcutHandler.nameToKeyCodeCache_ = map;
346 }
347
348 // Check if key is in cache.
349 return goog.ui.KeyboardShortcutHandler.nameToKeyCodeCache_[name];
350};

468/**
469 * Registers a keyboard shortcut.
470 * @param {string} identifier Identifier for the task performed by the keyboard
471 * combination. Multiple shortcuts can be provided for the same
472 * task by specifying the same identifier.
473 * @param {...(number|string|Array<number>)} var_args See below.
474 *
475 * param {number} keyCode Numeric code for key
476 * param {number=} opt_modifiers Bitmap indicating required modifier keys.
477 * goog.ui.KeyboardShortcutHandler.Modifiers.SHIFT, CONTROL,
478 * ALT, or META.
479 *
480 * The last two parameters can be repeated any number of times to create a
481 * shortcut using a sequence of strokes. Instead of varagrs the second parameter
482 * could also be an array where each element would be ragarded as a parameter.
483 *
484 * A string representation of the shortcut can be supplied instead of the last
485 * two parameters. In that case the method only takes two arguments, the
486 * identifier and the string.
487 *
488 * Examples:
489 * g registerShortcut(str, G_KEYCODE)
490 * Ctrl+g registerShortcut(str, G_KEYCODE, CTRL)
491 * Ctrl+Shift+g registerShortcut(str, G_KEYCODE, CTRL | SHIFT)
492 * Ctrl+g a registerShortcut(str, G_KEYCODE, CTRL, A_KEYCODE)
493 * Ctrl+g Shift+a registerShortcut(str, G_KEYCODE, CTRL, A_KEYCODE, SHIFT)
494 * g a registerShortcut(str, G_KEYCODE, NONE, A_KEYCODE)
495 *
496 * Examples using string representation for shortcuts:
497 * g registerShortcut(str, 'g')
498 * Ctrl+g registerShortcut(str, 'ctrl+g')
499 * Ctrl+Shift+g registerShortcut(str, 'ctrl+shift+g')
500 * Ctrl+g a registerShortcut(str, 'ctrl+g a')
501 * Ctrl+g Shift+a registerShortcut(str, 'ctrl+g shift+a')
502 * g a registerShortcut(str, 'g a').
503 */
504goog.ui.KeyboardShortcutHandler.prototype.registerShortcut = function(
505 identifier, var_args) {
506
507 // Add shortcut to shortcuts_ tree
508 goog.ui.KeyboardShortcutHandler.setShortcut_(
509 this.shortcuts_, this.interpretStrokes_(1, arguments), identifier);
510};

566/**
567 * Parses the variable arguments for registerShortcut and unregisterShortcut.
568 * @param {number} initialIndex The first index of "args" to treat as
569 * variable arguments.
570 * @param {Object} args The "arguments" array passed
571 * to registerShortcut or unregisterShortcut. Please see the comments in
572 * registerShortcut for list of allowed forms.
573 * @return {!Array<number>} The sequence of strokes, represented as numbers.
574 * @private
575 */
576goog.ui.KeyboardShortcutHandler.prototype.interpretStrokes_ = function(
577 initialIndex, args) {
578 var strokes;
579
580 // Build strokes array from string.
581 if (goog.isString(args[initialIndex])) {
582 strokes = goog.array.map(
583 goog.ui.KeyboardShortcutHandler.parseStringShortcut(args[initialIndex]),
584 function(stroke) {
585 goog.asserts.assertNumber(
586 stroke.keyCode, 'A non-modifier key is needed in each stroke.');
587 return goog.ui.KeyboardShortcutHandler.makeStroke_(
588 stroke.keyCode, stroke.modifiers);
589 });
590
591 // Build strokes array from arguments list or from array.
592 } else {
593 var strokesArgs = args, i = initialIndex;
594 if (goog.isArray(args[initialIndex])) {
595 strokesArgs = args[initialIndex];
596 i = 0;
597 }
598
599 strokes = [];
600 for (; i < strokesArgs.length; i += 2) {
601 strokes.push(goog.ui.KeyboardShortcutHandler.makeStroke_(
602 strokesArgs[i], strokesArgs[i + 1]));
603 }
604 }
605
606 return strokes;
607};

657/**
658 * Builds stroke array from string representation of shortcut.
659 * @param {string} s String representation of shortcut.
660 * @return {!Array<!{keyCode: ?number, modifiers: number}>} The stroke array. A
661 * null keyCode means no non-modifier key was part of the stroke.
662 */
663goog.ui.KeyboardShortcutHandler.parseStringShortcut = function(s) {
664 // Normalize whitespace and force to lower case.
665 s = s.replace(/[ +]*\+[ +]*/g, '+').replace(/[ ]+/g, ' ').toLowerCase();
666
667 // Build strokes array from string, space separates strokes, plus separates
668 // individual keys.
669 var groups = s.split(' ');
670 var strokes = [];
671 for (var group, i = 0; group = groups[i]; i++) {
672 var keys = group.split('+');
673 // Explicitly re-initialize key data (JS does not have block scoping).
674 var keyCode = null;
675 var modifiers = goog.ui.KeyboardShortcutHandler.Modifiers.NONE;
676 for (var key, j = 0; key = keys[j]; j++) {
677 switch (key) {
678 case 'shift':
679 modifiers |= goog.ui.KeyboardShortcutHandler.Modifiers.SHIFT;
680 continue;
681 case 'ctrl':
682 modifiers |= goog.ui.KeyboardShortcutHandler.Modifiers.CTRL;
683 continue;
684 case 'alt':
685 modifiers |= goog.ui.KeyboardShortcutHandler.Modifiers.ALT;
686 continue;
687 case 'meta':
688 modifiers |= goog.ui.KeyboardShortcutHandler.Modifiers.META;
689 continue;
690 }
691 if (!goog.isNull(keyCode)) {
692 goog.asserts.fail('At most one non-modifier key can be in a stroke.');
693 }
694 keyCode = goog.ui.KeyboardShortcutHandler.getKeyCode(key);
695 goog.asserts.assertNumber(
696 keyCode, 'Key name not found in goog.events.KeyNames: ' + key);
697 break;
698 }
699 strokes.push({keyCode: keyCode, modifiers: modifiers});
700 }
701
702 return strokes;
703};

706/**
707 * Adds a key event listener that triggers {@link #handleKeyDown_} when keys
708 * are pressed.
709 * @param {goog.events.EventTarget|EventTarget} keyTarget Event target that the
710 * event listener should be attached to.
711 * @protected
712 */
713goog.ui.KeyboardShortcutHandler.prototype.initializeKeyListener =
714 function(keyTarget) {
715 this.keyTarget_ = keyTarget;
716
717 goog.events.listen(this.keyTarget_, goog.events.EventType.KEYDOWN,
718 this.handleKeyDown_, false, this);
719
720 if (goog.userAgent.GECKO) {
721 goog.events.listen(this.keyTarget_, goog.events.EventType.KEYUP,
722 this.handleGeckoKeyUp_, false, this);
723 }
724
725 // Windows uses ctrl+alt keys (a.k.a. alt-graph keys) for typing characters
726 // on European keyboards (e.g. ctrl+alt+e for an an euro sign.) Unfortunately,
727 // Windows browsers except Firefox does not have any methods except listening
728 // keypress and keyup events to identify if ctrl+alt keys are really used for
729 // inputting characters. Therefore, we listen to these events and prevent

731 if (goog.userAgent.WINDOWS && !goog.userAgent.GECKO) {
732 goog.events.listen(this.keyTarget_, goog.events.EventType.KEYPRESS,
733 this.handleWindowsKeyPress_, false, this);
734 goog.events.listen(this.keyTarget_, goog.events.EventType.KEYUP,
735 this.handleWindowsKeyUp_, false, this);

740/**
741 * Handler for when a keyup event is fired in Firefox (Gecko).
742 * @param {goog.events.BrowserEvent} e The key event.
743 * @private
744 */
745goog.ui.KeyboardShortcutHandler.prototype.handleGeckoKeyUp_ = function(e) {
746 // Due to a bug in the way that Gecko on Mac handles cut/copy/paste key events
747 // using the meta key, it is necessary to fake the keyDown for the action keys
748 // (C,V,X) by capturing it on keyUp.
749 // This is because the keyDown events themselves are not fired by the browser
750 // in this case.
751 // Because users will often release the meta key a slight moment before they
752 // release the action key, we need to store whether the meta key has been
753 // released recently to avoid "flaky" cutting/pasting behavior.
754 if (goog.userAgent.MAC) {
755 if (e.keyCode == goog.events.KeyCodes.MAC_FF_META) {
756 this.metaKeyRecentlyReleased_ = true;
757 goog.Timer.callOnce(function() {
758 this.metaKeyRecentlyReleased_ = false;
759 }, 400, this);
760 return;
761 }
762
763 var metaKey = e.metaKey || this.metaKeyRecentlyReleased_;
764 if ((e.keyCode == goog.events.KeyCodes.C ||
765 e.keyCode == goog.events.KeyCodes.X ||
766 e.keyCode == goog.events.KeyCodes.V) && metaKey) {
767 e.metaKey = metaKey;
768 this.handleKeyDown_(e);
769 }
770 }
771
772 // Firefox triggers buttons on space keyUp instead of keyDown. So if space
773 // keyDown activated a shortcut, do NOT also trigger the focused button.
774 if (goog.events.KeyCodes.SPACE == this.activeShortcutKeyForGecko_ &&
775 goog.events.KeyCodes.SPACE == e.keyCode) {
776 e.preventDefault();
777 }
778 this.activeShortcutKeyForGecko_ = null;
779};

851/**
852 * Adds a shortcut stroke sequence to the given sequence tree. Recursive.
853 * @param {!goog.ui.KeyboardShortcutHandler.SequenceTree_} tree The stroke
854 * sequence tree to add to.
855 * @param {Array<number>} strokes Array of strokes for shortcut.
856 * @param {string} identifier Identifier for the task performed by shortcut.
857 * @private

859goog.ui.KeyboardShortcutHandler.setShortcut_ = function(
860 tree, strokes, identifier) {
861 var stroke = strokes.shift();
862 var node = tree[stroke];
863 if (node && (strokes.length == 0 || node.shortcut)) {
864 // This new shortcut would override an existing shortcut or shortcut prefix
865 // (since the new strokes end at an existing node), or an existing shortcut
866 // would be triggered by the prefix to this new shortcut (since there is
867 // already a terminal node on the path we are trying to create).
868 throw Error('Keyboard shortcut conflicts with existing shortcut');

873 goog.ui.KeyboardShortcutHandler.createInternalNode_());
874 goog.ui.KeyboardShortcutHandler.setShortcut_(
875 goog.asserts.assert(node.next, 'An internal node must have a next map'),
876 strokes, identifier);
877 } else {
878 // Add a terminal node.
879 tree[stroke] =
880 goog.ui.KeyboardShortcutHandler.createTerminalNode_(identifier);
881 }

885/**
886 * Removes a shortcut stroke sequence from the given sequence tree, pruning any
887 * dead branches of the tree. Recursive.
888 * @param {!goog.ui.KeyboardShortcutHandler.SequenceTree_} tree The stroke
889 * sequence tree to remove from.
890 * @param {Array<number>} strokes Array of strokes for shortcut to remove.
891 * @private
892 */
893goog.ui.KeyboardShortcutHandler.unsetShortcut_ = function(tree, strokes) {
894 var stroke = strokes.shift();
895 var node = tree[stroke];
896
897 if (!node) {
898 // The given stroke sequence is not in the tree.
899 return;
900 }
901 if (strokes.length == 0) {
902 // Base case - the end of the stroke sequence.
903 if (!node.shortcut) {
904 // The given stroke sequence does not end at a terminal node.
905 return;
906 }
907 delete tree[stroke];
908 } else {
909 if (!node.next) {
910 // The given stroke sequence is not in the tree.
911 return;
912 }
913 // Recursively remove the rest of the shortcut sequence from the node.next
914 // subtree.
915 goog.ui.KeyboardShortcutHandler.unsetShortcut_(node.next, strokes);
916 if (goog.object.isEmpty(node.next)) {
917 // The node.next subtree is now empty (the last stroke in it was just
918 // removed), so prune this dead branch of the tree.
919 delete tree[stroke];
920 }
921 }
922};

947/**
948 * Constructs key from key code and modifiers.
949 *
950 * The lower 8 bits are used for the key code, the following 3 for modifiers and
951 * the remaining bits are unused.
952 *
953 * @param {number} keyCode Numeric key code.
954 * @param {number} modifiers Required modifiers.
955 * @return {number} The key.
956 * @private
957 */
958goog.ui.KeyboardShortcutHandler.makeStroke_ = function(keyCode, modifiers) {
959 // Make sure key code is just 8 bits and OR it with the modifiers left shifted
960 // 8 bits.
961 return (keyCode & 255) | (modifiers << 8);

965/**
966 * Keypress handler.
967 * @param {goog.events.BrowserEvent} event Keypress event.
968 * @private
969 */
970goog.ui.KeyboardShortcutHandler.prototype.handleKeyDown_ = function(event) {
971 if (!this.isValidShortcut_(event)) {
972 return;
973 }
974 // For possible printable-key events, we cannot identify whether the events
975 // are used for typing characters until we receive respective keyup events.
976 // Therefore, we handle this event when we receive a succeeding keyup event
977 // to verify this event is not used for typing characters.
978 if (event.type == 'keydown' && this.isPossiblePrintableKey_(event)) {
979 this.isPrintableKey_ = false;
980 return;
981 }
982
983 var keyCode = goog.events.KeyCodes.normalizeKeyCode(event.keyCode);
984
985 var modifiers =
986 (event.shiftKey ? goog.ui.KeyboardShortcutHandler.Modifiers.SHIFT : 0) |
987 (event.ctrlKey ? goog.ui.KeyboardShortcutHandler.Modifiers.CTRL : 0) |
988 (event.altKey ? goog.ui.KeyboardShortcutHandler.Modifiers.ALT : 0) |
989 (event.metaKey ? goog.ui.KeyboardShortcutHandler.Modifiers.META : 0);
990 var stroke = goog.ui.KeyboardShortcutHandler.makeStroke_(keyCode, modifiers);
991
992 if (!this.currentTree_[stroke] || this.hasSequenceTimedOut_()) {
993 // Either this stroke does not continue any active sequence, or the
994 // currently active sequence has timed out. Reset shortcut tree progress.
995 this.setCurrentTree_(this.shortcuts_);
996 }
997
998 var node = this.currentTree_[stroke];
999 if (!node) {
1000 // This stroke does not correspond to a shortcut or continued sequence.
1001 return;
1002 }
1003 if (node.next) {
1004 // This stroke does not trigger a shortcut, but entered stroke(s) are a part
1005 // of a sequence. Progress in the sequence tree and record time to allow the
1006 // following stroke(s) to trigger the shortcut.
1007 this.setCurrentTree_(node.next);
1008 // Prevent default action so that the rest of the stroke sequence can be
1009 // completed.
1010 event.preventDefault();
1011 return;
1012 }
1013 // This stroke triggers a shortcut. Any active sequence has been completed, so
1014 // reset the sequence tree.
1015 this.setCurrentTree_(this.shortcuts_);
1016
1017 // Dispatch the triggered keyboard shortcut event. In addition to the generic
1018 // keyboard shortcut event a more specific fine grained one, specific for the
1019 // shortcut identifier, is fired.
1020 if (this.alwaysPreventDefault_) {
1021 event.preventDefault();
1022 }
1023
1024 if (this.alwaysStopPropagation_) {
1025 event.stopPropagation();
1026 }
1027
1028 var shortcut = goog.asserts.assertString(
1029 node.shortcut, 'A terminal node must have a string shortcut identifier.');
1030 // Dispatch SHORTCUT_TRIGGERED event
1031 var target = /** @type {Node} */ (event.target);
1032 var triggerEvent = new goog.ui.KeyboardShortcutEvent(
1033 goog.ui.KeyboardShortcutHandler.EventType.SHORTCUT_TRIGGERED, shortcut,
1034 target);
1035 var retVal = this.dispatchEvent(triggerEvent);
1036
1037 // Dispatch SHORTCUT_PREFIX_<identifier> event
1038 var prefixEvent = new goog.ui.KeyboardShortcutEvent(
1039 goog.ui.KeyboardShortcutHandler.EventType.SHORTCUT_PREFIX + shortcut,
1040 shortcut, target);
1041 retVal &= this.dispatchEvent(prefixEvent);
1042
1043 // The default action is prevented if 'preventDefault' was
1044 // called on either event, or if a listener returned false.
1045 if (!retVal) {
1046 event.preventDefault();
1047 }
1048
1049 // For Firefox, track which shortcut key was pushed.
1050 if (goog.userAgent.GECKO) {
1051 this.activeShortcutKeyForGecko_ = keyCode;
1052 }
1053};

1056/**
1057 * Checks if a given keypress event may be treated as a shortcut.
1058 * @param {goog.events.BrowserEvent} event Keypress event.
1059 * @return {boolean} Whether to attempt to process the event as a shortcut.
1060 * @private
1061 */
1062goog.ui.KeyboardShortcutHandler.prototype.isValidShortcut_ = function(event) {
1063 var keyCode = event.keyCode;
1064
1065 // Ignore Ctrl, Shift and ALT
1066 if (keyCode == goog.events.KeyCodes.SHIFT ||
1067 keyCode == goog.events.KeyCodes.CTRL ||
1068 keyCode == goog.events.KeyCodes.ALT) {
1069 return false;
1070 }
1071 var el = /** @type {Element} */ (event.target);
1072 var isFormElement =
1073 el.tagName == goog.dom.TagName.TEXTAREA ||
1074 el.tagName == goog.dom.TagName.INPUT ||
1075 el.tagName == goog.dom.TagName.BUTTON ||
1076 el.tagName == goog.dom.TagName.SELECT;
1077
1078 var isContentEditable = !isFormElement && (el.isContentEditable ||
1079 (el.ownerDocument && el.ownerDocument.designMode == 'on'));
1080
1081 if (!isFormElement && !isContentEditable) {
1082 return true;
1083 }
1084 // Always allow keys registered as global to be used (typically Esc, the
1085 // F-keys and other keys that are not typically used to manipulate text).
1086 if (this.globalKeys_[keyCode] || this.allShortcutsAreGlobal_) {
1087 return true;
1088 }
1089 if (isContentEditable) {
1090 // For events originating from an element in editing mode we only let
1091 // global key codes through.
1092 return false;
1093 }
1094 // Event target is one of (TEXTAREA, INPUT, BUTTON, SELECT).
1095 // Allow modifier shortcuts, unless we shouldn't.
1096 if (this.modifierShortcutsAreGlobal_ && (
1097 event.altKey || event.ctrlKey || event.metaKey)) {
1098 return true;
1099 }
1100 // Allow ENTER to be used as shortcut for text inputs.
1101 if (el.tagName == goog.dom.TagName.INPUT && this.textInputs_[el.type]) {
1102 return keyCode == goog.events.KeyCodes.ENTER;
1103 }
1104 // Checkboxes, radiobuttons and buttons. Allow all but SPACE as shortcut.
1105 if (el.tagName == goog.dom.TagName.INPUT ||
1106 el.tagName == goog.dom.TagName.BUTTON) {
1107 // TODO(gboyer): If more flexibility is needed, create protected helper
1108 // methods for each case (e.g. button, input, etc).
1109 if (this.allowSpaceKeyOnButtons_) {
1110 return true;
1111 } else {
1112 return keyCode != goog.events.KeyCodes.SPACE;
1113 }
1114 }
1115 // Don't allow any additional shortcut keys for textareas or selects.
1116 return false;
1117};
