59/**
60 * Decorates the given HTML element as a BidiInput. The HTML element can be an
61 * input element with type='text', a textarea element, or any contenteditable.
62 * Overrides {@link goog.ui.Component#decorateInternal}. Considered protected.
63 * @param {Element} element Element to decorate.
64 * @protected
65 * @override

67goog.ui.BidiInput.prototype.decorateInternal = function(element) {
68 goog.ui.BidiInput.superClass_.decorateInternal.call(this, element);
69 this.init_();
70};

73/**
74 * Creates the element for the text input.
75 * @protected
76 * @override
77 */
78goog.ui.BidiInput.prototype.createDom = function() {
79 this.setElementInternal(
80 this.getDomHelper().createDom(goog.dom.TagName.INPUT,
81 {'type': goog.dom.InputType.TEXT}));
82 this.init_();
83};

86/**
87 * Initializes the events and initial text direction.
88 * Called from either decorate or createDom, after the input field has
89 * been created.
90 * @private
91 */
92goog.ui.BidiInput.prototype.init_ = function() {
93 // Set initial direction by current text
94 this.setDirection_();
95
96 // Listen to value change events
97 this.inputHandler_ = new goog.events.InputHandler(this.getElement());
98 goog.events.listen(this.inputHandler_,
99 goog.events.InputHandler.EventType.INPUT,
100 this.setDirection_, false, this);
101};
102
103
104/**
105 * Set the direction of the input element based on the current value. If the
106 * value does not have any strongly directional characters, remove the dir
107 * attribute so that the direction is inherited instead.
108 * This method is called when the user changes the input element value, or
109 * when a program changes the value using
110 * {@link goog.ui.BidiInput#setValue}
111 * @private
112 */
113goog.ui.BidiInput.prototype.setDirection_ = function() {
114 var element = this.getElement();
115 if (element) {
116 var text = this.getValue();
117 goog.i18n.bidi.setElementDirByTextDirectionality(element, text);
118 }
119};
120
121
122/**
123 * Returns the direction of the input element.
124 * @return {?string} Return 'rtl' for right-to-left text,
125 * 'ltr' for left-to-right text, or null if the value itself is not
126 * enough to determine directionality (e.g. an empty value), and the
127 * direction is inherited from a parent element (typically the body
128 * element).
129 */
130goog.ui.BidiInput.prototype.getDirection = function() {
131 var dir = this.getElement().dir;
132 if (dir == '') {
133 dir = null;
134 }
135 return dir;
136};

139/**
140 * Sets the value of the underlying input field, and sets the direction
141 * according to the given value.
142 * @param {string} value The Value to set in the underlying input field.
143 */
144goog.ui.BidiInput.prototype.setValue = function(value) {
145 var element = this.getElement();
146 if (goog.isDefAndNotNull(element.value)) {
147 element.value = value;
148 } else {
149 goog.dom.setTextContent(element, value);
150 }
151 this.setDirection_();
152};
153
154
155/**
156 * Returns the value of the underlying input field.
157 * @return {string} Value of the underlying input field.
158 */
159goog.ui.BidiInput.prototype.getValue = function() {
160 var element = this.getElement();
161 return goog.isDefAndNotNull(element.value) ? element.value :
162 goog.dom.getRawTextContent(element);
163};

166/** @override */
167goog.ui.BidiInput.prototype.disposeInternal = function() {
168 if (this.inputHandler_) {
169 goog.events.removeAll(this.inputHandler_);
170 this.inputHandler_.dispose();
171 this.inputHandler_ = null;
172 goog.ui.BidiInput.superClass_.disposeInternal.call(this);
173 }
