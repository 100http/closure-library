21goog.provide('goog.ui.ToolbarMenuButton');

23goog.require('goog.ui.MenuButton');
24goog.require('goog.ui.ToolbarMenuButtonRenderer');

27
28
29/**
30 * A menu button control for a toolbar.
31 *
32 * @param {goog.ui.ControlContent} content Text caption or existing DOM
33 * structure to display as the button's caption.
34 * @param {goog.ui.Menu=} opt_menu Menu to render under the button when clicked.
35 * @param {goog.ui.ButtonRenderer=} opt_renderer Optional renderer used to
36 * render or decorate the button; defaults to
37 * {@link goog.ui.ToolbarMenuButtonRenderer}.
38 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper, used for
39 * document interaction.
40 * @constructor
41 * @extends {goog.ui.MenuButton}
42 */
43goog.ui.ToolbarMenuButton = function(
44 content, opt_menu, opt_renderer, opt_domHelper) {
45 goog.ui.MenuButton.call(this, content, opt_menu, opt_renderer ||
46 goog.ui.ToolbarMenuButtonRenderer.getInstance(), opt_domHelper);
47};
48goog.inherits(goog.ui.ToolbarMenuButton, goog.ui.MenuButton);
49
50
51// Registers a decorator factory function for toolbar menu buttons.
52goog.ui.registry.setDecoratorByClassName(
53 goog.ui.ToolbarMenuButtonRenderer.CSS_CLASS,
54 function() {
55 return new goog.ui.ToolbarMenuButton(null);
56 });
