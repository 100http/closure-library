39/**
40 * Creates an object that represents a prompt (used in place of javascript's
41 * prompt). The html structure of the prompt is the same as the layout for
42 * dialog.js except for the addition of a text box which is placed inside the
43 * "Content area" and has the default class-name 'modal-dialog-userInput'
44 *
45 * @param {string} promptTitle The title of the prompt.
46 * @param {string|!goog.html.SafeHtml} promptHtml The HTML body of the prompt.
47 * The variable is trusted and it should be already properly escaped.
48 * @param {Function} callback The function to call when the user selects Ok or
49 * Cancel. The function should expect a single argument which represents
50 * what the user entered into the prompt. If the user presses cancel, the
51 * value of the argument will be null.
52 * @param {string=} opt_defaultValue Optional default value that should be in
53 * the text box when the prompt appears.
54 * @param {string=} opt_class Optional prefix for the classes.
55 * @param {boolean=} opt_useIframeForIE For IE, workaround windowed controls
56 * z-index issue by using a an iframe instead of a div for bg element.
57 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper; see {@link
58 * goog.ui.Component} for semantics.
59 * @constructor
60 * @extends {goog.ui.Dialog}
61 */
62goog.ui.Prompt = function(promptTitle, promptHtml, callback, opt_defaultValue,
63 opt_class, opt_useIframeForIE, opt_domHelper) {
64 goog.ui.Prompt.base(this, 'constructor',
65 opt_class, opt_useIframeForIE, opt_domHelper);
66
67 /**
68 * The id of the input element.
69 * @type {string}
70 * @private
71 */
72 this.inputElementId_ = this.makeId('ie');
73
74 this.setTitle(promptTitle);
75
76 var label = goog.html.SafeHtml.create('label', {'for': this.inputElementId_},
77 promptHtml instanceof goog.html.SafeHtml ? promptHtml :
78 goog.html.legacyconversions.safeHtmlFromString(promptHtml));
79 var br = goog.html.SafeHtml.create('br');
80 this.setSafeHtmlContent(goog.html.SafeHtml.concat(label, br, br));
81
82 this.callback_ = callback;
83 this.defaultValue_ = goog.isDef(opt_defaultValue) ? opt_defaultValue : '';
84
85 /** @desc label for a dialog button. */
86 var MSG_PROMPT_OK = goog.getMsg('OK');
87 /** @desc label for a dialog button. */
88 var MSG_PROMPT_CANCEL = goog.getMsg('Cancel');
89 var buttonSet = new goog.ui.Dialog.ButtonSet(opt_domHelper);
90 buttonSet.set(goog.ui.Dialog.DefaultButtonKeys.OK, MSG_PROMPT_OK, true);
91 buttonSet.set(goog.ui.Dialog.DefaultButtonKeys.CANCEL,
92 MSG_PROMPT_CANCEL, false, true);
93 this.setButtonSet(buttonSet);
94};
95goog.inherits(goog.ui.Prompt, goog.ui.Dialog);
96goog.tagUnsealableClass(goog.ui.Prompt);

178/** @override */
179goog.ui.Prompt.prototype.enterDocument = function() {
180 if (this.inputDecoratorFn_) {
181 this.inputDecoratorFn_(this.userInputEl_);

215/**
216 * Set the number of rows in the user input element.
217 * A values of 1 means use an <input> element. If the prompt is already
218 * rendered then you cannot change from <input> to <textarea> or vice versa.
219 * @param {number} rows Number of rows for user input element.
220 * @throws {goog.ui.Component.Error.ALREADY_RENDERED} If the component is
221 * already rendered and an attempt to change between <input> and <textarea>
222 * is made.
223 */
224goog.ui.Prompt.prototype.setRows = function(rows) {
225 if (this.isInDocument()) {
226 if (this.userInputEl_.tagName == goog.dom.TagName.INPUT) {
227 if (rows > 1) {
228 throw Error(goog.ui.Component.Error.ALREADY_RENDERED);
229 }
230 } else {
231 if (rows <= 1) {
232 throw Error(goog.ui.Component.Error.ALREADY_RENDERED);
233 }
234 this.userInputEl_.rows = rows;
235 }
236 }
237 this.rows_ = rows;
238};

273/**
274 * Create the initial DOM representation for the prompt.
275 * @override
276 */
277goog.ui.Prompt.prototype.createDom = function() {
278 goog.ui.Prompt.superClass_.createDom.call(this);
279
280 var cls = this.getClass();
281
282 // add input box to the content
283 var attrs = {
284 'className': goog.getCssName(cls, 'userInput'),
285 'value': this.defaultValue_};
286 if (this.rows_ == 1) {
287 // If rows == 1 then use an input element.
288 this.userInputEl_ = /** @type {!HTMLInputElement} */
289 (this.getDomHelper().createDom(goog.dom.TagName.INPUT, attrs));
290 this.userInputEl_.type = goog.dom.InputType.TEXT;
291 if (this.cols_) {
292 this.userInputEl_.size = this.cols_;
293 }
294 } else {
295 // If rows > 1 then use a textarea.
296 this.userInputEl_ = /** @type {!HTMLInputElement} */
297 (this.getDomHelper().createDom(goog.dom.TagName.TEXTAREA, attrs));
298 this.userInputEl_.rows = this.rows_;
299 if (this.cols_) {
300 this.userInputEl_.cols = this.cols_;
301 }
302 }
303
304 this.userInputEl_.id = this.inputElementId_;
305 var contentEl = this.getContentElement();
306 contentEl.appendChild(this.getDomHelper().createDom(
307 goog.dom.TagName.DIV, {'style': 'overflow: auto'}, this.userInputEl_));
308};

375/**
376 * Handles the closing of the prompt, invoking the callback function that was
377 * registered to handle the value returned by the prompt.
378 * @param {goog.ui.Dialog.Event} e The dialog's selection event.
379 * @private
380 */
381goog.ui.Prompt.prototype.onPromptExit_ = function(e) {
382 /*
383 * The timeouts below are required for one edge case. If after the dialog
384 * hides, suppose validation of the input fails which displays an alert. If
385 * the user pressed the Enter key to dismiss the alert that was displayed it
386 * can trigger the event handler a second time. This timeout ensures that the
387 * alert is displayed only after the prompt is able to clean itself up.
388 */
389 if (!this.isClosing_) {
390 this.isClosing_ = true;
391 if (e.key == 'ok') {
392 goog.Timer.callOnce(
393 goog.bind(this.callback_, this, this.userInputEl_.value), 1);
394 } else {
395 goog.Timer.callOnce(goog.bind(this.callback_, this, null), 1);
396 }
397 }
398};
