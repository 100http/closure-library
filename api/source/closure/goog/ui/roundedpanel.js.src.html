42/**
43 * Factory method that returns an instance of a BaseRoundedPanel.
44 * @param {number} radius The radius of the rounded corner(s), in pixels.
45 * @param {number} borderWidth The thickness of the border, in pixels.
46 * @param {string} borderColor The border color of the panel.
47 * @param {string=} opt_backgroundColor The background color of the panel.
48 * @param {number=} opt_corners The corners of the panel to be rounded. Any
49 * corners not specified will be rendered as square corners. Will default
50 * to all square corners if not specified.
51 * @param {goog.dom.DomHelper=} opt_domHelper The DOM helper object for the
52 * document we want to render in.
53 * @return {!goog.ui.BaseRoundedPanel} An instance of a
54 * goog.ui.BaseRoundedPanel subclass.
55 */
56goog.ui.RoundedPanel.create = function(radius,
57 borderWidth,
58 borderColor,
59 opt_backgroundColor,
60 opt_corners,
61 opt_domHelper) {
62 // This variable checks for the presence of Safari 3.0+ or Gecko 1.9+,
63 // which can leverage special CSS styles to create rounded corners.
64 var isCssReady = goog.userAgent.WEBKIT &&
65 goog.userAgent.isVersionOrHigher('500') ||
66 goog.userAgent.GECKO && goog.userAgent.isVersionOrHigher('1.9a');
67
68 if (isCssReady) {
69 // Safari 3.0+ and Firefox 3.0+ support this instance.
70 return new goog.ui.CssRoundedPanel(
71 radius,
72 borderWidth,
73 borderColor,
74 opt_backgroundColor,
75 opt_corners,
76 opt_domHelper);
77 } else {
78 return new goog.ui.GraphicsRoundedPanel(
79 radius,
80 borderWidth,
81 borderColor,
82 opt_backgroundColor,
83 opt_corners,
84 opt_domHelper);
85 }
86};

121 * Base class for the hierarchy of RoundedPanel classes. Do not
122 * instantiate directly. Instead, call goog.ui.RoundedPanel.create().
123 * The HTML structure for the RoundedPanel is:
124 * <pre>
125 * - div (Contains the background and content. Class name: goog-roundedpanel)
126 * - div (Contains the background/rounded corners. Class name:

128 * - div (Contains the content. Class name: goog-roundedpanel-content)
129 * </pre>
130 * @param {number} radius The radius of the rounded corner(s), in pixels.
131 * @param {number} borderWidth The thickness of the border, in pixels.
132 * @param {string} borderColor The border color of the panel.
133 * @param {string=} opt_backgroundColor The background color of the panel.
134 * @param {number=} opt_corners The corners of the panel to be rounded. Any
135 * corners not specified will be rendered as square corners. Will default
136 * to all square corners if not specified.
137 * @param {goog.dom.DomHelper=} opt_domHelper The DOM helper object for the
138 * document we want to render in.
139 * @extends {goog.ui.Component}
140 * @constructor

142goog.ui.BaseRoundedPanel = function(radius,
143 borderWidth,
144 borderColor,
145 opt_backgroundColor,
146 opt_corners,
147 opt_domHelper) {
148 goog.ui.Component.call(this, opt_domHelper);
149
150 /**
151 * The radius of the rounded corner(s), in pixels.
152 * @type {number}
153 * @private
154 */
155 this.radius_ = radius;
156
157 /**
158 * The thickness of the border, in pixels.
159 * @type {number}
160 * @private
161 */
162 this.borderWidth_ = borderWidth;
163
164 /**
165 * The border color of the panel.
166 * @type {string}
167 * @private

213goog.ui.BaseRoundedPanel.prototype.decorateInternal = function(element) {
214 goog.ui.BaseRoundedPanel.superClass_.decorateInternal.call(this, element);
215 goog.dom.classlist.add(goog.asserts.assert(this.getElement()),
216 goog.ui.RoundedPanel.Classes_.PANEL);
217
218 // Create backgroundElement_, and add it to the DOM.
219 this.backgroundElement_ = this.getDomHelper().createElement(
220 goog.dom.TagName.DIV);
221 this.backgroundElement_.className = goog.ui.RoundedPanel.Classes_.BACKGROUND;
222 this.getElement().appendChild(this.backgroundElement_);
223
224 // Set contentElement_ by finding a child node within element_ with the
225 // proper class name. If none exists, create it and add it to the DOM.
226 this.contentElement_ = goog.dom.getElementsByTagNameAndClass(
227 null, goog.ui.RoundedPanel.Classes_.CONTENT, this.getElement())[0];
228 if (!this.contentElement_) {
229 this.contentElement_ = this.getDomHelper().createDom(goog.dom.TagName.DIV);
230 this.contentElement_.className = goog.ui.RoundedPanel.Classes_.CONTENT;
231 this.getElement().appendChild(this.contentElement_);
232 }
233};

314 } else {
315 var topLeftRadius =
316 this.corners_ & goog.ui.RoundedPanel.Corner.TOP_LEFT ?
317 this.radius_ :
318 0;
319 var cornerStyle = this.getStyle_(goog.ui.RoundedPanel.Corner.TOP_LEFT);
320 this.backgroundElement_.style[cornerStyle] = topLeftRadius + 'px';
321 var topRightRadius =
322 this.corners_ & goog.ui.RoundedPanel.Corner.TOP_RIGHT ?
323 this.radius_ :
324 0;
325 cornerStyle = this.getStyle_(goog.ui.RoundedPanel.Corner.TOP_RIGHT);
326 this.backgroundElement_.style[cornerStyle] = topRightRadius + 'px';
327 var bottomRightRadius =
328 this.corners_ & goog.ui.RoundedPanel.Corner.BOTTOM_RIGHT ?
329 this.radius_ :
330 0;
331 cornerStyle = this.getStyle_(goog.ui.RoundedPanel.Corner.BOTTOM_RIGHT);
332 this.backgroundElement_.style[cornerStyle] = bottomRightRadius + 'px';
333 var bottomLeftRadius =
334 this.corners_ & goog.ui.RoundedPanel.Corner.BOTTOM_LEFT ?

343/**
344 * This method returns the CSS style based on the corner of the panel, and the
345 * user-agent.
346 * @param {number} corner The corner whose style name to retrieve.
347 * @private
348 * @return {string} The CSS style based on the specified corner.
349 */
350goog.ui.CssRoundedPanel.prototype.getStyle_ = function(corner) {
351 // Determine the proper corner to work with.
352 var cssCorner, suffixLeft, suffixRight;
353 if (goog.userAgent.WEBKIT) {
354 suffixLeft = 'Left';
355 suffixRight = 'Right';
356 } else {
357 suffixLeft = 'left';
358 suffixRight = 'right';
359 }
360 switch (corner) {
361 case goog.ui.RoundedPanel.Corner.ALL:
362 cssCorner = '';
363 break;
364 case goog.ui.RoundedPanel.Corner.TOP_LEFT:
365 cssCorner = 'Top' + suffixLeft;
366 break;
367 case goog.ui.RoundedPanel.Corner.TOP_RIGHT:
368 cssCorner = 'Top' + suffixRight;
369 break;
370 case goog.ui.RoundedPanel.Corner.BOTTOM_LEFT:
371 cssCorner = 'Bottom' + suffixLeft;
372 break;
373 case goog.ui.RoundedPanel.Corner.BOTTOM_RIGHT:
374 cssCorner = 'Bottom' + suffixRight;
375 break;
376 }
377
378 return goog.userAgent.WEBKIT ?
379 'WebkitBorder' + cssCorner + 'Radius' :
380 'MozBorderRadius' + cssCorner;
381};

481/**
482 * This method performs all the necessary DOM manipulation to create the panel.
483 * Overrides {@link goog.ui.Component#decorateInternal}.
484 * @param {Element} element The element to decorate.
485 * @protected
486 * @override
487 */
488goog.ui.GraphicsRoundedPanel.prototype.decorateInternal =
489 function(element) {
490 goog.ui.GraphicsRoundedPanel.superClass_.decorateInternal.call(this,
491 element);
492
493 // Calculate the points and angles for creating the rounded corners. Then
494 // instantiate a Graphics object for drawing purposes.
495 var elementSize = goog.style.getSize(this.getElement());
496 this.calculateArcParameters_(elementSize);
497 this.graphics_ = goog.graphics.createGraphics(
498 /** @type {number} */ (elementSize.width),
499 /** @type {number} */ (elementSize.height),
500 /** @type {number} */ (elementSize.width),
501 /** @type {number} */ (elementSize.height),
502 this.getDomHelper());
503 this.graphics_.createDom();
504
505 // Create the path, starting from the bottom-right corner, moving clockwise.
506 // End with the top-right corner.
507 var path = new goog.graphics.Path();
508 for (var i = 0; i < 4; i++) {
509 if (this.radii_[i]) {
510 // If radius > 0, draw an arc, moving to the first point and drawing
511 // a line to the others.
512 var cx = this.arcCenters_[i].x;
513 var cy = this.arcCenters_[i].y;
514 var rx = this.radii_[i];
515 var ry = rx;
516 var fromAngle = this.startAngles_[i];
517 var extent = this.endAngles_[i] - fromAngle;
518 var startX = cx + goog.math.angleDx(fromAngle, rx);
519 var startY = cy + goog.math.angleDy(fromAngle, ry);
520 if (i > 0) {
521 var currentPoint = path.getCurrentPoint();
522 if (!currentPoint || startX != currentPoint[0] ||
523 startY != currentPoint[1]) {
524 path.lineTo(startX, startY);
525 }
526 } else {
527 path.moveTo(startX, startY);
528 }
529 path.arcTo(rx, ry, fromAngle, extent);
530 } else if (i == 0) {
531 // If we're just starting out (ie. i == 0), move to the starting point.
532 path.moveTo(this.cornerStarts_[i].x,
533 this.cornerStarts_[i].y);
534 } else {
535 // Otherwise, draw a line to the starting point.
536 path.lineTo(this.cornerStarts_[i].x,
537 this.cornerStarts_[i].y);
538 }
539 }
540
541 // Close the path, create a stroke object, and fill the enclosed area, if
542 // needed. Then render the path.
543 path.close();
544 var stroke = this.borderWidth_ ?
545 new goog.graphics.Stroke(this.borderWidth_, this.borderColor_) :
546 null;
547 var fill = this.backgroundColor_ ?
548 new goog.graphics.SolidFill(this.backgroundColor_, 1) :
549 null;
550 this.graphics_.drawPath(path, stroke, fill);
551 this.graphics_.render(this.backgroundElement_);
552};

568/**
569 * Calculates the start coordinates, circle centers, and angles, for the rounded
570 * corners at each corner of the panel.
571 * @param {goog.math.Size} elementSize The size of element_.
572 * @private
573 */
574goog.ui.GraphicsRoundedPanel.prototype.calculateArcParameters_ =
575 function(elementSize) {
576 // Initialize the arrays containing the key points and angles.
577 this.radii_ = [];
578 this.cornerStarts_ = [];
579 this.arcCenters_ = [];
580 this.startAngles_ = [];
581 this.endAngles_ = [];
582
583 // Set the start points, circle centers, and angles for the bottom-right,
584 // bottom-left, top-left and top-right corners, in that order.
585 var angleInterval = 90;
586 var borderWidthOffset = this.borderWidth_ *
587 goog.ui.GraphicsRoundedPanel.BORDER_WIDTH_FACTOR_;
588 var radius, xStart, yStart, xCenter, yCenter, startAngle, endAngle;
589 for (var i = 0; i < 4; i++) {
590 var corner = Math.pow(2, i); // Determines which corner we're dealing with.
591 var isLeft = corner & goog.ui.RoundedPanel.Corner.LEFT;
592 var isTop = corner & goog.ui.RoundedPanel.Corner.TOP;
593
594 // Calculate the radius and the start coordinates.
595 radius = corner & this.corners_ ? this.radius_ : 0;
596 switch (corner) {
597 case goog.ui.RoundedPanel.Corner.BOTTOM_LEFT:
598 xStart = borderWidthOffset + radius;
599 yStart = elementSize.height - borderWidthOffset;
600 break;
601 case goog.ui.RoundedPanel.Corner.TOP_LEFT:
602 xStart = borderWidthOffset;
603 yStart = radius + borderWidthOffset;
604 break;
605 case goog.ui.RoundedPanel.Corner.TOP_RIGHT:
606 xStart = elementSize.width - radius - borderWidthOffset;
607 yStart = borderWidthOffset;
608 break;
609 case goog.ui.RoundedPanel.Corner.BOTTOM_RIGHT:
610 xStart = elementSize.width - borderWidthOffset;
611 yStart = elementSize.height - radius - borderWidthOffset;
612 break;
613 }
614
615 // Calculate the circle centers and start/end angles.
616 xCenter = isLeft ?
617 radius + borderWidthOffset :
618 elementSize.width - radius - borderWidthOffset;
619 yCenter = isTop ?
620 radius + borderWidthOffset :
621 elementSize.height - radius - borderWidthOffset;
622 startAngle = angleInterval * i;
623 endAngle = startAngle + angleInterval;
624
625 // Append the radius, angles, and coordinates to their arrays.
626 this.radii_[i] = radius;
627 this.cornerStarts_[i] = new goog.math.Coordinate(xStart, yStart);
628 this.arcCenters_[i] = new goog.math.Coordinate(xCenter, yCenter);
629 this.startAngles_[i] = startAngle;
630 this.endAngles_[i] = endAngle;
631 }
