45/**
46 * Overrides the superclass implementation by wrapping the caption text or DOM
47 * structure in a color indicator element. Creates the following DOM structure:
48 * <div class="goog-inline-block goog-toolbar-menu-button-caption">
49 * <div class="goog-color-menu-button-indicator">
50 * Contents...
51 * </div>
52 * </div>
53 * @param {goog.ui.ControlContent} content Text caption or DOM structure.
54 * @param {goog.dom.DomHelper} dom DOM helper, used for document interaction.
55 * @return {!Element} Caption element.
56 * @see goog.ui.ToolbarColorMenuButtonRenderer#createColorIndicator
57 * @override
58 */
59goog.ui.ToolbarColorMenuButtonRenderer.prototype.createCaption = function(
60 content, dom) {
61 return goog.ui.MenuButtonRenderer.wrapCaption(
62 goog.ui.ColorMenuButtonRenderer.wrapCaption(content, dom),
63 this.getCssClass(),
64 dom);
65};

68/**
69 * Takes a color menu button control's root element and a value object
70 * (which is assumed to be a color), and updates the button's DOM to reflect
71 * the new color. Overrides {@link goog.ui.ButtonRenderer#setValue}.
72 * @param {Element} element The button control's root element (if rendered).
73 * @param {*} value New value; assumed to be a color spec string.
74 * @override
75 */
76goog.ui.ToolbarColorMenuButtonRenderer.prototype.setValue = function(element,
77 value) {
78 if (element) {
79 goog.ui.ColorMenuButtonRenderer.setCaptionValue(
80 this.getContentElement(element), value);
81 }
82};

85/**
86 * Initializes the button's DOM when it enters the document. Overrides the
87 * superclass implementation by making sure the button's color indicator is
88 * initialized.
89 * @param {goog.ui.Control} button goog.ui.ColorMenuButton whose DOM is to be
90 * initialized as it enters the document.
91 * @override
92 */
93goog.ui.ToolbarColorMenuButtonRenderer.prototype.initializeDom = function(
94 button) {
95 this.setValue(button.getElement(), button.getValue());
96 goog.dom.classlist.add(
97 goog.asserts.assert(button.getElement()),
98 goog.getCssName('goog-toolbar-color-menu-button'));
99 goog.ui.ToolbarColorMenuButtonRenderer.superClass_.initializeDom.call(this,
100 button);
101};
