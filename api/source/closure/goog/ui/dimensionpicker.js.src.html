36/**
37 * A dimension picker allows the user to visually select a row and column
38 * count using their mouse and keyboard.
39 *
40 * The currently selected dimension is controlled by an ACTION event. Event
41 * listeners may retrieve the selected item using the
42 * {@link #getValue} method.
43 *
44 * @param {goog.ui.DimensionPickerRenderer=} opt_renderer Renderer used to
45 * render or decorate the palette; defaults to
46 * {@link goog.ui.DimensionPickerRenderer}.
47 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper, used for
48 * document interaction.
49 * @constructor
50 * @extends {goog.ui.Control}
51 * @final
52 */
53goog.ui.DimensionPicker = function(opt_renderer, opt_domHelper) {
54 goog.ui.Control.call(this, null,
55 opt_renderer || goog.ui.DimensionPickerRenderer.getInstance(),
56 opt_domHelper);
57
58 this.size_ = new goog.math.Size(this.minColumns, this.minRows);
59};
60goog.inherits(goog.ui.DimensionPicker, goog.ui.Control);

119 var handler = this.getHandler();

128 handler.listen(parent, goog.ui.Component.EventType.SHOW, this.handleShow_);

173/**
174 * Handles mousemove events. Determines which palette size was moused over and
175 * highlights it.
176 * @param {goog.events.BrowserEvent} e Mouse event to handle.
177 * @protected
178 */
179goog.ui.DimensionPicker.prototype.handleMouseMove = function(e) {
180 var highlightedSizeX = this.getRenderer().getGridOffsetX(this,
181 this.isRightToLeft() ? e.target.offsetWidth - e.offsetX : e.offsetX);
182 var highlightedSizeY = this.getRenderer().getGridOffsetY(this, e.offsetY);
183
184 this.setValue(highlightedSizeX, highlightedSizeY);
185};

194goog.ui.DimensionPicker.prototype.handleWindowResize = function(e) {

199/**
200 * Handle key events if supported, so the user can use the keyboard to
201 * manipulate the highlighted rows and columns.
202 * @param {goog.events.KeyEvent} e The key event object.
203 * @return {boolean} Whether the key event was handled.
204 * @override
205 */
206goog.ui.DimensionPicker.prototype.handleKeyEvent = function(e) {
207 var rows = this.highlightedRows_;
208 var columns = this.highlightedColumns_;
209 switch (e.keyCode) {
210 case goog.events.KeyCodes.DOWN:
211 rows++;
212 break;
213 case goog.events.KeyCodes.UP:
214 rows--;
215 break;
216 case goog.events.KeyCodes.LEFT:
217 if (this.isRightToLeft()) {
218 columns++;
219 } else {
220 if (columns == 1) {
221 // Delegate to parent.
222 return false;
223 } else {
224 columns--;
225 }
226 }
227 break;
228 case goog.events.KeyCodes.RIGHT:
229 if (this.isRightToLeft()) {
230 if (columns == 1) {
231 // Delegate to parent.
232 return false;
233 } else {
234 columns--;
235 }
236 } else {
237 columns++;
238 }
239 break;
240 default:
241 return goog.ui.DimensionPicker.superClass_.handleKeyEvent.call(this, e);
242 }
243 this.setValue(columns, rows);
244 return true;
245};

267/**
268 * Sets the currently highlighted dimensions. If the dimensions are not valid
269 * (not between 1 and the maximum number of columns/rows to show), they will
270 * be changed to the closest valid value.
271 * @param {(number|!goog.math.Size)} columns The number of columns to highlight,
272 * or a goog.math.Size object containing both.
273 * @param {number=} opt_rows The number of rows to highlight. Can be
274 * omitted when columns is a good.math.Size object.
275 */
276goog.ui.DimensionPicker.prototype.setValue = function(columns,
277 opt_rows) {
278 if (!goog.isDef(opt_rows)) {
279 columns = /** @type {!goog.math.Size} */ (columns);
280 opt_rows = columns.height;
281 columns = columns.width;
282 } else {
283 columns = /** @type {number} */ (columns);
284 }
285
286 // Ensure that the row and column values are within the minimum value (1) and
287 // maxmimum values.
288 columns = Math.max(1, columns);
289 opt_rows = Math.max(1, opt_rows);
290 columns = Math.min(this.maxColumns, columns);
291 opt_rows = Math.min(this.maxRows, opt_rows);
292
293 if (this.highlightedColumns_ != columns ||
294 this.highlightedRows_ != opt_rows) {
295 var renderer = this.getRenderer();
296 // Show one more row/column than highlighted so the user understands the
297 // palette can grow.
298 this.size_.width = Math.max(
299 Math.min(columns + 1, this.maxColumns), this.minColumns);
300 this.size_.height = Math.max(
301 Math.min(opt_rows + 1, this.maxRows), this.minRows);
302 renderer.updateSize(this, this.getElement());
303
304 this.highlightedColumns_ = columns;
305 this.highlightedRows_ = opt_rows;
306 renderer.setHighlightedSize(this, columns, opt_rows);
307 }
