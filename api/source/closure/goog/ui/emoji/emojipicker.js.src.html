40/**
41 * Creates a new, empty emoji picker. An emoji picker is a grid of emoji, each
42 * cell of the grid containing a single emoji. The picker may contain multiple
43 * pages of emoji.
44 *
45 * When a user selects an emoji, by either clicking or pressing enter, the
46 * picker fires a goog.ui.Component.EventType.ACTION event with the id. The
47 * client listens on this event and in the handler can retrieve the id of the
48 * selected emoji and do something with it, for instance, inserting an image
49 * tag into a rich text control. An emoji picker does not maintain state. That
50 * is, once an emoji is selected, the emoji picker does not remember which emoji
51 * was selected.
52 *
53 * The emoji picker is implemented as a tabpane with each tabpage being a table.
54 * Each of the tables are the same size to prevent jittering when switching
55 * between pages.
56 *
57 * @param {string} defaultImgUrl Url of the img that should be used to fill up
58 * the cells in the emoji table, to prevent jittering. Should be the same
59 * size as the emoji.
60 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper.
61 * @extends {goog.ui.Component}
62 * @constructor
63 */
64goog.ui.emoji.EmojiPicker = function(defaultImgUrl, opt_domHelper) {
65 goog.ui.Component.call(this, opt_domHelper);
66
67 this.defaultImgUrl_ = defaultImgUrl;
68
69 /**
70 * Emoji that this picker displays.
71 *
72 * @type {Array<Object>}
73 * @private
74 */
75 this.emoji_ = [];
76
77 /**
78 * Pages of this emoji picker.
79 *
80 * @type {Array<goog.ui.emoji.EmojiPalette>}
81 * @private
82 */
83 this.pages_ = [];
84
85 /**
86 * Keeps track of which pages in the picker have been loaded. Used for delayed
87 * loading of tabs.
88 *
89 * @type {Array<boolean>}
90 * @private
91 */
92 this.pageLoadStatus_ = [];
93
94 /**
95 * Tabpane to hold the pages of this emojipicker.
96 *
97 * @type {goog.ui.TabPane}
98 * @private
99 */
100 this.tabPane_ = null;
101
102 this.getHandler().listen(this, goog.ui.Component.EventType.ACTION,
103 this.onEmojiPaletteAction_);
104};

416/**
417 * Adjusts the number of rows to be the maximum row count out of all the emoji
418 * groups, in order to prevent jitter in switching among the tabs.
419 *
420 * @private
421 */
422goog.ui.emoji.EmojiPicker.prototype.adjustNumRowsIfNecessary_ = function() {
423 var currentMax = 0;
424
425 for (var i = 0; i < this.emoji_.length; i++) {
426 var numEmoji = this.emoji_[i].emoji.length;
427 var rowsNeeded = Math.ceil(numEmoji / this.numCols_);
428 if (rowsNeeded > currentMax) {
429 currentMax = rowsNeeded;
430 }
431 }
432
433 this.setNumRows(currentMax);
434};

452/**
453 * @override
454 * @suppress {deprecated} Using deprecated goog.ui.TabPane.
455 */
456goog.ui.emoji.EmojiPicker.prototype.createDom = function() {
457 this.setElementInternal(this.getDomHelper().createDom(goog.dom.TagName.DIV));
458
459 if (this.autoSizeByColumnCount_) {
460 this.adjustNumRowsIfNecessary_();
461 }
462
463 if (this.emoji_.length == 0) {
464 throw Error('Must add some emoji to the picker');
465 }
466
467 // If there is more than one group of emoji, we construct a tabpane
468 if (this.emoji_.length > 1) {
469 // Give the tabpane a div to use as its content element, since tabpane
470 // overwrites the CSS class of the element it's passed
471 var div = this.getDomHelper().createDom(goog.dom.TagName.DIV);
472 this.getElement().appendChild(div);
473 this.tabPane_ = new goog.ui.TabPane(div,
474 this.tabLocation_,
475 this.getDomHelper(),
476 true /* use MOUSEDOWN */);
477 }
478
479 this.renderer_ = this.progressiveRender_ ?
480 new goog.ui.emoji.ProgressiveEmojiPaletteRenderer(this.defaultImgUrl_) :
481 new goog.ui.emoji.EmojiPaletteRenderer(this.defaultImgUrl_);
482
483 for (var i = 0; i < this.emoji_.length; i++) {
484 var emoji = this.emoji_[i].emoji;
485 var page = this.delayedLoad_ ?
486 this.createPlaceholderEmojiPage_(emoji) :
487 this.createEmojiPage_(emoji, i);
488 this.pages_.push(page);
489 }
490
491 this.activePage_ = 0;
492 this.getElement().tabIndex = 0;
493};

506/**
507 * Creates a page if it has not already been loaded. This has the side effects
508 * of setting the load status of the page to true.
509 *
510 * @param {Array<Array<string>>} emoji Emoji for this page. See
511 * {@link addEmojiGroup} for more details.
512 * @param {number} index Index of the page in the emojipicker.
513 * @return {goog.ui.emoji.EmojiPalette} the emoji page.
514 * @private
515 */
516goog.ui.emoji.EmojiPicker.prototype.createEmojiPage_ = function(emoji, index) {
517 // Safeguard against trying to create the same page twice
518 if (this.pageLoadStatus_[index]) {
519 return null;
520 }
521
522 var palette = new goog.ui.emoji.EmojiPalette(emoji,
523 this.urlPrefix_,
524 this.renderer_,
525 this.getDomHelper());
526 if (!this.manualLoadOfAnimatedEmoji_) {
527 palette.loadAnimatedEmoji();
528 }
529 palette.setSize(this.numCols_, this.numRows_);
530 palette.setSupportedState(goog.ui.Component.State.FOCUSED, this.focusable_);
531 palette.createDom();
532 palette.setParent(this);
533
534 this.pageLoadStatus_[index] = true;
535
536 return palette;
537};

540/**
541 * Returns an array of emoji whose real URLs have been replaced with the
542 * default img URL. Used for delayed loading.
543 *
544 * @param {Array<Array<string>>} emoji Original emoji array.
545 * @return {!Array<!Array<string>>} emoji array with all emoji pointing to the
546 * default img.
547 * @private
548 */
549goog.ui.emoji.EmojiPicker.prototype.getPlaceholderEmoji_ = function(emoji) {
550 var placeholderEmoji = [];
551
552 for (var i = 0; i < emoji.length; i++) {
553 placeholderEmoji.push([this.defaultImgUrl_, emoji[i][1]]);
554 }
555
556 return placeholderEmoji;

598/**
599 * @override
600 * @suppress {deprecated} Using deprecated goog.ui.TabPane.
601 */
602goog.ui.emoji.EmojiPicker.prototype.enterDocument = function() {
603 goog.ui.emoji.EmojiPicker.superClass_.enterDocument.call(this);
604
605 for (var i = 0; i < this.pages_.length; i++) {
606 this.pages_[i].enterDocument();
607 var pageElement = this.pages_[i].getElement();
608
609 // Add a new tab to the tabpane if there's more than one group of emoji.
610 // If there is just one group of emoji, then we simply use the single
611 // page's element as the content for the picker
612 if (this.pages_.length > 1) {
613 // Create a simple default title containg the page number if the title
614 // was not provided in the emoji group params
615 var title = this.emoji_[i].title || (i + 1);
616 this.tabPane_.addPage(new goog.ui.TabPane.TabPage(
617 pageElement, title, this.getDomHelper()));
618 } else {
619 this.getElement().appendChild(pageElement);
620 }
621 }
622
623 // Initialize listeners. Note that we need to initialize this listener
624 // after createDom, because addPage causes the goog.ui.TabPane.Events.CHANGE
625 // event to fire, but we only want the handler (which loads delayed images)
626 // to run after the picker has been constructed.
627 if (this.tabPane_) {
628 this.getHandler().listen(
629 this.tabPane_, goog.ui.TabPane.Events.CHANGE, this.onPageChanged_);
630
631 // Make the tabpane unselectable so that changing tabs doesn't disturb the
632 // cursor
633 goog.style.setUnselectable(this.tabPane_.getElement(), true);
634 }
635
636 this.getElement().unselectable = 'on';
637};
