28/**
29 * Progressively renders an emoji palette. The progressive renderer tries to
30 * use img tags instead of background-image for sprited emoji, since most
31 * browsers render img tags progressively (i.e., as the data comes in), while
32 * only very new browsers render background-image progressively.
33 *
34 * @param {string} defaultImgUrl Url of the img that should be used to fill up
35 * the cells in the emoji table, to prevent jittering. Will be stretched
36 * to the emoji cell size. A good image is a transparent dot.
37 * @constructor
38 * @extends {goog.ui.emoji.EmojiPaletteRenderer}
39 * @final
40 */
41goog.ui.emoji.ProgressiveEmojiPaletteRenderer = function(defaultImgUrl) {
42 goog.ui.emoji.EmojiPaletteRenderer.call(this, defaultImgUrl);
43};
44goog.inherits(goog.ui.emoji.ProgressiveEmojiPaletteRenderer,
45 goog.ui.emoji.EmojiPaletteRenderer);

48/** @override */
49goog.ui.emoji.ProgressiveEmojiPaletteRenderer.prototype.
50 buildElementFromSpriteMetadata = function(dom, spriteInfo, displayUrl) {
51 var width = spriteInfo.getWidthCssValue();
52 var height = spriteInfo.getHeightCssValue();
53 var x = spriteInfo.getXOffsetCssValue();
54 var y = spriteInfo.getYOffsetCssValue();
55 // Need this extra div for proper vertical centering.
56 var inner = dom.createDom(goog.dom.TagName.IMG, {'src': displayUrl});
57 var el = /** @type {!HTMLDivElement} */ (dom.createDom(goog.dom.TagName.DIV,
58 goog.getCssName('goog-palette-cell-extra'), inner));
59 goog.style.setStyle(el, {
60 'width': width,
61 'height': height,
62 'overflow': 'hidden',
63 'position': 'relative'
64 });
65 goog.style.setStyle(inner, {
66 'left': x,
67 'top': y,
68 'position': 'absolute'
69 });
70
71 return el;
72};

75/** @override */
76goog.ui.emoji.ProgressiveEmojiPaletteRenderer.prototype.
77 updateAnimatedPaletteItem = function(item, animatedImg) {
78 // Just to be safe, we check for the existence of the img element within this
79 // palette item before attempting to modify it.
80 var img;
81 var el = item.firstChild;
82 while (el) {
83 if ('IMG' == el.tagName) {
84 img = /** @type {!Element} */ (el);
85 break;
86 }
87 el = el.firstChild;
88 }
89 if (!el) {
90 return;
91 }
92
93 img.width = animatedImg.width;
94 img.height = animatedImg.height;
95 goog.style.setStyle(img, {
96 'left': 0,
97 'top': 0
98 });
99 img.src = animatedImg.src;
100};
