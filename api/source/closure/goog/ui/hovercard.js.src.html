37/**
38 * Create a hover card object. Hover cards extend tooltips in that they don't
39 * have to be manually attached to each element that can cause them to display.
40 * Instead, you can create a function that gets called when the mouse goes over
41 * any element on your page, and returns whether or not the hovercard should be
42 * shown for that element.
43 *
44 * Alternatively, you can define a map of tag names to the attribute name each
45 * tag should have for that tag to trigger the hover card. See example below.
46 *
47 * Hovercards can also be triggered manually by calling
48 * {@code triggerForElement}, shown without a delay by calling
49 * {@code showForElement}, or triggered over other elements by calling
50 * {@code attach}. For the latter two cases, the application is responsible
51 * for calling {@code detach} when finished.
52 *
53 * HoverCard objects fire a TRIGGER event when the mouse moves over an element
54 * that can trigger a hovercard, and BEFORE_SHOW when the hovercard is
55 * about to be shown. Clients can respond to these events and can prevent the
56 * hovercard from being triggered or shown.
57 *
58 * @param {Function|Object} isAnchor Function that returns true if a given
59 * element should trigger the hovercard. Alternatively, it can be a map of
60 * tag names to the attribute that the tag should have in order to trigger
61 * the hovercard, e.g., {A: 'href'} for all links. Tag names must be all
62 * upper case; attribute names are case insensitive.
63 * @param {boolean=} opt_checkDescendants Use false for a performance gain if
64 * you are sure that none of your triggering elements have child elements.
65 * Default is true.
66 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper to use for
67 * creating and rendering the hovercard element.
68 * @param {Document=} opt_triggeringDocument Optional document to use in place
69 * of the one included in the DomHelper for finding triggering elements.
70 * Defaults to the document included in the DomHelper.
71 * @constructor
72 * @extends {goog.ui.AdvancedTooltip}
73 */
74goog.ui.HoverCard = function(isAnchor, opt_checkDescendants, opt_domHelper,
75 opt_triggeringDocument) {
76 goog.ui.AdvancedTooltip.call(this, null, null, opt_domHelper);
77
78 if (goog.isFunction(isAnchor)) {
79 // Override default implementation of {@code isAnchor_}.
80 this.isAnchor_ = isAnchor;
81 } else {
82
83 /**
84 * Map of tag names to attribute names that will trigger a hovercard.
85 * @type {Object}
86 * @private
87 */
88 this.anchors_ = isAnchor;
89 }
90
91 /**
92 * Whether anchors may have child elements. If true, then we need to check
93 * the parent chain of any mouse over event to see if any of those elements
94 * could be anchors. Default is true.
95 * @type {boolean}
96 * @private
97 */
98 this.checkDescendants_ = opt_checkDescendants != false;
99
100 /**
101 * Array of anchor elements that should be detached when we are no longer
102 * associated with them.
103 * @type {!Array<Element>}
104 * @private
105 */
106 this.tempAttachedAnchors_ = [];
107
108 /**
109 * Document containing the triggering elements, to which we listen for
110 * mouseover events.
111 * @type {Document}
112 * @private
113 */
114 this.document_ = opt_triggeringDocument || (opt_domHelper ?
115 opt_domHelper.getDocument() : goog.dom.getDocument());
116
117 goog.events.listen(this.document_, goog.events.EventType.MOUSEOVER,
118 this.handleTriggerMouseOver_, false, this);
119};
120goog.inherits(goog.ui.HoverCard, goog.ui.AdvancedTooltip);

166/**
167 * This function can be overridden by passing a function as the first parameter
168 * to the constructor.
169 * @param {Node} node Node to test.
170 * @return {boolean} Whether or not hovercard should be shown.
171 * @private
172 */
173goog.ui.HoverCard.prototype.isAnchor_ = function(node) {
174 return node.tagName in this.anchors_ &&
175 !!node.getAttribute(this.anchors_[node.tagName]);
176};

179/**
180 * If the user mouses over an element with the correct tag and attribute, then
181 * trigger the hovercard for that element. If anchors could have children, then
182 * we also need to check the parent chain of the given element.
183 * @param {goog.events.Event} e Mouse over event.
184 * @private
185 */
186goog.ui.HoverCard.prototype.handleTriggerMouseOver_ = function(e) {
187 var target = /** @type {Element} */ (e.target);
188 // Target might be null when hovering over disabled input textboxes in IE.
189 if (!target) {
190 return;
191 }
192 if (this.isAnchor_(target)) {
193 this.setPosition(null);
194 this.triggerForElement(target);
195 } else if (this.checkDescendants_) {
196 var trigger = goog.dom.getAncestor(target,
197 goog.bind(this.isAnchor_, this),
198 false,
199 this.maxSearchSteps_);
200 if (trigger) {
201 this.setPosition(null);
202 this.triggerForElement(/** @type {!Element} */ (trigger));
203 }
204 }
205};

208/**
209 * Triggers the hovercard to show after a delay.
210 * @param {Element} anchorElement Element that is triggering the hovercard.
211 * @param {goog.positioning.AbstractPosition=} opt_pos Position to display
212 * hovercard.
213 * @param {Object=} opt_data Data to pass to the onTrigger event.
214 */
215goog.ui.HoverCard.prototype.triggerForElement = function(anchorElement,
216 opt_pos, opt_data) {
217 if (anchorElement == this.currentAnchor_) {
218 // Element is already showing, just make sure it doesn't hide.
219 this.clearHideTimer();
220 return;
221 }
222 if (anchorElement == this.anchor) {
223 // Hovercard is pending, no need to retrigger.
224 return;
225 }
226
227 // If a previous hovercard was being triggered, cancel it.
228 this.maybeCancelTrigger_();
229
230 // Create a new event for this trigger
231 var triggerEvent = new goog.ui.HoverCard.TriggerEvent(
232 goog.ui.HoverCard.EventType.TRIGGER, this, anchorElement, opt_data);
233
234 if (!this.getElements().contains(anchorElement)) {
235 this.attach(anchorElement);
236 this.tempAttachedAnchors_.push(anchorElement);
237 }
238 this.anchor = anchorElement;
239 if (!this.onTrigger(triggerEvent)) {
240 this.onCancelTrigger();
241 return;
242 }
243 var pos = opt_pos || this.position_;
244 this.startShowTimer(anchorElement,
245 /** @type {goog.positioning.AbstractPosition} */ (pos));
246};

249/**
250 * Sets the current anchor element at the time that the hovercard is shown.
251 * @param {Element} anchor New current anchor element, or null if there is
252 * no current anchor.
253 * @private
254 */
255goog.ui.HoverCard.prototype.setCurrentAnchor_ = function(anchor) {
256 if (anchor != this.currentAnchor_) {
257 this.detachTempAnchor_(this.currentAnchor_);
258 }
259 this.currentAnchor_ = anchor;

263/**
264 * If given anchor is in the list of temporarily attached anchors, then
265 * detach and remove from the list.
266 * @param {Element|undefined} anchor Anchor element that we may want to detach
267 * from.
268 * @private
269 */
270goog.ui.HoverCard.prototype.detachTempAnchor_ = function(anchor) {
271 var pos = goog.array.indexOf(this.tempAttachedAnchors_, anchor);
272 if (pos != -1) {
273 this.detach(anchor);
274 this.tempAttachedAnchors_.splice(pos, 1);
275 }

306goog.ui.HoverCard.prototype.maybeCancelTrigger_ = function() {
307 if (this.getState() == goog.ui.Tooltip.State.WAITING_TO_SHOW ||
308 this.getState() == goog.ui.Tooltip.State.UPDATING) {

314/**
315 * This method gets called when we detect that a trigger event will not lead
316 * to the hovercard being shown.
317 * @protected
318 */
319goog.ui.HoverCard.prototype.onCancelTrigger = function() {
320 var event = new goog.ui.HoverCard.TriggerEvent(
321 goog.ui.HoverCard.EventType.CANCEL_TRIGGER, this, this.anchor || null);
322 this.dispatchEvent(event);
323 this.detachTempAnchor_(this.anchor);
324 delete this.anchor;
325};

328/**
329 * Gets the DOM element that triggered the current hovercard. Note that in
330 * the TRIGGER or CANCEL_TRIGGER events, the current hovercard's anchor may not
331 * be the one that caused the event, so use the event's anchor property instead.
332 * @return {Element} Object that caused the currently displayed hovercard (or
333 * pending hovercard if none is displayed) to be triggered.
334 */
335goog.ui.HoverCard.prototype.getAnchorElement = function() {
336 // this.currentAnchor_ is only set if the hovercard is showing. If it isn't
337 // showing yet, then use this.anchor as the pending anchor.
338 return /** @type {Element} */ (this.currentAnchor_ || this.anchor);

354/**
355 * This mouse over event is only received if the anchor is already attached.
356 * If it was attached manually, then it may need to be triggered.
357 * @param {goog.events.BrowserEvent} event Mouse over event.
358 * @override
359 */
360goog.ui.HoverCard.prototype.handleMouseOver = function(event) {
361 // If this is a child of a triggering element, find the triggering element.
362 var trigger = this.getAnchorFromElement(
363 /** @type {Element} */ (event.target));
364
365 // If we moused over an element different from the one currently being
366 // triggered (if any), then trigger this new element.
367 if (trigger && trigger != this.anchor) {
368 this.triggerForElement(trigger);
369 return;
370 }
371
372 goog.ui.HoverCard.superClass_.handleMouseOver.call(this, event);
373};

376/**
377 * If the mouse moves out of the trigger while we're being triggered, then
378 * cancel it.
379 * @param {goog.events.BrowserEvent} event Mouse out or blur event.
380 * @override
381 */
382goog.ui.HoverCard.prototype.handleMouseOutAndBlur = function(event) {
383 // Get ready to see if a trigger should be cancelled.
384 var anchor = this.anchor;
385 var state = this.getState();
386 goog.ui.HoverCard.superClass_.handleMouseOutAndBlur.call(this, event);
387 if (state != this.getState() &&
388 (state == goog.ui.Tooltip.State.WAITING_TO_SHOW ||
389 state == goog.ui.Tooltip.State.UPDATING)) {
390 // Tooltip's handleMouseOutAndBlur method sets anchor to null. Reset
391 // so that the cancel trigger event will have the right data, and so that
392 // it will be properly detached.
393 this.anchor = anchor;
394 this.onCancelTrigger(); // This will remove and detach the anchor.
395 }
396};

399/**
400 * Called by timer from mouse over handler. If this is called and the hovercard
401 * is not shown for whatever reason, then send a cancel trigger event.
402 * @param {Element} el Element to show tooltip for.
403 * @param {goog.positioning.AbstractPosition=} opt_pos Position to display popup

407goog.ui.HoverCard.prototype.maybeShow = function(el, opt_pos) {
408 goog.ui.HoverCard.superClass_.maybeShow.call(this, el, opt_pos);

410 if (!this.isVisible()) {
411 this.cancelTrigger();
