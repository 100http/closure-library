15/**
16 * @fileoverview A menu class for showing popups. A single popup can be
17 * attached to multiple anchor points. The menu will try to reposition itself
18 * if it goes outside the viewport.
19 *
20 * Decoration is the same as goog.ui.Menu except that the outer DIV can have a
21 * 'for' property, which is the ID of the element which triggers the popup.
22 *
23 * Decorate Example:
24 * <button id="dButton">Decorated Popup</button>
25 * <div id="dMenu" for="dButton" class="goog-menu">
26 * <div class="goog-menuitem">A a</div>
27 * <div class="goog-menuitem">B b</div>
28 * <div class="goog-menuitem">C c</div>
29 * <div class="goog-menuitem">D d</div>
30 * <div class="goog-menuitem">E e</div>
31 * <div class="goog-menuitem">F f</div>
32 * </div>
33 *
34 * TESTED=FireFox 2.0, IE6, Opera 9, Chrome.
35 * TODO(user): Key handling is flakey in Opera and Chrome
36 *
37 * @see ../demos/popupmenu.html
38 */

60/**
61 * A basic menu class.
62 * @param {?goog.dom.DomHelper=} opt_domHelper Optional DOM helper.
63 * @param {?goog.ui.MenuRenderer=} opt_renderer Renderer used to render or
64 * decorate the container; defaults to {@link goog.ui.MenuRenderer}.
65 * @extends {goog.ui.Menu}
66 * @constructor
67 */

69 goog.ui.Menu.call(this, opt_domHelper, opt_renderer);
70
71 this.setAllowAutoFocus(true);
72

77 * Map of attachment points for the menu. Key -> Object

111/**
112 * Decorate an existing HTML structure with the menu. Menu items will be
113 * constructed from elements with classname 'goog-menuitem', separators will be
114 * made from HR elements.
115 * @param {?Element} element Element to decorate.
116 * @override
117 */
118goog.ui.PopupMenu.prototype.decorateInternal = function(element) {
119 goog.ui.PopupMenu.superClass_.decorateInternal.call(this, element);
120 // 'for' is a custom attribute for attaching the menu to a click target
121 var htmlFor = element.getAttribute('for') || element.htmlFor;
122 if (htmlFor) {
123 this.attach(
124 this.getDomHelper().getElement(htmlFor),
125 goog.positioning.Corner.BOTTOM_LEFT);
126 }
127};

130/** @override */
131goog.ui.PopupMenu.prototype.enterDocument = function() {
132 goog.ui.PopupMenu.superClass_.enterDocument.call(this);
133
134 this.targets_.forEach(this.attachEvent_, this);
135
136 var handler = this.getHandler();
137 handler.listen(
138 this, goog.ui.Component.EventType.ACTION, this.onAction_);
139 handler.listen(this.getDomHelper().getDocument(),
140 goog.events.EventType.MOUSEDOWN, this.onDocClick, true);
141
142 // Webkit doesn't fire a mousedown event when opening the context menu,
143 // but we need one to update menu visibility properly. So in Safari handle
144 // contextmenu mouse events like mousedown.
145 // {@link http://bugs.webkit.org/show_bug.cgi?id=6595}
146 if (goog.userAgent.WEBKIT) {
147 handler.listen(this.getDomHelper().getDocument(),
148 goog.events.EventType.CONTEXTMENU, this.onDocClick, true);
149 }
150};

153/**
154 * Attaches the menu to a new popup position and anchor element. A menu can
155 * only be attached to an element once, since attaching the same menu for
156 * multiple positions doesn't make sense.
157 *
158 * @param {?Element} element Element whose click event should trigger the menu.
159 * @param {?goog.positioning.Corner=} opt_targetCorner Corner of the target that
160 * the menu should be anchored to.
161 * @param {goog.positioning.Corner=} opt_menuCorner Corner of the menu that
162 * should be anchored.
163 * @param {boolean=} opt_contextMenu Whether the menu should show on
164 * {@link goog.events.EventType.CONTEXTMENU} events, false if it should
165 * show on {@link goog.events.EventType.MOUSEDOWN} events. Default is
166 * MOUSEDOWN.
167 * @param {?goog.math.Box=} opt_margin Margin for the popup used in positioning
168 * algorithms.
169 */
170goog.ui.PopupMenu.prototype.attach = function(
171 element, opt_targetCorner, opt_menuCorner, opt_contextMenu, opt_margin) {
172
173 if (this.isAttachTarget(element)) {
174 // Already in the popup, so just return.
175 return;
176 }
177
178 var target = this.createAttachTarget(element, opt_targetCorner,
179 opt_menuCorner, opt_contextMenu, opt_margin);
180
181 if (this.isInDocument()) {
182 this.attachEvent_(target);
183 }
184
185 // Add a listener for keyboard actions on the menu.
186 var handler = goog.partial(this.onMenuKeyboardAction_, element);
187 if (this.getElement()) {
188 this.getHandler().listen(this.getElement(), goog.events.EventType.KEYDOWN,
189 handler);
190 }
191};

194/**
195 * Handles keyboard actions on the PopupMenu, according to
196 * http://www.w3.org/WAI/PF/aria-practices/#menubutton.
197 *
198 * <p>If the ESC key is pressed, the menu is hidden (which is handled by
199 * this.onAction_), and the focus is returned to the element whose click event
200 * triggered opening of the menu.
201 *
202 * <p>If the SPACE or ENTER keys are pressed, the highlighted menu item's
203 * listeners are fired.
204 *
205 * @param {Element} element Element whose click event triggered the menu.
206 * @param {goog.events.BrowserEvent} e The key down event.
207 * @private
208 */
209goog.ui.PopupMenu.prototype.onMenuKeyboardAction_ = function(element, e) {
210 if (e.keyCode == goog.events.KeyCodes.ESC) {
211 element.focus();
212 return;
213 }
214 var highlightedItem = this.getChildAt(this.getHighlightedIndex());
215 var targetElement = highlightedItem.getElement();
216 // Create an event to pass to the menu item's listener.
217 var event = new goog.events.BrowserEvent(e.getBrowserEvent(), targetElement);
218 event.target = targetElement;
219 // If an item is highlighted and the user presses the SPACE/ENTER key, the
220 // event target is the menu rather than the menu item, so we manually fire
221 // the listener of the correct menu item.
222 if (e.keyCode == goog.events.KeyCodes.SPACE ||
223 e.keyCode == goog.events.KeyCodes.ENTER) {
224 goog.events.fireListeners(targetElement,
225 goog.events.EventType.KEYDOWN, false, event);
226 }
227 // After activating a menu item the PopupMenu should be hidden (already
228 // implemented in this.onAction_ for ENTER/MOUSEDOWN).
229 if (e.keyCode == goog.events.KeyCodes.SPACE) {
230 this.hide();
231 }
232};

235/**
236 * Creates an object describing how the popup menu should be attached to the
237 * anchoring element based on the given parameters. The created object is
238 * stored, keyed by {@code element} and is retrievable later by invoking
239 * {@link #getAttachTarget(element)} at a later point.
240 *
241 * Subclass may add more properties to the returned object, as needed.
242 *
243 * @param {?Element} element Element whose click event should trigger the menu.
244 * @param {?goog.positioning.Corner=} opt_targetCorner Corner of the target that
245 * the menu should be anchored to.
246 * @param {?goog.positioning.Corner=} opt_menuCorner Corner of the menu that
247 * should be anchored.
248 * @param {boolean=} opt_contextMenu Whether the menu should show on
249 * {@link goog.events.EventType.CONTEXTMENU} events, false if it should
250 * show on {@link goog.events.EventType.MOUSEDOWN} events. Default is
251 * MOUSEDOWN.
252 * @param {?goog.math.Box=} opt_margin Margin for the popup used in positioning
253 * algorithms.
254 *
255 * @return {?Object} An object that describes how the popup menu should be
256 * attached to the anchoring element.
257 *
258 * @protected
259 */
260goog.ui.PopupMenu.prototype.createAttachTarget = function(
261 element, opt_targetCorner, opt_menuCorner, opt_contextMenu, opt_margin) {
262 if (!element) {
263 return null;
264 }
265
266 var target = {
267 element_: element,
268 targetCorner_: opt_targetCorner,
269 menuCorner_: opt_menuCorner,
270 eventType_: opt_contextMenu ? goog.events.EventType.CONTEXTMENU :
271 goog.events.EventType.MOUSEDOWN,
272 margin_: opt_margin
273 };
274
275 this.targets_.set(goog.getUid(element), target);
276
277 return target;
278};

399/**
400 * Show the menu using given positioning object.
401 * @param {?goog.positioning.AbstractPosition} position The positioning
402 * instance.
403 * @param {goog.positioning.Corner=} opt_menuCorner The corner of the menu to be
404 * positioned.
405 * @param {?goog.math.Box=} opt_margin A margin specified in pixels.
406 * @param {?Element=} opt_anchor The element which acts as visual anchor for
407 * this menu.
408 */
409goog.ui.PopupMenu.prototype.showWithPosition = function(position,
410 opt_menuCorner, opt_margin, opt_anchor) {
411 var isVisible = this.isVisible();
412 if (this.isOrWasRecentlyVisible() && this.toggleMode_) {
413 this.hide();
414 return;
415 }
416
417 // Set current anchor before dispatching BEFORE_SHOW. This is typically useful
418 // when we would need to make modifications based on the current anchor to the
419 // menu just before displaying it.
420 this.currentAnchor_ = opt_anchor || null;
421
422 // Notify event handlers that the menu is about to be shown.
423 if (!this.dispatchEvent(goog.ui.Component.EventType.BEFORE_SHOW)) {
424 return;
425 }
426
427 var menuCorner = typeof opt_menuCorner != 'undefined' ?
428 opt_menuCorner :
429 goog.positioning.Corner.TOP_START;
430
431 // This is a little hacky so that we can position the menu with minimal
432 // flicker.
433
434 if (!isVisible) {
435 // On IE, setting visibility = 'hidden' on a visible menu
436 // will cause a blur, forcing the menu to close immediately.
437 this.getElement().style.visibility = 'hidden';
438 }
439
440 goog.style.setElementShown(this.getElement(), true);
441 position.reposition(this.getElement(), menuCorner, opt_margin);
442
443 if (!isVisible) {
444 this.getElement().style.visibility = 'visible';
445 }
446
447 this.setHighlightedIndex(-1);
448
449 // setVisible dispatches a goog.ui.Component.EventType.SHOW event, which may
450 // be canceled to prevent the menu from being shown.
451 this.setVisible(true);
452};

455/**
456 * Show the menu at a given attached target.
457 * @param {!Object} target Popup target.
458 * @param {number} x The client-X associated with the show event.
459 * @param {number} y The client-Y associated with the show event.
460 * @protected
461 */
462goog.ui.PopupMenu.prototype.showMenu = function(target, x, y) {
463 var position = goog.isDef(target.targetCorner_) ?
464 new goog.positioning.AnchoredViewportPosition(target.element_,
465 target.targetCorner_, true) :
466 new goog.positioning.ViewportClientPosition(x, y);
467 if (position.setLastResortOverflow) {
468 // This is a ViewportClientPosition, so we can set the overflow policy.
469 // Allow the menu to slide from the corner rather than clipping if it is
470 // completely impossible to fit it otherwise.
471 position.setLastResortOverflow(goog.positioning.Overflow.ADJUST_X |
472 goog.positioning.Overflow.ADJUST_Y);
473 }
474 this.showWithPosition(position, target.menuCorner_, target.margin_,
475 target.element_);
476};

508/**
509 * Hides the menu.
510 */
511goog.ui.PopupMenu.prototype.hide = function() {
512 if (!this.isVisible()) {
513 return;
514 }
515
516 // setVisible dispatches a goog.ui.Component.EventType.HIDE event, which may
517 // be canceled to prevent the menu from being hidden.
518 this.setVisible(false);
519 if (!this.isVisible()) {
520 // HIDE event wasn't canceled; the menu is now hidden.
521 this.lastHide_ = goog.now();
522 this.currentAnchor_ = null;
523 }
524};

573goog.ui.PopupMenu.prototype.onTargetKeyboardAction_ = function(e) {

575 e.keyCode == goog.events.KeyCodes.ENTER ||
576 e.keyCode == goog.events.KeyCodes.DOWN) {
577 this.onTargetActivation_(e);
578 }
579 // If the popupmenu is opened using the DOWN key, the focus should be on the
580 // first menu item.
581 if (e.keyCode == goog.events.KeyCodes.DOWN) {
582 this.highlightFirst();

587/**
588 * Handles a browser event on one of the popup targets.
589 * @param {goog.events.BrowserEvent} e The browser event.
590 * @private
591 */
592goog.ui.PopupMenu.prototype.onTargetActivation_ = function(e) {
593 var keys = this.targets_.getKeys();
594 for (var i = 0; i < keys.length; i++) {
595 var target = /** @type {!Object} */(this.targets_.get(keys[i]));
596 if (target.element_ == e.currentTarget) {
597 this.showMenu(target, (e.clientX), (e.clientY));
598 e.preventDefault();
599 e.stopPropagation();
600 return;
601 }
602 }
