617/**
618 * Sets the background fill.
619 *
620 * @param {Array<Object>} fill An array of background fill specification
621 * objects. Each object may have the following properties:
622 * {string} area The area to fill, either 'bg' for background or 'c' for
623 * chart area. The default is 'bg'.
624 * {string} color (required) The color of the background fill.
625 * // TODO(user): Add support for gradient/stripes, which requires
626 * // a different object structure.
627 */
628goog.ui.ServerChart.prototype.setBackgroundFill = function(fill) {
629 var value = [];
630 goog.array.forEach(fill, function(spec) {
631 spec.area = spec.area || 'bg';
632 spec.effect = spec.effect || 's';
633 value.push([spec.area, spec.effect, spec.color].join(','));
634 });
635 value = value.join('|');
636 this.setParameterValue(goog.ui.ServerChart.UriParam.BACKGROUND_FILL, value);
637};
638
639
640/**
641 * Returns the background fill.
642 *
643 * @return {!Array<Object>} An array of background fill specifications.
644 * If the fill specification string is in an unsupported format, the method
645 * returns an empty array.
646 */
647goog.ui.ServerChart.prototype.getBackgroundFill = function() {
648 var value =
649 this.uri_.getParameterValue(goog.ui.ServerChart.UriParam.BACKGROUND_FILL);
650 var result = [];
651 if (goog.isDefAndNotNull(value)) {
652 var fillSpecifications = value.split('|');
653 var valid = true;
654 goog.array.forEach(fillSpecifications, function(spec) {
655 var parts = spec.split(',');
656 if (valid && parts[1] == 's') {
657 result.push({area: parts[0], effect: parts[1], color: parts[2]});
658 } else {
659 // If the format is unsupported, return an empty array.
660 result = [];
661 valid = false;
662 }
663 });
664 }
665 return result;
666};

1069/**
1070 * Sets the weight function for a Venn Diagram along with the associated
1071 * colors and legend text. Weights are assigned as follows:
1072 * weights[0] is relative area of circle A.
1073 * weights[1] is relative area of circle B.
1074 * weights[2] is relative area of circle C.
1075 * weights[3] is relative area of overlap of circles A and B.
1076 * weights[4] is relative area of overlap of circles A and C.
1077 * weights[5] is relative area of overlap of circles B and C.
1078 * weights[6] is relative area of overlap of circles A, B and C.
1079 * For a two circle Venn Diagram the weights are assigned as follows:
1080 * weights[0] is relative area of circle A.
1081 * weights[1] is relative area of circle B.
1082 * weights[2] is relative area of overlap of circles A and B.
1083 *
1084 * @param {Array<number>} weights The relative weights of the circles.
1085 * @param {Array<string>=} opt_legendText The legend labels for the circles.
1086 * @param {Array<string>=} opt_colors The colors for the circles.
1087 */
1088goog.ui.ServerChart.prototype.setVennSeries = function(
1089 weights, opt_legendText, opt_colors) {
1090 if (this.getType() != goog.ui.ServerChart.ChartType.VENN) {
1091 throw Error('Can only set a weight function for a Venn diagram.');
1092 }
1093 var dataMin = this.arrayMin_(weights);
1094 if (dataMin < this.minValue_) {
1095 this.minValue_ = dataMin;
1096 }
1097 var dataMax = this.arrayMax_(weights);
1098 if (dataMax > this.maxValue_) {
1099 this.maxValue_ = dataMax;
1100 }
1101 if (goog.isDef(opt_legendText)) {
1102 goog.array.forEach(
1103 opt_legendText,
1104 goog.bind(function(legend) {
1105 this.setLegendTexts_.push(legend);
1106 }, this));
1107 this.uri_.setParameterValue(goog.ui.ServerChart.UriParam.LEGEND_TEXTS,
1108 this.setLegendTexts_.join('|'));
1109 }
1110 // If the caller only gave three weights, then they wanted a two circle
1111 // Venn Diagram. Create a 3 circle weight function where circle C has
1112 // area zero.
1113 if (weights.length == 3) {
1114 weights[3] = weights[2];
1115 weights[2] = 0.0;
1116 }
1117 this.dataSets_.push(weights);
1118 if (goog.isDef(opt_colors)) {
1119 goog.array.forEach(opt_colors, goog.bind(function(color) {
1120 this.setColors_.push(color);
1121 }, this));
1122 this.uri_.setParameterValue(goog.ui.ServerChart.UriParam.DATA_COLORS,
1123 this.setColors_.join(','));
1124 }

1669/**
1670 * Converts a single number to an encoded data value suitable for ChartServer.
1671 * The TEXT encoding is the number in decimal; the SIMPLE encoding is a single
1672 * character, and the EXTENDED encoding is two characters. See
1673 * https://developers.google.com/chart/image/docs/data_formats for the detailed
1674 * specification of these encoding formats.
1675 *
1676 * @private
1677 * @param {?number} value The value to convert (null for a missing data point).
1678 * @param {number} minValue The minimum value (used for normalization).
1679 * @param {number} maxValue The maximum value (used for normalization).
1680 * @param {goog.ui.ServerChart.EncodingType} encoding The data encoding to use;
1681 * must not be AUTOMATIC.
1682 * @return {string} The encoded data value.
1683 */
1684goog.ui.ServerChart.prototype.getConvertedValue_ = function(value,
1685 minValue,
1686 maxValue,
1687 encoding) {
1688 goog.asserts.assert(minValue <= maxValue,
1689 'minValue should be less than or equal to maxValue');
1690 var isExtended = (encoding == goog.ui.ServerChart.EncodingType.EXTENDED);
1691
1692 if (goog.isNull(value) || !goog.isDef(value) || isNaN(value) ||
1693 value < minValue || value > maxValue) {
1694 return isExtended ? '__' : '_';
1695 }
1696
1697 if (encoding == goog.ui.ServerChart.EncodingType.TEXT) {
1698 return String(value);
1699 }
1700
1701 var frac = goog.ui.ServerChart.DEFAULT_NORMALIZATION;
1702 if (maxValue > minValue) {
1703 frac = (value - minValue) / (maxValue - minValue);
1704 // Previous checks of value ensure that 0 <= frac <= 1 at this point.
1705 }
1706
1707 if (isExtended) {
1708 var maxIndex = goog.ui.ServerChart.CHART_VALUES_EXTENDED.length;
1709 var upperBound = goog.ui.ServerChart.EXTENDED_UPPER_BOUND;
1710 var index1 = Math.floor(frac * upperBound / maxIndex);
1711 var index2 = Math.floor((frac * upperBound) % maxIndex);
1712 var extendedVals = goog.ui.ServerChart.CHART_VALUES_EXTENDED;
1713 return extendedVals.charAt(index1) + extendedVals.charAt(index2);
1714 }
1715
1716 var index = Math.round(frac * (goog.ui.ServerChart.CHART_VALUES.length - 1));
1717 return goog.ui.ServerChart.CHART_VALUES.charAt(index);
1718};
