15/**
16 * @fileoverview A base ratings widget that allows the user to select a rating,
17 * like "star video" in Google Video. This fires a "change" event when the user
18 * selects a rating.
19 *
20 * Keyboard:
21 * ESC = Clear (if supported)
22 * Home = 1 star
23 * End = Full rating
24 * Left arrow = Decrease rating
25 * Right arrow = Increase rating
26 * 0 = Clear (if supported)
27 * 1 - 9 = nth star
28 *
29 * @see ../demos/ratings.html
30 */

127/**
128 * Decorate a HTML structure already in the document. Expects the structure:
129 * <pre>
130 * - div
131 * - select
132 * - option 1 #text = 1 star
133 * - option 2 #text = 2 stars
134 * - option 3 #text = 3 stars
135 * - option N (where N is max number of ratings)
136 * </pre>
137 *
138 * The div can contain other elements for graceful degredation, but they will be
139 * hidden when the decoration occurs.
140 *
141 * @param {Element} el Div element to decorate.
142 * @override
143 */
144goog.ui.Ratings.prototype.decorateInternal = function(el) {
145 var select = el.getElementsByTagName(goog.dom.TagName.SELECT)[0];
146 if (!select) {
147 throw Error('Can not decorate ' + el + ', with Ratings. Must ' +
148 'contain select box');
149 }
150 this.ratings_.length = 0;
151 for (var i = 0, n = select.options.length; i < n; i++) {
152 var option = select.options[i];
153 this.ratings_.push(option.text);
154 }
155 this.setSelectedIndex(select.selectedIndex);
156 select.style.display = 'none';
157 this.attachedFormField_ = select;
158 this.createDom();
159 el.insertBefore(this.getElement(), select);
160};

163/**
164 * Render the rating widget inside the provided element. This will override the
165 * current content of the element.
166 * @override
167 */
168goog.ui.Ratings.prototype.enterDocument = function() {
169 var el = this.getElement();
170 goog.asserts.assert(el, 'The DOM element for ratings cannot be null.');
171 el.tabIndex = 0;
172 goog.dom.classlist.add(el, this.getCssClass());
173 goog.a11y.aria.setRole(el, goog.a11y.aria.Role.SLIDER);
174 goog.a11y.aria.setState(el, goog.a11y.aria.State.VALUEMIN, 0);
175 var max = this.ratings_.length - 1;
176 goog.a11y.aria.setState(el, goog.a11y.aria.State.VALUEMAX, max);
177 var handler = this.getHandler();
178 handler.listen(el, 'keydown', this.onKeyDown_);
179
180 // Create the elements for the stars
181 for (var i = 0; i < this.ratings_.length; i++) {
182 var star = this.getDomHelper().createDom(goog.dom.TagName.SPAN, {
183 'title': this.ratings_[i],
184 'class': this.getClassName_(i, false),
185 'index': i});
186 this.stars_.push(star);
187 el.appendChild(star);
188 }
189
190 handler.listen(el, goog.events.EventType.CLICK, this.onClick_);
191 handler.listen(el, goog.events.EventType.MOUSEOUT, this.onMouseOut_);
192 handler.listen(el, goog.events.EventType.MOUSEOVER, this.onMouseOver_);
193
194 this.highlightIndex_(this.selectedIndex_);
195};

229/**
230 * Sets the selected index. If the provided index is greater than the number of
231 * ratings then the max is set. 0 is the first item, -1 is no selection.
232 * @param {number} index The index of the rating to select.
233 */
234goog.ui.Ratings.prototype.setSelectedIndex = function(index) {
235 index = Math.max(-1, Math.min(index, this.ratings_.length - 1));
236 if (index != this.selectedIndex_) {
237 this.selectedIndex_ = index;
238 this.highlightIndex_(this.selectedIndex_);
239 if (this.attachedFormField_) {
240 if (this.attachedFormField_.tagName == goog.dom.TagName.SELECT) {
241 this.attachedFormField_.selectedIndex = index;
242 } else {
243 this.attachedFormField_.value =
244 /** @type {string} */ (this.getValue());
245 }
246 var ratingsElement = this.getElement();
247 goog.asserts.assert(ratingsElement,
248 'The DOM ratings element cannot be null.');
249 goog.a11y.aria.setState(ratingsElement,
250 goog.a11y.aria.State.VALUENOW,
251 this.ratings_[index]);
252 }
253 this.dispatchEvent(goog.ui.Ratings.EventType.CHANGE);
254 }
255};

356/**
357 * Handle the mouse moving over a star.
358 * @param {goog.events.BrowserEvent} e The browser event.
359 * @private
360 */
361goog.ui.Ratings.prototype.onMouseOver_ = function(e) {
362 if (!this.isEnabled()) {
363 return;
364 }
365 if (goog.isDef(e.target.index)) {
366 var n = e.target.index;
367 if (this.highlightedIndex_ != n) {
368 this.highlightIndex_(n);
369 this.highlightedIndex_ = n;
370 this.dispatchEvent(goog.ui.Ratings.EventType.HIGHLIGHT_CHANGE);
371 this.dispatchEvent(goog.ui.Ratings.EventType.HIGHLIGHT);
372 }
373 }
374};

406/**
407 * Handle the key down event. 0 = unselected in this case, 1 = the first rating
408 * @param {goog.events.BrowserEvent} e The browser event.
409 * @private
410 */
411goog.ui.Ratings.prototype.onKeyDown_ = function(e) {
412 if (!this.isEnabled()) {
413 return;
414 }
415 switch (e.keyCode) {
416 case 27: // esc
417 this.setSelectedIndex(-1);
418 break;
419 case 36: // home
420 this.setSelectedIndex(0);
421 break;
422 case 35: // end
423 this.setSelectedIndex(this.ratings_.length);
424 break;
425 case 37: // left arrow
426 this.setSelectedIndex(this.getSelectedIndex() - 1);
427 break;
428 case 39: // right arrow
429 this.setSelectedIndex(this.getSelectedIndex() + 1);
430 break;
431 default:
432 // Detected a numeric key stroke, such as 0 - 9. 0 clears, 1 is first
433 // star, 9 is 9th star or last if there are less than 9 stars.
434 var num = parseInt(String.fromCharCode(e.keyCode), 10);
435 if (!isNaN(num)) {
436 this.setSelectedIndex(num - 1);
437 }
438 }
439};

467/**
468 * Get the class name for a given rating. All stars have the class:
469 * goog-ratings-star.
470 * Other possible classnames dependent on position and state are:
471 * goog-ratings-firststar-on
472 * goog-ratings-firststar-off
473 * goog-ratings-midstar-on
474 * goog-ratings-midstar-off
475 * goog-ratings-laststar-on
476 * goog-ratings-laststar-off
477 * @param {number} i Index to get class name for.
478 * @param {boolean} on Whether it should be on.
479 * @return {string} The class name.
480 * @private
481 */
482goog.ui.Ratings.prototype.getClassName_ = function(i, on) {
483 var className;
484 var enabledClassName;
485 var baseClass = this.getCssClass();
486
487 if (i === 0) {
488 className = goog.getCssName(baseClass, 'firststar');
489 } else if (i == this.ratings_.length - 1) {
490 className = goog.getCssName(baseClass, 'laststar');
491 } else {
492 className = goog.getCssName(baseClass, 'midstar');
493 }
494
495 if (on) {
496 className = goog.getCssName(className, 'on');
497 } else {
498 className = goog.getCssName(className, 'off');
499 }
500
501 if (this.isEnabled_) {
502 enabledClassName = goog.getCssName(baseClass, 'enabled');
503 } else {
504 enabledClassName = goog.getCssName(baseClass, 'disabled');
505 }
506
507 return goog.getCssName(baseClass, 'star') + ' ' + className +
508 ' ' + enabledClassName;
