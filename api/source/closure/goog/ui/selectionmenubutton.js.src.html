40/**
41 * A selection menu button control. Extends {@link goog.ui.MenuButton}.
42 * Menu contains 'select all' and 'select none' MenuItems for selecting all and
43 * no items by default. Other MenuItems can be added by user.
44 *
45 * The checkbox content fires the action events associated with the 'select all'
46 * and 'select none' menu items.
47 *
48 * @param {goog.ui.ButtonRenderer=} opt_renderer Renderer used to render or
49 * decorate the menu button; defaults to {@link goog.ui.MenuButtonRenderer}.
50 * @param {goog.ui.MenuItemRenderer=} opt_itemRenderer Optional menu item
51 * renderer.
52 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper, used for
53 * document interaction.
54 * @constructor
55 * @extends {goog.ui.MenuButton}
56 */
57goog.ui.SelectionMenuButton = function(opt_renderer,
58 opt_itemRenderer,
59 opt_domHelper) {
60 goog.ui.MenuButton.call(this,
61 null,
62 null,
63 opt_renderer,
64 opt_domHelper);
65 this.initialItemRenderer_ = opt_itemRenderer || null;
66};
67goog.inherits(goog.ui.SelectionMenuButton, goog.ui.MenuButton);
68goog.tagUnsealableClass(goog.ui.SelectionMenuButton);

129/**
130 * Gets the checkbox element. Needed because if decorating html, getContent()
131 * may include and comment/text elements in addition to the input element.
132 * @return {Element} Checkbox.
133 * @protected
134 */
135goog.ui.SelectionMenuButton.prototype.getCheckboxElement = function() {
136 var elements = this.getDomHelper().getElementsByTagNameAndClass(
137 goog.dom.TagName.INPUT,
138 goog.getCssName('goog-selectionmenubutton-checkbox'),
139 this.getContentElement());
140 return elements[0];

144/**
145 * Checkbox click handler.
146 * @param {goog.events.BrowserEvent} e Checkbox click event.
147 * @protected
148 */
149goog.ui.SelectionMenuButton.prototype.handleCheckboxClick = function(e) {
150 if (this.selectionState == goog.ui.SelectionMenuButton.SelectionState.NONE) {
151 this.setSelectionState(goog.ui.SelectionMenuButton.SelectionState.ALL);
152 if (this.getItemAt(0)) {
153 this.getItemAt(0).dispatchEvent( // 'All' item
154 goog.ui.Component.EventType.ACTION);
155 }
156 } else {
157 this.setSelectionState(goog.ui.SelectionMenuButton.SelectionState.NONE);
158 if (this.getItemAt(1)) {
159 this.getItemAt(1).dispatchEvent( // 'None' item
160 goog.ui.Component.EventType.ACTION);
161 }
162 }
163};

166/**
167 * Menu action handler to update checkbox checked state.
168 * @param {goog.events.Event} e Menu action event.
169 * @private
170 */
171goog.ui.SelectionMenuButton.prototype.handleMenuAction_ = function(e) {
172 if (e.target.getModel() == goog.ui.SelectionMenuButton.SelectionState.ALL) {
173 this.setSelectionState(goog.ui.SelectionMenuButton.SelectionState.ALL);
174 } else {
175 this.setSelectionState(goog.ui.SelectionMenuButton.SelectionState.NONE);
176 }
177};

180/**
181 * Set up events related to the menu items.
182 * @private
183 */
184goog.ui.SelectionMenuButton.prototype.addMenuEvent_ = function() {
185 if (this.getItemAt(0) && this.getItemAt(1)) {
186 this.getHandler().listen(this.getMenu(),
187 goog.ui.Component.EventType.ACTION,
188 this.handleMenuAction_);
189 this.getItemAt(0).setModel(goog.ui.SelectionMenuButton.SelectionState.ALL);
190 this.getItemAt(1).setModel(goog.ui.SelectionMenuButton.SelectionState.NONE);
191 }
192};

195/**
196 * Set up events related to the checkbox.

199goog.ui.SelectionMenuButton.prototype.addCheckboxEvent = function() {
200 this.getHandler().listen(this.getCheckboxElement(),
201 goog.events.EventType.CLICK,
202 this.handleCheckboxClick);

206/**
207 * Adds the checkbox to the button, and adds 2 items to the menu corresponding
208 * to 'select all' and 'select none'.
209 * @override
210 * @protected
211 */
212goog.ui.SelectionMenuButton.prototype.createDom = function() {
213 goog.ui.SelectionMenuButton.superClass_.createDom.call(this);
214
215 this.createCheckbox();
216
217 /** @desc Text for 'All' button, used to select all items in a list. */
218 var MSG_SELECTIONMENUITEM_ALL = goog.getMsg('All');
219 /** @desc Text for 'None' button, used to unselect all items in a list. */
220 var MSG_SELECTIONMENUITEM_NONE = goog.getMsg('None');
221
222 var itemAll = new goog.ui.MenuItem(MSG_SELECTIONMENUITEM_ALL,
223 null,
224 this.getDomHelper(),
225 this.initialItemRenderer_);
226 var itemNone = new goog.ui.MenuItem(MSG_SELECTIONMENUITEM_NONE,
227 null,
228 this.getDomHelper(),
229 this.initialItemRenderer_);
230 this.addItem(itemAll);
231 this.addItem(itemNone);
232
233 this.addCheckboxEvent();
234 this.addMenuEvent_();
235};

265/**
266 * Set selection state and update checkbox.
267 * @param {goog.ui.SelectionMenuButton.SelectionState} state Selection state.
268 */
269goog.ui.SelectionMenuButton.prototype.setSelectionState = function(state) {
270 if (this.selectionState != state) {
271 var checkbox = this.getCheckboxElement();
272 if (state == goog.ui.SelectionMenuButton.SelectionState.ALL) {
273 checkbox.checked = true;
274 goog.style.setOpacity(checkbox, 1);
275 } else if (state == goog.ui.SelectionMenuButton.SelectionState.SOME) {
276 checkbox.checked = true;
277 // TODO(user): Get UX help to style this
278 goog.style.setOpacity(checkbox, 0.5);
279 } else { // NONE
280 checkbox.checked = false;
281 goog.style.setOpacity(checkbox, 1);
282 }
283 this.selectionState = state;
284 }
