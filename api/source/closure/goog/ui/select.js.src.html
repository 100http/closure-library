41/**
42 * A selection control. Extends {@link goog.ui.MenuButton} by composing a
43 * menu with a selection model, and automatically updating the button's caption
44 * based on the current selection.
45 *
46 * Select fires the following events:
47 * CHANGE - after selection changes.
48 *
49 * @param {goog.ui.ControlContent=} opt_caption Default caption or existing DOM
50 * structure to display as the button's caption when nothing is selected.
51 * Defaults to no caption.
52 * @param {goog.ui.Menu=} opt_menu Menu containing selection options.
53 * @param {goog.ui.ButtonRenderer=} opt_renderer Renderer used to render or
54 * decorate the control; defaults to {@link goog.ui.MenuButtonRenderer}.
55 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper, used for
56 * document interaction.
57 * @param {!goog.ui.MenuRenderer=} opt_menuRenderer Renderer used to render or
58 * decorate the menu; defaults to {@link goog.ui.MenuRenderer}.
59 * @constructor
60 * @extends {goog.ui.MenuButton}
61 */
62goog.ui.Select = function(opt_caption, opt_menu, opt_renderer, opt_domHelper,
63 opt_menuRenderer) {
64 goog.ui.Select.base(this, 'constructor',
65 opt_caption, opt_menu, opt_renderer, opt_domHelper,
66 opt_menuRenderer ||
67 new goog.ui.MenuRenderer(goog.a11y.aria.Role.LISTBOX));
68 /**
69 * Default caption to show when no option is selected.
70 * @private {goog.ui.ControlContent}
71 */
72 this.defaultCaption_ = this.getContent();
73
74 /**
75 * The initial value of the aria label of the content element. This will be
76 * null until the caption is first populated and will be non-null thereafter.
77 * @private {?string}
78 */
79 this.initialAriaLabel_ = null;
80
81 this.setPreferredAriaRole(goog.a11y.aria.Role.LISTBOX);
82};
83goog.inherits(goog.ui.Select, goog.ui.MenuButton);
84goog.tagUnsealableClass(goog.ui.Select);

103/**
104 * Decorates the given element with this control. Overrides the superclass
105 * implementation by initializing the default caption on the select button.
106 * @param {Element} element Element to decorate.
107 * @override
108 */
109goog.ui.Select.prototype.decorateInternal = function(element) {
110 goog.ui.Select.superClass_.decorateInternal.call(this, element);
111 var caption = this.getCaption();
112 if (caption) {
113 // Initialize the default caption.
114 this.setDefaultCaption(caption);
115 } else if (!this.getSelectedItem()) {
116 // If there is no default caption and no selected item, select the first
117 // option (this is technically an arbitrary choice, but what most people
118 // would expect to happen).
119 this.setSelectedIndex(0);
120 }
121};

124/** @override */
125goog.ui.Select.prototype.disposeInternal = function() {
126 goog.ui.Select.superClass_.disposeInternal.call(this);
127
128 if (this.selectionModel_) {
129 this.selectionModel_.dispose();
130 this.selectionModel_ = null;
131 }
132
133 this.defaultCaption_ = null;
134};

137/**
138 * Handles {@link goog.ui.Component.EventType.ACTION} events dispatched by
139 * the menu item clicked by the user. Updates the selection model, calls
140 * the superclass implementation to hide the menu, stops the propagation of
141 * the event, and dispatches an ACTION event on behalf of the select control
142 * itself. Overrides {@link goog.ui.MenuButton#handleMenuAction}.
143 * @param {goog.events.Event} e Action event to handle.
144 * @override
145 */
146goog.ui.Select.prototype.handleMenuAction = function(e) {
147 this.setSelectedItem(/** @type {goog.ui.MenuItem} */ (e.target));
148 goog.ui.Select.base(this, 'handleMenuAction', e);
149
150 // NOTE(chrishenry): We should not stop propagation and then fire
151 // our own ACTION event. Fixing this without breaking anyone
152 // relying on this event is hard though.
153 e.stopPropagation();
154 this.dispatchEvent(goog.ui.Component.EventType.ACTION);
155};

158/**
159 * Handles {@link goog.events.EventType.SELECT} events raised by the
160 * selection model when the selection changes. Updates the contents of the
161 * select button.
162 * @param {goog.events.Event} e Selection event to handle.
163 */
164goog.ui.Select.prototype.handleSelectionChange = function(e) {
165 var item = this.getSelectedItem();
166 goog.ui.Select.superClass_.setValue.call(this, item && item.getValue());
167 this.updateCaption();

171/**
172 * Replaces the menu currently attached to the control (if any) with the given
173 * argument, and updates the selection model. Does nothing if the new menu is
174 * the same as the old one. Overrides {@link goog.ui.MenuButton#setMenu}.
175 * @param {goog.ui.Menu} menu New menu to be attached to the menu button.
176 * @return {goog.ui.Menu|undefined} Previous menu (undefined if none).
177 * @override
178 */
179goog.ui.Select.prototype.setMenu = function(menu) {
180 // Call superclass implementation to replace the menu.
181 var oldMenu = goog.ui.Select.superClass_.setMenu.call(this, menu);
182
183 // Do nothing unless the new menu is different from the current one.
184 if (menu != oldMenu) {
185 // Clear the old selection model (if any).
186 if (this.selectionModel_) {
187 this.selectionModel_.clear();
188 }
189
190 // Initialize new selection model (unless the new menu is null).
191 if (menu) {
192 if (this.selectionModel_) {
193 menu.forEachChild(function(child, index) {
194 this.setCorrectAriaRole_(
195 /** @type {goog.ui.MenuItem|goog.ui.MenuSeparator} */ (child));
196 this.selectionModel_.addItem(child);
197 }, this);
198 } else {
199 this.createSelectionModel_(menu);
200 }
201 }
202 }
203
204 return oldMenu;
205};

293/**
294 * Selects the specified option (assumed to be in the select menu), and
295 * deselects the previously selected option, if any. A null argument clears

297 * @param {goog.ui.MenuItem} item Option to be selected (null to clear

300goog.ui.Select.prototype.setSelectedItem = function(item) {

325/**
326 * Selects the first option found with an associated value equal to the
327 * argument, or clears the selection if no such option is found. A null
328 * argument also clears the selection. Overrides {@link
329 * goog.ui.Button#setValue}.
330 * @param {*} value Value of the option to be selected (null to clear
331 * the selection).
332 * @override
333 */
334goog.ui.Select.prototype.setValue = function(value) {
335 if (goog.isDefAndNotNull(value) && this.selectionModel_) {
336 for (var i = 0, item; item = this.selectionModel_.getItemAt(i); i++) {
337 if (item && typeof item.getValue == 'function' &&
338 item.getValue() == value) {
339 this.setSelectedItem(/** @type {!goog.ui.MenuItem} */ (item));
340 return;
341 }
342 }
343 }
344
345 this.setSelectedItem(null);
346};

349/**
350 * Gets the value associated with the currently selected option (null if none).
351 *
352 * Note that unlike {@link goog.ui.Button#getValue} which this method overrides,
353 * the "value" of a Select instance is the value of its selected menu item, not
354 * its own value. This makes a difference because the "value" of a Button is
355 * reset to the value of the element it decorates when it's added to the DOM
356 * (via ButtonRenderer), whereas the value of the selected item is unaffected.
357 * So while setValue() has no effect on a Button before it is added to the DOM,
358 * it will make a persistent change to a Select instance (which is consistent
359 * with any changes made by {@link goog.ui.Select#setSelectedItem} and
360 * {@link goog.ui.Select#setSelectedIndex}).
361 *
362 * @override
363 */
364goog.ui.Select.prototype.getValue = function() {
365 var selectedItem = this.getSelectedItem();
366 return selectedItem ? selectedItem.getValue() : null;
367};

399/**
400 * Creates a new selection model and sets up an event listener to handle
401 * {@link goog.events.EventType.SELECT} events dispatched by it.
402 * @param {goog.ui.Component=} opt_component If provided, will add the
403 * component's children as items to the selection model.
404 * @private
405 */
406goog.ui.Select.prototype.createSelectionModel_ = function(opt_component) {
407 this.selectionModel_ = new goog.ui.SelectionModel();
408 if (opt_component) {
409 opt_component.forEachChild(function(child, index) {
410 this.setCorrectAriaRole_(
411 /** @type {goog.ui.MenuItem|goog.ui.MenuSeparator} */ (child));
412 this.selectionModel_.addItem(child);
413 }, this);
414 }
415 this.listenToSelectionModelEvents_();
416};

431/**
432 * Updates the caption to be shown in the select button. If no option is
433 * selected and a default caption is set, sets the caption to the default
434 * caption; otherwise to the empty string.
435 * @protected
436 */
437goog.ui.Select.prototype.updateCaption = function() {
438 var item = this.getSelectedItem();
439 this.setContent(item ? item.getCaption() : this.defaultCaption_);
440
441 var contentElement = this.getRenderer().getContentElement(this.getElement());
442 // Despite the ControlRenderer interface indicating the return value is
443 // {Element}, many renderers cast element.firstChild to {Element} when it is
444 // really {Node}. Checking tagName verifies this is an {!Element}.
445 if (contentElement && this.getDomHelper().isElement(contentElement)) {
446 if (this.initialAriaLabel_ == null) {
447 this.initialAriaLabel_ = goog.a11y.aria.getLabel(contentElement);
448 }
449 var itemElement = item ? item.getElement() : null;
450 goog.a11y.aria.setLabel(contentElement, itemElement ?
451 goog.a11y.aria.getLabel(itemElement) : this.initialAriaLabel_);
452 this.updateAriaActiveDescendant_();
453 }
454};

457/**
458 * Updates the aria active descendant attribute.
459 * @private
460 */
461goog.ui.Select.prototype.updateAriaActiveDescendant_ = function() {
462 var renderer = this.getRenderer();
463 if (renderer) {
464 var contentElement = renderer.getContentElement(this.getElement());
465 if (contentElement) {
466 var buttonElement = this.getElementStrict();
467 if (!contentElement.id) {
468 contentElement.id = goog.ui.IdGenerator.getInstance().getNextUniqueId();
469 }
470 goog.a11y.aria.setRole(contentElement, goog.a11y.aria.Role.OPTION);
471 goog.a11y.aria.setState(buttonElement,
472 goog.a11y.aria.State.ACTIVEDESCENDANT, contentElement.id);
473 if (this.selectionModel_) {
474 // We can't use selectionmodel's getItemCount here because we need to
475 // skip separators.
476 var items = this.selectionModel_.getItems();
477 var menuItemCount = goog.array.count(items, function(item) {
478 return item instanceof goog.ui.MenuItem;
479 });
480 goog.a11y.aria.setState(contentElement, goog.a11y.aria.State.SETSIZE,
481 menuItemCount);
482 // Set a human-readable selection index.
483 goog.a11y.aria.setState(contentElement, goog.a11y.aria.State.POSINSET,
484 1 + this.selectionModel_.getSelectedIndex());
485 }
486 }
487 }
