222 * @override
223 */
224goog.ui.RichTextSpellChecker.prototype.check = function() {
225 this.blockReadyEvents();
226 this.preChargeDictionary_(this.rootNode_, this.dictionaryPreScanSize_);
227 this.unblockReadyEvents();
228
229 this.eventHandler_.listen(this.spellCheck,
230 goog.spell.SpellCheck.EventType.READY, this.onDictionaryCharged_, true);
231 this.spellCheck.processPending();
232};

235/**
236 * Processes nodes recursively.
237 *
238 * @param {Node} node Node to start with.
239 * @param {number} words Max number of words to process.
240 * @private
241 */
242goog.ui.RichTextSpellChecker.prototype.preChargeDictionary_ = function(node,
243 words) {
244 while (node) {
245 var next = this.nextNode_(node);
246 if (this.isExcluded_(node)) {
247 node = next;
248 continue;
249 }
250 if (node.nodeType == goog.dom.NodeType.TEXT) {
251 if (node.nodeValue) {
252 words -= this.populateDictionary(node.nodeValue, words);
253 if (words <= 0) {
254 return;
255 }
256 }
257 } else if (node.nodeType == goog.dom.NodeType.ELEMENT) {
258 if (node.firstChild) {
259 next = node.firstChild;
260 }
261 }
262 node = next;
263 }
264};

267/**
268 * Starts actual processing after the dictionary is charged.
269 * @param {goog.events.Event} e goog.spell.SpellCheck.EventType.READY event.
270 * @private
271 */
272goog.ui.RichTextSpellChecker.prototype.onDictionaryCharged_ = function(e) {
273 e.stopPropagation();
274 this.eventHandler_.unlisten(this.spellCheck,
275 goog.spell.SpellCheck.EventType.READY, this.onDictionaryCharged_, true);
276
277 // Now actually do the spell checking.
278 this.clearWordElements();
279 this.initializeAsyncMode();
280 this.elementsInserted_ = 0;
281 var result = this.processNode_(this.rootNode_);
282 if (result == goog.ui.AbstractSpellChecker.AsyncResult.PENDING) {
283 goog.Timer.callOnce(this.boundContinueAsyncFn_);
284 return;
285 }
286 this.finishAsyncProcessing();
287 this.finishCheck_();
288};

291/**
292 * Continues asynchrnonous spell checking.
293 * @private
294 */
295goog.ui.RichTextSpellChecker.prototype.continueAsync_ = function() {
296 var result = this.continueAsyncProcessing();
297 if (result == goog.ui.AbstractSpellChecker.AsyncResult.PENDING) {
298 goog.Timer.callOnce(this.boundContinueAsyncFn_);
299 return;
300 }
301 result = this.processNode_(this.currentNode_);
302 if (result == goog.ui.AbstractSpellChecker.AsyncResult.PENDING) {
303 goog.Timer.callOnce(this.boundContinueAsyncFn_);
304 return;
305 }
306 this.finishAsyncProcessing();
307 this.finishCheck_();
308};

361/** @override */
362goog.ui.RichTextSpellChecker.prototype.setExcludeMarker = function(marker) {
363 if (marker) {
364 if (typeof marker == 'string') {
365 marker = [marker];
366 }
367
368 this.excludeTags = [];
369 this.excludeMarker = [];
370 for (var i = 0; i < marker.length; i++) {
371 var parts = marker[i].split('.');
372 if (parts.length == 2) {
373 this.excludeTags.push(parts[0]);
374 this.excludeMarker.push(parts[1]);
375 } else {
376 this.excludeMarker.push(parts[0]);
377 this.excludeTags.push(undefined);
378 }
379 }
380 }
381};

384/**
385 * Determines if the node is excluded from checking.
386 *
387 * @param {Node} node The node to check.
388 * @return {boolean} Whether the node is excluded.
389 * @private
390 */
391goog.ui.RichTextSpellChecker.prototype.isExcluded_ = function(node) {
392 if (this.excludeMarker && node.className) {
393 for (var i = 0; i < this.excludeMarker.length; i++) {
394 var excludeTag = this.excludeTags[i];
395 var excludeClass = this.excludeMarker[i];
396 var isExcluded = !!(excludeClass &&
397 node.className.indexOf(excludeClass) != -1 &&
398 (!excludeTag || node.tagName == excludeTag));
399 if (isExcluded) {
400 return true;
401 }
402 }
403 }
404 return false;
405};

408/**
409 * Processes nodes recursively.
410 *
411 * @param {Node} node Node where to start.
412 * @return {goog.ui.AbstractSpellChecker.AsyncResult|undefined} Result code.
413 * @private
414 */
415goog.ui.RichTextSpellChecker.prototype.processNode_ = function(node) {
416 delete this.currentNode_;
417 while (node) {
418 var next = this.nextNode_(node);
419 if (this.isExcluded_(node)) {
420 node = next;
421 continue;
422 }
423 if (node.nodeType == goog.dom.NodeType.TEXT) {
424 var deleteNode = true;
425 if (node.nodeValue) {
426 var currentElements = this.elementsInserted_;
427 var result = this.processTextAsync(node, node.nodeValue);
428 if (result == goog.ui.AbstractSpellChecker.AsyncResult.PENDING) {
429 // This markes node for deletion (empty nodes get deleted couple
430 // of lines down this function). This is so our algorithm terminates.
431 // In this case the node may be needlessly recreated, but it
432 // happens rather infrequently and saves a lot of code.
433 node.nodeValue = '';
434 this.currentNode_ = node;
435 return result;
436 }
437 // If we did not add nodes in processing, the current element is still
438 // valid. Let's preserve it!
439 if (currentElements == this.elementsInserted_) {
440 deleteNode = false;
441 }
442 }
443 if (deleteNode) {
444 goog.dom.removeNode(node);
445 }
446 } else if (node.nodeType == goog.dom.NodeType.ELEMENT) {
447 // If this is a spell checker element...
448 if (node.className == this.wordClassName) {
449 // First, reconsolidate the text nodes inside the element - editing
450 // in IE splits them up.
451 var runner = node.firstChild;
452 while (runner) {
453 if (this.isTextLeaf_(runner)) {
454 while (this.isTextLeaf_(runner.nextSibling)) {
455 // Yes, this is not super efficient in IE, but it will almost
456 // never happen.
457 runner.nodeValue += runner.nextSibling.nodeValue;
458 goog.dom.removeNode(runner.nextSibling);
459 }
460 }
461 runner = runner.nextSibling;
462 }
463 // Move its contents out and reprocess it on the next iteration.
464 if (node.firstChild) {
465 next = node.firstChild;
466 while (node.firstChild) {
467 node.parentNode.insertBefore(node.firstChild, node);
468 }
469 }
470 // get rid of the empty shell.
471 goog.dom.removeNode(node);
472 } else {
473 if (node.firstChild) {
474 next = node.firstChild;
475 }
476 }
477 }
478 node = next;
479 }
480};

566/**
567 * Processes nodes recursively, removes all spell checker markup, and
568 * consolidates text nodes.
569 *
570 * @param {Node} node node on which to recurse.
571 * @private
572 */
573goog.ui.RichTextSpellChecker.prototype.restoreNode_ = function(node) {
574 while (node) {
575 if (this.isExcluded_(node)) {
576 node = node.nextSibling;
577 continue;
578 }
579 // Contents of the child of the element is usually 1 text element, but the
580 // user can actually add multiple nodes in it during editing. So we move
581 // all the children out, prepend, and reprocess (pointer is set back to
582 // the first node that's been moved out, and the loop repeats).
583 if (node.nodeType == goog.dom.NodeType.ELEMENT && node.className ==
584 this.wordClassName) {
585 var firstElement = node.firstChild;
586 var next;
587 for (var child = firstElement; child; child = next) {
588 next = child.nextSibling;
589 node.parentNode.insertBefore(child, node);
590 }
591 next = firstElement || node.nextSibling;
592 goog.dom.removeNode(node);
593 node = next;
594 continue;
595 }
596 // If this is a chain of text elements, we're trying to consolidate it.
597 var textLeaf = this.isTextLeaf_(node);
598 if (textLeaf) {
599 var textNodes = 1;
600 var next = node.nextSibling;
601 while (this.isTextLeaf_(node.previousSibling)) {
602 node = node.previousSibling;
603 ++textNodes;
604 }
605 while (this.isTextLeaf_(next)) {
606 next = next.nextSibling;
607 ++textNodes;
608 }
609 if (textNodes > 1) {
610 this.workBuffer_.append(node.nodeValue);
611 while (this.isTextLeaf_(node.nextSibling)) {
612 this.workBuffer_.append(node.nextSibling.nodeValue);
613 goog.dom.removeNode(node.nextSibling);
614 }
615 node.nodeValue = this.workBuffer_.toString();
616 this.workBuffer_.clear();
617 }
618 }
619 // Process child nodes, if any.
620 if (node.firstChild) {
621 this.restoreNode_(node.firstChild);
622 }
623 node = node.nextSibling;
624 }
625};

685/**
686 * Handles keyboard events inside the editor to allow keyboard navigation
687 * between misspelled words and activation of the suggestion menu.
688 *
689 * @param {goog.events.BrowserEvent} e the key event.
690 * @return {boolean} The handled value.
691 * @protected
692 */
693goog.ui.RichTextSpellChecker.prototype.handleRootNodeKeyEvent = function(e) {
694 var handled = false;
695 switch (e.keyCode) {
696 case goog.events.KeyCodes.RIGHT:
697 if (e.ctrlKey) {
698 handled = this.navigate(goog.ui.AbstractSpellChecker.Direction.NEXT);
699 }
700 break;
701
702 case goog.events.KeyCodes.LEFT:
703 if (e.ctrlKey) {
704 handled = this.navigate(
705 goog.ui.AbstractSpellChecker.Direction.PREVIOUS);
706 }
707 break;
708
709 case goog.events.KeyCodes.DOWN:
710 if (this.getFocusedElementIndex()) {
711 var el = this.editorDom_.getElement(this.makeElementId(
712 this.getFocusedElementIndex()));
713 if (el) {
714 var position = goog.style.getClientPosition(el);
715
716 if (this.isEditorIframe()) {
717 var iframePosition = goog.style.getClientPosition(
718 this.getElementStrict());
719 position = goog.math.Coordinate.sum(iframePosition, position);
720 }
721
722 var size = goog.style.getSize(el);
723 position.x += size.width / 2;
724 position.y += size.height / 2;
725 this.showSuggestionsMenu(el, position);
726 handled = true;
727 }
728 }
729 break;
730 }
731
732 if (handled) {
733 e.preventDefault();
734 }
735
736 return handled;
737};
