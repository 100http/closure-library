32/**
33 * A color palette is a grid of color swatches that the user can highlight or
34 * select via the keyboard or the mouse. The selection state of the palette is
35 * controlled by a selection model. When the user makes a selection, the
36 * component fires an ACTION event. Event listeners may retrieve the selected
37 * color using the {@link #getSelectedColor} method.
38 *
39 * @param {Array<string>=} opt_colors Array of colors in any valid CSS color
40 * format.
41 * @param {goog.ui.PaletteRenderer=} opt_renderer Renderer used to render or
42 * decorate the palette; defaults to {@link goog.ui.PaletteRenderer}.
43 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper, used for
44 * document interaction.
45 * @constructor
46 * @extends {goog.ui.Palette}
47 */
48goog.ui.ColorPalette = function(opt_colors, opt_renderer, opt_domHelper) {
49 /**
50 * Array of colors to show in the palette.
51 * @type {Array<string>}
52 * @private
53 */
54 this.colors_ = opt_colors || [];
55
56 goog.ui.Palette.call(this, null,
57 opt_renderer || goog.ui.PaletteRenderer.getInstance(), opt_domHelper);
58
59 // Set the colors separately from the super call since we need the correct
60 // DomHelper to be initialized for this class.
61 this.setColors(this.colors_);
62};
63goog.inherits(goog.ui.ColorPalette, goog.ui.Palette);
64goog.tagUnsealableClass(goog.ui.ColorPalette);

93/**
94 * Sets the colors that are contained in the palette.
95 * @param {Array<string>} colors Array of colors in any valid CSS color format.
96 * @param {Array<string>=} opt_labels The array of labels to be used as
97 * tooltips. When not provided, the color value will be used.
98 */
99goog.ui.ColorPalette.prototype.setColors = function(colors, opt_labels) {
100 this.colors_ = colors;
101 this.labels_ = opt_labels || null;
102 this.normalizedColors_ = null;
103 this.setContent(this.createColorNodes());
104};

107/**
108 * @return {?string} The current selected color in hex, or null.
109 */
110goog.ui.ColorPalette.prototype.getSelectedColor = function() {
111 var selectedItem = /** @type {Element} */ (this.getSelectedItem());
112 if (selectedItem) {
113 var color = goog.style.getStyle(selectedItem, 'background-color');
114 return goog.ui.ColorPalette.parseColor_(color);
115 } else {
116 return null;
117 }
118};

121/**
122 * Sets the selected color. Clears the selection if the argument is null or
123 * can't be parsed as a color.
124 * @param {?string} color The color to set as selected; null clears the
125 * selection.
126 */
127goog.ui.ColorPalette.prototype.setSelectedColor = function(color) {
128 var hexColor = goog.ui.ColorPalette.parseColor_(color);
129 if (!this.normalizedColors_) {
130 this.normalizedColors_ = goog.array.map(this.colors_, function(color) {
131 return goog.ui.ColorPalette.parseColor_(color);
132 });
133 }
134 this.setSelectedIndex(hexColor ?
135 goog.array.indexOf(this.normalizedColors_, hexColor) : -1);
136};

139/**
140 * @return {!Array<!Node>} An array of DOM nodes for each color.
141 * @protected
142 */
143goog.ui.ColorPalette.prototype.createColorNodes = function() {
144 return goog.array.map(this.colors_, function(color, index) {
145 var swatch = this.getDomHelper().createDom(goog.dom.TagName.DIV, {
146 'class': goog.getCssName(this.getRenderer().getCssClass(),
147 'colorswatch'),
148 'style': 'background-color:' + color
149 });
150 if (this.labels_ && this.labels_[index]) {
151 swatch.title = this.labels_[index];
152 } else {
153 swatch.title = color.charAt(0) == '#' ?
154 'RGB (' + goog.color.hexToRgb(color).join(', ') + ')' : color;
155 }
156 return swatch;
157 }, this);
158};

161/**
162 * Takes a string, attempts to parse it as a color spec, and returns a
163 * normalized hex color spec if successful (null otherwise).
164 * @param {?string} color String possibly containing a color spec; may be null.
165 * @return {?string} Normalized hex color spec, or null if the argument can't
166 * be parsed as a color.
167 * @private
168 */
169goog.ui.ColorPalette.parseColor_ = function(color) {
170 if (color) {
171 /** @preserveTry */
172 try {
173 return goog.color.parse(color).hex;
174 } catch (ex) {
175 // Fall through.
176 }
177 }
178 return null;
179};
