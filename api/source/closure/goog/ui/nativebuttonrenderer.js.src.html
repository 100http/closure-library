33/**
34 * Renderer for {@link goog.ui.Button}s. Renders and decorates native HTML
35 * button elements. Since native HTML buttons have built-in support for many
36 * features, overrides many expensive (and redundant) superclass methods to
37 * be no-ops.
38 * @constructor
39 * @extends {goog.ui.ButtonRenderer}
40 */
41goog.ui.NativeButtonRenderer = function() {
42 goog.ui.ButtonRenderer.call(this);
43};
44goog.inherits(goog.ui.NativeButtonRenderer, goog.ui.ButtonRenderer);
45goog.addSingletonGetter(goog.ui.NativeButtonRenderer);

51 return undefined;

55/**
56 * Returns the button's contents wrapped in a native HTML button element. Sets
57 * the button's disabled attribute as needed.
58 * @param {goog.ui.Control} button Button to render.
59 * @return {Element} Root element for the button (a native HTML button element).
60 * @override
61 */
62goog.ui.NativeButtonRenderer.prototype.createDom = function(button) {
63 this.setUpNativeButton_(button);
64 return button.getDomHelper().createDom(goog.dom.TagName.BUTTON, {
65 'class': this.getClassNames(button).join(' '),
66 'disabled': !button.isEnabled(),
67 'title': button.getTooltip() || '',
68 'value': button.getValue() || ''
69 }, button.getCaption() || '');
70};

73/**
74 * Overrides {@link goog.ui.ButtonRenderer#canDecorate} by returning true only
75 * if the element is an HTML button.
76 * @param {Element} element Element to decorate.
77 * @return {boolean} Whether the renderer can decorate the element.
78 * @override
79 */
80goog.ui.NativeButtonRenderer.prototype.canDecorate = function(element) {
81 return element.tagName == goog.dom.TagName.BUTTON ||
82 (element.tagName == goog.dom.TagName.INPUT && (
83 element.type == goog.dom.InputType.BUTTON ||
84 element.type == goog.dom.InputType.SUBMIT ||
85 element.type == goog.dom.InputType.RESET));
86};

89/** @override */
90goog.ui.NativeButtonRenderer.prototype.decorate = function(button, element) {
91 this.setUpNativeButton_(button);
92 if (element.disabled) {
93 // Add the marker class for the DISABLED state before letting the superclass
94 // implementation decorate the element, so its state will be correct.
95 var disabledClassName = goog.asserts.assertString(
96 this.getClassForState(goog.ui.Component.State.DISABLED));
97 goog.dom.classlist.add(element, disabledClassName);
98 }
99 return goog.ui.NativeButtonRenderer.superClass_.decorate.call(this, button,
100 element);
101};

104/**
105 * Native buttons natively support BiDi and keyboard focus.
106 * @suppress {visibility} getHandler and performActionInternal
107 * @override
108 */
109goog.ui.NativeButtonRenderer.prototype.initializeDom = function(button) {
110 // WARNING: This is a hack, and it is only applicable to native buttons,
111 // which are special because they do natively what most goog.ui.Controls
112 // do programmatically. Do not use your renderer's initializeDom method
113 // to hook up event handlers!
114 button.getHandler().listen(button.getElement(), goog.events.EventType.CLICK,
115 button.performActionInternal);
116};

150/**
151 * @override
152 * Native buttons also expose the DISABLED state in the HTML button's
153 * {@code disabled} attribute.
154 */
155goog.ui.NativeButtonRenderer.prototype.setState = function(button, state,
156 enable) {
157 goog.ui.NativeButtonRenderer.superClass_.setState.call(this, button, state,
158 enable);
159 var element = button.getElement();
160 if (element && state == goog.ui.Component.State.DISABLED) {
161 element.disabled = enable;
162 }
163};

202/**
