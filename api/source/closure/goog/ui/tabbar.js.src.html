36/**
37 * Tab bar UI component. A tab bar contains tabs, rendered above, below,
38 * before, or after tab contents. Tabs in tab bars dispatch the following
39 * events:
40 * <ul>
41 * <li>{@link goog.ui.Component.EventType.ACTION} when activated via the
42 * keyboard or the mouse,
43 * <li>{@link goog.ui.Component.EventType.SELECT} when selected, and
44 * <li>{@link goog.ui.Component.EventType.UNSELECT} when deselected.
45 * </ul>
46 * Clients may listen for all of the above events on the tab bar itself, and
47 * refer to the event target to identify the tab that dispatched the event.
48 * When an unselected tab is clicked for the first time, it dispatches both a
49 * {@code SELECT} event and an {@code ACTION} event; subsequent clicks on an
50 * already selected tab only result in {@code ACTION} events.
51 *
52 * @param {goog.ui.TabBar.Location=} opt_location Tab bar location; defaults to
53 * {@link goog.ui.TabBar.Location.TOP}.
54 * @param {goog.ui.TabBarRenderer=} opt_renderer Renderer used to render or
55 * decorate the container; defaults to {@link goog.ui.TabBarRenderer}.
56 * @param {goog.dom.DomHelper=} opt_domHelper DOM helper, used for document
57 * interaction.
58 * @constructor
59 * @extends {goog.ui.Container}
60 */
61goog.ui.TabBar = function(opt_location, opt_renderer, opt_domHelper) {
62 this.setLocation(opt_location || goog.ui.TabBar.Location.TOP);
63
64 goog.ui.Container.call(this, this.getOrientation(),
65 opt_renderer || goog.ui.TabBarRenderer.getInstance(),
66 opt_domHelper);
67
68 this.listenToTabEvents_();
69};
70goog.inherits(goog.ui.TabBar, goog.ui.Container);
71goog.tagUnsealableClass(goog.ui.TabBar);

132/**
133 * Removes the tab from the tab bar. Overrides the superclass implementation
134 * by deselecting the tab being removed. Since {@link #removeChildAt} uses
135 * {@link #removeChild} internally, we only need to override this method.
136 * @param {string|goog.ui.Component} tab Tab to remove.
137 * @param {boolean=} opt_unrender Whether to call {@code exitDocument} on the
138 * removed tab, and detach its DOM from the document (defaults to false).
139 * @return {goog.ui.Control} The removed tab, if any.
140 * @override
141 */
142goog.ui.TabBar.prototype.removeChild = function(tab, opt_unrender) {
143 // This actually only accepts goog.ui.Controls. There's a TODO
144 // on the superclass method to fix this.
145 this.deselectIfSelected(/** @type {goog.ui.Control} */ (tab));
146 return goog.ui.TabBar.superClass_.removeChild.call(this, tab, opt_unrender);
147};

217/**
218 * Selects the given tab.
219 * @param {goog.ui.Control?} tab Tab to select (null to select none).
220 */
221goog.ui.TabBar.prototype.setSelectedTab = function(tab) {
222 if (tab) {
223 // Select the tab and have it dispatch a SELECT event, to be handled in
224 // handleTabSelect() below.
225 tab.setSelected(true);
226 } else if (this.getSelectedTab()) {
227 // De-select the currently selected tab and have it dispatch an UNSELECT
228 // event, to be handled in handleTabUnselect() below.
229 this.getSelectedTab().setSelected(false);
230 }
231};

251/**
252 * If the specified tab is the currently selected tab, deselects it, and
253 * selects the closest selectable tab in the tab bar (first looking before,
254 * then after the deselected tab). Does nothing if the argument is not the
255 * currently selected tab. Called internally when a tab is removed, hidden,
256 * or disabled, to ensure that another tab is selected instead.
257 * @param {goog.ui.Control?} tab Tab to deselect (if any).
258 * @protected
259 */
260goog.ui.TabBar.prototype.deselectIfSelected = function(tab) {
261 if (tab && tab == this.getSelectedTab()) {
262 var index = this.indexOfChild(tab);
263 // First look for the closest selectable tab before this one.
264 for (var i = index - 1;
265 tab = /** @type {goog.ui.Tab} */ (this.getChildAt(i));
266 i--) {
267 if (this.isSelectableTab(tab)) {
268 this.setSelectedTab(tab);
269 return;
270 }
271 }
272 // Next, look for the closest selectable tab after this one.
273 for (var j = index + 1;
274 tab = /** @type {goog.ui.Tab} */ (this.getChildAt(j));
275 j++) {
276 if (this.isSelectableTab(tab)) {
277 this.setSelectedTab(tab);
278 return;
279 }
280 }
281 // If all else fails, just set the selection to null.
282 this.setSelectedTab(null);
283 }
284};

299/**
300 * Handles {@code SELECT} events dispatched by tabs as they become selected.
301 * @param {goog.events.Event} e Select event to handle.
302 * @protected
303 */
304goog.ui.TabBar.prototype.handleTabSelect = function(e) {
305 if (this.selectedTab_ && this.selectedTab_ != e.target) {
306 // Deselect currently selected tab.
307 this.selectedTab_.setSelected(false);
308 }
309 this.selectedTab_ = /** @type {goog.ui.Tab} */ (e.target);
310};

318goog.ui.TabBar.prototype.handleTabUnselect = function(e) {
319 if (e.target == this.selectedTab_) {
320 this.selectedTab_ = null;

345/**
346 * Handles focus events dispatched by the tab bar's key event target. If no tab
347 * is currently highlighted, highlights the selected tab or the first tab if no

361/**
362 * Subscribes to events dispatched by tabs.
363 * @private
364 */
365goog.ui.TabBar.prototype.listenToTabEvents_ = function() {
366 // Listen for SELECT, UNSELECT, DISABLE, and HIDE events dispatched by tabs.
367 this.getHandler().
368 listen(this, goog.ui.Component.EventType.SELECT, this.handleTabSelect).
369 listen(this,
370 goog.ui.Component.EventType.UNSELECT,
371 this.handleTabUnselect).
372 listen(this, goog.ui.Component.EventType.DISABLE, this.handleTabDisable).
373 listen(this, goog.ui.Component.EventType.HIDE, this.handleTabHide);
374};

377/**
378 * Returns the {@link goog.ui.Container.Orientation} that is implied by the
379 * given {@link goog.ui.TabBar.Location}.
380 * @param {goog.ui.TabBar.Location} location Tab bar location.
381 * @return {goog.ui.Container.Orientation} Corresponding orientation.
382 */
383goog.ui.TabBar.getOrientationFromLocation = function(location) {
384 return location == goog.ui.TabBar.Location.START ||
385 location == goog.ui.TabBar.Location.END ?
386 goog.ui.Container.Orientation.VERTICAL :
387 goog.ui.Container.Orientation.HORIZONTAL;
