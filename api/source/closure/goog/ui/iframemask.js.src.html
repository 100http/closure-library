35/**
36 * Controller for an iframe mask. The mask is only valid in the current
37 * document, or else the document of the given DOM helper.
38 *
39 * @param {goog.dom.DomHelper=} opt_domHelper The DOM helper for the relevant
40 * document.
41 * @param {goog.structs.Pool=} opt_iframePool An optional source of iframes.
42 * Iframes will be grabbed from the pool when they're needed and returned
43 * to the pool (but still attached to the DOM) when they're done.
44 * @constructor
45 * @extends {goog.Disposable}
46 */
47goog.ui.IframeMask = function(opt_domHelper, opt_iframePool) {
48 goog.Disposable.call(this);
49
50 /**
51 * The DOM helper for this document.
52 * @type {goog.dom.DomHelper}
53 * @private
54 */
55 this.dom_ = opt_domHelper || goog.dom.getDomHelper();
56
57 /**
58 * An Element to snap the mask to. If none is given, defaults to
59 * a full-screen iframe mask.
60 * @type {Element}
61 * @private
62 */
63 this.snapElement_ = this.dom_.getDocument().documentElement;
64
65 /**
66 * An event handler for listening to popups and the like.
67 * @type {goog.events.EventHandler<!goog.ui.IframeMask>}
68 * @private
69 */
70 this.handler_ = new goog.events.EventHandler(this);
71
72 /**
73 * An iframe pool.
74 * @type {goog.structs.Pool|undefined}
75 * @private
76 */
77 this.iframePool_ = opt_iframePool;
78};
79goog.inherits(goog.ui.IframeMask, goog.Disposable);
80goog.tagUnsealableClass(goog.ui.IframeMask);

108/**
109 * Removes the iframe from the DOM.
110 * @override
111 * @protected
112 */
113goog.ui.IframeMask.prototype.disposeInternal = function() {
114 if (this.iframePool_) {
115 this.iframePool_.releaseObject(
116 /** @type {HTMLIFrameElement} */ (this.iframe_));
117 } else {
118 goog.dom.removeNode(this.iframe_);
119 }
120 this.iframe_ = null;
121
122 this.handler_.dispose();
123 this.handler_ = null;
124
125 goog.ui.IframeMask.superClass_.disposeInternal.call(this);
126};

129/**
130 * CSS for a hidden iframe.
131 * @type {string}
132 * @private
133 */
134goog.ui.IframeMask.HIDDEN_CSS_TEXT_ =
135 'position:absolute;display:none;z-index:1';

138/**
139 * Removes the mask from the screen.
140 */
141goog.ui.IframeMask.prototype.hideMask = function() {
142 if (this.iframe_) {
143 this.iframe_.style.cssText = goog.ui.IframeMask.HIDDEN_CSS_TEXT_;
144 if (this.iframePool_) {
145 this.iframePool_.releaseObject(this.iframe_);
146 this.iframe_ = null;
147 }
148 }
149};

152/**
153 * Gets the iframe to use as a mask. Creates a new one if one has not been
154 * created yet.
155 * @return {!HTMLIFrameElement} The iframe.
156 * @private
157 */
158goog.ui.IframeMask.prototype.getIframe_ = function() {
159 if (!this.iframe_) {
160 this.iframe_ = this.iframePool_ ?
161 /** @type {HTMLIFrameElement} */ (this.iframePool_.getObject()) :
162 goog.dom.iframe.createBlank(this.dom_);
163 this.iframe_.style.cssText = goog.ui.IframeMask.HIDDEN_CSS_TEXT_;
164 this.dom_.getDocument().body.appendChild(this.iframe_);
165 }
166 return this.iframe_;
167};

170/**
171 * Applies the iframe mask to the screen.
172 */
173goog.ui.IframeMask.prototype.applyMask = function() {
174 var iframe = this.getIframe_();
175 var bounds = goog.style.getBounds(this.snapElement_);
176 iframe.style.cssText =
177 'position:absolute;' +
178 'left:' + bounds.left + 'px;' +
179 'top:' + bounds.top + 'px;' +
180 'width:' + bounds.width + 'px;' +
181 'height:' + bounds.height + 'px;' +
182 'z-index:' + this.zIndex_;
183 goog.style.setOpacity(iframe, this.opacity_);
184 iframe.style.display = 'block';
185};

209/**
210 * Sets the element to use as the bounds of the mask. Takes effect immediately.
211 * @param {Element} snapElement The snap element, which the iframe will be
212 * "snapped" around.
213 */
214goog.ui.IframeMask.prototype.setSnapElement = function(snapElement) {
215 this.snapElement_ = snapElement;
216 if (this.iframe_ && goog.style.isElementShown(this.iframe_)) {
217 this.applyMask();
218 }

222/**
223 * Listens on the specified target, hiding and showing the iframe mask
224 * when the given event types are dispatched.
225 * @param {goog.events.EventTarget} target The event target to listen on.
226 * @param {string} showEvent When this event fires, the mask will be applied.
227 * @param {string} hideEvent When this event fires, the mask will be hidden.
228 * @param {Element=} opt_snapElement When the mask is applied, it will
229 * automatically snap to this element. If no element is specified, it will
230 * use the default snap element.
231 */
232goog.ui.IframeMask.prototype.listenOnTarget = function(target, showEvent,
233 hideEvent, opt_snapElement) {
234 var timerKey;
235 this.handler_.listen(target, showEvent, function() {
236 if (opt_snapElement) {
237 this.setSnapElement(opt_snapElement);
238 }
239 // Check out the iframe asynchronously, so we don't block the SHOW
240 // event and cause a bounce.
241 timerKey = goog.Timer.callOnce(this.applyMask, 0, this);
242 });
243 this.handler_.listen(target, hideEvent, function() {
244 if (timerKey) {
245 goog.Timer.clear(timerKey);
246 timerKey = null;
247 }
248 this.hideMask();
249 });
250};

253/**
254 * Removes all handlers attached by listenOnTarget.
255 */
256goog.ui.IframeMask.prototype.removeHandlers = function() {
257 this.handler_.removeAll();
