50/**
51 * Tooltip widget. Can be attached to one or more elements and is shown, with a
52 * slight delay, when the the cursor is over the element or the element gains
53 * focus.
54 *
55 * @param {Element|string=} opt_el Element to display tooltip for, either
56 * element reference or string id.
57 * @param {?string=} opt_str Text message to display in tooltip.
58 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper.
59 * @constructor
60 * @extends {goog.ui.Popup}
61 */
62goog.ui.Tooltip = function(opt_el, opt_str, opt_domHelper) {
63 /**
64 * Dom Helper
65 * @type {goog.dom.DomHelper}
66 * @private
67 */
68 this.dom_ = opt_domHelper || (opt_el ?
69 goog.dom.getDomHelper(goog.dom.getElement(opt_el)) :
70 goog.dom.getDomHelper());
71
72 goog.ui.Popup.call(this, this.dom_.createDom(
73 goog.dom.TagName.DIV, {'style': 'position:absolute;display:none;'}));
74
75 /**
76 * Cursor position relative to the page.
77 * @type {!goog.math.Coordinate}
78 * @protected
79 */
80 this.cursorPosition = new goog.math.Coordinate(1, 1);
81
82 /**
83 * Elements this widget is attached to.
84 * @type {goog.structs.Set}
85 * @private
86 */
87 this.elements_ = new goog.structs.Set();
88
89 // Attach to element, if specified
90 if (opt_el) {
91 this.attach(opt_el);
92 }
93
94 // Set message, if specified.
95 if (opt_str != null) {
96 this.setText(opt_str);
97 }
98};
99goog.inherits(goog.ui.Tooltip, goog.ui.Popup);
100goog.tagUnsealableClass(goog.ui.Tooltip);

257/**
258 * Attach to element. Tooltip will be displayed when the cursor is over the
259 * element or when the element has been active for a few milliseconds.
260 *
261 * @param {Element|string} el Element to display tooltip for, either element

271 this.handleMouseOutAndBlur, false, this);
272 goog.events.listen(el, goog.events.EventType.MOUSEMOVE,
273 this.handleMouseMove, false, this);
274 goog.events.listen(el, goog.events.EventType.FOCUS,
275 this.handleFocus, false, this);
276 goog.events.listen(el, goog.events.EventType.BLUR,
277 this.handleMouseOutAndBlur, false, this);
278};

281/**

283 *
284 * @param {Element|string=} opt_el Element to detach from, either element
285 * reference or string id. If no element is
286 * specified all are detached.

288goog.ui.Tooltip.prototype.detach = function(opt_el) {
289 if (opt_el) {

291 this.detachElement_(el);

293 } else {
294 var a = this.elements_.getValues();

455/**
456 * Returns true if the coord is in the tooltip.
457 * @param {goog.math.Coordinate} coord Coordinate being tested.
458 * @return {boolean} Whether the coord is in the tooltip.
459 */
460goog.ui.Tooltip.prototype.isCoordinateInTooltip = function(coord) {
461 // Check if coord is inside the the tooltip
462 if (!this.isVisible()) {
463 return false;
464 }
465
466 var offset = goog.style.getPageOffset(this.getElement());
467 var size = goog.style.getSize(this.getElement());
468 return offset.x <= coord.x && coord.x <= offset.x + size.width &&
469 offset.y <= coord.y && coord.y <= offset.y + size.height;
470};

473/**
474 * Called before the popup is shown.
475 *
476 * @return {boolean} Whether tooltip should be shown.
477 * @protected
478 * @override
479 */
480goog.ui.Tooltip.prototype.onBeforeShow = function() {
481 if (!goog.ui.PopupBase.prototype.onBeforeShow.call(this)) {
482 return false;
483 }
484
485 // Hide all open tooltips except if this tooltip is triggered by an element
486 // inside another tooltip.
487 if (this.anchor) {
488 for (var tt, i = 0; tt = goog.ui.Tooltip.activeInstances_[i]; i++) {
489 if (!goog.dom.contains(tt.getElement(), this.anchor)) {
490 tt.setVisible(false);
491 }
492 }
493 }
494 goog.array.insert(goog.ui.Tooltip.activeInstances_, this);
495
496 var element = this.getElement();
497 element.className = this.className;
498 this.clearHideTimer();
499
500 // Register event handlers for tooltip. Used to prevent the tooltip from
501 // closing if the cursor is over the tooltip rather then the element that
502 // triggered it.
503 goog.events.listen(element, goog.events.EventType.MOUSEOVER,
504 this.handleTooltipMouseOver, false, this);
505 goog.events.listen(element, goog.events.EventType.MOUSEOUT,
506 this.handleTooltipMouseOut, false, this);
507
508 this.clearShowTimer();
509 return true;
510};

513/**
514 * Called after the popup is hidden.
515 *
516 * @protected
517 * @suppress {underscore|visibility}
518 * @override
519 */
520goog.ui.Tooltip.prototype.onHide_ = function() {
521 goog.array.remove(goog.ui.Tooltip.activeInstances_, this);
522
523 // Hide all open tooltips triggered by an element inside this tooltip.
524 var element = this.getElement();
525 for (var tt, i = 0; tt = goog.ui.Tooltip.activeInstances_[i]; i++) {
526 if (tt.anchor && goog.dom.contains(element, tt.anchor)) {
527 tt.setVisible(false);
528 }
529 }
530
531 // If this tooltip is inside another tooltip, start hide timer for that
532 // tooltip in case this tooltip was the only reason it was still showing.
533 if (this.parentTooltip_) {
534 this.parentTooltip_.startHideTimer();
535 }
536
537 goog.events.unlisten(element, goog.events.EventType.MOUSEOVER,
538 this.handleTooltipMouseOver, false, this);
539 goog.events.unlisten(element, goog.events.EventType.MOUSEOUT,
540 this.handleTooltipMouseOut, false, this);
541
542 this.anchor = undefined;
543 // If we are still waiting to show a different hovercard, don't abort it
544 // because you think you haven't seen a mouse move:
545 if (this.getState() == goog.ui.Tooltip.State.INACTIVE) {
546 this.seenInteraction_ = false;
547 }
548
549 goog.ui.PopupBase.prototype.onHide_.call(this);
550};

553/**
554 * Called by timer from mouse over handler. Shows tooltip if cursor is still
555 * over the same element.
556 *
557 * @param {Element} el Element to show tooltip for.
558 * @param {goog.positioning.AbstractPosition=} opt_pos Position to display popup
559 * at.
560 */
561goog.ui.Tooltip.prototype.maybeShow = function(el, opt_pos) {
562 // Assert that the mouse is still over the same element, and that we have not
563 // detached from the anchor in the meantime.
564 if (this.anchor == el && this.elements_.contains(this.anchor)) {
565 if (this.seenInteraction_ || !this.requireInteraction_) {
566 // If it is currently showing, then hide it, and abort if it doesn't hide.
567 this.setVisible(false);
568 if (!this.isVisible()) {
569 this.positionAndShow_(el, opt_pos);
570 }
571 } else {
572 this.anchor = undefined;
573 }
574 }
575 this.showTimer = undefined;
576};

676/**
677 * Handler for mouse over events.
678 *
679 * @param {goog.events.BrowserEvent} event Event object.
680 * @protected
681 */
682goog.ui.Tooltip.prototype.handleMouseOver = function(event) {
683 var el = this.getAnchorFromElement(/** @type {Element} */ (event.target));
684 this.activeEl_ = /** @type {Element} */ (el);
685 this.clearHideTimer();
686 if (el != this.anchor) {
687 this.anchor = el;
688 this.startShowTimer(/** @type {Element} */ (el));
689 this.checkForParentTooltip_();
690 this.saveCursorPosition_(event);
691 }

695/**
696 * Find anchor containing the given element, if any.
697 *
698 * @param {Element} el Element that triggered event.
699 * @return {Element} Element in elements_ array that contains given element,
700 * or null if not found.
701 * @protected
702 */
703goog.ui.Tooltip.prototype.getAnchorFromElement = function(el) {
704 // FireFox has a bug where mouse events relating to <input> elements are
705 // sometimes duplicated (often in FF2, rarely in FF3): once for the
706 // <input> element and once for a magic hidden <div> element. Javascript
707 // code does not have sufficient permissions to read properties on that
708 // magic element and thus will throw an error in this call to
709 // getAnchorFromElement_(). In that case we swallow the error.
710 // See https://bugzilla.mozilla.org/show_bug.cgi?id=330961
711 try {
712 while (el && !this.elements_.contains(el)) {
713 el = /** @type {Element} */ (el.parentNode);
714 }
715 return el;
716 } catch (e) {
717 return null;
718 }
719};

775/**
776 * Looks for an active tooltip whose element contains this tooltip's anchor.
777 * This allows us to prevent hides until they are really necessary.
778 *
779 * @private
780 */
781goog.ui.Tooltip.prototype.checkForParentTooltip_ = function() {
782 if (this.anchor) {
783 for (var tt, i = 0; tt = goog.ui.Tooltip.activeInstances_[i]; i++) {
784 if (goog.dom.contains(tt.getElement(), this.anchor)) {
785 tt.childTooltip_ = this;
