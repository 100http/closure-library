39/**
40 * The PopupBase class provides functionality for showing and hiding a generic
41 * container element. It also provides the option for hiding the popup element
42 * if the user clicks outside the popup or the popup loses focus.
43 *
44 * @constructor
45 * @extends {goog.events.EventTarget}
46 * @param {Element=} opt_element A DOM element for the popup.
47 * @param {goog.ui.PopupBase.Type=} opt_type Type of popup.
48 */
49goog.ui.PopupBase = function(opt_element, opt_type) {
50 goog.events.EventTarget.call(this);
51
52 /**
53 * An event handler to manage the events easily
54 * @type {goog.events.EventHandler<!goog.ui.PopupBase>}
55 * @private
56 */
57 this.handler_ = new goog.events.EventHandler(this);
58
59 this.setElement(opt_element || null);
60 if (opt_type) {
61 this.setType(opt_type);
62 }
63};
64goog.inherits(goog.ui.PopupBase, goog.events.EventTarget);
65goog.tagUnsealableClass(goog.ui.PopupBase);

202/**
203 * A time in ms used to debounce events that happen right after each other.
204 *
205 * A note about why this is necessary. There are two cases to consider.
206 * First case, a popup will usually see a focus event right after it's launched
207 * because it's typical for it to be launched in a mouse-down event which will
208 * then move focus to the launching button. We don't want to think this is a
209 * separate user action moving focus. Second case, a user clicks on the
210 * launcher button to close the menu. In that case, we'll close the menu in the
211 * focus event and then show it again because of the mouse down event, even
212 * though the intention is to just close the menu. This workaround appears to
213 * be the least intrusive fix.
214 *
215 * @type {number}
216 */
217goog.ui.PopupBase.DEBOUNCE_DELAY_MS = 150;

456/**
457 * Returns whether the popup is currently visible or was visible within about
458 * 150 ms ago. This is used by clients to handle a very specific, but common,
459 * popup scenario. The button that launches the popup should close the popup
460 * on mouse down if the popup is alrady open. The problem is that the popup
461 * closes itself during the capture phase of the mouse down and thus the button
462 * thinks it's hidden and this should show it again. This method provides a
463 * good heuristic for clients. Typically in their event handler they will have
464 * code that is:
465 *
466 * if (menu.isOrWasRecentlyVisible()) {
467 * menu.setVisible(false);
468 * } else {
469 * ... // code to position menu and initialize other state
470 * menu.setVisible(true);
471 * }
472 * @return {boolean} Whether the popup is currently visible or was visible
473 * within about 150 ms ago.
474 */
475goog.ui.PopupBase.prototype.isOrWasRecentlyVisible = function() {
476 return this.isVisible_ ||
477 (goog.now() - this.lastHideTime_ <
478 goog.ui.PopupBase.DEBOUNCE_DELAY_MS);
479};

509/**
510 * Does the work to show the popup.
511 * @private
512 */
513goog.ui.PopupBase.prototype.show_ = function() {
514 // Ignore call if we are already showing.
515 if (this.isVisible_) {
516 return;
517 }
518
519 // Give derived classes and handlers a chance to customize popup.
520 if (!this.onBeforeShow()) {
521 return;
522 }
523
524 // Allow callers to set the element in the BEFORE_SHOW event.
525 if (!this.element_) {
526 throw Error('Caller must call setElement before trying to show the popup');
527 }
528
529 // Call reposition after onBeforeShow, as it may change the style and/or
530 // content of the popup and thereby affecting the size which is used for the
531 // viewport calculation.
532 this.reposition();
533
534 var doc = goog.dom.getOwnerDocument(this.element_);
535
536 if (this.hideOnEscape_) {
537
538 // Handle the escape keys. Listen in the capture phase so that we can
539 // stop the escape key from propagating to other elements. For example,
540 // if there is a popup within a dialog box, we want the popup to be
541 // dismissed first, rather than the dialog.
542 this.handler_.listen(doc, goog.events.EventType.KEYDOWN,
543 this.onDocumentKeyDown_, true);
544 }
545
546 // Set up event handlers.
547 if (this.autoHide_) {
548
549 // Even if the popup is not in the focused document, we want to
550 // close it on mousedowns in the document it's in.
551 this.handler_.listen(doc, goog.events.EventType.MOUSEDOWN,
552 this.onDocumentMouseDown_, true);
553
554 if (goog.userAgent.IE) {
555 // We want to know about deactivates/mousedowns on the document with focus
556 // The top-level document won't get a deactivate event if the focus is
557 // in an iframe and the deactivate fires within that iframe.
558 // The active element in the top-level document will remain the iframe
559 // itself.
560 var activeElement;
561 /** @preserveTry */
562 try {
563 activeElement = doc.activeElement;
564 } catch (e) {
565 // There is an IE browser bug which can cause just the reading of
566 // document.activeElement to throw an Unspecified Error. This
567 // may have to do with loading a popup within a hidden iframe.
568 }
569 while (activeElement &&
570 activeElement.nodeName == goog.dom.TagName.IFRAME) {
571 /** @preserveTry */
572 try {
573 var tempDoc = goog.dom.getFrameContentDocument(activeElement);
574 } catch (e) {
575 // The frame is on a different domain that its parent document
576 // This way, we grab the lowest-level document object we can get
577 // a handle on given cross-domain security.
578 break;
579 }
580 doc = tempDoc;
581 activeElement = doc.activeElement;
582 }
583
584 // Handle mousedowns in the focused document in case the user clicks
585 // on the activeElement (in which case the popup should hide).
586 this.handler_.listen(doc, goog.events.EventType.MOUSEDOWN,
587 this.onDocumentMouseDown_, true);
588
589 // If the active element inside the focused document changes, then
590 // we probably need to hide the popup.
591 this.handler_.listen(doc, goog.events.EventType.DEACTIVATE,
592 this.onDocumentBlur_);
593
594 } else {
595 this.handler_.listen(doc, goog.events.EventType.BLUR,
596 this.onDocumentBlur_);
597 }
598 }
599
600 // Make the popup visible.
601 if (this.type_ == goog.ui.PopupBase.Type.TOGGLE_DISPLAY) {
602 this.showPopupElement();
603 } else if (this.type_ == goog.ui.PopupBase.Type.MOVE_OFFSCREEN) {
604 this.reposition();
605 }
606 this.isVisible_ = true;
607
608 this.lastShowTime_ = goog.now();
609 this.lastHideTime_ = -1;
610
611 // If there is transition to play, we play it and fire SHOW event after
612 // the transition is over.
613 if (this.showTransition_) {
614 goog.events.listenOnce(
615 /** @type {!goog.events.EventTarget} */ (this.showTransition_),
616 goog.fx.Transition.EventType.END, this.onShow_, false, this);
617 this.showTransition_.play();
618 } else {
619 // Notify derived classes and handlers.
620 this.onShow_();
621 }
622};

625/**
626 * Hides the popup. This call is idempotent.
627 *
628 * @param {Object=} opt_target Target of the event causing the hide.
629 * @return {boolean} Whether the popup was hidden and not cancelled.
630 * @private
631 */
632goog.ui.PopupBase.prototype.hide_ = function(opt_target) {
633 // Give derived classes and handlers a chance to cancel hiding.
634 if (!this.isVisible_ || !this.onBeforeHide_(opt_target)) {
635 return false;
636 }
637
638 // Remove any listeners we attached when showing the popup.
639 if (this.handler_) {
640 this.handler_.removeAll();
641 }
642
643 // Set visibility to hidden even if there is a transition.
644 this.isVisible_ = false;
645 this.lastHideTime_ = goog.now();
646
647 // If there is transition to play, we play it and only hide the element
648 // (and fire HIDE event) after the transition is over.
649 if (this.hideTransition_) {
650 goog.events.listenOnce(
651 /** @type {!goog.events.EventTarget} */ (this.hideTransition_),
652 goog.fx.Transition.EventType.END,
653 goog.partial(this.continueHidingPopup_, opt_target), false, this);
654 this.hideTransition_.play();
655 } else {
656 this.continueHidingPopup_(opt_target);
657 }
658
659 return true;
660};
661
662
663/**
664 * Continues hiding the popup. This is a continuation from hide_. It is
665 * a separate method so that we can add a transition before hiding.
666 * @param {Object=} opt_target Target of the event causing the hide.
667 * @private
668 */
669goog.ui.PopupBase.prototype.continueHidingPopup_ = function(opt_target) {
670 // Hide the popup.
671 if (this.type_ == goog.ui.PopupBase.Type.TOGGLE_DISPLAY) {
672 if (this.shouldHideAsync_) {
673 goog.Timer.callOnce(this.hidePopupElement, 0, this);
674 } else {
675 this.hidePopupElement();
676 }
677 } else if (this.type_ == goog.ui.PopupBase.Type.MOVE_OFFSCREEN) {
678 this.moveOffscreen_();
679 }
680
681 // Notify derived classes and handlers.
682 this.onHide_(opt_target);
683};

773/**
774 * Mouse down handler for the document on capture phase. Used to hide the
775 * popup for auto-hide mode.
776 *
777 * @param {goog.events.BrowserEvent} e The event object.
778 * @private
779 */
780goog.ui.PopupBase.prototype.onDocumentMouseDown_ = function(e) {
781 var target = e.target;
782
783 if (!goog.dom.contains(this.element_, target) &&
784 !this.isOrWithinAutoHidePartner_(target) &&
785 this.isWithinAutoHideRegion_(target) &&
786 !this.shouldDebounce_()) {
787
788 // Mouse click was outside popup and partners, so hide.
789 this.hide_(target);
790 }
791};

794/**
795 * Handles key-downs on the document to handle the escape key.
796 *
797 * @param {goog.events.BrowserEvent} e The event object.
798 * @private
799 */
800goog.ui.PopupBase.prototype.onDocumentKeyDown_ = function(e) {
801 if (e.keyCode == goog.events.KeyCodes.ESC) {
802 if (this.hide_(e.target)) {
803 // Eat the escape key, but only if this popup was actually closed.
804 e.preventDefault();

811/**
812 * Deactivate handler(IE) and blur handler (other browsers) for document.
813 * Used to hide the popup for auto-hide mode.
814 *
815 * @param {goog.events.BrowserEvent} e The event object.
816 * @private
817 */
818goog.ui.PopupBase.prototype.onDocumentBlur_ = function(e) {
819 if (!this.enableCrossIframeDismissal_) {
820 return;
821 }
822
823 var doc = goog.dom.getOwnerDocument(this.element_);
824
825 // Ignore blur events if the active element is still inside the popup or if
826 // there is no longer an active element. For example, a widget like a
827 // goog.ui.Button might programatically blur itself before losing tabIndex.
828 if (typeof document.activeElement != 'undefined') {
829 var activeElement = doc.activeElement;
830 if (!activeElement || goog.dom.contains(this.element_,
831 activeElement) || activeElement.tagName == goog.dom.TagName.BODY) {
832 return;
833 }
834
835 // Ignore blur events not for the document itself in non-IE browsers.
836 } else if (e.target != doc) {
837 return;
838 }
839
840 // Debounce the initial focus move.
841 if (this.shouldDebounce_()) {
842 return;
843 }
844
845 this.hide_();
846};
