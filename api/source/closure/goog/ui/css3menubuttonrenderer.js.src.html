15/**
16 * @fileoverview An alternative imageless button renderer that uses CSS3 rather
17 * than voodoo to render custom buttons with rounded corners and dimensionality
18 * (via a subtle flat shadow on the bottom half of the button) without the use
19 * of images.
20 *
21 * Based on the Custom Buttons 3.1 visual specification, see
22 * http://go/custombuttons
23 *
24 * Tested and verified to work in Gecko 1.9.2+ and WebKit 528+.
25 *
26 * @author eae@google.com (Emil A Eklund)
27 * @see ../demos/css3menubutton.html
28 */

41/**
42 * Custom renderer for {@link goog.ui.MenuButton}s. Css3 buttons can contain
43 * almost arbitrary HTML content, will flow like inline elements, but can be
44 * styled like block-level elements.
45 *
46 * @constructor
47 * @extends {goog.ui.MenuButtonRenderer}
48 * @final
49 */
50goog.ui.Css3MenuButtonRenderer = function() {
51 goog.ui.MenuButtonRenderer.call(this);
52};
53goog.inherits(goog.ui.Css3MenuButtonRenderer, goog.ui.MenuButtonRenderer);

65/**
66 * Default CSS class to be applied to the root element of components rendered
67 * by this renderer.
68 * @type {string}

70goog.ui.Css3MenuButtonRenderer.CSS_CLASS = goog.getCssName('goog-css3-button');

73/** @override */
74goog.ui.Css3MenuButtonRenderer.prototype.getContentElement = function(element) {
75 if (element) {
76 var captionElem = goog.dom.getElementsByTagNameAndClass(
77 '*', goog.getCssName(this.getCssClass(), 'caption'), element)[0];
78 return captionElem;
79 }
80 return null;
81};

84/**
85 * Returns true if this renderer can decorate the element. Overrides
86 * {@link goog.ui.MenuButtonRenderer#canDecorate} by returning true if the
87 * element is a DIV, false otherwise.
88 * @param {Element} element Element to decorate.
89 * @return {boolean} Whether the renderer can decorate the element.
90 * @override
91 */
92goog.ui.Css3MenuButtonRenderer.prototype.canDecorate = function(element) {
93 return element.tagName == goog.dom.TagName.DIV;
94};

97/**
98 * Takes a text caption or existing DOM structure, and returns the content
99 * wrapped in a pseudo-rounded-corner box. Creates the following DOM structure:
100 * <div class="goog-inline-block goog-css3-button goog-css3-menu-button">
101 * <div class="goog-css3-button-caption">Contents...</div>
102 * <div class="goog-css3-button-dropdown"></div>
103 * </div>
104 *
105 * Used by both {@link #createDom} and {@link #decorate}. To be overridden
106 * by subclasses.
107 * @param {goog.ui.ControlContent} content Text caption or DOM structure to wrap
108 * in a box.
109 * @param {goog.dom.DomHelper} dom DOM helper, used for document interaction.
110 * @return {!Element} Pseudo-rounded-corner box containing the content.
111 * @override
112 */
113goog.ui.Css3MenuButtonRenderer.prototype.createButton = function(content, dom) {
114 var baseClass = this.getCssClass();
115 var inlineBlock = goog.ui.INLINE_BLOCK_CLASSNAME + ' ';
116 return dom.createDom(goog.dom.TagName.DIV, inlineBlock,
117 dom.createDom(goog.dom.TagName.DIV, [
118 goog.getCssName(baseClass, 'caption'),
119 goog.getCssName('goog-inline-block')
120 ],
121 content),
122 dom.createDom(goog.dom.TagName.DIV, [
123 goog.getCssName(baseClass, 'dropdown'),
124 goog.getCssName('goog-inline-block')]));
125};

139// Register a decorator factory function for goog.ui.Css3MenuButtonRenderer.
140// Since we're using goog-css3-button as the base class in order to get the
141// same styling as goog.ui.Css3ButtonRenderer, we need to be explicit about
142// giving goog-css3-menu-button here.
143goog.ui.registry.setDecoratorByClassName(
144 goog.getCssName('goog-css3-menu-button'),
145 function() {
146 return new goog.ui.MenuButton(null, null,
147 goog.ui.Css3MenuButtonRenderer.getInstance());
148 });
