15/**
16 * @fileoverview A class for representing a separator, with renderers for both

33/**
34 * Class representing a separator. Although it extends {@link goog.ui.Control},
35 * the Separator class doesn't allocate any event handlers, nor does it change
36 * its appearance on mouseover, etc.
37 * @param {goog.ui.MenuSeparatorRenderer=} opt_renderer Renderer to render or
38 * decorate the separator; defaults to {@link goog.ui.MenuSeparatorRenderer}.
39 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper, used for
40 * document interaction.
41 * @constructor
42 * @extends {goog.ui.Control}
43 */
44goog.ui.Separator = function(opt_renderer, opt_domHelper) {
45 goog.ui.Control.call(this, null, opt_renderer ||
46 goog.ui.MenuSeparatorRenderer.getInstance(), opt_domHelper);
47
48 this.setSupportedState(goog.ui.Component.State.DISABLED, false);
49 this.setSupportedState(goog.ui.Component.State.HOVER, false);
50 this.setSupportedState(goog.ui.Component.State.ACTIVE, false);
51 this.setSupportedState(goog.ui.Component.State.FOCUSED, false);
52
53 // Separators are always considered disabled.
54 this.setStateInternal(goog.ui.Component.State.DISABLED);
55};
56goog.inherits(goog.ui.Separator, goog.ui.Control);
57
58
59/**
60 * Configures the component after its DOM has been rendered. Overrides
61 * {@link goog.ui.Control#enterDocument} by making sure no event handler
62 * is allocated.
63 * @override
64 */
65goog.ui.Separator.prototype.enterDocument = function() {
66 goog.ui.Separator.superClass_.enterDocument.call(this);
67 var element = this.getElement();
68 goog.asserts.assert(element,
69 'The DOM element for the separator cannot be null.');
70 goog.a11y.aria.setRole(element, 'separator');
71};

74// Register a decorator factory function for goog.ui.MenuSeparators.
75goog.ui.registry.setDecoratorByClassName(
76 goog.ui.MenuSeparatorRenderer.CSS_CLASS,
77 function() {
78 // Separator defaults to using MenuSeparatorRenderer.
79 return new goog.ui.Separator();
