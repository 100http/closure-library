48/**
49 * DatePicker widget. Allows a single date to be selected from a calendar like
50 * view.
51 *
52 * @param {goog.date.Date|Date=} opt_date Date to initialize the date picker
53 * with, defaults to the current date.
54 * @param {Object=} opt_dateTimeSymbols Date and time symbols to use.
55 * Defaults to goog.i18n.DateTimeSymbols if not set.
56 * @param {goog.dom.DomHelper=} opt_domHelper Optional DOM helper.
57 * @param {goog.ui.DatePickerRenderer=} opt_renderer Optional Date picker
58 * renderer.
59 * @constructor
60 * @extends {goog.ui.Component}
61 */
62goog.ui.DatePicker = function(opt_date, opt_dateTimeSymbols, opt_domHelper,
63 opt_renderer) {
64 goog.ui.Component.call(this, opt_domHelper);
65
66 /**
67 * Date and time symbols to use.
68 * @type {Object}
69 * @private
70 */
71 this.symbols_ = opt_dateTimeSymbols || goog.i18n.DateTimeSymbols;
72
73 this.wdayNames_ = this.symbols_.STANDALONESHORTWEEKDAYS;
74
75 // The DateTimeFormat object uses the global goog.i18n.DateTimeSymbols
76 // for initialization. So we save the original value, the global object,
77 // create the formatters, then restore the original value.
78 var tempSymbols = goog.i18n.DateTimeSymbols; // save
79 goog.i18n.DateTimeSymbols = this.symbols_;
80
81 // Formatters for the various areas of the picker
82 this.i18nDateFormatterDay_ = new goog.i18n.DateTimeFormat('d');
83 this.i18nDateFormatterDay2_ = new goog.i18n.DateTimeFormat('dd');
84 this.i18nDateFormatterWeek_ = new goog.i18n.DateTimeFormat('w');
85
86 // Previous implementation did not use goog.i18n.DateTimePatterns,
87 // so it is likely most developers did not set it.
88 // This is why the fallback to a hard-coded string (just in case).
89 var patYear = goog.i18n.DateTimePatterns.YEAR_FULL || 'y';
90 this.i18nDateFormatterYear_ = new goog.i18n.DateTimeFormat(patYear);
91 var patMMMMy = goog.i18n.DateTimePatterns.YEAR_MONTH_FULL || 'MMMM y';
92 this.i18nDateFormatterMonthYear_ = new goog.i18n.DateTimeFormat(patMMMMy);
93
94 goog.i18n.DateTimeSymbols = tempSymbols; // restore
95
96 /**
97 * @type {!goog.ui.DatePickerRenderer}
98 * @private
99 */
100 this.renderer_ = opt_renderer || new goog.ui.DefaultDatePickerRenderer(
101 this.getBaseCssClass(), this.getDomHelper());
102
103 /**
104 * Selected date.
105 * @type {goog.date.Date}
106 * @private
107 */
108 this.date_ = new goog.date.Date(opt_date);
109 this.date_.setFirstWeekCutOffDay(this.symbols_.FIRSTWEEKCUTOFFDAY);
110 this.date_.setFirstDayOfWeek(this.symbols_.FIRSTDAYOFWEEK);
111
112 /**
113 * Active month.
114 * @type {goog.date.Date}
115 * @private
116 */
117 this.activeMonth_ = this.date_.clone();
118 this.activeMonth_.setDate(1);
119
120 /**
121 * Class names to apply to the weekday columns.
122 * @type {Array<string>}
123 * @private
124 */
125 this.wdayStyles_ = ['', '', '', '', '', '', ''];
126 this.wdayStyles_[this.symbols_.WEEKENDRANGE[0]] =
127 goog.getCssName(this.getBaseCssClass(), 'wkend-start');
128 this.wdayStyles_[this.symbols_.WEEKENDRANGE[1]] =
129 goog.getCssName(this.getBaseCssClass(), 'wkend-end');
130
131 /**
132 * Object that is being used to cache key handlers.
133 * @type {Object}
134 * @private
135 */
136 this.keyHandlers_ = {};
137
138 /**
139 * Collection of dates that make up the date picker.
140 * @type {!Array<!Array<!goog.date.Date>>}
141 * @private
142 */
143 this.grid_ = [];
144
145 /** @private {Array<!Array<Element>>} */
146 this.elTable_;
147
148 /**
149 * TODO(tbreisacher): Remove external references to this field,
150 * and make it private.
151 * @type {Element}
152 */
153 this.tableBody_;
154
155 /** @private {Element} */
156 this.tableFoot_;
157
158 /** @private {Element} */
159 this.elYear_;
160
161 /** @private {Element} */
162 this.elMonth_;
163
164 /** @private {Element} */
165 this.elToday_;
166
167 /** @private {Element} */
168 this.elNone_;
169
170 /** @private {Element} */
171 this.menu_;
172
173 /** @private {Element} */
174 this.menuSelected_;
175
176 /** @private {function(Element)} */
177 this.menuCallback_;
178};

726/**
727 * Returns a date element given a row and column. In elTable_, the elements that
728 * represent dates are 1 indexed because of other elements such as headers.
729 * This corrects for the offset and makes the API 0 indexed.
730 *
731 * @param {number} row The row in the element table.
732 * @param {number} col The column in the element table.
733 * @return {Element} The element in the grid or null if there is none.
734 * @protected
735 */
736goog.ui.DatePicker.prototype.getDateElementAt = function(row, col) {
737 if (row < 0 || col < 0) {
738 return null;
739 }

741 return this.elTable_[adjustedRow] ?
742 this.elTable_[adjustedRow][col + 1] || null : null;
743};

746/**
747 * Sets the selected date.
748 *
749 * @param {goog.date.Date|Date} date Date to select or null to select nothing.
750 */
751goog.ui.DatePicker.prototype.setDate = function(date) {
752 // Check if the month has been changed.
753 var sameMonth = date == this.date_ || date && this.date_ &&
754 date.getFullYear() == this.date_.getFullYear() &&
755 date.getMonth() == this.date_.getMonth();
756
757 // Check if the date has been changed.
758 var sameDate = date == this.date_ || sameMonth &&
759 date.getDate() == this.date_.getDate();
760
761 // Set current date to clone of supplied goog.date.Date or Date.
762 this.date_ = date && new goog.date.Date(date);
763
764 // Set current month
765 if (date) {
766 this.activeMonth_.set(this.date_);
767 this.activeMonth_.setDate(1);
768 }
769
770 // Update calendar grid even if the date has not changed as even if today is
771 // selected another month can be displayed.
772 this.updateCalendarGrid_();
773
774 // TODO(eae): Standardize selection and change events with other components.
775 // Fire select event.
776 var selectEvent = new goog.ui.DatePickerEvent(
777 goog.ui.DatePicker.Events.SELECT, this, this.date_);
778 this.dispatchEvent(selectEvent);
779
780 // Fire change event.
781 if (!sameDate) {
782 var changeEvent = new goog.ui.DatePickerEvent(
783 goog.ui.DatePicker.Events.CHANGE, this, this.date_);
784 this.dispatchEvent(changeEvent);
785 }
786
787 // Fire change active month event.
788 if (!sameMonth) {
789 this.fireChangeActiveMonthEvent_();
790 }
791};

794/**
795 * Updates the navigation row (navigating months and maybe years) in the navRow_
796 * element of a created picker.
797 * @private
798 */
799goog.ui.DatePicker.prototype.updateNavigationRow_ = function() {
800 if (!this.elNavRow_) {
801 return;
802 }
803 var row = this.elNavRow_;
804
805 // Clear the navigation row.
806 while (row.firstChild) {
807 row.removeChild(row.firstChild);
808 }

1051 * Keypress handler for date grid.
1052 *
1053 * @param {goog.events.BrowserEvent} event Keypress event.
1054 * @private
1055 */
1056goog.ui.DatePicker.prototype.handleGridKeyPress_ = function(event) {
1057 var months, days;
1058 switch (event.keyCode) {
1059 case 33: // Page up
1060 event.preventDefault();
1061 months = -1;
1062 break;
1063 case 34: // Page down
1064 event.preventDefault();
1065 months = 1;
1066 break;
1067 case 37: // Left
1068 event.preventDefault();
1069 days = -1;
1070 break;
1071 case 39: // Right
1072 event.preventDefault();
1073 days = 1;
1074 break;
1075 case 38: // Down
1076 event.preventDefault();
1077 days = -7;
1078 break;
1079 case 40: // Up
1080 event.preventDefault();
1081 days = 7;
1082 break;
1083 case 36: // Home
1084 event.preventDefault();
1085 this.selectToday();
1086 case 46: // Delete
1087 event.preventDefault();
1088 this.selectNone();
1089 default:
1090 return;
1091 }
1092 var date;
1093 if (this.date_) {
1094 date = this.date_.clone();
1095 date.add(new goog.date.Interval(0, months, days));
1096 } else {
1097 date = this.activeMonth_.clone();
1098 date.setDate(1);
1099 }
1100 if (this.isUserSelectableDate_(date)) {
1101 this.setDate(date);

1315/**
1316 * Determines the dates/weekdays for the current month and builds an in memory
1317 * representation of the calendar.
1318 *
1319 * @private
1320 */
1321goog.ui.DatePicker.prototype.updateCalendarGrid_ = function() {
1322 if (!this.getElement()) {
1323 return;
1324 }
1325
1326 var date = this.activeMonth_.clone();
1327 date.setDate(1);
1328
1329 // Show year name of select month
1330 if (this.elMonthYear_) {
1331 goog.dom.setTextContent(this.elMonthYear_,
1332 this.i18nDateFormatterMonthYear_.format(date));
1333 }
1334 if (this.elMonth_) {
1335 goog.dom.setTextContent(this.elMonth_,
1336 this.symbols_.STANDALONEMONTHS[date.getMonth()]);
1337 }
1338 if (this.elYear_) {
1339 goog.dom.setTextContent(this.elYear_,
1340 this.i18nDateFormatterYear_.format(date));
1341 }
1342
1343 var wday = date.getWeekday();
1344 var days = date.getNumberOfDaysInMonth();
1345
1346 // Determine how many days to show for previous month
1347 date.add(new goog.date.Interval(goog.date.Interval.MONTHS, -1));
1348 date.setDate(date.getNumberOfDaysInMonth() - (wday - 1));
1349
1350 if (this.showFixedNumWeeks_ && !this.extraWeekAtEnd_ && days + wday < 33) {
1351 date.add(new goog.date.Interval(goog.date.Interval.DAYS, -7));
1352 }
1353
1354 // Create weekday/day grid
1355 var dayInterval = new goog.date.Interval(goog.date.Interval.DAYS, 1);
1356 this.grid_ = [];
1357 for (var y = 0; y < 6; y++) { // Weeks
1358 this.grid_[y] = [];
1359 for (var x = 0; x < 7; x++) { // Weekdays
1360 this.grid_[y][x] = date.clone();
1361 date.add(dayInterval);
1362 }
1363 }
1364
1365 this.redrawCalendarGrid_();
1366};

1369/**
1370 * Draws calendar view from in memory representation and applies class names
1371 * depending on the selection, weekday and whatever the day belongs to the
1372 * active month or not.
1373 * @private
1374 */
1375goog.ui.DatePicker.prototype.redrawCalendarGrid_ = function() {
1376 if (!this.getElement()) {
1377 return;
1378 }
1379
1380 var month = this.activeMonth_.getMonth();
1381 var today = new goog.date.Date();
1382 var todayYear = today.getFullYear();
1383 var todayMonth = today.getMonth();
1384 var todayDate = today.getDate();
1385
1386 // Draw calendar week by week, a worst case month has six weeks.
1387 for (var y = 0; y < 6; y++) {
1388
1389 // Draw week number, if enabled
1390 if (this.showWeekNum_) {
1391 goog.dom.setTextContent(this.elTable_[y + 1][0],
1392 this.i18nDateFormatterWeek_.format(this.grid_[y][0]));
1393 goog.dom.classlist.set(this.elTable_[y + 1][0],
1394 goog.getCssName(this.getBaseCssClass(), 'week'));
1395 } else {
1396 goog.dom.setTextContent(this.elTable_[y + 1][0], '');
1397 goog.dom.classlist.set(this.elTable_[y + 1][0], '');
1398 }
1399
1400 for (var x = 0; x < 7; x++) {
1401 var o = this.grid_[y][x];
1402 var el = this.elTable_[y + 1][x + 1];
1403
1404 // Assign a unique element id (required for setting the active descendant
1405 // ARIA role) unless already set.
1406 if (!el.id) {
1407 el.id = this.cellIdGenerator_.getNextUniqueId();
1408 }
1409 goog.asserts.assert(el, 'The table DOM element cannot be null.');
1410 goog.a11y.aria.setRole(el, 'gridcell');
1411 var classes = [goog.getCssName(this.getBaseCssClass(), 'date')];
1412 if (!this.isUserSelectableDate_(o)) {
1413 classes.push(goog.getCssName(this.getBaseCssClass(),
1414 'unavailable-date'));
1415 }
1416 if (this.showOtherMonths_ || o.getMonth() == month) {
1417 // Date belongs to previous or next month
1418 if (o.getMonth() != month) {
1419 classes.push(goog.getCssName(this.getBaseCssClass(), 'other-month'));
1420 }
1421
1422 // Apply styles set by setWeekdayClass
1423 var wday = (x + this.activeMonth_.getFirstDayOfWeek() + 7) % 7;
1424 if (this.wdayStyles_[wday]) {
1425 classes.push(this.wdayStyles_[wday]);
1426 }
1427
1428 // Current date
1429 if (o.getDate() == todayDate && o.getMonth() == todayMonth &&
1430 o.getFullYear() == todayYear) {
1431 classes.push(goog.getCssName(this.getBaseCssClass(), 'today'));
1432 }
1433
1434 // Selected date
1435 if (this.date_ && o.getDate() == this.date_.getDate() &&
1436 o.getMonth() == this.date_.getMonth() &&
1437 o.getFullYear() == this.date_.getFullYear()) {
1438 classes.push(goog.getCssName(this.getBaseCssClass(), 'selected'));
1439 goog.asserts.assert(this.tableBody_,
1440 'The table body DOM element cannot be null');
1441 goog.a11y.aria.setState(this.tableBody_, 'activedescendant', el.id);
1442 }
1443
1444 // Custom decorator
1445 if (this.decoratorFunction_) {
1446 var customClass = this.decoratorFunction_(o);
1447 if (customClass) {
1448 classes.push(customClass);
1449 }
1450 }
1451
1452 // Set cell text to the date and apply classes.
1453 var formatedDate = this.longDateFormat_ ?
1454 this.i18nDateFormatterDay2_.format(o) :
1455 this.i18nDateFormatterDay_.format(o);
1456 goog.dom.setTextContent(el, formatedDate);
1457 // Date belongs to previous or next month and showOtherMonths is false,
1458 // clear text and classes.
1459 } else {
1460 goog.dom.setTextContent(el, '');
1461 }
1462 goog.dom.classlist.set(el, classes.join(' '));
1463 }
1464
1465 // Hide the either the last one or last two weeks if they contain no days
1466 // from the active month and the showFixedNumWeeks is false. The first four
1467 // weeks are always shown as no month has less than 28 days).
1468 if (y >= 4) {
1469 var parentEl = /** @type {Element} */ (
1470 this.elTable_[y + 1][0].parentElement ||
1471 this.elTable_[y + 1][0].parentNode);
1472 goog.style.setElementShown(parentEl,
1473 this.grid_[y][0].getMonth() == month || this.showFixedNumWeeks_);
1474 }
1475 }
1476};
