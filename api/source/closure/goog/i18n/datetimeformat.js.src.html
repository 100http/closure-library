33/**
34 * Datetime formatting functions following the pattern specification as defined
35 * in JDK, ICU and CLDR, with minor modification for typical usage in JS.
36 * Pattern specification: (Refer to JDK/ICU/CLDR)
37 * <pre>
38 * Symbol Meaning Presentation Example
39 * ------ ------- ------------ -------
40 * G era designator (Text) AD
41 * y# year (Number) 1996
42 * Y* year (week of year) (Number) 1997
43 * u* extended year (Number) 4601
44 * M month in year (Text & Number) July & 07
45 * d day in month (Number) 10
46 * h hour in am/pm (1~12) (Number) 12
47 * H hour in day (0~23) (Number) 0
48 * m minute in hour (Number) 30
49 * s second in minute (Number) 55
50 * S fractional second (Number) 978
51 * E day of week (Text) Tuesday
52 * e* day of week (local 1~7) (Number) 2
53 * D* day in year (Number) 189
54 * F* day of week in month (Number) 2 (2nd Wed in July)
55 * w week in year (Number) 27
56 * W* week in month (Number) 2
57 * a am/pm marker (Text) PM
58 * k hour in day (1~24) (Number) 24
59 * K hour in am/pm (0~11) (Number) 0
60 * z time zone (Text) Pacific Standard Time
61 * Z time zone (RFC 822) (Number) -0800
62 * v time zone (generic) (Text) Pacific Time
63 * g* Julian day (Number) 2451334
64 * A* milliseconds in day (Number) 69540000
65 * ' escape for text (Delimiter) 'Date='
66 * '' single quote (Literal) 'o''clock'
67 *
68 * Item marked with '*' are not supported yet.
69 * Item marked with '#' works different than java
70 *
71 * The count of pattern letters determine the format.
72 * (Text): 4 or more, use full form, <4, use short or abbreviated form if it
73 * exists. (e.g., "EEEE" produces "Monday", "EEE" produces "Mon")
74 *
75 * (Number): the minimum number of digits. Shorter numbers are zero-padded to
76 * this amount (e.g. if "m" produces "6", "mm" produces "06"). Year is handled
77 * specially; that is, if the count of 'y' is 2, the Year will be truncated to
78 * 2 digits. (e.g., if "yyyy" produces "1997", "yy" produces "97".) Unlike other
79 * fields, fractional seconds are padded on the right with zero.
80 *
81 * (Text & Number): 3 or over, use text, otherwise use number. (e.g., "M"
82 * produces "1", "MM" produces "01", "MMM" produces "Jan", and "MMMM" produces
83 * "January".)
84 *
85 * Any characters in the pattern that are not in the ranges of ['a'..'z'] and
86 * ['A'..'Z'] will be treated as quoted text. For instance, characters like ':',
87 * '.', ' ', '#' and '@' will appear in the resulting time text even they are
88 * not embraced within single quotes.
89 * </pre>
90 */

94/**
95 * Construct a DateTimeFormat object based on current locale.
96 * @constructor
97 * @param {string|number} pattern pattern specification or pattern type.
98 * @param {!Object=} opt_dateTimeSymbols Optional symbols to use use for this
99 * instance rather than the global symbols.
100 * @final
101 */
102goog.i18n.DateTimeFormat = function(pattern, opt_dateTimeSymbols) {
103 goog.asserts.assert(goog.isDef(pattern), 'Pattern must be defined');
104 goog.asserts.assert(
105 goog.isDef(opt_dateTimeSymbols) || goog.isDef(goog.i18n.DateTimeSymbols),
106 'goog.i18n.DateTimeSymbols or explicit symbols must be defined');
107
108 this.patternParts_ = [];
109
110 /**
111 * Data structure that with all the locale info needed for date formatting.
112 * (day/month names, most common patterns, rules for week-end, etc.)
113 * @type {!Object}
114 * @private
115 */
116 this.dateTimeSymbols_ = opt_dateTimeSymbols || goog.i18n.DateTimeSymbols;
117 if (typeof pattern == 'number') {
118 this.applyStandardPattern_(pattern);
119 } else {
120 this.applyPattern_(pattern);
121 }
122};

145/**
146 * regular expression pattern for parsing pattern string
147 * @type {Array<RegExp>}
148 * @private
149 */
150goog.i18n.DateTimeFormat.TOKENS_ = [
151 //quote string
152 /^\'(?:[^\']|\'\')*\'/,
153 // pattern chars
154 /^(?:G+|y+|M+|k+|S+|E+|a+|h+|K+|H+|c+|L+|Q+|d+|m+|s+|v+|w+|z+|Z+)/,
155 // and all the other chars
156 /^[^\'GyMkSEahKHcLQdmsvwzZ]+/ // and all the other chars
157];

160/**
161 * These are token types, corresponding to above token definitions.
162 * @enum {number}
163 * @private
164 */
165goog.i18n.DateTimeFormat.PartTypes_ = {
166 QUOTED_STRING: 0,
167 FIELD: 1,
168 LITERAL: 2
169};

172/**
173 * Apply specified pattern to this formatter object.
174 * @param {string} pattern String specifying how the date should be formatted.
175 * @private
176 */
177goog.i18n.DateTimeFormat.prototype.applyPattern_ = function(pattern) {
178 if (goog.i18n.DateTimeFormat.removeRlmInPatterns_) {
179 // Remove RLM unicode control character from pattern.
180 pattern = pattern.replace(/\u200f/g, '');
181 }
182 // lex the pattern, once for all uses
183 while (pattern) {
184 for (var i = 0; i < goog.i18n.DateTimeFormat.TOKENS_.length; ++i) {
185 var m = pattern.match(goog.i18n.DateTimeFormat.TOKENS_[i]);
186 if (m) {
187 var part = m[0];
188 pattern = pattern.substring(part.length);
189 if (i == goog.i18n.DateTimeFormat.PartTypes_.QUOTED_STRING) {
190 if (part == "''") {
191 part = "'"; // '' -> '
192 } else {
193 part = part.substring(1, part.length - 1); // strip quotes
194 part = part.replace(/\'\'/, "'");
195 }
196 }
197 this.patternParts_.push({ text: part, type: i });
198 break;
199 }
200 }
201 }
202};

205/**
206 * Format the given date object according to preset pattern and current locale.
207 * @param {goog.date.DateLike} date The Date object that is being formatted.
208 * @param {goog.i18n.TimeZone=} opt_timeZone optional, if specified, time
209 * related fields will be formatted based on its setting. When this field
210 * is not specified, "undefined" will be pass around and those function
211 * that really need time zone service will create a default one.
212 * @return {string} Formatted string for the given date.
213 * Throws an error if the date is null or if one tries to format a date-only
214 * object (for instance goog.date.Date) using a pattern with time fields.
215 */
216goog.i18n.DateTimeFormat.prototype.format = function(date, opt_timeZone) {
217 if (!date)
218 throw Error('The date to format must be non-null.');
219
220 // We don't want to write code to calculate each date field because we
221 // want to maximize performance and minimize code size.
222 // JavaScript only provide API to render local time.
223 // Suppose target date is: 16:00 GMT-0400
224 // OS local time is: 12:00 GMT-0800
225 // We want to create a Local Date Object : 16:00 GMT-0800, and fix the
226 // time zone display ourselves.
227 // Thing get a little bit tricky when daylight time transition happens. For
228 // example, suppose OS timeZone is America/Los_Angeles, it is impossible to
229 // represent "2006/4/2 02:30" even for those timeZone that has no transition
230 // at this time. Because 2:00 to 3:00 on that day does not exist in
231 // America/Los_Angeles time zone. To avoid calculating date field through
232 // our own code, we uses 3 Date object instead, one for "Year, month, day",
233 // one for time within that day, and one for timeZone object since it need
234 // the real time to figure out actual time zone offset.
235 var diff = opt_timeZone ?
236 (date.getTimezoneOffset() - opt_timeZone.getOffset(date)) * 60000 : 0;
237 var dateForDate = diff ? new Date(date.getTime() + diff) : date;
238 var dateForTime = dateForDate;
239 // When the time manipulation applied above spans the DST on/off hour, this
240 // could alter the time incorrectly by adding or subtracting an additional
241 // hour.
242 // We can mitigate this by:
243 // - Adding the difference in timezone offset to the date. This ensures that
244 // the dateForDate is still within the right day if the extra DST hour
245 // affected the date.
246 // - Move the time one day forward if we applied a timezone offset backwards,
247 // or vice versa. This trick ensures that the time is in the same offset
248 // as the original date, so we remove the additional hour added or
249 // subtracted by the DST switch.
250 if (opt_timeZone &&
251 dateForDate.getTimezoneOffset() != date.getTimezoneOffset()) {
252 var dstDiff = (dateForDate.getTimezoneOffset() - date.getTimezoneOffset()) *
253 60000;
254 dateForDate = new Date(dateForDate.getTime() + dstDiff);
255
256 diff += diff > 0 ? -goog.date.MS_PER_DAY : goog.date.MS_PER_DAY;
257 dateForTime = new Date(date.getTime() + diff);
258 }
259
260 var out = [];
261 for (var i = 0; i < this.patternParts_.length; ++i) {
262 var text = this.patternParts_[i].text;
263 if (goog.i18n.DateTimeFormat.PartTypes_.FIELD ==
264 this.patternParts_[i].type) {
265 out.push(this.formatField_(text, date, dateForDate, dateForTime,
266 opt_timeZone));
267 } else {
268 out.push(text);
269 }
270 }
271 return out.join('');
272};

275/**
276 * Apply a predefined pattern as identified by formatType, which is stored in
277 * locale specific repository.
278 * @param {number} formatType A number that identified the predefined pattern.
279 * @private
280 */
281goog.i18n.DateTimeFormat.prototype.applyStandardPattern_ =
282 function(formatType) {
283 var pattern;
284 if (formatType < 4) {
285 pattern = this.dateTimeSymbols_.DATEFORMATS[formatType];
286 } else if (formatType < 8) {
287 pattern = this.dateTimeSymbols_.TIMEFORMATS[formatType - 4];
288 } else if (formatType < 12) {
289 pattern = this.dateTimeSymbols_.DATETIMEFORMATS[formatType - 8];
290 pattern = pattern.replace('{1}',
291 this.dateTimeSymbols_.DATEFORMATS[formatType - 8]);
292 pattern = pattern.replace('{0}',
293 this.dateTimeSymbols_.TIMEFORMATS[formatType - 8]);
294 } else {
295 this.applyStandardPattern_(goog.i18n.DateTimeFormat.Format.MEDIUM_DATETIME);
296 return;
297 }
298 this.applyPattern_(pattern);
299};

331/**
332 * Sets if the usage of Ascii digits in formatting should be enforced in
333 * formatted date/time even for locales where native digits are indicated.
334 * Also sets whether to remove RLM unicode control characters when using
335 * standard enumerated patterns (they exist e.g. in standard d/M/y for Arabic).
336 * Production code should call this once before any {@code DateTimeFormat}
337 * object is instantiated.
338 * Caveats:
339 * * Enforcing ASCII digits affects all future formatting by new or existing
340 * {@code DateTimeFormat} objects.
341 * * Removal of RLM characters only applies to {@code DateTimeFormat} objects
342 * instantiated after this call.
343 * @param {boolean} enforceAsciiDigits Whether Ascii digits should be enforced.
344 */
345goog.i18n.DateTimeFormat.setEnforceAsciiDigits = function(
346 enforceAsciiDigits) {
347 goog.i18n.DateTimeFormat.enforceAsciiDigits_ = enforceAsciiDigits;
348
349 // Also setting removal of RLM chracters when forcing ASCII digits since it's
350 // the right thing to do for Arabic standard patterns. One could add an
351 // optional argument here or to the {@code DateTimeFormat} constructor to
352 // enable an alternative behavior.
353 goog.i18n.DateTimeFormat.removeRlmInPatterns_ = enforceAsciiDigits;
354};

366/**
367 * Localizes a string potentially containing numbers, replacing ASCII digits
368 * with native digits if specified so by the locale. Leaves other characters.
369 * @param {number|string} input the string to be localized, using ASCII digits.
370 * @param {!Object=} opt_dateTimeSymbols Optional symbols to use use rather than
371 * the global symbols.
372 * @return {string} localized string, potentially using native digits.
373 */
374goog.i18n.DateTimeFormat.localizeNumbers =
375 function(input, opt_dateTimeSymbols) {
376 input = String(input);
377 var dateTimeSymbols = opt_dateTimeSymbols || goog.i18n.DateTimeSymbols;
378 if (dateTimeSymbols.ZERODIGIT === undefined ||
379 goog.i18n.DateTimeFormat.enforceAsciiDigits_) {
380 return input;
381 }
382
383 var parts = [];
384 for (var i = 0; i < input.length; i++) {
385 var c = input.charCodeAt(i);
386 parts.push((0x30 <= c && c <= 0x39) ? // '0' <= c <= '9'
387 String.fromCharCode(dateTimeSymbols.ZERODIGIT + c - 0x30) :
388 input.charAt(i));
389 }
390 return parts.join('');
391};

410/**
411 * Formats Year field according to pattern specified
412 * Javascript Date object seems incapable handling 1BC and
413 * year before. It can show you year 0 which does not exists.
414 * following we just keep consistent with javascript's
415 * toString method. But keep in mind those things should be
416 * unsupported.
417 * @param {number} count Number of time pattern char repeats, it controls
418 * how a field should be formatted.
419 * @param {!goog.date.DateLike} date It holds the date object to be formatted.
420 * @return {string} Formatted string that represent this field.
421 * @private
422 */
423goog.i18n.DateTimeFormat.prototype.formatYear_ = function(count, date) {
424 var value = date.getFullYear();
425 if (value < 0) {
426 value = -value;
427 }
428 if (count == 2) {
429 // See comment about special casing 'yy' at the start of the file, this
430 // matches ICU and CLDR behaviour. See also:
431 // http://icu-project.org/apiref/icu4j/com/ibm/icu/text/SimpleDateFormat.html
432 // http://www.unicode.org/reports/tr35/tr35-dates.html
433 value = value % 100;
434 }
435 return this.localizeNumbers_(goog.string.padNumber(value, count));
436};

460/**
461 * Validates is the goog.date.DateLike object to format has a time.
462 * DateLike means Date|goog.date.Date, and goog.date.DateTime inherits
463 * from goog.date.Date. But goog.date.Date does not have time related
464 * members (getHours, getMinutes, getSeconds).
465 * Formatting can be done, if there are no time placeholders in the pattern.
466 *
467 * @param {!goog.date.DateLike} date the object to validate.
468 * @private
469 */
470goog.i18n.DateTimeFormat.validateDateHasTime_ = function(date) {
471 if (date.getHours && date.getSeconds && date.getMinutes)
472 return;
473 // if (date instanceof Date || date instanceof goog.date.DateTime)
474 throw Error('The date to format has no time (probably a goog.date.Date). ' +
475 'Use Date or goog.date.DateTime, or use a pattern without time fields.');
476};

789/**
790 * Formatting one date field.
791 * @param {string} patternStr The pattern string for the field being formatted.
792 * @param {!goog.date.DateLike} date represents the real date to be formatted.
793 * @param {!goog.date.DateLike} dateForDate used to resolve date fields
794 * for formatting.
795 * @param {!goog.date.DateLike} dateForTime used to resolve time fields
796 * for formatting.
797 * @param {goog.i18n.TimeZone=} opt_timeZone This holds current time zone info.
798 * @return {string} string representation for the given field.
799 * @private
800 */
801goog.i18n.DateTimeFormat.prototype.formatField_ =
802 function(patternStr, date, dateForDate, dateForTime, opt_timeZone) {
803 var count = patternStr.length;
804 switch (patternStr.charAt(0)) {
805 case 'G': return this.formatEra_(count, dateForDate);
806 case 'y': return this.formatYear_(count, dateForDate);
807 case 'M': return this.formatMonth_(count, dateForDate);
808 case 'k': return this.format24Hours_(count, dateForTime);
809 case 'S': return this.formatFractionalSeconds_(count, dateForTime);
810 case 'E': return this.formatDayOfWeek_(count, dateForDate);
811 case 'a': return this.formatAmPm_(count, dateForTime);
812 case 'h': return this.format1To12Hours_(count, dateForTime);
813 case 'K': return this.format0To11Hours_(count, dateForTime);
814 case 'H': return this.format0To23Hours_(count, dateForTime);
815 case 'c': return this.formatStandaloneDay_(count, dateForDate);
816 case 'L': return this.formatStandaloneMonth_(count, dateForDate);
817 case 'Q': return this.formatQuarter_(count, dateForDate);
818 case 'd': return this.formatDate_(count, dateForDate);
819 case 'm': return this.formatMinutes_(count, dateForTime);
820 case 's': return this.formatSeconds_(count, dateForTime);
821 case 'v': return this.formatTimeZoneId_(date, opt_timeZone);
822 case 'w': return this.formatWeekOfYear_(count, dateForTime);
823 case 'z': return this.formatTimeZone_(count, date, opt_timeZone);
824 case 'Z': return this.formatTimeZoneRFC_(count, date, opt_timeZone);
825 default: return '';
826 }
