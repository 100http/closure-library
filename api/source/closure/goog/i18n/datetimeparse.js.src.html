30/**
31 * DateTimeParse is for parsing date in a locale-sensitive manner. It allows
32 * user to use any customized patterns to parse date-time string under certain
33 * locale. Things varies across locales like month name, weekname, field
34 * order, etc.
35 *
36 * This module is the counter-part of DateTimeFormat. They use the same
37 * date/time pattern specification, which is borrowed from ICU/JDK.
38 *
39 * This implementation could parse partial date/time.
40 *
41 * Time Format Syntax: To specify the time format use a time pattern string.
42 * In this pattern, following letters are reserved as pattern letters, which
43 * are defined as the following:
44 *
45 * <pre>
46 * Symbol Meaning Presentation Example
47 * ------ ------- ------------ -------
48 * G era designator (Text) AD
49 * y# year (Number) 1996
50 * M month in year (Text & Number) July & 07
51 * d day in month (Number) 10
52 * h hour in am/pm (1~12) (Number) 12
53 * H hour in day (0~23) (Number) 0
54 * m minute in hour (Number) 30
55 * s second in minute (Number) 55
56 * S fractional second (Number) 978
57 * E day of week (Text) Tuesday
58 * D day in year (Number) 189
59 * a am/pm marker (Text) PM
60 * k hour in day (1~24) (Number) 24
61 * K hour in am/pm (0~11) (Number) 0
62 * z time zone (Text) Pacific Standard Time
63 * Z time zone (RFC 822) (Number) -0800
64 * v time zone (generic) (Text) Pacific Time
65 * ' escape for text (Delimiter) 'Date='
66 * '' single quote (Literal) 'o''clock'
67 * </pre>
68 *
69 * The count of pattern letters determine the format. <p>
70 * (Text): 4 or more pattern letters--use full form,
71 * less than 4--use short or abbreviated form if one exists.
72 * In parsing, we will always try long format, then short. <p>
73 * (Number): the minimum number of digits. <p>
74 * (Text & Number): 3 or over, use text, otherwise use number. <p>
75 * Any characters that not in the pattern will be treated as quoted text. For
76 * instance, characters like ':', '.', ' ', '#' and '@' will appear in the
77 * resulting time text even they are not embraced within single quotes. In our
78 * current pattern usage, we didn't use up all letters. But those unused
79 * letters are strongly discouraged to be used as quoted text without quote.
80 * That's because we may use other letter for pattern in future. <p>
81 *
82 * Examples Using the US Locale:
83 *
84 * Format Pattern Result
85 * -------------- -------
86 * "yyyy.MM.dd G 'at' HH:mm:ss vvvv" ->> 1996.07.10 AD at 15:08:56 Pacific Time
87 * "EEE, MMM d, ''yy" ->> Wed, July 10, '96
88 * "h:mm a" ->> 12:08 PM
89 * "hh 'o''clock' a, zzzz" ->> 12 o'clock PM, Pacific Daylight Time
90 * "K:mm a, vvv" ->> 0:00 PM, PT
91 * "yyyyy.MMMMM.dd GGG hh:mm aaa" ->> 01996.July.10 AD 12:08 PM
92 *
93 * <p> When parsing a date string using the abbreviated year pattern ("yy"),
94 * DateTimeParse must interpret the abbreviated year relative to some
95 * century. It does this by adjusting dates to be within 80 years before and 20
96 * years after the time the parse function is called. For example, using a
97 * pattern of "MM/dd/yy" and a DateTimeParse instance created on Jan 1, 1997,
98 * the string "01/11/12" would be interpreted as Jan 11, 2012 while the string
99 * "05/04/64" would be interpreted as May 4, 1964. During parsing, only
100 * strings consisting of exactly two digits, as defined by {@link
101 * java.lang.Character#isDigit(char)}, will be parsed into the default
102 * century. Any other numeric string, such as a one digit string, a three or
103 * more digit string will be interpreted as its face value.
104 *
105 * <p> If the year pattern does not have exactly two 'y' characters, the year is
106 * interpreted literally, regardless of the number of digits. So using the
107 * pattern "MM/dd/yyyy", "01/11/12" parses to Jan 11, 12 A.D.
108 *
109 * <p> When numeric fields abut one another directly, with no intervening
110 * delimiter characters, they constitute a run of abutting numeric fields. Such
111 * runs are parsed specially. For example, the format "HHmmss" parses the input
112 * text "123456" to 12:34:56, parses the input text "12345" to 1:23:45, and
113 * fails to parse "1234". In other words, the leftmost field of the run is
114 * flexible, while the others keep a fixed width. If the parse fails anywhere in
115 * the run, then the leftmost field is shortened by one character, and the
116 * entire run is parsed again. This is repeated until either the parse succeeds
117 * or the leftmost field is one character in length. If the parse still fails at
118 * that point, the parse of the run fails.
119 *
120 * <p> Now timezone parsing only support GMT:hhmm, GMT:+hhmm, GMT:-hhmm
121 */

126 * Construct a DateTimeParse based on current locale.
127 * @param {string|number} pattern pattern specification or pattern type.

129 * @final

131goog.i18n.DateTimeParse = function(pattern) {
132 this.patternParts_ = [];
133 if (typeof pattern == 'number') {
134 this.applyStandardPattern_(pattern);
135 } else {
136 this.applyPattern_(pattern);

150/**
151 * Apply a pattern to this Parser. The pattern string will be parsed and saved
152 * in "compiled" form.
153 * Note: this method is somewhat similar to the pattern parsing method in
154 * datetimeformat. If you see something wrong here, you might want
155 * to check the other.
156 * @param {string} pattern It describes the format of date string that need to
157 * be parsed.
158 * @private
159 */
160goog.i18n.DateTimeParse.prototype.applyPattern_ = function(pattern) {
161 var inQuote = false;
162 var buf = '';
163
164 for (var i = 0; i < pattern.length; i++) {
165 var ch = pattern.charAt(i);
166
167 // handle space, add literal part (if exist), and add space part
168 if (ch == ' ') {
169 if (buf.length > 0) {
170 this.patternParts_.push({text: buf, count: 0, abutStart: false});
171 buf = '';
172 }
173 this.patternParts_.push({text: ' ', count: 0, abutStart: false});
174 while (i < pattern.length - 1 && pattern.charAt(i + 1) == ' ') {
175 i++;
176 }
177 } else if (inQuote) {
178 // inside quote, except '', just copy or exit
179 if (ch == '\'') {
180 if (i + 1 < pattern.length && pattern.charAt(i + 1) == '\'') {
181 // quote appeared twice continuously, interpret as one quote.
182 buf += '\'';
183 i++;
184 } else {
185 // exit quote
186 inQuote = false;
187 }
188 } else {
189 // literal
190 buf += ch;
191 }
192 } else if (goog.i18n.DateTimeParse.PATTERN_CHARS_.indexOf(ch) >= 0) {
193 // outside quote, it is a pattern char
194 if (buf.length > 0) {
195 this.patternParts_.push({text: buf, count: 0, abutStart: false});
196 buf = '';
197 }
198 var count = this.getNextCharCount_(pattern, i);
199 this.patternParts_.push({text: ch, count: count, abutStart: false});
200 i += count - 1;
201 } else if (ch == '\'') {
202 // Two consecutive quotes is a quote literal, inside or outside of quotes.
203 if (i + 1 < pattern.length && pattern.charAt(i + 1) == '\'') {
204 buf += '\'';
205 i++;
206 } else {
207 inQuote = true;
208 }
209 } else {
210 buf += ch;
211 }
212 }
213
214 if (buf.length > 0) {
215 this.patternParts_.push({text: buf, count: 0, abutStart: false});
216 }
217
218 this.markAbutStart_();
219};

222/**
223 * Apply a predefined pattern to this Parser.
224 * @param {number} formatType A constant used to identified the predefined
225 * pattern string stored in locale repository.
226 * @private
227 */
228goog.i18n.DateTimeParse.prototype.applyStandardPattern_ = function(formatType) {
229 var pattern;
230 // formatType constants are in consecutive numbers. So it can be used to
231 // index array in following way.
232
233 // if type is out of range, default to medium date/time format.
234 if (formatType > goog.i18n.DateTimeFormat.Format.SHORT_DATETIME) {
235 formatType = goog.i18n.DateTimeFormat.Format.MEDIUM_DATETIME;
236 }
237
238 if (formatType < 4) {
239 pattern = goog.i18n.DateTimeSymbols.DATEFORMATS[formatType];
240 } else if (formatType < 8) {
241 pattern = goog.i18n.DateTimeSymbols.TIMEFORMATS[formatType - 4];
242 } else {
243 pattern = goog.i18n.DateTimeSymbols.DATETIMEFORMATS[formatType - 8];
244 pattern = pattern.replace('{1}',
245 goog.i18n.DateTimeSymbols.DATEFORMATS[formatType - 8]);
246 pattern = pattern.replace('{0}',
247 goog.i18n.DateTimeSymbols.TIMEFORMATS[formatType - 8]);
248 }
249 this.applyPattern_(pattern);
250};

282/**
283 * Parse the given string and fill info into date object.
284 * @param {string} text The string being parsed.
285 * @param {goog.date.DateLike} date The Date object to hold the parsed date.
286 * @param {number} start The position from where parse should begin.
287 * @param {boolean} validation If true, input string need to be a valid
288 * date/time string.
289 * @return {number} How many characters parser advanced.
290 * @private
291 */
292goog.i18n.DateTimeParse.prototype.internalParse_ =
293 function(text, date, start, validation) {
294 var cal = new goog.i18n.DateTimeParse.MyDate_();
295 var parsePos = [start];
296
297 // For parsing abutting numeric fields. 'abutPat' is the
298 // offset into 'pattern' of the first of 2 or more abutting
299 // numeric fields. 'abutStart' is the offset into 'text'
300 // where parsing the fields begins. 'abutPass' starts off as 0
301 // and increments each time we try to parse the fields.
302 var abutPat = -1; // If >=0, we are in a run of abutting numeric fields
303 var abutStart = 0;
304 var abutPass = 0;
305
306 for (var i = 0; i < this.patternParts_.length; i++) {
307 if (this.patternParts_[i].count > 0) {
308 if (abutPat < 0 && this.patternParts_[i].abutStart) {
309 abutPat = i;
310 abutStart = start;
311 abutPass = 0;
312 }
313
314 // Handle fields within a run of abutting numeric fields. Take
315 // the pattern "HHmmss" as an example. We will try to parse
316 // 2/2/2 characters of the input text, then if that fails,
317 // 1/2/2. We only adjust the width of the leftmost field; the
318 // others remain fixed. This allows "123456" => 12:34:56, but
319 // "12345" => 1:23:45. Likewise, for the pattern "yyyyMMdd" we
320 // try 4/2/2, 3/2/2, 2/2/2, and finally 1/2/2.
321 if (abutPat >= 0) {
322 // If we are at the start of a run of abutting fields, then
323 // shorten this field in each pass. If we can't shorten
324 // this field any more, then the parse of this set of
325 // abutting numeric fields has failed.
326 var count = this.patternParts_[i].count;
327 if (i == abutPat) {
328 count -= abutPass;
329 abutPass++;
330 if (count == 0) {
331 // tried all possible width, fail now
332 return 0;
333 }
334 }
335
336 if (!this.subParse_(text, parsePos, this.patternParts_[i], count,
337 cal)) {
338 // If the parse fails anywhere in the run, back up to the
339 // start of the run and retry.
340 i = abutPat - 1;
341 parsePos[0] = abutStart;
342 continue;
343 }
344 }
345
346 // Handle non-numeric fields and non-abutting numeric fields.
347 else {
348 abutPat = -1;
349 if (!this.subParse_(text, parsePos, this.patternParts_[i], 0, cal)) {
350 return 0;
351 }
352 }
353 } else {
354 // Handle literal pattern characters. These are any
355 // quoted characters and non-alphabetic unquoted
356 // characters.
357 abutPat = -1;
358 // A run of white space in the pattern matches a run
359 // of white space in the input text.
360 if (this.patternParts_[i].text.charAt(0) == ' ') {
361 // Advance over run in input text
362 var s = parsePos[0];
363 this.skipSpace_(text, parsePos);
364
365 // Must see at least one white space char in input
366 if (parsePos[0] > s) {
367 continue;
368 }
369 } else if (text.indexOf(this.patternParts_[i].text, parsePos[0]) ==
370 parsePos[0]) {
371 parsePos[0] += this.patternParts_[i].text.length;
372 continue;
373 }
374 // We fall through to this point if the match fails
375 return 0;
376 }
377 }
378
379 // return progress
380 return cal.calcDate_(date, validation) ? parsePos[0] - start : 0;
381};

384/**
385 * Calculate character repeat count in pattern.
386 *
387 * @param {string} pattern It describes the format of date string that need to
388 * be parsed.
389 * @param {number} start The position of pattern character.
390 *
391 * @return {number} Repeat count.
392 * @private
393 */
394goog.i18n.DateTimeParse.prototype.getNextCharCount_ =
395 function(pattern, start) {
396 var ch = pattern.charAt(start);
397 var next = start + 1;
398 while (next < pattern.length && pattern.charAt(next) == ch) {
399 next++;
400 }
401 return next - start;
402};

419/**
420 * Check if the pattern part is a numeric field.
421 *
422 * @param {Object} part pattern part to be examined.
423 *
424 * @return {boolean} true if the pattern part is numeric field.
425 * @private
426 */
427goog.i18n.DateTimeParse.prototype.isNumericField_ = function(part) {
428 if (part.count <= 0) {
429 return false;
430 }
431 var i = goog.i18n.DateTimeParse.NUMERIC_FORMAT_CHARS_.indexOf(
432 part.text.charAt(0));
433 return i > 0 || i == 0 && part.count < 3;
434};

437/**
438 * Identify the start of an abutting numeric fields' run. Taking pattern
439 * "HHmmss" as an example. It will try to parse 2/2/2 characters of the input
440 * text, then if that fails, 1/2/2. We only adjust the width of the leftmost
441 * field; the others remain fixed. This allows "123456" => 12:34:56, but
442 * "12345" => 1:23:45. Likewise, for the pattern "yyyyMMdd" we try 4/2/2,
443 * 3/2/2, 2/2/2, and finally 1/2/2. The first field of connected numeric
444 * fields will be marked as abutStart, its width can be reduced to accommodate
445 * others.
446 *
447 * @private
448 */
449goog.i18n.DateTimeParse.prototype.markAbutStart_ = function() {
450 // abut parts are continuous numeric parts. abutStart is the switch
451 // point from non-abut to abut
452 var abut = false;
453
454 for (var i = 0; i < this.patternParts_.length; i++) {
455 if (this.isNumericField_(this.patternParts_[i])) {
456 // if next part is not following abut sequence, and isNumericField_
457 if (!abut && i + 1 < this.patternParts_.length &&
458 this.isNumericField_(this.patternParts_[i + 1])) {
459 abut = true;
460 this.patternParts_[i].abutStart = true;
461 }
462 } else {
463 abut = false;
464 }
465 }
466};

485/**
486 * Protected method that converts one field of the input string into a
487 * numeric field value.
488 *
489 * @param {string} text the time text to be parsed.
490 * @param {Array<number>} pos Parse position.
491 * @param {Object} part the pattern part for this field.
492 * @param {number} digitCount when > 0, numeric parsing must obey the count.
493 * @param {goog.i18n.DateTimeParse.MyDate_} cal object that holds parsed value.
494 *
495 * @return {boolean} True if it parses successfully.
496 * @private
497 */
498goog.i18n.DateTimeParse.prototype.subParse_ =
499 function(text, pos, part, digitCount, cal) {
500 this.skipSpace_(text, pos);
501
502 var start = pos[0];
503 var ch = part.text.charAt(0);
504
505 // parse integer value if it is a numeric field
506 var value = -1;
507 if (this.isNumericField_(part)) {
508 if (digitCount > 0) {
509 if ((start + digitCount) > text.length) {
510 return false;
511 }
512 value = this.parseInt_(
513 text.substring(0, start + digitCount), pos);
514 } else {
515 value = this.parseInt_(text, pos);
516 }
517 }
518
519 switch (ch) {
520 case 'G': // ERA
521 value = this.matchString_(text, pos, goog.i18n.DateTimeSymbols.ERAS);
522 if (value >= 0) {
523 cal.era = value;
524 }
525 return true;
526 case 'M': // MONTH
527 case 'L': // STANDALONEMONTH
528 return this.subParseMonth_(text, pos, cal, value);
529 case 'E':
530 return this.subParseDayOfWeek_(text, pos, cal);
531 case 'a': // AM_PM
532 value = this.matchString_(text, pos, goog.i18n.DateTimeSymbols.AMPMS);
533 if (value >= 0) {
534 cal.ampm = value;
535 }
536 return true;
537 case 'y': // YEAR
538 return this.subParseYear_(text, pos, start, value, part, cal);
539 case 'Q': // QUARTER
540 return this.subParseQuarter_(text, pos, cal, value);
541 case 'd': // DATE
542 if (value >= 0) {
543 cal.day = value;
544 }
545 return true;
546 case 'S': // FRACTIONAL_SECOND
547 return this.subParseFractionalSeconds_(value, pos, start, cal);
548 case 'h': // HOUR (1..12)
549 if (value == 12) {
550 value = 0;
551 }
552 case 'K': // HOUR (0..11)
553 case 'H': // HOUR_OF_DAY (0..23)
554 case 'k': // HOUR_OF_DAY (1..24)
555 if (value >= 0) {
556 cal.hours = value;
557 }
558 return true;
559 case 'm': // MINUTE
560 if (value >= 0) {
561 cal.minutes = value;
562 }
563 return true;
564 case 's': // SECOND
565 if (value >= 0) {
566 cal.seconds = value;
567 }
568 return true;
569
570 case 'z': // ZONE_OFFSET
571 case 'Z': // TIMEZONE_RFC
572 case 'v': // TIMEZONE_GENERIC
573 return this.subparseTimeZoneInGMT_(text, pos, cal);
574 default:
575 return false;
576 }
577};

580/**
581 * Parse year field. Year field is special because
582 * 1) two digit year need to be resolved.
583 * 2) we allow year to take a sign.
584 * 3) year field participate in abut processing.
585 *
586 * @param {string} text the time text to be parsed.
587 * @param {Array<number>} pos Parse position.
588 * @param {number} start where this field start.
589 * @param {number} value integer value of year.
590 * @param {Object} part the pattern part for this field.
591 * @param {goog.i18n.DateTimeParse.MyDate_} cal object to hold parsed value.
592 *
593 * @return {boolean} True if successful.
594 * @private
595 */
596goog.i18n.DateTimeParse.prototype.subParseYear_ =
597 function(text, pos, start, value, part, cal) {
598 var ch;
599 if (value < 0) {
600 //possible sign
601 ch = text.charAt(pos[0]);
602 if (ch != '+' && ch != '-') {
603 return false;
604 }
605 pos[0]++;
606 value = this.parseInt_(text, pos);
607 if (value < 0) {
608 return false;
609 }
610 if (ch == '-') {
611 value = -value;
612 }
613 }
614
615 // only if 2 digit was actually parsed, and pattern say it has 2 digit.
616 if (!ch && pos[0] - start == 2 && part.count == 2) {
617 cal.setTwoDigitYear_(value);
618 } else {
619 cal.year = value;
620 }
621 return true;
622};

625/**
626 * Parse Month field.
627 *
628 * @param {string} text the time text to be parsed.
629 * @param {Array<number>} pos Parse position.
630 * @param {goog.i18n.DateTimeParse.MyDate_} cal object to hold parsed value.
631 * @param {number} value numeric value if this field is expressed using
632 * numeric pattern, or -1 if not.
633 *
634 * @return {boolean} True if parsing successful.
635 * @private
636 */
637goog.i18n.DateTimeParse.prototype.subParseMonth_ =
638 function(text, pos, cal, value) {
639 // when month is symbols, i.e., MMM, MMMM, LLL or LLLL, value will be -1
640 if (value < 0) {
641 // Want to be able to parse both short and long forms.
642 // Try count == 4 first
643 var months = goog.i18n.DateTimeSymbols.MONTHS.concat(
644 goog.i18n.DateTimeSymbols.STANDALONEMONTHS).concat(
645 goog.i18n.DateTimeSymbols.SHORTMONTHS).concat(
646 goog.i18n.DateTimeSymbols.STANDALONESHORTMONTHS);
647 value = this.matchString_(text, pos, months);
648 if (value < 0) {
649 return false;
650 }
651 // The months variable is multiple of 12, so we have to get the actual
652 // month index by modulo 12.
653 cal.month = (value % 12);
654 return true;
655 } else {
656 cal.month = value - 1;
657 return true;
658 }
659};

662/**
663 * Parse Quarter field.
664 *
665 * @param {string} text the time text to be parsed.
666 * @param {Array<number>} pos Parse position.
667 * @param {goog.i18n.DateTimeParse.MyDate_} cal object to hold parsed value.
668 * @param {number} value numeric value if this field is expressed using
669 * numeric pattern, or -1 if not.
670 *
671 * @return {boolean} True if parsing successful.
672 * @private
673 */
674goog.i18n.DateTimeParse.prototype.subParseQuarter_ =
675 function(text, pos, cal, value) {
676 // value should be -1, since this is a non-numeric field.
677 if (value < 0) {
678 // Want to be able to parse both short and long forms.
679 // Try count == 4 first:
680 value = this.matchString_(text, pos, goog.i18n.DateTimeSymbols.QUARTERS);
681 if (value < 0) { // count == 4 failed, now try count == 3
682 value = this.matchString_(text, pos,
683 goog.i18n.DateTimeSymbols.SHORTQUARTERS);
684 }
685 if (value < 0) {
686 return false;
687 }
688 cal.month = value * 3; // First month of quarter.
689 cal.day = 1;
690 return true;
691 }
692 return false;
693};

698 * @param {string} text the time text to be parsed.

700 * @param {goog.i18n.DateTimeParse.MyDate_} cal object to hold parsed value.

744 * Parse GMT type timezone.
745 *
746 * @param {string} text the time text to be parsed.
747 * @param {Array<number>} pos Parse position.
748 * @param {goog.i18n.DateTimeParse.MyDate_} cal object to hold parsed value.

751 * @private
752 */
753goog.i18n.DateTimeParse.prototype.subparseTimeZoneInGMT_ =
754 function(text, pos, cal) {
755 // First try to parse generic forms such as GMT-07:00. Do this first
756 // in case localized DateFormatZoneData contains the string "GMT"
757 // for a zone; in that case, we don't want to match the first three
758 // characters of GMT+/-HH:MM etc.

760 // For time zones that have no known names, look for strings
761 // of the form:
762 // GMT[+-]hours:minutes or
763 // GMT[+-]hhmm or
764 // GMT.
765 if (text.indexOf('GMT', pos[0]) == pos[0]) {
766 pos[0] += 3; // 3 is the length of GMT
767 return this.parseTimeZoneOffset_(text, pos, cal);
768 }

770 // TODO(user): check for named time zones by looking through the locale
771 // data from the DateFormatZoneData strings. Should parse both short and long
772 // forms.
773 // subParseZoneString(text, start, cal);
774
775 // As a last resort, look for numeric timezones of the form
776 // [+-]hhmm as specified by RFC 822. This code is actually
777 // a little more permissive than RFC 822. It will try to do
778 // its best with numbers that aren't strictly 4 digits long.

783/**
784 * Parse time zone offset.
785 *
786 * @param {string} text the time text to be parsed.
787 * @param {Array<number>} pos Parse position.
788 * @param {goog.i18n.DateTimeParse.MyDate_} cal object to hold parsed value.
789 *
790 * @return {boolean} True if successful.
791 * @private
792 */
793goog.i18n.DateTimeParse.prototype.parseTimeZoneOffset_ =
794 function(text, pos, cal) {
795 if (pos[0] >= text.length) {
796 cal.tzOffset = 0;
797 return true;
798 }
799
800 var sign = 1;
801 switch (text.charAt(pos[0])) {
802 case '-': sign = -1; // fall through
803 case '+': pos[0]++;
804 }
805
806 // Look for hours:minutes or hhmm.
807 var st = pos[0];
808 var value = this.parseInt_(text, pos);
809 if (value < 0) {
810 return false;
811 }
812
813 var offset;
814 if (pos[0] < text.length && text.charAt(pos[0]) == ':') {
815 // This is the hours:minutes case
816 offset = value * 60;
817 pos[0]++;
818 value = this.parseInt_(text, pos);
819 if (value < 0) {
820 return false;
821 }
822 offset += value;
823 } else {
824 // This is the hhmm case.
825 offset = value;
826 // Assume "-23".."+23" refers to hours.
827 if (offset < 24 && (pos[0] - st) <= 2) {
828 offset *= 60;
829 } else {
830 // todo: this looks questionable, should have more error checking
831 offset = offset % 100 + offset / 100 * 60;
832 }
833 }
834
835 offset *= sign;
836 cal.tzOffset = -offset;
837 return true;
838};

841/**
842 * Parse an integer string and return integer value.
843 *
844 * @param {string} text string being parsed.
845 * @param {Array<number>} pos parse position.
846 *
847 * @return {number} Converted integer value or -1 if the integer cannot be
848 * parsed.
849 * @private
850 */
851goog.i18n.DateTimeParse.prototype.parseInt_ = function(text, pos) {
852 // Delocalizes the string containing native digits specified by the locale,
853 // replaces the native digits with ASCII digits. Leaves other characters.
854 // This is the reverse operation of localizeNumbers_ in datetimeformat.js.
855 if (goog.i18n.DateTimeSymbols.ZERODIGIT) {
856 var parts = [];
857 for (var i = pos[0]; i < text.length; i++) {
858 var c = text.charCodeAt(i) - goog.i18n.DateTimeSymbols.ZERODIGIT;
859 parts.push((0 <= c && c <= 9) ?
860 String.fromCharCode(c + 0x30) :
861 text.charAt(i));
862 }
863 text = parts.join('');
864 } else {
865 text = text.substring(pos[0]);
866 }
867
868 var m = text.match(/^\d+/);
869 if (!m) {
870 return -1;
871 }
872 pos[0] += m[0].length;
873 return parseInt(m[0], 10);
874};

877/**
878 * Attempt to match the text at a given position against an array of strings.
879 * Since multiple strings in the array may match (for example, if the array
880 * contains "a", "ab", and "abc", all will match the input string "abcd") the
881 * longest match is returned.
882 *
883 * @param {string} text The string to match to.
884 * @param {Array<number>} pos parsing position.
885 * @param {Array<string>} data The string array of matching patterns.
886 *
887 * @return {number} the new start position if matching succeeded; a negative
888 * number indicating matching failure.
889 * @private
890 */
891goog.i18n.DateTimeParse.prototype.matchString_ = function(text, pos, data) {
892 // There may be multiple strings in the data[] array which begin with
893 // the same prefix (e.g., Cerven and Cervenec (June and July) in Czech).
894 // We keep track of the longest match, and return that. Note that this
895 // unfortunately requires us to test all array elements.
896 var bestMatchLength = 0;
897 var bestMatch = -1;
898 var lower_text = text.substring(pos[0]).toLowerCase();
899 for (var i = 0; i < data.length; i++) {
900 var len = data[i].length;
901 // Always compare if we have no match yet; otherwise only compare
902 // against potentially better matches (longer strings).
903 if (len > bestMatchLength &&
904 lower_text.indexOf(data[i].toLowerCase()) == 0) {
905 bestMatch = i;
906 bestMatchLength = len;
907 }
908 }
909 if (bestMatch >= 0) {
910 pos[0] += bestMatchLength;
911 }
912 return bestMatch;
913};

1003/**
1004 * 2 digit year special handling. Assuming for example that the
1005 * defaultCenturyStart is 6/18/1903. This means that two-digit years will be
1006 * forced into the range 6/18/1903 to 6/17/2003. As a result, years 00, 01, and
1007 * 02 correspond to 2000, 2001, and 2002. Years 04, 05, etc. correspond
1008 * to 1904, 1905, etc. If the year is 03, then it is 2003 if the
1009 * other fields specify a date before 6/18, or 1903 if they specify a
1010 * date afterwards. As a result, 03 is an ambiguous year. All other
1011 * two-digit years are unambiguous.
1012 *
1013 * @param {number} year 2 digit year value before adjustment.
1014 * @return {number} disambiguated year.
1015 * @private
1016 */
1017goog.i18n.DateTimeParse.MyDate_.prototype.setTwoDigitYear_ = function(year) {
1018 var now = new Date();
1019 var defaultCenturyStartYear =
1020 now.getFullYear() - goog.i18n.DateTimeParse.ambiguousYearCenturyStart;
1021 var ambiguousTwoDigitYear = defaultCenturyStartYear % 100;
1022 this.ambiguousYear = (year == ambiguousTwoDigitYear);
1023 year += Math.floor(defaultCenturyStartYear / 100) * 100 +
1024 (year < ambiguousTwoDigitYear ? 100 : 0);
1025 return this.year = year;

1029/**
1030 * Based on the fields set, fill a Date object. For those fields that not
1031 * set, use the passed in date object's value.
1032 *
1033 * @param {goog.date.DateLike} date Date object to be filled.
1034 * @param {boolean} validation If true, input string will be checked to make
1035 * sure it is valid.
1036 *
1037 * @return {boolean} false if fields specify a invalid date.
1038 * @private
1039 */
1040goog.i18n.DateTimeParse.MyDate_.prototype.calcDate_ =
1041 function(date, validation) {
1042 // year 0 is 1 BC, and so on.
1043 if (this.era != undefined && this.year != undefined &&
1044 this.era == 0 && this.year > 0) {
1045 this.year = -(this.year - 1);
1046 }
1047
1048 if (this.year != undefined) {
1049 date.setFullYear(this.year);
1050 }
1051
1052 // The setMonth and setDate logic is a little tricky. We need to make sure
1053 // day of month is smaller enough so that it won't cause a month switch when
1054 // setting month. For example, if data in date is Nov 30, when month is set
1055 // to Feb, because there is no Feb 30, JS adjust it to Mar 2. So Feb 12 will
1056 // become Mar 12.
1057 var orgDate = date.getDate();
1058
1059 // Every month has a 1st day, this can actually be anything less than 29.
1060 date.setDate(1);
1061
1062 if (this.month != undefined) {
1063 date.setMonth(this.month);
1064 }
1065
1066 if (this.day != undefined) {
1067 date.setDate(this.day);
1068 } else {
1069 var maxDate =
1070 goog.date.getNumberOfDaysInMonth(date.getFullYear(), date.getMonth());
1071 date.setDate(orgDate > maxDate ? maxDate : orgDate);
1072 }
1073
1074 if (goog.isFunction(date.setHours)) {
1075 if (this.hours == undefined) {
1076 this.hours = date.getHours();
1077 }
1078 // adjust ampm
1079 if (this.ampm != undefined && this.ampm > 0 && this.hours < 12) {
1080 this.hours += 12;
1081 }
1082 date.setHours(this.hours);
1083 }
1084
1085 if (goog.isFunction(date.setMinutes) && this.minutes != undefined) {
1086 date.setMinutes(this.minutes);
1087 }
1088
1089 if (goog.isFunction(date.setSeconds) && this.seconds != undefined) {
1090 date.setSeconds(this.seconds);
1091 }
1092
1093 if (goog.isFunction(date.setMilliseconds) &&
1094 this.milliseconds != undefined) {
1095 date.setMilliseconds(this.milliseconds);
1096 }
1097
1098 // If validation is needed, verify that the uncalculated date fields
1099 // match the calculated date fields. We do this before we set the
1100 // timezone offset, which will skew all of the dates.
1101 //
1102 // Don't need to check the day of week as it is guaranteed to be
1103 // correct or return false below.
1104 if (validation &&
1105 (this.year != undefined && this.year != date.getFullYear() ||
1106 this.month != undefined && this.month != date.getMonth() ||
1107 this.day != undefined && this.day != date.getDate() ||
1108 this.hours >= 24 || this.minutes >= 60 || this.seconds >= 60 ||
1109 this.milliseconds >= 1000)) {
1110 return false;
1111 }
1112
1113 // adjust time zone
1114 if (this.tzOffset != undefined) {
1115 var offset = date.getTimezoneOffset();
1116 date.setTime(date.getTime() + (this.tzOffset - offset) * 60 * 1000);
1117 }
1118
1119 // resolve ambiguous year if needed
1120 if (this.ambiguousYear) { // the two-digit year == the default start year
1121 var defaultCenturyStart = new Date();
1122 defaultCenturyStart.setFullYear(
1123 defaultCenturyStart.getFullYear() -
1124 goog.i18n.DateTimeParse.ambiguousYearCenturyStart);
1125 if (date.getTime() < defaultCenturyStart.getTime()) {
1126 date.setFullYear(defaultCenturyStart.getFullYear() + 100);
1127 }
1128 }
1129
1130 // dayOfWeek, validation only
1131 if (this.dayOfWeek != undefined) {
1132 if (this.day == undefined) {
1133 // adjust to the nearest day of the week
1134 var adjustment = (7 + this.dayOfWeek - date.getDay()) % 7;
1135 if (adjustment > 3) {
1136 adjustment -= 7;
1137 }
1138 var orgMonth = date.getMonth();
1139 date.setDate(date.getDate() + adjustment);
1140
1141 // don't let it switch month
1142 if (date.getMonth() != orgMonth) {
1143 date.setDate(date.getDate() + (adjustment > 0 ? -7 : 7));
1144 }
1145 } else if (this.dayOfWeek != date.getDay()) {
1146 return false;
1147 }
1148 }
1149 return true;
