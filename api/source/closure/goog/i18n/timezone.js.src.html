28/**
29 * TimeZone class implemented a time zone resolution and name information
30 * source for client applications. The time zone object is initiated from
31 * a time zone information object. Application can initiate a time zone
32 * statically, or it may choose to initiate from a data obtained from server.
33 * Each time zone information array is small, but the whole set of data
34 * is too much for client application to download. If end user is allowed to
35 * change time zone setting, dynamic retrieval should be the method to use.
36 * In case only time zone offset is known, there is a decent fallback
37 * that only use the time zone offset to create a TimeZone object.
38 * A whole set of time zone information array was available under
39 * http://go/js_locale_data. It is generated based on CLDR and
40 * Olson time zone data base (through pytz), and will be updated timely.
41 *
42 * @constructor
43 * @final
44 */
45goog.i18n.TimeZone = function() {
46 /**
47 * The standard time zone id.
48 * @type {string}
49 * @private
50 */
51 this.timeZoneId_;
52
53
54 /**
55 * The standard, non-daylight time zone offset, in minutes WEST of UTC.
56 * @type {number}
57 * @private
58 */
59 this.standardOffset_;
60
61
62 /**
63 * An array of strings that can have 2 or 4 elements. The first two elements
64 * are the long and short names for standard time in this time zone, and the
65 * last two elements (if present) are the long and short names for daylight
66 * time in this time zone.
67 * @type {Array<string>}
68 * @private
69 */
70 this.tzNames_;
71
72
73 /**
74 * This array specifies the Daylight Saving Time transitions for this time
75 * zone. This is a flat array of numbers which are interpreted in pairs:
76 * [time1, adjustment1, time2, adjustment2, ...] where each time is a DST
77 * transition point given as a number of hours since 00:00 UTC, January 1,
78 * 1970, and each adjustment is the adjustment to apply for times after the
79 * DST transition, given as minutes EAST of UTC.
80 * @type {Array<number>}
81 * @private
82 */
83 this.transitions_;
84};

107/**
108 * This factory method creates a time zone instance. It takes either an object
109 * containing complete time zone information, or a single number representing a
110 * constant time zone offset. If the latter form is used, DST functionality is
111 * not available.
112 *
113 * @param {number|Object} timeZoneData If this parameter is a number, it should
114 * indicate minutes WEST of UTC to be used as a constant time zone offset.
115 * Otherwise, it should be an object with these four fields:
116 * <ul>
117 * <li>id: A string ID for the time zone.
118 * <li>std_offset: The standard time zone offset in minutes EAST of UTC.
119 * <li>names: An array of four names (standard short name, standard long
120 * name, daylight short name, daylight long, name)
121 * <li>transitions: An array of numbers which are interpreted in pairs:
122 * [time1, adjustment1, time2, adjustment2, ...] where each time is
123 * a DST transition point given as a number of hours since 00:00 UTC,
124 * January 1, 1970, and each adjustment is the adjustment to apply
125 * for times after the DST transition, given as minutes EAST of UTC.
126 * </ul>
127 * @return {!goog.i18n.TimeZone} A goog.i18n.TimeZone object for the given
128 * time zone data.
129 */
130goog.i18n.TimeZone.createTimeZone = function(timeZoneData) {
131 if (typeof timeZoneData == 'number') {
132 return goog.i18n.TimeZone.createSimpleTimeZone_(timeZoneData);
133 }
134 var tz = new goog.i18n.TimeZone();
135 tz.timeZoneId_ = timeZoneData['id'];
136 tz.standardOffset_ = -timeZoneData['std_offset'];
137 tz.tzNames_ = timeZoneData['names'];
138 tz.transitions_ = timeZoneData['transitions'];
139 return tz;
140};

143/**
144 * This factory method creates a time zone object with a constant offset.
145 * @param {number} timeZoneOffsetInMinutes Offset in minutes WEST of UTC.
146 * @return {!goog.i18n.TimeZone} A time zone object with the given constant
147 * offset. Note that the time zone ID of this object will use the POSIX
148 * convention, which has a reversed sign ("Etc/GMT+8" means UTC-8 or PST).
149 * @private
150 */
151goog.i18n.TimeZone.createSimpleTimeZone_ = function(timeZoneOffsetInMinutes) {
152 var tz = new goog.i18n.TimeZone();
153 tz.standardOffset_ = timeZoneOffsetInMinutes;
154 tz.timeZoneId_ =
155 goog.i18n.TimeZone.composePosixTimeZoneID_(timeZoneOffsetInMinutes);
156 var str = goog.i18n.TimeZone.composeUTCString_(timeZoneOffsetInMinutes);
157 tz.tzNames_ = [str, str];
158 tz.transitions_ = [];
159 return tz;
160};

163/**
164 * Generate a GMT-relative string for a constant time zone offset.
165 * @param {number} offset The time zone offset in minutes WEST of UTC.
166 * @return {string} The GMT string for this offset, which will indicate
167 * hours EAST of UTC.
168 * @private
169 */
170goog.i18n.TimeZone.composeGMTString_ = function(offset) {
171 var parts = ['GMT'];
172 parts.push(offset <= 0 ? '+' : '-');
173 offset = Math.abs(offset);
174 parts.push(goog.string.padNumber(Math.floor(offset / 60) % 100, 2),
175 ':', goog.string.padNumber(offset % 60, 2));
176 return parts.join('');
177};

180/**
181 * Generate a POSIX time zone ID for a constant time zone offset.
182 * @param {number} offset The time zone offset in minutes WEST of UTC.
183 * @return {string} The POSIX time zone ID for this offset, which will indicate
184 * hours WEST of UTC.
185 * @private
186 */
187goog.i18n.TimeZone.composePosixTimeZoneID_ = function(offset) {
188 if (offset == 0) {
189 return 'Etc/GMT';
190 }
191 var parts = ['Etc/GMT', offset < 0 ? '-' : '+'];
192 offset = Math.abs(offset);
193 parts.push(Math.floor(offset / 60) % 100);
194 offset = offset % 60;
195 if (offset != 0) {
196 parts.push(':', goog.string.padNumber(offset, 2));
197 }
198 return parts.join('');
199};

202/**
203 * Generate a UTC-relative string for a constant time zone offset.
204 * @param {number} offset The time zone offset in minutes WEST of UTC.
205 * @return {string} The UTC string for this offset, which will indicate
206 * hours EAST of UTC.
207 * @private
208 */
209goog.i18n.TimeZone.composeUTCString_ = function(offset) {
210 if (offset == 0) {
211 return 'UTC';
212 }
213 var parts = ['UTC', offset < 0 ? '+' : '-'];
214 offset = Math.abs(offset);
215 parts.push(Math.floor(offset / 60) % 100);
216 offset = offset % 60;
217 if (offset != 0) {
218 parts.push(':', offset);
219 }
220 return parts.join('');

224/**
225 * Convert the contents of time zone object to a timeZoneData object, suitable
226 * for passing to goog.i18n.TimeZone.createTimeZone.
227 * @return {!Object} A timeZoneData object (see the documentation for
228 * goog.i18n.TimeZone.createTimeZone).
229 */
230goog.i18n.TimeZone.prototype.getTimeZoneData = function() {
231 return {
232 'id': this.timeZoneId_,
233 'std_offset': -this.standardOffset_, // note createTimeZone flips the sign
234 'names': goog.array.clone(this.tzNames_), // avoid aliasing the array
235 'transitions': goog.array.clone(this.transitions_) // avoid aliasing
236 };

240/**
241 * Return the DST adjustment to the time zone offset for a given time.
242 * While Daylight Saving Time is in effect, this number is positive.
243 * Otherwise, it is zero.
244 * @param {goog.date.DateLike} date The time to check.
245 * @return {number} The DST adjustment in minutes EAST of UTC.
246 */
247goog.i18n.TimeZone.prototype.getDaylightAdjustment = function(date) {
248 var timeInMs = Date.UTC(date.getUTCFullYear(), date.getUTCMonth(),
249 date.getUTCDate(), date.getUTCHours(),
250 date.getUTCMinutes());
251 var timeInHours = timeInMs / goog.i18n.TimeZone.MILLISECONDS_PER_HOUR_;
252 var index = 0;
253 while (index < this.transitions_.length &&
254 timeInHours >= this.transitions_[index]) {
255 index += 2;
256 }
257 return (index == 0) ? 0 : this.transitions_[index - 1];
258};
