35/**
36 * Constructor of NumberFormat.
37 * @param {number|string} pattern The number that indicates a predefined
38 * number format pattern.
39 * @param {string=} opt_currency Optional international currency
40 * code. This determines the currency code/symbol used in format/parse. If
41 * not given, the currency code for current locale will be used.
42 * @param {number=} opt_currencyStyle currency style, value defined in
43 * goog.i18n.NumberFormat.CurrencyStyle.
44 * @constructor
45 */
46goog.i18n.NumberFormat = function(pattern, opt_currency, opt_currencyStyle) {
47 /** @private {string} */
48 this.intlCurrencyCode_ = opt_currency ||
49 goog.i18n.NumberFormatSymbols.DEF_CURRENCY_CODE;
50
51 /** @private {number} */
52 this.currencyStyle_ = opt_currencyStyle ||
53 goog.i18n.NumberFormat.CurrencyStyle.LOCAL;
54
55 /** @private {number} */
56 this.maximumIntegerDigits_ = 40;
57 /** @private {number} */
58 this.minimumIntegerDigits_ = 1;
59 /** @private {number} */
60 this.significantDigits_ = 0; // invariant, <= maximumFractionDigits
61 /** @private {number} */
62 this.maximumFractionDigits_ = 3; // invariant, >= minFractionDigits
63 /** @private {number} */
64 this.minimumFractionDigits_ = 0;
65 /** @private {number} */
66 this.minExponentDigits_ = 0;
67 /** @private {boolean} */
68 this.useSignForPositiveExponent_ = false;
69
70 /**
71 * Whether to show trailing zeros in the fraction when significantDigits_ is
72 * positive.
73 * @private {boolean}
74 */
75 this.showTrailingZeros_ = false;
76
77 /** @private {string} */
78 this.positivePrefix_ = '';
79 /** @private {string} */
80 this.positiveSuffix_ = '';
81 /** @private {string} */
82 this.negativePrefix_ = '-';
83 /** @private {string} */
84 this.negativeSuffix_ = '';
85
86 // The multiplier for use in percent, per mille, etc.
87 /** @private {number} */
88 this.multiplier_ = 1;
89 /** @private {number} */
90 this.groupingSize_ = 3;
91 /** @private {boolean} */
92 this.decimalSeparatorAlwaysShown_ = false;
93 /** @private {boolean} */
94 this.useExponentialNotation_ = false;
95 /** @private {goog.i18n.NumberFormat.CompactStyle} */
96 this.compactStyle_ = goog.i18n.NumberFormat.CompactStyle.NONE;
97
98 /**
99 * The number to base the formatting on when using compact styles, or null
100 * if formatting should not be based on another number.
101 * @type {?number}
102 * @private
103 */
104 this.baseFormattingNumber_ = null;
105
106 /** @private {string} */
107 this.pattern_;
108
109 if (typeof pattern == 'number') {
110 this.applyStandardPattern_(pattern);
111 } else {
112 this.applyPattern_(pattern);
113 }
114};

280/**
281 * Apply provided pattern, result are stored in member variables.
282 *
283 * @param {string} pattern String pattern being applied.
284 * @private
285 */
286goog.i18n.NumberFormat.prototype.applyPattern_ = function(pattern) {
287 this.pattern_ = pattern.replace(/ /g, '\u00a0');
288 var pos = [0];
289
290 this.positivePrefix_ = this.parseAffix_(pattern, pos);
291 var trunkStart = pos[0];
292 this.parseTrunk_(pattern, pos);
293 var trunkLen = pos[0] - trunkStart;
294 this.positiveSuffix_ = this.parseAffix_(pattern, pos);
295 if (pos[0] < pattern.length &&
296 pattern.charAt(pos[0]) == goog.i18n.NumberFormat.PATTERN_SEPARATOR_) {
297 pos[0]++;
298 this.negativePrefix_ = this.parseAffix_(pattern, pos);
299 // we assume this part is identical to positive part.
300 // user must make sure the pattern is correctly constructed.
301 pos[0] += trunkLen;
302 this.negativeSuffix_ = this.parseAffix_(pattern, pos);
303 } else {
304 // if no negative affix specified, they share the same positive affix
305 this.negativePrefix_ = this.positivePrefix_ + this.negativePrefix_;
306 this.negativeSuffix_ += this.positiveSuffix_;
307 }
308};

311/**
312 * Apply a predefined pattern to NumberFormat object.
313 * @param {number} patternType The number that indicates a predefined number
314 * format pattern.
315 * @private
316 */
317goog.i18n.NumberFormat.prototype.applyStandardPattern_ = function(patternType) {
318 switch (patternType) {
319 case goog.i18n.NumberFormat.Format.DECIMAL:
320 this.applyPattern_(goog.i18n.NumberFormatSymbols.DECIMAL_PATTERN);
321 break;
322 case goog.i18n.NumberFormat.Format.SCIENTIFIC:
323 this.applyPattern_(goog.i18n.NumberFormatSymbols.SCIENTIFIC_PATTERN);
324 break;
325 case goog.i18n.NumberFormat.Format.PERCENT:
326 this.applyPattern_(goog.i18n.NumberFormatSymbols.PERCENT_PATTERN);
327 break;
328 case goog.i18n.NumberFormat.Format.CURRENCY:
329 this.applyPattern_(goog.i18n.currency.adjustPrecision(
330 goog.i18n.NumberFormatSymbols.CURRENCY_PATTERN,
331 this.intlCurrencyCode_));
332 break;
333 case goog.i18n.NumberFormat.Format.COMPACT_SHORT:
334 this.applyCompactStyle_(goog.i18n.NumberFormat.CompactStyle.SHORT);
335 break;
336 case goog.i18n.NumberFormat.Format.COMPACT_LONG:
337 this.applyCompactStyle_(goog.i18n.NumberFormat.CompactStyle.LONG);
338 break;
339 default:
340 throw Error('Unsupported pattern type.');
341 }
342};

360/**
361 * Parses text string to produce a Number.
362 *
363 * This method attempts to parse text starting from position "opt_pos" if it
364 * is given. Otherwise the parse will start from the beginning of the text.
365 * When opt_pos presents, opt_pos will be updated to the character next to where
366 * parsing stops after the call. If an error occurs, opt_pos won't be updated.
367 *
368 * @param {string} text The string to be parsed.
369 * @param {Array<number>=} opt_pos Position to pass in and get back.
370 * @return {number} Parsed number. This throws an error if the text cannot be
371 * parsed.
372 */
373goog.i18n.NumberFormat.prototype.parse = function(text, opt_pos) {
374 var pos = opt_pos || [0];
375
376 if (this.compactStyle_ != goog.i18n.NumberFormat.CompactStyle.NONE) {
377 throw Error('Parsing of compact numbers is unimplemented');
378 }
379
380 var ret = NaN;
381
382 // we don't want to handle 2 kind of space in parsing, normalize it to nbsp
383 text = text.replace(/ /g, '\u00a0');
384
385 var gotPositive = text.indexOf(this.positivePrefix_, pos[0]) == pos[0];
386 var gotNegative = text.indexOf(this.negativePrefix_, pos[0]) == pos[0];
387
388 // check for the longest match
389 if (gotPositive && gotNegative) {
390 if (this.positivePrefix_.length > this.negativePrefix_.length) {
391 gotNegative = false;
392 } else if (this.positivePrefix_.length < this.negativePrefix_.length) {
393 gotPositive = false;
394 }
395 }
396
397 if (gotPositive) {
398 pos[0] += this.positivePrefix_.length;
399 } else if (gotNegative) {
400 pos[0] += this.negativePrefix_.length;
401 }
402
403 // process digits or Inf, find decimal position
404 if (text.indexOf(goog.i18n.NumberFormatSymbols.INFINITY, pos[0]) == pos[0]) {
405 pos[0] += goog.i18n.NumberFormatSymbols.INFINITY.length;
406 ret = Infinity;
407 } else {
408 ret = this.parseNumber_(text, pos);
409 }
410
411 // check for suffix
412 if (gotPositive) {
413 if (!(text.indexOf(this.positiveSuffix_, pos[0]) == pos[0])) {
414 return NaN;
415 }
416 pos[0] += this.positiveSuffix_.length;
417 } else if (gotNegative) {
418 if (!(text.indexOf(this.negativeSuffix_, pos[0]) == pos[0])) {
419 return NaN;
420 }
421 pos[0] += this.negativeSuffix_.length;
422 }
423
424 return gotNegative ? -ret : ret;
425};

428/**
429 * This function will parse a "localized" text into a Number. It needs to
430 * handle locale specific decimal, grouping, exponent and digits.
431 *
432 * @param {string} text The text that need to be parsed.
433 * @param {Array<number>} pos In/out parsing position. In case of failure,
434 * pos value won't be changed.
435 * @return {number} Number value, or NaN if nothing can be parsed.
436 * @private
437 */
438goog.i18n.NumberFormat.prototype.parseNumber_ = function(text, pos) {
439 var sawDecimal = false;
440 var sawExponent = false;
441 var sawDigit = false;
442 var scale = 1;
443 var decimal = goog.i18n.NumberFormatSymbols.DECIMAL_SEP;
444 var grouping = goog.i18n.NumberFormatSymbols.GROUP_SEP;
445 var exponentChar = goog.i18n.NumberFormatSymbols.EXP_SYMBOL;
446
447 if (this.compactStyle_ != goog.i18n.NumberFormat.CompactStyle.NONE) {
448 throw Error('Parsing of compact style numbers is not implemented');
449 }
450
451 var normalizedText = '';
452 for (; pos[0] < text.length; pos[0]++) {
453 var ch = text.charAt(pos[0]);
454 var digit = this.getDigit_(ch);
455 if (digit >= 0 && digit <= 9) {
456 normalizedText += digit;
457 sawDigit = true;
458 } else if (ch == decimal.charAt(0)) {
459 if (sawDecimal || sawExponent) {
460 break;
461 }
462 normalizedText += '.';
463 sawDecimal = true;
464 } else if (ch == grouping.charAt(0) &&
465 ('\u00a0' != grouping.charAt(0) ||
466 pos[0] + 1 < text.length &&
467 this.getDigit_(text.charAt(pos[0] + 1)) >= 0)) {
468 // Got a grouping character here. When grouping character is nbsp, need
469 // to make sure the character following it is a digit.
470 if (sawDecimal || sawExponent) {
471 break;
472 }
473 continue;
474 } else if (ch == exponentChar.charAt(0)) {
475 if (sawExponent) {
476 break;
477 }
478 normalizedText += 'E';
479 sawExponent = true;
480 } else if (ch == '+' || ch == '-') {
481 normalizedText += ch;
482 } else if (ch == goog.i18n.NumberFormatSymbols.PERCENT.charAt(0)) {
483 if (scale != 1) {
484 break;
485 }
486 scale = 100;
487 if (sawDigit) {
488 pos[0]++; // eat this character if parse end here
489 break;
490 }
491 } else if (ch == goog.i18n.NumberFormatSymbols.PERMILL.charAt(0)) {
492 if (scale != 1) {
493 break;
494 }
495 scale = 1000;
496 if (sawDigit) {
497 pos[0]++; // eat this character if parse end here
498 break;
499 }
500 } else {
501 break;
502 }
503 }
504 return parseFloat(normalizedText) / scale;
505};

508/**
509 * Formats a Number to produce a string.
510 *
511 * @param {number} number The Number to be formatted.
512 * @return {string} The formatted number string.
513 */
514goog.i18n.NumberFormat.prototype.format = function(number) {
515 if (isNaN(number)) {
516 return goog.i18n.NumberFormatSymbols.NAN;
517 }
518
519 var parts = [];
520 var baseFormattingNumber = goog.isNull(this.baseFormattingNumber_) ?
521 number :
522 this.baseFormattingNumber_;
523 var unit = this.getUnitAfterRounding_(baseFormattingNumber, number);
524 number /= Math.pow(10, unit.divisorBase);
525
526 parts.push(unit.prefix);
527
528 // in icu code, it is commented that certain computation need to keep the
529 // negative sign for 0.
530 var isNegative = number < 0.0 || number == 0.0 && 1 / number < 0.0;
531
532 parts.push(isNegative ? this.negativePrefix_ : this.positivePrefix_);
533
534 if (!isFinite(number)) {
535 parts.push(goog.i18n.NumberFormatSymbols.INFINITY);
536 } else {
537 // convert number to non-negative value
538 number *= isNegative ? -1 : 1;
539
540 number *= this.multiplier_;
541 this.useExponentialNotation_ ?
542 this.subformatExponential_(number, parts) :
543 this.subformatFixed_(number, this.minimumIntegerDigits_, parts);
544 }
545
546 parts.push(isNegative ? this.negativeSuffix_ : this.positiveSuffix_);
547 parts.push(unit.suffix);
548
549 return parts.join('');
550};

553/**
554 * Round a number into an integer and fractional part
555 * based on the rounding rules for this NumberFormat.
556 * @param {number} number The number to round.
557 * @return {{intValue: number, fracValue: number}} The integer and fractional
558 * part after rounding.
559 * @private
560 */
561goog.i18n.NumberFormat.prototype.roundNumber_ = function(number) {
562 var power = Math.pow(10, this.maximumFractionDigits_);
563 var shiftedNumber = this.significantDigits_ <= 0 ?
564 Math.round(number * power) :
565 Math.round(this.roundToSignificantDigits_(
566 number * power,
567 this.significantDigits_,
568 this.maximumFractionDigits_));
569
570 var intValue, fracValue;
571 if (isFinite(shiftedNumber)) {
572 intValue = Math.floor(shiftedNumber / power);
573 fracValue = Math.floor(shiftedNumber - intValue * power);
574 } else {
575 intValue = number;
576 fracValue = 0;
577 }
578 return {intValue: intValue, fracValue: fracValue};
579};

582/**
583 * Formats a Number in fraction format.
584 *
585 * @param {number} number
586 * @param {number} minIntDigits Minimum integer digits.
587 * @param {Array<string>} parts
588 * This array holds the pieces of formatted string.
589 * This function will add its formatted pieces to the array.
590 * @private
591 */
592goog.i18n.NumberFormat.prototype.subformatFixed_ =
593 function(number, minIntDigits, parts) {
594 if (this.minimumFractionDigits_ > this.maximumFractionDigits_) {
595 throw Error('Min value must be less than max value');
596 }
597
598 var rounded = this.roundNumber_(number);
599 var power = Math.pow(10, this.maximumFractionDigits_);
600 var intValue = rounded.intValue;
601 var fracValue = rounded.fracValue;
602
603 var numIntDigits = (intValue == 0) ? 0 : this.intLog10_(intValue) + 1;
604 var fractionPresent = this.minimumFractionDigits_ > 0 || fracValue > 0 ||
605 (this.showTrailingZeros_ && numIntDigits < this.significantDigits_);
606 var minimumFractionDigits = this.minimumFractionDigits_;
607 if (fractionPresent) {
608 if (this.showTrailingZeros_ && this.significantDigits_ > 0) {
609 minimumFractionDigits = this.significantDigits_ - numIntDigits;
610 } else {
611 minimumFractionDigits = this.minimumFractionDigits_;
612 }
613 }
614
615 var intPart = '';
616 var translatableInt = intValue;
617 while (translatableInt > 1E20) {
618 // here it goes beyond double precision, add '0' make it look better
619 intPart = '0' + intPart;
620 translatableInt = Math.round(translatableInt / 10);
621 }
622 intPart = translatableInt + intPart;
623
624 var decimal = goog.i18n.NumberFormatSymbols.DECIMAL_SEP;
625 var grouping = goog.i18n.NumberFormatSymbols.GROUP_SEP;
626 var zeroCode = goog.i18n.NumberFormat.enforceAsciiDigits_ ?
627 48 /* ascii '0' */ :
628 goog.i18n.NumberFormatSymbols.ZERO_DIGIT.charCodeAt(0);
629 var digitLen = intPart.length;
630
631 if (intValue > 0 || minIntDigits > 0) {
632 for (var i = digitLen; i < minIntDigits; i++) {
633 parts.push(String.fromCharCode(zeroCode));
634 }
635
636 for (var i = 0; i < digitLen; i++) {
637 parts.push(String.fromCharCode(zeroCode + intPart.charAt(i) * 1));
638
639 if (digitLen - i > 1 && this.groupingSize_ > 0 &&
640 ((digitLen - i) % this.groupingSize_ == 1)) {
641 parts.push(grouping);
642 }
643 }
644 } else if (!fractionPresent) {
645 // If there is no fraction present, and we haven't printed any
646 // integer digits, then print a zero.
647 parts.push(String.fromCharCode(zeroCode));
648 }
649
650 // Output the decimal separator if we always do so.
651 if (this.decimalSeparatorAlwaysShown_ || fractionPresent) {
652 parts.push(decimal);
653 }
654
655 var fracPart = '' + (fracValue + power);
656 var fracLen = fracPart.length;
657 while (fracPart.charAt(fracLen - 1) == '0' &&
658 fracLen > minimumFractionDigits + 1) {
659 fracLen--;
660 }
661
662 for (var i = 1; i < fracLen; i++) {
663 parts.push(String.fromCharCode(zeroCode + fracPart.charAt(i) * 1));
664 }
665};

668/**
669 * Formats exponent part of a Number.
670 *
671 * @param {number} exponent Exponential value.
672 * @param {Array<string>} parts The array that holds the pieces of formatted
673 * string. This function will append more formatted pieces to the array.
674 * @private
675 */
676goog.i18n.NumberFormat.prototype.addExponentPart_ = function(exponent, parts) {
677 parts.push(goog.i18n.NumberFormatSymbols.EXP_SYMBOL);
678
679 if (exponent < 0) {
680 exponent = -exponent;
681 parts.push(goog.i18n.NumberFormatSymbols.MINUS_SIGN);
682 } else if (this.useSignForPositiveExponent_) {
683 parts.push(goog.i18n.NumberFormatSymbols.PLUS_SIGN);
684 }
685
686 var exponentDigits = '' + exponent;
687 var zeroChar = goog.i18n.NumberFormat.enforceAsciiDigits_ ? '0' :
688 goog.i18n.NumberFormatSymbols.ZERO_DIGIT;
689 for (var i = exponentDigits.length; i < this.minExponentDigits_; i++) {
690 parts.push(zeroChar);
691 }
692 parts.push(exponentDigits);
693};

695
696/**
697 * Formats Number in exponential format.
698 *
699 * @param {number} number Value need to be formated.
700 * @param {Array<string>} parts The array that holds the pieces of formatted
701 * string. This function will append more formatted pieces to the array.
702 * @private
703 */
704goog.i18n.NumberFormat.prototype.subformatExponential_ =
705 function(number, parts) {
706 if (number == 0.0) {
707 this.subformatFixed_(number, this.minimumIntegerDigits_, parts);
708 this.addExponentPart_(0, parts);
709 return;
710 }
711
712 var exponent = goog.math.safeFloor(Math.log(number) / Math.log(10));
713 number /= Math.pow(10, exponent);
714
715 var minIntDigits = this.minimumIntegerDigits_;
716 if (this.maximumIntegerDigits_ > 1 &&
717 this.maximumIntegerDigits_ > this.minimumIntegerDigits_) {
718 // A repeating range is defined; adjust to it as follows.
719 // If repeat == 3, we have 6,5,4=>3; 3,2,1=>0; 0,-1,-2=>-3;
720 // -3,-4,-5=>-6, etc. This takes into account that the
721 // exponent we have here is off by one from what we expect;
722 // it is for the format 0.MMMMMx10^n.
723 while ((exponent % this.maximumIntegerDigits_) != 0) {
724 number *= 10;
725 exponent--;
726 }
727 minIntDigits = 1;
728 } else {
729 // No repeating range is defined; use minimum integer digits.
730 if (this.minimumIntegerDigits_ < 1) {
731 exponent++;
732 number /= 10;
733 } else {
734 exponent -= this.minimumIntegerDigits_ - 1;
735 number *= Math.pow(10, this.minimumIntegerDigits_ - 1);
736 }
737 }
738 this.subformatFixed_(number, minIntDigits, parts);
739 this.addExponentPart_(exponent, parts);
740};

863/**
864 * Parses affix part of pattern.
865 *
866 * @param {string} pattern Pattern string that need to be parsed.
867 * @param {Array<number>} pos One element position array to set and receive
868 * parsing position.
869 *
870 * @return {string} Affix received from parsing.
871 * @private
872 */
873goog.i18n.NumberFormat.prototype.parseAffix_ = function(pattern, pos) {
874 var affix = '';
875 var inQuote = false;
876 var len = pattern.length;
877
878 for (; pos[0] < len; pos[0]++) {
879 var ch = pattern.charAt(pos[0]);
880 if (ch == goog.i18n.NumberFormat.QUOTE_) {
881 if (pos[0] + 1 < len &&
882 pattern.charAt(pos[0] + 1) == goog.i18n.NumberFormat.QUOTE_) {
883 pos[0]++;
884 affix += '\''; // 'don''t'
885 } else {
886 inQuote = !inQuote;
887 }
888 continue;
889 }
890
891 if (inQuote) {
892 affix += ch;
893 } else {
894 switch (ch) {
895 case goog.i18n.NumberFormat.PATTERN_DIGIT_:
896 case goog.i18n.NumberFormat.PATTERN_ZERO_DIGIT_:
897 case goog.i18n.NumberFormat.PATTERN_GROUPING_SEPARATOR_:
898 case goog.i18n.NumberFormat.PATTERN_DECIMAL_SEPARATOR_:
899 case goog.i18n.NumberFormat.PATTERN_SEPARATOR_:
900 return affix;
901 case goog.i18n.NumberFormat.PATTERN_CURRENCY_SIGN_:
902 if ((pos[0] + 1) < len &&
903 pattern.charAt(pos[0] + 1) ==
904 goog.i18n.NumberFormat.PATTERN_CURRENCY_SIGN_) {
905 pos[0]++;
906 affix += this.intlCurrencyCode_;
907 } else {
908 switch (this.currencyStyle_) {
909 case goog.i18n.NumberFormat.CurrencyStyle.LOCAL:
910 affix += goog.i18n.currency.getLocalCurrencySign(
911 this.intlCurrencyCode_);
912 break;
913 case goog.i18n.NumberFormat.CurrencyStyle.GLOBAL:
914 affix += goog.i18n.currency.getGlobalCurrencySign(
915 this.intlCurrencyCode_);
916 break;
917 case goog.i18n.NumberFormat.CurrencyStyle.PORTABLE:
918 affix += goog.i18n.currency.getPortableCurrencySign(
919 this.intlCurrencyCode_);
920 break;
921 default:
922 break;
923 }
924 }
925 break;
926 case goog.i18n.NumberFormat.PATTERN_PERCENT_:
927 if (this.multiplier_ != 1) {
928 throw Error('Too many percent/permill');
929 }
930 this.multiplier_ = 100;
931 affix += goog.i18n.NumberFormatSymbols.PERCENT;
932 break;
933 case goog.i18n.NumberFormat.PATTERN_PER_MILLE_:
934 if (this.multiplier_ != 1) {
935 throw Error('Too many percent/permill');
936 }
937 this.multiplier_ = 1000;
938 affix += goog.i18n.NumberFormatSymbols.PERMILL;
939 break;
940 default:
941 affix += ch;
942 }
943 }
944 }
945
946 return affix;

950/**
951 * Parses the trunk part of a pattern.
952 *
953 * @param {string} pattern Pattern string that need to be parsed.
954 * @param {Array<number>} pos One element position array to set and receive
955 * parsing position.
956 * @private
957 */
958goog.i18n.NumberFormat.prototype.parseTrunk_ = function(pattern, pos) {
959 var decimalPos = -1;
960 var digitLeftCount = 0;
961 var zeroDigitCount = 0;
962 var digitRightCount = 0;
963 var groupingCount = -1;
964
965 var len = pattern.length;
966 for (var loop = true; pos[0] < len && loop; pos[0]++) {
967 var ch = pattern.charAt(pos[0]);
968 switch (ch) {
969 case goog.i18n.NumberFormat.PATTERN_DIGIT_:
970 if (zeroDigitCount > 0) {
971 digitRightCount++;
972 } else {
973 digitLeftCount++;
974 }
975 if (groupingCount >= 0 && decimalPos < 0) {
976 groupingCount++;
977 }
978 break;
979 case goog.i18n.NumberFormat.PATTERN_ZERO_DIGIT_:
980 if (digitRightCount > 0) {
981 throw Error('Unexpected "0" in pattern "' + pattern + '"');
982 }
983 zeroDigitCount++;
984 if (groupingCount >= 0 && decimalPos < 0) {
985 groupingCount++;
986 }
987 break;
988 case goog.i18n.NumberFormat.PATTERN_GROUPING_SEPARATOR_:
989 groupingCount = 0;
990 break;
991 case goog.i18n.NumberFormat.PATTERN_DECIMAL_SEPARATOR_:
992 if (decimalPos >= 0) {
993 throw Error('Multiple decimal separators in pattern "' +
994 pattern + '"');
995 }
996 decimalPos = digitLeftCount + zeroDigitCount + digitRightCount;
997 break;
998 case goog.i18n.NumberFormat.PATTERN_EXPONENT_:
999 if (this.useExponentialNotation_) {
1000 throw Error('Multiple exponential symbols in pattern "' +
1001 pattern + '"');
1002 }
1003 this.useExponentialNotation_ = true;
1004 this.minExponentDigits_ = 0;
1005
1006 // exponent pattern can have a optional '+'.
1007 if ((pos[0] + 1) < len && pattern.charAt(pos[0] + 1) ==
1008 goog.i18n.NumberFormat.PATTERN_PLUS_) {
1009 pos[0]++;
1010 this.useSignForPositiveExponent_ = true;
1011 }
1012
1013 // Use lookahead to parse out the exponential part
1014 // of the pattern, then jump into phase 2.
1015 while ((pos[0] + 1) < len && pattern.charAt(pos[0] + 1) ==
1016 goog.i18n.NumberFormat.PATTERN_ZERO_DIGIT_) {
1017 pos[0]++;
1018 this.minExponentDigits_++;
1019 }
1020
1021 if ((digitLeftCount + zeroDigitCount) < 1 ||
1022 this.minExponentDigits_ < 1) {
1023 throw Error('Malformed exponential pattern "' + pattern + '"');
1024 }
1025 loop = false;
1026 break;
1027 default:
1028 pos[0]--;
1029 loop = false;
1030 break;
1031 }
1032 }
1033
1034 if (zeroDigitCount == 0 && digitLeftCount > 0 && decimalPos >= 0) {
1035 // Handle '###.###' and '###.' and '.###'
1036 var n = decimalPos;
1037 if (n == 0) { // Handle '.###'
1038 n++;
1039 }
1040 digitRightCount = digitLeftCount - n;
1041 digitLeftCount = n - 1;
1042 zeroDigitCount = 1;
1043 }
1044
1045 // Do syntax checking on the digits.
1046 if (decimalPos < 0 && digitRightCount > 0 ||
1047 decimalPos >= 0 && (decimalPos < digitLeftCount ||
1048 decimalPos > digitLeftCount + zeroDigitCount) ||
1049 groupingCount == 0) {
1050 throw Error('Malformed pattern "' + pattern + '"');
1051 }
1052 var totalDigits = digitLeftCount + zeroDigitCount + digitRightCount;
1053
1054 this.maximumFractionDigits_ = decimalPos >= 0 ? totalDigits - decimalPos : 0;
1055 if (decimalPos >= 0) {
1056 this.minimumFractionDigits_ = digitLeftCount + zeroDigitCount - decimalPos;
1057 if (this.minimumFractionDigits_ < 0) {
1058 this.minimumFractionDigits_ = 0;
1059 }
1060 }
1061
1062 // The effectiveDecimalPos is the position the decimal is at or would be at
1063 // if there is no decimal. Note that if decimalPos<0, then digitTotalCount ==
1064 // digitLeftCount + zeroDigitCount.
1065 var effectiveDecimalPos = decimalPos >= 0 ? decimalPos : totalDigits;
1066 this.minimumIntegerDigits_ = effectiveDecimalPos - digitLeftCount;
1067 if (this.useExponentialNotation_) {
1068 this.maximumIntegerDigits_ = digitLeftCount + this.minimumIntegerDigits_;
1069
1070 // in exponential display, we need to at least show something.
1071 if (this.maximumFractionDigits_ == 0 && this.minimumIntegerDigits_ == 0) {
1072 this.minimumIntegerDigits_ = 1;
1073 }
1074 }
1075
1076 this.groupingSize_ = Math.max(0, groupingCount);
1077 this.decimalSeparatorAlwaysShown_ = decimalPos == 0 ||
1078 decimalPos == totalDigits;
1079};

1100/**
1101 * Get compact unit for a certain number of digits
1102 *
1103 * @param {number} base The number of digits to get the unit for.
1104 * @param {string} plurality The plurality of the number.
1105 * @return {!goog.i18n.NumberFormat.CompactNumberUnit} The compact unit.
1106 * @private
1107 */
1108goog.i18n.NumberFormat.prototype.getUnitFor_ = function(base, plurality) {
1109 var table = this.compactStyle_ == goog.i18n.NumberFormat.CompactStyle.SHORT ?
1110 goog.i18n.CompactNumberFormatSymbols.COMPACT_DECIMAL_SHORT_PATTERN :
1111 goog.i18n.CompactNumberFormatSymbols.COMPACT_DECIMAL_LONG_PATTERN;
1112
1113 if (base < 3) {
1114 return goog.i18n.NumberFormat.NULL_UNIT_;
1115 } else {
1116 base = Math.min(14, base);
1117 var patterns = table[Math.pow(10, base)];
1118 if (!patterns) {
1119 return goog.i18n.NumberFormat.NULL_UNIT_;
1120 }
1121
1122 var pattern = patterns[plurality];
1123 if (!pattern || pattern == '0') {
1124 return goog.i18n.NumberFormat.NULL_UNIT_;
1125 }
1126
1127 var parts = /([^0]*)(0+)(.*)/.exec(pattern);
1128 if (!parts) {
1129 return goog.i18n.NumberFormat.NULL_UNIT_;
1130 }
1131
1132 return {
1133 prefix: parts[1],
1134 suffix: parts[3],
1135 divisorBase: base - (parts[2].length - 1)
1136 };
1137 }

1141/**
1142 * Get the compact unit divisor, accounting for rounding of the quantity.
1143 *
1144 * @param {number} formattingNumber The number to base the formatting on. The
1145 * unit will be calculated from this number.
1146 * @param {number} pluralityNumber The number to use for calculating the
1147 * plurality.
1148 * @return {!goog.i18n.NumberFormat.CompactNumberUnit} The unit after rounding.
1149 * @private
1150 */
1151goog.i18n.NumberFormat.prototype.getUnitAfterRounding_ =
1152 function(formattingNumber, pluralityNumber) {
1153 if (this.compactStyle_ == goog.i18n.NumberFormat.CompactStyle.NONE) {
1154 return goog.i18n.NumberFormat.NULL_UNIT_;
1155 }
1156
1157 formattingNumber = Math.abs(formattingNumber);
1158 pluralityNumber = Math.abs(pluralityNumber);
1159
1160 var initialPlurality = this.pluralForm_(formattingNumber);
1161 // Compute the exponent from the formattingNumber, to compute the unit.
1162 var base = formattingNumber <= 1 ? 0 : this.intLog10_(formattingNumber);
1163 var initialDivisor = this.getUnitFor_(base, initialPlurality).divisorBase;
1164 // Round both numbers based on the unit used.
1165 var pluralityAttempt = pluralityNumber / Math.pow(10, initialDivisor);
1166 var pluralityRounded = this.roundNumber_(pluralityAttempt);
1167 var formattingAttempt = formattingNumber / Math.pow(10, initialDivisor);
1168 var formattingRounded = this.roundNumber_(formattingAttempt);
1169 // Compute the plurality of the pluralityNumber when formatted using the name
1170 // units as the formattingNumber.
1171 var finalPlurality =
1172 this.pluralForm_(pluralityRounded.intValue + pluralityRounded.fracValue);
1173 // Get the final unit, using the rounded formatting number to get the correct
1174 // unit, and the plurality computed from the pluralityNumber.
1175 return this.getUnitFor_(
1176 initialDivisor + this.intLog10_(formattingRounded.intValue),
1177 finalPlurality);
