15/**
16 * @fileoverview Functions for encoding strings according to MIME
17 * standards, especially RFC 1522.
18 */
19goog.provide('goog.i18n.mime');
20goog.provide('goog.i18n.mime.encode');
21
22goog.require('goog.array');

25/**
26 * Regular expression for matching those characters that are outside the
27 * range that can be used in the quoted-printable encoding of RFC 1522:
28 * anything outside the 7-bit ASCII encoding, plus ?, =, _ or space.
29 * @type {RegExp}
30 * @private
31 */
32goog.i18n.mime.NONASCII_ = /[^!-<>@-^`-~]/g;
33
34
35/**
36 * Like goog.i18n.NONASCII_ but also omits double-quotes.
37 * @type {RegExp}
38 * @private
39 */
40goog.i18n.mime.NONASCII_NOQUOTE_ = /[^!#-<>@-^`-~]/g;
41
42
43/**
44 * Encodes a string for inclusion in a MIME header. The string is encoded
45 * in UTF-8 according to RFC 1522, using quoted-printable form.
46 * @param {string} str The string to encode.
47 * @param {boolean=} opt_noquote Whether double-quote characters should also
48 * be escaped (should be true if the result will be placed inside a
49 * quoted string for a parameter value in a MIME header).
50 * @return {string} The encoded string.
51 */
52goog.i18n.mime.encode = function(str, opt_noquote) {
53 var nonascii = opt_noquote ?
54 goog.i18n.mime.NONASCII_NOQUOTE_ : goog.i18n.mime.NONASCII_;
55
56 if (str.search(nonascii) >= 0) {
57 str = '=?UTF-8?Q?' + str.replace(nonascii,
58 /**
59 * @param {string} c The matched char.
60 * @return {string} The quoted-printable form of utf-8 encoding.
61 */
62 function(c) {
63 var i = c.charCodeAt(0);
64 if (i == 32) {
65 // Special case for space, which can be encoded as _ not =20
66 return '_';
67 }
68 var a = goog.array.concat('', goog.i18n.mime.getHexCharArray(c));
69 return a.join('=');
70 }) + '?=';
71 }
72 return str;
73};

76/**
77 * Get an array of UTF-8 hex codes for a given character.
78 * @param {string} c The matched character.
79 * @return {!Array<string>} A hex array representing the character.
80 */
81goog.i18n.mime.getHexCharArray = function(c) {
82 var i = c.charCodeAt(0);
83 var a = [];
84 // First convert the UCS-2 character into its UTF-8 bytes
85 if (i < 128) {
86 a.push(i);
87 } else if (i <= 0x7ff) {
88 a.push(
89 0xc0 + ((i >> 6) & 0x3f),
90 0x80 + (i & 0x3f));
91 } else if (i <= 0xffff) {
92 a.push(
93 0xe0 + ((i >> 12) & 0x3f),
94 0x80 + ((i >> 6) & 0x3f),
95 0x80 + (i & 0x3f));
96 } else {
97 // (This is defensive programming, since ecmascript isn't supposed
98 // to handle code points that take more than 16 bits.)
99 a.push(
100 0xf0 + ((i >> 18) & 0x3f),
101 0x80 + ((i >> 12) & 0x3f),
102 0x80 + ((i >> 6) & 0x3f),
103 0x80 + (i & 0x3f));
104 }
105 // Now convert those bytes into hex strings (don't do anything with
106 // a[0] as that's got the empty string that lets us use join())
107 for (i = a.length - 1; i >= 0; --i) {
108 a[i] = a[i].toString(16);
109 }
110 return a;
111};
