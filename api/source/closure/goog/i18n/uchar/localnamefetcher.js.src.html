15/**
16 * @fileoverview Object which fetches Unicode codepoint names that are locally
17 * stored in a bundled database. Currently, only invisible characters are
18 * covered by this database. See the goog.i18n.uChar.RemoteNameFetcher class for
19 * a remote database option.
20 */

22goog.provide('goog.i18n.uChar.LocalNameFetcher');

30/**
31 * Builds the NameFetcherLocal object. This is a simple object which retrieves
32 * character names from a local bundled database. This database only covers
33 * invisible characters. See the goog.i18n.uChar class for more details.
34 *
35 * @constructor
36 * @implements {goog.i18n.uChar.NameFetcher}
37 * @final
38 */
39goog.i18n.uChar.LocalNameFetcher = function() {
40};

43/**
44 * A reference to the LocalNameFetcher logger.
45 *
46 * @type {goog.log.Logger}
47 * @private
48 */
49goog.i18n.uChar.LocalNameFetcher.logger_ =

53/** @override */
54goog.i18n.uChar.LocalNameFetcher.prototype.prefetch = function(character) {
55};

58/** @override */
59goog.i18n.uChar.LocalNameFetcher.prototype.getName = function(character,
60 callback) {
61 var localName = goog.i18n.uChar.toName(character);
62 if (!localName) {
63 goog.i18n.uChar.LocalNameFetcher.logger_.
64 warning('No local name defined for character ' + character);
65 }
66 callback(localName);
67};
68
69
70/** @override */
71goog.i18n.uChar.LocalNameFetcher.prototype.isNameAvailable = function(
72 character) {
73 return !!goog.i18n.uChar.toName(character);
74};
