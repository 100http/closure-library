28/**
29 * Creates a 2D affine transform. An affine transform performs a linear
30 * mapping from 2D coordinates to other 2D coordinates that preserves the
31 * "straightness" and "parallelness" of lines.
32 *
33 * Such a coordinate transformation can be represented by a 3 row by 3 column
34 * matrix with an implied last row of [ 0 0 1 ]. This matrix transforms source
35 * coordinates (x,y) into destination coordinates (x',y') by considering them
36 * to be a column vector and multiplying the coordinate vector by the matrix
37 * according to the following process:
38 * <pre>
39 * [ x'] [ m00 m01 m02 ] [ x ] [ m00x + m01y + m02 ]
40 * [ y'] = [ m10 m11 m12 ] [ y ] = [ m10x + m11y + m12 ]
41 * [ 1 ] [ 0 0 1 ] [ 1 ] [ 1 ]
42 * </pre>
43 *
44 * This class is optimized for speed and minimizes calculations based on its
45 * knowledge of the underlying matrix (as opposed to say simply performing
46 * matrix multiplication).
47 *
48 * @param {number=} opt_m00 The m00 coordinate of the transform.
49 * @param {number=} opt_m10 The m10 coordinate of the transform.
50 * @param {number=} opt_m01 The m01 coordinate of the transform.
51 * @param {number=} opt_m11 The m11 coordinate of the transform.
52 * @param {number=} opt_m02 The m02 coordinate of the transform.
53 * @param {number=} opt_m12 The m12 coordinate of the transform.
54 * @constructor
55 * @final
56 */
57goog.graphics.AffineTransform = function(opt_m00, opt_m10, opt_m01,
58 opt_m11, opt_m02, opt_m12) {
59 if (arguments.length == 6) {
60 this.setTransform(/** @type {number} */ (opt_m00),
61 /** @type {number} */ (opt_m10),
62 /** @type {number} */ (opt_m01),
63 /** @type {number} */ (opt_m11),
64 /** @type {number} */ (opt_m02),
65 /** @type {number} */ (opt_m12));
66 } else if (arguments.length != 0) {
67 throw Error('Insufficient matrix parameters');
68 } else {
69 this.m00_ = this.m11_ = 1;
70 this.m10_ = this.m01_ = this.m02_ = this.m12_ = 0;
71 }
72};

75/**
76 * @return {boolean} Whether this transform is the identity transform.

78goog.graphics.AffineTransform.prototype.isIdentity = function() {
79 return this.m00_ == 1 && this.m10_ == 0 && this.m01_ == 0 &&
80 this.m11_ == 1 && this.m02_ == 0 && this.m12_ == 0;

93/**
94 * Sets this transform to the matrix specified by the 6 values.
95 *
96 * @param {number} m00 The m00 coordinate of the transform.
97 * @param {number} m10 The m10 coordinate of the transform.
98 * @param {number} m01 The m01 coordinate of the transform.
99 * @param {number} m11 The m11 coordinate of the transform.
100 * @param {number} m02 The m02 coordinate of the transform.
101 * @param {number} m12 The m12 coordinate of the transform.
102 * @return {!goog.graphics.AffineTransform} This affine transform.
103 */
104goog.graphics.AffineTransform.prototype.setTransform = function(m00, m10, m01,
105 m11, m02, m12) {
106 if (!goog.isNumber(m00) || !goog.isNumber(m10) || !goog.isNumber(m01) ||
107 !goog.isNumber(m11) || !goog.isNumber(m02) || !goog.isNumber(m12)) {
108 throw Error('Invalid transform parameters');
109 }
110 this.m00_ = m00;
111 this.m10_ = m10;
112 this.m01_ = m01;
113 this.m11_ = m11;
114 this.m02_ = m02;
115 this.m12_ = m12;
116 return this;
117};

243/**
244 * Concatenates this transform with a shear transformation.
245 *
246 * @param {number} shx The x shear factor.
247 * @param {number} shy The y shear factor.
248 * @return {!goog.graphics.AffineTransform} This affine transform.
249 */
250goog.graphics.AffineTransform.prototype.shear = function(shx, shy) {
251 var m00 = this.m00_;
252 var m10 = this.m10_;
253 this.m00_ += shy * this.m01_;
254 this.m10_ += shy * this.m11_;
255 this.m01_ += shx * m00;
256 this.m11_ += shx * m10;
257 return this;
258};

261/**
262 * Pre-concatenates this transform with a shear transformation.
263 * i.e. calculates the following matrix product:
264 *
265 * <pre>
266 * [ 1 shx 0] [m00 m01 m02]
267 * [shy 1 0] [m10 m11 m12]
268 * [ 0 0 1] [ 0 0 1]
269 * </pre>
270 *
271 * @param {number} shx The x shear factor.
272 * @param {number} shy The y shear factor.
273 * @return {!goog.graphics.AffineTransform} This affine transform.
274 */
275goog.graphics.AffineTransform.prototype.preShear = function(shx, shy) {
276 var m00 = this.m00_;
277 var m01 = this.m01_;
278 var m02 = this.m02_;
279 this.m00_ += shx * this.m10_;
280 this.m01_ += shx * this.m11_;
281 this.m02_ += shx * this.m12_;
282 this.m10_ += shy * m00;
283 this.m11_ += shy * m01;
284 this.m12_ += shy * m02;
285 return this;
286};

351/**
352 * Concatenates an affine transform to this transform.
353 *
354 * @param {!goog.graphics.AffineTransform} tx The transform to concatenate.
355 * @return {!goog.graphics.AffineTransform} This affine transform.
356 */
357goog.graphics.AffineTransform.prototype.concatenate = function(tx) {
358 var m0 = this.m00_;
359 var m1 = this.m01_;
360 this.m00_ = tx.m00_ * m0 + tx.m10_ * m1;
361 this.m01_ = tx.m01_ * m0 + tx.m11_ * m1;
362 this.m02_ += tx.m02_ * m0 + tx.m12_ * m1;
363
364 m0 = this.m10_;
365 m1 = this.m11_;
366 this.m10_ = tx.m00_ * m0 + tx.m10_ * m1;
367 this.m11_ = tx.m01_ * m0 + tx.m11_ * m1;
368 this.m12_ += tx.m02_ * m0 + tx.m12_ * m1;
369 return this;
370};

373/**
374 * Pre-concatenates an affine transform to this transform.
375 *
376 * @param {!goog.graphics.AffineTransform} tx The transform to preconcatenate.
377 * @return {!goog.graphics.AffineTransform} This affine transform.
378 */
379goog.graphics.AffineTransform.prototype.preConcatenate = function(tx) {
380 var m0 = this.m00_;
381 var m1 = this.m10_;
382 this.m00_ = tx.m00_ * m0 + tx.m01_ * m1;
383 this.m10_ = tx.m10_ * m0 + tx.m11_ * m1;
384
385 m0 = this.m01_;
386 m1 = this.m11_;
387 this.m01_ = tx.m00_ * m0 + tx.m01_ * m1;
388 this.m11_ = tx.m10_ * m0 + tx.m11_ * m1;
389
390 m0 = this.m02_;
391 m1 = this.m12_;
392 this.m02_ = tx.m00_ * m0 + tx.m01_ * m1 + tx.m02_;
393 this.m12_ = tx.m10_ * m0 + tx.m11_ * m1 + tx.m12_;
394 return this;
395};

398/**
399 * Transforms an array of coordinates by this transform and stores the result
400 * into a destination array.
401 *
402 * @param {!Array<number>} src The array containing the source points
403 * as x, y value pairs.
404 * @param {number} srcOff The offset to the first point to be transformed.
405 * @param {!Array<number>} dst The array into which to store the transformed
406 * point pairs.
407 * @param {number} dstOff The offset of the location of the first transformed
408 * point in the destination array.
409 * @param {number} numPts The number of points to tranform.
410 */
411goog.graphics.AffineTransform.prototype.transform = function(src, srcOff, dst,
412 dstOff, numPts) {
413 var i = srcOff;
414 var j = dstOff;
415 var srcEnd = srcOff + 2 * numPts;
416 while (i < srcEnd) {
417 var x = src[i++];
418 var y = src[i++];
419 dst[j++] = x * this.m00_ + y * this.m01_ + this.m02_;
420 dst[j++] = x * this.m10_ + y * this.m11_ + this.m12_;
421 }
422};

428goog.graphics.AffineTransform.prototype.getDeterminant = function() {

433/**
434 * Returns whether the transform is invertible. A transform is not invertible
435 * if the determinant is 0 or any value is non-finite or NaN.
436 *
437 * @return {boolean} Whether the transform is invertible.
438 */
439goog.graphics.AffineTransform.prototype.isInvertible = function() {
440 var det = this.getDeterminant();
441 return goog.math.isFiniteNumber(det) &&
442 goog.math.isFiniteNumber(this.m02_) &&
443 goog.math.isFiniteNumber(this.m12_) &&
444 det != 0;
445};

448/**
449 * @return {!goog.graphics.AffineTransform} An AffineTransform object
450 * representing the inverse transformation.
451 */
452goog.graphics.AffineTransform.prototype.createInverse = function() {
453 var det = this.getDeterminant();
454 return new goog.graphics.AffineTransform(
455 this.m11_ / det,
456 -this.m10_ / det,
457 -this.m01_ / det,
458 this.m00_ / det,
459 (this.m01_ * this.m12_ - this.m11_ * this.m02_) / det,
460 (this.m10_ * this.m02_ - this.m00_ * this.m12_) / det);
461};

569/**
570 * Compares two affine transforms for equality.
571 *
572 * @param {goog.graphics.AffineTransform} tx The other affine transform.
573 * @return {boolean} whether the two transforms are equal.
574 */
575goog.graphics.AffineTransform.prototype.equals = function(tx) {
576 if (this == tx) {
577 return true;
578 }
579 if (!tx) {
580 return false;
581 }
582 return this.m00_ == tx.m00_ &&
583 this.m01_ == tx.m01_ &&
584 this.m02_ == tx.m02_ &&
585 this.m10_ == tx.m10_ &&
586 this.m11_ == tx.m11_ &&
587 this.m12_ == tx.m12_;
