33/**
34 * Base class for the different graphics. You should never construct objects
35 * of this class. Instead us goog.graphics.createGraphics
36 * @param {number|string} width The width in pixels or percent.
37 * @param {number|string} height The height in pixels or percent.
38 * @param {?number=} opt_coordWidth Optional coordinate system width - if
39 * omitted or null, defaults to same as width.
40 * @param {?number=} opt_coordHeight Optional coordinate system height - if
41 * omitted or null, defaults to same as height.
42 * @param {goog.dom.DomHelper=} opt_domHelper The DOM helper object for the
43 * document we want to render in.
44 * @constructor
45 * @extends {goog.ui.Component}
46 */
47goog.graphics.AbstractGraphics = function(width, height,
48 opt_coordWidth, opt_coordHeight,
49 opt_domHelper) {
50 goog.ui.Component.call(this, opt_domHelper);
51
52 /**
53 * Width of graphics in pixels or percentage points.
54 * @type {number|string}
55 * @protected
56 */
57 this.width = width;
58
59 /**
60 * Height of graphics in pixels or precentage points.
61 * @type {number|string}
62 * @protected
63 */
64 this.height = height;
65
66 /**
67 * Width of coordinate system in units.
68 * @type {?number}
69 * @protected
70 */
71 this.coordWidth = opt_coordWidth || null;
72
73 /**
74 * Height of coordinate system in units.
75 * @type {?number}
76 * @protected
77 */
78 this.coordHeight = opt_coordHeight || null;
79};
80goog.inherits(goog.graphics.AbstractGraphics, goog.ui.Component);

173/**
174 * @return {goog.math.Size?} Returns the number of pixels spanned by the
175 * surface, or null if the size could not be computed due to the size being
176 * specified in percentage points and the component not being in the
177 * document.
178 */
179goog.graphics.AbstractGraphics.prototype.getPixelSize = function() {
180 if (this.isInDocument()) {
181 return goog.style.getSize(this.getElement());
182 }
183 if (goog.isNumber(this.width) && goog.isNumber(this.height)) {
184 return new goog.math.Size(this.width, this.height);
185 }
186 return null;
187};

322/**
323 * Draw a text string within a rectangle (drawing is horizontal)
324 *
325 * @param {string} text The text to draw.
326 * @param {number} x X coordinate (left).
327 * @param {number} y Y coordinate (top).
328 * @param {number} width Width of rectangle.
329 * @param {number} height Height of rectangle.
330 * @param {string} align Horizontal alignment: left (default), center, right.
331 * @param {string} vAlign Vertical alignment: top (default), center, bottom.
332 * @param {goog.graphics.Font} font Font describing the font properties.
333 * @param {goog.graphics.Stroke?} stroke Stroke object describing the
334 * stroke.
335 * @param {goog.graphics.Fill?} fill Fill object describing the fill.
336 * @param {goog.graphics.GroupElement=} opt_group The group wrapper element to
337 * append to. If not specified, appends to the main canvas.
338 *
339 * @return {goog.graphics.TextElement} The newly created element.
340 */
341goog.graphics.AbstractGraphics.prototype.drawText = function(
342 text, x, y, width, height, align, vAlign, font, stroke, fill, opt_group) {
343 var baseline = font.size / 2; // Baseline is middle of line
344 var textY;
345 if (vAlign == 'bottom') {
346 textY = y + height - baseline;
347 } else if (vAlign == 'center') {
348 textY = y + height / 2;
349 } else {
350 textY = y + baseline;
351 }
352
353 return this.drawTextOnLine(text, x, textY, x + width, textY, align,
354 font, stroke, fill, opt_group);
355};
