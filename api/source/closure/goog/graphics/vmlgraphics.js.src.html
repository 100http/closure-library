50/**
51 * A Graphics implementation for drawing using VML.
52 * @param {string|number} width The (non-zero) width in pixels. Strings
53 * expressing percentages of parent with (e.g. '80%') are also accepted.
54 * @param {string|number} height The (non-zero) height in pixels. Strings
55 * expressing percentages of parent with (e.g. '80%') are also accepted.
56 * @param {?number=} opt_coordWidth The coordinate width - if
57 * omitted or null, defaults to same as width.
58 * @param {?number=} opt_coordHeight The coordinate height - if
59 * omitted or null, defaults to same as height.
60 * @param {goog.dom.DomHelper=} opt_domHelper The DOM helper object for the
61 * document we want to render in.
62 * @constructor
63 * @extends {goog.graphics.AbstractGraphics}
64 * @deprecated goog.graphics is deprecated. It existed to abstract over browser
65 * differences before the canvas tag was widely supported. See
66 * http://en.wikipedia.org/wiki/Canvas_element for details.
67 * @final
68 */
69goog.graphics.VmlGraphics = function(width, height,
70 opt_coordWidth, opt_coordHeight,
71 opt_domHelper) {
72 goog.graphics.AbstractGraphics.call(this, width, height,
73 opt_coordWidth, opt_coordHeight,
74 opt_domHelper);
75 this.handler_ = new goog.events.EventHandler(this);
76 this.registerDisposable(this.handler_);
77};
78goog.inherits(goog.graphics.VmlGraphics, goog.graphics.AbstractGraphics);

248/**
249 * Resets the graphics so they will display properly on IE8. Noop in older
250 * versions.
251 * @private
252 */
253goog.graphics.VmlGraphics.prototype.updateGraphics_ = function() {
254 if (goog.graphics.VmlGraphics.IE8_MODE_ && this.isInDocument()) {
255 // There's a risk of mXSS here, as the browser is not guaranteed to
256 // return the HTML that was originally written, when innerHTML is read.
257 // However, given that this a deprecated API and affects only IE, it seems
258 // an acceptable risk.
259 var html = goog.html.uncheckedconversions
260 .safeHtmlFromStringKnownToSatisfyTypeContract(
261 goog.string.Const.from('Assign innerHTML to itself'),
262 this.getElement().innerHTML);
263 goog.dom.safe.setInnerHtml(
264 /** @type {!Element} */ (this.getElement()), html);
265 }

284/**
285 * Sets the fill for the given element.
286 * @param {goog.graphics.StrokeAndFillElement} element The element wrapper.
287 * @param {goog.graphics.Fill?} fill The fill object.
288 * @override
289 */
290goog.graphics.VmlGraphics.prototype.setElementFill = function(element, fill) {
291 var vmlElement = element.getElement();
292 goog.graphics.VmlGraphics.removeFill_(vmlElement);
293 if (fill instanceof goog.graphics.SolidFill) {
294 // NOTE(arv): VML does not understand 'transparent' so hard code support
295 // for it.
296 if (fill.getColor() == 'transparent') {
297 vmlElement.filled = false;
298 } else if (fill.getOpacity() != 1) {
299 vmlElement.filled = true;
300 // Set opacity (number 0-1 is translated to percent)
301 var fillNode = this.createVmlElement('fill');
302 fillNode.opacity = Math.round(fill.getOpacity() * 100) + '%';
303 fillNode.color = fill.getColor();
304 vmlElement.appendChild(fillNode);
305 } else {
306 vmlElement.filled = true;
307 vmlElement.fillcolor = fill.getColor();
308 }
309 } else if (fill instanceof goog.graphics.LinearGradient) {
310 vmlElement.filled = true;
311 // Add a 'fill' element
312 var gradient = this.createVmlElement('fill');
313 gradient.color = fill.getColor1();
314 gradient.color2 = fill.getColor2();
315 if (goog.isNumber(fill.getOpacity1())) {
316 gradient.opacity = fill.getOpacity1();
317 }
318 if (goog.isNumber(fill.getOpacity2())) {
319 gradient.opacity2 = fill.getOpacity2();
320 }
321 var angle = goog.math.angle(fill.getX1(), fill.getY1(),
322 fill.getX2(), fill.getY2());
323 // Our angles start from 0 to the right, and grow clockwise.
324 // MSIE starts from 0 to top, and grows anti-clockwise.
325 angle = Math.round(goog.math.standardAngle(270 - angle));
326 gradient.angle = angle;
327 gradient.type = 'gradient';
328 vmlElement.appendChild(gradient);
329 } else {
330 vmlElement.filled = false;
331 }
332 this.updateGraphics_();
333};

336/**
337 * Sets the stroke for the given element.
338 * @param {goog.graphics.StrokeAndFillElement} element The element wrapper.
339 * @param {goog.graphics.Stroke?} stroke The stroke object.
340 * @override
341 */
342goog.graphics.VmlGraphics.prototype.setElementStroke = function(element,
343 stroke) {
344 var vmlElement = element.getElement();
345 if (stroke) {
346 vmlElement.stroked = true;
347
348 var width = stroke.getWidth();
349 if (goog.isString(width) && width.indexOf('px') == -1) {
350 width = parseFloat(width);
351 } else {
352 width = width * this.getPixelScaleX();
353 }
354
355 var strokeElement = vmlElement.getElementsByTagName('stroke')[0];
356 if (!strokeElement) {
357 strokeElement = strokeElement || this.createVmlElement('stroke');
358 vmlElement.appendChild(strokeElement);
359 }
360 strokeElement.opacity = stroke.getOpacity();
361 strokeElement.weight = width + 'px';
362 strokeElement.color = stroke.getColor();
363 } else {
364 vmlElement.stroked = false;
365 }
366 this.updateGraphics_();
367};

370/**
371 * Set the translation and rotation of an element.
372 *
373 * If a more general affine transform is needed than this provides
374 * (e.g. skew and scale) then use setElementAffineTransform.
375 * @param {goog.graphics.Element} element The element wrapper.
376 * @param {number} x The x coordinate of the translation transform.
377 * @param {number} y The y coordinate of the translation transform.
378 * @param {number} angle The angle of the rotation transform.
379 * @param {number} centerX The horizontal center of the rotation transform.
380 * @param {number} centerY The vertical center of the rotation transform.
381 * @override
382 */
383goog.graphics.VmlGraphics.prototype.setElementTransform = function(element, x,
384 y, angle, centerX, centerY) {
385 var el = element.getElement();
386
387 el.style.left = goog.graphics.VmlGraphics.toPosPx(x);
388 el.style.top = goog.graphics.VmlGraphics.toPosPx(y);
389 if (angle || el.rotation) {
390 el.rotation = angle;
391 el.coordsize = goog.graphics.VmlGraphics.toSizeCoord(centerX * 2) + ' ' +
392 goog.graphics.VmlGraphics.toSizeCoord(centerY * 2);
393 }
394};

397/**
398 * Set the transformation of an element.
399 * @param {!goog.graphics.Element} element The element wrapper.
400 * @param {!goog.graphics.AffineTransform} affineTransform The
401 * transformation applied to this element.
402 * @override
403 */
404goog.graphics.VmlGraphics.prototype.setElementAffineTransform = function(
405 element, affineTransform) {
406 var t = affineTransform;
407 var vmlElement = element.getElement();
408 goog.graphics.VmlGraphics.removeSkew_(vmlElement);
409 var skewNode = this.createVmlElement('skew');
410 skewNode.on = 'true';
411 // Move the transform origin to 0px,0px of the graphics.
412 // In VML, 0,0 means the center of the element, -0.5,-0.5 left top conner of
413 // it.
414 skewNode.origin =
415 (-vmlElement.style.pixelLeft / vmlElement.style.pixelWidth - 0.5) + ',' +
416 (-vmlElement.style.pixelTop / vmlElement.style.pixelHeight - 0.5);
417 skewNode.offset = t.getTranslateX().toFixed(1) + 'px,' +
418 t.getTranslateY().toFixed(1) + 'px';
419 skewNode.matrix = [t.getScaleX().toFixed(6), t.getShearX().toFixed(6),
420 t.getShearY().toFixed(6), t.getScaleY().toFixed(6),
421 0, 0].join(',');
422 vmlElement.appendChild(skewNode);
423 this.updateGraphics_();
424};

513/**
514 * Creates the DOM representation of the graphics area.
515 * @override
516 */
517goog.graphics.VmlGraphics.prototype.createDom = function() {
518 var doc = this.dom_.getDocument();
519
520 // Add the namespace.
521 if (!doc.namespaces[goog.graphics.VmlGraphics.VML_PREFIX_]) {
522 if (goog.graphics.VmlGraphics.IE8_MODE_) {
523 doc.namespaces.add(goog.graphics.VmlGraphics.VML_PREFIX_,
524 goog.graphics.VmlGraphics.VML_NS_,
525 goog.graphics.VmlGraphics.VML_IMPORT_);
526 } else {
527 doc.namespaces.add(goog.graphics.VmlGraphics.VML_PREFIX_,
528 goog.graphics.VmlGraphics.VML_NS_);
529 }
530
531 // We assume that we only need to add the CSS if the namespace was not
532 // present
533 var ss = doc.createStyleSheet();
534 ss.cssText = goog.graphics.VmlGraphics.VML_PREFIX_ + '\\:*' +
535 '{behavior:url(#default#VML)}';
536 }
537
538 // Outer a DIV with overflow hidden for clipping.
539 // All inner elements are absolutly positioned on-top of this div.
540 var pixelWidth = this.width;
541 var pixelHeight = this.height;
542 var divElement = this.dom_.createDom(goog.dom.TagName.DIV, {
543 'style': 'overflow:hidden;position:relative;width:' +
544 goog.graphics.VmlGraphics.toCssSize(pixelWidth) + ';height:' +
545 goog.graphics.VmlGraphics.toCssSize(pixelHeight)
546 });
547
548 this.setElementInternal(divElement);
549
550 var group = this.createVmlElement('group');
551 var style = group.style;
552
553 style.position = 'absolute';
554 style.left = style.top = 0;
555 style.width = this.width;
556 style.height = this.height;
557 if (this.coordWidth) {
558 group.coordsize =
559 goog.graphics.VmlGraphics.toSizeCoord(this.coordWidth) + ' ' +
560 goog.graphics.VmlGraphics.toSizeCoord(
561 /** @type {number} */ (this.coordHeight));
562 } else {
563 group.coordsize = goog.graphics.VmlGraphics.toSizeCoord(pixelWidth) + ' ' +
564 goog.graphics.VmlGraphics.toSizeCoord(pixelHeight);
565 }
566
567 if (goog.isDef(this.coordLeft)) {
568 group.coordorigin = goog.graphics.VmlGraphics.toSizeCoord(this.coordLeft) +
569 ' ' + goog.graphics.VmlGraphics.toSizeCoord(this.coordTop);
570 } else {
571 group.coordorigin = '0 0';
572 }
573 divElement.appendChild(group);
574
575 this.canvasElement = new goog.graphics.VmlGroupElement(group, this);
576
577 goog.events.listen(divElement, goog.events.EventType.RESIZE, goog.bind(
578 this.handleContainerResize_, this));
579};

582/**
583 * Changes the canvas element size to match the container element size.
584 * @private
585 */
586goog.graphics.VmlGraphics.prototype.handleContainerResize_ = function() {
587 var size = goog.style.getSize(this.getElement());
588 var style = this.canvasElement.getElement().style;
589
590 if (size.width) {
591 style.width = size.width + 'px';
592 style.height = size.height + 'px';
593 } else {
594 var current = this.getElement();
595 while (current && current.currentStyle &&
596 current.currentStyle.display != 'none') {
597 current = current.parentNode;
598 }
599 if (current && current.currentStyle) {
600 this.handler_.listen(current, 'propertychange',
601 this.handleContainerResize_);
602 }
603 }
604
605 this.dispatchEvent(goog.events.EventType.RESIZE);
606};

690/**
691 * Draw an ellipse.
692 *
693 * @param {number} cx Center X coordinate.
694 * @param {number} cy Center Y coordinate.
695 * @param {number} rx Radius length for the x-axis.
696 * @param {number} ry Radius length for the y-axis.
697 * @param {goog.graphics.Stroke?} stroke Stroke object describing the
698 * stroke.
699 * @param {goog.graphics.Fill?} fill Fill object describing the fill.
700 * @param {goog.graphics.GroupElement=} opt_group The group wrapper element
701 * to append to. If not specified, appends to the main canvas.
702 *
703 * @return {!goog.graphics.EllipseElement} The newly created element.
704 * @override
705 */
706goog.graphics.VmlGraphics.prototype.drawEllipse = function(cx, cy, rx, ry,
707 stroke, fill, opt_group) {
708 var element = this.createVmlElement('oval');
709 goog.graphics.VmlGraphics.setPositionAndSize(element, cx - rx, cy - ry,
710 rx * 2, ry * 2);
711 var wrapper = new goog.graphics.VmlEllipseElement(element, this,

846/**
847 * Returns a string representation of a logical path suitable for use in
848 * a VML element.
849 *
850 * @param {goog.graphics.Path} path The logical path.
851 * @return {string} The VML path representation.
852 * @suppress {deprecated} goog.graphics is deprecated.
853 */
854goog.graphics.VmlGraphics.getVmlPath = function(path) {
855 var list = [];
856 path.forEachSegment(function(segment, args) {
857 switch (segment) {
858 case goog.graphics.Path.Segment.MOVETO:
859 list.push('m');
860 Array.prototype.push.apply(list, goog.array.map(args,
861 goog.graphics.VmlGraphics.toSizeCoord));
862 break;
863 case goog.graphics.Path.Segment.LINETO:
864 list.push('l');
865 Array.prototype.push.apply(list, goog.array.map(args,
866 goog.graphics.VmlGraphics.toSizeCoord));
867 break;
868 case goog.graphics.Path.Segment.CURVETO:
869 list.push('c');
870 Array.prototype.push.apply(list, goog.array.map(args,
871 goog.graphics.VmlGraphics.toSizeCoord));
872 break;
873 case goog.graphics.Path.Segment.CLOSE:
874 list.push('x');
875 break;
876 case goog.graphics.Path.Segment.ARCTO:
877 var toAngle = args[2] + args[3];
878 var cx = goog.graphics.VmlGraphics.toSizeCoord(
879 args[4] - goog.math.angleDx(toAngle, args[0]));
880 var cy = goog.graphics.VmlGraphics.toSizeCoord(
881 args[5] - goog.math.angleDy(toAngle, args[1]));
882 var rx = goog.graphics.VmlGraphics.toSizeCoord(args[0]);
883 var ry = goog.graphics.VmlGraphics.toSizeCoord(args[1]);
884 // VML angles are in fd units (see http://www.w3.org/TR/NOTE-VML) and
885 // are positive counter-clockwise.
886 var fromAngle = Math.round(args[2] * -65536);
887 var extent = Math.round(args[3] * -65536);
888 list.push('ae', cx, cy, rx, ry, fromAngle, extent);
889 break;
890 }
891 });
892 return list.join(' ');
893};
