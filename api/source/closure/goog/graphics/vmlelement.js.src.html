39/**
40 * Returns the VML element corresponding to this object. This method is added
41 * to several classes below. Note that the return value of this method may
42 * change frequently in IE8, so it should not be cached externally.
43 * @return {Element} The VML element corresponding to this object.
44 * @this {goog.graphics.VmlGroupElement|goog.graphics.VmlEllipseElement|
45 * goog.graphics.VmlRectElement|goog.graphics.VmlPathElement|
46 * goog.graphics.VmlTextElement|goog.graphics.VmlImageElement}
47 * @private
48 */
49goog.graphics.vmlGetElement_ = function() {
50 this.element_ = this.getGraphics().getVmlElement(this.id_) || this.element_;
51 return this.element_;
52};

101/**
102 * Set the size of the group element.
103 * @param {number|string} width The width of the group element.
104 * @param {number|string} height The height of the group element.
105 * @override
106 */
107goog.graphics.VmlGroupElement.prototype.setSize = function(width, height) {
108 var element = this.getElement();
109
110 var style = element.style;
111 style.width = /** @suppress {missingRequire} */ (
112 goog.graphics.VmlGraphics.toSizePx(width));
113 style.height = /** @suppress {missingRequire} */ (
114 goog.graphics.VmlGraphics.toSizePx(height));
115
116 element.coordsize = /** @suppress {missingRequire} */
117 goog.graphics.VmlGraphics.toSizeCoord(width) +
118 ' ' +
119 /** @suppress {missingRequire} */
120 goog.graphics.VmlGraphics.toSizeCoord(height);
121
122 // Don't overwrite the root element's origin.
123 if (!this.isRootElement_()) {
124 element.coordorigin = '0 0';
125 }
126};

138 * @param {number} cx Center X coordinate.
139 * @param {number} cy Center Y coordinate.
140 * @param {number} rx Radius length for the x-axis.
141 * @param {number} ry Radius length for the y-axis.
142 * @param {goog.graphics.Stroke?} stroke The stroke to use for this element.
143 * @param {goog.graphics.Fill?} fill The fill to use for this element.
144 * @constructor

149 * @final

151goog.graphics.VmlEllipseElement = function(element, graphics,
152 cx, cy, rx, ry, stroke, fill) {
153 this.id_ = element.id;
154
155 goog.graphics.EllipseElement.call(this, element, graphics, stroke, fill);
156
157 // Store center and radius for future calls to setRadius or setCenter.
158
159 /**
160 * X coordinate of the ellipse center.
161 * @type {number}

168 * @type {number}
169 */
170 this.cy = cy;

173 /**
174 * Radius length for the x-axis.
175 * @type {number}
176 */
177 this.rx = rx;
178
179
180 /**
181 * Radius length for the y-axis.
182 * @type {number}
183 */
184 this.ry = ry;

194/**
195 * Update the center point of the ellipse.
196 * @param {number} cx Center X coordinate.
197 * @param {number} cy Center Y coordinate.
198 * @override
199 */
200goog.graphics.VmlEllipseElement.prototype.setCenter = function(cx, cy) {
201 this.cx = cx;
202 this.cy = cy;
203 /** @suppress {missingRequire} */
204 goog.graphics.VmlGraphics.setPositionAndSize(this.getElement(),
205 cx - this.rx, cy - this.ry, this.rx * 2, this.ry * 2);
206};
