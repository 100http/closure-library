29/**
30 * Creates a path object. A path is a sequence of segments and may be open or
31 * closed. Path uses the EVEN-ODD fill rule for determining the interior of the
32 * path. A path must start with a moveTo command.
33 *
34 * A "simple" path does not contain any arcs and may be transformed using
35 * the {@code transform} method.
36 *
37 * @constructor
38 */
39goog.graphics.Path = function() {
40 /**
41 * The segment types that constitute this path.
42 * @type {!Array<number>}
43 * @private
44 */
45 this.segments_ = [];
46
47 /**
48 * The number of repeated segments of the current type.
49 * @type {!Array<number>}
50 * @private
51 */
52 this.count_ = [];
53
54 /**
55 * The arguments corresponding to each of the segments.
56 * @type {!Array<number>}
57 * @private
58 */
59 this.arguments_ = [];
60};

128/**
129 * Appends another path to the end of this path.
130 *
131 * @param {!goog.graphics.Path} path The path to append.
132 * @return {!goog.graphics.Path} This path.
133 */
134goog.graphics.Path.prototype.appendPath = function(path) {
135 if (path.currentPoint_) {
136 Array.prototype.push.apply(this.segments_, path.segments_);
137 Array.prototype.push.apply(this.count_, path.count_);
138 Array.prototype.push.apply(this.arguments_, path.arguments_);
139 this.currentPoint_ = path.currentPoint_.concat();
140 this.closePoint_ = path.closePoint_.concat();
141 this.simple_ = this.simple_ && path.simple_;
142 }
143 return this;
144};

163/**
164 * Adds a point to the path by moving to the specified point. Repeated moveTo
165 * commands are collapsed into a single moveTo.
166 *
167 * @param {number} x X coordinate of destination point.
168 * @param {number} y Y coordinate of destination point.
169 * @return {!goog.graphics.Path} The path itself.
170 */
171goog.graphics.Path.prototype.moveTo = function(x, y) {
172 if (goog.array.peek(this.segments_) == goog.graphics.Path.Segment.MOVETO) {
173 this.arguments_.length -= 2;
174 } else {
175 this.segments_.push(goog.graphics.Path.Segment.MOVETO);
176 this.count_.push(1);
177 }
178 this.arguments_.push(x, y);
179 this.currentPoint_ = this.closePoint_ = [x, y];
180 return this;
181};

184/**
185 * Adds points to the path by drawing a straight line to each point.
186 *
187 * @param {...number} var_args The coordinates of each destination point as x, y
188 * value pairs.
189 * @return {!goog.graphics.Path} The path itself.
190 */
191goog.graphics.Path.prototype.lineTo = function(var_args) {
192 var lastSegment = goog.array.peek(this.segments_);
193 if (lastSegment == null) {
194 throw Error('Path cannot start with lineTo');
195 }
196 if (lastSegment != goog.graphics.Path.Segment.LINETO) {
197 this.segments_.push(goog.graphics.Path.Segment.LINETO);
198 this.count_.push(0);
199 }
200 for (var i = 0; i < arguments.length; i += 2) {
201 var x = arguments[i];
202 var y = arguments[i + 1];
203 this.arguments_.push(x, y);
204 }
205 this.count_[this.count_.length - 1] += i / 2;
206 this.currentPoint_ = [x, y];
207 return this;
208};

211/**
212 * Adds points to the path by drawing cubic Bezier curves. Each curve is
213 * specified using 3 points (6 coordinates) - two control points and the end
214 * point of the curve.
215 *
216 * @param {...number} var_args The coordinates specifiying each curve in sets of
217 * 6 points: {@code [x1, y1]} the first control point, {@code [x2, y2]} the
218 * second control point and {@code [x, y]} the end point.
219 * @return {!goog.graphics.Path} The path itself.
220 */
221goog.graphics.Path.prototype.curveTo = function(var_args) {
222 var lastSegment = goog.array.peek(this.segments_);
223 if (lastSegment == null) {
224 throw Error('Path cannot start with curve');
225 }
226 if (lastSegment != goog.graphics.Path.Segment.CURVETO) {
227 this.segments_.push(goog.graphics.Path.Segment.CURVETO);
228 this.count_.push(0);
229 }
230 for (var i = 0; i < arguments.length; i += 6) {
231 var x = arguments[i + 4];
232 var y = arguments[i + 5];
233 this.arguments_.push(arguments[i], arguments[i + 1],
234 arguments[i + 2], arguments[i + 3], x, y);
235 }
236 this.count_[this.count_.length - 1] += i / 6;
237 this.currentPoint_ = [x, y];
238 return this;
239};

244 * last point to the first point.
245 *
246 * @return {!goog.graphics.Path} The path itself.

248goog.graphics.Path.prototype.close = function() {
249 var lastSegment = goog.array.peek(this.segments_);

251 throw Error('Path cannot start with close');
252 }
253 if (lastSegment != goog.graphics.Path.Segment.CLOSE) {

262/**
263 * Adds a path command to draw an arc centered at the point {@code (cx, cy)}
264 * with radius {@code rx} along the x-axis and {@code ry} along the y-axis from
265 * {@code startAngle} through {@code extent} degrees. Positive rotation is in
266 * the direction from positive x-axis to positive y-axis.
267 *
268 * @param {number} cx X coordinate of center of ellipse.
269 * @param {number} cy Y coordinate of center of ellipse.
270 * @param {number} rx Radius of ellipse on x axis.
271 * @param {number} ry Radius of ellipse on y axis.
272 * @param {number} fromAngle Starting angle measured in degrees from the
273 * positive x-axis.
274 * @param {number} extent The span of the arc in degrees.
275 * @param {boolean} connect If true, the starting point of the arc is connected
276 * to the current point.
277 * @return {!goog.graphics.Path} The path itself.
278 * @deprecated Use {@code arcTo} or {@code arcToAsCurves} instead.
279 */
280goog.graphics.Path.prototype.arc = function(cx, cy, rx, ry,
281 fromAngle, extent, connect) {
282 var startX = cx + goog.math.angleDx(fromAngle, rx);
283 var startY = cy + goog.math.angleDy(fromAngle, ry);
284 if (connect) {
285 if (!this.currentPoint_ || startX != this.currentPoint_[0] ||
286 startY != this.currentPoint_[1]) {
287 this.lineTo(startX, startY);
288 }
289 } else {
290 this.moveTo(startX, startY);
291 }
292 return this.arcTo(rx, ry, fromAngle, extent);
293};

296/**
297 * Adds a path command to draw an arc starting at the path's current point,
298 * with radius {@code rx} along the x-axis and {@code ry} along the y-axis from
299 * {@code startAngle} through {@code extent} degrees. Positive rotation is in
300 * the direction from positive x-axis to positive y-axis.
301 *
302 * This method makes the path non-simple.
303 *
304 * @param {number} rx Radius of ellipse on x axis.
305 * @param {number} ry Radius of ellipse on y axis.
306 * @param {number} fromAngle Starting angle measured in degrees from the
307 * positive x-axis.
308 * @param {number} extent The span of the arc in degrees.
309 * @return {!goog.graphics.Path} The path itself.
310 */
311goog.graphics.Path.prototype.arcTo = function(rx, ry, fromAngle, extent) {
312 var cx = this.currentPoint_[0] - goog.math.angleDx(fromAngle, rx);
313 var cy = this.currentPoint_[1] - goog.math.angleDy(fromAngle, ry);
314 var ex = cx + goog.math.angleDx(fromAngle + extent, rx);
315 var ey = cy + goog.math.angleDy(fromAngle + extent, ry);
316 this.segments_.push(goog.graphics.Path.Segment.ARCTO);
317 this.count_.push(1);
318 this.arguments_.push(rx, ry, fromAngle, extent, ex, ey);
319 this.simple_ = false;
320 this.currentPoint_ = [ex, ey];
321 return this;
322};

325/**
326 * Same as {@code arcTo}, but approximates the arc using bezier curves.
327.* As a result, this method does not affect the simplified status of this path.
328 * The algorithm is adapted from {@code java.awt.geom.ArcIterator}.
329 *
330 * @param {number} rx Radius of ellipse on x axis.
331 * @param {number} ry Radius of ellipse on y axis.
332 * @param {number} fromAngle Starting angle measured in degrees from the
333 * positive x-axis.
334 * @param {number} extent The span of the arc in degrees.
335 * @return {!goog.graphics.Path} The path itself.
336 */
337goog.graphics.Path.prototype.arcToAsCurves = function(
338 rx, ry, fromAngle, extent) {
339 var cx = this.currentPoint_[0] - goog.math.angleDx(fromAngle, rx);
340 var cy = this.currentPoint_[1] - goog.math.angleDy(fromAngle, ry);
341 var extentRad = goog.math.toRadians(extent);
342 var arcSegs = Math.ceil(Math.abs(extentRad) / Math.PI * 2);
343 var inc = extentRad / arcSegs;
344 var angle = goog.math.toRadians(fromAngle);
345 for (var j = 0; j < arcSegs; j++) {
346 var relX = Math.cos(angle);
347 var relY = Math.sin(angle);
348 var z = 4 / 3 * Math.sin(inc / 2) / (1 + Math.cos(inc / 2));
349 var c0 = cx + (relX - z * relY) * rx;
350 var c1 = cy + (relY + z * relX) * ry;
351 angle += inc;
352 relX = Math.cos(angle);
353 relY = Math.sin(angle);
354 this.curveTo(c0, c1,
355 cx + (relX + z * relY) * rx,
356 cy + (relY - z * relX) * ry,
357 cx + relX * rx,
358 cy + relY * ry);
359 }
360 return this;
361};

364/**
365 * Iterates over the path calling the supplied callback once for each path
366 * segment. The arguments to the callback function are the segment type and
367 * an array of its arguments.
368 *
369 * The {@code LINETO} and {@code CURVETO} arrays can contain multiple
370 * segments of the same type. The number of segments is the length of the
371 * array divided by the segment length (2 for lines, 6 for curves).
372 *
373 * As a convenience the {@code ARCTO} segment also includes the end point as the
374 * last two arguments: {@code rx, ry, fromAngle, extent, x, y}.
375 *
376 * @param {function(number, Array)} callback The function to call with each
377 * path segment.
378 */
379goog.graphics.Path.prototype.forEachSegment = function(callback) {
380 var points = this.arguments_;
381 var index = 0;
382 for (var i = 0, length = this.segments_.length; i < length; i++) {
383 var seg = this.segments_[i];
384 var n = goog.graphics.Path.segmentArgCounts_[seg] * this.count_[i];
385 callback(seg, points.slice(index, index + n));
386 index += n;
387 }
388};

428/**
429 * A map from segment type to the path function to call to simplify a path.
430 * @type {!Object}
431 * @private
432 * @suppress {deprecated} goog.graphics.Path is deprecated.

434goog.graphics.Path.simplifySegmentMap_ = (function() {
435 var map = {};
436 map[goog.graphics.Path.Segment.MOVETO] = goog.graphics.Path.prototype.moveTo;
437 map[goog.graphics.Path.Segment.LINETO] = goog.graphics.Path.prototype.lineTo;
438 map[goog.graphics.Path.Segment.CLOSE] = goog.graphics.Path.prototype.close;
439 map[goog.graphics.Path.Segment.CURVETO] =
440 goog.graphics.Path.prototype.curveTo;
441 map[goog.graphics.Path.Segment.ARCTO] =
442 goog.graphics.Path.prototype.arcToAsCurves;
443 return map;

447/**
448 * Creates a copy of the given path, replacing {@code arcTo} with
449 * {@code arcToAsCurves}. The resulting path is simplified and can
450 * be transformed.
451 *
452 * @param {!goog.graphics.Path} src The path to simplify.
453 * @return {!goog.graphics.Path} A new simplified path.
454 * @suppress {deprecated} goog.graphics is deprecated.
455 */
456goog.graphics.Path.createSimplifiedPath = function(src) {
457 if (src.isSimple()) {
458 return src.clone();
459 }
460 var path = new goog.graphics.Path();
461 src.forEachSegment(function(segment, args) {
462 goog.graphics.Path.simplifySegmentMap_[segment].apply(path, args);
463 });
464 return path;
465};

483/**
484 * Transforms the path. Only simple paths are transformable. Attempting
485 * to transform a non-simple path will throw an error.
486 *
487 * @param {!goog.graphics.AffineTransform} tx The transformation to perform.
488 * @return {!goog.graphics.Path} The path itself.
489 */
490goog.graphics.Path.prototype.transform = function(tx) {
491 if (!this.isSimple()) {
492 throw Error('Non-simple path');
493 }
494 tx.transform(this.arguments_, 0, this.arguments_, 0,
495 this.arguments_.length / 2);
496 if (this.closePoint_) {
497 tx.transform(this.closePoint_, 0, this.closePoint_, 0, 1);
498 }
499 if (this.currentPoint_ && this.closePoint_ != this.currentPoint_) {
500 tx.transform(this.currentPoint_, 0, this.currentPoint_, 0, 1);
501 }
502 return this;
503};
