32/**
33 * Class for a Trie datastructure. Trie data structures are made out of trees
34 * of Trie classes.
35 *
36 * @param {goog.structs.Trie<VALUE>|Object<string, VALUE>=} opt_trie Optional
37 * goog.structs.Trie or Object to initialize trie with.
38 * @constructor
39 * @template VALUE
40 */
41goog.structs.Trie = function(opt_trie) {
42 /**
43 * This trie's value. For the base trie, this will be the value of the
44 * empty key, if defined.
45 * @private {VALUE}
46 */
47 this.value_ = undefined;
48
49 /**
50 * This trie's child nodes.
51 * @private {!Object<!goog.structs.Trie<VALUE>>}
52 */
53 this.childNodes_ = {};
54
55 if (opt_trie) {
56 this.setAll(opt_trie);
57 }
58};

83/**
84 * Helper function for set and add. Adds the given key/value pair to
85 * the trie, or, if the key already exists, sets the value of the key. If
86 * opt_add is true, then throws an exception if the key already has a value in
87 * the trie. O(L), where L is the length of the key.
88 * @param {string} key The key.
89 * @param {VALUE} value The value.
90 * @param {boolean=} opt_add Throw exception if key is already in the trie.
91 * @private
92 */
93goog.structs.Trie.prototype.setOrAdd_ = function(key, value, opt_add) {
94 var node = this;
95 for (var characterPosition = 0; characterPosition < key.length;
96 characterPosition++) {
97 var currentCharacter = key.charAt(characterPosition);
98 if (!node.childNodes_[currentCharacter]) {
99 node.childNodes_[currentCharacter] = new goog.structs.Trie();
100 }
101 node = node.childNodes_[currentCharacter];
102 }
103 if (opt_add && node.value_ !== undefined) {
104 throw Error('The collection already contains the key "' + key + '"');
105 } else {
106 node.value_ = value;
107 }
108};

111/**
112 * Adds multiple key/value pairs from another goog.structs.Trie or Object.
113 * O(N) where N is the number of nodes in the trie.
114 * @param {!Object<string, VALUE>|!goog.structs.Trie<VALUE>} trie Object
115 * containing the data to add.
116 */
117goog.structs.Trie.prototype.setAll = function(trie) {
118 var keys = goog.structs.getKeys(trie);
119 var values = goog.structs.getValues(trie);
120
121 for (var i = 0; i < keys.length; i++) {
122 this.set(keys[i], values[i]);
123 }
124};

127/**
128 * Traverse along the given path, returns the child node at ending.
129 * Returns undefined if node for the path doesn't exist.
130 * @param {string} path The path to traverse.
131 * @return {!goog.structs.Trie<VALUE>|undefined}
132 * @private
133 */
134goog.structs.Trie.prototype.getChildNode_ = function(path) {
135 var node = this;
136 for (var characterPosition = 0; characterPosition < path.length;
137 characterPosition++) {
138 var currentCharacter = path.charAt(characterPosition);
139 node = node.childNodes_[currentCharacter];
140 if (!node) {
141 return undefined;
142 }
143 }
144 return node;
145};

148/**
149 * Retrieves a value from the trie given a key. O(L), where L is the length of
150 * the key.
151 * @param {string} key The key to retrieve from the trie.
152 * @return {VALUE|undefined} The value of the key in the trie, or undefined if
153 * the trie does not contain this key.
154 */
155goog.structs.Trie.prototype.get = function(key) {
156 var node = this.getChildNode_(key);
157 return node ? node.value_ : undefined;

161/**
162 * Retrieves all values from the trie that correspond to prefixes of the given
163 * input key. O(L), where L is the length of the key.
164 *
165 * @param {string} key The key to use for lookup. The given key as well as all
166 * prefixes of the key are retrieved.
167 * @param {?number=} opt_keyStartIndex Optional position in key to start lookup
168 * from. Defaults to 0 if not specified.
169 * @return {!Object<string, VALUE>} Map of end index of matching prefixes and
170 * corresponding values. Empty if no match found.
171 */
172goog.structs.Trie.prototype.getKeyAndPrefixes = function(key,
173 opt_keyStartIndex) {
174 var node = this;
175 var matches = {};
176 var characterPosition = opt_keyStartIndex || 0;
177
178 if (node.value_ !== undefined) {
179 matches[characterPosition] = node.value_;
180 }
181
182 for (; characterPosition < key.length; characterPosition++) {
183 var currentCharacter = key.charAt(characterPosition);
184 if (!(currentCharacter in node.childNodes_)) {
185 break;
186 }
187 node = node.childNodes_[currentCharacter];
188 if (node.value_ !== undefined) {
189 matches[characterPosition] = node.value_;
190 }
191 }
192
193 return matches;
194};

197/**
198 * Gets the values of the trie. Not returned in any reliable order. O(N) where

209/**
210 * Gets the values of the trie. Not returned in any reliable order. O(N) where
211 * N is the number of nodes in the trie. Builds the values as it goes.
212 * @param {!Array<VALUE>} allValues Array to place values into.
213 * @private
214 */
215goog.structs.Trie.prototype.getValuesInternal_ = function(allValues) {
216 if (this.value_ !== undefined) {
217 allValues.push(this.value_);
218 }
219 for (var childNode in this.childNodes_) {
220 this.childNodes_[childNode].getValuesInternal_(allValues);
221 }
222};

225/**
226 * Gets the keys of the trie. Not returned in any reliable order. O(N) where
227 * N is the number of nodes in the trie (or prefix subtree).
228 * @param {string=} opt_prefix Find only keys with this optional prefix.
229 * @return {!Array<string>} The keys in the trie.
230 */
231goog.structs.Trie.prototype.getKeys = function(opt_prefix) {
232 var allKeys = [];
233 if (opt_prefix) {
234 // Traverse to the given prefix, then call getKeysInternal_ to dump the
235 // keys below that point.
236 var node = this;
237 for (var characterPosition = 0; characterPosition < opt_prefix.length;
238 characterPosition++) {
239 var currentCharacter = opt_prefix.charAt(characterPosition);
240 if (!node.childNodes_[currentCharacter]) {
241 return [];
242 }
243 node = node.childNodes_[currentCharacter];
244 }
245 node.getKeysInternal_(opt_prefix, allKeys);
246 } else {
247 this.getKeysInternal_('', allKeys);
248 }
249 return allKeys;
250};

253/**
254 * Private method to get keys from the trie. Builds the keys as it goes.
255 * @param {string} keySoFar The partial key (prefix) traversed so far.
256 * @param {!Array<string>} allKeys The partially built array of keys seen so
257 * far.
258 * @private
259 */
260goog.structs.Trie.prototype.getKeysInternal_ = function(keySoFar, allKeys) {
261 if (this.value_ !== undefined) {
262 allKeys.push(keySoFar);
263 }
264 for (var childNode in this.childNodes_) {
265 this.childNodes_[childNode].getKeysInternal_(keySoFar + childNode, allKeys);
266 }
267};

281/**
282 * Checks to see if a certain prefix is in the trie. O(L), where L is the length
283 * of the prefix.
284 * @param {string} prefix A prefix that may be in the trie.
285 * @return {boolean} Whether any key of the trie has the prefix.
286 */
287goog.structs.Trie.prototype.containsPrefix = function(prefix) {
288 // Empty string is any key's prefix.
289 if (prefix.length == 0) {
290 return !this.isEmpty();
291 }
292 return !!this.getChildNode_(prefix);
293};

296/**
297 * Checks to see if a certain value is in the trie. Worst case is O(N) where
298 * N is the number of nodes in the trie.
299 * @param {VALUE} value A value that may be in the trie.
300 * @return {boolean} Whether the trie contains the value.
301 */
302goog.structs.Trie.prototype.containsValue = function(value) {
303 if (this.value_ === value) {
304 return true;
305 }
306 for (var childNode in this.childNodes_) {
307 if (this.childNodes_[childNode].containsValue(value)) {
308 return true;
309 }
310 }
311 return false;
312};

324/**
325 * Removes a key from the trie or throws an exception if the key is not in the
326 * trie. O(L), where L is the length of the key.
327 * @param {string} key A key that should be removed from the trie.
328 * @return {VALUE} The value whose key was removed.
329 */
330goog.structs.Trie.prototype.remove = function(key) {
331 var node = this;
332 var parents = [];
333 for (var characterPosition = 0; characterPosition < key.length;
334 characterPosition++) {
335 var currentCharacter = key.charAt(characterPosition);
336 if (!node.childNodes_[currentCharacter]) {
337 throw Error('The collection does not have the key "' + key + '"');
338 }
339
340 // Archive the current parent and child name (key in childNodes_) so that
341 // we may remove the following node and its parents if they are empty.
342 parents.push([node, currentCharacter]);
343
344 node = node.childNodes_[currentCharacter];
345 }
346 var oldValue = node.value_;
347 delete node.value_;
348
349 while (parents.length > 0) {
350 var currentParentAndCharacter = parents.pop();
351 var currentParent = currentParentAndCharacter[0];
352 var currentCharacter = currentParentAndCharacter[1];
353 if (currentParent.childNodes_[currentCharacter].isEmpty()) {
354 // If the child is empty, then remove it.
355 delete currentParent.childNodes_[currentCharacter];
356 } else {
357 // No point of traversing back any further, since we can't remove this
358 // path.
359 break;
360 }
361 }
362 return oldValue;
363};
