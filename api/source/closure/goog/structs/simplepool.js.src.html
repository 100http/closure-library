31/**
32 * A generic pool class. Simpler and more efficient than goog.structs.Pool
33 * because it doesn't maintain a list of objects that are in use. This class
34 * has constant overhead and doesn't create any additional objects as part of
35 * the pool management after construction time.
36 *
37 * IMPORTANT: If the objects being pooled are arrays or maps that can have
38 * unlimited number of properties, they need to be cleaned before being
39 * returned to the pool.
40 *
41 * Also note that {@see goog.object.clean} actually allocates an array to clean
42 * the object passed to it, so simply using this function would defy the
43 * purpose of using the pool.
44 *
45 * @param {number} initialCount Initial number of objects to populate the free
46 * pool at construction time.
47 * @param {number} maxCount Maximum number of objects to keep in the free pool.
48 * @constructor
49 * @extends {goog.Disposable}
50 * @template T
51 */
52goog.structs.SimplePool = function(initialCount, maxCount) {
53 goog.Disposable.call(this);
54
55 /**
56 * Function for overriding createObject. The avoids a common case requiring
57 * subclassing this class.
58 * @private {Function}
59 */
60 this.createObjectFn_ = null;
61
62 /**
63 * Function for overriding disposeObject. The avoids a common case requiring
64 * subclassing this class.
65 * @private {Function}
66 */
67 this.disposeObjectFn_ = null;
68
69 /**
70 * Maximum number of objects allowed
71 * @private {number}
72 */
73 this.maxCount_ = maxCount;
74
75 /**
76 * Queue used to store objects that are currently in the pool and available
77 * to be used.
78 * @private {Array<T>}
79 */
80 this.freeQueue_ = [];
81
82 this.createInitial_(initialCount);
83};
84goog.inherits(goog.structs.SimplePool, goog.Disposable);

110/**
111 * Gets an unused object from the the pool, if there is one available,
112 * otherwise creates a new one.
113 * @return {T} An object from the pool or a new one if necessary.
114 */
115goog.structs.SimplePool.prototype.getObject = function() {
116 if (this.freeQueue_.length) {
117 return this.freeQueue_.pop();
118 }
119 return this.createObject();
120};

123/**
124 * Returns an object to the pool so that it can be reused. If the pool is
125 * already full, the object is disposed instead.
126 * @param {T} obj The object to release.
127 */
128goog.structs.SimplePool.prototype.releaseObject = function(obj) {
129 if (this.freeQueue_.length < this.maxCount_) {
130 this.freeQueue_.push(obj);
131 } else {
132 this.disposeObject(obj);
133 }
134};

137/**
138 * Populates the pool with initialCount objects.
139 * @param {number} initialCount The number of objects to add to the pool.
140 * @private
141 */
142goog.structs.SimplePool.prototype.createInitial_ = function(initialCount) {
143 if (initialCount > this.maxCount_) {
144 throw Error('[goog.structs.SimplePool] Initial cannot be greater than max');
145 }
146 for (var i = 0; i < initialCount; i++) {
147 this.freeQueue_.push(this.createObject());
148 }
149};

152/**
153 * Should be overridden by sub-classes to return an instance of the object type
154 * that is expected in the pool.
155 * @return {T} The created object.
156 */
157goog.structs.SimplePool.prototype.createObject = function() {
158 if (this.createObjectFn_) {
159 return this.createObjectFn_();
160 } else {
161 return {};
162 }
163};
164
165
166/**
167 * Should be overrideen to dispose of an object. Default implementation is to
168 * remove all of the object's members, which should render it useless. Calls the
169 * object's dispose method, if available.
170 * @param {T} obj The object to dispose.
171 */
172goog.structs.SimplePool.prototype.disposeObject = function(obj) {
173 if (this.disposeObjectFn_) {
174 this.disposeObjectFn_(obj);
175 } else if (goog.isObject(obj)) {
176 if (goog.isFunction(obj.dispose)) {
177 obj.dispose();
178 } else {
179 for (var i in obj) {
180 delete obj[i];
181 }
182 }
183 }
184};

186
187/**
188 * Disposes of the pool and all objects currently held in the pool.
189 * @override
190 * @protected
191 */
192goog.structs.SimplePool.prototype.disposeInternal = function() {
193 goog.structs.SimplePool.superClass_.disposeInternal.call(this);
194 // Call disposeObject on each object held by the pool.
195 var freeQueue = this.freeQueue_;
196 while (freeQueue.length) {
197 this.disposeObject(freeQueue.pop());
198 }
199 delete this.freeQueue_;
200};
