15/**
16 * @fileoverview This event wrapper will dispatch an event when the user uses
17 * the mouse wheel to scroll an element. You can get the direction by checking
18 * the deltaX and deltaY properties of the event.
19 *
20 * This class aims to smooth out inconsistencies between browser platforms with
21 * regards to mousewheel events, but we do not cover every possible
22 * software/hardware combination out there, some of which occasionally produce
23 * very large deltas in mousewheel events. If your application wants to guard
24 * against extremely large deltas, use the setMaxDeltaX and setMaxDeltaY APIs
25 * to set maximum values that make sense for your application.
26 *
27 * @author arv@google.com (Erik Arvidsson)
28 * @see ../demos/mousewheelhandler.html
29 */

45/**
46 * This event handler allows you to catch mouse wheel events in a consistent
47 * manner.
48 * @param {Element|Document} element The element to listen to the mouse wheel
49 * event on.
50 * @param {boolean=} opt_capture Whether to handle the mouse wheel event in
51 * capture phase.
52 * @constructor
53 * @extends {goog.events.EventTarget}
54 */
55goog.events.MouseWheelHandler = function(element, opt_capture) {
56 goog.events.EventTarget.call(this);
57
58 /**
59 * This is the element that we will listen to the real mouse wheel events on.
60 * @type {Element|Document}
61 * @private
62 */
63 this.element_ = element;
64
65 var rtlElement = goog.dom.isElement(this.element_) ?
66 /** @type {Element} */ (this.element_) :
67 (this.element_ ? /** @type {Document} */ (this.element_).body : null);
68
69 /**
70 * True if the element exists and is RTL, false otherwise.
71 * @type {boolean}
72 * @private
73 */
74 this.isRtl_ = !!rtlElement && goog.style.isRightToLeft(rtlElement);
75
76 var type = goog.userAgent.GECKO ? 'DOMMouseScroll' : 'mousewheel';
77
78 /**
79 * The key returned from the goog.events.listen.
80 * @type {goog.events.Key}
81 * @private
82 */
83 this.listenKey_ = goog.events.listen(this.element_, type, this, opt_capture);
84};
85goog.inherits(goog.events.MouseWheelHandler, goog.events.EventTarget);

131/**
132 * Handles the events on the element.
133 * @param {goog.events.BrowserEvent} e The underlying browser event.
134 */
135goog.events.MouseWheelHandler.prototype.handleEvent = function(e) {
136 var deltaX = 0;
137 var deltaY = 0;
138 var detail = 0;
139 var be = e.getBrowserEvent();
140 if (be.type == 'mousewheel') {
141 var wheelDeltaScaleFactor = 1;
142 if (goog.userAgent.IE ||
143 goog.userAgent.WEBKIT &&
144 (goog.userAgent.WINDOWS || goog.userAgent.isVersionOrHigher('532.0'))) {
145 // In IE we get a multiple of 120; we adjust to a multiple of 3 to
146 // represent number of lines scrolled (like Gecko).
147 // Newer versions of Webkit match IE behavior, and WebKit on
148 // Windows also matches IE behavior.
149 // See bug https://bugs.webkit.org/show_bug.cgi?id=24368
150 wheelDeltaScaleFactor = 40;
151 }
152
153 detail = goog.events.MouseWheelHandler.smartScale_(
154 -be.wheelDelta, wheelDeltaScaleFactor);
155 if (goog.isDef(be.wheelDeltaX)) {
156 // Webkit has two properties to indicate directional scroll, and
157 // can scroll both directions at once.
158 deltaX = goog.events.MouseWheelHandler.smartScale_(
159 -be.wheelDeltaX, wheelDeltaScaleFactor);
160 deltaY = goog.events.MouseWheelHandler.smartScale_(
161 -be.wheelDeltaY, wheelDeltaScaleFactor);
162 } else {
163 deltaY = detail;
164 }
165
166 // Historical note: Opera (pre 9.5) used to negate the detail value.
167 } else { // Gecko
168 // Gecko returns multiple of 3 (representing the number of lines scrolled)
169 detail = be.detail;
170
171 // Gecko sometimes returns really big values if the user changes settings to
172 // scroll a whole page per scroll
173 if (detail > 100) {
174 detail = 3;
175 } else if (detail < -100) {
176 detail = -3;
177 }
178
179 // Firefox 3.1 adds an axis field to the event to indicate direction of
180 // scroll. See https://developer.mozilla.org/en/Gecko-Specific_DOM_Events
181 if (goog.isDef(be.axis) && be.axis === be.HORIZONTAL_AXIS) {
182 deltaX = detail;
183 } else {
184 deltaY = detail;
185 }
186 }
187
188 if (goog.isNumber(this.maxDeltaX_)) {
189 deltaX = goog.math.clamp(deltaX, -this.maxDeltaX_, this.maxDeltaX_);
190 }
191 if (goog.isNumber(this.maxDeltaY_)) {
192 deltaY = goog.math.clamp(deltaY, -this.maxDeltaY_, this.maxDeltaY_);
193 }
194 // Don't clamp 'detail', since it could be ambiguous which axis it refers to
195 // and because it's informally deprecated anyways.
196
197 // For horizontal scrolling we need to flip the value for RTL grids.
198 if (this.isRtl_) {
199 deltaX = -deltaX;
200 }
201 var newEvent = new goog.events.MouseWheelEvent(detail, be, deltaX, deltaY);
202 this.dispatchEvent(newEvent);
203};

206/**
207 * Helper for scaling down a mousewheel delta by a scale factor, if appropriate.
208 * @param {number} mouseWheelDelta Delta from a mouse wheel event. Expected to
209 * be an integer.
210 * @param {number} scaleFactor Factor to scale the delta down by. Expected to
211 * be an integer.
212 * @return {number} Scaled-down delta value, or the original delta if the
213 * scaleFactor does not appear to be applicable.
214 * @private
215 */
216goog.events.MouseWheelHandler.smartScale_ = function(mouseWheelDelta,
217 scaleFactor) {
218 // The basic problem here is that in Webkit on Mac and Linux, we can get two
219 // very different types of mousewheel events: from continuous devices
220 // (touchpads, Mighty Mouse) or non-continuous devices (normal wheel mice).
221 //
222 // Non-continuous devices in Webkit get their wheel deltas scaled up to
223 // behave like IE. Continuous devices return much smaller unscaled values
224 // (which most of the time will not be cleanly divisible by the IE scale
225 // factor), so we should not try to normalize them down.
226 //
227 // Detailed discussion:
228 // https://bugs.webkit.org/show_bug.cgi?id=29601
229 // http://trac.webkit.org/browser/trunk/WebKit/chromium/src/mac/WebInputEventFactory.mm#L1063
230 if (goog.userAgent.WEBKIT &&
231 (goog.userAgent.MAC || goog.userAgent.LINUX) &&
232 (mouseWheelDelta % scaleFactor) != 0) {
233 return mouseWheelDelta;
234 } else {
235 return mouseWheelDelta / scaleFactor;
236 }
237};

249/**
250 * A base class for mouse wheel events. This is used with the
251 * MouseWheelHandler.
252 *
253 * @param {number} detail The number of rows the user scrolled.
254 * @param {Event} browserEvent Browser event object.
255 * @param {number} deltaX The number of rows the user scrolled in the X
256 * direction.
257 * @param {number} deltaY The number of rows the user scrolled in the Y
258 * direction.
259 * @constructor
260 * @extends {goog.events.BrowserEvent}
261 * @final
262 */
263goog.events.MouseWheelEvent = function(detail, browserEvent, deltaX, deltaY) {
264 goog.events.BrowserEvent.call(this, browserEvent);
265
266 this.type = goog.events.MouseWheelHandler.EventType.MOUSEWHEEL;
267
268 /**
269 * The number of lines the user scrolled
270 * @type {number}
271 * NOTE: Informally deprecated. Use deltaX and deltaY instead, they provide
272 * more information.
273 */
274 this.detail = detail;
275
276 /**
277 * The number of "lines" scrolled in the X direction.
278 *
279 * Note that not all browsers provide enough information to distinguish
280 * horizontal and vertical scroll events, so for these unsupported browsers,
281 * we will always have a deltaX of 0, even if the user scrolled their mouse
282 * wheel or trackpad sideways.
283 *
284 * Currently supported browsers are Webkit and Firefox 3.1 or later.
285 *
286 * @type {number}
287 */
288 this.deltaX = deltaX;
289
290 /**
291 * The number of lines scrolled in the Y direction.
292 * @type {number}
293 */
294 this.deltaY = deltaY;
295};
296goog.inherits(goog.events.MouseWheelEvent, goog.events.BrowserEvent);
