44/**
45 * A class representing a group of one or more "drag lists" with items that can
46 * be dragged within them and between them.
47 *
48 * Example usage:
49 * var dragListGroup = new goog.fx.DragListGroup();
50 * dragListGroup.setDragItemHandleHoverClass(className1, className2);
51 * dragListGroup.setDraggerElClass(className3);
52 * dragListGroup.addDragList(vertList, goog.fx.DragListDirection.DOWN);
53 * dragListGroup.addDragList(horizList, goog.fx.DragListDirection.RIGHT);
54 * dragListGroup.init();
55 *
56 * @extends {goog.events.EventTarget}
57 * @constructor
58 * @struct
59 */
60goog.fx.DragListGroup = function() {
61 goog.fx.DragListGroup.base(this, 'constructor');
62
63 /**
64 * The user-supplied CSS classes to add to a drag item on hover (not during a
65 * drag action).
66 * @private {Array|undefined}
67 */
68 this.dragItemHoverClasses_;
69
70 /**
71 * The user-supplied CSS classes to add to a drag item handle on hover (not
72 * during a drag action).
73 * @private {Array|undefined}
74 */
75 this.dragItemHandleHoverClasses_;
76
77 /**
78 * The user-supplied CSS classes to add to the current drag item (during a
79 * drag action).
80 * @private {Array|undefined}
81 */
82 this.currDragItemClasses_;
83
84 /**
85 * The user-supplied CSS classes to add to the clone of the current drag item
86 * that's actually being dragged around (during a drag action).
87 * @private {Array<string>|undefined}
88 */
89 this.draggerElClasses_;
90
91 /**
92 * The current drag item being moved.
93 * Note: This is only defined while a drag action is happening.
94 * @private {Element}
95 */
96 this.currDragItem_;
97
98 /**
99 * The drag list that {@code this.currDragItem_} is currently hovering over,
100 * or null if it is not hovering over a list.
101 * @private {Element}
102 */
103 this.currHoverList_;
104
105 /**
106 * The original drag list that the current drag item came from. We need to
107 * remember this in case the user drops the item outside of any lists, in
108 * which case we return the item to its original location.
109 * Note: This is only defined while a drag action is happening.
110 * @private {Element}
111 */
112 this.origList_;
113
114 /**
115 * The original next item in the original list that the current drag item came
116 * from. We need to remember this in case the user drops the item outside of
117 * any lists, in which case we return the item to its original location.
118 * Note: This is only defined while a drag action is happening.
119 * @private {Element}
120 */
121 this.origNextItem_;
122
123 /**
124 * The current item in the list we are hovering over. We need to remember
125 * this in case we do not update the position of the current drag item while
126 * dragging (see {@code updateWhileDragging_}). In this case the current drag
127 * item will be inserted into the list before this element when the drag ends.
128 * @private {Element}
129 */
130 this.currHoverItem_;
131
132 /**
133 * The clone of the current drag item that's actually being dragged around.
134 * Note: This is only defined while a drag action is happening.
135 * @private {HTMLElement}
136 */
137 this.draggerEl_;
138
139 /**
140 * The dragger object.
141 * Note: This is only defined while a drag action is happening.
142 * @private {goog.fx.Dragger}
143 */
144 this.dragger_;
145
146 /**
147 * The amount of distance, in pixels, after which a mousedown or touchstart is
148 * considered a drag.
149 * @private {number}
150 */
151 this.hysteresisDistance_ = 0;
152
153
154 /**
155 * The drag lists.
156 * @private {Array<Element>}
157 */
158 this.dragLists_ = [];
159
160 /**
161 * All the drag items. Set by init().
162 * @private {Array<Element>}
163 */
164 this.dragItems_ = [];
165
166 /**
167 * Which drag item corresponds to a given handle. Set by init().
168 * Specifically, this maps from the unique ID (as given by goog.getUid)
169 * of the handle to the drag item.
170 * @private {Object}
171 */
172 this.dragItemForHandle_ = {};
173
174 /**
175 * The event handler for this instance.
176 * @private {goog.events.EventHandler<!goog.fx.DragListGroup>}
177 */
178 this.eventHandler_ = new goog.events.EventHandler(this);
179
180 /**
181 * Whether the setup has been done to make all items in all lists draggable.
182 * @private {boolean}
183 */
184 this.isInitialized_ = false;
185
186 /**
187 * Whether the currDragItem is always displayed. By default the list
188 * collapses, the currDragItem's display is set to none, when we do not
189 * hover over a draglist.
190 * @private {boolean}
191 */
192 this.isCurrDragItemAlwaysDisplayed_ = false;
193
194 /**
195 * Whether to update the position of the currDragItem as we drag, i.e.,
196 * insert the currDragItem each time to the position where it would land if
197 * we were to end the drag at that point. Defaults to true.
198 * @private {boolean}
199 */
200 this.updateWhileDragging_ = true;
201};

368/**
369 * Performs the initial setup to make all items in all lists draggable.
370 */
371goog.fx.DragListGroup.prototype.init = function() {
372 if (this.isInitialized_) {
373 return;
374 }
375
376 for (var i = 0, numLists = this.dragLists_.length; i < numLists; i++) {
377 var dragList = this.dragLists_[i];
378
379 var dragItems = goog.dom.getChildren(dragList);
380 for (var j = 0, numItems = dragItems.length; j < numItems; ++j) {
381 this.listenForDragEvents(dragItems[j]);
382 }
383 }
384
385 this.isInitialized_ = true;
386};

389/**
390 * Adds a single item to the given drag list and sets up the drag listeners for
391 * it.
392 * If opt_index is specified the item is inserted at this index, otherwise the
393 * item is added as the last child of the list.
394 *
395 * @param {!Element} list The drag list where to add item to.
396 * @param {!Element} item The new element to add.
397 * @param {number=} opt_index Index where to insert the item in the list. If not
398 * specified item is inserted as the last child of list.
399 */
400goog.fx.DragListGroup.prototype.addItemToDragList = function(list, item,
401 opt_index) {
402 if (goog.isDef(opt_index)) {
403 goog.dom.insertChildAt(list, item, opt_index);
404 } else {
405 goog.dom.appendChild(list, item);
406 }
407 this.listenForDragEvents(item);
408};

411/** @override */
412goog.fx.DragListGroup.prototype.disposeInternal = function() {
413 this.eventHandler_.dispose();
414
415 for (var i = 0, n = this.dragLists_.length; i < n; i++) {
416 var dragList = this.dragLists_[i];
417 // Note: IE doesn't allow 'delete' for fields on HTML elements (because
418 // they're not real JS objects in IE), so we just set them to undefined.
419 dragList.dlgGrowthDirection_ = undefined;
420 dragList.dlgDragHoverClass_ = undefined;
421 }
422
423 this.dragLists_.length = 0;
424 this.dragItems_.length = 0;
425 this.dragItemForHandle_ = null;
426
427 // In the case where a drag event is currently in-progress and dispose is
428 // called, this cleans up the extra state.
429 this.cleanupDragDom_();
430
431 goog.fx.DragListGroup.superClass_.disposeInternal.call(this);
432};

435/**
436 * Caches the heights of each drag list and drag item, except for the current
437 * drag item.
438 *
439 * @param {Element} currDragItem The item currently being dragged.
440 * @private
441 */
442goog.fx.DragListGroup.prototype.recacheListAndItemBounds_ = function(
443 currDragItem) {
444 for (var i = 0, n = this.dragLists_.length; i < n; i++) {
445 var dragList = this.dragLists_[i];
446 dragList.dlgBounds_ = goog.style.getBounds(dragList);
447 }
448
449 for (var i = 0, n = this.dragItems_.length; i < n; i++) {
450 var dragItem = this.dragItems_[i];
451 if (dragItem != currDragItem) {
452 dragItem.dlgBounds_ = goog.style.getBounds(dragItem);
453 }
454 }
455};

458/**
459 * Listens for drag events on the given drag item. This method is currently used
460 * to initialize drag items.
461 *
462 * @param {Element} dragItem the element to initialize. This element has to be
463 * in one of the drag lists.
464 * @protected
465 */
466goog.fx.DragListGroup.prototype.listenForDragEvents = function(dragItem) {
467 var dragItemHandle = this.getHandleForDragItem_(dragItem);
468 var uid = goog.getUid(dragItemHandle);
469 this.dragItemForHandle_[uid] = dragItem;
470
471 if (this.dragItemHoverClasses_) {
472 this.eventHandler_.listen(
473 dragItem, goog.events.EventType.MOUSEOVER,
474 this.handleDragItemMouseover_);
475 this.eventHandler_.listen(
476 dragItem, goog.events.EventType.MOUSEOUT,
477 this.handleDragItemMouseout_);
478 }
479 if (this.dragItemHandleHoverClasses_) {
480 this.eventHandler_.listen(
481 dragItemHandle, goog.events.EventType.MOUSEOVER,
482 this.handleDragItemHandleMouseover_);
483 this.eventHandler_.listen(
484 dragItemHandle, goog.events.EventType.MOUSEOUT,
485 this.handleDragItemHandleMouseout_);
486 }
487
488 this.dragItems_.push(dragItem);
489 this.eventHandler_.listen(dragItemHandle,
490 [goog.events.EventType.MOUSEDOWN, goog.events.EventType.TOUCHSTART],
491 this.handlePotentialDragStart_);
492};

495/**
496 * Handles mouse and touch events which may start a drag action.
497 * @param {!goog.events.BrowserEvent} e MOUSEDOWN or TOUCHSTART event.
498 * @private
499 */
500goog.fx.DragListGroup.prototype.handlePotentialDragStart_ = function(e) {
501 var uid = goog.getUid(/** @type {Node} */ (e.currentTarget));
502 this.currDragItem_ = /** @type {Element} */ (this.dragItemForHandle_[uid]);
503
504 this.draggerEl_ = /** @type {!HTMLElement} */ (
505 this.createDragElementInternal(this.currDragItem_));
506 if (this.draggerElClasses_) {
507 // Add CSS class for the clone, if any.
508 goog.dom.classlist.addAll(
509 goog.asserts.assert(this.draggerEl_), this.draggerElClasses_ || []);
510 }
511
512 // Place the clone (i.e. draggerEl) at the same position as the actual
513 // current drag item. This is a bit tricky since
514 // goog.style.getPageOffset() gets the left-top pos of the border, but
515 // goog.style.setPageOffset() sets the left-top pos of the margin.
516 // It's difficult to adjust for the margins of the clone because it's
517 // difficult to read it: goog.style.getComputedStyle() doesn't work for IE.
518 // Instead, our workaround is simply to set the clone's margins to 0px.
519 this.draggerEl_.style.margin = '0';
520 this.draggerEl_.style.position = 'absolute';
521 this.draggerEl_.style.visibility = 'hidden';
522 var doc = goog.dom.getOwnerDocument(this.currDragItem_);
523 doc.body.appendChild(this.draggerEl_);
524
525 // Important: goog.style.setPageOffset() only works correctly for IE when the
526 // element is already in the document.
527 var currDragItemPos = goog.style.getPageOffset(this.currDragItem_);
528 goog.style.setPageOffset(this.draggerEl_, currDragItemPos);
529
530 this.dragger_ = new goog.fx.Dragger(this.draggerEl_);
531 this.dragger_.setHysteresis(this.hysteresisDistance_);
532
533 // Listen to events on the dragger. These handlers will be unregistered at
534 // DRAGEND, when the dragger is disposed of. We can't use eventHandler_,
535 // because it creates new references to the handler functions at each
536 // dragging action, and keeps them until DragListGroup is disposed of.
537 goog.events.listen(this.dragger_, goog.fx.Dragger.EventType.START,
538 this.handleDragStart_, false, this);
539 goog.events.listen(this.dragger_, goog.fx.Dragger.EventType.END,
540 this.handleDragEnd_, false, this);
541 goog.events.listen(this.dragger_, goog.fx.Dragger.EventType.EARLY_CANCEL,
542 this.cleanup_, false, this);
543 this.dragger_.startDrag(e);
544};

577/**
578 * Handles the start of a drag action.
579 * @param {!goog.fx.DragEvent} e goog.fx.Dragger.EventType.START event.
580 * @private
581 */
582goog.fx.DragListGroup.prototype.handleDragStart_ = function(e) {
583 if (!this.dispatchEvent(new goog.fx.DragListGroupEvent(
584 goog.fx.DragListGroup.EventType.BEFOREDRAGSTART, this, e.browserEvent,
585 this.currDragItem_, null, null))) {
586 e.preventDefault();
587 this.cleanup_();
588 return;
589 }
590
591 // Record the original location of the current drag item.
592 // Note: this.origNextItem_ may be null.
593 this.origList_ = /** @type {Element} */ (this.currDragItem_.parentNode);
594 this.origNextItem_ = goog.dom.getNextElementSibling(this.currDragItem_);
595 this.currHoverItem_ = this.origNextItem_;
596 this.currHoverList_ = this.origList_;
597
598 // If there's a CSS class specified for the current drag item, add it.
599 // Otherwise, make the actual current drag item hidden (takes up space).
600 if (this.currDragItemClasses_) {
601 goog.dom.classlist.addAll(
602 goog.asserts.assert(this.currDragItem_),
603 this.currDragItemClasses_ || []);
604 } else {
605 this.currDragItem_.style.visibility = 'hidden';
606 }
607
608 // Precompute distances from top-left corner to center for efficiency.
609 var draggerElSize = goog.style.getSize(this.draggerEl_);
610 this.draggerEl_.halfWidth = draggerElSize.width / 2;
611 this.draggerEl_.halfHeight = draggerElSize.height / 2;
612
613 this.draggerEl_.style.visibility = '';
614
615 // Record the bounds of all the drag lists and all the other drag items. This
616 // caching is for efficiency, so that we don't have to recompute the bounds on
617 // each drag move. Do this in the state where the current drag item is not in
618 // any of the lists, except when update while dragging is disabled, as in this
619 // case the current drag item does not get removed until drag ends.
620 if (this.updateWhileDragging_) {
621 this.currDragItem_.style.display = 'none';
622 }
623 this.recacheListAndItemBounds_(this.currDragItem_);
624 this.currDragItem_.style.display = '';
625
626 // Listen to events on the dragger.
627 goog.events.listen(this.dragger_, goog.fx.Dragger.EventType.DRAG,
628 this.handleDragMove_, false, this);
629
630 this.dispatchEvent(
631 new goog.fx.DragListGroupEvent(
632 goog.fx.DragListGroup.EventType.DRAGSTART, this, e.browserEvent,
633 this.currDragItem_, this.draggerEl_, this.dragger_));
634};

637/**
638 * Handles a drag movement (i.e. DRAG event fired by the dragger).
639 *
640 * @param {goog.fx.DragEvent} dragEvent Event object fired by the dragger.
641 * @return {boolean} The return value for the event.
642 * @private
643 */
644goog.fx.DragListGroup.prototype.handleDragMove_ = function(dragEvent) {
645
646 // Compute the center of the dragger element (i.e. the cloned drag item).
647 var draggerElPos = goog.style.getPageOffset(this.draggerEl_);
648 var draggerElCenter = new goog.math.Coordinate(
649 draggerElPos.x + this.draggerEl_.halfWidth,
650 draggerElPos.y + this.draggerEl_.halfHeight);
651
652 // Check whether the center is hovering over one of the drag lists.
653 var hoverList = this.getHoverDragList_(draggerElCenter);
654
655 // If hovering over a list, find the next item (if drag were to end now).
656 var hoverNextItem =
657 hoverList ? this.getHoverNextItem_(hoverList, draggerElCenter) : null;
658
659 var rv = this.dispatchEvent(
660 new goog.fx.DragListGroupEvent(
661 goog.fx.DragListGroup.EventType.BEFOREDRAGMOVE, this, dragEvent,
662 this.currDragItem_, this.draggerEl_, this.dragger_,
663 draggerElCenter, hoverList, hoverNextItem));
664 if (!rv) {
665 return false;
666 }
667
668 if (hoverList) {
669 if (this.updateWhileDragging_) {
670 this.insertCurrDragItem_(hoverList, hoverNextItem);
671 } else {
672 // If update while dragging is disabled do not insert
673 // the dragged item, but update the hovered item instead.
674 this.updateCurrHoverItem(hoverNextItem, draggerElCenter);
675 }
676 this.currDragItem_.style.display = '';
677 // Add drag list's hover class (if any).
678 if (hoverList.dlgDragHoverClass_) {
679 goog.dom.classlist.add(
680 goog.asserts.assert(hoverList), hoverList.dlgDragHoverClass_);
681 }
682
683 } else {
684 // Not hovering over a drag list, so remove the item altogether unless
685 // specified otherwise by the user.
686 if (!this.isCurrDragItemAlwaysDisplayed_) {
687 this.currDragItem_.style.display = 'none';
688 }
689
690 // Remove hover classes (if any) from all drag lists.
691 for (var i = 0, n = this.dragLists_.length; i < n; i++) {
692 var dragList = this.dragLists_[i];
693 if (dragList.dlgDragHoverClass_) {
694 goog.dom.classlist.remove(

886/**
887 * Helper for handleDragMove_().
888 * Given the position of the center of the dragger element, figures out whether
889 * it's currently hovering over any of the drag lists.
890 *
891 * @param {goog.math.Coordinate} draggerElCenter The center position of the
892 * dragger element.
893 * @return {Element} If currently hovering over a drag list, returns the drag
894 * list element. Else returns null.
895 * @private
896 */
897goog.fx.DragListGroup.prototype.getHoverDragList_ = function(draggerElCenter) {
898
899 // If the current drag item was in a list last time we did this, then check
900 // that same list first.
901 var prevHoverList = null;
902 if (this.currDragItem_.style.display != 'none') {
903 prevHoverList = /** @type {Element} */ (this.currDragItem_.parentNode);
904 // Important: We can't use the cached bounds for this list because the
905 // cached bounds are based on the case where the current drag item is not
906 // in the list. Since the current drag item is known to be in this list, we
907 // must recompute the list's bounds.
908 var prevHoverListBounds = goog.style.getBounds(prevHoverList);
909 if (this.isInRect_(draggerElCenter, prevHoverListBounds)) {
910 return prevHoverList;
911 }
912 }
913
914 for (var i = 0, n = this.dragLists_.length; i < n; i++) {
915 var dragList = this.dragLists_[i];
916 if (dragList == prevHoverList) {
917 continue;
918 }
919 if (this.isInRect_(draggerElCenter, dragList.dlgBounds_)) {
920 return dragList;
921 }
922 }
923
924 return null;
925};

977/**
978 * Helper for handleDragMove_().
979 * Given the position of the center of the dragger element, plus the drag list
980 * that it's currently hovering over, figures out the next drag item in the
981 * list that follows the current position of the dragger element. (I.e. if
982 * the drag action ends right now, it would become the item after the current
983 * drag item.)
984 *
985 * @param {Element} hoverList The drag list that we're hovering over.
986 * @param {goog.math.Coordinate} draggerElCenter The center position of the
987 * dragger element.
988 * @return {Element} Returns the earliest item in the hover list that belongs
989 * after the current position of the dragger element. If all items in the
990 * list should come before the current drag item, then returns null.
991 * @private
992 */
993goog.fx.DragListGroup.prototype.getHoverNextItem_ = function(
994 hoverList, draggerElCenter) {
995 if (hoverList == null) {
996 throw Error('getHoverNextItem_ called with null hoverList.');
997 }
998
999 // The definition of what it means for the draggerEl to be "before" a given
1000 // item in the hover drag list is not always the same. It changes based on
1001 // the growth direction of the hover drag list in question.
1002 /** @type {number} */
1003 var relevantCoord;
1004 var getRelevantBoundFn;
1005 var isBeforeFn;
1006 var pickClosestRow = false;
1007 var distanceToClosestRow = undefined;
1008 switch (hoverList.dlgGrowthDirection_) {
1009 case goog.fx.DragListDirection.DOWN:
1010 // "Before" means draggerElCenter.y is less than item's bottom y-value.
1011 relevantCoord = draggerElCenter.y;
1012 getRelevantBoundFn = goog.fx.DragListGroup.getBottomBound_;
1013 isBeforeFn = goog.fx.DragListGroup.isLessThan_;
1014 break;
1015 case goog.fx.DragListDirection.RIGHT_2D:
1016 pickClosestRow = true;
1017 case goog.fx.DragListDirection.RIGHT:
1018 // "Before" means draggerElCenter.x is less than item's right x-value.
1019 relevantCoord = draggerElCenter.x;
1020 getRelevantBoundFn = goog.fx.DragListGroup.getRightBound_;
1021 isBeforeFn = goog.fx.DragListGroup.isLessThan_;
1022 break;
1023 case goog.fx.DragListDirection.LEFT_2D:
1024 pickClosestRow = true;
1025 case goog.fx.DragListDirection.LEFT:
1026 // "Before" means draggerElCenter.x is greater than item's left x-value.
1027 relevantCoord = draggerElCenter.x;
1028 getRelevantBoundFn = goog.fx.DragListGroup.getLeftBound_;
1029 isBeforeFn = goog.fx.DragListGroup.isGreaterThan_;
1030 break;
1031 }
1032
1033 // This holds the earliest drag item found so far that should come after
1034 // this.currDragItem_ in the hover drag list (based on draggerElCenter).
1035 var earliestAfterItem = null;
1036 // This is the position of the relevant bound for the earliestAfterItem,
1037 // where "relevant" is determined by the growth direction of hoverList.
1038 var earliestAfterItemRelevantBound;
1039
1040 var hoverListItems = goog.dom.getChildren(hoverList);
1041 for (var i = 0, n = hoverListItems.length; i < n; i++) {
1042 var item = hoverListItems[i];
1043 if (item == this.currDragItem_) {
1044 continue;
1045 }
1046
1047 var relevantBound = getRelevantBoundFn(item.dlgBounds_);
1048 // When the hoverlist is broken into multiple rows (i.e., in the case of
1049 // LEFT_2D and RIGHT_2D) it is no longer enough to only look at the
1050 // x-coordinate alone in order to find the {@earliestAfterItem} in the
1051 // hoverlist. Make sure it is chosen from the row closest to the
1052 // {@code draggerElCenter}.
1053 if (pickClosestRow) {
1054 var distanceToRow = goog.fx.DragListGroup.verticalDistanceFromItem_(item,
1055 draggerElCenter);
1056 // Initialize the distance to the closest row to the current value if
1057 // undefined.
1058 if (!goog.isDef(distanceToClosestRow)) {
1059 distanceToClosestRow = distanceToRow;
1060 }
1061 if (isBeforeFn(relevantCoord, relevantBound) &&
1062 (earliestAfterItemRelevantBound == undefined ||
1063 (distanceToRow < distanceToClosestRow) ||
1064 ((distanceToRow == distanceToClosestRow) &&
1065 (isBeforeFn(relevantBound, earliestAfterItemRelevantBound) ||
1066 relevantBound == earliestAfterItemRelevantBound)))) {
1067 earliestAfterItem = item;
1068 earliestAfterItemRelevantBound = relevantBound;
1069 }
1070 // Update distance to closest row.
1071 if (distanceToRow < distanceToClosestRow) {
1072 distanceToClosestRow = distanceToRow;
1073 }
1074 } else if (isBeforeFn(relevantCoord, relevantBound) &&
1075 (earliestAfterItemRelevantBound == undefined ||
1076 isBeforeFn(relevantBound, earliestAfterItemRelevantBound))) {
1077 earliestAfterItem = item;
1078 earliestAfterItemRelevantBound = relevantBound;
1079 }
1080 }
1081 // If we ended up picking an element that is not in the closest row it can
1082 // only happen if we should have picked the last one in which case there is
1083 // no consecutive element.
1084 if (!goog.isNull(earliestAfterItem) &&
1085 goog.fx.DragListGroup.verticalDistanceFromItem_(
1086 earliestAfterItem, draggerElCenter) > distanceToClosestRow) {
1087 return null;
1088 } else {
1089 return earliestAfterItem;
1090 }
1091};
