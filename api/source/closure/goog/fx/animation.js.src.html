37/**
38 * Constructor for an animation object.
39 * @param {Array<number>} start Array for start coordinates.
40 * @param {Array<number>} end Array for end coordinates.
41 * @param {number} duration Length of animation in milliseconds.
42 * @param {Function=} opt_acc Acceleration function, returns 0-1 for inputs 0-1.
43 * @constructor
44 * @implements {goog.fx.anim.Animated}
45 * @implements {goog.fx.Transition}
46 * @extends {goog.fx.TransitionBase}
47 */
48goog.fx.Animation = function(start, end, duration, opt_acc) {
49 goog.fx.Animation.base(this, 'constructor');
50
51 if (!goog.isArray(start) || !goog.isArray(end)) {
52 throw Error('Start and end parameters must be arrays');
53 }
54
55 if (start.length != end.length) {
56 throw Error('Start and end points must be the same length');
57 }

59 /**
60 * Start point.
61 * @type {Array<number>}
62 * @protected
63 */
64 this.startPoint = start;
65
66 /**
67 * End point.
68 * @type {Array<number>}
69 * @protected
70 */
71 this.endPoint = end;
72
73 /**
74 * Duration of animation in milliseconds.
75 * @type {number}
76 * @protected
77 */
78 this.duration = duration;
79
80 /**
81 * Acceleration function, which must return a number between 0 and 1 for
82 * inputs between 0 and 1.
83 * @type {Function|undefined}
84 * @private
85 */
86 this.accel_ = opt_acc;
87
88 /**
89 * Current coordinate for animation.
90 * @type {Array<number>}
91 * @protected
92 */
93 this.coords = [];

95 /**
96 * Whether the animation should use "right" rather than "left" to position
97 * elements in RTL. This is a temporary flag to allow clients to transition
98 * to the new behavior at their convenience. At some point it will be the
99 * default.
100 * @type {boolean}
101 * @private
102 */
103 this.useRightPositioningForRtl_ = false;

105 /**
106 * Current frame rate.
107 * @private {number}
108 */
109 this.fps_ = 0;

111 /**
112 * Percent of the way through the animation.
113 * @protected {number}

118 * Timestamp for when last frame was run.

237/**
238 * Starts or resumes an animation.
239 * @param {boolean=} opt_restart Whether to restart the
240 * animation from the beginning if it has been paused.
241 * @return {boolean} Whether animation was started.
242 * @override
243 */
244goog.fx.Animation.prototype.play = function(opt_restart) {
245 if (opt_restart || this.isStopped()) {
246 this.progress = 0;
247 this.coords = this.startPoint;
248 } else if (this.isPlaying()) {
249 return false;
250 }
251
252 goog.fx.anim.unregisterAnimation(this);
253
254 var now = /** @type {number} */ (goog.now());
255
256 this.startTime = now;
257 if (this.isPaused()) {
258 this.startTime -= this.duration * this.progress;
259 }
260
261 this.endTime = this.startTime + this.duration;
262 this.lastFrame = this.startTime;
263
264 if (!this.progress) {
265 this.onBegin();
266 }
267
268 this.onPlay();
269
270 if (this.isPaused()) {
271 this.onResume();
272 }
273
274 this.setStatePlaying();
275
276 goog.fx.anim.registerAnimation(this);
277 this.cycle(now);
278
279 return true;
280};

283/**
284 * Stops the animation.
285 * @param {boolean=} opt_gotoEnd If true the animation will move to the
286 * end coords.
287 * @override
288 */
289goog.fx.Animation.prototype.stop = function(opt_gotoEnd) {
290 goog.fx.anim.unregisterAnimation(this);
291 this.setStateStopped();
292
293 if (!!opt_gotoEnd) {
294 this.progress = 1;
295 }
296
297 this.updateCoords_(this.progress);
298
299 this.onStop();
300 this.onEnd();
301};

326/**
327 * Sets the progress of the animation.
328 * @param {number} progress The new progress of the animation.
329 */
330goog.fx.Animation.prototype.setProgress = function(progress) {
331 this.progress = progress;
332 if (this.isPlaying()) {
333 var now = goog.now();
334 // If the animation is already playing, we recompute startTime and endTime
335 // such that the animation plays consistently, that is:
336 // now = startTime + progress * duration.
337 this.startTime = now - this.duration * this.progress;
338 this.endTime = this.startTime + this.duration;
339 }
340};

374/**
375 * Handles the actual iteration of the animation in a timeout
376 * @param {number} now The current time.
377 */
378goog.fx.Animation.prototype.cycle = function(now) {
379 this.progress = (now - this.startTime) / (this.endTime - this.startTime);
380
381 if (this.progress >= 1) {
382 this.progress = 1;
383 }
384
385 this.fps_ = 1000 / (now - this.lastFrame);
386 this.lastFrame = now;
387
388 this.updateCoords_(this.progress);
389
390 // Animation has finished.
391 if (this.progress == 1) {
392 this.setStateStopped();
393 goog.fx.anim.unregisterAnimation(this);
394
395 this.onFinish();
396 this.onEnd();
397
398 // Animation is still under way.
399 } else if (this.isPlaying()) {
400 this.onAnimate();
401 }
402};

405/**
406 * Calculates current coordinates, based on the current state. Applies
407 * the accelleration function if it exists.
408 * @param {number} t Percentage of the way through the animation as a decimal.
409 * @private
410 */
411goog.fx.Animation.prototype.updateCoords_ = function(t) {
412 if (goog.isFunction(this.accel_)) {
413 t = this.accel_(t);
414 }
415 this.coords = new Array(this.startPoint.length);
416 for (var i = 0; i < this.startPoint.length; i++) {
417 this.coords[i] = (this.endPoint[i] - this.startPoint[i]) * t +
418 this.startPoint[i];
419 }
420};

450/**
451 * Class for an animation event object.
452 * @param {string} type Event type.
453 * @param {goog.fx.Animation} anim An animation object.
454 * @constructor
455 * @extends {goog.events.Event}
456 */
457goog.fx.AnimationEvent = function(type, anim) {
458 goog.fx.AnimationEvent.base(this, 'constructor', type);
459
460 /**
461 * The current coordinates.
462 * @type {Array<number>}
463 */
464 this.coords = anim.coords;
465
466 /**
467 * The x coordinate.
468 * @type {number}
469 */
470 this.x = anim.coords[0];
471
472 /**
473 * The y coordinate.
474 * @type {number}
475 */
476 this.y = anim.coords[1];
477
478 /**
479 * The z coordinate.
480 * @type {number}
481 */
482 this.z = anim.coords[2];
483
484 /**
485 * The current duration.
486 * @type {number}
487 */
488 this.duration = anim.duration;
489
490 /**
491 * The current progress.
492 * @type {number}
493 */
494 this.progress = anim.getProgress();
495
496 /**
497 * Frames per second so far.
498 */
499 this.fps = anim.fps_;
500
501 /**
502 * The state of the animation.
503 * @type {number}
504 */
505 this.state = anim.getStateInternal();
506
507 /**
508 * The animation object.
509 * @type {goog.fx.Animation}
510 */
511 // TODO(arv): This can be removed as this is the same as the target
512 this.anim = anim;
513};
514goog.inherits(goog.fx.AnimationEvent, goog.events.Event);
