44/**
45 * A class that allows mouse or touch-based dragging (moving) of an element
46 *
47 * @param {Element} target The element that will be dragged.
48 * @param {Element=} opt_handle An optional handle to control the drag, if null
49 * the target is used.
50 * @param {goog.math.Rect=} opt_limits Object containing left, top, width,
51 * and height.
52 *
53 * @extends {goog.events.EventTarget}
54 * @constructor
55 * @struct
56 */
57goog.fx.Dragger = function(target, opt_handle, opt_limits) {
58 goog.fx.Dragger.base(this, 'constructor');
59
60 /**
61 * Reference to drag target element.
62 * @type {?Element}
63 */
64 this.target = target;
65
66 /**
67 * Reference to the handler that initiates the drag.
68 * @type {?Element}
69 */
70 this.handle = opt_handle || target;
71
72 /**
73 * Object representing the limits of the drag region.
74 * @type {goog.math.Rect}
75 */
76 this.limits = opt_limits || new goog.math.Rect(NaN, NaN, NaN, NaN);
77
78 /**
79 * Reference to a document object to use for the events.
80 * @private {Document}
81 */
82 this.document_ = goog.dom.getOwnerDocument(target);
83
84 /** @private {!goog.events.EventHandler} */
85 this.eventHandler_ = new goog.events.EventHandler(this);
86 this.registerDisposable(this.eventHandler_);
87
88 /**
89 * Whether the element is rendered right-to-left. We initialize this lazily.
90 * @private {boolean|undefined}}
91 */
92 this.rightToLeft_;
93
94 /**
95 * Current x position of mouse or touch relative to viewport.
96 * @type {number}
97 */
98 this.clientX = 0;
99
100 /**
101 * Current y position of mouse or touch relative to viewport.
102 * @type {number}
103 */
104 this.clientY = 0;
105
106 /**
107 * Current x position of mouse or touch relative to screen. Deprecated because
108 * it doesn't take into affect zoom level or pixel density.
109 * @type {number}
110 * @deprecated Consider switching to clientX instead.
111 */
112 this.screenX = 0;
113
114 /**
115 * Current y position of mouse or touch relative to screen. Deprecated because
116 * it doesn't take into affect zoom level or pixel density.
117 * @type {number}
118 * @deprecated Consider switching to clientY instead.
119 */
120 this.screenY = 0;
121
122 /**
123 * The x position where the first mousedown or touchstart occurred.
124 * @type {number}
125 */
126 this.startX = 0;
127
128 /**
129 * The y position where the first mousedown or touchstart occurred.
130 * @type {number}
131 */
132 this.startY = 0;
133
134 /**
135 * Current x position of drag relative to target's parent.
136 * @type {number}
137 */
138 this.deltaX = 0;
139
140 /**
141 * Current y position of drag relative to target's parent.
142 * @type {number}
143 */
144 this.deltaY = 0;
145
146 /**
147 * The current page scroll value.
148 * @type {?goog.math.Coordinate}
149 */
150 this.pageScroll;
151
152 /**
153 * Whether dragging is currently enabled.
154 * @private {boolean}
155 */
156 this.enabled_ = true;
157
158 /**
159 * Whether object is currently being dragged.
160 * @private {boolean}
161 */
162 this.dragging_ = false;
163
164 /**
165 * Whether mousedown should be default prevented.
166 * @private {boolean}
167 **/
168 this.preventMouseDown_ = true;
169
170 /**
171 * The amount of distance, in pixels, after which a mousedown or touchstart is
172 * considered a drag.
173 * @private {number}
174 */
175 this.hysteresisDistanceSquared_ = 0;
176
177 /**
178 * The SCROLL event target used to make drag element follow scrolling.
179 * @private {?EventTarget}
180 */
181 this.scrollTarget_;
182
183 /**
184 * Whether IE drag events cancelling is on.
185 * @private {boolean}
186 */
187 this.ieDragStartCancellingOn_ = false;
188
189 /**
190 * Whether the dragger implements the changes described in http://b/6324964,
191 * making it truly RTL. This is a temporary flag to allow clients to
192 * transition to the new behavior at their convenience. At some point it will
193 * be the default.
194 * @private {boolean}
195 */
196 this.useRightPositioningForRtl_ = false;
197
198 // Add listener. Do not use the event handler here since the event handler is
199 // used for listeners added and removed during the drag operation.
200 goog.events.listen(this.handle,
201 [goog.events.EventType.TOUCHSTART, goog.events.EventType.MOUSEDOWN],
202 this.startDrag, false, this);
203};

222/**
223 * Creates copy of node being dragged. This is a utility function to be used
224 * wherever it is inappropriate for the original source to follow the mouse
225 * cursor itself.
226 *
227 * @param {Element} sourceEl Element to copy.
228 * @return {!Element} The clone of {@code sourceEl}.
229 */
230goog.fx.Dragger.cloneNode = function(sourceEl) {
231 var clonedEl = /** @type {Element} */ (sourceEl.cloneNode(true)),
232 origTexts = sourceEl.getElementsByTagName(goog.dom.TagName.TEXTAREA),
233 dragTexts = clonedEl.getElementsByTagName(goog.dom.TagName.TEXTAREA);
234 // Cloning does not copy the current value of textarea elements, so correct
235 // this manually.
236 for (var i = 0; i < origTexts.length; i++) {
237 dragTexts[i].value = origTexts[i].value;
238 }
239 switch (sourceEl.tagName) {
240 case goog.dom.TagName.TR:
241 return goog.dom.createDom(goog.dom.TagName.TABLE, null,
242 goog.dom.createDom(goog.dom.TagName.TBODY,
243 null, clonedEl));
244 case goog.dom.TagName.TD:
245 case goog.dom.TagName.TH:
246 return goog.dom.createDom(
247 goog.dom.TagName.TABLE, null, goog.dom.createDom(
248 goog.dom.TagName.TBODY, null, goog.dom.createDom(
249 goog.dom.TagName.TR, null, clonedEl)));
250 case goog.dom.TagName.TEXTAREA:
251 clonedEl.value = sourceEl.value;
252 default:
253 return clonedEl;
254 }
255};

411/**
412 * Event handler that is used to start the drag
413 * @param {goog.events.BrowserEvent} e Event object.
414 */
415goog.fx.Dragger.prototype.startDrag = function(e) {
416 var isMouseDown = e.type == goog.events.EventType.MOUSEDOWN;
417
418 // Dragger.startDrag() can be called by AbstractDragDrop with a mousemove
419 // event and IE does not report pressed mouse buttons on mousemove. Also,
420 // it does not make sense to check for the button if the user is already
421 // dragging.
422
423 if (this.enabled_ && !this.dragging_ &&
424 (!isMouseDown || e.isMouseActionButton())) {
425 if (this.hysteresisDistanceSquared_ == 0) {
426 if (this.fireDragStart_(e)) {
427 this.dragging_ = true;
428 if (this.preventMouseDown_) {
429 e.preventDefault();
430 }
431 } else {
432 // If the start drag is cancelled, don't setup for a drag.
433 return;
434 }
435 } else if (this.preventMouseDown_) {
436 // Need to preventDefault for hysteresis to prevent page getting selected.
437 e.preventDefault();
438 }
439 this.setupDragHandlers();
440
441 this.clientX = this.startX = e.clientX;
442 this.clientY = this.startY = e.clientY;
443 this.screenX = e.screenX;
444 this.screenY = e.screenY;
445 this.computeInitialPosition();
446 this.pageScroll = goog.dom.getDomHelper(this.document_).getDocumentScroll();
447 } else {
448 this.dispatchEvent(goog.fx.Dragger.EventType.EARLY_CANCEL);
449 }
450};

453/**
454 * Sets up event handlers when dragging starts.
455 * @protected
456 */
457goog.fx.Dragger.prototype.setupDragHandlers = function() {
458 var doc = this.document_;
459 var docEl = doc.documentElement;
460 // Use bubbling when we have setCapture since we got reports that IE has
461 // problems with the capturing events in combination with setCapture.
462 var useCapture = !goog.fx.Dragger.HAS_SET_CAPTURE_;
463
464 this.eventHandler_.listen(doc,
465 [goog.events.EventType.TOUCHMOVE, goog.events.EventType.MOUSEMOVE],
466 this.handleMove_, useCapture);
467 this.eventHandler_.listen(doc,
468 [goog.events.EventType.TOUCHEND, goog.events.EventType.MOUSEUP],
469 this.endDrag, useCapture);
470
471 if (goog.fx.Dragger.HAS_SET_CAPTURE_) {
472 docEl.setCapture(false);
473 this.eventHandler_.listen(docEl,
474 goog.events.EventType.LOSECAPTURE,
475 this.endDrag);
476 } else {
477 // Make sure we stop the dragging if the window loses focus.
478 // Don't use capture in this listener because we only want to end the drag
479 // if the actual window loses focus. Since blur events do not bubble we use
480 // a bubbling listener on the window.
481 this.eventHandler_.listen(goog.dom.getWindow(doc),
482 goog.events.EventType.BLUR,
483 this.endDrag);
484 }
485
486 if (goog.userAgent.IE && this.ieDragStartCancellingOn_) {
487 // Cancel IE's 'ondragstart' event.
488 this.eventHandler_.listen(doc, goog.events.EventType.DRAGSTART,
489 goog.events.Event.preventDefault);
490 }
491
492 if (this.scrollTarget_) {
493 this.eventHandler_.listen(this.scrollTarget_, goog.events.EventType.SCROLL,
494 this.onScroll_, useCapture);
495 }
496};

524/**
525 * Event handler that is used to end the drag.
526 * @param {goog.events.BrowserEvent} e Event object.
527 * @param {boolean=} opt_dragCanceled Whether the drag has been canceled.
528 */
529goog.fx.Dragger.prototype.endDrag = function(e, opt_dragCanceled) {
530 this.cleanUpAfterDragging_();
531
532 if (this.dragging_) {
533 this.dragging_ = false;
534
535 var x = this.limitX(this.deltaX);
536 var y = this.limitY(this.deltaY);
537 var dragCanceled = opt_dragCanceled ||
538 e.type == goog.events.EventType.TOUCHCANCEL;
539 this.dispatchEvent(new goog.fx.DragEvent(
540 goog.fx.Dragger.EventType.END, this, e.clientX, e.clientY, e, x, y,
541 dragCanceled));
542 } else {
543 this.dispatchEvent(goog.fx.Dragger.EventType.EARLY_CANCEL);
544 }
545};

557/**
558 * Event handler that is used on mouse / touch move to update the drag
559 * @param {goog.events.BrowserEvent} e Event object.
560 * @private
561 */
562goog.fx.Dragger.prototype.handleMove_ = function(e) {
563 if (this.enabled_) {
564 // dx in right-to-left cases is relative to the right.
565 var sign = this.useRightPositioningForRtl_ &&
566 this.isRightToLeft_() ? -1 : 1;
567 var dx = sign * (e.clientX - this.clientX);
568 var dy = e.clientY - this.clientY;
569 this.clientX = e.clientX;
570 this.clientY = e.clientY;
571 this.screenX = e.screenX;
572 this.screenY = e.screenY;
573
574 if (!this.dragging_) {
575 var diffX = this.startX - this.clientX;
576 var diffY = this.startY - this.clientY;
577 var distance = diffX * diffX + diffY * diffY;
578 if (distance > this.hysteresisDistanceSquared_) {
579 if (this.fireDragStart_(e)) {
580 this.dragging_ = true;
581 } else {
582 // DragListGroup disposes of the dragger if BEFOREDRAGSTART is
583 // canceled.
584 if (!this.isDisposed()) {
585 this.endDrag(e);
586 }
587 return;
588 }
589 }
590 }
591
592 var pos = this.calculatePosition_(dx, dy);
593 var x = pos.x;
594 var y = pos.y;
595
596 if (this.dragging_) {
597
598 var rv = this.dispatchEvent(new goog.fx.DragEvent(
599 goog.fx.Dragger.EventType.BEFOREDRAG, this, e.clientX, e.clientY,
600 e, x, y));
601
602 // Only do the defaultAction and dispatch drag event if predrag didn't
603 // prevent default
604 if (rv) {
605 this.doDrag(e, x, y, false);
606 e.preventDefault();
607 }
608 }
609 }
610};

613/**
614 * Calculates the drag position.
615 *
616 * @param {number} dx The horizontal movement delta.
617 * @param {number} dy The vertical movement delta.
618 * @return {!goog.math.Coordinate} The newly calculated drag element position.
619 * @private
620 */
621goog.fx.Dragger.prototype.calculatePosition_ = function(dx, dy) {
622 // Update the position for any change in body scrolling
623 var pageScroll = goog.dom.getDomHelper(this.document_).getDocumentScroll();
624 dx += pageScroll.x - this.pageScroll.x;
625 dy += pageScroll.y - this.pageScroll.y;
626 this.pageScroll = pageScroll;
627
628 this.deltaX += dx;
629 this.deltaY += dy;
630
631 var x = this.limitX(this.deltaX);
632 var y = this.limitY(this.deltaY);
633 return new goog.math.Coordinate(x, y);
634};

642goog.fx.Dragger.prototype.onScroll_ = function(e) {

666/**
667 * Returns the 'real' x after limits are applied (allows for some
668 * limits to be undefined).
669 * @param {number} x X-coordinate to limit.
670 * @return {number} The 'real' X-coordinate after limits are applied.
671 */
672goog.fx.Dragger.prototype.limitX = function(x) {
673 var rect = this.limits;
674 var left = !isNaN(rect.left) ? rect.left : null;
675 var width = !isNaN(rect.width) ? rect.width : 0;
676 var maxX = left != null ? left + width : Infinity;
677 var minX = left != null ? left : -Infinity;
678 return Math.min(maxX, Math.max(minX, x));
679};
680
681
682/**
683 * Returns the 'real' y after limits are applied (allows for some
684 * limits to be undefined).
685 * @param {number} y Y-coordinate to limit.
686 * @return {number} The 'real' Y-coordinate after limits are applied.
687 */
688goog.fx.Dragger.prototype.limitY = function(y) {
689 var rect = this.limits;
690 var top = !isNaN(rect.top) ? rect.top : null;
691 var height = !isNaN(rect.height) ? rect.height : 0;
692 var maxY = top != null ? top + height : Infinity;
693 var minY = top != null ? top : -Infinity;
694 return Math.min(maxY, Math.max(minY, y));
695};

711/**
712 * Overridable function for handling the default action of the drag behaviour.
713 * Normally this is simply moving the element to x,y though in some cases it
714 * might be used to resize the layer. This is basically a shortcut to
715 * implementing a default ondrag event handler.
716 * @param {number} x X-coordinate for target element. In right-to-left, x this
717 * is the number of pixels the target should be moved to from the right.
718 * @param {number} y Y-coordinate for target element.
719 */
720goog.fx.Dragger.prototype.defaultAction = function(x, y) {
721 if (this.useRightPositioningForRtl_ && this.isRightToLeft_()) {
722 this.target.style.right = x + 'px';
723 } else {
724 this.target.style.left = x + 'px';
725 }
726 this.target.style.top = y + 'px';
727};
