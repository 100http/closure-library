35/**
36 * Topic-based publish/subscribe messaging implementation that provides
37 * communication between browsing contexts that share the same origin.
38 *
39 * Wrapper around PubSub that utilizes localStorage to broadcast publications to
40 * all browser windows with the same origin as the publishing context. This
41 * allows for topic-based publish/subscribe implementation of strings shared by
42 * all browser contexts that share the same origin.
43 *
44 * Delivery is guaranteed on all browsers except IE8 where topics expire after a
45 * timeout. Publishing of a topic within a callback function provides no
46 * guarantee on ordering in that there is a possiblilty that separate origin
47 * contexts may see topics in a different order.
48 *
49 * This class is not secure and in certain cases (e.g., a browser crash) data
50 * that is published can persist in localStorage indefinitely. Do not use this
51 * class to communicate private or confidential information.
52 *
53 * On IE8, localStorage is shared by the http and https origins. An attacker
54 * could possibly leverage this to publish to the secure origin.
55 *
56 * goog.labs.pubsub.BroadcastPubSub wraps an instance of PubSub rather than
57 * subclassing because the base PubSub class allows publishing of arbitrary
58 * objects.
59 *
60 * Special handling is done for the IE8 browsers. See the IE8_EVENTS_KEY_
61 * constant and the {@code publish} function for more information.
62 *
63 *
64 * @constructor @struct @extends {goog.Disposable}
65 */
66goog.labs.pubsub.BroadcastPubSub = function() {
67 goog.labs.pubsub.BroadcastPubSub.base(this, 'constructor');
68 goog.labs.pubsub.BroadcastPubSub.instances_.push(this);
69
70 /** @private @const */
71 this.pubSub_ = new goog.pubsub.PubSub();
72 this.registerDisposable(this.pubSub_);
73
74 /** @private @const */
75 this.handler_ = new goog.events.EventHandler(this);
76 this.registerDisposable(this.handler_);
77
78 /** @private @const */
79 this.logger_ = goog.log.getLogger('goog.labs.pubsub.BroadcastPubSub');
80
81 /** @private @const */
82 this.mechanism_ = new goog.storage.mechanism.HTML5LocalStorage();
83
84 /** @private {goog.storage.Storage} */
85 this.storage_ = null;
86
87 /** @private {Object<string, number>} */
88 this.ie8LastEventTimes_ = null;
89
90 /** @private {number} */
91 this.ie8StartupTimestamp_ = goog.now() - 1;
92
93 if (this.mechanism_.isAvailable()) {
94 this.storage_ = new goog.storage.Storage(this.mechanism_);
95
96 var target = window;
97 if (goog.labs.pubsub.BroadcastPubSub.IS_IE8_) {
98 this.ie8LastEventTimes_ = {};
99
100 target = document;
101 }
102 this.handler_.listen(target,
103 goog.events.EventType.STORAGE,
104 this.handleStorageEvent_);
105 }
106};
107goog.inherits(goog.labs.pubsub.BroadcastPubSub, goog.Disposable);

121/**
122 * Handle the storage event and possibly dispatch topics.
123 * @param {!goog.events.BrowserEvent} e Event object.
124 * @private
125 */
126goog.labs.pubsub.BroadcastPubSub.prototype.handleStorageEvent_ =
127 function(e) {
128 if (goog.labs.pubsub.BroadcastPubSub.IS_IE8_) {
129 // Even though we have the event, IE8 doesn't update our localStorage until
130 // after we handle the actual event.
131 goog.async.run(this.handleIe8StorageEvent_, this);
132 return;
133 }
134
135 var browserEvent = e.getBrowserEvent();
136 if (browserEvent.key !=
137 goog.labs.pubsub.BroadcastPubSub.STORAGE_KEY_) {
138 return;
139 }
140
141 var data = goog.json.parse(browserEvent.newValue);
142 var args = goog.isObject(data) && data['args'];
143 if (goog.isArray(args) && goog.array.every(args, goog.isString)) {
144 this.dispatch_(args);
145 } else {
146 goog.log.warning(this.logger_, 'storage event contained invalid arguments');
147 }
148};

161/**
162 * Publishes a message to a topic. Remote subscriptions in other tabs/windows
163 * are dispatched via local storage events. Local subscriptions are called
164 * asynchronously via Timer event in order to simulate remote behavior locally.
165 * @param {string} topic Topic to publish to.
166 * @param {...string} var_args String arguments that are applied to each
167 * subscription function.
168 */
169goog.labs.pubsub.BroadcastPubSub.prototype.publish =
170 function(topic, var_args) {
171 var args = goog.array.toArray(arguments);
172
173 // Dispatch to localStorage.
174 if (this.storage_) {
175 // Update topics to use the optional prefix.
176 var now = goog.now();
177 var data = {
178 'args': args,
179 'timestamp': now
180 };
181
182 if (!goog.labs.pubsub.BroadcastPubSub.IS_IE8_) {
183 // Generated events will contain all the data in modern browsers.
184 this.storage_.set(goog.labs.pubsub.BroadcastPubSub.STORAGE_KEY_, data);
185 this.storage_.remove(goog.labs.pubsub.BroadcastPubSub.STORAGE_KEY_);
186 } else {
187 // With IE8 we need to manage our own events queue.
188 var events = null;
189 /** @preserveTry */
190 try {
191 events = this.storage_.get(
192 goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_);
193 } catch (ex) {
194 goog.log.error(this.logger_,
195 'publish encountered invalid event queue at ' +
196 goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_);
197 }
198 if (!goog.isArray(events)) {
199 events = [];
200 }
201 // Avoid a race condition where we're publishing in the same
202 // millisecond that another event that may be getting
203 // processed. In short, we try go guarantee that whatever event
204 // we put on the event queue has a timestamp that is older than
205 // any other timestamp in the queue.
206 var lastEvent = events[events.length - 1];
207 var lastTimestamp = lastEvent && lastEvent['timestamp'] ||
208 this.ie8StartupTimestamp_;
209 if (lastTimestamp >= now) {
210 now = lastTimestamp +
211 goog.labs.pubsub.BroadcastPubSub.IE8_TIMESTAMP_UNIQUE_OFFSET_MS_;
212 data['timestamp'] = now;
213 }
214 events.push(data);
215 this.storage_.set(
216 goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_, events);
217
218 // Cleanup this event in IE8_EVENT_LIFETIME_MS_ milliseconds.
219 goog.Timer.callOnce(goog.bind(this.cleanupIe8StorageEvents_, this, now),
220 goog.labs.pubsub.BroadcastPubSub.IE8_EVENT_LIFETIME_MS_);
221 }
222 }
223
224 // W3C spec is to not dispatch the storage event to the same window that
225 // modified localStorage. For conforming browsers we have to manually dispatch
226 // the publish event to subscriptions on instances of BroadcastPubSub in the
227 // current window.
228 if (!goog.userAgent.IE) {
229 // Dispatch the publish event to local instances asynchronously to fix some
230 // quirks with timings. The result is that all subscriptions are dispatched
231 // before any future publishes are processed. The effect is that
232 // subscriptions in the same window are dispatched as if they are the result
233 // of a publish from another tab.
234 goog.array.forEach(goog.labs.pubsub.BroadcastPubSub.instances_,
235 function(instance) {
236 goog.async.run(goog.bind(instance.dispatch_, instance, args));
237 });
238 }
239};

371/**
372 * Name for this window/tab's storage key that stores its IE8 event queue.
373 *
374 * The browsers storage events are supposed to track the key which was changed,
375 * the previous value for that key, and the new value of that key. Our
376 * implementation is dependent on this information but IE8 doesn't provide it.
377 * We implement our own event queue using local storage to track this
378 * information in IE8. Since all instances share the same localStorage context
379 * in a particular tab, we share the events queue.
380 *
381 * This key is a static member shared by all instances of BroadcastPubSub in the
382 * same Window context. To avoid read-update-write contention, this key is only
383 * written in a single context in the cleanupIe8StorageEvents_ function. Since
384 * instances in other contexts will read this key there is code in the {@code
385 * publish} function to make sure timestamps are unique even within the same
386 * millisecond.
387 *
388 * @private @const
389 */
390goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_ =
391 goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_PREFIX_ +
392 goog.math.randomInt(1e9);

458/**
459 * Processes the events array for key if all elements are valid IE8 events.
460 * @param {string} key The key in localStorage where the event queue is stored.
461 * @param {!Array<!Object>} events Array of possible events stored at key.
462 * @return {boolean} Return true if all elements in the array are valid
463 * events, false otherwise.
464 * @private
465 */
466goog.labs.pubsub.BroadcastPubSub.prototype.maybeProcessIe8Events_ =
467 function(key, events) {
468 if (!events.length) {
469 return false;
470 }
471
472 var validEvents =
473 goog.labs.pubsub.BroadcastPubSub.filterValidIe8Events_(events);
474 if (validEvents.length == events.length) {
475 var lastTimestamp = goog.array.peek(validEvents)['timestamp'];
476 var previousTime =
477 this.ie8LastEventTimes_[key] || this.ie8StartupTimestamp_;
478 if (lastTimestamp > previousTime -
479 goog.labs.pubsub.BroadcastPubSub.IE8_QUEUE_LIFETIME_MS_) {
480 this.ie8LastEventTimes_[key] = lastTimestamp;
481 validEvents = goog.labs.pubsub.BroadcastPubSub.filterNewIe8Events_(
482 previousTime, validEvents);
483 for (var i = 0, event; event = validEvents[i]; i++) {
484 this.dispatch_(event['args']);
485 }
486 return true;
487 }
488 } else {
489 goog.log.warning(this.logger_, 'invalid events found in queue ' + key);
490 }
491
492 return false;
493};

496/**
497 * Handle the storage event and possibly dispatch events. Looks through all keys
498 * in localStorage for valid keys.
499 * @private
500 */
501goog.labs.pubsub.BroadcastPubSub.prototype.handleIe8StorageEvent_ = function() {
502 var numKeys = this.mechanism_.getCount();
503 for (var idx = 0; idx < numKeys; idx++) {
504 var key = this.mechanism_.key(idx);
505 // Don't process events we generated. The W3C standard says that storage
506 // events should be queued by the browser for each window whose document's
507 // storage object is affected by a change in localStorage. Chrome, Firefox,
508 // and modern IE don't dispatch the event to the window which made the
509 // change. This code simulates that behavior in IE8.
510 if (!(goog.isString(key) && goog.string.startsWith(
511 key, goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_PREFIX_))) {
512 continue;
513 }
514
515 var events = null;
516 /** @preserveTry */
517 try {
518 events = this.storage_.get(key);
519 } catch (ex) {
520 goog.log.warning(this.logger_, 'invalid remote event queue ' + key);
521 }
522
523 if (!(goog.isArray(events) && this.maybeProcessIe8Events_(key, events))) {
524 // Events is not an array, empty, contains invalid events, or expired.
525 this.storage_.remove(key);
526 }
527 }

531/**
532 * Cleanup our IE8 event queue by removing any events that come at or before the
533 * given timestamp.
534 * @param {number} timestamp Maximum timestamp to remove from the queue.
535 * @private
536 */
537goog.labs.pubsub.BroadcastPubSub.prototype.cleanupIe8StorageEvents_ =
538 function(timestamp) {
539 var events = null;
540 /** @preserveTry */
541 try {
542 events = this.storage_.get(
543 goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_);
544 } catch (ex) {
545 goog.log.error(this.logger_,
546 'cleanup encountered invalid event queue key ' +
547 goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_);
548 }
549 if (!goog.isArray(events)) {
550 this.storage_.remove(goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_);
551 return;
552 }
553
554 events = goog.labs.pubsub.BroadcastPubSub.filterNewIe8Events_(
555 timestamp, goog.labs.pubsub.BroadcastPubSub.filterValidIe8Events_(
556 events));
557
558 if (events.length > 0) {
559 this.storage_.set(goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_, events);
560 } else {
561 this.storage_.remove(goog.labs.pubsub.BroadcastPubSub.IE8_EVENTS_KEY_);
562 }
