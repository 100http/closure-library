15/**
16 * @fileoverview Provides a convenient API for data persistence with data
17 * expiration and number of items limit.
18 *
19 * Setting and removing values keeps a max number of items invariant.
20 * Collecting values can be user initiated. If oversize, first removes
21 * expired items, if still oversize than removes the oldest items until a size
22 * constraint is fulfilled.
23 *
24 */
25

41/**
42 * Provides a storage with bounded number of elements, expiring keys and
43 * a collection method.
44 *
45 * @param {!goog.storage.mechanism.IterableMechanism} mechanism The underlying
46 * storage mechanism.
47 * @param {number} maxItems Maximum number of items in storage.
48 * @constructor
49 * @struct
50 * @extends {goog.storage.CollectableStorage}
51 * @final
52 */
53storage.BoundedCollectableStorage = function(mechanism, maxItems) {
54 storage.BoundedCollectableStorage.base(this, 'constructor', mechanism);
55
56 /**
57 * A maximum number of items that should be stored.
58 * @private {number}
59 */
60 this.maxItems_ = maxItems;
61};
62goog.inherits(storage.BoundedCollectableStorage,
63 goog.storage.CollectableStorage);

66/**
67 * An item key used to store a list of keys.
68 * @const
69 * @private
70 */
71storage.BoundedCollectableStorage.KEY_LIST_KEY_ = 'bounded-collectable-storage';

74/**
75 * Recreates a list of keys in order of creation.
76 *
77 * @return {!Array<string>} a list of unexpired keys.
78 * @private
79 */
80storage.BoundedCollectableStorage.prototype.rebuildIndex_ = function() {
81 var keys = [];
82 goog.iter.forEach(/** @type {goog.storage.mechanism.IterableMechanism} */ (
83 this.mechanism).__iterator__(true), function(key) {
84 if (storage.BoundedCollectableStorage.KEY_LIST_KEY_ == key) {
85 return;
86 }
87
88 var wrapper;
89 /** @preserveTry */
90 try {
91 wrapper = this.getWrapper(key, true);
92 } catch (ex) {
93 if (ex == goog.storage.ErrorCode.INVALID_VALUE) {
94 // Skip over bad wrappers and continue.
95 return;
96 }
97 // Unknown error, escalate.
98 throw ex;
99 }
100 goog.asserts.assert(wrapper);
101
102 var creationTime =
103 goog.storage.ExpiringStorage.getCreationTime(wrapper);
104 keys.push({key: key, created: creationTime});
105 }, this);
106
107 goog.array.sort(keys, function(a, b) {
108 return a.created - b.created;
109 });
110
111 return goog.array.map(keys, function(v) {
112 return v.key;
113 });
114};

117/**
118 * Gets key list from a local storage. If an item does not exist,
119 * may recreate it.
120 *
121 * @param {boolean} rebuild Whether to rebuild a index if no index item exists.
122 * @return {!Array<string>} a list of keys if index exist, otherwise undefined.
123 * @private
124 */
125storage.BoundedCollectableStorage.prototype.getKeys_ = function(rebuild) {
126 var keys = storage.BoundedCollectableStorage.superClass_.get.call(this,
127 storage.BoundedCollectableStorage.KEY_LIST_KEY_) || null;
128 if (!keys || !goog.isArray(keys)) {
129 if (rebuild) {
130 keys = this.rebuildIndex_();
131 } else {
132 keys = [];
133 }
134 }
135 return /** @type {!Array<string>} */ (keys);
136};

151/**
152 * Remove subsequence from a sequence.
153 *
154 * @param {!Array<string>} keys is a sequence.
155 * @param {!Array<string>} keysToRemove subsequence of keys, the order must
156 * be kept.
157 * @return {!Array<string>} a keys sequence after removing keysToRemove.
158 * @private
159 */
160storage.BoundedCollectableStorage.removeSubsequence_ =
161 function(keys, keysToRemove) {
162 if (keysToRemove.length == 0) {
163 return goog.array.clone(keys);
164 }
165 var keysToKeep = [];
166 var keysIdx = 0;
167 var keysToRemoveIdx = 0;
168
169 while (keysToRemoveIdx < keysToRemove.length && keysIdx < keys.length) {
170 var key = keysToRemove[keysToRemoveIdx];
171 while (keysIdx < keys.length && keys[keysIdx] != key) {
172 keysToKeep.push(keys[keysIdx]);
173 ++keysIdx;
174 }
175 ++keysToRemoveIdx;
176 }
177
178 goog.asserts.assert(keysToRemoveIdx == keysToRemove.length);
179 goog.asserts.assert(keysIdx < keys.length);
180 return goog.array.concat(keysToKeep, goog.array.slice(keys, keysIdx + 1));
181};

184/**
185 * Keeps the number of items in storage under maxItems. Removes elements in the
186 * order of creation.
187 *
188 * @param {!Array<string>} keys a list of keys in order of creation.
189 * @param {number} maxSize a number of items to keep.
190 * @return {!Array<string>} keys left after removing oversize data.
191 * @private
192 */
193storage.BoundedCollectableStorage.prototype.collectOversize_ =
194 function(keys, maxSize) {
195 if (keys.length <= maxSize) {
196 return goog.array.clone(keys);
197 }
198 var keysToRemove = goog.array.slice(keys, 0, keys.length - maxSize);
199 goog.array.forEach(keysToRemove, function(key) {
200 storage.BoundedCollectableStorage.superClass_.remove.call(this, key);
201 }, this);
202 return storage.BoundedCollectableStorage.removeSubsequence_(
203 keys, keysToRemove);
204};

207/**
208 * Cleans up the storage by removing expired keys.
209 *
210 * @param {boolean=} opt_strict Also remove invalid keys.
211 * @override
212 */
213storage.BoundedCollectableStorage.prototype.collect =
214 function(opt_strict) {
215 var keys = this.getKeys_(true);
216 var keysToRemove = this.collectInternal(keys, opt_strict);
217 keys = storage.BoundedCollectableStorage.removeSubsequence_(
218 keys, keysToRemove);
219 this.setKeys_(keys);
220};

223/**
224 * Ensures that we keep only maxItems number of items in a local storage.
225 * @param {boolean=} opt_skipExpired skip removing expired items first.
226 * @param {boolean=} opt_strict Also remove invalid keys.
227 */
228storage.BoundedCollectableStorage.prototype.collectOversize =
229 function(opt_skipExpired, opt_strict) {
230 var keys = this.getKeys_(true);
231 if (!opt_skipExpired) {
232 var keysToRemove = this.collectInternal(keys, opt_strict);
233 keys = storage.BoundedCollectableStorage.removeSubsequence_(
234 keys, keysToRemove);
235 }
236 keys = this.collectOversize_(keys, this.maxItems_);
237 this.setKeys_(keys);
238};

241/**
242 * Set an item in the storage.
243 *
244 * @param {string} key The key to set.
245 * @param {*} value The value to serialize to a string and save.
246 * @param {number=} opt_expiration The number of miliseconds since epoch
247 * (as in goog.now()) when the value is to expire. If the expiration
248 * time is not provided, the value will persist as long as possible.
249 * @override
250 */
251storage.BoundedCollectableStorage.prototype.set =
252 function(key, value, opt_expiration) {
253 storage.BoundedCollectableStorage.base(
254 this, 'set', key, value, opt_expiration);
255 var keys = this.getKeys_(true);
256 goog.array.remove(keys, key);
257
258 if (goog.isDef(value)) {
259 keys.push(key);
260 if (keys.length >= this.maxItems_) {
261 var keysToRemove = this.collectInternal(keys);
262 keys = storage.BoundedCollectableStorage.removeSubsequence_(
263 keys, keysToRemove);
264 keys = this.collectOversize_(keys, this.maxItems_);
265 }
266 }
267 this.setKeys_(keys);
268};

271/**
272 * Remove an item from the data storage.
273 *
274 * @param {string} key The key to remove.
275 * @override
276 */
277storage.BoundedCollectableStorage.prototype.remove = function(key) {
278 storage.BoundedCollectableStorage.base(this, 'remove', key);
279
280 var keys = this.getKeys_(false);
281 if (goog.isDef(keys)) {
282 goog.array.remove(keys, key);
283 this.setKeys_(keys);
284 }
