538/**
539 * Disconnects and closes the channel.
540 */
541WebChannelBase.prototype.disconnect = function() {
542 this.channelDebug_.debug('disconnect()');
543
544 this.cancelRequests_();
545
546 if (this.state_ == WebChannelBase.State.OPENED) {
547 var rid = this.nextRid_++;
548 var uri = this.forwardChannelUri_.clone();
549 uri.setParameterValue('SID', this.sid_);
550 uri.setParameterValue('RID', rid);
551 uri.setParameterValue('TYPE', 'terminate');
552
553 // Add the reconnect parameters.
554 this.addAdditionalParams_(uri);
555
556 var request = ChannelRequest.createChannelRequest(
557 this, this.channelDebug_, this.sid_, rid);
558 request.sendUsingImgTag(uri);
559 }
560
561 this.onClose_();
562};

761/**
762 * Sends a request to the server. The format of the request is a Map data
763 * structure of key/value pairs. These maps are then encoded in a format
764 * suitable for the wire and then reconstituted as a Map data structure that
765 * the server can process.
766 * @param {!Object|!goog.structs.Map} map The map to send.
767 * @param {!Object=} opt_context The context associated with the map.
768 */
769WebChannelBase.prototype.sendMap = function(map, opt_context) {
770 goog.asserts.assert(this.state_ != WebChannelBase.State.CLOSED,
771 'Invalid operation: sending map when state is closed');
772
773 // We can only send 1000 maps per POST, but typically we should never have
774 // that much to send, so warn if we exceed that (we still send all the maps).
775 if (this.outgoingMaps_.length == WebChannelBase.MAX_MAPS_PER_REQUEST_) {
776 // severe() is temporary so that we get these uploaded and can figure out
777 // what's causing them. Afterwards can change to warning().
778 this.channelDebug_.severe(
779 'Already have ' + WebChannelBase.MAX_MAPS_PER_REQUEST_ +
780 ' queued maps upon queueing ' + goog.json.serialize(map));
781 }
782
783 this.outgoingMaps_.push(
784 new Wire.QueuedMap(this.nextMapId_++, map, opt_context));
785 if (this.state_ == WebChannelBase.State.OPENING ||
786 this.state_ == WebChannelBase.State.OPENED) {
787 this.ensureForwardChannel_();
788 }

792/**
793 * When set to true, this changes the behavior of the forward channel so it
794 * will not retry requests; it will fail after one network failure, and if
795 * there was already one network failure, the request will fail immediately.
796 * @param {boolean} failFast Whether or not to fail fast.
797 */
798WebChannelBase.prototype.setFailFast = function(failFast) {
799 this.failFast_ = failFast;
800 this.channelDebug_.info('setFailFast: ' + failFast);
801 if ((this.forwardChannelRequestPool_.hasPendingRequest() ||
802 this.forwardChannelTimerId_) &&
803 this.forwardChannelRetryCount_ > this.getForwardChannelMaxRetries()) {
804 this.channelDebug_.info(
805 'Retry count ' + this.forwardChannelRetryCount_ +
806 ' > new maxRetries ' + this.getForwardChannelMaxRetries() +
807 '. Fail immediately!');
808
809 if (!this.forwardChannelRequestPool_.forceComplete(
810 goog.bind(this.onRequestComplete, this))) {
811 // i.e., this.forwardChannelTimerId_
812 goog.global.clearTimeout(this.forwardChannelTimerId_);

981/**

983 * @param {ChannelRequest=} opt_retryRequest A failed request to retry.
984 * @private
985 */
986WebChannelBase.prototype.startForwardChannel_ = function(
987 opt_retryRequest) {
988 this.channelDebug_.debug('startForwardChannel_');
989 if (!this.okToMakeRequest_()) {
990 return; // channel is cancelled
991 } else if (this.state_ == WebChannelBase.State.INIT) {
992 if (opt_retryRequest) {
993 this.channelDebug_.severe('Not supposed to retry the open');
994 return;
995 }
996 this.open_();
997 this.state_ = WebChannelBase.State.OPENING;
998 } else if (this.state_ == WebChannelBase.State.OPENED) {
999 if (opt_retryRequest) {
1000 this.makeForwardChannelRequest_(opt_retryRequest);
1001 return;
1002 }
1003
1004 if (this.outgoingMaps_.length == 0) {
1005 this.channelDebug_.debug('startForwardChannel_ returned: ' +
1006 'nothing to send');
1007 // no need to start a new forward channel request
1008 return;
1009 }
1010
1011 if (this.forwardChannelRequestPool_.isFull()) {
1012 // Should be impossible to be called in this state.
1013 this.channelDebug_.severe('startForwardChannel_ returned: ' +
1014 'connection already in progress');
1015 return;
1016 }
1017
1018 this.makeForwardChannelRequest_();
1019 this.channelDebug_.debug('startForwardChannel_ finished, sent request');
1020 }

1026 * @private

1031

1051/**
1052 * Makes a forward channel request using XMLHTTP.
1053 * @param {!ChannelRequest=} opt_retryRequest A failed request to retry.
1054 * @private
1055 */
1056WebChannelBase.prototype.makeForwardChannelRequest_ =
1057 function(opt_retryRequest) {
1058 var rid;
1059 var requestText;
1060 if (opt_retryRequest) {
1061 this.requeuePendingMaps_();
1062 rid = this.nextRid_ - 1; // Must use last RID
1063 requestText = this.dequeueOutgoingMaps_();
1064 } else {
1065 rid = this.nextRid_++;
1066 requestText = this.dequeueOutgoingMaps_();
1067 }
1068
1069 var uri = this.forwardChannelUri_.clone();
1070 uri.setParameterValue('SID', this.sid_);
1071 uri.setParameterValue('RID', rid);
1072 uri.setParameterValue('AID', this.lastArrayId_);
1073 // Add the additional reconnect parameters.
1074 this.addAdditionalParams_(uri);
1075
1076 var request = ChannelRequest.createChannelRequest(this, this.channelDebug_,
1077 this.sid_, rid, this.forwardChannelRetryCount_ + 1);
1078 request.setExtraHeaders(this.extraHeaders_);
1079
1080 // Randomize from 50%-100% of the forward channel timeout to avoid
1081 // a big hit if servers happen to die at once.
1082 request.setTimeout(
1083 Math.round(this.forwardChannelRequestTimeoutMs_ * 0.50) +
1084 Math.round(this.forwardChannelRequestTimeoutMs_ * 0.50 * Math.random()));
1085 this.forwardChannelRequestPool_.addRequest(request);
1086 request.xmlHttpPost(uri, requestText, true);
1087};

1320/**
1321 * Handles a POST response from the server.
1322 * @param {Array<number>} responseValues The key value pairs in
1323 * the POST response.
1324 * @param {!ChannelRequest} forwardReq The forward channel request that
1325 * triggers this function call.
1326 * @private
1327 */
1328WebChannelBase.prototype.handlePostResponse_ = function(
1329 responseValues, forwardReq) {
1330 // The first response value is set to 0 if server is missing backchannel.
1331 if (responseValues[0] == 0) {
1332 this.handleBackchannelMissing_(forwardReq);
1333 return;
1334 }
1335 this.lastPostResponseArrayId_ = responseValues[1];
1336 var outstandingArrays = this.lastPostResponseArrayId_ - this.lastArrayId_;
1337 if (0 < outstandingArrays) {
1338 var numOutstandingBackchannelBytes = responseValues[2];
1339 this.channelDebug_.debug(numOutstandingBackchannelBytes + ' bytes (in ' +
1340 outstandingArrays + ' arrays) are outstanding on the BackChannel');
1341 if (!this.shouldRetryBackChannel_(numOutstandingBackchannelBytes)) {
1342 return;
1343 }
1344 if (!this.deadBackChannelTimerId_) {
1345 // We expect to receive data within 2 RTTs or we retry the backchannel.
1346 this.deadBackChannelTimerId_ = requestStats.setTimeout(
1347 goog.bind(this.onBackChannelDead_, this),
1348 2 * WebChannelBase.RTT_ESTIMATE);
1349 }
1350 }

1354/**
1355 * Handles a POST response from the server telling us that it has detected that
1356 * we have no hanging GET connection.
1357 * @param {!ChannelRequest} forwardReq The forward channel request that
1358 * triggers this function call.
1359 * @private
1360 */
1361WebChannelBase.prototype.handleBackchannelMissing_ = function(forwardReq) {
1362 // As long as the back channel was started before the POST was sent,
1363 // we should retry the backchannel. We give a slight buffer of RTT_ESTIMATE
1364 // so as not to excessively retry the backchannel
1365 this.channelDebug_.debug('Server claims our backchannel is missing.');
1366 if (this.backChannelTimerId_) {
1367 this.channelDebug_.debug('But we are currently starting the request.');
1368 return;
1369 } else if (!this.backChannelRequest_) {
1370 this.channelDebug_.warning(
1371 'We do not have a BackChannel established');
1372 } else if (this.backChannelRequest_.getRequestStartTime() +
1373 WebChannelBase.RTT_ESTIMATE < forwardReq.getRequestStartTime()) {
1374 this.clearDeadBackchannelTimer_();
1375 this.backChannelRequest_.cancel();
1376 this.backChannelRequest_ = null;
1377 } else {
1378 return;
1379 }
1380 this.maybeRetryBackChannel_();
1381 requestStats.notifyStatEvent(requestStats.Stat.BACKCHANNEL_MISSING);
1382};

1469/**
1470 * @override
1471 */
1472WebChannelBase.prototype.onRequestComplete = function(request) {
1473 this.channelDebug_.debug('Request complete');
1474 var type;
1475 if (this.backChannelRequest_ == request) {
1476 this.clearDeadBackchannelTimer_();
1477 this.backChannelRequest_ = null;
1478 type = WebChannelBase.ChannelType_.BACK_CHANNEL;
1479 } else if (this.forwardChannelRequestPool_.hasRequest(request)) {
1480 this.forwardChannelRequestPool_.removeRequest(request);
1481 type = WebChannelBase.ChannelType_.FORWARD_CHANNEL;
1482 } else {
1483 // return if it was an old request from a previous session
1484 return;
1485 }
1486
1487 this.lastStatusCode_ = request.getLastStatusCode();
1488
1489 if (this.state_ == WebChannelBase.State.CLOSED) {
1490 return;
1491 }
1492
1493 if (request.getSuccess()) {
1494 // Yay!
1495 if (type == WebChannelBase.ChannelType_.FORWARD_CHANNEL) {
1496 var size = request.getPostData() ? request.getPostData().length : 0;
1497 requestStats.notifyTimingEvent(size,
1498 goog.now() - request.getRequestStartTime(),
1499 this.forwardChannelRetryCount_);
1500 this.ensureForwardChannel_();
1501 this.onSuccess_();
1502 this.pendingMaps_.length = 0;
1503 } else { // i.e., back-channel
1504 this.ensureBackChannel_();
1505 }
1506 return;
1507 }
1508 // Else unsuccessful. Fall through.
1509
1510 var lastError = request.getLastError();
1511 if (!WebChannelBase.isFatalError_(lastError, this.lastStatusCode_)) {
1512 // Maybe retry.
1513 this.channelDebug_.debug('Maybe retrying, last error: ' +
1514 ChannelRequest.errorStringFromCode(lastError, this.lastStatusCode_));
1515 if (type == WebChannelBase.ChannelType_.FORWARD_CHANNEL) {
1516 if (this.maybeRetryForwardChannel_(request)) {
1517 return;
1518 }
1519 }
1520 if (type == WebChannelBase.ChannelType_.BACK_CHANNEL) {
1521 if (this.maybeRetryBackChannel_()) {
1522 return;
1523 }
1524 }
1525 // Else exceeded max retries. Fall through.
1526 this.channelDebug_.debug('Exceeded max number of retries');
1527 } else {
1528 // Else fatal error. Fall through and mark the pending maps as failed.
1529 this.channelDebug_.debug('Not retrying due to error type');
1530 }
1531
1532
1533 // Can't save this session. :(
1534 this.channelDebug_.debug('Error: HTTP request failed');
1535 switch (lastError) {
1536 case ChannelRequest.Error.NO_DATA:
1537 this.signalError_(WebChannelBase.Error.NO_DATA);
1538 break;
1539 case ChannelRequest.Error.BAD_DATA:
1540 this.signalError_(WebChannelBase.Error.BAD_DATA);
1541 break;

1582/**
1583 * Processes the data returned by the server.
1584 * @param {!Array<!Array<?>>} respArray The response array returned
1585 * by the server.
1586 * @private
1587 */
1588WebChannelBase.prototype.onInput_ = function(respArray) {
1589 var batch = this.handler_ && this.handler_.channelHandleMultipleArrays ?
1590 [] : null;
1591 for (var i = 0; i < respArray.length; i++) {
1592 var nextArray = respArray[i];
1593 this.lastArrayId_ = nextArray[0];
1594 nextArray = nextArray[1];
1595 if (this.state_ == WebChannelBase.State.OPENING) {
1596 if (nextArray[0] == 'c') {
1597 this.sid_ = nextArray[1];
1598 this.hostPrefix_ = this.correctHostPrefix(nextArray[2]);
1599 var negotiatedVersion = nextArray[3];
1600 if (goog.isDefAndNotNull(negotiatedVersion)) {
1601 this.channelVersion_ = negotiatedVersion;
1602 }
1603 this.state_ = WebChannelBase.State.OPENED;
1604 if (this.handler_) {
1605 this.handler_.channelOpened(this);
1606 }
1607 this.backChannelUri_ = this.getBackChannelUri(
1608 this.hostPrefix_, /** @type {string} */ (this.path_));
1609 // Open connection to receive data
1610 this.ensureBackChannel_();
1611 } else if (nextArray[0] == 'stop') {
1612 this.signalError_(WebChannelBase.Error.STOP);
1613 }
1614 } else if (this.state_ == WebChannelBase.State.OPENED) {
1615 if (nextArray[0] == 'stop') {
1616 if (batch && !goog.array.isEmpty(batch)) {
1617 this.handler_.channelHandleMultipleArrays(this, batch);
1618 batch.length = 0;
1619 }
1620 this.signalError_(WebChannelBase.Error.STOP);
1621 } else if (nextArray[0] == 'noop') {
1622 // ignore - noop to keep connection happy
1623 } else {
1624 if (batch) {
1625 batch.push(nextArray);
1626 } else if (this.handler_) {
1627 this.handler_.channelHandleArray(this, nextArray);
1628 }
1629 }
1630 // We have received useful data on the back-channel, so clear its retry
1631 // count. We do this because back-channels by design do not complete
1632 // quickly, so on a flaky connection we could have many fail to complete
1633 // fully but still deliver a lot of data before they fail. We don't want
1634 // to count such failures towards the retry limit, because we don't want
1635 // to give up on a session if we can still receive data.
1636 this.backChannelRetryCount_ = 0;
1637 }
1638 }
1639 if (batch && !goog.array.isEmpty(batch)) {
1640 this.handler_.channelHandleMultipleArrays(this, batch);
1641 }
