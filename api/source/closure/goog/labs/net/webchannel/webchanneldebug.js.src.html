173/**
174 * Removes potentially private data from a response so that we don't
175 * accidentally save private and personal data to the server logs.
176 * @param {?string} responseText A JSON response to clean.
177 * @return {?string} The cleaned response.
178 * @private
179 */
180WebChannelDebug.prototype.redactResponse_ = function(responseText) {
181 if (!responseText) {
182 return null;
183 }
184 /** @preserveTry */
185 try {
186 var responseArray = goog.json.unsafeParse(responseText);
187 if (responseArray) {
188 for (var i = 0; i < responseArray.length; i++) {
189 if (goog.isArray(responseArray[i])) {
190 this.maybeRedactArray_(responseArray[i]);
191 }
192 }
193 }
194
195 return goog.json.serialize(responseArray);
196 } catch (e) {
197 this.debug('Exception parsing expected JS array - probably was not JS');
198 return responseText;
199 }
200};
201
202
203/**
204 * Removes data from a response array that may be sensitive.
205 * @param {!Array<?>} array The array to clean.
206 * @private
207 */
208WebChannelDebug.prototype.maybeRedactArray_ = function(array) {
209 if (array.length < 2) {
210 return;
211 }
212 var dataPart = array[1];
213 if (!goog.isArray(dataPart)) {
214 return;
215 }
216 if (dataPart.length < 1) {
217 return;
218 }
219
220 var type = dataPart[0];
221 if (type != 'noop' && type != 'stop') {
222 // redact all fields in the array
223 for (var i = 1; i < dataPart.length; i++) {
224 dataPart[i] = '';
225 }
226 }
227};

230/**
231 * Removes potentially private data from a request POST body so that we don't
232 * accidentally save private and personal data to the server logs.
233 * @param {?string} data The data string to clean.
234 * @return {?string} The data string with sensitive data replaced by 'redacted'.
235 * @private
236 */
237WebChannelDebug.prototype.maybeRedactPostData_ = function(data) {
238 if (!data) {
239 return null;
240 }
241 var out = '';
242 var params = data.split('&');
243 for (var i = 0; i < params.length; i++) {
244 var param = params[i];
245 var keyValue = param.split('=');
246 if (keyValue.length > 1) {
247 var key = keyValue[0];
248 var value = keyValue[1];
249
250 var keyParts = key.split('_');
251 if (keyParts.length >= 2 && keyParts[1] == 'type') {
252 out += key + '=' + value + '&';
253 } else {
254 out += key + '=' + 'redacted' + '&';
255 }
256 }
257 }
258 return out;
259};
