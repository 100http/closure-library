151/**
152 * Iterates over all the unknown fields in the message.
153 *
154 * @param {function(number, *)} callback A callback method
155 * which gets invoked for each unknown field.
156 * @param {Object=} opt_scope The scope under which to execute the callback.

160 var scope = opt_scope || this;

163 if (!this.fields_[keyNum]) {
164 callback.call(scope, keyNum, this.values_[key]);

320/**
321 * Compares this message with another one ignoring the unknown fields.
322 * @param {*} other The other message.
323 * @return {boolean} Whether they are equal. Returns false if the {@code other}
324 * argument is a different type of message or not a message.
325 */
326goog.proto2.Message.prototype.equals = function(other) {
327 if (!other || this.constructor != other.constructor) {
328 return false;
329 }
330
331 var fields = this.getDescriptor().getFields();
332 for (var i = 0; i < fields.length; i++) {
333 var field = fields[i];
334 var tag = field.getTag();
335 if (this.has$Value(tag) != other.has$Value(tag)) {
336 return false;
337 }
338
339 if (this.has$Value(tag)) {
340 var isComposite = field.isCompositeType();
341
342 var fieldsEqual = function(value1, value2) {
343 return isComposite ? value1.equals(value2) : value1 == value2;
344 };
345
346 var thisValue = this.getValueForTag_(tag);
347 var otherValue = other.getValueForTag_(tag);
348
349 if (field.isRepeated()) {
350 // In this case thisValue and otherValue are arrays.
351 if (thisValue.length != otherValue.length) {
352 return false;
353 }
354 for (var j = 0; j < thisValue.length; j++) {
355 if (!fieldsEqual(thisValue[j], otherValue[j])) {
356 return false;
357 }
358 }
359 } else if (!fieldsEqual(thisValue, otherValue)) {
360 return false;
361 }
362 }
363 }
364
365 return true;
366};

388/**
389 * Merges the given message into this message.
390 *
391 * Singular fields will be overwritten, except for embedded messages which will
392 * be merged. Repeated fields will be concatenated.
393 * @param {!goog.proto2.Message} message The source message.
394 */
395goog.proto2.Message.prototype.mergeFrom = function(message) {
396 goog.asserts.assert(this.constructor == message.constructor,
397 'The source message must have the same type.');
398 var fields = this.getDescriptor().getFields();
399
400 for (var i = 0; i < fields.length; i++) {
401 var field = fields[i];
402 var tag = field.getTag();
403 if (message.has$Value(tag)) {
404 if (this.deserializedFields_) {
405 delete this.deserializedFields_[field.getTag()];
406 }
407
408 var isComposite = field.isCompositeType();
409 if (field.isRepeated()) {
410 var values = message.array$Values(tag);
411 for (var j = 0; j < values.length; j++) {
412 this.add$Value(tag, isComposite ? values[j].clone() : values[j]);
413 }
414 } else {
415 var value = message.getValueForTag_(tag);
416 if (isComposite) {
417 var child = this.getValueForTag_(tag);
418 if (child) {
419 child.mergeFrom(value);
420 } else {
421 this.set$Value(tag, value.clone());
422 }
423 } else {
424 this.set$Value(tag, value);
425 }
426 }
427 }
428 }
429};

444/**
445 * Fills in the protocol buffer with default values. Any fields that are
446 * already set will not be overridden.
447 * @param {boolean} simpleFieldsToo If true, all fields will be initialized;
448 * if false, only the nested messages and groups.
449 */
450goog.proto2.Message.prototype.initDefaults = function(simpleFieldsToo) {
451 var fields = this.getDescriptor().getFields();
452 for (var i = 0; i < fields.length; i++) {
453 var field = fields[i];
454 var tag = field.getTag();
455 var isComposite = field.isCompositeType();
456
457 // Initialize missing fields.
458 if (!this.has$Value(tag) && !field.isRepeated()) {
459 if (isComposite) {
460 this.values_[tag] = new /** @type {Function} */ (field.getNativeType());
461 } else if (simpleFieldsToo) {
462 this.values_[tag] = field.getDefaultValue();
463 }
464 }
465
466 // Fill in the existing composite fields recursively.
467 if (isComposite) {
468 if (field.isRepeated()) {
469 var values = this.array$Values(tag);
470 for (var j = 0; j < values.length; j++) {
471 values[j].initDefaults(simpleFieldsToo);
472 }
473 } else {
474 this.get$Value(tag).initDefaults(simpleFieldsToo);
475 }
476 }
477 }

496/**
497 * Returns the value for the given tag number. If a lazy deserializer is
498 * instantiated, lazily deserializes the field if required before returning the
499 * value.
500 *
501 * @param {number} tag The tag number.
502 * @return {*} The corresponding value, if any.
503 * @private
504 */
505goog.proto2.Message.prototype.getValueForTag_ = function(tag) {
506 // Retrieve the current value, which may still be serialized.
507 var value = this.values_[tag];
508 if (!goog.isDefAndNotNull(value)) {
509 return null;
510 }
511
512 // If we have a lazy deserializer, then ensure that the field is
513 // properly deserialized.
514 if (this.lazyDeserializer_) {
515 // If the tag is not deserialized, then we must do so now. Deserialize
516 // the field's value via the deserializer.
517 if (!(tag in this.deserializedFields_)) {
518 var deserializedValue = this.lazyDeserializer_.deserializeField(
519 this, this.fields_[tag], value);
520 this.deserializedFields_[tag] = deserializedValue;
521 return deserializedValue;
522 }
523
524 return this.deserializedFields_[tag];
525 }
526
527 // Otherwise, just return the value.
528 return value;
529};

701/**
702 * Creates the metadata descriptor representing the definition of this message.
703 *
704 * @param {function(new:goog.proto2.Message)} messageType Constructor for the

706 * @param {!Object} metadataObj The object containing the metadata.

709goog.proto2.Message.createDescriptor = function(messageType, metadataObj) {
710 var fields = [];
711 var descriptorInfo = metadataObj[0];
712
713 for (var key in metadataObj) {
714 if (key != 0) {
715 // Create the field descriptor.
716 fields.push(
717 new goog.proto2.FieldDescriptor(messageType, key, metadataObj[key]));
718 }
719 }
720
721 return new goog.proto2.Descriptor(messageType, descriptorInfo, fields);
