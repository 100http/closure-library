38/**
39 * TextFormatSerializer, a serializer which turns Messages into the human
40 * readable text format.
41 * @param {boolean=} opt_ignoreMissingFields If true, then fields that cannot be
42 * found on the proto when parsing the text format will be ignored.
43 * @param {boolean=} opt_useEnumValues If true, serialization code for enums
44 * will use enum integer values instead of human-readable symbolic names.
45 * @constructor
46 * @extends {goog.proto2.Serializer}
47 * @final
48 */
49goog.proto2.TextFormatSerializer = function(
50 opt_ignoreMissingFields, opt_useEnumValues) {
51 /**
52 * Whether to ignore fields not defined on the proto when parsing the text
53 * format.
54 * @type {boolean}
55 * @private
56 */
57 this.ignoreMissingFields_ = !!opt_ignoreMissingFields;
58
59 /**
60 * Whether to use integer enum values during enum serialization.
61 * If false, symbolic names will be used.
62 * @type {boolean}
63 * @private
64 */
65 this.useEnumValues_ = !!opt_useEnumValues;
66};
67goog.inherits(goog.proto2.TextFormatSerializer, goog.proto2.Serializer);
68
69
70/**
71 * Deserializes a message from text format and places the data in the message.
72 * @param {goog.proto2.Message} message The message in which to
73 * place the information.
74 * @param {*} data The text format data.
75 * @return {?string} The parse error or null on success.
76 * @override
77 */
78goog.proto2.TextFormatSerializer.prototype.deserializeTo =
79 function(message, data) {
80 var descriptor = message.getDescriptor();
81 var textData = data.toString();
82 var parser = new goog.proto2.TextFormatSerializer.Parser();
83 if (!parser.parse(message, textData, this.ignoreMissingFields_)) {
84 return parser.getError();
85 }
86
87 return null;
88};

91/**
92 * Serializes a message to a string.
93 * @param {goog.proto2.Message} message The message to be serialized.
94 * @return {string} The serialized form of the message.
95 * @override
96 */
97goog.proto2.TextFormatSerializer.prototype.serialize = function(message) {
98 var printer = new goog.proto2.TextFormatSerializer.Printer_();
99 this.serializeMessage_(message, printer);
100 return printer.toString();
101};

103
104/**
105 * Serializes the message and prints the text form into the given printer.
106 * @param {goog.proto2.Message} message The message to serialize.
107 * @param {goog.proto2.TextFormatSerializer.Printer_} printer The printer to
108 * which the text format will be printed.
109 * @private
110 */
111goog.proto2.TextFormatSerializer.prototype.serializeMessage_ =
112 function(message, printer) {
113 var descriptor = message.getDescriptor();
114 var fields = descriptor.getFields();
115
116 // Add the defined fields, recursively.
117 goog.array.forEach(fields, function(field) {
118 this.printField_(message, field, printer);
119 }, this);
120
121 // Add the unknown fields, if any.
122 message.forEachUnknown(function(tag, value) {
123 this.serializeUnknown_(tag, value, printer);
124 }, this);
125};
126
127
128/**
129 * Serializes an unknown field. When parsed from the JsPb object format, this
130 * manifests as either a primitive type, an array, or a raw object with integer
131 * keys. There is no descriptor available to interpret the types of nested
132 * messages.
133 * @param {number} tag The tag for the field. Since it's unknown, this is a
134 * number rather than a string.
135 * @param {*} value The value of the field.
136 * @param {!goog.proto2.TextFormatSerializer.Printer_} printer The printer to
137 * which the text format will be serialized.
138 * @private
139 */
140goog.proto2.TextFormatSerializer.prototype.serializeUnknown_ =
141 function(tag, value, printer) {
142 if (!goog.isDefAndNotNull(value)) {
143 return;
144 }
145
146 if (goog.isArray(value)) {
147 goog.array.forEach(value, function(val) {
148 this.serializeUnknown_(tag, val, printer);
149 }, this);
150 return;
151 }
152
153 if (goog.isObject(value)) {
154 printer.append(tag);
155 printer.append(' {');
156 printer.appendLine();
157 printer.indent();
158 if (value instanceof goog.proto2.Message) {
159 // Note(user): This conditional is here to make the
160 // testSerializationOfUnknown unit test pass, but in practice we should
161 // never have a Message for an "unknown" field.
162 this.serializeMessage_(value, printer);
163 } else {
164 // For an unknown message, fields are keyed by positive integers. We
165 // don't have a 'length' property to use for enumeration, so go through
166 // all properties and ignore the ones that aren't legal keys.
167 for (var key in value) {
168 var keyAsNumber = goog.string.parseInt(key);
169 goog.asserts.assert(goog.math.isInt(keyAsNumber));
170 this.serializeUnknown_(keyAsNumber, value[key], printer);
171 }
172 }
173 printer.dedent();
174 printer.append('}');
175 printer.appendLine();
176 return;
177 }
178
179 if (goog.isString(value)) {
180 value = goog.string.quote(value);
181 }
182 printer.append(tag);
183 printer.append(': ');
184 printer.append(value.toString());
185 printer.appendLine();
186};

189/**
190 * Prints the serialized value for the given field to the printer.
191 * @param {*} value The field's value.
192 * @param {goog.proto2.FieldDescriptor} field The field whose value is being
193 * printed.
194 * @param {goog.proto2.TextFormatSerializer.Printer_} printer The printer to
195 * which the value will be printed.
196 * @private
197 */
198goog.proto2.TextFormatSerializer.prototype.printFieldValue_ =
199 function(value, field, printer) {
200 switch (field.getFieldType()) {
201 case goog.proto2.FieldDescriptor.FieldType.DOUBLE:
202 case goog.proto2.FieldDescriptor.FieldType.FLOAT:
203 case goog.proto2.FieldDescriptor.FieldType.INT64:
204 case goog.proto2.FieldDescriptor.FieldType.UINT64:
205 case goog.proto2.FieldDescriptor.FieldType.INT32:
206 case goog.proto2.FieldDescriptor.FieldType.UINT32:
207 case goog.proto2.FieldDescriptor.FieldType.FIXED64:
208 case goog.proto2.FieldDescriptor.FieldType.FIXED32:
209 case goog.proto2.FieldDescriptor.FieldType.BOOL:
210 case goog.proto2.FieldDescriptor.FieldType.SFIXED32:
211 case goog.proto2.FieldDescriptor.FieldType.SFIXED64:
212 case goog.proto2.FieldDescriptor.FieldType.SINT32:
213 case goog.proto2.FieldDescriptor.FieldType.SINT64:
214 printer.append(value);
215 break;
216
217 case goog.proto2.FieldDescriptor.FieldType.BYTES:
218 case goog.proto2.FieldDescriptor.FieldType.STRING:
219 value = goog.string.quote(value.toString());
220 printer.append(value);
221 break;
222
223 case goog.proto2.FieldDescriptor.FieldType.ENUM:
224 if (!this.useEnumValues_) {
225 // Search the enum type for a matching key.
226 var found = false;
227 goog.object.forEach(field.getNativeType(), function(eValue, key) {
228 if (eValue == value) {
229 printer.append(key);
230 found = true;
231 }
232 });
233 }
234
235 if (!found || this.useEnumValues_) {
236 // Otherwise, just print the numeric value.
237 printer.append(value.toString());
238 }
239 break;
240
241 case goog.proto2.FieldDescriptor.FieldType.GROUP:
242 case goog.proto2.FieldDescriptor.FieldType.MESSAGE:
243 this.serializeMessage_(
244 /** @type {goog.proto2.Message} */ (value), printer);
245 break;
246 }
247};
248
249
250/**
251 * Prints the serialized field to the printer.
252 * @param {goog.proto2.Message} message The parent message.
253 * @param {goog.proto2.FieldDescriptor} field The field to print.
254 * @param {goog.proto2.TextFormatSerializer.Printer_} printer The printer to
255 * which the field will be printed.
256 * @private
257 */
258goog.proto2.TextFormatSerializer.prototype.printField_ =
259 function(message, field, printer) {
260 // Skip fields not present.
261 if (!message.has(field)) {
262 return;
263 }
264
265 var count = message.countOf(field);
266 for (var i = 0; i < count; ++i) {
267 // Field name.
268 printer.append(field.getName());
269
270 // Field delimiter.
271 if (field.getFieldType() == goog.proto2.FieldDescriptor.FieldType.MESSAGE ||
272 field.getFieldType() == goog.proto2.FieldDescriptor.FieldType.GROUP) {
273 printer.append(' {');
274 printer.appendLine();
275 printer.indent();
276 } else {
277 printer.append(': ');
278 }
279
280 // Write the field value.
281 this.printFieldValue_(message.get(field, i), field, printer);
282
283 // Close the field.
284 if (field.getFieldType() == goog.proto2.FieldDescriptor.FieldType.MESSAGE ||
285 field.getFieldType() == goog.proto2.FieldDescriptor.FieldType.GROUP) {
286 printer.dedent();
287 printer.append('}');
288 printer.appendLine();
289 } else {
290 printer.appendLine();
291 }
292 }
293};

300/**
301 * Helper class used by the text format serializer for pretty-printing text.
302 * @constructor
303 * @private
304 */
305goog.proto2.TextFormatSerializer.Printer_ = function() {
306 /**
307 * The current indentation count.
308 * @type {number}
309 * @private
310 */
311 this.indentation_ = 0;
312
313 /**
314 * The buffer of string pieces.
315 * @type {Array<string>}
316 * @private
317 */
318 this.buffer_ = [];
319
320 /**
321 * Whether indentation is required before the next append of characters.
322 * @type {boolean}
323 * @private
324 */
325 this.requiresIndentation_ = true;
326};
327
328
329/**
330 * @return {string} The contents of the printer.
331 * @override
332 */
333goog.proto2.TextFormatSerializer.Printer_.prototype.toString = function() {
334 return this.buffer_.join('');
335};
336
337
338/**
339 * Increases the indentation in the printer.
340 */
341goog.proto2.TextFormatSerializer.Printer_.prototype.indent = function() {
342 this.indentation_ += 2;
343};
344
345
346/**
347 * Decreases the indentation in the printer.
348 */
349goog.proto2.TextFormatSerializer.Printer_.prototype.dedent = function() {
350 this.indentation_ -= 2;
351 goog.asserts.assert(this.indentation_ >= 0);
352};
353
354
355/**
356 * Appends the given value to the printer.
357 * @param {*} value The value to append.
358 */
359goog.proto2.TextFormatSerializer.Printer_.prototype.append = function(value) {
360 if (this.requiresIndentation_) {
361 for (var i = 0; i < this.indentation_; ++i) {
362 this.buffer_.push(' ');
363 }
364 this.requiresIndentation_ = false;
365 }
366
367 this.buffer_.push(value.toString());
368};
369
370
371/**
372 * Appends a newline to the printer.
373 */
374goog.proto2.TextFormatSerializer.Printer_.prototype.appendLine = function() {
375 this.buffer_.push('\n');
376 this.requiresIndentation_ = true;
377};

384/**
385 * Helper class for tokenizing the text format.
386 * @param {string} data The string data to tokenize.
387 * @param {boolean=} opt_ignoreWhitespace If true, whitespace tokens will not
388 * be reported by the tokenizer.
389 * @param {boolean=} opt_ignoreComments If true, comment tokens will not be
390 * reported by the tokenizer.
391 * @constructor
392 * @private
393 */
394goog.proto2.TextFormatSerializer.Tokenizer_ =
395 function(data, opt_ignoreWhitespace, opt_ignoreComments) {
396
397 /**
398 * Whether to skip whitespace tokens on output.
399 * @private {boolean}
400 */
401 this.ignoreWhitespace_ = !!opt_ignoreWhitespace;
402
403 /**
404 * Whether to skip comment tokens on output.
405 * @private {boolean}
406 */
407 this.ignoreComments_ = !!opt_ignoreComments;
408
409 /**
410 * The data being tokenized.
411 * @private {string}
412 */
413 this.data_ = data;
414
415 /**
416 * The current index in the data.
417 * @private {number}
418 */
419 this.index_ = 0;
420
421 /**
422 * The data string starting at the current index.
423 * @private {string}
424 */
425 this.currentData_ = data;
426
427 /**
428 * The current token type.
429 * @private {goog.proto2.TextFormatSerializer.Tokenizer_.Token}
430 */
431 this.current_ = {
432 type: goog.proto2.TextFormatSerializer.Tokenizer_.TokenTypes.END,
433 value: null
434 };
435};
436
437
438/**
439 * @typedef {{type: goog.proto2.TextFormatSerializer.Tokenizer_.TokenTypes,
440 * value: ?string}}
441 */
442goog.proto2.TextFormatSerializer.Tokenizer_.Token;
443
444
445/**
446 * @return {goog.proto2.TextFormatSerializer.Tokenizer_.Token} The current
447 * token.
448 */
449goog.proto2.TextFormatSerializer.Tokenizer_.prototype.getCurrent = function() {
450 return this.current_;
451};
452
453
454/**
455 * An enumeration of all the token types.
456 * @enum {!RegExp}
457 */
458goog.proto2.TextFormatSerializer.Tokenizer_.TokenTypes = {
459 END: /---end---/,
460 // Leading "-" to identify "-infinity"."
461 IDENTIFIER: /^-?[a-zA-Z][a-zA-Z0-9_]*/,
462 NUMBER: /^(0x[0-9a-f]+)|(([-])?[0-9][0-9]*(\.?[0-9]+)?(e[+-]?[0-9]+|[f])?)/,
463 COMMENT: /^#.*/,
464 OPEN_BRACE: /^{/,
465 CLOSE_BRACE: /^}/,
466 OPEN_TAG: /^</,
467 CLOSE_TAG: /^>/,
468 OPEN_LIST: /^\[/,
469 CLOSE_LIST: /^\]/,
470 STRING: new RegExp('^"([^"\\\\]|\\\\.)*"'),
471 COLON: /^:/,
472 COMMA: /^,/,
473 SEMI: /^;/,
474 WHITESPACE: /^\s/
475};
476
477
478/**
479 * Advances to the next token.
480 * @return {boolean} True if a valid token was found, false if the end was
481 * reached or no valid token was found.
482 */
483goog.proto2.TextFormatSerializer.Tokenizer_.prototype.next = function() {
484 var types = goog.proto2.TextFormatSerializer.Tokenizer_.TokenTypes;
485
486 // Skip any whitespace if requested.
487 while (this.nextInternal_()) {
488 var type = this.getCurrent().type;
489 if ((type != types.WHITESPACE && type != types.COMMENT) ||
490 (type == types.WHITESPACE && !this.ignoreWhitespace_) ||
491 (type == types.COMMENT && !this.ignoreComments_)) {
492 return true;
493 }
494 }
495
496 // If we reach this point, set the current token to END.
497 this.current_ = {
498 type: goog.proto2.TextFormatSerializer.Tokenizer_.TokenTypes.END,
499 value: null
500 };
501
502 return false;
503};
504
505
506/**
507 * Internal method for determining the next token.
508 * @return {boolean} True if a next token was found, false otherwise.
509 * @private
510 */
511goog.proto2.TextFormatSerializer.Tokenizer_.prototype.nextInternal_ =
512 function() {
513 if (this.index_ >= this.data_.length) {
514 return false;
515 }
516
517 var data = this.currentData_;
518 var types = goog.proto2.TextFormatSerializer.Tokenizer_.TokenTypes;
519 var next = null;
520
521 // Loop through each token type and try to match the beginning of the string
522 // with the token's regular expression.
523 goog.object.some(types, function(type, id) {
524 if (next || type == types.END) {
525 return false;
526 }
527
528 // Note: This regular expression check is at, minimum, O(n).
529 var info = type.exec(data);
530 if (info && info.index == 0) {
531 next = {
532 type: type,
533 value: info[0]
534 };
535 }
536
537 return !!next;
538 });
539
540 // Advance the index by the length of the token.
541 if (next) {
542 this.current_ =
543 /** @type {goog.proto2.TextFormatSerializer.Tokenizer_.Token} */ (next);
544 this.index_ += next.value.length;
545 this.currentData_ = this.currentData_.substring(next.value.length);
546 }
547
548 return !!next;
549};

556/**
557 * Helper class for parsing the text format.
558 * @constructor
559 * @final
560 */
561goog.proto2.TextFormatSerializer.Parser = function() {
562 /**
563 * The error during parsing, if any.
564 * @type {?string}
565 * @private
566 */
567 this.error_ = null;
568
569 /**
570 * The current tokenizer.
571 * @type {goog.proto2.TextFormatSerializer.Tokenizer_}
572 * @private
573 */
574 this.tokenizer_ = null;
575
576 /**
577 * Whether to ignore missing fields in the proto when parsing.
578 * @type {boolean}
579 * @private
580 */
581 this.ignoreMissingFields_ = false;
582};
583
584
585/**
586 * Parses the given data, filling the message as it goes.
587 * @param {goog.proto2.Message} message The message to fill.
588 * @param {string} data The text format data.
589 * @param {boolean=} opt_ignoreMissingFields If true, fields missing in the
590 * proto will be ignored.
591 * @return {boolean} True on success, false on failure. On failure, the
592 * getError method can be called to get the reason for failure.
593 */
594goog.proto2.TextFormatSerializer.Parser.prototype.parse =
595 function(message, data, opt_ignoreMissingFields) {
596 this.error_ = null;
597 this.ignoreMissingFields_ = !!opt_ignoreMissingFields;
598 this.tokenizer_ =
599 new goog.proto2.TextFormatSerializer.Tokenizer_(data, true, true);
600 this.tokenizer_.next();
601 return this.consumeMessage_(message, '');
602};
603
604
605/**
606 * @return {?string} The parse error, if any.
607 */
608goog.proto2.TextFormatSerializer.Parser.prototype.getError = function() {
609 return this.error_;
610};
611
612
613/**
614 * Reports a parse error.
615 * @param {string} msg The error message.
616 * @private
617 */
618goog.proto2.TextFormatSerializer.Parser.prototype.reportError_ =
619 function(msg) {
620 this.error_ = msg;
621};
622
623
624/**
625 * Attempts to consume the given message.
626 * @param {goog.proto2.Message} message The message to consume and fill. If
627 * null, then the message contents will be consumed without ever being set
628 * to anything.
629 * @param {string} delimiter The delimiter expected at the end of the message.
630 * @return {boolean} True on success, false otherwise.
631 * @private
632 */
633goog.proto2.TextFormatSerializer.Parser.prototype.consumeMessage_ =
634 function(message, delimiter) {
635 var types = goog.proto2.TextFormatSerializer.Tokenizer_.TokenTypes;
636 while (!this.lookingAt_('>') && !this.lookingAt_('}') &&
637 !this.lookingAtType_(types.END)) {
638 if (!this.consumeField_(message)) { return false; }
639 }
640
641 if (delimiter) {
642 if (!this.consume_(delimiter)) { return false; }
643 } else {
644 if (!this.lookingAtType_(types.END)) {
645 this.reportError_('Expected END token');
646 }
647 }
648
649 return true;
650};
651
652
653/**
654 * Attempts to consume the value of the given field.
655 * @param {goog.proto2.Message} message The parent message.
656 * @param {goog.proto2.FieldDescriptor} field The field.
657 * @return {boolean} True on success, false otherwise.
658 * @private
659 */
660goog.proto2.TextFormatSerializer.Parser.prototype.consumeFieldValue_ =
661 function(message, field) {
662 var value = this.getFieldValue_(field);
663 if (goog.isNull(value)) { return false; }
664
665 if (field.isRepeated()) {
666 message.add(field, value);
667 } else {
668 message.set(field, value);
669 }
670
671 return true;
672};

674
675/**
676 * Attempts to convert a string to a number.
677 * @param {string} num in hexadecimal or float format.
678 * @return {!number} The converted number or null on error.
679 * @private
680 */
681goog.proto2.TextFormatSerializer.Parser.getNumberFromString_ =
682 function(num) {
683
684 var returnValue = goog.string.contains(num, '.') ?
685 parseFloat(num) : // num is a float.
686 goog.string.parseInt(num); // num is an int.
687
688 goog.asserts.assert(!isNaN(returnValue));
689 goog.asserts.assert(isFinite(returnValue));
690
691 return returnValue;
692};
693
694
695/**
696 * Parse NaN, positive infinity, or negative infinity from a string.
697 * @param {string} identifier An identifier string to check.
698 * @return {?number} Infinity, negative infinity, NaN, or null if none
699 * of the constants could be parsed.
700 * @private
701 */
702goog.proto2.TextFormatSerializer.Parser.parseNumericalConstant_ =
703 function(identifier) {
704 if (/^-?inf(?:inity)?f?$/i.test(identifier)) {
705 return Infinity * (goog.string.startsWith(identifier, '-') ? -1 : 1);
706 }
707
708 if (/^nanf?$/i.test(identifier)) {
709 return NaN;
710 }
711
712 return null;
713};
714
715
716/**
717 * Attempts to parse the given field's value from the stream.
718 * @param {goog.proto2.FieldDescriptor} field The field.
719 * @return {*} The field's value or null if none.
720 * @private
721 */
722goog.proto2.TextFormatSerializer.Parser.prototype.getFieldValue_ =
723 function(field) {
724 var types = goog.proto2.TextFormatSerializer.Tokenizer_.TokenTypes;
725 switch (field.getFieldType()) {
726 case goog.proto2.FieldDescriptor.FieldType.DOUBLE:
727 case goog.proto2.FieldDescriptor.FieldType.FLOAT:
728
729 var identifier = this.consumeIdentifier_();
730 if (identifier) {
731 var numericalIdentifier =
732 goog.proto2.TextFormatSerializer.Parser.parseNumericalConstant_(
733 identifier);
734 // Use isDefAndNotNull since !!NaN is false.
735 if (goog.isDefAndNotNull(numericalIdentifier)) {
736 return numericalIdentifier;
737 }
738 }
739
740 case goog.proto2.FieldDescriptor.FieldType.INT32:
741 case goog.proto2.FieldDescriptor.FieldType.UINT32:
742 case goog.proto2.FieldDescriptor.FieldType.FIXED32:
743 case goog.proto2.FieldDescriptor.FieldType.SFIXED32:
744 case goog.proto2.FieldDescriptor.FieldType.SINT32:
745 var num = this.consumeNumber_();
746 if (!num) {
747 return null;
748 }
749
750 return goog.proto2.TextFormatSerializer.Parser.getNumberFromString_(num);
751
752 case goog.proto2.FieldDescriptor.FieldType.INT64:
753 case goog.proto2.FieldDescriptor.FieldType.UINT64:
754 case goog.proto2.FieldDescriptor.FieldType.FIXED64:
755 case goog.proto2.FieldDescriptor.FieldType.SFIXED64:
756 case goog.proto2.FieldDescriptor.FieldType.SINT64:
757 var num = this.consumeNumber_();
758 if (!num) {
759 return null;
760 }
761
762 if (field.getNativeType() == Number) {
763 // 64-bit number stored as a number.
764 return goog.proto2.TextFormatSerializer.Parser.getNumberFromString_(
765 num);
766 }
767
768 return num; // 64-bit numbers are by default stored as strings.
769
770 case goog.proto2.FieldDescriptor.FieldType.BOOL:
771 var ident = this.consumeIdentifier_();
772 if (!ident) {
773 return null;
774 }
775
776 switch (ident) {
777 case 'true': return true;
778 case 'false': return false;
779 default:
780 this.reportError_('Unknown type for bool: ' + ident);
781 return null;
782 }
783
784 case goog.proto2.FieldDescriptor.FieldType.ENUM:
785 if (this.lookingAtType_(types.NUMBER)) {
786 var num = this.consumeNumber_();
787 if (!num) {
788 return null;
789 }
790
791 return goog.proto2.TextFormatSerializer.Parser.getNumberFromString_(
792 num);
793 } else {
794 // Search the enum type for a matching key.
795 var name = this.consumeIdentifier_();
796 if (!name) {
797 return null;
798 }
799
800 var enumValue = field.getNativeType()[name];
801 if (enumValue == null) {
802 this.reportError_('Unknown enum value: ' + name);
803 return null;
804 }
805
806 return enumValue;
807 }
808
809 case goog.proto2.FieldDescriptor.FieldType.BYTES:
810 case goog.proto2.FieldDescriptor.FieldType.STRING:
811 return this.consumeString_();
812 }
813};

816/**
817 * Attempts to consume a nested message.
818 * @param {goog.proto2.Message} message The parent message.
819 * @param {goog.proto2.FieldDescriptor} field The field.
820 * @return {boolean} True on success, false otherwise.
821 * @private
822 */
823goog.proto2.TextFormatSerializer.Parser.prototype.consumeNestedMessage_ =
824 function(message, field) {
825 var delimiter = '';
826
827 // Messages support both < > and { } as delimiters for legacy reasons.
828 if (this.tryConsume_('<')) {
829 delimiter = '>';
830 } else {
831 if (!this.consume_('{')) { return false; }
832 delimiter = '}';
833 }
834
835 var msg = field.getFieldMessageType().createMessageInstance();
836 var result = this.consumeMessage_(msg, delimiter);
837 if (!result) { return false; }
838
839 // Add the message to the parent message.
840 if (field.isRepeated()) {
841 message.add(field, msg);
842 } else {
843 message.set(field, msg);
844 }
845
846 return true;

850/**
851 * Attempts to consume the value of an unknown field. This method uses
852 * heuristics to try to consume just the right tokens.
853 * @return {boolean} True on success, false otherwise.
854 * @private
855 */
856goog.proto2.TextFormatSerializer.Parser.prototype.consumeUnknownFieldValue_ =
857 function() {
858 // : is optional.
859 this.tryConsume_(':');
860
861 // Handle form: [.. , ... , ..]
862 if (this.tryConsume_('[')) {
863 while (true) {
864 this.tokenizer_.next();
865 if (this.tryConsume_(']')) {
866 break;
867 }
868 if (!this.consume_(',')) { return false; }
869 }
870
871 return true;
872 }
873
874 // Handle nested messages/groups.
875 if (this.tryConsume_('<')) {
876 return this.consumeMessage_(null /* unknown */, '>');
877 } else if (this.tryConsume_('{')) {
878 return this.consumeMessage_(null /* unknown */, '}');
879 } else {
880 // Otherwise, consume a single token for the field value.
881 this.tokenizer_.next();
882 }
883
884 return true;

888/**
889 * Attempts to consume a field under a message.
890 * @param {goog.proto2.Message} message The parent message. If null, then the
891 * field value will be consumed without being assigned to anything.
892 * @return {boolean} True on success, false otherwise.
893 * @private
894 */
895goog.proto2.TextFormatSerializer.Parser.prototype.consumeField_ =
896 function(message) {
897 var fieldName = this.consumeIdentifier_();
898 if (!fieldName) {
899 this.reportError_('Missing field name');
900 return false;
901 }
902
903 var field = null;
904 if (message) {
905 field = message.getDescriptor().findFieldByName(fieldName.toString());
906 }
907
908 if (field == null) {
909 if (this.ignoreMissingFields_) {
910 return this.consumeUnknownFieldValue_();
911 } else {
912 this.reportError_('Unknown field: ' + fieldName);
913 return false;
914 }
915 }
916
917 if (field.getFieldType() == goog.proto2.FieldDescriptor.FieldType.MESSAGE ||
918 field.getFieldType() == goog.proto2.FieldDescriptor.FieldType.GROUP) {
919 // : is optional here.
920 this.tryConsume_(':');
921 if (!this.consumeNestedMessage_(message, field)) { return false; }
922 } else {
923 // Long Format: "someField: 123"
924 // Short Format: "someField: [123, 456, 789]"
925 if (!this.consume_(':')) { return false; }
926
927 if (field.isRepeated() && this.tryConsume_('[')) {
928 // Short repeated format, e.g. "foo: [1, 2, 3]"
929 while (true) {
930 if (!this.consumeFieldValue_(message, field)) { return false; }
931 if (this.tryConsume_(']')) {
932 break;
933 }
934 if (!this.consume_(',')) { return false; }
935 }
936 } else {
937 // Normal field format.
938 if (!this.consumeFieldValue_(message, field)) { return false; }
939 }
940 }
941
942 // For historical reasons, fields may optionally be separated by commas or
943 // semicolons.
944 this.tryConsume_(',') || this.tryConsume_(';');
945 return true;
946};
