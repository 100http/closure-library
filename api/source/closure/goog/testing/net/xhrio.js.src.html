105/**
106 * Simulates the static XhrIo send method.
107 * @param {string} url Uri to make request to.
108 * @param {Function=} opt_callback Callback function for when request is
109 * complete.
110 * @param {string=} opt_method Send method, default: GET.
111 * @param {string=} opt_content Post data.
112 * @param {Object|goog.structs.Map=} opt_headers Map of headers to add to the
113 * request.
114 * @param {number=} opt_timeoutInterval Number of milliseconds after which an
115 * incomplete request will be aborted; 0 means no timeout is set.
116 * @param {boolean=} opt_withCredentials Whether to send credentials with the
117 * request. Default to false. See {@link goog.net.XhrIo#setWithCredentials}.
118 * @return {!goog.testing.net.XhrIo} The mocked sent XhrIo.
119 */
120goog.testing.net.XhrIo.send = function(url, opt_callback, opt_method,
121 opt_content, opt_headers,
122 opt_timeoutInterval,
123 opt_withCredentials) {
124 var x = new goog.testing.net.XhrIo();
125 goog.testing.net.XhrIo.sendInstances_.push(x);
126 if (opt_callback) {
127 goog.events.listen(x, goog.net.EventType.COMPLETE, opt_callback);
128 }
129 goog.events.listen(x,
130 goog.net.EventType.READY,
131 goog.partial(goog.testing.net.XhrIo.cleanupSend_, x));
132 if (opt_timeoutInterval) {
133 x.setTimeoutInterval(opt_timeoutInterval);
134 }
135 x.setWithCredentials(Boolean(opt_withCredentials));
136 x.send(url, opt_method, opt_content, opt_headers);
137
138 return x;
139};

385/**
386 * Simulates the XhrIo send.
387 * @param {string} url Uri to make request too.
388 * @param {string=} opt_method Send method, default: GET.
389 * @param {string=} opt_content Post data.
390 * @param {Object|goog.structs.Map=} opt_headers Map of headers to add to the
391 * request.
392 */
393goog.testing.net.XhrIo.prototype.send = function(url, opt_method, opt_content,
394 opt_headers) {
395 if (this.xhr_) {
396 throw Error('[goog.net.XhrIo] Object is active with another request');
397 }
398
399 this.lastUri_ = url;
400 this.lastMethod_ = opt_method || 'GET';
401 this.lastContent_ = opt_content;
402 this.lastHeaders_ = opt_headers;
403
404 if (this.testQueue_) {
405 this.testQueue_.enqueue(['s', url, opt_method, opt_content, opt_headers]);
406 }
407 this.xhr_ = true;
408 this.active_ = true;
409 this.readyState_ = goog.net.XmlHttp.ReadyState.UNINITIALIZED;
410 this.simulateReadyStateChange(goog.net.XmlHttp.ReadyState.LOADING);
411};

425/**
426 * Simulates changing to the new ready state.
427 * @param {number} readyState Ready state to change to.
428 */
429goog.testing.net.XhrIo.prototype.simulateReadyStateChange =
430 function(readyState) {
431 if (readyState < this.readyState_) {
432 throw Error('Readystate cannot go backwards');
433 }
434
435 // INTERACTIVE can be dispatched repeatedly as more data is reported.
436 if (readyState == goog.net.XmlHttp.ReadyState.INTERACTIVE &&
437 readyState == this.readyState_) {
438 this.dispatchEvent(goog.net.EventType.READY_STATE_CHANGE);
439 return;
440 }
441
442 while (this.readyState_ < readyState) {
443 this.readyState_++;
444 this.dispatchEvent(goog.net.EventType.READY_STATE_CHANGE);
445
446 if (this.readyState_ == goog.net.XmlHttp.ReadyState.COMPLETE) {
447 this.active_ = false;
448 this.dispatchEvent(goog.net.EventType.COMPLETE);
449 }
450 }
451};

469/**
470 * Simulates receiving a response.
471 * @param {number} statusCode Simulated status code.
472 * @param {string|Document|ArrayBuffer|null} response Simulated response.
473 * @param {Object=} opt_headers Simulated response headers.
474 */
475goog.testing.net.XhrIo.prototype.simulateResponse = function(statusCode,
476 response, opt_headers) {
477 this.statusCode_ = statusCode;
478 this.response_ = response || '';
479 this.responseHeaders_ = opt_headers || {};
480
481 try {
482 if (this.isSuccess()) {
483 this.simulateReadyStateChange(goog.net.XmlHttp.ReadyState.COMPLETE);
484 this.dispatchEvent(goog.net.EventType.SUCCESS);
485 } else {
486 this.lastErrorCode_ = goog.net.ErrorCode.HTTP_ERROR;
487 this.lastError_ = this.getStatusText() + ' [' + this.getStatus() + ']';
488 this.simulateReadyStateChange(goog.net.XmlHttp.ReadyState.COMPLETE);
489 this.dispatchEvent(goog.net.EventType.ERROR);
490 }
491 } finally {
492 this.simulateReady();
493 }
494};

729/**
730 * Returns all response headers as a key-value map.
731 * Multiple values for the same header key can be combined into one,
732 * separated by a comma and a space.
733 * Note that the native getResponseHeader method for retrieving a single header
734 * does a case insensitive match on the header name. This method does not
735 * include any case normalization logic, it will just return a key-value
736 * representation of the headers.
737 * See: http://www.w3.org/TR/XMLHttpRequest/#the-getresponseheader()-method
738 * @return {!Object<string, string>} An object with the header keys as keys
739 * and header values as values.
740 */
741goog.testing.net.XhrIo.prototype.getResponseHeaders = function() {
742 var headersObject = {};
743 goog.object.forEach(this.responseHeaders_, function(value, key) {
744 if (headersObject[key]) {
745 headersObject[key] += ', ' + value;
746 } else {
747 headersObject[key] = value;
748 }
749 });
750 return headersObject;
