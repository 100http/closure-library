15/**
16 * @fileoverview This file defines a factory that can be used to mock and
17 * replace an entire class. This allows for mocks to be used effectively with
18 * "new" instead of having to inject all instances. Essentially, a given class
19 * is replaced with a proxy to either a loose or strict mock. Proxies locate
20 * the appropriate mock based on constructor arguments.
21 *
22 * The usage is:
23 * <ul>
24 * <li>Create a mock with one of the provided methods with a specifc set of
25 * constructor arguments
26 * <li>Set expectations by calling methods on the mock object
27 * <li>Call $replay() on the mock object
28 * <li>Instantiate the object as normal
29 * <li>Call $verify() to make sure that expectations were met
30 * <li>Call reset on the factory to revert all classes back to their original
31 * state
32 * </ul>
33 *
34 * For examples, please see the unit test.
35 *
36 */
37

51/**
52 * A record that represents all the data associated with a mock replacement of

179/**
180 * Finds the mock corresponding to a given argument set. Throws an error if
181 * there is no appropriate match found.
182 * @param {Array<?>} args An array of function arguments.
183 * @return {goog.testing.StrictMock|goog.testing.LooseMock|null} The mock
184 * corresponding to a given argument set.
185 */
186goog.testing.MockClassRecord.prototype.findMockInstance = function(args) {
187 for (var i = 0; i < this.instancesByArgs_.length; i++) {
188 var instanceArgs = this.instancesByArgs_[i].args;
189 if (goog.testing.mockmatchers.flexibleArrayMatcher(instanceArgs, args)) {
190 return this.instancesByArgs_[i].mock;
191 }
192 }
193
194 return null;
195};

256/**
257 * Iterates through a namespace to find the name of a given class. This is done
258 * solely to support compilation since string identifiers would break down.
259 * Tests usually aren't compiled, but the functionality is supported.
260 * @param {Object} namespace A javascript namespace (e.g. goog.testing).
261 * @param {Function} classToMock The class whose name should be returned.
262 * @return {string} The name of the class.
263 * @private
264 */
265goog.testing.MockClassFactory.prototype.getClassName_ = function(namespace,
266 classToMock) {
267 var namespaces;
268 if (namespace === goog.global) {
269 namespaces = goog.testing.TestCase.getGlobals();
270 } else {
271 namespaces = [namespace];
272 }
273 for (var i = 0; i < namespaces.length; i++) {
274 for (var prop in namespaces[i]) {
275 if (namespaces[i][prop] === classToMock) {
276 return prop;
277 }
278 }
279 }
280
281 throw Error('Class is not a part of the given namespace');
282};

296/**
297 * Returns a proxy constructor closure. Since this is a constructor, "this"
298 * refers to the local scope of the constructed object thus bind cannot be
299 * used.
300 * @param {string} className The name of the class.
301 * @param {Function} mockFinder A bound function that returns the mock
302 * associated with a class given the constructor's argument list.
303 * @return {!Function} A proxy constructor.
304 * @private
305 */
306goog.testing.MockClassFactory.prototype.getProxyCtor_ = function(className,
307 mockFinder) {
308 return function() {
309 this.$mock_ = mockFinder(className, arguments);
310 if (!this.$mock_) {
311 // The "arguments" variable is not a proper Array so it must be converted.
312 var args = Array.prototype.slice.call(arguments, 0);
313 throw Error('No mock found for ' + className + ' with arguments ' +
314 args.join(', '));
315 }
316 };
317};

320/**
321 * Returns a proxy function for a mock class instance. This function cannot
322 * be used with bind since "this" must refer to the scope of the proxy
323 * constructor.
324 * @param {string} fnName The name of the function that should be proxied.
325 * @return {!Function} A proxy function.
326 * @private
327 */
328goog.testing.MockClassFactory.prototype.getProxyFunction_ = function(fnName) {
329 return function() {
330 return this.$mock_[fnName].apply(this.$mock_, arguments);
331 };

349/**
350 * Create a proxy class. A proxy will pass functions to the mock for a class.
351 * The proxy class only covers prototype methods. A static mock is not build
352 * simultaneously since it might be strict or loose. The proxy class inherits
353 * from the target class in order to preserve instanceof checks.
354 * @param {Object} namespace A javascript namespace (e.g. goog.testing).
355 * @param {Function} classToMock The class that will be proxied.
356 * @param {string} className The name of the class.
357 * @return {!Function} The proxy for provided class.
358 * @private
359 */
360goog.testing.MockClassFactory.prototype.createProxy_ = function(namespace,
361 classToMock, className) {
362 var proxy = this.getProxyCtor_(className,
363 goog.bind(this.findMockInstance_, this));
364 var protoToProxy = classToMock.prototype;
365 goog.inherits(proxy, classToMock);
366
367 for (var prop in protoToProxy) {
368 if (goog.isFunction(protoToProxy[prop])) {
369 proxy.prototype[prop] = this.getProxyFunction_(prop);
370 }
371 }
372
373 // For IE the for-in-loop does not contain any properties that are not
374 // enumerable on the prototype object (for example isPrototypeOf from
375 // Object.prototype) and it will also not include 'replace' on objects that
376 // extend String and change 'replace' (not that it is common for anyone to
377 // extend anything except Object).
378 // TODO (arv): Implement goog.object.getIterator and replace this loop.
379
380 goog.array.forEach(goog.testing.MockClassFactory.PROTOTYPE_FIELDS_,
381 function(field) {
382 if (Object.prototype.hasOwnProperty.call(protoToProxy, field)) {
383 proxy.prototype[field] = this.getProxyFunction_(field);
384 }
385 }, this);
386
387 this.mockClassRecords_[className] = new goog.testing.MockClassRecord(
388 namespace, className, classToMock, proxy);
389 namespace[className] = proxy;
390 return proxy;
391};

394/**
395 * Gets either a loose or strict mock for a given class based on a set of
396 * arguments.
397 * @param {Object} namespace A javascript namespace (e.g. goog.testing).
398 * @param {Function} classToMock The class that will be mocked.
399 * @param {boolean} isStrict Whether or not the mock should be strict.
400 * @param {goog.array.ArrayLike} ctorArgs The arguments associated with this
401 * instance's constructor.
402 * @return {!goog.testing.StrictMock|!goog.testing.LooseMock} The mock created
403 * for the provided class.
404 * @private
405 */
406goog.testing.MockClassFactory.prototype.getMockClass_ =
407 function(namespace, classToMock, isStrict, ctorArgs) {
408 var className = this.getClassName_(namespace, classToMock);
409
410 // The namespace and classToMock variables should be removed from the
411 // passed in argument stack.
412 ctorArgs = goog.array.slice(ctorArgs, 2);
413
414 if (goog.isFunction(classToMock)) {
415 var mock = isStrict ? new goog.testing.StrictMock(classToMock) :
416 new goog.testing.LooseMock(classToMock);
417
418 if (!this.classHasMock_(className)) {
419 this.createProxy_(namespace, classToMock, className);
420 } else {
421 var instance = this.findMockInstance_(className, ctorArgs);
422 if (instance) {
423 throw Error('Mock instance already created for ' + className +
424 ' with arguments ' + ctorArgs.join(', '));
425 }
426 }
427 this.mockClassRecords_[className].addMockInstance(ctorArgs, mock);
428
429 return mock;
430 } else {
431 throw Error('Cannot create a mock class for ' + className +
432 ' of type ' + typeof classToMock);
433 }
434};

467/**
468 * Creates either a loose or strict mock for the static functions of a given
469 * class.
470 * @param {Function} classToMock The class whose static functions will be
471 * mocked. This should be the original class and not the proxy.
472 * @param {string} className The name of the class.
473 * @param {Function} proxy The proxy that will replace the original class.
474 * @param {boolean} isStrict Whether or not the mock should be strict.
475 * @return {!goog.testing.StrictMock|!goog.testing.LooseMock} The mock created
476 * for the static functions of the provided class.
477 * @private
478 */
479goog.testing.MockClassFactory.prototype.createStaticMock_ =
480 function(classToMock, className, proxy, isStrict) {
481 var mock = isStrict ? new goog.testing.StrictMock(classToMock, true) :
482 new goog.testing.LooseMock(classToMock, false, true);
483
484 for (var prop in classToMock) {
485 if (goog.isFunction(classToMock[prop])) {
486 proxy[prop] = goog.bind(mock.$mockMethod, mock, prop);
487 } else if (classToMock[prop] !== classToMock.prototype) {
488 proxy[prop] = classToMock[prop];
489 }
490 }
491
492 this.mockClassRecords_[className].setStaticMock(mock);
493 return mock;
494};

497/**
498 * Gets either a loose or strict mock for the static functions of a given class.
499 * @param {Object} namespace A javascript namespace (e.g. goog.testing).
500 * @param {Function} classToMock The class whose static functions will be
501 * mocked. This should be the original class and not the proxy.
502 * @param {boolean} isStrict Whether or not the mock should be strict.
503 * @return {goog.testing.StrictMock|goog.testing.LooseMock} The mock created
504 * for the static functions of the provided class.
505 * @private
506 */
507goog.testing.MockClassFactory.prototype.getStaticMock_ = function(namespace,
508 classToMock, isStrict) {
509 var className = this.getClassName_(namespace, classToMock);
510
511 if (goog.isFunction(classToMock)) {
512 if (!this.classHasMock_(className)) {
513 var proxy = this.createProxy_(namespace, classToMock, className);
514 var mock = this.createStaticMock_(classToMock, className, proxy,
515 isStrict);
516 return mock;
517 }
518
519 if (!this.mockClassRecords_[className].getStaticMock()) {
520 var proxy = this.mockClassRecords_[className].getProxy();
521 var originalClass = this.mockClassRecords_[className].getOriginalClass();
522 var mock = this.createStaticMock_(originalClass, className, proxy,
523 isStrict);
524 return mock;
525 } else {
526 var mock = this.mockClassRecords_[className].getStaticMock();
527 var mockIsStrict = mock instanceof goog.testing.StrictMock;
528
529 if (mockIsStrict != isStrict) {
530 var mockType = mock instanceof goog.testing.StrictMock ? 'strict' :
531 'loose';
532 var requestedType = isStrict ? 'strict' : 'loose';
533 throw Error('Requested a ' + requestedType + ' static mock, but a ' +
534 mockType + ' mock already exists.');
535 }
536
537 return mock;
538 }
539 } else {
540 throw Error('Cannot create a mock for the static functions of ' +
541 className + ' of type ' + typeof classToMock);
542 }
543};
