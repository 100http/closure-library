15/**
16 * @fileoverview A wrapper for MockControl that provides mocks and assertions
17 * for testing asynchronous code. All assertions will only be verified when
18 * $verifyAll is called on the wrapped MockControl.
19 *
20 * This class is meant primarily for testing code that exposes asynchronous APIs
21 * without being truly asynchronous (using asynchronous primitives like browser
22 * events or timeouts). This is often the case when true asynchronous
23 * depedencies have been mocked out. This means that it doesn't rely on
24 * AsyncTestCase or DeferredTestCase, although it can be used with those as
25 * well.
26 *
27 * Example usage:
28 *
29 * <pre>
30 * var mockControl = new goog.testing.MockControl();
31 * var asyncMockControl = new goog.testing.async.MockControl(mockControl);
32 *
33 * myAsyncObject.onSuccess(asyncMockControl.asyncAssertEquals(
34 * 'callback should run and pass the correct value',
35 * 'http://someurl.com');
36 * asyncMockControl.assertDeferredEquals(
37 * 'deferred object should be resolved with the correct value',
38 * 'http://someurl.com',
39 * myAsyncObject.getDeferredUrl());
40 * asyncMockControl.run();
41 * mockControl.$verifyAll();
42 * </pre>
43 *
44 */
45

75/**
76 * Returns a function that will assert that it will be called, and run the given
77 * callback when it is.
78 *
79 * @param {string} name The name of the callback mock.
80 * @param {function(...*) : *} callback The wrapped callback. This will be
81 * called when the returned function is called.
82 * @param {Object=} opt_selfObj The object which this should point to when the
83 * callback is run.
84 * @return {!Function} The mock callback.
85 * @suppress {missingProperties} Mocks do not fit in the type system well.
86 */
87goog.testing.async.MockControl.prototype.createCallbackMock = function(
88 name, callback, opt_selfObj) {
89 goog.asserts.assert(
90 goog.isString(name),
91 'name parameter ' + goog.debug.deepExpose(name) + ' should be a string');
92
93 var ignored = new goog.testing.mockmatchers.IgnoreArgument();
94
95 // Use everyone's favorite "double-cast" trick to subvert the type system.
96 var obj = /** @type {Object} */ (this.mockControl_.createFunctionMock(name));
97 var fn = /** @type {Function} */ (obj);
98
99 fn(ignored).$does(function(args) {
100 if (opt_selfObj) {
101 callback = goog.bind(callback, opt_selfObj);
102 }
103 return callback.apply(this, args);
104 });
105 fn.$replay();
106 return function() { return fn(arguments); };
107};

110/**
111 * Returns a function that will assert that its arguments are equal to the
112 * arguments given to asyncAssertEquals. In addition, the function also asserts
113 * that it will be called.
114 *
115 * @param {string} message A message to print if the arguments are wrong.
116 * @param {...*} var_args The arguments to assert.
117 * @return {function(...*) : void} The mock callback.
118 */
119goog.testing.async.MockControl.prototype.asyncAssertEquals = function(
120 message, var_args) {
121 var expectedArgs = Array.prototype.slice.call(arguments, 1);
122 return this.createCallbackMock('asyncAssertEquals', function() {
123 assertObjectEquals(
124 message, expectedArgs, Array.prototype.slice.call(arguments));
125 });
126};

129/**
130 * Asserts that a deferred object will have an error and call its errback
131 * function.
132 * @param {goog.async.Deferred} deferred The deferred object.
133 * @param {function() : void} fn A function wrapping the code in which the error
134 * will occur.
135 */
136goog.testing.async.MockControl.prototype.assertDeferredError = function(
137 deferred, fn) {
138 deferred.addErrback(this.createCallbackMock(
139 'assertDeferredError', function() {}));
140 goog.testing.asserts.callWithoutLogging(fn);
141};

144/**
145 * Asserts that a deferred object will call its callback with the given value.
146 *
147 * @param {string} message A message to print if the arguments are wrong.
148 * @param {goog.async.Deferred|*} expected The expected value. If this is a
149 * deferred object, then the expected value is the deferred value.
150 * @param {goog.async.Deferred|*} actual The actual value. If this is a deferred
151 * object, then the actual value is the deferred value. Either this or
152 * 'expected' must be deferred.
153 */
154goog.testing.async.MockControl.prototype.assertDeferredEquals = function(
155 message, expected, actual) {
156 if (expected instanceof goog.async.Deferred &&
157 actual instanceof goog.async.Deferred) {
158 // Assert that the first deferred is resolved.
159 expected.addCallback(this.createCallbackMock(
160 'assertDeferredEquals', function(exp) {
161 // Assert that the second deferred is resolved, and that the value is
162 // as expected.
163 actual.addCallback(this.asyncAssertEquals(message, exp));
164 }, this));
165 } else if (expected instanceof goog.async.Deferred) {
166 expected.addCallback(this.createCallbackMock(
167 'assertDeferredEquals', function(exp) {
168 assertObjectEquals(message, exp, actual);
169 }));
170 } else if (actual instanceof goog.async.Deferred) {
171 actual.addCallback(this.asyncAssertEquals(message, expected));
172 } else {
173 throw Error('Either expected or actual must be deferred');
174 }
175};
