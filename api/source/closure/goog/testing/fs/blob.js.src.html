59/**
60 * Creates a blob with bytes of a blob ranging from the optional start
61 * parameter up to but not including the optional end parameter, and with a type
62 * attribute that is the value of the optional contentType parameter.
63 * @see http://www.w3.org/TR/FileAPI/#dfn-slice
64 * @param {number=} opt_start The start byte offset.
65 * @param {number=} opt_end The end point of a slice.
66 * @param {string=} opt_contentType The type of the resulting Blob.
67 * @return {!goog.testing.fs.Blob} The result blob of the slice operation.
68 */
69goog.testing.fs.Blob.prototype.slice = function(
70 opt_start, opt_end, opt_contentType) {
71 var relativeStart;
72 if (goog.isNumber(opt_start)) {
73 relativeStart = (opt_start < 0) ?
74 Math.max(this.data_.length + opt_start, 0) :
75 Math.min(opt_start, this.data_.length);
76 } else {
77 relativeStart = 0;
78 }
79 var relativeEnd;
80 if (goog.isNumber(opt_end)) {
81 relativeEnd = (opt_end < 0) ?
82 Math.max(this.data_.length + opt_end, 0) :
83 Math.min(opt_end, this.data_.length);
84 } else {
85 relativeEnd = this.data_.length;
86 }
87 var span = Math.max(relativeEnd - relativeStart, 0);
88 return new goog.testing.fs.Blob(
89 this.data_.substr(relativeStart, span),
90 opt_contentType);
91};

103/**
104 * @return {!ArrayBuffer} The string data encapsulated by the blob as an
105 * ArrayBuffer.
106 */
107goog.testing.fs.Blob.prototype.toArrayBuffer = function() {
108 var buf = new ArrayBuffer(this.data_.length * 2);
109 var arr = new Uint16Array(buf);
110 for (var i = 0; i < this.data_.length; i++) {
111 arr[i] = this.data_.charCodeAt(i);
112 }
113 return buf;
114};
