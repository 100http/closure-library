75/**
76 * Helper that returns the previous or next (in document order) node from the
77 * given node that is a non-empty text node, or null if none is found or
78 * opt_stopAt is not an ancestor of node. Note that if the given node has
79 * children, the search will start from the end or start tag of the node
80 * (depending on whether it's searching for the previous or next node), meaning
81 * all its descendants will be included in the search, unless
82 * opt_skipDescendants is true.
83 * @param {Node} node Node to start searching from.
84 * @param {Node=} opt_stopAt Node to stop searching at (search will be
85 * restricted to this node's subtree), defaults to the body of the document
86 * containing node.
87 * @param {boolean=} opt_skipDescendants Whether to skip searching the given
88 * node's descentants.
89 * @param {boolean=} opt_isPrevious Whether to search for the previous non-empty
90 * text node instead of the next one.
91 * @return {Text} The next (in document order) node from the given node that
92 * is a non-empty text node, or null if none is found or opt_stopAt is not
93 * an ancestor of node.
94 * @private
95 */
96goog.testing.editor.dom.getPreviousNextNonEmptyTextNodeHelper_ = function(
97 node, opt_stopAt, opt_skipDescendants, opt_isPrevious) {
98 opt_stopAt = opt_stopAt || node.ownerDocument.body;
99 // Initializing the iterator to iterate over the children of opt_stopAt
100 // makes it stop only when it finishes iterating through all of that
101 // node's children, even though we will start at a different node and exit
102 // that starting node's subtree in the process.
103 var iter = new goog.dom.TagIterator(opt_stopAt, opt_isPrevious);
104
105 // TODO(user): Move this logic to a new method in TagIterator such as
106 // skipToNode().
107 // Then we set the iterator to start at the given start node, not opt_stopAt.
108 var walkType; // Let TagIterator set the initial walk type by default.
109 var depth = goog.testing.editor.dom.getRelativeDepth_(node, opt_stopAt);
110 if (depth == -1) {
111 return null; // Fail because opt_stopAt is not an ancestor of node.
112 }
113 if (node.nodeType == goog.dom.NodeType.ELEMENT) {
114 if (opt_skipDescendants) {
115 // Specifically set the initial walk type so that we skip the descendant
116 // subtree by starting at the start if going backwards or at the end if
117 // going forwards.
118 walkType = opt_isPrevious ? goog.dom.TagWalkType.START_TAG :
119 goog.dom.TagWalkType.END_TAG;
120 } else {
121 // We're starting "inside" an element node so the depth needs to be one
122 // deeper than the node's actual depth. That's how TagIterator works!
123 depth++;
124 }
125 }
126 iter.setPosition(node, walkType, depth);
127
128 // Advance the iterator so it skips the start node.
129 try {
130 iter.next();
131 } catch (e) {
132 return null; // It could have been a leaf node.
133 }
134 // Now just get the first non-empty text node the iterator finds.
135 var filter = goog.iter.filter(iter,
136 goog.testing.editor.dom.isNonEmptyTextNode_);
137 try {
138 return /** @type {Text} */ (filter.next());
139 } catch (e) { // No next item is available so return null.
140 return null;
141 }
142};

156/**
157 * Returns the depth of the given node relative to the given parent node, or -1
158 * if the given node is not a descendant of the given parent node. E.g. if
159 * node == parentNode returns 0, if node.parentNode == parentNode returns 1,
160 * etc.
161 * @param {Node} node Node whose depth to get.
162 * @param {Node} parentNode Node relative to which the depth should be
163 * calculated.
164 * @return {number} The depth of the given node relative to the given parent
165 * node, or -1 if the given node is not a descendant of the given parent
166 * node.
167 * @private
168 */
169goog.testing.editor.dom.getRelativeDepth_ = function(node, parentNode) {
170 var depth = 0;
171 while (node) {
172 if (node == parentNode) {
173 return depth;
174 }
175 node = node.parentNode;
176 depth++;
177 }
178 return -1;
179};

182/**
183 * Assert that the range is surrounded by the given strings. This is useful
184 * because different browsers can place the range endpoints inside different
185 * nodes even when visually the range looks the same. Also, there may be empty
186 * text nodes in the way (again depending on the browser) making it difficult to
187 * use assertRangeEquals.
188 * @param {string} before String that should occur immediately before the start
189 * point of the range. If this is the empty string, assert will only succeed
190 * if there is no text before the start point of the range.
191 * @param {string} after String that should occur immediately after the end
192 * point of the range. If this is the empty string, assert will only succeed
193 * if there is no text after the end point of the range.
194 * @param {goog.dom.AbstractRange} range The range to be tested.
195 * @param {Node=} opt_stopAt Node to stop searching at (search will be
196 * restricted to this node's subtree).
197 */
198goog.testing.editor.dom.assertRangeBetweenText = function(before,
199 after,
200 range,
201 opt_stopAt) {
202 var previousText =
203 goog.testing.editor.dom.getTextFollowingRange_(range, true, opt_stopAt);
204 if (before == '') {
205 assertNull('Expected nothing before range but found <' + previousText + '>',
206 previousText);
207 } else {
208 assertNotNull('Expected <' + before + '> before range but found nothing',
209 previousText);
210 assertTrue('Expected <' + before + '> before range but found <' +
211 previousText + '>',
212 goog.string.endsWith(
213 /** @type {string} */ (previousText), before));
214 }
215 var nextText =
216 goog.testing.editor.dom.getTextFollowingRange_(range, false, opt_stopAt);
217 if (after == '') {
218 assertNull('Expected nothing after range but found <' + nextText + '>',
219 nextText);
220 } else {
221 assertNotNull('Expected <' + after + '> after range but found nothing',
222 nextText);
223 assertTrue('Expected <' + after + '> after range but found <' +
224 nextText + '>',
225 goog.string.startsWith(
226 /** @type {string} */ (nextText), after));
227 }
228};

231/**
232 * Returns the text that follows the given range, where the term "follows" means
233 * "comes immediately before the start of the range" if isBefore is true, and
234 * "comes immediately after the end of the range" if isBefore is false, or null
235 * if no non-empty text node is found.
236 * @param {goog.dom.AbstractRange} range The range to search from.
237 * @param {boolean} isBefore Whether to search before the range instead of
238 * after it.
239 * @param {Node=} opt_stopAt Node to stop searching at (search will be
240 * restricted to this node's subtree).
241 * @return {?string} The text that follows the given range, or null if no
242 * non-empty text node is found.
243 * @private
244 */
245goog.testing.editor.dom.getTextFollowingRange_ = function(range,
246 isBefore,
247 opt_stopAt) {
248 var followingTextNode;
249 var endpointNode = isBefore ? range.getStartNode() : range.getEndNode();
250 var endpointOffset = isBefore ? range.getStartOffset() : range.getEndOffset();
251 var getFollowingTextNode =
252 isBefore ? goog.testing.editor.dom.getPreviousNonEmptyTextNode :
253 goog.testing.editor.dom.getNextNonEmptyTextNode;
254
255 if (endpointNode.nodeType == goog.dom.NodeType.TEXT) {
256 // Range endpoint is in a text node.
257 var endText = endpointNode.nodeValue;
258 if (isBefore ? endpointOffset > 0 : endpointOffset < endText.length) {
259 // There is text in this node following the endpoint so return the portion
260 // that follows the endpoint.
261 return isBefore ? endText.substr(0, endpointOffset) :
262 endText.substr(endpointOffset);
263 } else {
264 // There is no text following the endpoint so look for the follwing text
265 // node.
266 followingTextNode = getFollowingTextNode(endpointNode, opt_stopAt);
267 return followingTextNode && followingTextNode.nodeValue;
268 }
269 } else {
270 // Range endpoint is in an element node.
271 var numChildren = endpointNode.childNodes.length;
272 if (isBefore ? endpointOffset > 0 : endpointOffset < numChildren) {
273 // There is at least one child following the endpoint.
274 var followingChild =
275 endpointNode.childNodes[isBefore ? endpointOffset - 1 :
276 endpointOffset];
277 if (goog.testing.editor.dom.isNonEmptyTextNode_(followingChild)) {
278 // The following child has text so return that.
279 return followingChild.nodeValue;
280 } else {
281 // The following child has no text so look for the following text node.
282 followingTextNode = getFollowingTextNode(followingChild, opt_stopAt);
283 return followingTextNode && followingTextNode.nodeValue;
284 }
285 } else {
286 // There is no child following the endpoint, so search from the endpoint
287 // node, but don't search its children because they are not following the
288 // endpoint!
289 followingTextNode = getFollowingTextNode(endpointNode, opt_stopAt, true);
290 return followingTextNode && followingTextNode.nodeValue;
291 }
292 }
