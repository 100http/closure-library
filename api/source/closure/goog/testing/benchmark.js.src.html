15goog.provide('goog.testing.benchmark');
16goog.setTestOnly('goog.testing.benchmark');
17
18goog.require('goog.dom');
19goog.require('goog.dom.TagName');
20goog.require('goog.testing.PerformanceTable');
21goog.require('goog.testing.PerformanceTimer');
22goog.require('goog.testing.TestCase');
23
24
25/**
26 * Run the benchmarks.
27 * @private
28 */
29goog.testing.benchmark.run_ = function() {
30 // Parse the 'times' query parameter if it's set.
31 var times = 200;
32 var search = window.location.search;
33 var timesMatch = search.match(/(?:\?|&)times=([^?&]+)/i);
34 if (timesMatch) {
35 times = Number(timesMatch[1]);
36 }
37
38 var prefix = 'benchmark';
39
40 // First, get the functions.
41 var testSources = goog.testing.TestCase.getGlobals();
42
43 var benchmarks = {};
44 var names = [];
45
46 for (var i = 0; i < testSources.length; i++) {
47 var testSource = testSources[i];
48 for (var name in testSource) {
49 if ((new RegExp('^' + prefix)).test(name)) {
50 var ref;
51 try {
52 ref = testSource[name];
53 } catch (ex) {
54 // NOTE(brenneman): When running tests from a file:// URL on Firefox
55 // 3.5 for Windows, any reference to window.sessionStorage raises
56 // an "Operation is not supported" exception. Ignore any exceptions
57 // raised by simply accessing global properties.
58 ref = undefined;
59 }
60
61 if (goog.isFunction(ref)) {
62 benchmarks[name] = ref;
63 names.push(name);
64 }
65 }
66 }
67 }
68
69 document.body.appendChild(
70 goog.dom.createTextNode(
71 'Running ' + names.length + ' benchmarks ' + times + ' times each.'));
72 document.body.appendChild(goog.dom.createElement(goog.dom.TagName.BR));
73
74 names.sort();
75
76 // Build a table and timer.
77 var performanceTimer = new goog.testing.PerformanceTimer(times);
78 performanceTimer.setDiscardOutliers(true);
79
80 var performanceTable = new goog.testing.PerformanceTable(document.body,
81 performanceTimer, 2);
82
83 // Next, run the benchmarks.
84 for (var i = 0; i < names.length; i++) {
85 performanceTable.run(benchmarks[names[i]], names[i]);
86 }
87};
88
89
90/**
91 * Onload handler that runs the benchmarks.
92 * @param {Event} e The event object.
93 */
94window.onload = function(e) {
95 goog.testing.benchmark.run_();
96};
