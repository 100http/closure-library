30/**
31 * Uses document.write to add an iFrame to the page with the reference path in
32 * the src attribute. Used for loading an html file containing reference
33 * structures to test against into the page. Should be called within the body of
34 * the jsunit test page.
35 * @param {string} referencePath A path to a reference HTML file.
36 */
37goog.testing.ui.style.writeReferenceFrame = function(referencePath) {
38 document.write('<iframe id="reference" name="reference" ' +
39 'src="' + referencePath + '"></iframe>');
40};

43/**
44 * Returns a reference to the first element child of a node with the given id
45 * from the page loaded into the reference iFrame. Used to retrieve a particular
46 * reference DOM structure to test against.
47 * @param {string} referenceId The id of a container element for a reference
48 * structure in the reference page.
49 * @return {Node} The root element of the reference structure.
50 */
51goog.testing.ui.style.getReferenceNode = function(referenceId) {
52 return goog.dom.getFirstElementChild(
53 window.frames['reference'].document.getElementById(referenceId));
54};

56
57/**
58 * Returns an array of all element children of a given node.
59 * @param {Node} element The node to get element children of.
60 * @return {!Array<!Node>} An array of all the element children.
61 */
62goog.testing.ui.style.getElementChildren = function(element) {
63 var first = goog.dom.getFirstElementChild(element);
64 if (!first) {
65 return [];
66 }
67 var children = [first], next;
68 while (next = goog.dom.getNextElementSibling(children[children.length - 1])) {
69 children.push(next);
70 }
71 return children;
72};

75/**
76 * Tests whether a given node is a "content" node of a reference structure,
77 * which means it is allowed to have arbitrary children.
78 * @param {Node} element The node to test.
79 * @return {boolean} Whether the given node is a content node or not.
80 */
81goog.testing.ui.style.isContentNode = function(element) {
82 return element.className.indexOf('content') != -1;
83};

86/**
87 * Tests that the structure, node names, and classes of the given element are
88 * the same as the reference structure with the given id. Throws an error if the
89 * element doesn't have the same nodes at each level of the DOM with the same
90 * classes on each. The test ignores all DOM structure within content nodes.
91 * @param {Node} element The root node of the DOM structure to test.
92 * @param {string} referenceId The id of the container for the reference
93 * structure to test against.
94 */
95goog.testing.ui.style.assertStructureMatchesReference = function(element,
96 referenceId) {
97 goog.testing.ui.style.assertStructureMatchesReferenceInner_(element,
98 goog.testing.ui.style.getReferenceNode(referenceId));
99};

102/**
103 * A recursive function for comparing structure, node names, and classes between
104 * a test and reference DOM structure. Throws an error if one of these things
105 * doesn't match. Used internally by
106 * {@link goog.testing.ui.style.assertStructureMatchesReference}.
107 * @param {Node} element DOM element to test.
108 * @param {Node} reference DOM element to use as a reference (test against).
109 * @private
110 */
111goog.testing.ui.style.assertStructureMatchesReferenceInner_ = function(element,
112 reference) {
113 if (!element && !reference) {
114 return;
115 }
116 assertTrue('Expected two elements.', !!element && !!reference);
117 assertEquals('Expected nodes to have the same nodeName.',
118 element.nodeName, reference.nodeName);
119 var testElem = goog.asserts.assertElement(element);
120 var refElem = goog.asserts.assertElement(reference);
121 var elementClasses = goog.dom.classlist.get(testElem);
122 goog.array.forEach(goog.dom.classlist.get(refElem), function(referenceClass) {
123 assertContains('Expected test node to have all reference classes.',
124 referenceClass, elementClasses);
125 });
126 // Call assertStructureMatchesReferenceInner_ on all element children
127 // unless this is a content node
128 var elChildren = goog.testing.ui.style.getElementChildren(element),
129 refChildren = goog.testing.ui.style.getElementChildren(reference);
130 if (!goog.testing.ui.style.isContentNode(reference)) {
131 if (elChildren.length != refChildren.length) {
132 assertEquals('Expected same number of children for a non-content node.',
133 elChildren.length, refChildren.length);
134 }
135 for (var i = 0; i < elChildren.length; i++) {
136 goog.testing.ui.style.assertStructureMatchesReferenceInner_(elChildren[i],
137 refChildren[i]);
138 }
139 }
