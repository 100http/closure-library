32/**
33 * A UI widget that runs performance tests and displays the results.
34 * @param {Element} root The element where the table should be attached.
35 * @param {goog.testing.PerformanceTimer=} opt_timer A timer to use for
36 * executing functions and profiling them.
37 * @param {number=} opt_precision Number of digits of precision to include in
38 * results. Defaults to 0.
39 * @param {number=} opt_numSamples The number of samples to take. Defaults to 5.
40 * @constructor
41 * @final
42 */
43goog.testing.PerformanceTable = function(
44 root, opt_timer, opt_precision, opt_numSamples) {
45 /**
46 * Where the table should be attached.
47 * @private {Element}
48 */
49 this.root_ = root;
50
51 /**
52 * Number of digits of precision to include in results.
53 * Defaults to 0.
54 * @private {number}
55 */
56 this.precision_ = opt_precision || 0;
57
58 var timer = opt_timer;
59 if (!timer) {
60 timer = new goog.testing.PerformanceTimer();
61 timer.setNumSamples(opt_numSamples || 5);
62 timer.setDiscardOutliers(true);
63 }
64
65 /**
66 * A timer for running the tests.
67 * @private {goog.testing.PerformanceTimer}
68 */
69 this.timer_ = timer;
70
71 this.initRoot_();
72};
73
74
75/**
76 * @return {goog.testing.PerformanceTimer} The timer being used.
77 */
78goog.testing.PerformanceTable.prototype.getTimer = function() {
79 return this.timer_;
80};

83/**
84 * Render the initial table.
85 * @private
86 */
87goog.testing.PerformanceTable.prototype.initRoot_ = function() {
88 this.root_.innerHTML =
89 '<table class="test-results" cellspacing="1">' +
90 ' <thead>' +
91 ' <tr>' +
92 ' <th rowspan="2">Test Description</th>' +
93 ' <th rowspan="2">Runs</th>' +
94 ' <th colspan="4">Results (ms)</th>' +
95 ' </tr>' +
96 ' <tr>' +
97 ' <th>Average</th>' +
98 ' <th>Std Dev</th>' +
99 ' <th>Minimum</th>' +
100 ' <th>Maximum</th>' +
101 ' </tr>' +
102 ' </thead>' +
103 ' <tbody>' +
104 ' </tbody>' +
105 '</table>';
106};

118/**
119 * Round to the specified precision.
120 * @param {number} num The number to round.
121 * @return {string} The rounded number, as a string.
122 * @private
123 */
124goog.testing.PerformanceTable.prototype.round_ = function(num) {
125 var factor = Math.pow(10, this.precision_);
126 return String(Math.round(num * factor) / factor);
127};

130/**

142/**
143 * Run the given task with the performance timer, and show the results.
144 * @param {goog.testing.PerformanceTimer.Task} task The performance timer task
145 * to run.
146 * @param {string=} opt_desc A description to associate with this run.
147 */
148goog.testing.PerformanceTable.prototype.runTask = function(task, opt_desc) {
149 var results = this.timer_.runTask(task);
150 this.recordResults(results, opt_desc);
151};

154/**
155 * Record a performance timer results object to the performance table. See
156 * {@code goog.testing.PerformanceTimer} for details of the format of this
157 * object.
158 * @param {Object} results The performance timer results object.
159 * @param {string=} opt_desc A description to associate with these results.
160 */
161goog.testing.PerformanceTable.prototype.recordResults = function(
162 results, opt_desc) {
163 var average = results['average'];
164 var standardDeviation = results['standardDeviation'];
165 var isSuspicious = average < 0 || standardDeviation > average * .5;
166 var resultsRow = goog.dom.createDom(goog.dom.TagName.TR, null,
167 goog.dom.createDom(goog.dom.TagName.TD, 'test-description',
168 opt_desc || 'No description'),
169 goog.dom.createDom(goog.dom.TagName.TD, 'test-count',
170 String(results['count'])),
171 goog.dom.createDom(goog.dom.TagName.TD, 'test-average',
172 this.round_(average)),
173 goog.dom.createDom(goog.dom.TagName.TD, 'test-standard-deviation',
174 this.round_(standardDeviation)),
175 goog.dom.createDom(goog.dom.TagName.TD, 'test-minimum',
176 String(results['minimum'])),
177 goog.dom.createDom(goog.dom.TagName.TD, 'test-maximum',
178 String(results['maximum'])));
179 if (isSuspicious) {
180 resultsRow.className = 'test-suspicious';
181 }
182 this.getTableBody_().appendChild(resultsRow);
183};

186/**
187 * Report an error in the table.
188 * @param {*} reason The reason for the error.
189 */
190goog.testing.PerformanceTable.prototype.reportError = function(reason) {
191 this.getTableBody_().appendChild(
192 goog.dom.createDom(goog.dom.TagName.TR, null,
193 goog.dom.createDom(goog.dom.TagName.TD,
194 {'class': 'test-error', 'colSpan': 5},
195 String(reason))));
196};
