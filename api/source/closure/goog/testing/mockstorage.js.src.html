15/**
16 * @fileoverview Provides a JS storage class implementing the HTML5 Storage
17 * interface.
18 */
19
20
21goog.require('goog.structs.Map');
22
23
24goog.provide('goog.testing.MockStorage');
25
26
27
28/**
29 * A JS storage instance, implementing the HTML5 Storage interface.
30 * See http://www.w3.org/TR/webstorage/ for details.
31 *
32 * @constructor
33 * @implements {Storage}
34 * @final
35 */
36goog.testing.MockStorage = function() {
37 /**
38 * The underlying storage object.
39 * @type {goog.structs.Map}
40 * @private
41 */
42 this.store_ = new goog.structs.Map();
43
44 /**
45 * The number of elements in the storage.
46 * @type {number}
47 */
48 this.length = 0;
49};
50
51
52/**
53 * Sets an item to the storage.
54 * @param {string} key Storage key.
55 * @param {*} value Storage value. Must be convertible to string.
56 * @override
57 */
58goog.testing.MockStorage.prototype.setItem = function(key, value) {
59 this.store_.set(key, String(value));
60 this.length = this.store_.getCount();
61};
62
63
64/**
65 * Gets an item from the storage. The item returned is the "structured clone"
66 * of the value from setItem. In practice this means it's the value cast to a
67 * string.
68 * @param {string} key Storage key.
69 * @return {?string} Storage value for key; null if does not exist.
70 * @override
71 */
72goog.testing.MockStorage.prototype.getItem = function(key) {
73 var val = this.store_.get(key);
74 // Enforce that getItem returns string values.
75 return (val != null) ? /** @type {string} */ (val) : null;
76};
77
78
79/**
80 * Removes and item from the storage.
81 * @param {string} key Storage key.
82 * @override
83 */
84goog.testing.MockStorage.prototype.removeItem = function(key) {
85 this.store_.remove(key);
86 this.length = this.store_.getCount();
87};
88
89
90/**
91 * Clears the storage.
92 * @override
93 */
94goog.testing.MockStorage.prototype.clear = function() {
95 this.store_.clear();
96 this.length = 0;
97};
98
99
100/**
101 * Returns the key at the given index.
102 * @param {number} index The index for the key.
103 * @return {?string} Key at the given index, null if not found.
104 * @override
105 */
106goog.testing.MockStorage.prototype.key = function(index) {
107 return this.store_.getKeys()[index] || null;
108};
