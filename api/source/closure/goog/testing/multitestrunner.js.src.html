16 * @fileoverview Utility for running multiple test files that utilize the same
17 * interface as goog.testing.TestRunner. Each test is run in series and their
18 * results aggregated. The main usecase for the MultiTestRunner is to allow
19 * the testing of all tests in a project locally.
20 *
21 */

635/**
636 * Starts executing the tests.
637 */
638goog.testing.MultiTestRunner.prototype.start = function() {
639 this.startButtonEl_.disabled = true;
640 this.stopButtonEl_.disabled = false;
641 this.stopped_ = false;
642 this.active_ = true;
643 this.finished_ = {};
644 this.activeTests_ = this.getTestsToRun();
645 this.startedCount_ = 0;
646 this.resultCount_ = 0;
647 this.passes_ = 0;
648 this.stats_ = [];
649 this.startTime_ = goog.now();
650
651 this.resetProgressDom_();
652 goog.dom.removeChildren(this.logEl_);
653
654 this.resetReport_();
655 this.clearStats_();
656 this.showTab_(0);
657
658 // Ensure the pool isn't too big.
659 while (this.getChildCount() > this.poolSize_) {
660 this.removeChildAt(0, true).dispose();
661 }
662
663 // Start a test in each runner.
664 for (var i = 0; i < this.poolSize_; i++) {
665 if (i >= this.getChildCount()) {
666 var testFrame = new goog.testing.MultiTestRunner.TestFrame(
667 this.basePath_, this.timeoutMs_, this.verbosePasses_, this.dom_);
668 this.addChild(testFrame, true);
669 }
670 this.runNextTest_(
671 /** @type {goog.testing.MultiTestRunner.TestFrame} */
672 (this.getChildAt(i)));
673 }
674};

677/**
678 * Logs a message to the log window.
679 * @param {string} msg A message to log.
680 */
681goog.testing.MultiTestRunner.prototype.log = function(msg) {
682 if (msg != '.') {
683 msg = this.getTimeStamp_() + ' : ' + msg;
684 }
685
686 this.logEl_.appendChild(this.dom_.createDom(goog.dom.TagName.DIV, null, msg));
687
688 // Autoscroll if we're near the bottom.
689 var top = this.logEl_.scrollTop;
690 var height = /** @type {!HTMLElement} */ (this.logEl_).scrollHeight -
691 /** @type {!HTMLElement} */ (this.logEl_).offsetHeight;
692 if (top == 0 || top > height - 50) {
693 this.logEl_.scrollTop = height;
694 }

698/**
699 * Processes a result returned from a TestFrame. If there are tests remaining
700 * it will trigger the next one to be run, otherwise if there are no tests and
701 * all results have been recieved then it will call finish.
702 * @param {goog.testing.MultiTestRunner.TestFrame} frame The frame that just
703 * finished.
704 */
705goog.testing.MultiTestRunner.prototype.processResult = function(frame) {
706 var success = frame.isSuccess();
707 var report = frame.getReport();
708 var test = frame.getTestFile();
709
710 this.stats_.push(frame.getStats());
711 this.finished_[test] = true;
712
713 var prefix = success ? '' : '*** FAILURE *** ';
714 this.log(prefix +
715 this.trimFileName_(test) + ' : ' + (success ? 'Passed' : 'Failed'));
716
717 this.resultCount_++;
718
719 if (success) {
720 this.passes_++;
721 }
722
723 this.drawProgressSegment_(test, success);
724 this.writeCurrentSummary_();
725 if (!(success && this.hidePasses_)) {
726 this.drawTestResult_(test, success, report);
727 }
728
729 if (!this.stopped_ && this.startedCount_ < this.activeTests_.length) {
730 this.runNextTest_(frame);
731 } else if (this.resultCount_ == this.activeTests_.length) {
732 this.finish_();
733 }
734};

737/**
738 * Runs the next available test, if there are any left.
739 * @param {goog.testing.MultiTestRunner.TestFrame} frame Where to run the test.
740 * @private
741 */
742goog.testing.MultiTestRunner.prototype.runNextTest_ = function(frame) {
743 if (this.startedCount_ < this.activeTests_.length) {
744 var nextTest = this.activeTests_[this.startedCount_++];
745 this.log(this.trimFileName_(nextTest) + ' : Loading');
746 frame.runTest(nextTest);
747 }
748};

751/**
752 * Handles the test finishing, processing the results and rendering the report.
753 * @private
754 */
755goog.testing.MultiTestRunner.prototype.finish_ = function() {
756 if (this.stopped_) {
757 this.log('Stopped');
758 } else {
759 this.log('Finished');
760 }
761
762 this.startButtonEl_.disabled = false;
763 this.stopButtonEl_.disabled = true;
764 this.active_ = false;
765
766 this.showTab_(1);
767 this.drawStats_();
768
769 // Remove all the test frames
770 while (this.getChildCount() > 0) {
771 this.removeChildAt(0, true).dispose();
772 }
773
774 // Compute tests that did not finish before the stop button was hit.
775 var unfinished = [];
776 for (var i = 0; i < this.activeTests_.length; i++) {
777 var test = this.activeTests_[i];
778 if (!this.finished_[test]) {
779 unfinished.push(test);
780 }
781 }
782
783 if (unfinished.length) {
784 this.reportEl_.appendChild(goog.dom.createDom(
785 goog.dom.TagName.PRE, undefined,
786 'Theses tests did not finish:\n' + unfinished.join('\n')));
787 }
788};

804/**
805 * Draws the stats for the test run.
806 * @private
807 */
808goog.testing.MultiTestRunner.prototype.drawStats_ = function() {
809 this.drawFilesHistogram_();
810

849/**
850 * Draws a stats histogram.
851 * @param {string} statsField Field of the stats object to graph.
852 * @param {number} bucketSize The size for the histogram's buckets.
853 * @param {function(number, ...*): *} valueTransformFn Function for
854 * transforming the x-labels value for display.
855 * @param {number} width The width in pixels of the graph.
856 * @param {string} title The graph's title.
857 * @private
858 */
859goog.testing.MultiTestRunner.prototype.drawStatsHistogram_ = function(
860 statsField, bucketSize, valueTransformFn, width, title) {
861
862 var hist = {}, data = [], xlabels = [], ylabels = [];
863 var max = 0;
864 for (var i = 0; i < this.stats_.length; i++) {
865 var num = this.stats_[i][statsField];
866 var bucket = Math.floor(num / bucketSize) * bucketSize;
867 if (bucket > max) {
868 max = bucket;
869 }
870 if (!hist[bucket]) {
871 hist[bucket] = 1;
872 } else {
873 hist[bucket]++;
874 }
875 }
876 var maxBucketSize = 0;
877 for (var i = 0; i <= max; i += bucketSize) {
878 xlabels.push(valueTransformFn(i));
879 var count = hist[i] || 0;
880 if (count > maxBucketSize) {
881 maxBucketSize = count;
882 }
883 data.push(count);
884 }
885 var diff = Math.max(1, Math.ceil(maxBucketSize / 10));
886 for (var i = 0; i <= maxBucketSize; i += diff) {
887 ylabels.push(i);
888 }
889 var chart = new goog.ui.ServerChart(
890 goog.ui.ServerChart.ChartType.VERTICAL_STACKED_BAR, width, 250, null,
891 goog.ui.ServerChart.CHART_SERVER_HTTPS_URI);
892 chart.setTitle(title);
893 chart.addDataSet(data, 'ff9900');
894 chart.setLeftLabels(ylabels);
895 chart.setGridY(ylabels.length - 1);
896 chart.setXLabels(xlabels);
897 chart.render(this.statsEl_);

1410/**
1411 * Checks the active test for completion, keeping track of the tests' various
1412 * execution stages.
1413 * @private
1414 */
1415goog.testing.MultiTestRunner.TestFrame.prototype.checkForCompletion_ =
1416 function() {
1417 var js = goog.dom.getFrameContentWindow(this.iframeEl_);
1418 switch (this.currentState_) {
1419 case 0:
1420 if (this.iframeLoaded_ && js['G_testRunner']) {
1421 this.lastStateTime_ = goog.now();
1422 this.currentState_++;
1423 }
1424 break;
1425 case 1:
1426 if (js['G_testRunner']['isInitialized']()) {
1427 this.lastStateTime_ = goog.now();
1428 this.currentState_++;
1429 }
1430 break;
1431 case 2:
1432 if (js['G_testRunner']['isFinished']()) {
1433 var tr = js['G_testRunner'];
1434 this.isSuccess_ = tr['isSuccess']();
1435 this.report_ = tr['getReport'](this.verbosePasses_);
1436 this.runTime_ = tr['getRunTime']();
1437 this.numFilesLoaded_ = tr['getNumFilesLoaded']();
1438 this.finish_();
1439 return;
1440 }
1441 }
1442
1443 // Check to see if the test has timed out.
1444 if (goog.now() - this.lastStateTime_ > this.timeoutMs_) {
1445 this.report_ = this.testFile_ + ' timed out ' +
1446 goog.testing.MultiTestRunner.STATES[this.currentState_];
1447 this.isSuccess_ = false;
1448 this.finish_();
1449 return;
1450 }
1451
1452 // Check again in 100ms.
1453 goog.Timer.callOnce(this.checkForCompletion_, 100, this);
