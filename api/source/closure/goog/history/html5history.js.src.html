36/**
37 * An implementation compatible with goog.History that uses the HTML5
38 * history APIs.
39 *
40 * @param {Window=} opt_win The window to listen/dispatch history events on.
41 * @param {goog.history.Html5History.TokenTransformer=} opt_transformer
42 * The token transformer that is used to create URL from the token
43 * when storing token without using hash fragment.
44 * @constructor
45 * @extends {goog.events.EventTarget}
46 * @final
47 */
48goog.history.Html5History = function(opt_win, opt_transformer) {
49 goog.events.EventTarget.call(this);
50 goog.asserts.assert(goog.history.Html5History.isSupported(opt_win),
51 'HTML5 history is not supported.');
52
53 /**
54 * The window object to use for history tokens. Typically the top window.
55 * @type {Window}
56 * @private
57 */
58 this.window_ = opt_win || window;
59
60 /**
61 * The token transformer that is used to create URL from the token
62 * when storing token without using hash fragment.
63 * @type {goog.history.Html5History.TokenTransformer}
64 * @private
65 */
66 this.transformer_ = opt_transformer || null;
67
68 goog.events.listen(this.window_, goog.events.EventType.POPSTATE,
69 this.onHistoryEvent_, false, this);
70 goog.events.listen(this.window_, goog.events.EventType.HASHCHANGE,
71 this.onHistoryEvent_, false, this);
72};
73goog.inherits(goog.history.Html5History, goog.events.EventTarget);

76/**
77 * Returns whether Html5History is supported.
78 * @param {Window=} opt_win Optional window to check.
79 * @return {boolean} Whether html5 history is supported.
80 */
81goog.history.Html5History.isSupported = function(opt_win) {
82 var win = opt_win || window;
83 return !!(win.history && win.history.pushState);
84};

112/**
113 * Starts or stops the History. When enabled, the History object
114 * will immediately fire an event for the current location. The caller can set
115 * up event listeners between the call to the constructor and the call to
116 * setEnabled.
117 *
118 * @param {boolean} enable Whether to enable history.
119 */
120goog.history.Html5History.prototype.setEnabled = function(enable) {
121 if (enable == this.enabled_) {
122 return;
123 }
124
125 this.enabled_ = enable;
126
127 if (enable) {
128 this.dispatchEvent(new goog.history.Event(this.getToken(), false));
129 }
130};

133/**
134 * Returns the current token.
135 * @return {string} The current token.
136 */
137goog.history.Html5History.prototype.getToken = function() {
138 if (this.useFragment_) {
139 var loc = this.window_.location.href;
140 var index = loc.indexOf('#');
141 return index < 0 ? '' : loc.substring(index + 1);
142 } else {
143 return this.transformer_ ?
144 this.transformer_.retrieveToken(
145 this.pathPrefix_, this.window_.location) :
146 this.window_.location.pathname.substr(this.pathPrefix_.length);
147 }
148};

151/**
152 * Sets the history state.
153 * @param {string} token The history state identifier.
154 * @param {string=} opt_title Optional title to associate with history entry.
155 */
156goog.history.Html5History.prototype.setToken = function(token, opt_title) {
157 if (token == this.getToken()) {
158 return;
159 }
160
161 // Per externs/gecko_dom.js document.title can be null.
162 this.window_.history.pushState(null,
163 opt_title || this.window_.document.title || '', this.getUrl_(token));
164 this.dispatchEvent(new goog.history.Event(token, false));
165};

168/**
169 * Replaces the current history state without affecting the rest of the history
170 * stack.
171 * @param {string} token The history state identifier.
172 * @param {string=} opt_title Optional title to associate with history entry.
173 */
174goog.history.Html5History.prototype.replaceToken = function(token, opt_title) {
175 // Per externs/gecko_dom.js document.title can be null.
176 this.window_.history.replaceState(null,
177 opt_title || this.window_.document.title || '', this.getUrl_(token));
178 this.dispatchEvent(new goog.history.Event(token, false));
179};

193/**
194 * Sets whether to use the fragment to store tokens.
195 * @param {boolean} useFragment Whether to use the fragment.
196 */
197goog.history.Html5History.prototype.setUseFragment = function(useFragment) {
198 if (this.useFragment_ != useFragment) {
199 if (useFragment) {
200 goog.events.listen(this.window_, goog.events.EventType.HASHCHANGE,
201 this.onHistoryEvent_, false, this);
202 } else {
203 goog.events.unlisten(this.window_, goog.events.EventType.HASHCHANGE,
204 this.onHistoryEvent_, false, this);
205 }
206 this.useFragment_ = useFragment;
207 }
208};

230/**
231 * Gets the URL to set when calling history.pushState
232 * @param {string} token The history token.
233 * @return {string} The URL.
234 * @private
235 */
236goog.history.Html5History.prototype.getUrl_ = function(token) {
237 if (this.useFragment_) {
238 return '#' + token;
239 } else {
240 return this.transformer_ ?
241 this.transformer_.createUrl(
242 token, this.pathPrefix_, this.window_.location) :
243 this.pathPrefix_ + token + this.window_.location.search;
244 }
245};
