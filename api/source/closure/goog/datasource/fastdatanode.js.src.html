15/**
16 * @fileoverview
17 * Efficient implementation of DataNode API.
18 *
19 * The implementation consists of three concrete classes for modelling
20 * DataNodes with different characteristics: FastDataNode,
21 * FastPrimitiveDataNode and FastListNode.
22 *
23 * FastDataNode is for bean-like or map-like objects that consists of
24 * key/value mappings and where the primary access pattern is by key.
25 *
26 * FastPrimitiveDataNode wraps primitives like strings, boolean, and numbers.
27 *
28 * FastListNode is for array-like data nodes. It also supports key-based
29 * lookups if the data nodes have an "id" property or if child nodes are
30 * explicitly added by name. It is most efficient if these features are not
31 * used.
32 *
33 * FastDataNodes can be constructed from JSON-like objects via the function
34 * goog.ds.FastDataNode.fromJs.
35
36 */
37

48/*
49 * Implementation note: In order to reduce the number of objects,
50 * FastDataNode stores its key/value mappings directly in the FastDataNode
51 * object iself (instead of a separate map). To make this work we have to
52 * sure that there are no name clashes with other attribute names used by
53 * FastDataNode (like dataName and parent). This is especially difficult in
54 * the light of automatic renaming by the JavaScript compiler. For this reason,
55 * all internal attributes start with "__" so that they are not renamed
56 * by the compiler.
57 */

59/**
60 * Creates a new abstract data node.
61 * @param {string} dataName Name of the datanode.
62 * @param {goog.ds.DataNode=} opt_parent Parent of this data node.
63 * @constructor
64 * @extends {goog.ds.DataNodeList}
65 */
66// TODO(arv): Use interfaces when available.
67goog.ds.AbstractFastDataNode = function(dataName, opt_parent) {
68 if (!dataName) {
69 throw Error('Cannot create a fast data node without a data name');
70 }
71 this['__dataName'] = dataName;
72 this['__parent'] = opt_parent;
73};

96/**
97 * Get the path leading to this data node.
98 * @return {string} Data path.
99 * @override
100 */
101goog.ds.AbstractFastDataNode.prototype.getDataPath = function() {
102 var parentPath;
103 if (this['__parent']) {
104 parentPath = this['__parent'].getDataPath() + goog.ds.STR_PATH_SEPARATOR;
105 } else {
106 parentPath = '';
107 }
108 return parentPath + this.getDataName();
109};

140/**
141 * Creates a new FastDataNode structure initialized from object. This will
142 * return an instance of the most suitable sub-class of FastDataNode.
143 *
144 * You should not modify object after creating a fast data node from it
145 * or assume that changing object changes the data node. Doing so results
146 * in undefined behaviour.
147 *
148 * @param {Object|number|boolean|string} object Object to initialize data
149 * node from.
150 * @param {string} dataName Name of data node.
151 * @param {goog.ds.DataNode=} opt_parent Parent of data node.
152 * @return {!goog.ds.AbstractFastDataNode} Data node representing object.
153 */
154goog.ds.FastDataNode.fromJs = function(object, dataName, opt_parent) {
155 if (goog.isArray(object)) {
156 return new goog.ds.FastListNode(object, dataName, opt_parent);
157 } else if (goog.isObject(object)) {
158 return new goog.ds.FastDataNode(object, dataName, opt_parent);
159 } else {
160 return new goog.ds.PrimitiveFastDataNode(object || !!object,
161 dataName,
162 opt_parent);
163 }
164};

185/** @override */
186goog.ds.FastDataNode.prototype.getChildNodes = function(opt_selector) {
187 if (!opt_selector || opt_selector == goog.ds.STR_ALL_CHILDREN_SELECTOR) {
188 return this;
189 } else if (opt_selector.indexOf(goog.ds.STR_WILDCARD) == -1) {
190 var child = this.getChildNode(opt_selector);
191 return child ? new goog.ds.FastListNode([child], '') :
192 new goog.ds.EmptyNodeList();
193 } else {
194 throw Error('Unsupported selector: ' + opt_selector);
195 }
196};

199/**
200 * Makes sure that a named child is wrapped in a data node structure.
201 * @param {string} name Name of child to wrap.
202 * @private
203 */
204goog.ds.FastDataNode.prototype.wrapChild_ = function(name) {
205 var child = this[name];
206 if (child != null && !child.getDataName) {
207 this[name] = goog.ds.FastDataNode.fromJs(this[name], name, this);
208 }
209};

212/**
213 * Get a child node by name.
214 * @param {string} name Name of child node.
215 * @param {boolean=} opt_create Whether to create the child if it does not
216 * exist.
217 * @return {goog.ds.DataNode} Child node.
218 * @override
219 */
220goog.ds.FastDataNode.prototype.getChildNode = function(name, opt_create) {
221 this.wrapChild_(name);
222 // this[name] always is a data node object, so using "||" is fine.
223 var child = this[name] || null;
224 if (child == null && opt_create) {
225 child = new goog.ds.FastDataNode({}, name, this);
226 this[name] = child;
227 }
228 return child;
229};

233 * Sets a child node. Creates the child if it does not exist.
234 *
235 * Calling this function makes any child nodes previously obtained for name
236 * invalid. You should not use these child nodes but instead obtain a new
237 * instance by calling getChildNode.

239 * @override

241goog.ds.FastDataNode.prototype.setChildNode = function(name, value) {
242 if (value != null) {

283 * not modify the object returned by this function.
284 * @return {!Object} Javascript object representation of this data node.
285 */
286goog.ds.FastDataNode.prototype.getJsObject = function() {
287 var result = {};
288 for (var key in this) {
289 if (!goog.string.startsWith(key, '__') && !goog.isFunction(this[key])) {
290 result[key] = (this[key]['__dataName'] ? this[key].getJsObject() :
291 this[key]);
292 }
293 }
294 return result;

340/**
341 * Gets a child node by index. This method has a complexity of O(n) where
342 * n is the number of children. If you need a faster implementation of this
343 * method, you should use goog.ds.FastListNode.
344 * @param {number} index Index of child node (starting from 0).
345 * @return {goog.ds.DataNode} Child node at specified index.
346 * @override
347 */
348goog.ds.FastDataNode.prototype.getByIndex = function(index) {
349 var i = 0;
350 for (var key in this) {
351 if (!goog.string.startsWith(key, '__') && !goog.isFunction(this[key])) {
352 if (i == index) {
353 this.wrapChild_(key);
354 return this[key];
355 }
356 ++i;
357 }
358 }
359 return null;
360};

363/**
364 * Gets the number of child nodes. This method has a complexity of O(n) where
365 * n is the number of children. If you need a faster implementation of this
366 * method, you should use goog.ds.FastListNode.
367 * @return {number} Number of child nodes.
368 * @override
369 */
370goog.ds.FastDataNode.prototype.getCount = function() {
371 var count = 0;
372 for (var key in this) {
373 if (!goog.string.startsWith(key, '__') && !goog.isFunction(this[key])) {
374 ++count;
375 }
376 }
377 // maybe cache this?
378 return count;

510/**
511 * Creates a new list node from an array.
512 * @param {Array<?>} values values hold by this list node.
513 * @param {string} dataName name of this node.
514 * @param {goog.ds.DataNode=} opt_parent parent of this node.
515 * @extends {goog.ds.AbstractFastDataNode}
516 * @constructor
517 * @final
518 */
519// TODO(arv): Use interfaces when available. This implements DataNodeList
520// as well.
521goog.ds.FastListNode = function(values, dataName, opt_parent) {
522 this.values_ = [];
523 for (var i = 0; i < values.length; ++i) {
524 var name = values[i].id || ('[' + i + ']');
525 this.values_.push(goog.ds.FastDataNode.fromJs(values[i], name, this));
526 if (values[i].id) {
527 if (!this.map_) {
528 this.map_ = {};
529 }
530 this.map_[values[i].id] = i;
531 }
532 }
533 goog.ds.AbstractFastDataNode.call(this, dataName, opt_parent);
534};
535goog.inherits(goog.ds.FastListNode, goog.ds.AbstractFastDataNode);

567goog.ds.FastListNode.prototype.getChildNode = function(key, opt_create) {

595/**
596 * Tries to interpret key as a numeric index enclosed by square brakcets.
597 * @param {string} key Key that should be interpreted as a number.
598 * @return {?number} Numeric index or null if key is not of the form
599 * described above.
600 * @private
601 */
602goog.ds.FastListNode.prototype.getKeyAsNumber_ = function(key) {
603 if (key.charAt(0) == '[' && key.charAt(key.length - 1) == ']') {
604 return Number(key.substring(1, key.length - 1));
605 } else {
606 return null;
607 }

611/**
612 * Sets a child node. Creates the child if it does not exist. To set
613 * children at a certain index, use a key of the form '[index]'. Note, that
614 * you can only set values at existing numeric indices. To add a new node
615 * to this list, you have to use the add method.
616 *
617 * Calling this function makes any child nodes previously obtained for name
618 * invalid. You should not use these child nodes but instead obtain a new
619 * instance by calling getChildNode.
620 *
621 * @override
622 */
623goog.ds.FastListNode.prototype.setChildNode = function(key, value) {
624 var count = this.values_.length;
625 if (value != null) {
626 if (!value.getDataName) {
627 value = goog.ds.FastDataNode.fromJs(value, key, this);
628 }
629 var index = this.getKeyAsNumber_(key);
630 if (index != null) {
631 if (index < 0 || index >= this.values_.length) {
632 throw Error('List index out of bounds: ' + index);
633 }
634 this.values_[key] = value;
635 } else {
636 if (!this.map_) {
637 this.map_ = {};
638 }
639 this.values_.push(value);
640 this.map_[key] = this.values_.length - 1;
641 }
642 } else {
643 this.removeNode(key);
644 }
645 var dm = goog.ds.DataManager.getInstance();
646 dm.fireDataChange(this.getDataPath() + goog.ds.STR_PATH_SEPARATOR + key);
647 if (this.values_.length != count) {
648 this.listSizeChanged_();
649 }
650 return null;
651};

765/**
766 * Removes a child node.
767 * @override
768 */
769goog.ds.FastListNode.prototype.removeNode = function(name) {
770 var index = this.getKeyAsNumber_(name);
771 if (index == null && this.map_) {
772 index = this.map_[name];
773 }
774 if (index != null) {
775 this.values_.splice(index, 1);
776 if (this.map_) {
777 var keyToDelete = null;
778 for (var key in this.map_) {
779 if (this.map_[key] == index) {
780 keyToDelete = key;
781 } else if (this.map_[key] > index) {
782 --this.map_[key];
783 }
784 }
