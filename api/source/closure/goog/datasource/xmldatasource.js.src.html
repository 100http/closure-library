37/**
38 * Data source whose backing is an xml node
39 *
40 * @param {Node} node The XML node. Can be null.
41 * @param {goog.ds.XmlDataSource} parent Parent of XML element. Can be null.
42 * @param {string=} opt_name The name of this node relative to the parent node.
43 *
44 * @extends {goog.ds.DataNode}
45 * @constructor
46 */
47// TODO(arv): Use interfaces when available.
48goog.ds.XmlDataSource = function(node, parent, opt_name) {
49 this.parent_ = parent;
50 this.dataName_ = opt_name || (node ? node.nodeName : '');
51 this.setNode_(node);

63/**
64 * Set the current root nodeof the data source.
65 * Can be an attribute node, text node, or element node
66 * @param {Node} node The node. Can be null.
67 *
68 * @private
69 */
70goog.ds.XmlDataSource.prototype.setNode_ = function(node) {
71 this.node_ = node;
72 if (node != null) {
73 switch (node.nodeType) {
74 case goog.dom.NodeType.ATTRIBUTE:
75 case goog.dom.NodeType.TEXT:
76 this.value_ = node.nodeValue;
77 break;
78 case goog.dom.NodeType.ELEMENT:
79 if (node.childNodes.length == 1 &&
80 node.firstChild.nodeType == goog.dom.NodeType.TEXT) {
81 this.value_ = node.firstChild.nodeValue;
82 }
83 }
84 }
85};

88/**
89 * Creates the DataNodeList with the child nodes for this element.
90 * Allows for only building list as needed.
91 *
92 * @private
93 */
94goog.ds.XmlDataSource.prototype.createChildNodes_ = function() {
95 if (this.childNodeList_) {
96 return;
97 }
98 var childNodeList = new goog.ds.BasicNodeList();
99 if (this.node_ != null) {
100 var childNodes = this.node_.childNodes;
101 for (var i = 0, childNode; childNode = childNodes[i]; i++) {
102 if (childNode.nodeType != goog.dom.NodeType.TEXT ||
103 !goog.ds.XmlDataSource.isEmptyTextNodeValue_(childNode.nodeValue)) {
104 var newNode = new goog.ds.XmlDataSource(childNode,
105 this, childNode.nodeName);
106 childNodeList.add(newNode);
107 }
108 }
109 }
110 this.childNodeList_ = childNodeList;
111};

114/**
115 * Creates the DataNodeList with the attributes for the element
116 * Allows for only building list as needed.
117 *
118 * @private
119 */
120goog.ds.XmlDataSource.prototype.createAttributes_ = function() {
121 if (this.attributes_) {
122 return;
123 }
124 var attributes = new goog.ds.BasicNodeList();
125 if (this.node_ != null && this.node_.attributes != null) {
126 var atts = this.node_.attributes;
127 for (var i = 0, att; att = atts[i]; i++) {
128 var newNode = new goog.ds.XmlDataSource(att, this, att.nodeName);
129 attributes.add(newNode);
130 }
131 }
132 this.attributes_ = attributes;
133};

157/** @override */
158goog.ds.XmlDataSource.prototype.getChildNodes = function(opt_selector) {
159 if (opt_selector && opt_selector ==
160 goog.ds.XmlDataSource.ATTRIBUTE_SELECTOR_) {
161 this.createAttributes_();
162 return this.attributes_;
163 } else if (opt_selector == null ||
164 opt_selector == goog.ds.STR_ALL_CHILDREN_SELECTOR) {
165 this.createChildNodes_();
166 return this.childNodeList_;
167 } else {
168 throw Error('Unsupported selector');
169 }

174/**
175 * Gets a named child node of the current node
176 * @param {string} name The node name.
177 * @return {goog.ds.DataNode} The child node, or null if
178 * no node of this name exists.
179 * @override
180 */
181goog.ds.XmlDataSource.prototype.getChildNode = function(name) {
182 if (goog.string.startsWith(name, goog.ds.STR_ATTRIBUTE_START)) {
183 var att = this.node_.getAttributeNode(name.substring(1));
184 return att ? new goog.ds.XmlDataSource(att, this) : null;
185 } else {
186 return /** @type {goog.ds.DataNode} */ (this.getChildNodes().get(name));
187 }
188};

191/**
192 * Gets the value of a child node
193 * @param {string} name The node name.
194 * @return {*} The value of the node, or null if no value or the child node
195 * doesn't exist.
196 * @override
197 */
198goog.ds.XmlDataSource.prototype.getChildNodeValue = function(name) {
199 if (goog.string.startsWith(name, goog.ds.STR_ATTRIBUTE_START)) {
200 var node = this.node_.getAttributeNode(name.substring(1));
201 return node ? node.nodeValue : null;
202 } else {
203 var node = this.getChildNode(name);
204 return node ? node.get() : null;
205 }

235 var parentPath = '';

265/**
266 * Check whether a node is an empty text node. Nodes consisting of only white
267 * space (#x20, #xD, #xA, #x9) can generally be collapsed to a zero length
268 * text string.
269 * @param {string} str String to match.
270 * @return {boolean} True if string equates to empty text node.

274 return /^[\r\n\t ]*$/.test(str);

292/**
293 * Data source whose backing is an XMLHttpRequest,
294 *
295 * A URI of an empty string will mean that no request is made
296 * and the data source will be a single, empty node.
297 *
298 * @param {(string|goog.Uri)} uri URL of the XMLHttpRequest.
299 * @param {string} name Name of the datasource.
300 *
301 * implements goog.ds.XmlHttpDataSource.
302 * @constructor
303 * @extends {goog.ds.XmlDataSource}
304 * @final
305 */
306goog.ds.XmlHttpDataSource = function(uri, name) {
307 goog.ds.XmlDataSource.call(this, null, null, name);
308 if (uri) {
309 this.uri_ = new goog.Uri(uri);
310 } else {
311 this.uri_ = null;
312 }
313};
314goog.inherits(goog.ds.XmlHttpDataSource, goog.ds.XmlDataSource);

324/**
325 * Load or reload the backing data for this node.
326 * Fires the XMLHttpRequest
327 * @override
328 */
329goog.ds.XmlHttpDataSource.prototype.load = function() {
330 if (this.uri_) {
331 goog.log.info(goog.ds.logger, 'Sending XML request for DataSource ' +
332 this.getDataName() + ' to ' + this.uri_);
333 this.loadState_ = goog.ds.LoadState.LOADING;
334
335 goog.net.XhrIo.send(this.uri_, goog.bind(this.complete_, this));
336 } else {
337 this.node_ = goog.ds.XmlDataSource.createChildlessDocument_();
338 this.loadState_ = goog.ds.LoadState.NOT_LOADED;
339 }
340};

353/**
354 * Handles the completion of an XhrIo request. Dispatches to success or load
355 * based on the result.
356 * @param {!goog.events.Event} e The XhrIo event object.
357 * @private
358 */
359goog.ds.XmlHttpDataSource.prototype.complete_ = function(e) {
360 var xhr = /** @type {goog.net.XhrIo} */ (e.target);
361 if (xhr && xhr.isSuccess()) {
362 this.success_(xhr);
363 } else {
364 this.failure_();
365 }

369/**
370 * Success result. Checks whether valid XML was returned
371 * and sets the XML and loadstate.
372 *
373 * @param {!goog.net.XhrIo} xhr The successful XhrIo object.
374 * @private
375 */
376goog.ds.XmlHttpDataSource.prototype.success_ = function(xhr) {
377 goog.log.info(goog.ds.logger,
378 'Got data for DataSource ' + this.getDataName());
379 var xml = xhr.getResponseXml();
380
381 // Fix for case where IE returns valid XML as text but
382 // doesn't parse by default
383 if (xml && !xml.hasChildNodes() &&
384 goog.isObject(xhr.getResponseText())) {
385 xml = goog.dom.xml.loadXml(xhr.getResponseText());
386 }
387 // Failure result
388 if (!xml || !xml.hasChildNodes()) {
389 this.loadState_ = goog.ds.LoadState.FAILED;
390 this.node_ = goog.ds.XmlDataSource.createChildlessDocument_();
391 } else {
392 this.loadState_ = goog.ds.LoadState.LOADED;
393 this.node_ = xml.documentElement;
394 }
395
396 if (this.getDataName()) {
397 goog.ds.DataManager.getInstance().fireDataChange(this.getDataName());
398 }
399};

402/**
403 * Failure result
404 *
405 * @private
406 */
407goog.ds.XmlHttpDataSource.prototype.failure_ = function() {
408 goog.log.info(goog.ds.logger, 'Data retrieve failed for DataSource ' +
409 this.getDataName());
410
411 this.loadState_ = goog.ds.LoadState.FAILED;
412 this.node_ = goog.ds.XmlDataSource.createChildlessDocument_();
413
414 if (this.getDataName()) {
415 goog.ds.DataManager.getInstance().fireDataChange(this.getDataName());
416 }
