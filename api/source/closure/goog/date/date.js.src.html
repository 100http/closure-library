87/**
88 * Regular expression for splitting date parts from ISO 8601 styled string.
89 * Examples: '20060210' or '2005-02-22' or '20050222' or '2005-08'
90 * or '2005-W22' or '2005W22' or '2005-W22-4', etc.
91 * For explanation and more examples, see:
92 * {@link http://en.wikipedia.org/wiki/ISO_8601}
93 *
94 * @type {RegExp}
95 * @private
96 */
97goog.date.splitDateStringRegex_ = new RegExp(
98 '^(\\d{4})(?:(?:-?(\\d{2})(?:-?(\\d{2}))?)|' +
99 '(?:-?(\\d{3}))|(?:-?W(\\d{2})(?:-?([1-7]))?))?$');

104 * Examples: '18:46:39.994' or '184639.994'

106 * @type {RegExp}

108 */
109goog.date.splitTimeStringRegex_ =
110 /^(\d{2})(?::?(\d{2})(?::?(\d{2})(\.\d+)?)?)?$/;

154/**
155 * Returns whether the given year is a long ISO year.
156 * See {@link http://www.phys.uu.nl/~vgent/calendar/isocalendar_text3.htm}.
157 *
158 * @param {number} year Full year part of date.
159 * @return {boolean} Whether the given year is a long ISO year.
160 */
161goog.date.isLongIsoYear = function(year) {
162 var n = 5 * year + 12 - 4 * (Math.floor(year / 100) - Math.floor(year / 400));
163 n += Math.floor((year - 100) / 400) - Math.floor((year - 102) / 400);
164 n += Math.floor((year - 200) / 400) - Math.floor((year - 199) / 400);
165
166 return n % 28 < 5;
167};

172 *
173 * @param {number} year Year part of date.
174 * @param {number} month Month part of date.
175 * @return {number} The number of days for the given month.
176 */
177goog.date.getNumberOfDaysInMonth = function(year, month) {
178 switch (month) {
179 case goog.date.month.FEB:
180 return goog.date.isLeapYear(year) ? 29 : 28;
181 case goog.date.month.JUN:
182 case goog.date.month.SEP:
183 case goog.date.month.NOV:
184 case goog.date.month.APR:
185 return 30;
186 }
187 return 31;

229/**
230 * Static function for week number calculation. ISO 8601 implementation.
231 *
232 * @param {number} year Year part of date.
233 * @param {number} month Month part of date (0-11).
234 * @param {number} date Day part of date (1-31).
235 * @param {number=} opt_weekDay Cut off weekday, defaults to Thursday.
236 * @param {number=} opt_firstDayOfWeek First day of the week, defaults to
237 * Monday.
238 * Monday=0, Sunday=6.
239 * @return {number} The week number (1-53).
240 */
241goog.date.getWeekNumber = function(year, month, date, opt_weekDay,
242 opt_firstDayOfWeek) {
243 var d = new Date(year, month, date);
244
245 // Default to Thursday for cut off as per ISO 8601.
246 var cutoff = opt_weekDay || goog.date.weekDay.THU;
247
248 // Default to Monday for first day of the week as per ISO 8601.
249 var firstday = opt_firstDayOfWeek || goog.date.weekDay.MON;
250
251 // The d.getDay() has to be converted first to ISO weekday (Monday=0).
252 var isoday = (d.getDay() + 6) % 7;
253
254 // Position of given day in the picker grid w.r.t. first day of week
255 var daypos = (isoday - firstday + 7) % 7;
256
257 // Position of cut off day in the picker grid w.r.t. first day of week
258 var cutoffpos = (cutoff - firstday + 7) % 7;
259
260 // Unix timestamp of the midnight of the cutoff day in the week of 'd'.
261 // There might be +-1 hour shift in the result due to the daylight saving,
262 // but it doesn't affect the year.
263 var cutoffSameWeek = d.valueOf() +
264 (cutoffpos - daypos) * goog.date.MS_PER_DAY;
265
266 // Unix timestamp of January 1 in the year of 'cutoffSameWeek'.
267 var jan1 = new Date(new Date(cutoffSameWeek).getFullYear(), 0, 1).valueOf();
268
269 // Number of week. The round() eliminates the effect of daylight saving.
270 return Math.floor(Math.round(
271 (cutoffSameWeek - jan1) / goog.date.MS_PER_DAY) / 7) + 1;
272};

309/**
310 * Parses a datetime string expressed in ISO 8601 format. Overwrites the date
311 * and optionally the time part of the given object with the parsed values.
312 *
313 * @param {!goog.date.DateTime} dateTime Object whose fields will be set.
314 * @param {string} formatted A date or datetime expressed in ISO 8601 format.
315 * @return {boolean} Whether the parsing succeeded.
316 */
317goog.date.setIso8601DateTime = function(dateTime, formatted) {
318 formatted = goog.string.trim(formatted);
319 var delim = formatted.indexOf('T') == -1 ? ' ' : 'T';
320 var parts = formatted.split(delim);
321 return goog.date.setIso8601DateOnly_(dateTime, parts[0]) &&
322 (parts.length < 2 || goog.date.setIso8601TimeOnly_(dateTime, parts[1]));
323};

326/**
327 * Sets date fields based on an ISO 8601 format string.
328 *
329 * @param {!goog.date.DateTime} d Object whose fields will be set.
330 * @param {string} formatted A date expressed in ISO 8601 format.
331 * @return {boolean} Whether the parsing succeeded.
332 * @private
333 */
334goog.date.setIso8601DateOnly_ = function(d, formatted) {
335 // split the formatted ISO date string into its date fields
336 var parts = formatted.match(goog.date.splitDateStringRegex_);
337 if (!parts) {
338 return false;
339 }
340
341 var year = Number(parts[1]);
342 var month = Number(parts[2]);
343 var date = Number(parts[3]);
344 var dayOfYear = Number(parts[4]);
345 var week = Number(parts[5]);
346 // ISO weekdays start with 1, native getDay() values start with 0
347 var dayOfWeek = Number(parts[6]) || 1;
348
349 d.setFullYear(year);
350
351 if (dayOfYear) {
352 d.setDate(1);
353 d.setMonth(0);
354 var offset = dayOfYear - 1; // offset, so 1-indexed, i.e., skip day 1
355 d.add(new goog.date.Interval(goog.date.Interval.DAYS, offset));
356 } else if (week) {
357 goog.date.setDateFromIso8601Week_(d, week, dayOfWeek);
358 } else {
359 if (month) {
360 d.setDate(1);
361 d.setMonth(month - 1);
362 }
363 if (date) {
364 d.setDate(date);
365 }
366 }
367
368 return true;
369};

372/**
373 * Sets date fields based on an ISO 8601 week string.
374 * See {@link http://en.wikipedia.org/wiki/ISO_week_date}, "Relation with the
375 * Gregorian Calendar". The first week of a new ISO year is the week with the
376 * majority of its days in the new Gregorian year. I.e., ISO Week 1's Thursday
377 * is in that year. ISO weeks always start on Monday. So ISO Week 1 can
378 * contain a few days from the previous Gregorian year. And ISO weeks always
379 * end on Sunday, so the last ISO week (Week 52 or 53) can have a few days from
380 * the following Gregorian year.
381 * Example: '1997-W01' lasts from 1996-12-30 to 1997-01-05. January 1, 1997 is
382 * a Wednesday. So W01's Monday is Dec.30, 1996, and Sunday is January 5, 1997.
383 *
384 * @param {goog.date.DateTime} d Object whose fields will be set.
385 * @param {number} week ISO week number.
386 * @param {number} dayOfWeek ISO day of week.
387 * @private
388 */
389goog.date.setDateFromIso8601Week_ = function(d, week, dayOfWeek) {
390 // calculate offset for first week
391 d.setMonth(0);
392 d.setDate(1);
393 var jsDay = d.getDay();
394 // switch Sunday (0) to index 7; ISO days are 1-indexed
395 var jan1WeekDay = jsDay || 7;
396
397 var THURSDAY = 4;
398 if (jan1WeekDay <= THURSDAY) {
399 // was extended back to Monday
400 var startDelta = 1 - jan1WeekDay; // e.g., Thu(4) ==> -3
401 } else {
402 // was extended forward to Monday
403 startDelta = 8 - jan1WeekDay; // e.g., Fri(5) ==> +3
404 }
405
406 // find the absolute number of days to offset from the start of year
407 // to arrive close to the Gregorian equivalent (pending adjustments above)
408 // Note: decrement week multiplier by one because 1st week is
409 // represented by dayOfWeek value
410 var absoluteDays = Number(dayOfWeek) + (7 * (Number(week) - 1));
411
412 // convert from ISO weekday format to Gregorian calendar date
413 // note: subtract 1 because 1-indexed; offset should not include 1st of month
414 var delta = startDelta + absoluteDays - 1;
415 var interval = new goog.date.Interval(goog.date.Interval.DAYS, delta);
416 d.add(interval);
417};

420/**
421 * Sets time fields based on an ISO 8601 format string.
422 * Note: only time fields, not date fields.
423 *
424 * @param {!goog.date.DateTime} d Object whose fields will be set.
425 * @param {string} formatted A time expressed in ISO 8601 format.
426 * @return {boolean} Whether the parsing succeeded.
427 * @private
428 */
429goog.date.setIso8601TimeOnly_ = function(d, formatted) {
430 // first strip timezone info from the end
431 var parts = formatted.match(goog.date.splitTimezoneStringRegex_);
432
433 var offset = 0; // local time if no timezone info
434 if (parts) {
435 if (parts[0] != 'Z') {
436 offset = parts[2] * 60 + Number(parts[3]);
437 offset *= parts[1] == '-' ? 1 : -1;
438 }
439 offset -= d.getTimezoneOffset();
440 formatted = formatted.substr(0, formatted.length - parts[0].length);
441 }
442
443 // then work out the time
444 parts = formatted.match(goog.date.splitTimeStringRegex_);
445 if (!parts) {
446 return false;
447 }
448
449 d.setHours(Number(parts[1]));
450 d.setMinutes(Number(parts[2]) || 0);
451 d.setSeconds(Number(parts[3]) || 0);
452 d.setMilliseconds(parts[4] ? parts[4] * 1000 : 0);
453
454 if (offset != 0) {
455 // adjust the date and time according to the specified timezone
456 d.setTime(d.getTime() + offset * 60000);
457 }
458
459 return true;
460};

464/**
465 * Class representing a date/time interval. Used for date calculations.
466 * <pre>
467 * new goog.date.Interval(0, 1) // One month
468 * new goog.date.Interval(0, 0, 3, 1) // Three days and one hour
469 * new goog.date.Interval(goog.date.Interval.DAYS, 1) // One day
470 * </pre>
471 *
472 * @param {number|string=} opt_years Years or string representing date part.
473 * @param {number=} opt_months Months or number of whatever date part specified
474 * by first parameter.
475 * @param {number=} opt_days Days.
476 * @param {number=} opt_hours Hours.
477 * @param {number=} opt_minutes Minutes.
478 * @param {number=} opt_seconds Seconds.
479 * @constructor
480 * @struct
481 * @final
482 */
483goog.date.Interval = function(opt_years, opt_months, opt_days, opt_hours,
484 opt_minutes, opt_seconds) {
485 if (goog.isString(opt_years)) {
486 var type = opt_years;
487 var interval = /** @type {number} */ (opt_months);
488 this.years = type == goog.date.Interval.YEARS ? interval : 0;
489 this.months = type == goog.date.Interval.MONTHS ? interval : 0;
490 this.days = type == goog.date.Interval.DAYS ? interval : 0;
491 this.hours = type == goog.date.Interval.HOURS ? interval : 0;
492 this.minutes = type == goog.date.Interval.MINUTES ? interval : 0;
493 this.seconds = type == goog.date.Interval.SECONDS ? interval : 0;
494 } else {
495 this.years = /** @type {number} */ (opt_years) || 0;
496 this.months = opt_months || 0;
497 this.days = opt_days || 0;
498 this.hours = opt_hours || 0;
499 this.minutes = opt_minutes || 0;
500 this.seconds = opt_seconds || 0;
501 }
502};

505/**
506 * Parses an XML Schema duration (ISO 8601 extended).
507 * @see http://www.w3.org/TR/xmlschema-2/#duration
508 *
509 * @param {string} duration An XML schema duration in textual format.
510 * Recurring durations and weeks are not supported.
511 * @return {goog.date.Interval} The duration as a goog.date.Interval or null
512 * if the parse fails.
513 */
514goog.date.Interval.fromIsoString = function(duration) {
515 var parts = duration.match(goog.date.splitDurationRegex_);
516 if (!parts) {
517 return null;
518 }
519
520 var timeEmpty = !(parts[6] || parts[7] || parts[8]);
521 var dateTimeEmpty = timeEmpty && !(parts[2] || parts[3] || parts[4]);
522 if (dateTimeEmpty || timeEmpty && parts[5]) {
523 return null;
524 }
525
526 var negative = parts[1];
527 var years = parseInt(parts[2], 10) || 0;
528 var months = parseInt(parts[3], 10) || 0;
529 var days = parseInt(parts[4], 10) || 0;
530 var hours = parseInt(parts[6], 10) || 0;
531 var minutes = parseInt(parts[7], 10) || 0;
532 var seconds = parseFloat(parts[8]) || 0;
533 return negative ? new goog.date.Interval(-years, -months, -days,
534 -hours, -minutes, -seconds) :
535 new goog.date.Interval(years, months, days,
536 hours, minutes, seconds);
537};

540/**
541 * Serializes goog.date.Interval into XML Schema duration (ISO 8601 extended).
542 * @see http://www.w3.org/TR/xmlschema-2/#duration
543 *
544 * @param {boolean=} opt_verbose Include zero fields in the duration string.
545 * @return {?string} An XML schema duration in ISO 8601 extended format,
546 * or null if the interval contains both positive and negative fields.
547 */
548goog.date.Interval.prototype.toIsoString = function(opt_verbose) {
549 var minField = Math.min(this.years, this.months, this.days,
550 this.hours, this.minutes, this.seconds);
551 var maxField = Math.max(this.years, this.months, this.days,
552 this.hours, this.minutes, this.seconds);
553 if (minField < 0 && maxField > 0) {
554 return null;
555 }
556
557 // Return 0 seconds if all fields are zero.
558 if (!opt_verbose && minField == 0 && maxField == 0) {
559 return 'PT0S';
560 }
561
562 var res = [];
563
564 // Add sign and 'P' prefix.
565 if (minField < 0) {
566 res.push('-');
567 }
568 res.push('P');
569
570 // Add date.
571 if (this.years || opt_verbose) {
572 res.push(Math.abs(this.years) + 'Y');
573 }
574 if (this.months || opt_verbose) {
575 res.push(Math.abs(this.months) + 'M');
576 }
577 if (this.days || opt_verbose) {
578 res.push(Math.abs(this.days) + 'D');
579 }
580
581 // Add time.
582 if (this.hours || this.minutes || this.seconds || opt_verbose) {
583 res.push('T');
584 if (this.hours || opt_verbose) {
585 res.push(Math.abs(this.hours) + 'H');
586 }
587 if (this.minutes || opt_verbose) {
588 res.push(Math.abs(this.minutes) + 'M');
589 }
590 if (this.seconds || opt_verbose) {
591 res.push(Math.abs(this.seconds) + 'S');
592 }
593 }
594
595 return res.join('');
596};

733/**
734 * Class representing a date. Defaults to current date if none is specified.
735 *
736 * Implements most methods of the native js Date object (except the time related
737 * ones, {@see goog.date.DateTime}) and can be used interchangeably with it just
738 * as if goog.date.Date was a synonym of Date. To make this more transparent,
739 * Closure APIs should accept goog.date.DateLike instead of the real Date
740 * object.
741 *
742 * To allow goog.date.Date objects to be passed as arguments to methods
743 * expecting Date objects this class is marked as extending the built in Date
744 * object even though that's not strictly true.
745 *
746 * @param {number|goog.date.DateLike=} opt_year Four digit year or a date-like
747 * object. If not set, the created object will contain the date
748 * determined by goog.now().
749 * @param {number=} opt_month Month, 0 = Jan, 11 = Dec.
750 * @param {number=} opt_date Date of month, 1 - 31.
751 * @constructor
752 * @struct
753 * @see goog.date.DateTime
754 */
755goog.date.Date = function(opt_year, opt_month, opt_date) {
756 /** @protected {!Date} The wrapped date or datetime. */
757 this.date;
758 // goog.date.DateTime assumes that only this.date is added in this ctor.
759 if (goog.isNumber(opt_year)) {
760 this.date = this.buildDate_(opt_year, opt_month || 0, opt_date || 1);
761 this.maybeFixDst_(opt_date || 1);
762 } else if (goog.isObject(opt_year)) {
763 this.date = this.buildDate_(opt_year.getFullYear(), opt_year.getMonth(),
764 opt_year.getDate());
765 this.maybeFixDst_(opt_year.getDate());
766 } else {
767 this.date = new Date(goog.now());
768 this.date.setHours(0);
769 this.date.setMinutes(0);
770 this.date.setSeconds(0);
771 this.date.setMilliseconds(0);
772 }
773};

776/**
777 * new Date(y, m, d) treats years in the interval [0, 100) as two digit years,
778 * adding 1900 to them. This method ensures that calling the date constructor
779 * as a copy constructor returns a value that is equal to the passed in
780 * date value by explicitly setting the full year.
781 * @private
782 * @param {number} fullYear The full year (including century).
783 * @param {number} month The month, from 0-11.
784 * @param {number} date The day of the month.
785 * @return {!Date} The constructed Date object.
786 */
787goog.date.Date.prototype.buildDate_ = function(fullYear, month, date) {
788 var d = new Date(fullYear, month, date);
789 if (fullYear >= 0 && fullYear < 100) {
790 // Can't just setFullYear as new Date() can flip over for e.g. month = 13.
791 d.setFullYear(d.getFullYear() - 1900);
792 }
793 return d;
794};

1164/**
1165 * Performs date calculation by adding the supplied interval to the date.
1166 *
1167 * @param {goog.date.Interval} interval Date interval to add.
1168 */
1169goog.date.Date.prototype.add = function(interval) {
1170 if (interval.years || interval.months) {
1171 // As months have different number of days adding a month to Jan 31 by just
1172 // setting the month would result in a date in early March rather than Feb
1173 // 28 or 29. Doing it this way overcomes that problem.
1174
1175 // adjust year and month, accounting for both directions
1176 var month = this.getMonth() + interval.months + interval.years * 12;
1177 var year = this.getYear() + Math.floor(month / 12);
1178 month %= 12;
1179 if (month < 0) {
1180 month += 12;
1181 }
1182
1183 var daysInTargetMonth = goog.date.getNumberOfDaysInMonth(year, month);
1184 var date = Math.min(daysInTargetMonth, this.getDate());
1185
1186 // avoid inadvertently causing rollovers to adjacent months
1187 this.setDate(1);
1188
1189 this.setFullYear(year);
1190 this.setMonth(month);
1191 this.setDate(date);
1192 }
1193
1194 if (interval.days) {
1195 // Convert the days to milliseconds and add it to the UNIX timestamp.
1196 // Taking noon helps to avoid 1 day error due to the daylight saving.
1197 var noon = new Date(this.getYear(), this.getMonth(), this.getDate(), 12);
1198 var result = new Date(noon.getTime() + interval.days * 86400000);
1199
1200 // Set date to 1 to prevent rollover caused by setting the year or month.
1201 this.setDate(1);
1202 this.setFullYear(result.getFullYear());
1203 this.setMonth(result.getMonth());
1204 this.setDate(result.getDate());
1205
1206 this.maybeFixDst_(result.getDate());
1207 }
1208};

1674/**
1675 * Generates time label for the datetime, e.g., '5:30am'.
1676 * By default this does not pad hours (e.g., to '05:30') and it does add
1677 * an am/pm suffix.
1678 * TODO(user): i18n -- hardcoding time format like this is bad. E.g., in CJK
1679 * locales, need Chinese characters for hour and minute units.
1680 * @param {boolean=} opt_padHours Whether to pad hours, e.g., '05:30' vs '5:30'.
1681 * @param {boolean=} opt_showAmPm Whether to show the 'am' and 'pm' suffix.
1682 * @param {boolean=} opt_omitZeroMinutes E.g., '5:00pm' becomes '5pm',
1683 * but '5:01pm' remains '5:01pm'.
1684 * @return {string} The time label.
1685 */
1686goog.date.DateTime.prototype.toUsTimeString = function(opt_padHours,
1687 opt_showAmPm,
1688 opt_omitZeroMinutes) {
1689 var hours = this.getHours();
1690
1691 // show am/pm marker by default
1692 if (!goog.isDef(opt_showAmPm)) {
1693 opt_showAmPm = true;
1694 }
1695
1696 // 12pm
1697 var isPM = hours == 12;
1698
1699 // change from 1-24 to 1-12 basis
1700 if (hours > 12) {
1701 hours -= 12;
1702 isPM = true;
1703 }
1704
1705 // midnight is expressed as "12am", but if am/pm marker omitted, keep as '0'
1706 if (hours == 0 && opt_showAmPm) {
1707 hours = 12;
1708 }
1709

1711 var minutes = this.getMinutes();
1712 if (!opt_omitZeroMinutes || minutes > 0) {
1713 label += ':' + goog.string.padNumber(minutes, 2);
1714 }
1715
1716 // by default, show am/pm suffix

1719 * @desc Suffix for morning times.
1720 */
