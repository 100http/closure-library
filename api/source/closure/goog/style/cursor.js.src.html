15/**
16 * @fileoverview Functions to create special cursor styles, like "draggable"
17 * (open hand) or "dragging" (closed hand).
18 *
19 * @author dgajda@google.com (Damian Gajda) Ported to closure.
20 */
21
22goog.provide('goog.style.cursor');
23
24goog.require('goog.userAgent');
25
26
27/**
28 * The file name for the open-hand (draggable) cursor.
29 * @type {string}
30 */
31goog.style.cursor.OPENHAND_FILE = 'openhand.cur';
32
33
34/**
35 * The file name for the close-hand (dragging) cursor.
36 * @type {string}
37 */
38goog.style.cursor.CLOSEDHAND_FILE = 'closedhand.cur';
39
40
41/**
42 * Create the style for the draggable cursor based on browser and OS.
43 * The value can be extended to be '!important' if needed.
44 *
45 * @param {string} absoluteDotCurFilePath The absolute base path of
46 * 'openhand.cur' file to be used if the browser supports it.
47 * @param {boolean=} opt_obsolete Just for compiler backward compatibility.
48 * @return {string} The "draggable" mouse cursor style value.
49 */
50goog.style.cursor.getDraggableCursorStyle = function(
51 absoluteDotCurFilePath, opt_obsolete) {
52 return goog.style.cursor.getCursorStyle_(
53 '-moz-grab',
54 absoluteDotCurFilePath + goog.style.cursor.OPENHAND_FILE,
55 'default');
56};
57
58
59/**
60 * Create the style for the dragging cursor based on browser and OS.
61 * The value can be extended to be '!important' if needed.
62 *
63 * @param {string} absoluteDotCurFilePath The absolute base path of
64 * 'closedhand.cur' file to be used if the browser supports it.
65 * @param {boolean=} opt_obsolete Just for compiler backward compatibility.
66 * @return {string} The "dragging" mouse cursor style value.
67 */
68goog.style.cursor.getDraggingCursorStyle = function(
69 absoluteDotCurFilePath, opt_obsolete) {
70 return goog.style.cursor.getCursorStyle_(
71 '-moz-grabbing',
72 absoluteDotCurFilePath + goog.style.cursor.CLOSEDHAND_FILE,
73 'move');
74};

77/**
78 * Create the style for the cursor based on browser and OS.
79 *
80 * @param {string} geckoNonWinBuiltInStyleValue The Gecko on non-Windows OS,
81 * built in cursor style.
82 * @param {string} absoluteDotCurFilePath The .cur file absolute file to be
83 * used if the browser supports it.
84 * @param {string} defaultStyle The default fallback cursor style.
85 * @return {string} The computed mouse cursor style value.
86 * @private
87 */
88goog.style.cursor.getCursorStyle_ = function(geckoNonWinBuiltInStyleValue,
89 absoluteDotCurFilePath, defaultStyle) {
90 // Use built in cursors for Gecko on non Windows OS.
91 // We prefer our custom cursor, but Firefox Mac and Firefox Linux
92 // cannot do custom cursors. They do have a built-in hand, so use it:
93 if (goog.userAgent.GECKO && !goog.userAgent.WINDOWS) {
94 return geckoNonWinBuiltInStyleValue;
95 }
96
97 // Use the custom cursor file.
98 var cursorStyleValue = 'url("' + absoluteDotCurFilePath + '")';
99 // Change hot-spot for Safari.
100 if (goog.userAgent.WEBKIT) {
101 // Safari seems to ignore the hotspot specified in the .cur file (it uses
102 // 0,0 instead). This causes the cursor to jump as it transitions between
103 // openhand and pointer which is especially annoying when trying to hover
104 // over the route for draggable routes. We specify the hotspot here as 7,5
105 // in the css - unfortunately ie6 can't understand this and falls back to
106 // the builtin cursors so we just do this for safari (but ie DOES correctly
107 // use the hotspot specified in the file so this is ok). The appropriate
108 // coordinates were determined by looking at a hex dump and the format
109 // description from wikipedia.
110 cursorStyleValue += ' 7 5';
111 }
112 // Add default cursor fallback.
113 cursorStyleValue += ', ' + defaultStyle;
114 return cursorStyleValue;
115};
