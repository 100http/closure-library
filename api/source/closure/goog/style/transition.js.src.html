35/**
36 * A typedef to represent a CSS3 transition property. Duration and delay
37 * are both in seconds. Timing is CSS3 timing function string, such as
38 * 'easein', 'linear'.
39 *
40 * Alternatively, specifying string in the form of '[property] [duration]
41 * [timing] [delay]' as specified in CSS3 transition is fine too.
42 *
43 * @typedef { {
44 * property: string,
45 * duration: number,
46 * timing: string,
47 * delay: number
48 * } | string }
49 */
50goog.style.transition.Css3Property;
51

53/**
54 * Sets the element CSS3 transition to properties.
55 * @param {Element} element The element to set transition on.
56 * @param {goog.style.transition.Css3Property|
57 * Array<goog.style.transition.Css3Property>} properties A single CSS3
58 * transition property or array of properties.
59 */
60goog.style.transition.set = function(element, properties) {
61 if (!goog.isArray(properties)) {
62 properties = [properties];
63 }
64 goog.asserts.assert(
65 properties.length > 0, 'At least one Css3Property should be specified.');
66
67 var values = goog.array.map(
68 properties, function(p) {
69 if (goog.isString(p)) {
70 return p;
71 } else {
72 goog.asserts.assertObject(p,
73 'Expected css3 property to be an object.');
74 var propString = p.property + ' ' + p.duration + 's ' + p.timing +
75 ' ' + p.delay + 's';
76 goog.asserts.assert(p.property && goog.isNumber(p.duration) &&
77 p.timing && goog.isNumber(p.delay),
78 'Unexpected css3 property value: %s', propString);
79 return propString;
80 }
81 });
82 goog.style.transition.setPropertyValue_(element, values.join(','));
83};

87 * Removes any programmatically-added CSS3 transition in the given element.
88 * @param {Element} element The element to remove transition from.

90goog.style.transition.removeAll = function(element) {

95/**
96 * @return {boolean} Whether CSS3 transition is supported.
97 */
98goog.style.transition.isSupported = goog.functions.cacheReturnValue(function() {
99 // Since IE would allow any attribute, we need to explicitly check the
100 // browser version here instead.
101 if (goog.userAgent.IE) {
102 return goog.userAgent.isVersionOrHigher('10.0');
103 }
104
105 // We create a test element with style=-vendor-transition
106 // We then detect whether those style properties are recognized and
107 // available from js.
108 var el = document.createElement(goog.dom.TagName.DIV);
109 var transition = 'opacity 1s linear';
110 var vendorPrefix = goog.dom.vendor.getVendorPrefix();
111 var style = {'transition': transition};
112 if (vendorPrefix) {
113 style[vendorPrefix + '-transition'] = transition;
114 }
115 goog.dom.safe.setInnerHtml(el,
116 goog.html.SafeHtml.create('div', {'style': style}));
117
118 var testElement = /** @type {Element} */ (el.firstChild);
119 goog.asserts.assert(testElement.nodeType == Node.ELEMENT_NODE);
120
121 return goog.style.getStyle(testElement, 'transition') != '';
122});
