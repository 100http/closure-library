27/**
28 * Returns the normalized scrollLeft position for a scrolled element.
29 * @param {Element} element The scrolled element.
30 * @return {number} The number of pixels the element is scrolled. 0 indicates
31 * that the element is not scrolled at all (which, in general, is the
32 * left-most position in ltr and the right-most position in rtl).
33 */
34goog.style.bidi.getScrollLeft = function(element) {
35 var isRtl = goog.style.isRightToLeft(element);
36 if (isRtl && goog.userAgent.GECKO) {
37 // ScrollLeft starts at 0 and then goes negative as the element is scrolled
38 // towards the left.
39 return -element.scrollLeft;
40 } else if (isRtl &&
41 !(goog.userAgent.IE && goog.userAgent.isVersionOrHigher('8'))) {
42 // ScrollLeft starts at the maximum positive value and decreases towards
43 // 0 as the element is scrolled towards the left. However, for overflow
44 // visible, there is no scrollLeft and the value always stays correctly at 0
45 var overflowX = goog.style.getComputedOverflowX(element);
46 if (overflowX == 'visible') {
47 return element.scrollLeft;
48 } else {
49 return element.scrollWidth - element.clientWidth - element.scrollLeft;
50 }
51 }
52 // ScrollLeft behavior is identical in rtl and ltr, it starts at 0 and
53 // increases as the element is scrolled away from the start.
54 return element.scrollLeft;
55};

58/**
59 * Returns the "offsetStart" of an element, analagous to offsetLeft but
60 * normalized for right-to-left environments and various browser
61 * inconsistencies. This value returned can always be passed to setScrollOffset
62 * to scroll to an element's left edge in a left-to-right offsetParent or
63 * right edge in a right-to-left offsetParent.
64 *
65 * For example, here offsetStart is 10px in an LTR environment and 5px in RTL:
66 *
67 * <pre>
68 * | xxxxxxxxxx |
69 * ^^^^^^^^^^ ^^^^ ^^^^^
70 * 10px elem 5px
71 * </pre>
72 *
73 * If an element is positioned before the start of its offsetParent, the
74 * startOffset may be negative. This can be used with setScrollOffset to
75 * reliably scroll to an element:
76 *
77 * <pre>
78 * var scrollOffset = goog.style.bidi.getOffsetStart(element);
79 * goog.style.bidi.setScrollOffset(element.offsetParent, scrollOffset);
80 * </pre>
81 *
82 * @see setScrollOffset
83 *
84 * @param {Element} element The element for which we need to determine the
85 * offsetStart position.
86 * @return {number} The offsetStart for that element.
87 */
88goog.style.bidi.getOffsetStart = function(element) {
89 element = /** @type {!HTMLElement} */ (element);
90 var offsetLeftForReal = element.offsetLeft;
91
92 // The element might not have an offsetParent.
93 // For example, the node might not be attached to the DOM tree,
94 // and position:fixed children do not have an offset parent.
95 // Just try to do the best we can with what we have.
96 var bestParent = element.offsetParent;
97
98 if (!bestParent && goog.style.getComputedPosition(element) == 'fixed') {
99 bestParent = goog.dom.getOwnerDocument(element).documentElement;
100 }
101
102 // Just give up in this case.
103 if (!bestParent) {
104 return offsetLeftForReal;
105 }
106
107 if (goog.userAgent.GECKO) {
108 // When calculating an element's offsetLeft, Firefox erroneously subtracts
109 // the border width from the actual distance. So we need to add it back.
110 var borderWidths = goog.style.getBorderBox(bestParent);
111 offsetLeftForReal += borderWidths.left;
112 } else if (goog.userAgent.isDocumentModeOrHigher(8) &&
113 !goog.userAgent.isDocumentModeOrHigher(9)) {
114 // When calculating an element's offsetLeft, IE8/9-Standards Mode
115 // erroneously adds the border width to the actual distance. So we need to
116 // subtract it.
117 var borderWidths = goog.style.getBorderBox(bestParent);
118 offsetLeftForReal -= borderWidths.left;
119 }
120
121 if (goog.style.isRightToLeft(bestParent)) {
122 // Right edge of the element relative to the left edge of its parent.
123 var elementRightOffset = offsetLeftForReal + element.offsetWidth;
124
125 // Distance from the parent's right edge to the element's right edge.
126 return bestParent.clientWidth - elementRightOffset;
127 }
128
129 return offsetLeftForReal;
130};

133/**
134 * Sets the element's scrollLeft attribute so it is correctly scrolled by
135 * offsetStart pixels. This takes into account whether the element is RTL and
136 * the nuances of different browsers. To scroll to the "beginning" of an
137 * element use getOffsetStart to obtain the element's offsetStart value and then
138 * pass the value to setScrollOffset.
139 * @see getOffsetStart
140 * @param {Element} element The element to set scrollLeft on.
141 * @param {number} offsetStart The number of pixels to scroll the element.
142 * If this value is < 0, 0 is used.
143 */
144goog.style.bidi.setScrollOffset = function(element, offsetStart) {
145 offsetStart = Math.max(offsetStart, 0);
146 // In LTR and in "mirrored" browser RTL (such as IE), we set scrollLeft to
147 // the number of pixels to scroll.
148 // Otherwise, in RTL, we need to account for different browser behavior.
149 if (!goog.style.isRightToLeft(element)) {
150 element.scrollLeft = offsetStart;
151 } else if (goog.userAgent.GECKO) {
152 // Negative scroll-left positions in RTL.
153 element.scrollLeft = -offsetStart;
154 } else if (!(goog.userAgent.IE && goog.userAgent.isVersionOrHigher('8'))) {
155 // Take the current scrollLeft value and move to the right by the
156 // offsetStart to get to the left edge of the element, and then by
157 // the clientWidth of the element to get to the right edge.
158 element.scrollLeft =
159 element.scrollWidth - offsetStart - element.clientWidth;
160 } else {
161 element.scrollLeft = offsetStart;
162 }
163};

166/**
167 * Sets the element's left style attribute in LTR or right style attribute in
168 * RTL. Also clears the left attribute in RTL and the right attribute in LTR.
169 * @param {Element} elem The element to position.
170 * @param {number} left The left position in LTR; will be set as right in RTL.
171 * @param {?number} top The top position. If null only the left/right is set.
172 * @param {boolean} isRtl Whether we are in RTL mode.
173 */
174goog.style.bidi.setPosition = function(elem, left, top, isRtl) {
175 if (!goog.isNull(top)) {
176 elem.style.top = top + 'px';
177 }
178 if (isRtl) {
179 elem.style.right = left + 'px';
180 elem.style.left = '';
181 } else {
182 elem.style.left = left + 'px';
183 elem.style.right = '';
184 }
185};
