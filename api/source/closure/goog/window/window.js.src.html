53/**
54 * Opens a new window.
55 *
56 * @param {goog.html.SafeUrl|string|Object} linkRef If an Object with an 'href'
57 * attribute (such as HTMLAnchorElement) is passed then the value of 'href'
58 * is used, otherwise otherwise its toString method is called. Note that
59 * if a string|Object is used, an untrusted script execution might result
60 * (e.g. from javascript: URLs).
61 *
62 * @param {Object=} opt_options supports the following options:
63 * 'target': (string) target (window name). If null, linkRef.target will
64 * be used.
65 * 'width': (number) window width.
66 * 'height': (number) window height.
67 * 'top': (number) distance from top of screen
68 * 'left': (number) distance from left of screen
69 * 'toolbar': (boolean) show toolbar
70 * 'scrollbars': (boolean) show scrollbars
71 * 'location': (boolean) show location
72 * 'statusbar': (boolean) show statusbar
73 * 'menubar': (boolean) show menubar
74 * 'resizable': (boolean) resizable
75 * 'noreferrer': (boolean) whether to attempt to remove the referrer header
76 * from the request headers. Does this by opening a blank window that
77 * then redirects to the target url, so users may see some flickering.
78 *
79 * @param {Window=} opt_parentWin Parent window that should be used to open the
80 * new window.
81 *
82 * @return {Window} Returns the window object that was opened. This returns
83 * null if a popup blocker prevented the window from being
84 * opened. In case when a new window is opened in a different
85 * browser sandbox (such as iOS standalone mode), the returned
86 * object is a emulated Window object that functions as if
87 * a cross-origin window has been opened.
88 */
89goog.window.open = function(linkRef, opt_options, opt_parentWin) {
90 if (!opt_options) {
91 opt_options = {};
92 }
93 var parentWin = opt_parentWin || window;
94
95 var url;
96 if (linkRef instanceof goog.html.SafeUrl) {
97 url = goog.html.SafeUrl.unwrap(linkRef);
98 } else {
99 // HTMLAnchorElement has a toString() method with the same behavior as
100 // goog.Uri in all browsers except for Safari, which returns
101 // '[object HTMLAnchorElement]'. We check for the href first, then
102 // assume that it's a goog.Uri or String otherwise.
103 url = typeof linkRef.href != 'undefined' ? linkRef.href : String(linkRef);
104 }
105
106 var target = opt_options.target || linkRef.target;
107
108 var sb = [];
109 for (var option in opt_options) {
110 switch (option) {
111 case 'width':
112 case 'height':
113 case 'top':
114 case 'left':
115 sb.push(option + '=' + opt_options[option]);
116 break;
117 case 'target':
118 case 'noreferrer':
119 break;
120 default:
121 sb.push(option + '=' + (opt_options[option] ? 1 : 0));
122 }
123 }
124 var optionString = sb.join(',');
125
126 var newWin;
127 if (goog.labs.userAgent.platform.isIos() &&
128 parentWin.navigator && parentWin.navigator['standalone'] &&
129 target && target != '_self') {
130 // iOS in standalone mode disregards "target" in window.open and always
131 // opens new URL in the same window. The workout around is to create an "A"
132 // element and send a click event to it.
133 // Notice that the "A" tag does NOT have to be added to the DOM.
134 var a = parentWin.document.createElement(goog.dom.TagName.A);
135 // TODO(user): Sanitize URL. See cl/89965465 for code and test which
136 // was reverted.
137 a.setAttribute('href', url);
138 a.setAttribute('target', target);
139 if (opt_options['noreferrer']) {
140 a.setAttribute('rel', 'noreferrer');
141 }
142 var click = document.createEvent('MouseEvent');
143 click.initMouseEvent('click',
144 true, // canBubble
145 true, // cancelable
146 parentWin,
147 1); // detail = mousebutton
148 a.dispatchEvent(click);
149 // New window is not available in this case. Instead, a fake Window object
150 // is returned. In particular, it will have window.document undefined. In
151 // general, it will appear to most of clients as a Window for a different
152 // origin. Since iOS standalone web apps are run in their own sandbox, this
153 // is the most appropriate return value.
154 newWin = /** @type {!Window} */ ({});
155 } else if (opt_options['noreferrer']) {
156 // Use a meta-refresh to stop the referrer from being included in the
157 // request headers. This seems to be the only cross-browser way to
158 // remove the referrer. It also allows for the opener to be set to null
159 // in the new window, thus disallowing the opened window from navigating
160 // its opener.
161 //
162 // Detecting user agent and then using a different strategy per browser
163 // would allow the referrer to leak in case of an incorrect/missing user
164 // agent.
165 newWin = parentWin.open('', target, optionString);
166 if (newWin) {
167 if (goog.userAgent.EDGE_OR_IE) {
168 // IE/EDGE can't parse the content attribute if the url contains
169 // a semicolon. We can fix this by adding quotes around the url, but
170 // then we can't parse quotes in the URL correctly. We take a
171 // best-effort approach.
172 //
173 // If the URL has semicolons, wrap it in single quotes to protect
174 // the semicolons.
175 // If the URL has semicolons and single quotes, url-encode the single
176 // quotes as well.
177 //
178 // This is imperfect. Notice that both ' and ; are reserved characters
179 // in URIs, so this could do the wrong thing, but at least it will
180 // do the wrong thing in only rare cases.
181 // ugh.
182 if (goog.string.contains(url, ';')) {
183 url = "'" + url.replace(/'/g, '%27') + "'";
184 }
185 }
186 newWin.opener = null;
187 var escapedUrl = goog.string.htmlEscape(url);
188 var safeHtml = goog.html.uncheckedconversions
189 .safeHtmlFromStringKnownToSatisfyTypeContract(
190 goog.string.Const.from('b/12014412, meta tag with sanitized URL'),
191 '<META HTTP-EQUIV="refresh" content="0; url=' +
192 escapedUrl + '">');
193 goog.dom.safe.documentWrite(newWin.document, safeHtml);
194 newWin.document.close();
195 }
196 } else {
197 newWin = parentWin.open(url, target, optionString);
198 }
199 // newWin is null if a popup blocker prevented the window open.
200 return newWin;
201};

204/**
205 * Opens a new window without any real content in it.
206 *
207 * This can be used to get around popup blockers if you need to open a window
208 * in response to a user event, but need to do asynchronous work to determine
209 * the URL to open, and then set the URL later.
210 *
211 * Example usage:
212 *
213 * var newWin = goog.window.openBlank('Loading...');
214 * setTimeout(
215 * function() {
216 * newWin.location.href = 'http://www.google.com';
217 * }, 100);
218 *
219 * @param {string=} opt_message String to show in the new window. This string
220 * will be HTML-escaped to avoid XSS issues.
221 * @param {Object=} opt_options Options to open window with.
222 * {@see goog.window.open for exact option semantics}.
223 * @param {Window=} opt_parentWin Parent window that should be used to open the
224 * new window.
225 * @return {Window} Returns the window object that was opened. This returns
226 * null if a popup blocker prevented the window from being
227 * opened.
228 */
229goog.window.openBlank = function(opt_message, opt_options, opt_parentWin) {
230 // Open up a window with the loading message and nothing else.
231 // This will be interpreted as HTML content type with a missing doctype
232 // and html/body tags, but is otherwise acceptable.
233 //
234 // IMPORTANT: The order of escaping is crucial here in order to avoid XSS.
235 // First, HTML-escaping is needed because the result of the JS expression
236 // is evaluated as HTML. Second, JS-string escaping is needed; this avoids
237 // \u escaping from inserting HTML tags and \ from escaping the final ".
238 // Finally, URL percent-encoding is done with encodeURI(); this
239 // avoids percent-encoding from bypassing HTML and JS escaping.
240 //
241 // Note: There are other ways the same result could be achieved but the
242 // current behavior was preserved when this code was refactored to use
243 // SafeUrl, in order to avoid breakage.
244 var loadingMessage;
245 if (!opt_message) {
246 loadingMessage = '';
247 } else {
248 loadingMessage =
249 goog.string.escapeString(goog.string.htmlEscape(opt_message));
250 }
251 var url = goog.html.uncheckedconversions
252 .safeUrlFromStringKnownToSatisfyTypeContract(
253 goog.string.Const.from(
254 'b/12014412, encoded string in javascript: URL'),
255 'javascript:"' + encodeURI(loadingMessage) + '"');
256 return /** @type {Window} */ (goog.window.open(
257 url, opt_options, opt_parentWin));
258};

261/**
262 * Raise a help popup window, defaulting to "Google standard" size and name.
263 *
264 * (If your project is using GXPs, consider using {@link PopUpLink.gxp}.)
265 *
266* @param {goog.html.SafeUrl|string|Object} linkRef If an Object with an 'href'
267 * attribute (such as HTMLAnchorElement) is passed then the value of 'href'
268 * is used, otherwise otherwise its toString method is called. Note that
269 * if a string|Object is used, an untrusted script execution might result
270 * (e.g. from javascript: URLs).
271 *
272 * @param {Object=} opt_options Options to open window with.
273 * {@see goog.window.open for exact option semantics}
274 * Additional wrinkles to the options:
275 * - if 'target' field is null, linkRef.target will be used. If *that's*
276 * null, the default is "google_popup".
277 * - if 'width' field is not specified, the default is 690.
278 * - if 'height' field is not specified, the default is 500.
279 *
280 * @return {boolean} true if the window was not popped up, false if it was.
281 */
282goog.window.popup = function(linkRef, opt_options) {
283 if (!opt_options) {
284 opt_options = {};
285 }
286
287 // set default properties
288 opt_options['target'] = opt_options['target'] ||
289 linkRef['target'] || goog.window.DEFAULT_POPUP_TARGET;
290 opt_options['width'] = opt_options['width'] ||
291 goog.window.DEFAULT_POPUP_WIDTH;
292 opt_options['height'] = opt_options['height'] ||
293 goog.window.DEFAULT_POPUP_HEIGHT;
294
295 var newWin = goog.window.open(linkRef, opt_options);
296 if (!newWin) {
297 return true;
298 }
299 newWin.focus();
300
301 return false;
302};
