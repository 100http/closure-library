25/**
26 * Parses an alpha color out of a string.
27 * @param {string} str Color in some format.
28 * @return {{hex: string, type: string}} 'hex' is a string containing
29 * a hex representation of the color, and 'type' is a string
30 * containing the type of color format passed in ('hex', 'rgb', 'named').
31 */
32goog.color.alpha.parse = function(str) {
33 var result = {};
34 str = String(str);
35
36 var maybeHex = goog.color.prependHashIfNecessaryHelper(str);
37 if (goog.color.alpha.isValidAlphaHexColor_(maybeHex)) {
38 result.hex = goog.color.alpha.normalizeAlphaHex_(maybeHex);
39 result.type = 'hex';
40 return result;
41 } else {
42 var rgba = goog.color.alpha.isValidRgbaColor_(str);
43 if (rgba.length) {
44 result.hex = goog.color.alpha.rgbaArrayToHex(rgba);
45 result.type = 'rgba';
46 return result;
47 } else {
48 var hsla = goog.color.alpha.isValidHslaColor_(str);
49 if (hsla.length) {
50 result.hex = goog.color.alpha.hslaArrayToHex(hsla);
51 result.type = 'hsla';
52 return result;
53 }
54 }
55 }
56 throw Error(str + ' is not a valid color string');
57};

71/**
72 * Gets the hex color part of an alpha hex color. For example, from '#abcdef55'
73 * return '#abcdef'.
74 * @param {string} colorWithAlpha The alpha hex color to get the hex color from.
75 * @return {string} The hex color where the alpha part has been stripped off.
76 */
77goog.color.alpha.extractHexColor = function(colorWithAlpha) {
78 if (goog.color.alpha.isValidAlphaHexColor_(colorWithAlpha)) {
79 var fullColor = goog.color.prependHashIfNecessaryHelper(colorWithAlpha);
80 var normalizedColor = goog.color.alpha.normalizeAlphaHex_(fullColor);
81 return normalizedColor.substring(0, 7);
82 } else {
83 throw Error(colorWithAlpha + ' is not a valid 8-hex color string');
84 }
85};

113/**
114 * Normalize a hex representation of an alpha color.
115 * @param {string} hexColor an alpha hex color string.
116 * @return {string} hex color in the format '#rrggbbaa' with all lowercase
117 * literals.
118 * @private
119 */
120goog.color.alpha.normalizeAlphaHex_ = function(hexColor) {
121 if (!goog.color.alpha.isValidAlphaHexColor_(hexColor)) {
122 throw Error("'" + hexColor + "' is not a valid alpha hex color");
123 }
124 if (hexColor.length == 5) { // of the form #RGBA
125 hexColor = hexColor.replace(goog.color.alpha.hexQuadrupletRe_,
126 '#$1$1$2$2$3$3$4$4');
127 }
128 return hexColor.toLowerCase();
129};

132/**
133 * Converts an 8-hex representation of a color to RGBA.
134 * @param {string} hexColor Color to convert.
135 * @return {!Array<number>} array containing [r, g, b, a].
136 * r, g, b are ints between 0
137 * and 255, and a is a value between 0 and 1.
138 */
139goog.color.alpha.hexToRgba = function(hexColor) {
140 // TODO(user): Enhance code sharing with goog.color, for example by
141 // adding a goog.color.genericHexToRgb method.
142 hexColor = goog.color.alpha.normalizeAlphaHex_(hexColor);
143 var r = parseInt(hexColor.substr(1, 2), 16);
144 var g = parseInt(hexColor.substr(3, 2), 16);
145 var b = parseInt(hexColor.substr(5, 2), 16);
146 var a = parseInt(hexColor.substr(7, 2), 16);
147
148 return [r, g, b, a / 255];
149};

152/**
153 * Converts a color from RGBA to hex representation.
154 * @param {number} r Amount of red, int between 0 and 255.
155 * @param {number} g Amount of green, int between 0 and 255.
156 * @param {number} b Amount of blue, int between 0 and 255.
157 * @param {number} a Amount of alpha, float between 0 and 1.
158 * @return {string} hex representation of the color.
159 */
160goog.color.alpha.rgbaToHex = function(r, g, b, a) {
161 var intAlpha = Math.floor(a * 255);
162 if (isNaN(intAlpha) || intAlpha < 0 || intAlpha > 255) {
163 // TODO(user): The CSS spec says the value should be clamped.
164 throw Error('"(' + r + ',' + g + ',' + b + ',' + a +
165 '") is not a valid RGBA color');
166 }
167 var hexA = goog.color.prependZeroIfNecessaryHelper(intAlpha.toString(16));
168 return goog.color.rgbToHex(r, g, b) + hexA;
169};

373/**
374 * Checks if a string is a valid rgba color. We expect strings of the format
375 * '(r, g, b, a)', or 'rgba(r, g, b, a)', where r, g, b are ints in [0, 255]
376 * and a is a float in [0, 1].
377 * @param {string} str String to check.
378 * @return {!Array<number>} the integers [r, g, b, a] for valid colors or the
379 * empty array for invalid colors.
380 * @private
381 */
382goog.color.alpha.isValidRgbaColor_ = function(str) {
383 // Each component is separate (rather than using a repeater) so we can
384 // capture the match. Also, we explicitly set each component to be either 0,
385 // or start with a non-zero, to prevent octal numbers from slipping through.
386 var regExpResultArray = str.match(goog.color.alpha.rgbaColorRe_);
387 if (regExpResultArray) {
388 var r = Number(regExpResultArray[1]);
389 var g = Number(regExpResultArray[2]);
390 var b = Number(regExpResultArray[3]);
391 var a = Number(regExpResultArray[4]);
392 if (r >= 0 && r <= 255 &&
393 g >= 0 && g <= 255 &&
394 b >= 0 && b <= 255 &&
395 a >= 0 && a <= 1) {
396 return [r, g, b, a];
397 }
398 }
399 return [];
400};

434 * Takes an array of [r, g, b, a] and converts it into a string appropriate for
