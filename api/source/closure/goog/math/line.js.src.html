29/**
30 * Object representing a line.
31 * @param {number} x0 X coordinate of the start point.
32 * @param {number} y0 Y coordinate of the start point.
33 * @param {number} x1 X coordinate of the end point.
34 * @param {number} y1 Y coordinate of the end point.
35 * @struct
36 * @constructor
37 * @final
38 */
39goog.math.Line = function(x0, y0, x1, y1) {
40 /**
41 * X coordinate of the first point.
42 * @type {number}
43 */
44 this.x0 = x0;
45
46 /**
47 * Y coordinate of the first point.
48 * @type {number}
49 */
50 this.y0 = y0;
51
52 /**
53 * X coordinate of the first control point.
54 * @type {number}
55 */
56 this.x1 = x1;
57
58 /**
59 * Y coordinate of the first control point.
60 * @type {number}
61 */
62 this.y1 = y1;
63};

74/**
75 * Tests whether the given line is exactly the same as this one.
76 * @param {goog.math.Line} other The other line.
77 * @return {boolean} Whether the given line is the same as this one.
78 */
79goog.math.Line.prototype.equals = function(other) {
80 return this.x0 == other.x0 && this.y0 == other.y0 &&
81 this.x1 == other.x1 && this.y1 == other.y1;
82};

85/**
86 * @return {number} The squared length of the line segment used to define the
87 * line.
88 */
89goog.math.Line.prototype.getSegmentLengthSquared = function() {
90 var xdist = this.x1 - this.x0;
91 var ydist = this.y1 - this.y0;
92 return xdist * xdist + ydist * ydist;
93};

96/**
97 * @return {number} The length of the line segment used to define the line.
98 */
99goog.math.Line.prototype.getSegmentLength = function() {
100 return Math.sqrt(this.getSegmentLengthSquared());
101};

104/**
105 * Computes the interpolation parameter for the point on the line closest to
106 * a given point.
107 * @param {number|goog.math.Coordinate} x The x coordinate of the point, or
108 * a coordinate object.
109 * @param {number=} opt_y The y coordinate of the point - required if x is a
110 * number, ignored if x is a goog.math.Coordinate.
111 * @return {number} The interpolation parameter of the point on the line
112 * closest to the given point.
113 * @private
114 */
115goog.math.Line.prototype.getClosestLinearInterpolation_ = function(x, opt_y) {
116 var y;
117 if (x instanceof goog.math.Coordinate) {
118 y = x.y;
119 x = x.x;
120 } else {
121 y = opt_y;
122 }
123
124 var x0 = this.x0;
125 var y0 = this.y0;
126
127 var xChange = this.x1 - x0;
128 var yChange = this.y1 - y0;
129
130 return ((x - x0) * xChange + (y - y0) * yChange) /
131 this.getSegmentLengthSquared();
132};

135/**
136 * Returns the point on the line segment proportional to t, where for t = 0 we
137 * return the starting point and for t = 1 we return the end point. For t < 0
138 * or t > 1 we extrapolate along the line defined by the line segment.
139 * @param {number} t The interpolation parameter along the line segment.
140 * @return {!goog.math.Coordinate} The point on the line segment at t.
141 */
142goog.math.Line.prototype.getInterpolatedPoint = function(t) {
143 return new goog.math.Coordinate(
144 goog.math.lerp(this.x0, this.x1, t),
145 goog.math.lerp(this.y0, this.y1, t));
146};

149/**
150 * Computes the point on the line closest to a given point. Note that a line
151 * in this case is defined as the infinite line going through the start and end
152 * points. To find the closest point on the line segment itself see
153 * {@see #getClosestSegmentPoint}.
154 * @param {number|goog.math.Coordinate} x The x coordinate of the point, or
155 * a coordinate object.
156 * @param {number=} opt_y The y coordinate of the point - required if x is a
157 * number, ignored if x is a goog.math.Coordinate.
158 * @return {!goog.math.Coordinate} The point on the line closest to the given
159 * point.
160 */
161goog.math.Line.prototype.getClosestPoint = function(x, opt_y) {
162 return this.getInterpolatedPoint(
163 this.getClosestLinearInterpolation_(x, opt_y));
164};

167/**
168 * Computes the point on the line segment closest to a given point.
169 * @param {number|goog.math.Coordinate} x The x coordinate of the point, or
170 * a coordinate object.
171 * @param {number=} opt_y The y coordinate of the point - required if x is a
172 * number, ignored if x is a goog.math.Coordinate.
173 * @return {!goog.math.Coordinate} The point on the line segment closest to the
174 * given point.
175 */
176goog.math.Line.prototype.getClosestSegmentPoint = function(x, opt_y) {
177 return this.getInterpolatedPoint(
178 goog.math.clamp(this.getClosestLinearInterpolation_(x, opt_y), 0, 1));
179};
