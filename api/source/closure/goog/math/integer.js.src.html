28/**
29 * Constructs a two's-complement integer an array containing bits of the
30 * integer in 32-bit (signed) pieces, given in little-endian order (i.e.,
31 * lowest-order bits in the first piece), and the sign of -1 or 0.
32 *
33 * See the from* functions below for other convenient ways of constructing
34 * Integers.
35 *
36 * The internal representation of an integer is an array of 32-bit signed
37 * pieces, along with a sign (0 or -1) that indicates the contents of all the
38 * other 32-bit pieces out to infinity. We use 32-bit pieces because these are
39 * the size of integers on which Javascript performs bit-operations. For
40 * operations like addition and multiplication, we split each number into 16-bit
41 * pieces, which can easily be multiplied within Javascript's floating-point
42 * representation without overflow or change in sign.
43 *
44 * @struct
45 * @constructor
46 * @param {Array<number>} bits Array containing the bits of the number.
47 * @param {number} sign The sign of the number: -1 for negative and 0 positive.
48 * @final
49 */
50goog.math.Integer = function(bits, sign) {
51 /**
52 * @type {!Array<number>}
53 * @private
54 */
55 this.bits_ = [];
56
57 /**
58 * @type {number}
59 * @private
60 */
61 this.sign_ = sign;
62
63 // Copy the 32-bit signed integer values passed in. We prune out those at the
64 // top that equal the sign since they are redundant.
65 var top = true;
66 for (var i = bits.length - 1; i >= 0; i--) {
67 var val = bits[i] | 0;
68 if (!top || val != sign) {
69 this.bits_[i] = val;
70 top = false;
71 }
72 }
73};

88/**
89 * Returns an Integer representing the given (32-bit) integer value.
90 * @param {number} value A 32-bit integer value.
91 * @return {!goog.math.Integer} The corresponding Integer value.
92 */
93goog.math.Integer.fromInt = function(value) {
94 if (-128 <= value && value < 128) {
95 var cachedObj = goog.math.Integer.IntCache_[value];
96 if (cachedObj) {
97 return cachedObj;
98 }
99 }
100
101 var obj = new goog.math.Integer([value | 0], value < 0 ? -1 : 0);
102 if (-128 <= value && value < 128) {
103 goog.math.Integer.IntCache_[value] = obj;
104 }
105 return obj;
106};

109/**
110 * Returns an Integer representing the given value, provided that it is a finite
111 * number. Otherwise, zero is returned.
112 * @param {number} value The value in question.
113 * @return {!goog.math.Integer} The corresponding Integer value.
114 */
115goog.math.Integer.fromNumber = function(value) {
116 if (isNaN(value) || !isFinite(value)) {
117 return goog.math.Integer.ZERO;
118 } else if (value < 0) {
119 return goog.math.Integer.fromNumber(-value).negate();
120 } else {
121 var bits = [];
122 var pow = 1;
123 for (var i = 0; value >= pow; i++) {
124 bits[i] = (value / pow) | 0;
125 pow *= goog.math.Integer.TWO_PWR_32_DBL_;
126 }
127 return new goog.math.Integer(bits, 0);
128 }
129};

132/**
133 * Returns a Integer representing the value that comes by concatenating the
134 * given entries, each is assumed to be 32 signed bits, given in little-endian
135 * order (lowest order bits in the lowest index), and sign-extending the highest
136 * order 32-bit value.
137 * @param {Array<number>} bits The bits of the number, in 32-bit signed pieces,
138 * in little-endian order.
139 * @return {!goog.math.Integer} The corresponding Integer value.
140 */
141goog.math.Integer.fromBits = function(bits) {
142 var high = bits[bits.length - 1];
143 return new goog.math.Integer(bits, high & (1 << 31) ? -1 : 0);
144};

147/**
148 * Returns an Integer representation of the given string, written using the
149 * given radix.
150 * @param {string} str The textual representation of the Integer.
151 * @param {number=} opt_radix The radix in which the text is written.
152 * @return {!goog.math.Integer} The corresponding Integer value.
153 */
154goog.math.Integer.fromString = function(str, opt_radix) {
155 if (str.length == 0) {
156 throw Error('number format error: empty string');
157 }
158
159 var radix = opt_radix || 10;
160 if (radix < 2 || 36 < radix) {
161 throw Error('radix out of range: ' + radix);
162 }
163
164 if (str.charAt(0) == '-') {
165 return goog.math.Integer.fromString(str.substring(1), radix).negate();
166 } else if (str.indexOf('-') >= 0) {
167 throw Error('number format error: interior "-" character');
168 }
169
170 // Do several (8) digits each time through the loop, so as to
171 // minimize the calls to the very expensive emulated div.
172 var radixToPower = goog.math.Integer.fromNumber(Math.pow(radix, 8));
173
174 var result = goog.math.Integer.ZERO;
175 for (var i = 0; i < str.length; i += 8) {
176 var size = Math.min(8, str.length - i);
177 var value = parseInt(str.substring(i, i + size), radix);
178 if (size < 8) {
179 var power = goog.math.Integer.fromNumber(Math.pow(radix, size));
180 result = result.multiply(power).add(goog.math.Integer.fromNumber(value));
181 } else {
182 result = result.multiply(radixToPower);
183 result = result.add(goog.math.Integer.fromNumber(value));
184 }
185 }
186 return result;
187};

223/** @return {number} The closest floating-point representation to this value. */
224goog.math.Integer.prototype.toNumber = function() {
225 if (this.isNegative()) {
226 return -this.negate().toNumber();
227 } else {
228 var val = 0;
229 var pow = 1;
230 for (var i = 0; i < this.bits_.length; i++) {
231 val += this.getBitsUnsigned(i) * pow;
232 pow *= goog.math.Integer.TWO_PWR_32_DBL_;
233 }
234 return val;
235 }

239/**
240 * @param {number=} opt_radix The radix in which the text should be written.
241 * @return {string} The textual representation of this value.
242 * @override
243 */
244goog.math.Integer.prototype.toString = function(opt_radix) {
245 var radix = opt_radix || 10;
246 if (radix < 2 || 36 < radix) {
247 throw Error('radix out of range: ' + radix);
248 }
249
250 if (this.isZero()) {
251 return '0';
252 } else if (this.isNegative()) {
253 return '-' + this.negate().toString(radix);
254 }
255
256 // Do several (6) digits each time through the loop, so as to
257 // minimize the calls to the very expensive emulated div.
258 var radixToPower = goog.math.Integer.fromNumber(Math.pow(radix, 6));
259
260 var rem = this;
261 var result = '';
262 while (true) {
263 var remDiv = rem.divide(radixToPower);
264 var intval = rem.subtract(remDiv.multiply(radixToPower)).toInt();
265 var digits = intval.toString(radix);
266
267 rem = remDiv;
268 if (rem.isZero()) {
269 return digits + result;
270 } else {
271 while (digits.length < 6) {
272 digits = '0' + digits;
273 }
274 result = '' + digits + result;
275 }
276 }
277};

281 * Returns the index-th 32-bit (signed) piece of the Integer according to
282 * little-endian order (i.e., index 0 contains the smallest bits).
283 * @param {number} index The index in question.
284 * @return {number} The requested 32-bits as a signed number.

286goog.math.Integer.prototype.getBits = function(index) {
287 if (index < 0) {
288 return 0; // Allowing this simplifies bit shifting operations below...
289 } else if (index < this.bits_.length) {
290 return this.bits_[index];

422/**
423 * Returns an integer with only the first numBits bits of this value, sign
424 * extended from the final bit.
425 * @param {number} numBits The number of bits by which to shift.
426 * @return {!goog.math.Integer} The shorted integer value.
427 */
428goog.math.Integer.prototype.shorten = function(numBits) {
429 var arr_index = (numBits - 1) >> 5;
430 var bit_index = (numBits - 1) % 32;
431 var bits = [];
432 for (var i = 0; i < arr_index; i++) {
433 bits[i] = this.getBits(i);
434 }
435 var sigBits = bit_index == 31 ? 0xFFFFFFFF : (1 << (bit_index + 1)) - 1;
436 var val = this.getBits(arr_index) & sigBits;
437 if (val & (1 << bit_index)) {
438 val |= 0xFFFFFFFF - sigBits;
439 bits[arr_index] = val;
440 return new goog.math.Integer(bits, -1);
441 } else {
442 bits[arr_index] = val;
443 return new goog.math.Integer(bits, 0);
444 }
445};

454/**
455 * Returns the sum of this and the given Integer.
456 * @param {goog.math.Integer} other The Integer to add to this.
457 * @return {!goog.math.Integer} The Integer result.
458 */
459goog.math.Integer.prototype.add = function(other) {
460 var len = Math.max(this.bits_.length, other.bits_.length);
461 var arr = [];
462 var carry = 0;
463
464 for (var i = 0; i <= len; i++) {
465 var a1 = this.getBits(i) >>> 16;
466 var a0 = this.getBits(i) & 0xFFFF;
467
468 var b1 = other.getBits(i) >>> 16;
469 var b0 = other.getBits(i) & 0xFFFF;
470
471 var c0 = carry + a0 + b0;
472 var c1 = (c0 >>> 16) + a1 + b1;
473 carry = c1 >>> 16;
474 c0 &= 0xFFFF;
475 c1 &= 0xFFFF;
476 arr[i] = (c1 << 16) | c0;
477 }
478 return goog.math.Integer.fromBits(arr);
479};

492/**
493 * Returns the product of this and the given Integer.
494 * @param {goog.math.Integer} other The Integer to multiply against this.
495 * @return {!goog.math.Integer} The product of this and the other.
496 */
497goog.math.Integer.prototype.multiply = function(other) {
498 if (this.isZero()) {
499 return goog.math.Integer.ZERO;
500 } else if (other.isZero()) {
501 return goog.math.Integer.ZERO;
502 }
503
504 if (this.isNegative()) {
505 if (other.isNegative()) {
506 return this.negate().multiply(other.negate());
507 } else {
508 return this.negate().multiply(other).negate();
509 }
510 } else if (other.isNegative()) {
511 return this.multiply(other.negate()).negate();
512 }
513
514 // If both numbers are small, use float multiplication
515 if (this.lessThan(goog.math.Integer.TWO_PWR_24_) &&
516 other.lessThan(goog.math.Integer.TWO_PWR_24_)) {
517 return goog.math.Integer.fromNumber(this.toNumber() * other.toNumber());
518 }
519
520 // Fill in an array of 16-bit products.
521 var len = this.bits_.length + other.bits_.length;
522 var arr = [];
523 for (var i = 0; i < 2 * len; i++) {
524 arr[i] = 0;
525 }
526 for (var i = 0; i < this.bits_.length; i++) {
527 for (var j = 0; j < other.bits_.length; j++) {
528 var a1 = this.getBits(i) >>> 16;
529 var a0 = this.getBits(i) & 0xFFFF;
530
531 var b1 = other.getBits(j) >>> 16;
532 var b0 = other.getBits(j) & 0xFFFF;
533
534 arr[2 * i + 2 * j] += a0 * b0;
535 goog.math.Integer.carry16_(arr, 2 * i + 2 * j);
536 arr[2 * i + 2 * j + 1] += a1 * b0;
537 goog.math.Integer.carry16_(arr, 2 * i + 2 * j + 1);
538 arr[2 * i + 2 * j + 1] += a0 * b1;
539 goog.math.Integer.carry16_(arr, 2 * i + 2 * j + 1);
540 arr[2 * i + 2 * j + 2] += a1 * b1;
541 goog.math.Integer.carry16_(arr, 2 * i + 2 * j + 2);
542 }
543 }
544
545 // Combine the 16-bit values into 32-bit values.
546 for (var i = 0; i < len; i++) {
547 arr[i] = (arr[2 * i + 1] << 16) | arr[2 * i];
548 }
549 for (var i = len; i < 2 * len; i++) {
550 arr[i] = 0;
551 }
552 return new goog.math.Integer(arr, 0);
553};

556/**
557 * Carries any overflow from the given index into later entries.
558 * @param {Array<number>} bits Array of 16-bit values in little-endian order.
559 * @param {number} index The index in question.
560 * @private
561 */
562goog.math.Integer.carry16_ = function(bits, index) {
563 while ((bits[index] & 0xFFFF) != bits[index]) {
564 bits[index + 1] += bits[index] >>> 16;
565 bits[index] &= 0xFFFF;
566 }
567};

570/**
571 * Returns this Integer divided by the given one.
572 * @param {goog.math.Integer} other Th Integer to divide this by.
573 * @return {!goog.math.Integer} This value divided by the given one.
574 */
575goog.math.Integer.prototype.divide = function(other) {
576 if (other.isZero()) {
577 throw Error('division by zero');
578 } else if (this.isZero()) {
579 return goog.math.Integer.ZERO;
580 }
581
582 if (this.isNegative()) {
583 if (other.isNegative()) {
584 return this.negate().divide(other.negate());
585 } else {
586 return this.negate().divide(other).negate();
587 }
588 } else if (other.isNegative()) {
589 return this.divide(other.negate()).negate();
590 }
591
592 // Repeat the following until the remainder is less than other: find a
593 // floating-point that approximates remainder / other *from below*, add this
594 // into the result, and subtract it from the remainder. It is critical that
595 // the approximate value is less than or equal to the real value so that the
596 // remainder never becomes negative.
597 var res = goog.math.Integer.ZERO;
598 var rem = this;
599 while (rem.greaterThanOrEqual(other)) {
600 // Approximate the result of division. This may be a little greater or
601 // smaller than the actual value.
602 var approx = Math.max(1, Math.floor(rem.toNumber() / other.toNumber()));
603
604 // We will tweak the approximate result by changing it in the 48-th digit or
605 // the smallest non-fractional digit, whichever is larger.
606 var log2 = Math.ceil(Math.log(approx) / Math.LN2);
607 var delta = (log2 <= 48) ? 1 : Math.pow(2, log2 - 48);
608
609 // Decrease the approximation until it is smaller than the remainder. Note
610 // that if it is too large, the product overflows and is negative.
611 var approxRes = goog.math.Integer.fromNumber(approx);
612 var approxRem = approxRes.multiply(other);
613 while (approxRem.isNegative() || approxRem.greaterThan(rem)) {
614 approx -= delta;
615 approxRes = goog.math.Integer.fromNumber(approx);
616 approxRem = approxRes.multiply(other);
617 }
618
619 // We know the answer can't be zero... and actually, zero would cause
620 // infinite recursion since we would make no progress.
621 if (approxRes.isZero()) {
622 approxRes = goog.math.Integer.ONE;
623 }
624
625 res = res.add(approxRes);
626 rem = rem.subtract(approxRem);
627 }
628 return res;
629};

698/**
699 * Returns this value with bits shifted to the left by the given amount.
700 * @param {number} numBits The number of bits by which to shift.
701 * @return {!goog.math.Integer} This shifted to the left by the given amount.
702 */
703goog.math.Integer.prototype.shiftLeft = function(numBits) {
704 var arr_delta = numBits >> 5;
705 var bit_delta = numBits % 32;
706 var len = this.bits_.length + arr_delta + (bit_delta > 0 ? 1 : 0);
707 var arr = [];
708 for (var i = 0; i < len; i++) {
709 if (bit_delta > 0) {
710 arr[i] = (this.getBits(i - arr_delta) << bit_delta) |
711 (this.getBits(i - arr_delta - 1) >>> (32 - bit_delta));
712 } else {
713 arr[i] = this.getBits(i - arr_delta);
714 }
715 }
