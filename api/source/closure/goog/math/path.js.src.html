29/**
30 * Creates a path object. A path is a sequence of segments and may be open or
31 * closed. Path uses the EVEN-ODD fill rule for determining the interior of the
32 * path. A path must start with a moveTo command.
33 *
34 * A "simple" path does not contain any arcs and may be transformed using
35 * the {@code transform} method.
36 *
37 * @struct
38 * @constructor
39 * @final
40 */
41goog.math.Path = function() {
42 /**
43 * The segment types that constitute this path.
44 * @private {!Array<goog.math.Path.Segment>}
45 */
46 this.segments_ = [];
47
48 /**
49 * The number of repeated segments of the current type.
50 * @type {!Array<number>}
51 * @private
52 */
53 this.count_ = [];
54
55 /**
56 * The arguments corresponding to each of the segments.
57 * @type {!Array<number>}
58 * @private
59 */
60 this.arguments_ = [];
61
62 /**
63 * The coordinates of the point which closes the path (the point of the
64 * last moveTo command).
65 * @type {Array<number>?}
66 * @private
67 */
68 this.closePoint_ = null;
69
70 /**
71 * The coordinates most recently added to the end of the path.
72 * @type {Array<number>?}
73 * @private
74 */
75 this.currentPoint_ = null;
76
77 /**
78 * Flag for whether this is a simple path (contains no arc segments).
79 * @type {boolean}
80 * @private
81 */
82 this.simple_ = true;
83};

160/**
161 * Appends another path to the end of this path.
162 *
163 * @param {!goog.math.Path} path The path to append.
164 * @return {!goog.math.Path} This path.
165 */
166goog.math.Path.prototype.appendPath = function(path) {
167 if (path.currentPoint_) {
168 Array.prototype.push.apply(this.segments_, path.segments_);
169 Array.prototype.push.apply(this.count_, path.count_);
170 Array.prototype.push.apply(this.arguments_, path.arguments_);
171 this.currentPoint_ = path.currentPoint_.concat();
172 this.closePoint_ = path.closePoint_.concat();
173 this.simple_ = this.simple_ && path.simple_;
174 }
175 return this;
176};

195/**
196 * Adds a point to the path by moving to the specified point. Repeated moveTo
197 * commands are collapsed into a single moveTo.
198 *
199 * @param {number} x X coordinate of destination point.
200 * @param {number} y Y coordinate of destination point.
201 * @return {!goog.math.Path} The path itself.
202 */
203goog.math.Path.prototype.moveTo = function(x, y) {
204 if (goog.array.peek(this.segments_) == goog.math.Path.Segment.MOVETO) {
205 this.arguments_.length -= 2;
206 } else {
207 this.segments_.push(goog.math.Path.Segment.MOVETO);
208 this.count_.push(1);
209 }
210 this.arguments_.push(x, y);
211 this.currentPoint_ = this.closePoint_ = [x, y];
212 return this;
213};

240/**
241 * Adds points to the path by drawing a straight line to each point.
242 *
243 * @param {!Array<number>|Arguments} coordinates The coordinates of each
244 * destination point as x, y value pairs.
245 * @return {!goog.math.Path} The path itself.
246 * @private
247 */
248goog.math.Path.prototype.lineTo_ = function(coordinates) {
249 var lastSegment = goog.array.peek(this.segments_);
250 if (lastSegment == null) {
251 throw Error('Path cannot start with lineTo');
252 }
253 if (lastSegment != goog.math.Path.Segment.LINETO) {
254 this.segments_.push(goog.math.Path.Segment.LINETO);
255 this.count_.push(0);
256 }
257 for (var i = 0; i < coordinates.length; i += 2) {
258 var x = coordinates[i];
259 var y = coordinates[i + 1];
260 this.arguments_.push(x, y);
261 }
262 this.count_[this.count_.length - 1] += i / 2;
263 this.currentPoint_ = [x, y];
264 return this;
265};

299/**
300 * Adds points to the path by drawing cubic Bezier curves. Each curve is
301 * specified using 3 points (6 coordinates) - two control points and the end
302 * point of the curve.
303 *
304 * @param {!Array<number>|Arguments} coordinates The coordinates specifiying
305 * each curve in sets of 6 points: {@code [x1, y1]} the first control point,
306 * {@code [x2, y2]} the second control point and {@code [x, y]} the end
307 * point.
308 * @return {!goog.math.Path} The path itself.
309 * @private
310 */
311goog.math.Path.prototype.curveTo_ = function(coordinates) {
312 var lastSegment = goog.array.peek(this.segments_);
313 if (lastSegment == null) {
314 throw Error('Path cannot start with curve');
315 }
316 if (lastSegment != goog.math.Path.Segment.CURVETO) {
317 this.segments_.push(goog.math.Path.Segment.CURVETO);
318 this.count_.push(0);
319 }
320 for (var i = 0; i < coordinates.length; i += 6) {
321 var x = coordinates[i + 4];
322 var y = coordinates[i + 5];
323 this.arguments_.push(coordinates[i], coordinates[i + 1],
324 coordinates[i + 2], coordinates[i + 3], x, y);
325 }
326 this.count_[this.count_.length - 1] += i / 6;
327 this.currentPoint_ = [x, y];
328 return this;
329};

352/**
353 * Adds a path command to draw an arc centered at the point {@code (cx, cy)}
354 * with radius {@code rx} along the x-axis and {@code ry} along the y-axis from
355 * {@code startAngle} through {@code extent} degrees. Positive rotation is in
356 * the direction from positive x-axis to positive y-axis.
357 *
358 * @param {number} cx X coordinate of center of ellipse.
359 * @param {number} cy Y coordinate of center of ellipse.
360 * @param {number} rx Radius of ellipse on x axis.
361 * @param {number} ry Radius of ellipse on y axis.
362 * @param {number} fromAngle Starting angle measured in degrees from the
363 * positive x-axis.
364 * @param {number} extent The span of the arc in degrees.
365 * @param {boolean} connect If true, the starting point of the arc is connected
366 * to the current point.
367 * @return {!goog.math.Path} The path itself.
368 * @deprecated Use {@code arcTo} or {@code arcToAsCurves} instead.
369 */
370goog.math.Path.prototype.arc = function(cx, cy, rx, ry,
371 fromAngle, extent, connect) {
372 var startX = cx + goog.math.angleDx(fromAngle, rx);
373 var startY = cy + goog.math.angleDy(fromAngle, ry);
374 if (connect) {
375 if (!this.currentPoint_ || startX != this.currentPoint_[0] ||
376 startY != this.currentPoint_[1]) {
377 this.lineTo(startX, startY);
378 }
379 } else {
380 this.moveTo(startX, startY);
381 }
382 return this.arcTo(rx, ry, fromAngle, extent);
383};

386/**
387 * Adds a path command to draw an arc starting at the path's current point,
388 * with radius {@code rx} along the x-axis and {@code ry} along the y-axis from
389 * {@code startAngle} through {@code extent} degrees. Positive rotation is in
390 * the direction from positive x-axis to positive y-axis.
391 *
392 * This method makes the path non-simple.
393 *
394 * @param {number} rx Radius of ellipse on x axis.
395 * @param {number} ry Radius of ellipse on y axis.
396 * @param {number} fromAngle Starting angle measured in degrees from the
397 * positive x-axis.
398 * @param {number} extent The span of the arc in degrees.
399 * @return {!goog.math.Path} The path itself.
400 */
401goog.math.Path.prototype.arcTo = function(rx, ry, fromAngle, extent) {
402 var cx = this.currentPoint_[0] - goog.math.angleDx(fromAngle, rx);
403 var cy = this.currentPoint_[1] - goog.math.angleDy(fromAngle, ry);
404 var ex = cx + goog.math.angleDx(fromAngle + extent, rx);
405 var ey = cy + goog.math.angleDy(fromAngle + extent, ry);
406 this.segments_.push(goog.math.Path.Segment.ARCTO);
407 this.count_.push(1);
408 this.arguments_.push(rx, ry, fromAngle, extent, ex, ey);
409 this.simple_ = false;
410 this.currentPoint_ = [ex, ey];
411 return this;
412};

415/**
416 * Same as {@code arcTo}, but approximates the arc using bezier curves.
417.* As a result, this method does not affect the simplified status of this path.
418 * The algorithm is adapted from {@code java.awt.geom.ArcIterator}.
419 *
420 * @param {number} rx Radius of ellipse on x axis.
421 * @param {number} ry Radius of ellipse on y axis.
422 * @param {number} fromAngle Starting angle measured in degrees from the
423 * positive x-axis.
424 * @param {number} extent The span of the arc in degrees.
425 * @return {!goog.math.Path} The path itself.
426 */
427goog.math.Path.prototype.arcToAsCurves = function(
428 rx, ry, fromAngle, extent) {
429 var cx = this.currentPoint_[0] - goog.math.angleDx(fromAngle, rx);
430 var cy = this.currentPoint_[1] - goog.math.angleDy(fromAngle, ry);
431 var extentRad = goog.math.toRadians(extent);
432 var arcSegs = Math.ceil(Math.abs(extentRad) / Math.PI * 2);
433 var inc = extentRad / arcSegs;
434 var angle = goog.math.toRadians(fromAngle);
435 for (var j = 0; j < arcSegs; j++) {
436 var relX = Math.cos(angle);
437 var relY = Math.sin(angle);
438 var z = 4 / 3 * Math.sin(inc / 2) / (1 + Math.cos(inc / 2));
439 var c0 = cx + (relX - z * relY) * rx;
440 var c1 = cy + (relY + z * relX) * ry;
441 angle += inc;
442 relX = Math.cos(angle);
443 relY = Math.sin(angle);
444 this.curveTo(c0, c1,
445 cx + (relX + z * relY) * rx,
446 cy + (relY - z * relX) * ry,
447 cx + relX * rx,
448 cy + relY * ry);
449 }
450 return this;
451};

454/**
455 * Iterates over the path calling the supplied callback once for each path
456 * segment. The arguments to the callback function are the segment type and
457 * an array of its arguments.
458 *
459 * The {@code LINETO} and {@code CURVETO} arrays can contain multiple
460 * segments of the same type. The number of segments is the length of the
461 * array divided by the segment length (2 for lines, 6 for curves).
462 *
463 * As a convenience the {@code ARCTO} segment also includes the end point as the
464 * last two arguments: {@code rx, ry, fromAngle, extent, x, y}.
465 *
466 * @param {function(!goog.math.Path.Segment, !Array<number>)} callback
467 * The function to call with each path segment.
468 */
469goog.math.Path.prototype.forEachSegment = function(callback) {
470 var points = this.arguments_;
471 var index = 0;
472 for (var i = 0, length = this.segments_.length; i < length; i++) {
473 var seg = this.segments_[i];
474 var n = goog.math.Path.segmentArgCounts_[seg] * this.count_[i];
475 callback(seg, points.slice(index, index + n));
476 index += n;
477 }
478};

535/**
536 * Creates a copy of the given path, replacing {@code arcTo} with
537 * {@code arcToAsCurves}. The resulting path is simplified and can
538 * be transformed.
539 *
540 * @param {!goog.math.Path} src The path to simplify.
541 * @return {!goog.math.Path} A new simplified path.
542 */
543goog.math.Path.createSimplifiedPath = function(src) {
544 if (src.isSimple()) {
545 return src.clone();
546 }
547 var path = new goog.math.Path();
548 src.forEachSegment(function(segment, args) {
549 goog.math.Path.simplifySegmentMap_[segment].apply(path, args);
550 });
551 return path;
552};

570/**
571 * Transforms the path. Only simple paths are transformable. Attempting
572 * to transform a non-simple path will throw an error.
573 *
574 * @param {!goog.math.AffineTransform} tx The transformation to perform.
575 * @return {!goog.math.Path} The path itself.
576 */
577goog.math.Path.prototype.transform = function(tx) {
578 if (!this.isSimple()) {
579 throw Error('Non-simple path');
580 }
581 tx.transform(this.arguments_, 0, this.arguments_, 0,
582 this.arguments_.length / 2);
583 if (this.closePoint_) {
584 tx.transform(this.closePoint_, 0, this.closePoint_, 0, 1);
585 }
586 if (this.currentPoint_ && this.closePoint_ != this.currentPoint_) {
587 tx.transform(this.currentPoint_, 0, this.currentPoint_, 0, 1);
588 }
589 return this;
590};
