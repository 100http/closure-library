260/**
261 * Returns a formatted string for the event.
262 * @param {number} startTime The start time of the trace to generate relative
263 * times.
264 * @param {number} prevTime The completion time of the previous event or -1.
265 * @param {string} indent Extra indent for the message
266 * if there was no previous event.
267 * @return {string} The formatted tracer string.
268 */
269goog.debug.Trace_.Event_.prototype.toTraceString = function(startTime, prevTime,
270 indent) {
271 var sb = [];
272
273 if (prevTime == -1) {
274 sb.push(' ');
275 } else {
276 sb.push(goog.debug.Trace_.longToPaddedString_(this.eventTime - prevTime));
277 }
278
279 sb.push(' ', goog.debug.Trace_.formatTime_(this.eventTime - startTime));
280 if (this.eventType == goog.debug.Trace_.EventType.START) {
281 sb.push(' Start ');
282 } else if (this.eventType == goog.debug.Trace_.EventType.STOP) {
283 sb.push(' Done ');
284 var delta = this.stopTime - this.startTime;
285 sb.push(goog.debug.Trace_.longToPaddedString_(delta), ' ms ');
286 } else {
287 sb.push(' Comment ');
288 }
289
290 sb.push(indent, this);
291 if (this.totalVarAlloc > 0) {
292 sb.push('[VarAlloc ', this.totalVarAlloc, '] ');
293 }
294 return sb.join('');
295};

341/**
342 * Resets the trace.
343 * @param {number} defaultThreshold The default threshold below which the
344 * tracer output will be supressed. Can be overridden on a per-Tracer basis.
345 */
346goog.debug.Trace_.prototype.reset = function(defaultThreshold) {
347 this.defaultThreshold_ = defaultThreshold;
348
349 for (var i = 0; i < this.events_.length; i++) {
350 var id = /** @type {Object} */ (this.eventPool_).id;
351 if (id) {
352 this.idPool_.releaseObject(id);
353 }
354 this.eventPool_.releaseObject(this.events_[i]);
355 }
356
357 this.events_.length = 0;
358 this.outstandingEvents_.clear();
359 this.startTime_ = goog.debug.Trace_.now();
360 this.tracerOverheadStart_ = 0;
361 this.tracerOverheadEnd_ = 0;
362 this.tracerOverheadComment_ = 0;
363 this.tracerCount_ = 0;
364 this.commentCount_ = 0;
365
366 var keys = this.stats_.getKeys();
367 for (var i = 0; i < keys.length; i++) {
368 var key = keys[i];
369 var stat = this.stats_.get(key);
370 stat.count = 0;
371 stat.time = 0;
372 stat.varAlloc = 0;
373 this.statPool_.releaseObject(/** @type {Object} */ (stat));
374 }
375 this.stats_.clear();
376};

379/**
380 * Starts a tracer
381 * @param {string} comment A comment used to identify the tracer. Does not
382 * need to be unique.
383 * @param {string=} opt_type Type used to identify the tracer. If a Trace is
384 * given a type (the first argument to the constructor) and multiple Traces
385 * are done on that type then a "TOTAL line will be produced showing the
386 * total number of traces and the sum of the time
387 * ("TOTAL Database 2 (37 ms)" in our example). These traces should be
388 * mutually exclusive or else the sum won't make sense (the time will
389 * be double counted if the second starts before the first ends).
390 * @return {number} The identifier for the tracer that should be passed to the
391 * the stopTracer method.
392 */
393goog.debug.Trace_.prototype.startTracer = function(comment, opt_type) {
394 var tracerStartTime = goog.debug.Trace_.now();
395 var varAlloc = this.getTotalVarAlloc();
396 var outstandingEventCount = this.outstandingEvents_.getCount();
397 if (this.events_.length + outstandingEventCount > this.MAX_TRACE_SIZE) {
398 goog.log.warning(this.logger_,
399 'Giant thread trace. Clearing to avoid memory leak.');
400 // This is the more likely case. This usually means that we
401 // either forgot to clear the trace or else we are performing a
402 // very large number of events
403 if (this.events_.length > this.MAX_TRACE_SIZE / 2) {
404 for (var i = 0; i < this.events_.length; i++) {
405 var event = this.events_[i];
406 if (event.id) {
407 this.idPool_.releaseObject(event.id);
408 }
409 this.eventPool_.releaseObject(event);
410 }
411 this.events_.length = 0;
412 }
413
414 // This is less likely and probably indicates that a lot of traces
415 // aren't being closed. We want to avoid unnecessarily clearing
416 // this though in case the events do eventually finish.
417 if (outstandingEventCount > this.MAX_TRACE_SIZE / 2) {
418 this.outstandingEvents_.clear();
419 }
420 }
421
422 goog.debug.Logger.logToProfilers('Start : ' + comment);
423
424 var event = /** @type {goog.debug.Trace_.Event_} */ (
425 this.eventPool_.getObject());
426 event.totalVarAlloc = varAlloc;
427 event.eventType = goog.debug.Trace_.EventType.START;
428 event.id = Number(this.idPool_.getObject());
429 event.comment = comment;
430 event.type = opt_type;
431 this.events_.push(event);
432 this.outstandingEvents_.set(String(event.id), event);
433 this.tracerCount_++;
434 var now = goog.debug.Trace_.now();
435 event.startTime = event.eventTime = now;
436 this.tracerOverheadStart_ += now - tracerStartTime;
437 return event.id;
438};

441/**
442 * Stops a tracer
443 * @param {number|undefined|null} id The id of the tracer that is ending.
444 * @param {number=} opt_silenceThreshold Threshold below which the tracer is
445 * silenced.
446 * @return {?number} The elapsed time for the tracer or null if the tracer
447 * identitifer was not recognized.
448 */
449goog.debug.Trace_.prototype.stopTracer = function(id, opt_silenceThreshold) {
450 // this used to call goog.isDef(opt_silenceThreshold) but that causes an
451 // object allocation in IE for some reason (doh!). The following code doesn't
452 // cause an allocation
453 var now = goog.debug.Trace_.now();
454 var silenceThreshold;
455 if (opt_silenceThreshold === 0) {
456 silenceThreshold = 0;
457 } else if (opt_silenceThreshold) {
458 silenceThreshold = opt_silenceThreshold;
459 } else {
460 silenceThreshold = this.defaultThreshold_;
461 }
462
463 var startEvent = this.outstandingEvents_.get(String(id));
464 if (startEvent == null) {
465 return null;
466 }
467
468 this.outstandingEvents_.remove(String(id));
469
470 var stopEvent;
471 var elapsed = now - startEvent.startTime;
472 if (elapsed < silenceThreshold) {
473 var count = this.events_.length;
474 for (var i = count - 1; i >= 0; i--) {
475 var nextEvent = this.events_[i];
476 if (nextEvent == startEvent) {
477 this.events_.splice(i, 1);
478 this.idPool_.releaseObject(startEvent.id);
479 this.eventPool_.releaseObject(/** @type {Object} */ (startEvent));
480 break;
481 }
482 }
483
484 } else {
485 stopEvent = /** @type {goog.debug.Trace_.Event_} */ (
486 this.eventPool_.getObject());
487 stopEvent.eventType = goog.debug.Trace_.EventType.STOP;
488 stopEvent.startTime = startEvent.startTime;
489 stopEvent.comment = startEvent.comment;
490 stopEvent.type = startEvent.type;
491 stopEvent.stopTime = stopEvent.eventTime = now;
492
493 this.events_.push(stopEvent);
494 }
495
496 var type = startEvent.type;
497 var stat = null;
498 if (type) {
499 stat = this.getStat_(type);
500 stat.count++;
501 stat.time += elapsed;
502 }
503 if (stopEvent) {
504 goog.debug.Logger.logToProfilers('Stop : ' + stopEvent.comment);
505
506 stopEvent.totalVarAlloc = this.getTotalVarAlloc();
507
508 if (stat) {
509 stat.varAlloc += (stopEvent.totalVarAlloc - startEvent.totalVarAlloc);
510 }
511 }
512 var tracerFinishTime = goog.debug.Trace_.now();
513 this.tracerOverheadEnd_ += tracerFinishTime - now;
514 return elapsed;
515};

542/**
543 * Adds a comment to the trace. Makes it possible to see when a specific event
544 * happened in relation to the traces.
545 * @param {string} comment A comment that is inserted into the trace.
546 * @param {?string=} opt_type Type used to identify the tracer. If a comment is
547 * given a type and multiple comments are done on that type then a "TOTAL
548 * line will be produced showing the total number of comments of that type.
549 * @param {?number=} opt_timeStamp The timestamp to insert the comment. If not
550 * specified, the current time wil be used.
551 */
552goog.debug.Trace_.prototype.addComment = function(comment, opt_type,
553 opt_timeStamp) {
554 var now = goog.debug.Trace_.now();
555 var timeStamp = opt_timeStamp ? opt_timeStamp : now;
556
557 var eventComment = /** @type {goog.debug.Trace_.Event_} */ (
558 this.eventPool_.getObject());
559 eventComment.eventType = goog.debug.Trace_.EventType.COMMENT;
560 eventComment.eventTime = timeStamp;
561 eventComment.type = opt_type;
562 eventComment.comment = comment;
563 eventComment.totalVarAlloc = this.getTotalVarAlloc();
564 this.commentCount_++;
565
566 if (opt_timeStamp) {
567 var numEvents = this.events_.length;
568 for (var i = 0; i < numEvents; i++) {
569 var event = this.events_[i];
570 var eventTime = event.eventTime;
571
572 if (eventTime > timeStamp) {
573 goog.array.insertAt(this.events_, eventComment, i);
574 break;
575 }
576 }
577 if (i == numEvents) {
578 this.events_.push(eventComment);
579 }
580 } else {
581 this.events_.push(eventComment);
582 }
583
584 var type = eventComment.type;
585 if (type) {
586 var stat = this.getStat_(type);
587 stat.count++;
588 }
589
590 this.tracerOverheadComment_ += goog.debug.Trace_.now() - now;
591};

623/**
624 * Returns a formatted string that describes the thread trace.
625 * @return {string} A formatted string.
626 * @override
627 */
628goog.debug.Trace_.prototype.toString = function() {
629 var sb = [];
630 var etime = -1;
631 var indent = [];
632 for (var i = 0; i < this.events_.length; i++) {
633 var e = this.events_[i];
634 if (e.eventType == goog.debug.Trace_.EventType.STOP) {
635 indent.pop();
636 }
637 sb.push(' ', e.toTraceString(this.startTime_, etime, indent.join('')));
638 etime = e.eventTime;
639 sb.push('\n');
640 if (e.eventType == goog.debug.Trace_.EventType.START) {
641 indent.push('| ');
642 }
643 }
644
645 if (this.outstandingEvents_.getCount() != 0) {
646 var now = goog.debug.Trace_.now();
647
648 sb.push(' Unstopped timers:\n');
649 goog.iter.forEach(this.outstandingEvents_, function(startEvent) {
650 sb.push(' ', startEvent, ' (', now - startEvent.startTime,
651 ' ms, started at ',
652 goog.debug.Trace_.formatTime_(startEvent.startTime),
653 ')\n');
654 });
655 }
656
657 var statKeys = this.stats_.getKeys();
658 for (var i = 0; i < statKeys.length; i++) {
659 var stat = this.stats_.get(statKeys[i]);
660 if (stat.count > 1) {
661 sb.push(' TOTAL ', stat, '\n');
662 }
663 }
664
665 sb.push('Total tracers created ', this.tracerCount_, '\n',
666 'Total comments created ', this.commentCount_, '\n',
667 'Overhead start: ', this.tracerOverheadStart_, ' ms\n',
668 'Overhead end: ', this.tracerOverheadEnd_, ' ms\n',
669 'Overhead comment: ', this.tracerOverheadComment_, ' ms\n');
670
671 return sb.join('');
672};

693/**
694 * Return the sec.ms part of time (if time = "20:06:11.566", "11.566
695 * @param {number} time The time in MS.
696 * @return {string} A formatted string as sec.ms'.
697 * @private

699goog.debug.Trace_.formatTime_ = function(time) {
700 time = Math.round(time);
701 var sec = (time / 1000) % 60;
702 var ms = time % 1000;
703
704 // TODO their must be a nicer way to get zero padded integers
705 return String(100 + sec).substring(1, 3) + '.' +
706 String(1000 + ms).substring(1, 4);
