15/**
16 * @fileoverview Runtime development CSS Compiler emulation, via javascript.
17 * This class provides an approximation to CSSCompiler's functionality by
18 * hacking the live CSSOM.
19 * This code is designed to be inserted in the DOM immediately after the last
20 * style block in HEAD when in development mode, i.e. you are not using a
21 * running instance of a CSS Compiler to pass your CSS through.
22 */

38/**
39 * A class for solving development CSS issues/emulating the CSS Compiler.
40 * @param {goog.debug.DevCss.UserAgent=} opt_userAgent The user agent, if not
41 * passed in, will be determined using goog.userAgent.
42 * @param {number|string=} opt_userAgentVersion The user agent's version.
43 * If not passed in, will be determined using goog.userAgent.
44 * @throws {Error} When userAgent detection fails.
45 * @constructor
46 * @final
47 */
48goog.debug.DevCss = function(opt_userAgent, opt_userAgentVersion) {
49 if (!opt_userAgent) {
50 // Walks through the known goog.userAgents.
51 if (goog.userAgent.IE) {
52 opt_userAgent = goog.debug.DevCss.UserAgent.IE;
53 } else if (goog.userAgent.GECKO) {
54 opt_userAgent = goog.debug.DevCss.UserAgent.GECKO;
55 } else if (goog.userAgent.WEBKIT) {
56 opt_userAgent = goog.debug.DevCss.UserAgent.WEBKIT;
57 } else if (goog.userAgent.MOBILE) {
58 opt_userAgent = goog.debug.DevCss.UserAgent.MOBILE;
59 } else if (goog.userAgent.OPERA) {
60 opt_userAgent = goog.debug.DevCss.UserAgent.OPERA;
61 }
62 }
63 switch (opt_userAgent) {
64 case goog.debug.DevCss.UserAgent.OPERA:
65 case goog.debug.DevCss.UserAgent.IE:
66 case goog.debug.DevCss.UserAgent.GECKO:
67 case goog.debug.DevCss.UserAgent.FIREFOX:
68 case goog.debug.DevCss.UserAgent.WEBKIT:
69 case goog.debug.DevCss.UserAgent.SAFARI:
70 case goog.debug.DevCss.UserAgent.MOBILE:
71 break;
72 default:
73 throw Error('Could not determine the user agent from known UserAgents');
74 }
75
76 /**
77 * One of goog.debug.DevCss.UserAgent.
78 * @type {string}
79 * @private
80 */
81 this.userAgent_ = opt_userAgent;
82
83 /**
84 * @const @private
85 */
86 this.userAgentTokens_ = {};
87
88 /**
89 * @type {number|string}
90 * @private
91 */
92 this.userAgentVersion_ = opt_userAgentVersion || goog.userAgent.VERSION;
93 this.generateUserAgentTokens_();
94
95 /**
96 * @type {boolean}
97 * @private
98 */
99 this.isIe6OrLess_ = this.userAgent_ == goog.debug.DevCss.UserAgent.IE &&
100 goog.string.compareVersions('7', this.userAgentVersion_) > 0;
101
102 if (this.isIe6OrLess_) {
103 /**
104 * @type {Array<{classNames,combinedClassName,els}>}
105 * @private
106 */
107 this.ie6CombinedMatches_ = [];
108 }
109};

112/**
113 * Rewrites the CSSOM as needed to activate any useragent-specific selectors.
114 * @param {boolean=} opt_enableIe6ReadyHandler If true(the default), and the
115 * userAgent is ie6, we set a document "ready" event handler to walk the DOM
116 * and make combined selector className changes. Having this parameter also
117 * aids unit testing.
118 */
119goog.debug.DevCss.prototype.activateBrowserSpecificCssRules = function(
120 opt_enableIe6ReadyHandler) {
121 var enableIe6EventHandler = goog.isDef(opt_enableIe6ReadyHandler) ?
122 opt_enableIe6ReadyHandler : true;
123 var cssRules = goog.cssom.getAllCssStyleRules();
124
125 for (var i = 0, cssRule; cssRule = cssRules[i]; i++) {
126 this.replaceBrowserSpecificClassNames_(cssRule);
127 }
128
129 // Since we may have manipulated the rules above, we'll have to do a
130 // complete sweep again if we're in IE6. Luckily performance doesn't
131 // matter for this tool.
132 if (this.isIe6OrLess_) {
133 cssRules = goog.cssom.getAllCssStyleRules();
134 for (var i = 0, cssRule; cssRule = cssRules[i]; i++) {
135 this.replaceIe6CombinedSelectors_(cssRule);
136 }
137 }
138
139 // Add an event listener for document ready to rewrite any necessary
140 // combined classnames in IE6.
141 if (this.isIe6OrLess_ && enableIe6EventHandler) {
142 goog.events.listen(document, goog.events.EventType.LOAD, goog.bind(
143 this.addIe6CombinedClassNames_, this));
144 }
145};

180/**
181 * Generates user agent token match strings with comparison and version bits.
182 * For example:
183 * userAgentTokens_.ANY will be like 'GECKO'
184 * userAgentTokens_.LESS_THAN will be like 'GECKO-LT3' etc...
185 * @private
186 */
187goog.debug.DevCss.prototype.generateUserAgentTokens_ = function() {
188 this.userAgentTokens_.ANY = goog.debug.DevCss.CssToken_.USERAGENT +
189 goog.debug.DevCss.CssToken_.SEPARATOR + this.userAgent_;
190 this.userAgentTokens_.EQUALS = this.userAgentTokens_.ANY +
191 goog.debug.DevCss.CssToken_.SEPARATOR;
192 this.userAgentTokens_.LESS_THAN = this.userAgentTokens_.ANY +
193 goog.debug.DevCss.CssToken_.SEPARATOR +
194 goog.debug.DevCss.CssToken_.LESS_THAN;
195 this.userAgentTokens_.LESS_THAN_OR_EQUAL = this.userAgentTokens_.ANY +
196 goog.debug.DevCss.CssToken_.SEPARATOR +
197 goog.debug.DevCss.CssToken_.LESS_THAN_OR_EQUAL;
198 this.userAgentTokens_.GREATER_THAN = this.userAgentTokens_.ANY +
199 goog.debug.DevCss.CssToken_.SEPARATOR +
200 goog.debug.DevCss.CssToken_.GREATER_THAN;
201 this.userAgentTokens_.GREATER_THAN_OR_EQUAL = this.userAgentTokens_.ANY +
202 goog.debug.DevCss.CssToken_.SEPARATOR +
203 goog.debug.DevCss.CssToken_.GREATER_THAN_OR_EQUAL;
204};

207/**
208 * Gets the version number bit from a selector matching userAgentToken.
209 * @param {string} selectorText The selector text of a CSS rule.
210 * @param {string} userAgentToken Includes the LTE/GTE bit to see if it matches.
211 * @return {string|undefined} The version number.
212 * @private
213 */
214goog.debug.DevCss.prototype.getVersionNumberFromSelectorText_ = function(
215 selectorText, userAgentToken) {
216 var regex = new RegExp(userAgentToken + '([\\d\\.]+)');
217 var matches = regex.exec(selectorText);
218 if (matches && matches.length == 2) {
219 return matches[1];
220 }
221};

224/**
225 * Extracts a rule version from the selector text, and if it finds one, calls
226 * compareVersions against it and the passed in token string to provide the
227 * value needed to determine if we have a match or not.
228 * @param {CSSRule} cssRule The rule to test against.
229 * @param {string} token The match token to test against the rule.
230 * @return {!Array|undefined} A tuple with the result of the compareVersions
231 * call and the matched ruleVersion.
232 * @private
233 */
234goog.debug.DevCss.prototype.getRuleVersionAndCompare_ = function(cssRule,
235 token) {
236 if (!cssRule.selectorText.match(token)) {
237 return;
238 }
239 var ruleVersion = this.getVersionNumberFromSelectorText_(
240 cssRule.selectorText, token);
241 if (!ruleVersion) {
242 return;
243 }
244
245 var comparison = goog.string.compareVersions(this.userAgentVersion_,
246 ruleVersion);
247 return [comparison, ruleVersion];
248};

251/**
252 * Replaces a CSS selector if we have matches based on our useragent/version.
253 * Example: With a selector like ".USERAGENT-IE-LTE6 .class { prop: value }" if
254 * we are running IE6 we'll end up with ".class { prop: value }", thereby
255 * "activating" the selector.
256 * @param {CSSRule} cssRule The cssRule to potentially replace.
257 * @private
258 */
259goog.debug.DevCss.prototype.replaceBrowserSpecificClassNames_ = function(
260 cssRule) {
261
262 // If we don't match the browser token, we can stop now.
263 if (!cssRule.selectorText.match(this.userAgentTokens_.ANY)) {
264 return;
265 }
266
267 // We know it will begin as a classname.
268 var additionalRegexString;
269
270 // Tests "Less than or equals".
271 var compared = this.getRuleVersionAndCompare_(cssRule,
272 this.userAgentTokens_.LESS_THAN_OR_EQUAL);
273 if (compared && compared.length) {
274 if (compared[0] > 0) {
275 return;
276 }
277 additionalRegexString = this.userAgentTokens_.LESS_THAN_OR_EQUAL +
278 compared[1];
279 }
280
281 // Tests "Less than".
282 compared = this.getRuleVersionAndCompare_(cssRule,
283 this.userAgentTokens_.LESS_THAN);
284 if (compared && compared.length) {
285 if (compared[0] > -1) {
286 return;
287 }
288 additionalRegexString = this.userAgentTokens_.LESS_THAN + compared[1];
289 }
290
291 // Tests "Greater than or equals".
292 compared = this.getRuleVersionAndCompare_(cssRule,
293 this.userAgentTokens_.GREATER_THAN_OR_EQUAL);
294 if (compared && compared.length) {
295 if (compared[0] < 0) {
296 return;
297 }
298 additionalRegexString = this.userAgentTokens_.GREATER_THAN_OR_EQUAL +
299 compared[1];
300 }
301
302 // Tests "Greater than".
303 compared = this.getRuleVersionAndCompare_(cssRule,
304 this.userAgentTokens_.GREATER_THAN);
305 if (compared && compared.length) {
306 if (compared[0] < 1) {
307 return;
308 }
309 additionalRegexString = this.userAgentTokens_.GREATER_THAN + compared[1];
310 }
311
312 // Tests "Equals".
313 compared = this.getRuleVersionAndCompare_(cssRule,
314 this.userAgentTokens_.EQUALS);
315 if (compared && compared.length) {
316 if (compared[0] != 0) {
317 return;
318 }
319 additionalRegexString = this.userAgentTokens_.EQUALS + compared[1];
320 }
321
322 // If we got to here without generating the additionalRegexString, then
323 // we did not match any of our comparison token strings, and we want a
324 // general browser token replacement.
325 if (!additionalRegexString) {
326 additionalRegexString = this.userAgentTokens_.ANY;
327 }
328
329 // We need to match at least a single whitespace character to know that
330 // we are matching the entire useragent string token.
331 var regexString = '\\.' + additionalRegexString + '\\s+';
332 var re = new RegExp(regexString, 'g');
333
334 var currentCssText = goog.cssom.getCssTextFromCssRule(cssRule);
335
336 // Replacing the token with '' activates the selector for this useragent.
337 var newCssText = currentCssText.replace(re, '');
338
339 if (newCssText != currentCssText) {
340 goog.cssom.replaceCssRule(cssRule, newCssText);
341 }
342};

345/**
346 * Replaces IE6 combined selector rules with a workable development alternative.
347 * IE6 actually parses .class1.class2 {} to simply .class2 {} which is nasty.
348 * To fully support combined selectors in IE6 this function needs to be paired
349 * with a call to replace the relevant DOM elements classNames as well.
350 * @see {this.addIe6CombinedClassNames_}
351 * @param {CSSRule} cssRule The rule to potentially fix.
352 * @private
353 */
354goog.debug.DevCss.prototype.replaceIe6CombinedSelectors_ = function(cssRule) {
355 // This match only ever works in IE because other UA's won't have our
356 // IE6_SELECTOR_TEXT in the cssText property.
357 if (cssRule.style.cssText &&
358 cssRule.style.cssText.match(
359 goog.debug.DevCss.CssToken_.IE6_SELECTOR_TEXT)) {
360 var cssText = goog.cssom.getCssTextFromCssRule(cssRule);
361 var combinedSelectorText = this.getIe6CombinedSelectorText_(cssText);
362 if (combinedSelectorText) {
363 var newCssText = combinedSelectorText + '{' + cssRule.style.cssText + '}';
364 goog.cssom.replaceCssRule(cssRule, newCssText);
365 }
366 }
367};

370/**
371 * Gets the appropriate new combined selector text for IE6.
372 * Also adds an entry onto ie6CombinedMatches_ with relevant info for the
373 * likely following call to walk the DOM and rewrite the class attribute.
374 * Example: With a selector like
375 * ".class2 { -goog-ie6-selector: .class1.class2; prop: value }".
376 * this function will return:
377 * ".class1_class2 { prop: value }".
378 * @param {string} cssText The CSS selector text and css rule text combined.
379 * @return {?string} The rewritten css rule text.
380 * @private
381 */
382goog.debug.DevCss.prototype.getIe6CombinedSelectorText_ = function(cssText) {
383 var regex = new RegExp(goog.debug.DevCss.CssToken_.IE6_SELECTOR_TEXT +
384 '\\s*:\\s*\\"([^\\"]+)\\"', 'gi');
385 var matches = regex.exec(cssText);
386 if (matches) {
387 var combinedSelectorText = matches[1];
388 // To aid in later fixing the DOM, we need to split up the possible
389 // selector groups by commas.
390 var groupedSelectors = combinedSelectorText.split(/\s*\,\s*/);
391 for (var i = 0, selector; selector = groupedSelectors[i]; i++) {
392 // Strips off the leading ".".
393 var combinedClassName = selector.substr(1);
394 var classNames = combinedClassName.split(
395 goog.debug.DevCss.CssToken_.IE6_COMBINED_GLUE);
396 var entry = {
397 classNames: classNames,
398 combinedClassName: combinedClassName,
399 els: []
400 };
401 this.ie6CombinedMatches_.push(entry);
402 }
403 return combinedSelectorText;
404 }
405 return null;
406};

409/**
410 * Adds combined selectors with underscores to make them "work" in IE6.
411 * @see {this.replaceIe6CombinedSelectors_}
412 * @private
413 */
414goog.debug.DevCss.prototype.addIe6CombinedClassNames_ = function() {
415 if (!this.ie6CombinedMatches_.length) {
416 return;
417 }
418 var allEls = document.getElementsByTagName('*');
419 // Match nodes for all classNames.
420 for (var i = 0, classNameEntry; classNameEntry =
421 this.ie6CombinedMatches_[i]; i++) {
422 for (var j = 0, el; el = allEls[j]; j++) {
423 var classNamesLength = classNameEntry.classNames.length;
424 for (var k = 0, className; className = classNameEntry.classNames[k];
425 k++) {
426 if (!goog.dom.classlist.contains(goog.asserts.assert(el), className)) {
427 break;
428 }
429 if (k == classNamesLength - 1) {
430 classNameEntry.els.push(el);
431 }
432 }
433 }
434 // Walks over our matching nodes and fixes them.
435 if (classNameEntry.els.length) {
436 for (var j = 0, el; el = classNameEntry.els[j]; j++) {
437 goog.asserts.assert(el);
438 if (!goog.dom.classlist.contains(el,
439 classNameEntry.combinedClassName)) {
440 goog.dom.classlist.add(el, classNameEntry.combinedClassName);
441 }
442 }
443 }
