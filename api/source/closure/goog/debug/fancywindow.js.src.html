47/**
48 * Provides a Fancy extension to the DebugWindow class. Allows filtering based
49 * on loggers and levels.
50 *
51 * @param {string=} opt_identifier Idenitifier for this logging class.
52 * @param {string=} opt_prefix Prefix pre-pended to messages.
53 * @constructor
54 * @extends {goog.debug.DebugWindow}
55 */
56goog.debug.FancyWindow = function(opt_identifier, opt_prefix) {
57 this.readOptionsFromLocalStorage_();
58 goog.debug.FancyWindow.base(this, 'constructor', opt_identifier, opt_prefix);
59 /** @private {goog.dom.DomHelper} */
60 this.dh_ = null;
61};
62goog.inherits(goog.debug.FancyWindow, goog.debug.DebugWindow);

65/**
66 * Constant indicating if we are able to use localStorage to persist filters
67 * @type {boolean}
68 */
69goog.debug.FancyWindow.HAS_LOCAL_STORE = (function() {
70 /** @preserveTry */
71 try {
72 return !!window['localStorage'].getItem;
73 } catch (e) {}
74 return false;
75})();

85/** @override */
86goog.debug.FancyWindow.prototype.writeBufferToLog = function() {
87 this.lastCall = goog.now();
88 if (this.hasActiveWindow()) {
89 var logel = this.dh_.getElement('log');
90
91 // Work out if scrolling is needed before we add the content
92 var scroll =
93 logel.scrollHeight - (logel.scrollTop + logel.offsetHeight) <= 100;
94
95 for (var i = 0; i < this.outputBuffer.length; i++) {
96 var div = this.dh_.createDom(goog.dom.TagName.DIV, 'logmsg');
97 goog.dom.safe.setInnerHtml(div, this.outputBuffer[i]);
98 logel.appendChild(div);
99 }
100 this.outputBuffer.length = 0;
101 this.resizeStuff_();
102
103 if (scroll) {
104 logel.scrollTop = logel.scrollHeight;
105 }
106 }
107};

110/** @override */
111goog.debug.FancyWindow.prototype.writeInitialDocument = function() {
112 if (!this.hasActiveWindow()) {
113 return;
114 }
115
116 var doc = this.win.document;
117 doc.open();
118 goog.dom.safe.documentWrite(doc, this.getHtml_());
119 doc.close();
120
121 (goog.userAgent.IE ? doc.body : this.win).onresize =
122 goog.bind(this.resizeStuff_, this);
123
124 // Create a dom helper for the logging window
125 this.dh_ = new goog.dom.DomHelper(doc);
126
127 // Don't use events system to reduce dependencies
128 this.dh_.getElement('openbutton').onclick =
129 goog.bind(this.openOptions_, this);
130 this.dh_.getElement('closebutton').onclick =
131 goog.bind(this.closeOptions_, this);
132 this.dh_.getElement('clearbutton').onclick =
133 goog.bind(this.clear, this);
134 this.dh_.getElement('exitbutton').onclick =
135 goog.bind(this.exit_, this);
136
137 this.writeSavedMessages();
138};

141/**
142 * Show the options menu.
143 * @return {boolean} false.
144 * @private
145 */
146goog.debug.FancyWindow.prototype.openOptions_ = function() {
147 var el = goog.asserts.assert(this.dh_.getElement('optionsarea'));
148 goog.dom.safe.setInnerHtml(el, goog.html.SafeHtml.EMPTY);
149
150 var loggers = goog.debug.FancyWindow.getLoggers_();
151 var dh = this.dh_;
152 for (var i = 0; i < loggers.length; i++) {
153 var logger = loggers[i];
154 var curlevel = logger.getLevel() ? logger.getLevel().name : 'INHERIT';
155 var div = dh.createDom(goog.dom.TagName.DIV, {},
156 this.getDropDown_('sel' + logger.getName(), curlevel),
157 dh.createDom(goog.dom.TagName.SPAN, {}, logger.getName() || '(root)'));
158 el.appendChild(div);
159 }
160
161 this.dh_.getElement('options').style.display = 'block';
162 return false;
163};

166/**
167 * Make a drop down for the log levels.
168 * @param {string} id Logger id.
169 * @param {string} selected What log level is currently selected.
170 * @return {Element} The newly created 'select' DOM element.
171 * @private
172 */
173goog.debug.FancyWindow.prototype.getDropDown_ = function(id, selected) {
174 var dh = this.dh_;
175 var sel = dh.createDom(goog.dom.TagName.SELECT, {'id': id});
176 var levels = goog.debug.Logger.Level.PREDEFINED_LEVELS;
177 for (var i = 0; i < levels.length; i++) {
178 var level = levels[i];
179 var option = dh.createDom(goog.dom.TagName.OPTION, {}, level.name);
180 if (selected == level.name) {
181 option.selected = true;
182 }
183 sel.appendChild(option);
184 }
185 sel.appendChild(dh.createDom(goog.dom.TagName.OPTION,
186 {'selected': selected == 'INHERIT'}, 'INHERIT'));
187 return sel;
188};

191/**
192 * Close the options menu.
193 * @return {boolean} The value false.
194 * @private
195 */
196goog.debug.FancyWindow.prototype.closeOptions_ = function() {
197 this.dh_.getElement('options').style.display = 'none';
198 var loggers = goog.debug.FancyWindow.getLoggers_();
199 var dh = this.dh_;
200 for (var i = 0; i < loggers.length; i++) {
201 var logger = loggers[i];
202 var sel = dh.getElement('sel' + logger.getName());
203 var level = sel.options[sel.selectedIndex].text;
204 if (level == 'INHERIT') {
205 logger.setLevel(null);
206 } else {
207 logger.setLevel(goog.debug.Logger.Level.getPredefinedLevel(level));
208 }
209 }
210 this.writeOptionsToLocalStorage_();
211 return false;
212};

275goog.debug.FancyWindow.prototype.getHtml_ = function() {
276 var SafeHtml = goog.html.SafeHtml;
277 var head = SafeHtml.create('head', {}, SafeHtml.concat(
278 SafeHtml.create('title', {}, 'Logging: ' + this.identifier),
279 SafeHtml.createStyle(this.getStyleRules())));
280
281 var body = SafeHtml.create('body', {}, SafeHtml.concat(
282 SafeHtml.create('div',
283 {'id': 'log', 'style': goog.string.Const.from('overflow:auto')}),
284 SafeHtml.create('div', {'id': 'head'}, SafeHtml.concat(
285 SafeHtml.create('p', {},
286 SafeHtml.create('b', {}, 'Logging: ' + this.identifier)),
287 SafeHtml.create('p', {}, this.welcomeMessage),
288 SafeHtml.create('span', {'id': 'clearbutton'}, 'clear'),
289 SafeHtml.create('span', {'id': 'exitbutton'}, 'exit'),
290 SafeHtml.create('span', {'id': 'openbutton'}, 'options'))),
291 SafeHtml.create('div', {'id': 'options'}, SafeHtml.concat(
292 SafeHtml.create('big', {},
293 SafeHtml.create('b', {}, 'Options:')),
294 SafeHtml.create('div', {'id': 'optionsarea'}),
295 SafeHtml.create('span', {'id': 'closebutton'}, 'save and close')))));
296
297 return SafeHtml.create('html', {}, SafeHtml.concat(head, body));

301/**
302 * Write logger levels to localStorage if possible.
303 * @private
304 */
305goog.debug.FancyWindow.prototype.writeOptionsToLocalStorage_ = function() {
306 if (!goog.debug.FancyWindow.HAS_LOCAL_STORE) {
307 return;
308 }
309 var loggers = goog.debug.FancyWindow.getLoggers_();
310 var storedKeys = goog.debug.FancyWindow.getStoredKeys_();
311 for (var i = 0; i < loggers.length; i++) {
312 var key = goog.debug.FancyWindow.LOCAL_STORE_PREFIX + loggers[i].getName();
313 var level = loggers[i].getLevel();
314 if (key in storedKeys) {
315 if (!level) {
316 window.localStorage.removeItem(key);
317 } else if (window.localStorage.getItem(key) != level.name) {
318 window.localStorage.setItem(key, level.name);
319 }
320 } else if (level) {
321 window.localStorage.setItem(key, level.name);
322 }
323 }
324};

327/**
328 * Sync logger levels with any values stored in localStorage.
329 * @private
330 */
331goog.debug.FancyWindow.prototype.readOptionsFromLocalStorage_ = function() {
332 if (!goog.debug.FancyWindow.HAS_LOCAL_STORE) {
333 return;
334 }
335 var storedKeys = goog.debug.FancyWindow.getStoredKeys_();
336 for (var key in storedKeys) {
337 var loggerName = key.replace(goog.debug.FancyWindow.LOCAL_STORE_PREFIX, '');
338 var logger = goog.debug.LogManager.getLogger(loggerName);
339 var curLevel = logger.getLevel();
340 var storedLevel = window.localStorage.getItem(key).toString();
341 if (!curLevel || curLevel.toString() != storedLevel) {
342 logger.setLevel(goog.debug.Logger.Level.getPredefinedLevel(storedLevel));
343 }
344 }
345};

348/**
349 * Helper function to create a list of locally stored keys. Used to avoid
350 * expensive localStorage.getItem() calls.
351 * @return {!Object} List of keys.
352 * @private
353 */
354goog.debug.FancyWindow.getStoredKeys_ = function() {
355 var storedKeys = {};
356 for (var i = 0, len = window.localStorage.length; i < len; i++) {
357 var key = window.localStorage.key(i);
358 if (key != null && goog.string.startsWith(
359 key, goog.debug.FancyWindow.LOCAL_STORE_PREFIX)) {
360 storedKeys[key] = true;
361 }
362 }
363 return storedKeys;
364};

367/**
368 * Gets a sorted array of all the loggers registered.
369 * @return {!Array<!goog.debug.Logger>} Array of logger instances.
370 * @private
371 */
372goog.debug.FancyWindow.getLoggers_ = function() {
373 var loggers = goog.object.getValues(goog.debug.LogManager.getLoggers());
374
375 /**
376 * @param {!goog.debug.Logger} a
377 * @param {!goog.debug.Logger} b
378 * @return {number}
379 */
380 var loggerSort = function(a, b) {
381 return goog.array.defaultCompare(a.getName(), b.getName());
382 };
383 goog.array.sort(loggers, loggerSort);
384 return loggers;
385};
