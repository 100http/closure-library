15/**
16 * @fileoverview goog.dom.animationFrame permits work to be done in-sync with
17 * the render refresh rate of the browser and to divide work up globally based
18 * on whether the intent is to measure or to mutate the DOM. The latter avoids
19 * repeated style recalculation which can be really slow.
20 *
21 * Goals of the API:
22 * <ul>
23 * <li>Make it easy to schedule work for the next animation frame.
24 * <li>Make it easy to only do work once per animation frame, even if two
25 * events fire that trigger the same work.
26 * <li>Make it easy to do all work in two phases to avoid repeated style
27 * recalculation caused by interleaved reads and writes.
28 * <li>Avoid creating closures per schedule operation.
29 * </ul>
30 *
31 *
32 * Programmatic:
33 * <pre>
34 * var animationTask = goog.dom.animationFrame.createTask({
35 * measure: function(state) {
36 * state.width = goog.style.getSize(elem).width;
37 * this.animationTask();
38 * },
39 * mutate: function(state) {
40 * goog.style.setWidth(elem, Math.floor(state.width / 2));
41 * }
42 * }, this);
43 * });
44 * </pre>
45 *
46 * See also
47 * https://developer.mozilla.org/en-US/docs/Web/API/window.requestAnimationFrame
48 */

142/**
143 * Returns a function that schedules the two passed-in functions to be run upon
144 * the next animation frame. Calling the function again during the same
145 * animation frame does nothing.
146 *
147 * The function under the "measure" key will run first and together with all
148 * other functions scheduled under this key and the function under "mutate" will
149 * run after that.
150 *
151 * @param {{
152 * measure: (function(this:THIS, !goog.dom.animationFrame.State)|undefined),
153 * mutate: (function(this:THIS, !goog.dom.animationFrame.State)|undefined)
154 * }} spec
155 * @param {THIS=} opt_context Context in which to run the function.
156 * @return {function(...?)}
157 * @template THIS
158 */
159goog.dom.animationFrame.createTask = function(spec, opt_context) {
160 var id = goog.dom.animationFrame.taskId_++;
161 var measureTask = {
162 id: id,
163 fn: spec.measure,
164 context: opt_context
165 };
166 var mutateTask = {
167 id: id,
168 fn: spec.mutate,
169 context: opt_context
170 };
171
172 var taskSet = {
173 measureTask: measureTask,
174 mutateTask: mutateTask,
175 state: {},
176 args: undefined,
177 isScheduled: false
178 };
179
180 return function() {
181 // Default the context to the one that was used to call the tasks scheduler
182 // (this function).
183 if (!opt_context) {
184 measureTask.context = this;
185 mutateTask.context = this;
186 }
187
188 // Save args and state.
189 if (arguments.length > 0) {
190 // The state argument goes last. That is kinda horrible but compatible
191 // with {@see wiz.async.method}.
192 if (!taskSet.args) {
193 taskSet.args = [];
194 }
195 taskSet.args.length = 0;
196 taskSet.args.push.apply(taskSet.args, arguments);
197 taskSet.args.push(taskSet.state);
198 } else {
199 if (!taskSet.args || taskSet.args.length == 0) {
200 taskSet.args = [taskSet.state];
201 } else {
202 taskSet.args[0] = taskSet.state;
203 taskSet.args.length = 1;
204 }
205 }
206 if (!taskSet.isScheduled) {
207 taskSet.isScheduled = true;
208 var tasksArray = goog.dom.animationFrame.tasks_[
209 goog.dom.animationFrame.doubleBufferIndex_];
210 tasksArray.push(taskSet);
211 }
212 goog.dom.animationFrame.requestAnimationFrame_();
213 };
214};

217/**
218 * Run scheduled tasks.
219 * @private
220 */
221goog.dom.animationFrame.runTasks_ = function() {
222 goog.dom.animationFrame.running_ = true;
223 goog.dom.animationFrame.requestedFrame_ = false;
224 var tasksArray = goog.dom.animationFrame
225 .tasks_[goog.dom.animationFrame.doubleBufferIndex_];
226 var taskLength = tasksArray.length;
227
228 // During the runTasks_, if there is a recursive call to queue up more
229 // task(s) for the next frame, we use double-buffering for that.
230 goog.dom.animationFrame.doubleBufferIndex_ =
231 (goog.dom.animationFrame.doubleBufferIndex_ + 1) % 2;
232
233 var task;
234
235 // Run all the measure tasks first.
236 for (var i = 0; i < taskLength; ++i) {
237 task = tasksArray[i];
238 var measureTask = task.measureTask;
239 task.isScheduled = false;
240 if (measureTask.fn) {
241 // TODO (perumaal): Handle any exceptions thrown by the lambda.
242 measureTask.fn.apply(measureTask.context, task.args);
243 }
244 }
245
246 // Run the mutate tasks next.
247 for (var i = 0; i < taskLength; ++i) {
248 task = tasksArray[i];
249 var mutateTask = task.mutateTask;
250 task.isScheduled = false;
251 if (mutateTask.fn) {
252 // TODO (perumaal): Handle any exceptions thrown by the lambda.
253 mutateTask.fn.apply(mutateTask.context, task.args);
254 }
255
256 // Clear state for next vsync.
257 task.state = {};
258 }
259
260 // Clear the tasks array as we have finished processing all the tasks.
261 tasksArray.length = 0;
262 goog.dom.animationFrame.running_ = false;
263};

276/**
277 * Request {@see goog.dom.animationFrame.runTasks_} to be called upon the
278 * next animation frame if we haven't done so already.
279 * @private
280 */
281goog.dom.animationFrame.requestAnimationFrame_ = function() {
282 if (goog.dom.animationFrame.requestedFrame_) {
283 return;
284 }
285 goog.dom.animationFrame.requestedFrame_ = true;
286 window.requestAnimationFrame(goog.dom.animationFrame.runTasks_);
