43/**
44 * Safe source for a new blank iframe that may not cause a new load of the
45 * iframe. This is different from {@code goog.dom.iframe.BLANK_SOURCE} in that
46 * it will allow an iframe to be loaded synchronously in more browsers, notably
47 * Gecko, following the javascript protocol spec.
48 *
49 * NOTE: This should not be used to replace the source of an existing iframe.
50 * The new src value will be ignored, per the spec.
51 *
52 * Due to cross-browser differences, the load is not guaranteed to be
53 * synchronous. If code depends on the load of the iframe,
54 * then {@code goog.net.IframeLoadMonitor} or a similar technique should be
55 * used.
56 *
57 * According to
58 * http://www.whatwg.org/specs/web-apps/current-work/multipage/webappapis.html#javascript-protocol
59 * the 'javascript:""' URL should trigger a new load of the iframe, which may be
60 * asynchronous. A void src, such as 'javascript:undefined', does not change
61 * the browsing context document's, and thus should not trigger another load.
62 *
63 * Intentionally not about:blank, which also triggers a load.
64 *
65 * NOTE: 'javascript:' URL handling spec compliance varies per browser. IE
66 * throws an error with 'javascript:undefined'. Webkit browsers will reload the
67 * iframe when setting this source on an existing iframe.
68 *
69 * @type {string}
70 */
71goog.dom.iframe.BLANK_SOURCE_NEW_FRAME = goog.userAgent.IE ?
72 'javascript:""' :
73 'javascript:undefined';

84/**
85 * Creates a completely blank iframe element.
86 *
87 * The iframe will not caused mixed-content warnings for IE6 under HTTPS.
88 * The iframe will also have no borders or padding, so that the styled width
89 * and height will be the actual width and height of the iframe.
90 *
91 * This function currently only attempts to create a blank iframe. There
92 * are no guarantees to the contents of the iframe or whether it is rendered
93 * in quirks mode.
94 *
95 * @param {goog.dom.DomHelper} domHelper The dom helper to use.
96 * @param {!goog.html.SafeStyle|string=} opt_styles CSS styles for the
97 * iframe. If possible pass a SafeStyle; string is supported for
98 * backwards-compatibility only and uses goog.html.legacyconversions.
99 * @return {!HTMLIFrameElement} A completely blank iframe.
100 */
101goog.dom.iframe.createBlank = function(domHelper, opt_styles) {
102 var styles;
103 if (goog.isString(opt_styles)) {
104 styles = goog.html.legacyconversions.safeStyleFromString(opt_styles)
105 .getTypedStringValue();
106 } else if (opt_styles instanceof goog.html.SafeStyle) {
107 // SafeStyle has to be converted back to a string for now, since there's
108 // no safe alternative to createDom().
109 styles = goog.html.SafeStyle.unwrap(opt_styles);
110 } else { // undefined.
111 styles = '';
112 }
113 return /** @type {!HTMLIFrameElement} */ (domHelper.createDom('iframe', {
114 'frameborder': 0,
115 // Since iframes are inline elements, we must align to bottom to
116 // compensate for the line descent.
117 'style': goog.dom.iframe.STYLES_ + styles,
118 'src': goog.dom.iframe.BLANK_SOURCE
119 }));
120};

123/**
124 * Writes the contents of a blank iframe that has already been inserted
125 * into the document. If possible use {@link #writeSafeContent},
126 * this function exists for backwards-compatibility only and uses
127 * goog.html.legacyconversions.
128 * @param {!HTMLIFrameElement} iframe An iframe with no contents, such as
129 * one created by goog.dom.iframe.createBlank, but already appended to
130 * a parent document.
131 * @param {string} content Content to write to the iframe, from doctype to
132 * the HTML close tag.
133 */
134goog.dom.iframe.writeContent = function(iframe, content) {
135 goog.dom.iframe.writeSafeContent(
136 iframe, goog.html.legacyconversions.safeHtmlFromString(content));
137};

140/**
141 * Writes the contents of a blank iframe that has already been inserted
142 * into the document.
143 * @param {!HTMLIFrameElement} iframe An iframe with no contents, such as
144 * one created by {@link #createBlank}, but already appended to
145 * a parent document.
146 * @param {!goog.html.SafeHtml} content Content to write to the iframe,
147 * from doctype to the HTML close tag.
148 */
149goog.dom.iframe.writeSafeContent = function(iframe, content) {
150 var doc = goog.dom.getFrameContentDocument(iframe);
151 doc.open();
152 goog.dom.safe.documentWrite(doc, content);
153 doc.close();
154};

159/**
160 * Creates a same-domain iframe containing preloaded content.
161 *
162 * This is primarily useful for DOM sandboxing. One use case is to embed
163 * a trusted Javascript app with potentially conflicting CSS styles. The
164 * second case is to reduce the cost of layout passes by the browser -- for
165 * example, you can perform sandbox sizing of characters in an iframe while
166 * manipulating a heavy DOM in the main window. The iframe and parent frame
167 * can access each others' properties and functions without restriction.
168 *
169 * @param {!Element} parentElement The parent element in which to append the
170 * iframe.
171 * @param {!goog.html.SafeHtml|string=} opt_headContents Contents to go into
172 * the iframe's head. If possible pass a SafeHtml; string is supported for
173 * backwards-compatibility only and uses goog.html.legacyconversions.
174 * @param {!goog.html.SafeHtml|string=} opt_bodyContents Contents to go into
175 * the iframe's body. If possible pass a SafeHtml; string is supported for
176 * backwards-compatibility only and uses goog.html.legacyconversions.
177 * @param {!goog.html.SafeStyle|string=} opt_styles CSS styles for the iframe
178 * itself, before adding to the parent element. If possible pass a
179 * SafeStyle; string is supported for backwards-compatibility only and
180 * uses goog.html.legacyconversions.
181 * @param {boolean=} opt_quirks Whether to use quirks mode (false by default).
182 * @return {!HTMLIFrameElement} An iframe that has the specified contents.
183 */
184goog.dom.iframe.createWithContent = function(
185 parentElement, opt_headContents, opt_bodyContents, opt_styles, opt_quirks) {
186 var domHelper = goog.dom.getDomHelper(parentElement);
187
188 if (goog.isString(opt_headContents)) {
189 opt_headContents =
190 goog.html.legacyconversions.safeHtmlFromString(opt_headContents);
191 }
192 if (goog.isString(opt_bodyContents)) {
193 opt_bodyContents =
194 goog.html.legacyconversions.safeHtmlFromString(opt_bodyContents);
195 }
196 if (goog.isString(opt_styles)) {
197 opt_styles =
198 goog.html.legacyconversions.safeStyleFromString(opt_styles);
199 }
200
201 var content = goog.html.SafeHtml.create('html', {}, goog.html.SafeHtml.concat(
202 goog.html.SafeHtml.create('head', {}, opt_headContents),
203 goog.html.SafeHtml.create('body', {}, opt_bodyContents)));
204 if (!opt_quirks) {
205 content =
206 goog.html.SafeHtml.concat(goog.html.SafeHtml.DOCTYPE_HTML, content);
207 }
208
209 var iframe = goog.dom.iframe.createBlank(domHelper, opt_styles);
210
211 // Cannot manipulate iframe content until it is in a document.
212 parentElement.appendChild(iframe);
213 goog.dom.iframe.writeSafeContent(iframe, content);
214
215 return iframe;
216};
