33/**
34 * Utility function to match a string against either a string or a regular
35 * expression.
36 *
37 * @param {string|RegExp} obj Either a string or a regular expression.
38 * @param {string} str The string to match.
39 * @return {boolean} Whether the strings are equal, or if the string matches
40 * the regular expression.
41 */
42goog.dom.pattern.matchStringOrRegex = function(obj, str) {
43 if (goog.isString(obj)) {
44 // Match a string
45 return str == obj;
46 } else {
47 // Match a regular expression
48 return !!(str && str.match(obj));
49 }
50};

53/**
54 * Utility function to match a DOM attribute against either a string or a
55 * regular expression. Conforms to the interface spec for
56 * {@link goog.object#every}.
57 *
58 * @param {string|RegExp} elem Either a string or a regular expression.
59 * @param {string} index The attribute name to match.
60 * @param {Object} orig The original map of matches to test.
61 * @return {boolean} Whether the strings are equal, or if the attribute matches
62 * the regular expression.
63 * @this {Element} Called using goog.object every on an Element.
64 */
65goog.dom.pattern.matchStringOrRegexMap = function(elem, index, orig) {
66 return goog.dom.pattern.matchStringOrRegex(elem,
67 index in this ? this[index] :
68 (this.getAttribute ? this.getAttribute(index) : null));
69};

72/**
73 * When matched to a token, a pattern may return any of the following statuses:
74 * <ol>
75 * <li><code>NO_MATCH</code> - The pattern does not match. This is the only
76 * value that evaluates to <code>false</code> in a boolean context.
77 * <li><code>MATCHING</code> - The token is part of an incomplete match.
78 * <li><code>MATCH</code> - The token completes a match.
79 * <li><code>BACKTRACK_MATCH</code> - The token does not match, but indicates
80 * the end of a repetitive match. For instance, in regular expressions,
81 * the pattern <code>/a+/</code> would match <code>'aaaaaaaab'</code>.
82 * Every <code>'a'</code> token would give a status of
83 * <code>MATCHING</code> while the <code>'b'</code> token would give a
84 * status of <code>BACKTRACK_MATCH</code>.
85 * </ol>
86 * @enum {number}
87 */
88goog.dom.pattern.MatchType = {
89 NO_MATCH: 0,
90 MATCHING: 1,
91 MATCH: 2,
92 BACKTRACK_MATCH: 3
93};
