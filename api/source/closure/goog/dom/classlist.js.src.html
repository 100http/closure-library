15/**
16 * @fileoverview Utilities for detecting, adding and removing classes. Prefer
17 * this over goog.dom.classes for new code since it attempts to use classList
18 * (DOMTokenList: http://dom.spec.whatwg.org/#domtokenlist) which is faster
19 * and requires less code.
20 *
21 * Note: these utilities are meant to operate on HTMLElements
22 * and may have unexpected behavior on elements with differing interfaces
23 * (such as SVGElements).
24 */

33 * Override this define at build-time if you know your target supports it.
34 * @define {boolean} Whether to use the classList property (DOMTokenList).
35 */
36goog.define('goog.dom.classlist.ALWAYS_USE_DOM_TOKEN_LIST', false);

39/**
40 * Gets an array-like object of class names on an element.
41 * @param {Element} element DOM node to get the classes of.
42 * @return {!goog.array.ArrayLike} Class names on {@code element}.
43 */
44goog.dom.classlist.get = function(element) {
45 if (goog.dom.classlist.ALWAYS_USE_DOM_TOKEN_LIST || element.classList) {
46 return element.classList;
47 }
48
49 var className = element.className;
50 // Some types of elements don't have a className in IE (e.g. iframes).
51 // Furthermore, in Firefox, className is not a string when the element is
52 // an SVG element.
53 return goog.isString(className) && className.match(/\S+/g) || [];
54};

67/**
68 * Returns true if an element has a class. This method may throw a DOM
69 * exception for an invalid or empty class name if DOMTokenList is used.
70 * @param {Element} element DOM node to test.
71 * @param {string} className Class name to test for.
72 * @return {boolean} Whether element has the class.
73 */
74goog.dom.classlist.contains = function(element, className) {
75 if (goog.dom.classlist.ALWAYS_USE_DOM_TOKEN_LIST || element.classList) {
76 return element.classList.contains(className);
77 }
78 return goog.array.contains(goog.dom.classlist.get(element), className);
79};

82/**
83 * Adds a class to an element. Does not add multiples of class names. This
84 * method may throw a DOM exception for an invalid or empty class name if
85 * DOMTokenList is used.
86 * @param {Element} element DOM node to add class to.
87 * @param {string} className Class name to add.
88 */
89goog.dom.classlist.add = function(element, className) {
90 if (goog.dom.classlist.ALWAYS_USE_DOM_TOKEN_LIST || element.classList) {
91 element.classList.add(className);
92 return;
93 }
94
95 if (!goog.dom.classlist.contains(element, className)) {
96 // Ensure we add a space if this is not the first class name added.
97 element.className += element.className.length > 0 ?
98 (' ' + className) : className;
99 }
100};

103/**
104 * Convenience method to add a number of class names at once.
105 * @param {Element} element The element to which to add classes.
106 * @param {goog.array.ArrayLike<string>} classesToAdd An array-like object
107 * containing a collection of class names to add to the element.
108 * This method may throw a DOM exception if classesToAdd contains invalid
109 * or empty class names.
110 */
111goog.dom.classlist.addAll = function(element, classesToAdd) {
112 if (goog.dom.classlist.ALWAYS_USE_DOM_TOKEN_LIST || element.classList) {
113 goog.array.forEach(classesToAdd, function(className) {
114 goog.dom.classlist.add(element, className);
115 });
116 return;
117 }
118
119 var classMap = {};
120
121 // Get all current class names into a map.
122 goog.array.forEach(goog.dom.classlist.get(element),
123 function(className) {
124 classMap[className] = true;
125 });
126
127 // Add new class names to the map.
128 goog.array.forEach(classesToAdd,
129 function(className) {
130 classMap[className] = true;
131 });
132
133 // Flatten the keys of the map into the className.
134 element.className = '';
135 for (var className in classMap) {
136 element.className += element.className.length > 0 ?
137 (' ' + className) : className;
138 }
139};

142/**
143 * Removes a class from an element. This method may throw a DOM exception
144 * for an invalid or empty class name if DOMTokenList is used.
145 * @param {Element} element DOM node to remove class from.
146 * @param {string} className Class name to remove.
147 */
148goog.dom.classlist.remove = function(element, className) {
149 if (goog.dom.classlist.ALWAYS_USE_DOM_TOKEN_LIST || element.classList) {
150 element.classList.remove(className);
151 return;
152 }
153
154 if (goog.dom.classlist.contains(element, className)) {
155 // Filter out the class name.
156 element.className = goog.array.filter(
157 goog.dom.classlist.get(element),
158 function(c) {
159 return c != className;
160 }).join(' ');
161 }
162};

165/**
166 * Removes a set of classes from an element. Prefer this call to
167 * repeatedly calling {@code goog.dom.classlist.remove} if you want to remove
168 * a large set of class names at once.
169 * @param {Element} element The element from which to remove classes.
170 * @param {goog.array.ArrayLike<string>} classesToRemove An array-like object
171 * containing a collection of class names to remove from the element.
172 * This method may throw a DOM exception if classesToRemove contains invalid
173 * or empty class names.
174 */
175goog.dom.classlist.removeAll = function(element, classesToRemove) {
176 if (goog.dom.classlist.ALWAYS_USE_DOM_TOKEN_LIST || element.classList) {
177 goog.array.forEach(classesToRemove, function(className) {
178 goog.dom.classlist.remove(element, className);
179 });
180 return;
181 }
182 // Filter out those classes in classesToRemove.
183 element.className = goog.array.filter(
184 goog.dom.classlist.get(element),
185 function(className) {
186 // If this class is not one we are trying to remove,
187 // add it to the array of new class names.
188 return !goog.array.contains(classesToRemove, className);
189 }).join(' ');
190};

193/**
194 * Adds or removes a class depending on the enabled argument. This method
195 * may throw a DOM exception for an invalid or empty class name if DOMTokenList
196 * is used.
197 * @param {Element} element DOM node to add or remove the class on.
198 * @param {string} className Class name to add or remove.
199 * @param {boolean} enabled Whether to add or remove the class (true adds,
200 * false removes).
201 */
202goog.dom.classlist.enable = function(element, className, enabled) {
203 if (enabled) {
204 goog.dom.classlist.add(element, className);
205 } else {
206 goog.dom.classlist.remove(element, className);
207 }
208};

211/**
212 * Adds or removes a set of classes depending on the enabled argument. This
213 * method may throw a DOM exception for an invalid or empty class name if
214 * DOMTokenList is used.
215 * @param {!Element} element DOM node to add or remove the class on.
216 * @param {goog.array.ArrayLike<string>} classesToEnable An array-like object
217 * containing a collection of class names to add or remove from the element.
218 * @param {boolean} enabled Whether to add or remove the classes (true adds,
219 * false removes).
220 */
221goog.dom.classlist.enableAll = function(element, classesToEnable, enabled) {
222 var f = enabled ? goog.dom.classlist.addAll :
223 goog.dom.classlist.removeAll;
224 f(element, classesToEnable);
225};

228/**
229 * Switches a class on an element from one to another without disturbing other
230 * classes. If the fromClass isn't removed, the toClass won't be added. This
231 * method may throw a DOM exception if the class names are empty or invalid.
232 * @param {Element} element DOM node to swap classes on.
233 * @param {string} fromClass Class to remove.
234 * @param {string} toClass Class to add.
235 * @return {boolean} Whether classes were switched.
236 */
237goog.dom.classlist.swap = function(element, fromClass, toClass) {
238 if (goog.dom.classlist.contains(element, fromClass)) {
239 goog.dom.classlist.remove(element, fromClass);
240 goog.dom.classlist.add(element, toClass);
241 return true;
242 }
243 return false;
244};

247/**
248 * Removes a class if an element has it, and adds it the element doesn't have
249 * it. Won't affect other classes on the node. This method may throw a DOM
250 * exception if the class name is empty or invalid.
251 * @param {Element} element DOM node to toggle class on.
252 * @param {string} className Class to toggle.
253 * @return {boolean} True if class was added, false if it was removed
254 * (in other words, whether element has the class after this function has
255 * been called).
256 */
257goog.dom.classlist.toggle = function(element, className) {
258 var add = !goog.dom.classlist.contains(element, className);
259 goog.dom.classlist.enable(element, className, add);
260 return add;
261};

264/**
265 * Adds and removes a class of an element. Unlike
266 * {@link goog.dom.classlist.swap}, this method adds the classToAdd regardless
267 * of whether the classToRemove was present and had been removed. This method
268 * may throw a DOM exception if the class names are empty or invalid.
269 *
270 * @param {Element} element DOM node to swap classes on.
271 * @param {string} classToRemove Class to remove.
272 * @param {string} classToAdd Class to add.
273 */
274goog.dom.classlist.addRemove = function(element, classToRemove, classToAdd) {
275 goog.dom.classlist.remove(element, classToRemove);
276 goog.dom.classlist.add(element, classToAdd);
