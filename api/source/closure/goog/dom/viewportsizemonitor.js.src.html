32/**
33 * This class can be used to monitor changes in the viewport size. Instances
34 * dispatch a {@link goog.events.EventType.RESIZE} event when the viewport size
35 * changes. Handlers can call {@link goog.dom.ViewportSizeMonitor#getSize} to
36 * get the new viewport size.
37 *
38 * Use this class if you want to execute resize/reflow logic each time the
39 * user resizes the browser window. This class is guaranteed to only dispatch
40 * {@code RESIZE} events when the pixel dimensions of the viewport change.
41 * (Internet Explorer fires resize events if any element on the page is resized,
42 * even if the viewport dimensions are unchanged, which can lead to infinite
43 * resize loops.)
44 *
45 * Example usage:
46 * <pre>
47 * var vsm = new goog.dom.ViewportSizeMonitor();
48 * goog.events.listen(vsm, goog.events.EventType.RESIZE, function(e) {
49 * alert('Viewport size changed to ' + vsm.getSize());
50 * });
51 * </pre>
52 *
53 * Manually verified on IE6, IE7, FF2, Opera 11, Safari 4 and Chrome.
54 *
55 * @param {Window=} opt_window The window to monitor; defaults to the window in
56 * which this code is executing.
57 * @constructor
58 * @extends {goog.events.EventTarget}
59 */
60goog.dom.ViewportSizeMonitor = function(opt_window) {
61 goog.dom.ViewportSizeMonitor.base(this, 'constructor');
62
63 /**
64 * The window to monitor. Defaults to the window in which the code is running.
65 * @private {Window}
66 */
67 this.window_ = opt_window || window;
68
69 /**
70 * Event listener key for window the window resize handler, as returned by
71 * {@link goog.events.listen}.
72 * @private {goog.events.Key}
73 */
74 this.listenerKey_ = goog.events.listen(this.window_,
75 goog.events.EventType.RESIZE, this.handleResize_, false, this);
76
77 /**
78 * The most recently recorded size of the viewport, in pixels.
79 * @private {goog.math.Size}
80 */
81 this.size_ = goog.dom.getViewportSize(this.window_);
82};
83goog.inherits(goog.dom.ViewportSizeMonitor, goog.events.EventTarget);
84

86/**
87 * Returns a viewport size monitor for the given window. A new one is created
88 * if it doesn't exist already. This prevents the unnecessary creation of
89 * multiple spooling monitors for a window.
90 * @param {Window=} opt_window The window to monitor; defaults to the window in
91 * which this code is executing.
92 * @return {!goog.dom.ViewportSizeMonitor} Monitor for the given window.
93 */
94goog.dom.ViewportSizeMonitor.getInstanceForWindow = function(opt_window) {
95 var currentWindow = opt_window || window;
96 var uid = goog.getUid(currentWindow);
97
98 return goog.dom.ViewportSizeMonitor.windowInstanceMap_[uid] =
99 goog.dom.ViewportSizeMonitor.windowInstanceMap_[uid] ||
100 new goog.dom.ViewportSizeMonitor(currentWindow);
101};
102
103
104/**
105 * Removes and disposes a viewport size monitor for the given window if one
106 * exists.
107 * @param {Window=} opt_window The window whose monitor should be removed;
108 * defaults to the window in which this code is executing.
109 */
110goog.dom.ViewportSizeMonitor.removeInstanceForWindow = function(opt_window) {
111 var uid = goog.getUid(opt_window || window);
112
113 goog.dispose(goog.dom.ViewportSizeMonitor.windowInstanceMap_[uid]);
114 delete goog.dom.ViewportSizeMonitor.windowInstanceMap_[uid];
115};

118/**
119 * Map of window hash code to viewport size monitor for that window, if
120 * created.
121 * @type {Object<number,goog.dom.ViewportSizeMonitor>}
122 * @private
123 */
124goog.dom.ViewportSizeMonitor.windowInstanceMap_ = {};

127/**
128 * Returns the most recently recorded size of the viewport, in pixels. May
129 * return null if no window resize event has been handled yet.
130 * @return {goog.math.Size} The viewport dimensions, in pixels.
131 */
132goog.dom.ViewportSizeMonitor.prototype.getSize = function() {
133 // Return a clone instead of the original to preserve encapsulation.
134 return this.size_ ? this.size_.clone() : null;
135};

137
138/** @override */
139goog.dom.ViewportSizeMonitor.prototype.disposeInternal = function() {
140 goog.dom.ViewportSizeMonitor.superClass_.disposeInternal.call(this);
141
142 if (this.listenerKey_) {
143 goog.events.unlistenByKey(this.listenerKey_);
144 this.listenerKey_ = null;
145 }
146
147 this.window_ = null;
148 this.size_ = null;
149};

152/**
153 * Handles window resize events by measuring the dimensions of the
154 * viewport and dispatching a {@link goog.events.EventType.RESIZE} event if the
155 * current dimensions are different from the previous ones.
156 * @param {goog.events.Event} event The window resize event to handle.
157 * @private
158 */
159goog.dom.ViewportSizeMonitor.prototype.handleResize_ = function(event) {
160 var size = goog.dom.getViewportSize(this.window_);
161 if (!goog.math.Size.equals(size, this.size_)) {
162 this.size_ = size;
163 this.dispatchEvent(goog.events.EventType.RESIZE);
164 }
165};
