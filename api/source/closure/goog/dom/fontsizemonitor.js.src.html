35/**
36 * This class can be used to monitor changes in font size. Instances will
37 * dispatch a {@code goog.dom.FontSizeMonitor.EventType.CHANGE} event.
38 * Example usage:
39 * <pre>
40 * var fms = new goog.dom.FontSizeMonitor();
41 * goog.events.listen(fms, goog.dom.FontSizeMonitor.EventType.CHANGE,
42 * function(e) {
43 * alert('Font size was changed');
44 * });
45 * </pre>
46 * @param {goog.dom.DomHelper=} opt_domHelper DOM helper object that is used to
47 * determine where to insert the DOM nodes used to determine when the font
48 * size changes.
49 * @constructor
50 * @extends {goog.events.EventTarget}
51 * @final
52 */
53goog.dom.FontSizeMonitor = function(opt_domHelper) {
54 goog.events.EventTarget.call(this);
55
56 var dom = opt_domHelper || goog.dom.getDomHelper();
57
58 /**
59 * Offscreen iframe which we use to detect resize events.
60 * @type {HTMLElement}
61 * @private
62 */
63 this.sizeElement_ = /** @type {!HTMLElement} */ (dom.createDom(
64 // The size of the iframe is expressed in em, which are font size relative
65 // which will cause the iframe to be resized when the font size changes.
66 // The actual values are not relevant as long as we can ensure that the
67 // iframe has a non zero size and is completely off screen.
68 goog.userAgent.IE ? goog.dom.TagName.DIV : goog.dom.TagName.IFRAME, {
69 'style': 'position:absolute;width:9em;height:9em;top:-99em',
70 'tabIndex': -1,
71 'aria-hidden': 'true'
72 }));
73 var p = dom.getDocument().body;
74 p.insertBefore(this.sizeElement_, p.firstChild);
75
76 /**
77 * The object that we listen to resize events on.
78 * @type {Element|Window}
79 * @private
80 */
81 var resizeTarget = this.resizeTarget_ =
82 goog.userAgent.IE ? this.sizeElement_ :
83 goog.dom.getFrameContentWindow(
84 /** @type {HTMLIFrameElement} */ (this.sizeElement_));
85
86 // We need to open and close the document to get Firefox 2 to work. We must
87 // not do this for IE in case we are using HTTPS since accessing the document
88 // on an about:blank iframe in IE using HTTPS raises a Permission Denied
89 // error.
90 if (goog.userAgent.GECKO) {
91 var doc = resizeTarget.document;
92 doc.open();
93 doc.close();
94 }
95
96 // Listen to resize event on the window inside the iframe.
97 goog.events.listen(resizeTarget, goog.events.EventType.RESIZE,
98 this.handleResize_, false, this);
99
100 /**
101 * Last measured width of the iframe element.
102 * @type {number}
103 * @private
104 */
105 this.lastWidth_ = this.sizeElement_.offsetWidth;
106};
107goog.inherits(goog.dom.FontSizeMonitor, goog.events.EventTarget);

112 * @enum {string}

114goog.dom.FontSizeMonitor.EventType = {

116 CHANGE: 'fontsizechange'

121 * Constant for the change event.
122 * @type {string}
123 * @deprecated Use {@code goog.dom.FontSizeMonitor.EventType.CHANGE} instead.

125goog.dom.FontSizeMonitor.CHANGE_EVENT =
126 goog.dom.FontSizeMonitor.EventType.CHANGE;

128
129/** @override */
130goog.dom.FontSizeMonitor.prototype.disposeInternal = function() {
131 goog.dom.FontSizeMonitor.superClass_.disposeInternal.call(this);
132
133 goog.events.unlisten(this.resizeTarget_, goog.events.EventType.RESIZE,
134 this.handleResize_, false, this);
135 this.resizeTarget_ = null;
136
137 // Firefox 2 crashes if the iframe is removed during the unload phase.
138 if (!goog.userAgent.GECKO ||
139 goog.userAgent.isVersionOrHigher('1.9')) {
140 goog.dom.removeNode(this.sizeElement_);
141 }
142 delete this.sizeElement_;
143};

146/**
147 * Handles the onresize event of the iframe and dispatches a change event in
148 * case its size really changed.
149 * @param {goog.events.BrowserEvent} e The event object.
150 * @private
151 */
152goog.dom.FontSizeMonitor.prototype.handleResize_ = function(e) {
153 // Only dispatch the event if the size really changed. Some newer browsers do
154 // not really change the font-size, instead they zoom the whole page. This
155 // does trigger window resize events on the iframe but the logical pixel size
156 // remains the same (the device pixel size changes but that is irrelevant).
157 var currentWidth = this.sizeElement_.offsetWidth;
158 if (this.lastWidth_ != currentWidth) {
159 this.lastWidth_ = currentWidth;
160 this.dispatchEvent(goog.dom.FontSizeMonitor.EventType.CHANGE);
161 }
