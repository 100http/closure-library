101/**
102 * Returns a browser range spanning the given node's contents.
103 * @param {Node} node The node to select.
104 * @return {!TextRange} A browser range spanning the node's contents.
105 * @private
106 */
107goog.dom.browserrange.IeRange.getBrowserRangeForNode_ = function(node) {
108 var nodeRange = goog.dom.getOwnerDocument(node).body.createTextRange();
109 if (node.nodeType == goog.dom.NodeType.ELEMENT) {
110 // Elements are easy.
111 nodeRange.moveToElementText(node);
112 // Note(user) : If there are no child nodes of the element, the
113 // range.htmlText includes the element's outerHTML. The range created above
114 // is not collapsed, and should be collapsed explicitly.
115 // Example : node = <div></div>
116 // But if the node is sth like <br>, it shouldnt be collapsed.
117 if (goog.dom.browserrange.canContainRangeEndpoint(node) &&
118 !node.childNodes.length) {
119 nodeRange.collapse(false);
120 }
121 } else {
122 // Text nodes are hard.
123 // Compute the offset from the nearest element related position.
124 var offset = 0;
125 var sibling = node;
126 while (sibling = sibling.previousSibling) {
127 var nodeType = sibling.nodeType;
128 if (nodeType == goog.dom.NodeType.TEXT) {
129 offset += sibling.length;
130 } else if (nodeType == goog.dom.NodeType.ELEMENT) {
131 // Move to the space after this element.
132 nodeRange.moveToElementText(sibling);
133 break;
134 }
135 }
136
137 if (!sibling) {
138 nodeRange.moveToElementText(node.parentNode);
139 }
140
141 nodeRange.collapse(!sibling);
142
143 if (offset) {
144 nodeRange.move('character', offset);
145 }
146
147 nodeRange.moveEnd('character', node.length);
148 }
149
150 return nodeRange;
151};

154/**
155 * Returns a browser range spanning the given nodes.
156 * @param {Node} startNode The node to start with.
157 * @param {number} startOffset The offset within the start node.
158 * @param {Node} endNode The node to end with.
159 * @param {number} endOffset The offset within the end node.
160 * @return {!TextRange} A browser range spanning the node's contents.
161 * @private
162 */
163goog.dom.browserrange.IeRange.getBrowserRangeForNodes_ = function(startNode,
164 startOffset, endNode, endOffset) {
165 // Create a range starting at the correct start position.
166 var child, collapse = false;
167 if (startNode.nodeType == goog.dom.NodeType.ELEMENT) {
168 if (startOffset > startNode.childNodes.length) {
169 goog.log.error(goog.dom.browserrange.IeRange.logger_,
170 'Cannot have startOffset > startNode child count');
171 }
172 child = startNode.childNodes[startOffset];
173 collapse = !child;
174 startNode = child || startNode.lastChild || startNode;
175 startOffset = 0;
176 }
177 var leftRange = goog.dom.browserrange.IeRange.
178 getBrowserRangeForNode_(startNode);
179
180 // This happens only when startNode is a text node.
181 if (startOffset) {
182 leftRange.move('character', startOffset);
183 }
184
185
186 // The range movements in IE are still an approximation to the standard W3C
187 // behavior, and IE has its trickery when it comes to htmlText and text
188 // properties of the range. So we short-circuit computation whenever we can.
189 if (startNode == endNode && startOffset == endOffset) {
190 leftRange.collapse(true);
191 return leftRange;
192 }
193
194 // This can happen only when the startNode is an element, and there is no node
195 // at the given offset. We start at the last point inside the startNode in
196 // that case.
197 if (collapse) {
198 leftRange.collapse(false);
199 }
200
201 // Create a range that ends at the right position.
202 collapse = false;
203 if (endNode.nodeType == goog.dom.NodeType.ELEMENT) {
204 if (endOffset > endNode.childNodes.length) {
205 goog.log.error(goog.dom.browserrange.IeRange.logger_,
206 'Cannot have endOffset > endNode child count');
207 }
208 child = endNode.childNodes[endOffset];
209 endNode = child || endNode.lastChild || endNode;
210 endOffset = 0;
211 collapse = !child;
212 }
213 var rightRange = goog.dom.browserrange.IeRange.
214 getBrowserRangeForNode_(endNode);
215 rightRange.collapse(!collapse);
216 if (endOffset) {
217 rightRange.moveEnd('character', endOffset);
218 }
219
220 // Merge and return.
221 leftRange.setEndPoint('EndToEnd', rightRange);
222 return leftRange;
223};

226/**
227 * Create a range object that selects the given node's text.
228 * @param {Node} node The node to select.
229 * @return {!goog.dom.browserrange.IeRange} An IE range wrapper object.
230 */
231goog.dom.browserrange.IeRange.createFromNodeContents = function(node) {
232 var range = new goog.dom.browserrange.IeRange(
233 goog.dom.browserrange.IeRange.getBrowserRangeForNode_(node),
234 goog.dom.getOwnerDocument(node));
235
236 if (!goog.dom.browserrange.canContainRangeEndpoint(node)) {
237 range.startNode_ = range.endNode_ = range.parentNode_ = node.parentNode;
238 range.startOffset_ = goog.array.indexOf(range.parentNode_.childNodes, node);
239 range.endOffset_ = range.startOffset_ + 1;
240 } else {
241 // Note(user) : Emulate the behavior of W3CRange - Go to deepest possible
242 // range containers on both edges. It seems W3CRange did this to match the
243 // IE behavior, and now it is a circle. Changing W3CRange may break clients
244 // in all sorts of ways.
245 var tempNode, leaf = node;
246 while ((tempNode = leaf.firstChild) &&
247 goog.dom.browserrange.canContainRangeEndpoint(tempNode)) {
248 leaf = tempNode;
249 }
250 range.startNode_ = leaf;
251 range.startOffset_ = 0;
252
253 leaf = node;
254 while ((tempNode = leaf.lastChild) &&
255 goog.dom.browserrange.canContainRangeEndpoint(tempNode)) {
256 leaf = tempNode;
257 }
258 range.endNode_ = leaf;
259 range.endOffset_ = leaf.nodeType == goog.dom.NodeType.ELEMENT ?
260 leaf.childNodes.length : leaf.length;
261 range.parentNode_ = node;
262 }
263 return range;
264};

319/** @override */
320goog.dom.browserrange.IeRange.prototype.getContainer = function() {
321 if (!this.parentNode_) {
322 var selectText = this.range_.text;
323
324 // If the selection ends with spaces, we need to remove these to get the
325 // parent container of only the real contents. This is to get around IE's
326 // inconsistency where it selects the spaces after a word when you double
327 // click, but leaves out the spaces during execCommands.
328 var range = this.range_.duplicate();
329 // We can't use goog.string.trimRight, as that will remove other whitespace
330 // too.
331 var rightTrimmedSelectText = selectText.replace(/ +$/, '');
332 var numSpacesAtEnd = selectText.length - rightTrimmedSelectText.length;
333 if (numSpacesAtEnd) {
334 range.moveEnd('character', -numSpacesAtEnd);
335 }
336
337 // Get the parent node. This should be the end, but alas, it is not.
338 var parent = range.parentElement();
339
340 var htmlText = range.htmlText;
341 var htmlTextLen = goog.string.stripNewlines(htmlText).length;
342 if (this.isCollapsed() && htmlTextLen > 0) {
343 return (this.parentNode_ = parent);
344 }
345
346 // Deal with selection bug where IE thinks one of the selection's children
347 // is actually the selection's parent. Relies on the assumption that the
348 // HTML text of the parent container is longer than the length of the
349 // selection's HTML text.
350
351 // Also note IE will sometimes insert \r and \n whitespace, which should be
352 // disregarded. Otherwise the loop may run too long and return wrong parent
353 while (htmlTextLen > goog.string.stripNewlines(parent.outerHTML).length) {
354 parent = parent.parentNode;
355 }
356
357 // Deal with IE's selecting the outer tags when you double click
358 // If the innerText is the same, then we just want the inner node
359 while (parent.childNodes.length == 1 &&
360 parent.innerText == goog.dom.browserrange.IeRange.getNodeText_(
361 parent.firstChild)) {
362 // A container should be an element which can have children or a text
363 // node. Elements like IMG, BR, etc. can not be containers.
364 if (!goog.dom.browserrange.canContainRangeEndpoint(parent.firstChild)) {
365 break;
366 }
367 parent = parent.firstChild;
368 }
369
370 // If the selection is empty, we may need to do extra work to position it
371 // properly.
372 if (selectText.length == 0) {
373 parent = this.findDeepestContainer_(parent);
374 }
375
376 this.parentNode_ = parent;
377 }
378
379 return this.parentNode_;
380};

383/**
384 * Helper method to find the deepest parent for this range, starting
385 * the search from {@code node}, which must contain the range.
386 * @param {Node} node The node to start the search from.
387 * @return {Node} The deepest parent for this range.
388 * @private
389 */
390goog.dom.browserrange.IeRange.prototype.findDeepestContainer_ = function(node) {
391 var childNodes = node.childNodes;
392 for (var i = 0, len = childNodes.length; i < len; i++) {
393 var child = childNodes[i];
394
395 if (goog.dom.browserrange.canContainRangeEndpoint(child)) {
396 var childRange =
397 goog.dom.browserrange.IeRange.getBrowserRangeForNode_(child);
398 var start = goog.dom.RangeEndpoint.START;
399 var end = goog.dom.RangeEndpoint.END;
400
401 // There are two types of erratic nodes where the range over node has
402 // different htmlText than the node's outerHTML.
403 // Case 1 - A node with magic &nbsp; child. In this case :
404 // nodeRange.htmlText shows &nbsp; ('<p>&nbsp;</p>), while
405 // node.outerHTML doesn't show the magic node (<p></p>).
406 // Case 2 - Empty span. In this case :
407 // node.outerHTML shows '<span></span>'
408 // node.htmlText is just empty string ''.
409 var isChildRangeErratic = (childRange.htmlText != child.outerHTML);
410
411 // Moreover the inRange comparison fails only when the
412 var isNativeInRangeErratic = this.isCollapsed() && isChildRangeErratic;
413
414 // In case 2 mentioned above, childRange is also collapsed. So we need to
415 // compare start of this range with both start and end of child range.
416 var inChildRange = isNativeInRangeErratic ?
417 (this.compareBrowserRangeEndpoints(childRange, start, start) >= 0 &&
418 this.compareBrowserRangeEndpoints(childRange, start, end) <= 0) :
419 this.range_.inRange(childRange);
420 if (inChildRange) {
421 return this.findDeepestContainer_(child);
422 }
423 }
424 }
425
426 return node;
427};

492/**
493 * Recurses to find the correct node for the given endpoint.
494 * @param {goog.dom.RangeEndpoint} endpoint The endpoint to get the node for.
495 * @param {Node=} opt_node Optional node to start the search from.
496 * @return {Node} The deepest node containing the endpoint.
497 * @private
498 */
499goog.dom.browserrange.IeRange.prototype.getEndpointNode_ = function(endpoint,
500 opt_node) {
501
502 /** @type {Node} */
503 var node = opt_node || this.getContainer();
504
505 // If we're at a leaf in the DOM, we're done.
506 if (!node || !node.firstChild) {
507 return node;
508 }
509
510 var start = goog.dom.RangeEndpoint.START, end = goog.dom.RangeEndpoint.END;
511 var isStartEndpoint = endpoint == start;
512
513 // Find the first/last child that overlaps the selection.
514 // NOTE(user) : One of the children can be the magic &nbsp; node. This
515 // node will have only nodeType property as valid and accessible. All other
516 // dom related properties like ownerDocument, parentNode, nextSibling etc
517 // cause error when accessed. Therefore use the for-loop on childNodes to
518 // iterate.
519 for (var j = 0, length = node.childNodes.length; j < length; j++) {
520 var i = isStartEndpoint ? j : length - j - 1;
521 var child = node.childNodes[i];
522 var childRange;
523 try {
524 childRange = goog.dom.browserrange.createRangeFromNodeContents(child);
525 } catch (e) {
526 // If the child is the magic &nbsp; node, then the above will throw
527 // error. The magic node exists only when editing using keyboard, so can
528 // not add any unit test.
529 continue;
530 }
531 var ieRange = childRange.getBrowserRange();
532
533 // Case 1 : Finding end points when this range is collapsed.
534 // Note that in case of collapsed range, getEnd{Node,Offset} call
535 // getStart{Node,Offset}.
536 if (this.isCollapsed()) {
537 // Handle situations where caret is not in a text node. In such cases,
538 // the adjacent child won't be a valid range endpoint container.
539 if (!goog.dom.browserrange.canContainRangeEndpoint(child)) {
540 // The following handles a scenario like <div><BR>[caret]<BR></div>,
541 // where point should be (div, 1).
542 if (this.compareBrowserRangeEndpoints(ieRange, start, start) == 0) {
543 this.startOffset_ = this.endOffset_ = i;
544 return node;
545 }
546 } else if (childRange.containsRange(this)) {
547 // For collapsed range, we should invert the containsRange check with
548 // childRange.
549 return this.getEndpointNode_(endpoint, child);
550 }
551
552 // Case 2 - The first child encountered to have overlap this range is
553 // contained entirely in this range.
554 } else if (this.containsRange(childRange)) {
555 // If it is an element which can not be a range endpoint container, the
556 // current child offset can be used to deduce the endpoint offset.
557 if (!goog.dom.browserrange.canContainRangeEndpoint(child)) {
558
559 // Container can't be any deeper, so current node is the container.
560 if (isStartEndpoint) {
561 this.startOffset_ = i;
562 } else {
563 this.endOffset_ = i + 1;
564 }
565 return node;
566 }
567
568 // If child can contain range endpoints, recurse inside this child.
569 return this.getEndpointNode_(endpoint, child);
570
571 // Case 3 - Partial non-adjacency overlap.
572 } else if (this.compareBrowserRangeEndpoints(ieRange, start, end) < 0 &&
573 this.compareBrowserRangeEndpoints(ieRange, end, start) > 0) {
574 // If this child overlaps the selection partially, recurse down to find
575 // the first/last child the next level down that overlaps the selection
576 // completely. We do not consider edge-adjacency (== 0) as overlap.
577 return this.getEndpointNode_(endpoint, child);
578 }
579
580 }
581
582 // None of the children of this node overlapped the selection, that means
583 // the selection starts/ends in this node directly.
584 return node;
585};

588/**
589 * Compares one endpoint of this range with the endpoint of a node.
590 * For internal methods, we should prefer this method to containsNode.
591 * containsNode has a lot of false negatives when we're dealing with
592 * {@code <br>} tags.
593 *
594 * @param {Node} node The node to compare against.
595 * @param {goog.dom.RangeEndpoint} thisEndpoint The endpoint of this range
596 * to compare with.
597 * @param {goog.dom.RangeEndpoint} otherEndpoint The endpoint of the node
598 * to compare with.
599 * @return {number} 0 if the endpoints are equal, negative if this range
600 * endpoint comes before the other node endpoint, and positive otherwise.
601 * @private
602 */
603goog.dom.browserrange.IeRange.prototype.compareNodeEndpoints_ =
604 function(node, thisEndpoint, otherEndpoint) {
605 return this.range_.compareEndPoints(
606 (thisEndpoint == goog.dom.RangeEndpoint.START ? 'Start' : 'End') +
607 'To' +
608 (otherEndpoint == goog.dom.RangeEndpoint.START ? 'Start' : 'End'),
609 goog.dom.browserrange.createRangeFromNodeContents(node).
610 getBrowserRange());
611};

614/**
615 * Returns the offset into the start/end container.
616 * @param {goog.dom.RangeEndpoint} endpoint The endpoint to get the offset for.
617 * @param {Node=} opt_container The container to get the offset relative to.
618 * Defaults to the value returned by getStartNode/getEndNode.
619 * @return {number} The offset.
620 * @private
621 */
622goog.dom.browserrange.IeRange.prototype.getOffset_ = function(endpoint,
623 opt_container) {
624 var isStartEndpoint = endpoint == goog.dom.RangeEndpoint.START;
625 var container = opt_container ||
626 (isStartEndpoint ? this.getStartNode() : this.getEndNode());
627
628 if (container.nodeType == goog.dom.NodeType.ELEMENT) {
629 // Find the first/last child that overlaps the selection
630 var children = container.childNodes;
631 var len = children.length;
632 var edge = isStartEndpoint ? 0 : len - 1;
633 var sign = isStartEndpoint ? 1 : - 1;
634
635 // We find the index in the child array of the endpoint of the selection.
636 for (var i = edge; i >= 0 && i < len; i += sign) {
637 var child = children[i];
638 // Ignore the child nodes, which could be end point containers.
639 if (goog.dom.browserrange.canContainRangeEndpoint(child)) {
640 continue;
641 }
642 // Stop looping when we reach the edge of the selection.
643 var endPointCompare =
644 this.compareNodeEndpoints_(child, endpoint, endpoint);
645 if (endPointCompare == 0) {
646 return isStartEndpoint ? i : i + 1;
647 }
648 }
649
650 // When starting from the end in an empty container, we erroneously return
651 // -1: fix this to return 0.
652 return i == -1 ? 0 : i;
653 } else {
654 // Get a temporary range object.
655 var range = this.range_.duplicate();
656
657 // Create a range that selects the entire container.
658 var nodeRange = goog.dom.browserrange.IeRange.getBrowserRangeForNode_(
659 container);
660
661 // Now, intersect our range with the container range - this should give us
662 // the part of our selection that is in the container.
663 range.setEndPoint(isStartEndpoint ? 'EndToEnd' : 'StartToStart', nodeRange);
664
665 var rangeLength = range.text.length;
666 return isStartEndpoint ? container.length - rangeLength : rangeLength;
667 }
668};

699/** @override */
700goog.dom.browserrange.IeRange.prototype.isCollapsed = function() {
701 // Note(user) : The earlier implementation used (range.text == ''), but this
702 // fails when (range.htmlText == '<br>')
703 // Alternative: this.range_.htmlText == '';

730/** @override */
731goog.dom.browserrange.IeRange.prototype.removeContents = function() {
732 // NOTE: Sometimes htmlText is non-empty, but the range is actually empty.
733 // TODO(gboyer): The htmlText check is probably unnecessary, but I left it in
734 // for paranoia.
735 if (!this.isCollapsed() && this.range_.htmlText) {
736 // Store some before-removal state.
737 var startNode = this.getStartNode();
738 var endNode = this.getEndNode();
739 var oldText = this.range_.text;
740
741 // IE sometimes deletes nodes unrelated to the selection. This trick fixes
742 // that problem most of the time. Even though it looks like a no-op, it is
743 // somehow changing IE's internal state such that empty unrelated nodes are
744 // no longer deleted.
745 var clone = this.range_.duplicate();
746 clone.moveStart('character', 1);
747 clone.moveStart('character', -1);
748
749 // However, sometimes moving the start back and forth ends up changing the
750 // range.
751 // TODO(gboyer): This condition used to happen for empty ranges, but (1)
752 // never worked, and (2) the isCollapsed call should protect against empty
753 // ranges better than before. However, this is left for paranoia.
754 if (clone.text == oldText) {
755 this.range_ = clone;
756 }
757
758 // Use the browser's native deletion code.
759 this.range_.text = '';
760 this.clearCachedValues_();
761
762 // Unfortunately, when deleting a portion of a single text node, IE creates
763 // an extra text node unlike other browsers which just change the text in
764 // the node. We normalize for that behavior here, making IE behave like all
765 // the other browsers.
766 var newStartNode = this.getStartNode();
767 var newStartOffset = this.getStartOffset();
768 /** @preserveTry */
769 try {
770 var sibling = startNode.nextSibling;
771 if (startNode == endNode && startNode.parentNode &&
772 startNode.nodeType == goog.dom.NodeType.TEXT &&
773 sibling && sibling.nodeType == goog.dom.NodeType.TEXT) {
774 startNode.nodeValue += sibling.nodeValue;
775 goog.dom.removeNode(sibling);
776
777 // Make sure to reselect the appropriate position.
778 this.range_ = goog.dom.browserrange.IeRange.getBrowserRangeForNode_(
779 newStartNode);
780 this.range_.move('character', newStartOffset);
781 this.clearCachedValues_();
782 }
783 } catch (e) {
784 // IE throws errors on orphaned nodes.
785 }
786 }
787};

801/**
802 * Pastes the given element into the given range, returning the resulting
803 * element.
804 * @param {TextRange} range The range to paste into.
805 * @param {Element} element The node to insert a copy of.
806 * @param {goog.dom.DomHelper=} opt_domHelper DOM helper object for the document
807 * the range resides in.
808 * @return {Element} The resulting copy of element.
809 * @private
810 */
811goog.dom.browserrange.IeRange.pasteElement_ = function(range, element,
812 opt_domHelper) {
813 opt_domHelper = opt_domHelper || goog.dom.browserrange.IeRange.getDomHelper_(
814 range);
815
816 // Make sure the node has a unique id.
817 var id;
818 var originalId = id = element.id;
819 if (!id) {
820 id = element.id = goog.string.createUniqueString();
821 }
822
823 // Insert (a clone of) the node.
824 range.pasteHTML(element.outerHTML);
825
826 // Pasting the outerHTML of the modified element into the document creates
827 // a clone of the element argument. We want to return a reference to the
828 // clone, not the original. However we need to remove the temporary ID
829 // first.
830 element = opt_domHelper.getElement(id);
831
832 // If element is null here, we failed.
833 if (element) {
834 if (!originalId) {
835 element.removeAttribute('id');
836 }
837 }
838
839 return element;
840};

863/**
864 * Internal handler for inserting a node.
865 * @param {TextRange} clone A clone of this range's browser range object.
866 * @param {Node} node The node to insert.
867 * @param {boolean} before Whether to insert the node before or after the range.
868 * @param {goog.dom.DomHelper=} opt_domHelper The dom helper to use.
869 * @return {Node} The resulting copy of node.
870 * @private
871 */
872goog.dom.browserrange.IeRange.insertNode_ = function(clone, node,
873 before, opt_domHelper) {
874 // Get a DOM helper.
875 opt_domHelper = opt_domHelper || goog.dom.browserrange.IeRange.getDomHelper_(
876 clone);
877
878 // If it's not an element, wrap it in one.
879 var isNonElement;
880 if (node.nodeType != goog.dom.NodeType.ELEMENT) {
881 isNonElement = true;
882 node = opt_domHelper.createDom(goog.dom.TagName.DIV, null, node);
883 }
884
885 clone.collapse(before);
886 node = goog.dom.browserrange.IeRange.pasteElement_(clone,
887 /** @type {!Element} */ (node), opt_domHelper);
888
889 // If we didn't want an element, unwrap the element and return the node.
890 if (isNonElement) {
891 // pasteElement_() may have returned a copy of the wrapper div, and the
892 // node it wraps could also be a new copy. So we must extract that new
893 // node from the new wrapper.
894 var newNonElement = node.firstChild;
895 opt_domHelper.flattenElement(node);
896 node = newNonElement;
897 }
898
899 return node;
900};
