15/**
16 * @fileoverview MD5 cryptographic hash.
17 * Implementation of http://tools.ietf.org/html/rfc1321 with common
18 * optimizations and tweaks (see http://en.wikipedia.org/wiki/MD5).
19 *
20 * Usage:
21 * var md5 = new goog.crypt.Md5();
22 * md5.update(bytes);
23 * var hash = md5.digest();
24 *
25 * Performance:
26 * Chrome 23 ~680 Mbit/s
27 * Chrome 13 (in a VM) ~250 Mbit/s
28 * Firefox 6.0 (in a VM) ~100 Mbit/s
29 * IE9 (in a VM) ~27 Mbit/s
30 * Firefox 3.6 ~15 Mbit/s
31 * IE8 (in a VM) ~13 Mbit/s
32 *

352/** @override */
353goog.crypt.Md5.prototype.update = function(bytes, opt_length) {
354 if (!goog.isDef(opt_length)) {
355 opt_length = bytes.length;
356 }
357 var lengthMinusBlock = opt_length - this.blockSize;
358
359 // Copy some object properties to local variables in order to save on access
360 // time from inside the loop (~10% speedup was observed on Chrome 11).
361 var block = this.block_;
362 var blockLength = this.blockLength_;
363 var i = 0;
364
365 // The outer while loop should execute at most twice.
366 while (i < opt_length) {
367 // When we have no data in the block to top up, we can directly process the
368 // input buffer (assuming it contains sufficient data). This gives ~30%
369 // speedup on Chrome 14 and ~70% speedup on Firefox 6.0, but requires that
370 // the data is provided in large chunks (or in multiples of 64 bytes).
371 if (blockLength == 0) {
372 while (i <= lengthMinusBlock) {
373 this.compress_(bytes, i);
374 i += this.blockSize;
375 }
376 }
377
378 if (goog.isString(bytes)) {
379 while (i < opt_length) {
380 block[blockLength++] = bytes.charCodeAt(i++);
381 if (blockLength == this.blockSize) {
382 this.compress_(block);
383 blockLength = 0;
384 // Jump to the outer loop so we use the full-block optimization.
385 break;
386 }
387 }
388 } else {
389 while (i < opt_length) {
390 block[blockLength++] = bytes[i++];
391 if (blockLength == this.blockSize) {
392 this.compress_(block);
393 blockLength = 0;
394 // Jump to the outer loop so we use the full-block optimization.
395 break;
396 }
397 }
398 }
399 }
400
401 this.blockLength_ = blockLength;
402 this.totalLength_ += opt_length;
403};

406/** @override */
407goog.crypt.Md5.prototype.digest = function() {
408 // This must accommodate at least 1 padding byte (0x80), 8 bytes of
409 // total bitlength, and must end at a 64-byte boundary.
410 var pad = new Array((this.blockLength_ < 56 ?
411 this.blockSize :
412 this.blockSize * 2) - this.blockLength_);
413
414 // Add padding: 0x80 0x00*
415 pad[0] = 0x80;
416 for (var i = 1; i < pad.length - 8; ++i) {
417 pad[i] = 0;
418 }
419 // Add the total number of bits, little endian 64-bit integer.
420 var totalBits = this.totalLength_ * 8;
421 for (var i = pad.length - 8; i < pad.length; ++i) {
422 pad[i] = totalBits & 0xff;
423 totalBits /= 0x100; // Don't use bit-shifting here!
424 }
425 this.update(pad);
426
427 var digest = new Array(16);
428 var n = 0;
429 for (var i = 0; i < 4; ++i) {
430 for (var j = 0; j < 32; j += 8) {
431 digest[n++] = (this.chain_[i] >>> j) & 0xff;
432 }
433 }
434 return digest;
