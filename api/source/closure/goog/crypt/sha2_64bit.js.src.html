36/**
37 * Constructs a SHA-2 64-bit cryptographic hash.
38 * This class should not be used. Rather, one should use one of its
39 * subclasses.
40 * @constructor
41 * @param {number} numHashBlocks The size of the output in 16-byte blocks
42 * @param {!Array<number>} initHashBlocks The hash-specific initialization
43 * vector, as a sequence of sixteen 32-bit numbers.
44 * @extends {goog.crypt.Hash}
45 * @struct
46 */
47goog.crypt.Sha2_64bit = function(numHashBlocks, initHashBlocks) {
48 goog.crypt.Sha2_64bit.base(this, 'constructor');
49
50 /**
51 * The number of bytes that are digested in each pass of this hasher.
52 * @const {number}
53 */
54 this.blockSize = goog.crypt.Sha2_64bit.BLOCK_SIZE_;
55
56 /**
57 * A chunk holding the currently processed message bytes. Once the chunk has
58 * {@code this.blocksize} bytes, we feed it into [@code computeChunk_}.
59 * @private {!Uint8Array|!Array<number>}
60 */
61 this.chunk_ = goog.isDef(goog.global.Uint8Array) ?
62 new Uint8Array(goog.crypt.Sha2_64bit.BLOCK_SIZE_) :
63 new Array(goog.crypt.Sha2_64bit.BLOCK_SIZE_);
64
65 /**
66 * Current number of bytes in {@code this.chunk_}.
67 * @private {number}
68 */
69 this.chunkBytes_ = 0;
70
71 /**
72 * Total number of bytes in currently processed message.
73 * @private {number}
74 */
75 this.total_ = 0;
76
77 /**
78 * Holds the previous values of accumulated hash a-h in the
79 * {@code computeChunk_} function.
80 * @private {!Array<!goog.math.Long>}
81 */
82 this.hash_ = [];
83
84 /**
85 * The number of blocks of output produced by this hash function, where each
86 * block is eight bytes long.
87 * @private {number}
88 */
89 this.numHashBlocks_ = numHashBlocks;
90
91 /**
92 * Temporary array used in chunk computation. Allocate here as a
93 * member rather than as a local within computeChunk_() as a
94 * performance optimization to reduce the number of allocations and
95 * reduce garbage collection.
96 * @type {!Array<!goog.math.Long>}
97 * @private
98 */
99 this.w_ = [];
100
101 /**
102 * The value to which {@code this.hash_} should be reset when this
103 * Hasher is reset.
104 * @private @const {!Array<!goog.math.Long>}
105 */
106 this.initHashBlocks_ = goog.crypt.Sha2_64bit.toLongArray_(initHashBlocks);
107
108 /**
109 * If true, we have taken the digest from this hasher, but we have not
110 * yet reset it.
111 *
112 * @private {boolean}
113 */
114 this.needsReset_ = false;
115
116 this.reset();
117};

148/** @override */
149goog.crypt.Sha2_64bit.prototype.update = function(message, opt_length) {
150 var length = goog.isDef(opt_length) ? opt_length : message.length;
151
152 // Make sure this hasher is usable.
153 if (this.needsReset_) {
154 throw Error('this hasher needs to be reset');
155 }
156 // Process the message from left to right up to |length| bytes.
157 // When we get a 512-bit chunk, compute the hash of it and reset
158 // this.chunk_. The message might not be multiple of 512 bits so we
159 // might end up with a chunk that is less than 512 bits. We store
160 // such partial chunk in chunk_ and it will be filled up later
161 // in digest().
162 var chunkBytes = this.chunkBytes_;
163
164 // The input message could be either byte array or string.
165 if (goog.isString(message)) {
166 for (var i = 0; i < length; i++) {
167 var b = message.charCodeAt(i);
168 if (b > 255) {
169 throw Error('Characters must be in range [0,255]');
170 }
171 this.chunk_[chunkBytes++] = b;
172 if (chunkBytes == this.blockSize) {
173 this.computeChunk_();
174 chunkBytes = 0;
175 }
176 }
177 } else if (goog.isArray(message)) {
178 for (var i = 0; i < length; i++) {
179 var b = message[i];
180 // Hack: b|0 coerces b to an integer, so the last part confirms that
181 // b has no fractional part.
182 if (!goog.isNumber(b) || b < 0 || b > 255 || b != (b | 0)) {
183 throw Error('message must be a byte array');
184 }
185 this.chunk_[chunkBytes++] = b;
186 if (chunkBytes == this.blockSize) {
187 this.computeChunk_();
188 chunkBytes = 0;
189 }
190 }
191 } else {
192 throw Error('message must be string or array');
193 }
194
195 // Record the current bytes in chunk to support partial update.
196 this.chunkBytes_ = chunkBytes;
197
198 // Record total message bytes we have processed so far.
199 this.total_ += length;
200};

203/** @override */
204goog.crypt.Sha2_64bit.prototype.digest = function() {
205 if (this.needsReset_) {
206 throw Error('this hasher needs to be reset');
207 }
208 var totalBits = this.total_ * 8;
209
210 // Append pad 0x80 0x00* until this.chunkBytes_ == 112
211 if (this.chunkBytes_ < 112) {
212 this.update(goog.crypt.Sha2_64bit.PADDING_, 112 - this.chunkBytes_);
213 } else {
214 // the rest of this block, plus 112 bytes of next block
215 this.update(goog.crypt.Sha2_64bit.PADDING_,
216 this.blockSize - this.chunkBytes_ + 112);
217 }
218
219 // Append # bits in the 64-bit big-endian format.
220 for (var i = 127; i >= 112; i--) {
221 this.chunk_[i] = totalBits & 255;
222 totalBits /= 256; // Don't use bit-shifting here!
223 }
224 this.computeChunk_();
225
226 // Finally, output the result digest.
227 var n = 0;
228 var digest = new Array(8 * this.numHashBlocks_);
229 for (var i = 0; i < this.numHashBlocks_; i++) {
230 var block = this.hash_[i];
231 var high = block.getHighBits();
232 var low = block.getLowBits();
233 for (var j = 24; j >= 0; j -= 8) {
234 digest[n++] = ((high >> j) & 255);
235 }
236 for (var j = 24; j >= 0; j -= 8) {
237 digest[n++] = ((low >> j) & 255);
238 }
239 }
240
241 // The next call to this hasher must be a reset
242 this.needsReset_ = true;
243 return digest;
244};

247/**
248 * Updates this hash by processing the 1024-bit message chunk in this.chunk_.
249 * @private
250 */
251goog.crypt.Sha2_64bit.prototype.computeChunk_ = function() {
252 var chunk = this.chunk_;
253 var K_ = goog.crypt.Sha2_64bit.K_;
254
255 // Divide the chunk into 16 64-bit-words.
256 var w = this.w_;
257 for (var i = 0; i < 16; i++) {
258 var offset = i * 8;
259 w[i] = new goog.math.Long(
260 (chunk[offset + 4] << 24) | (chunk[offset + 5] << 16) |
261 (chunk[offset + 6] << 8) | (chunk[offset + 7]),
262 (chunk[offset] << 24) | (chunk[offset + 1] << 16) |
263 (chunk[offset + 2] << 8) | (chunk[offset + 3]));
264
265 }
266
267 // Extend the w[] array to be the number of rounds.
268 for (var i = 16; i < 80; i++) {
269 var s0 = this.sigma0_(w[i - 15]);
270 var s1 = this.sigma1_(w[i - 2]);
271 w[i] = this.sum_(w[i - 16], w[i - 7], s0, s1);
272 }
273
274 var a = this.hash_[0];
275 var b = this.hash_[1];
276 var c = this.hash_[2];
277 var d = this.hash_[3];
278 var e = this.hash_[4];
279 var f = this.hash_[5];
280 var g = this.hash_[6];
281 var h = this.hash_[7];
282 for (var i = 0; i < 80; i++) {
283 var S0 = this.Sigma0_(a);
284 var maj = this.majority_(a, b, c);
285 var t2 = S0.add(maj);
286 var S1 = this.Sigma1_(e);
287 var ch = this.choose_(e, f, g);
288 var t1 = this.sum_(h, S1, ch, K_[i], w[i]);
289 h = g;
290 g = f;
291 f = e;
292 e = d.add(t1);
293 d = c;
294 c = b;
295 b = a;
296 a = t1.add(t2);
297 }
298
299 this.hash_[0] = this.hash_[0].add(a);
300 this.hash_[1] = this.hash_[1].add(b);
301 this.hash_[2] = this.hash_[2].add(c);
302 this.hash_[3] = this.hash_[3].add(d);
303 this.hash_[4] = this.hash_[4].add(e);
304 this.hash_[5] = this.hash_[5].add(f);
305 this.hash_[6] = this.hash_[6].add(g);
306 this.hash_[7] = this.hash_[7].add(h);
307};

310/**
311 * Calculates the SHA2 64-bit sigma0 function.
312 * rotateRight(value, 1) ^ rotateRight(value, 8) ^ (value >>> 7)
313 *
314 * @private
315 * @param {!goog.math.Long} value
316 * @return {!goog.math.Long}
317 */
318goog.crypt.Sha2_64bit.prototype.sigma0_ = function(value) {
319 var valueLow = value.getLowBits();
320 var valueHigh = value.getHighBits();
321 // Implementation note: We purposely do not use the shift operations defined
322 // in goog.math.Long. Inlining the code for specific values of shifting and
323 // not generating the intermediate results doubles the speed of this code.
324 var low = (valueLow >>> 1) ^ (valueHigh << 31) ^
325 (valueLow >>> 8) ^ (valueHigh << 24) ^
326 (valueLow >>> 7) ^ (valueHigh << 25);
327 var high = (valueHigh >>> 1) ^ (valueLow << 31) ^
328 (valueHigh >>> 8) ^ (valueLow << 24) ^
329 (valueHigh >>> 7);
330 return new goog.math.Long(low, high);
331};

334/**
335 * Calculates the SHA2 64-bit sigma1 function.
336 * rotateRight(value, 19) ^ rotateRight(value, 61) ^ (value >>> 6)
337 *
338 * @private
339 * @param {!goog.math.Long} value
340 * @return {!goog.math.Long}
341 */
342goog.crypt.Sha2_64bit.prototype.sigma1_ = function(value) {
343 var valueLow = value.getLowBits();
344 var valueHigh = value.getHighBits();
345 // Implementation note: See _sigma0() above
346 var low = (valueLow >>> 19) ^ (valueHigh << 13) ^
347 (valueHigh >>> 29) ^ (valueLow << 3) ^
348 (valueLow >>> 6) ^ (valueHigh << 26);
349 var high = (valueHigh >>> 19) ^ (valueLow << 13) ^
350 (valueLow >>> 29) ^ (valueHigh << 3) ^
351 (valueHigh >>> 6);
352 return new goog.math.Long(low, high);

400/**
401 * Calculates the SHA-2 64-bit choose function.
402 *
403 * This function uses {@code value} as a mask to choose bits from either
404 * {@code one} if the bit is set or {@code two} if the bit is not set.
405 *
406 * @private
407 * @param {!goog.math.Long} value
408 * @param {!goog.math.Long} one
409 * @param {!goog.math.Long} two
410 * @return {!goog.math.Long}
411 */
412goog.crypt.Sha2_64bit.prototype.choose_ = function(value, one, two) {
413 var valueLow = value.getLowBits();
414 var valueHigh = value.getHighBits();
415 return new goog.math.Long(
416 (valueLow & one.getLowBits()) | (~valueLow & two.getLowBits()),
417 (valueHigh & one.getHighBits()) | (~valueHigh & two.getHighBits()));
418};

421/**
422 * Calculates the SHA-2 64-bit majority function.
423 * This function returns, for each bit position, the bit held by the majority
424 * of its three arguments.
425 *
426 * @private
427 * @param {!goog.math.Long} one
428 * @param {!goog.math.Long} two
429 * @param {!goog.math.Long} three
430 * @return {!goog.math.Long}
431 */
432goog.crypt.Sha2_64bit.prototype.majority_ = function(one, two, three) {
433 return new goog.math.Long(
434 (one.getLowBits() & two.getLowBits()) |
435 (two.getLowBits() & three.getLowBits()) |
436 (one.getLowBits() & three.getLowBits()),
437 (one.getHighBits() & two.getHighBits()) |
438 (two.getHighBits() & three.getHighBits()) |
439 (one.getHighBits() & three.getHighBits()));
440};

443/**
444 * Adds two or more goog.math.Long values.
445 *
446 * @private
447 * @param {!goog.math.Long} one first summand
448 * @param {!goog.math.Long} two second summand
449 * @param {...goog.math.Long} var_args more arguments to sum
450 * @return {!goog.math.Long} The resulting sum.
451 */
452goog.crypt.Sha2_64bit.prototype.sum_ = function(one, two, var_args) {
453 // The low bits may be signed, but they represent a 32-bit unsigned quantity.
454 // We must be careful to normalize them.
455 // This doesn't matter for the high bits.
456 // Implementation note: Performance testing shows that this method runs
457 // fastest when the first two arguments are pulled out of the loop.
458 var low = (one.getLowBits() ^ 0x80000000) + (two.getLowBits() ^ 0x80000000);
459 var high = one.getHighBits() + two.getHighBits();
460 for (var i = arguments.length - 1; i >= 2; --i) {
461 low += arguments[i].getLowBits() ^ 0x80000000;
462 high += arguments[i].getHighBits();
463 }
464 // Because of the ^0x80000000, each value we added is 0x80000000 too small.
465 // Add arguments.length * 0x80000000 to the current sum. We can do this
466 // quickly by adding 0x80000000 to low when the number of arguments is
467 // odd, and adding (number of arguments) >> 1 to high.
468 if (arguments.length & 1) {
469 low += 0x80000000;
470 }
471 high += arguments.length >> 1;
472
473 // If low is outside the range [0, 0xFFFFFFFF], its overflow or underflow
474 // should be added to high. We don't actually need to modify low or
475 // normalize high because the goog.math.Long constructor already does that.
476 high += Math.floor(low / 0x100000000);
477 return new goog.math.Long(low, high);
478};
