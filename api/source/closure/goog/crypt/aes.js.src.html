29/**
30 * Implementation of AES in JavaScript.
31 * See http://en.wikipedia.org/wiki/Advanced_Encryption_Standard
32 *
33 * WARNING: This is ECB mode only. If you are encrypting something
34 * longer than 16 bytes, or encrypting more than one value with the same key
35 * (so basically, always) you need to use this with a block cipher mode of
36 * operation. See goog.crypt.Cbc.
37 *
38 * See http://en.wikipedia.org/wiki/Block_cipher_modes_of_operation for more
39 * information.
40 *
41 * @constructor
42 * @implements {goog.crypt.BlockCipher}
43 * @param {!Array<number>} key The key as an array of integers in {0, 255}.
44 * The key must have lengths of 16, 24, or 32 integers for 128-,
45 * 192-, or 256-bit encryption, respectively.
46 * @final
47 * @struct
48 */
49goog.crypt.Aes = function(key) {
50 goog.crypt.Aes.assertKeyArray_(key);
51
52 /**
53 * The AES key.
54 * @type {!Array<number>}
55 * @private
56 */
57 this.key_ = key;
58
59 /**
60 * Key length, in words.
61 * @type {number}
62 * @private
63 */
64 this.keyLength_ = this.key_.length / 4;
65
66 /**
67 * Number of rounds. Based on key length per AES spec.
68 * @type {number}
69 * @private
70 */
71 this.numberOfRounds_ = this.keyLength_ + 6;
72
73 /**
74 * 4x4 byte array containing the current state.
75 * @type {!Array<!Array<number>>}
76 * @private
77 */
78 this.state_ = [[], [], [], []];
79
80 /**
81 * Scratch temporary array for calculation.
82 * @type {!Array<!Array<number>>}
83 * @private
84 */
85 this.temp_ = [[], [], [], []];
86
87 /**
88 * The key schedule.
89 * @type {!Array<!Array<number>>}
90 * @private
91 */
92 this.keySchedule_;
93
94 this.keyExpansion_();
95};
96

98/**
99 * @define {boolean} Whether to call test method stubs. This can be enabled
100 * for unit testing.
101 */
102goog.define('goog.crypt.Aes.ENABLE_TEST_MODE', false);
103
104
105/**
106 * @override
107 */
108goog.crypt.Aes.prototype.encrypt = function(input) {
109
110 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
111 this.testKeySchedule_(0, this.keySchedule_, 0);
112 }
113
114 this.copyInput_(input);
115 this.addRoundKey_(0);
116
117 for (var round = 1; round < this.numberOfRounds_; ++round) {
118 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
119 this.testKeySchedule_(round, this.keySchedule_, round);
120 this.testStartRound_(round, this.state_);
121 }
122
123 this.subBytes_(goog.crypt.Aes.SBOX_);
124 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
125 this.testAfterSubBytes_(round, this.state_);
126 }
127
128 this.shiftRows_();
129 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
130 this.testAfterShiftRows_(round, this.state_);
131 }
132
133 this.mixColumns_();
134 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
135 this.testAfterMixColumns_(round, this.state_);
136 }
137
138 this.addRoundKey_(round);
139 }
140
141 this.subBytes_(goog.crypt.Aes.SBOX_);
142 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
143 this.testAfterSubBytes_(round, this.state_);
144 }
145
146 this.shiftRows_();
147 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
148 this.testAfterShiftRows_(round, this.state_);
149 }
150
151 this.addRoundKey_(this.numberOfRounds_);
152
153 return this.generateOutput_();
154};
155
156
157/**
158 * @override
159 */
160goog.crypt.Aes.prototype.decrypt = function(input) {
161
162 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
163 this.testKeySchedule_(0, this.keySchedule_, this.numberOfRounds_);
164 }
165
166 this.copyInput_(input);
167 this.addRoundKey_(this.numberOfRounds_);
168
169 for (var round = 1; round < this.numberOfRounds_; ++round) {
170 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
171 this.testKeySchedule_(round, this.keySchedule_,
172 this.numberOfRounds_ - round);
173 this.testStartRound_(round, this.state_);
174 }
175
176 this.invShiftRows_();
177 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
178 this.testAfterShiftRows_(round, this.state_);
179 }
180
181 this.subBytes_(goog.crypt.Aes.INV_SBOX_);
182 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
183 this.testAfterSubBytes_(round, this.state_);
184 }
185
186 this.addRoundKey_(this.numberOfRounds_ - round);
187 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
188 this.testAfterAddRoundKey_(round, this.state_);
189 }
190
191 this.invMixColumns_();
192 }
193
194 this.invShiftRows_();
195 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
196 this.testAfterShiftRows_(round, this.state_);
197 }
198
199 this.subBytes_(goog.crypt.Aes.INV_SBOX_);
200 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
201 this.testAfterSubBytes_(this.numberOfRounds_, this.state_);
202 }
203
204 if (goog.crypt.Aes.ENABLE_TEST_MODE) {
205 this.testKeySchedule_(this.numberOfRounds_, this.keySchedule_, 0);
206 }
207
208 this.addRoundKey_(0);
209
210 return this.generateOutput_();
211};

222/**
223 * Asserts that the key's array of integers is in the correct format.

228 if (goog.asserts.ENABLE_ASSERTS) {
229 goog.asserts.assert(arr.length == 16 || arr.length == 24 ||
230 arr.length == 32,
231 'Key must have length 16, 24, or 32.');
232 for (var i = 0; i < arr.length; i++) {

234 goog.asserts.assert(arr[i] >= 0 && arr[i] <= 255);
235 }

307/**
308 * Helper to copy input into the AES state matrix.
309 * @param {!Array<number>|!Uint8Array} input Byte array to copy into the state
310 * matrix.
311 * @private
312 */
313goog.crypt.Aes.prototype.copyInput_ = function(input) {
314 var v, p;
315
316 goog.asserts.assert(input.length == goog.crypt.Aes.BLOCK_SIZE_ * 4,
317 'Expecting input of 4 times block size.');
318
319 for (var r = 0; r < goog.crypt.Aes.BLOCK_SIZE_; r++) {
320 for (var c = 0; c < 4; c++) {
321 p = c * 4 + r;
322 v = input[p];
323
324 goog.asserts.assert(
325 v <= 255 && v >= 0,
326 'Invalid input. Value %s at position %s is not a byte.', v, p);
327
328 this.state_[r][c] = v;
329 }
330 }
331};
332
333
334/**
335 * Helper to copy the state matrix into an output array.
336 * @return {!Array<number>} Output byte array.
337 * @private
338 */
339goog.crypt.Aes.prototype.generateOutput_ = function() {
340 var output = [];
341 for (var r = 0; r < goog.crypt.Aes.BLOCK_SIZE_; r++) {
342 for (var c = 0; c < 4; c++) {
343 output[c * 4 + r] = this.state_[r][c];
344 }
345 }
346 return output;
347};

350/**
351 * AES's AddRoundKey procedure. Add the current round key to the state.
352 * @param {number} round The current round.
353 * @private
354 */
355goog.crypt.Aes.prototype.addRoundKey_ = function(round) {
356 for (var r = 0; r < 4; r++) {
357 for (var c = 0; c < 4; c++) {
358 this.state_[r][c] ^= this.keySchedule_[round * 4 + c][r];
359 }
360 }
361};

364/**
365 * AES's SubBytes procedure. Substitute bytes from the precomputed SBox lookup
366 * into the state.
367 * @param {!Array<number>} box The SBox or invSBox.
368 * @private
369 */
370goog.crypt.Aes.prototype.subBytes_ = function(box) {
371 for (var r = 0; r < 4; r++) {
372 for (var c = 0; c < 4; c++) {
373 this.state_[r][c] = box[this.state_[r][c]];
374 }
375 }
376};

379/**
380 * AES's ShiftRows procedure. Shift the values in each row to the right. Each
381 * row is shifted one more slot than the one above it.
382 * @private
383 */
384goog.crypt.Aes.prototype.shiftRows_ = function() {
385 for (var r = 1; r < 4; r++) {
386 for (var c = 0; c < 4; c++) {
387 this.temp_[r][c] = this.state_[r][c];
388 }
389 }
390
391 for (var r = 1; r < 4; r++) {
392 for (var c = 0; c < 4; c++) {
393 this.state_[r][c] = this.temp_[r][(c + r) %
394 goog.crypt.Aes.BLOCK_SIZE_];
395 }
396 }
397};
398
399
400/**
401 * AES's InvShiftRows procedure. Shift the values in each row to the right.
402 * @private
403 */
404goog.crypt.Aes.prototype.invShiftRows_ = function() {
405 for (var r = 1; r < 4; r++) {
406 for (var c = 0; c < 4; c++) {
407 this.temp_[r][(c + r) % goog.crypt.Aes.BLOCK_SIZE_] =
408 this.state_[r][c];
409 }
410 }
411
412 for (var r = 1; r < 4; r++) {
413 for (var c = 0; c < 4; c++) {
414 this.state_[r][c] = this.temp_[r][c];
415 }
416 }
417};

420/**
421 * AES's MixColumns procedure. Mix the columns of the state using magic.
422 * @private
423 */
424goog.crypt.Aes.prototype.mixColumns_ = function() {
425 var s = this.state_;
426 var t = this.temp_[0];
427
428 for (var c = 0; c < 4; c++) {
429 t[0] = s[0][c];
430 t[1] = s[1][c];
431 t[2] = s[2][c];
432 t[3] = s[3][c];
433
434 s[0][c] = (goog.crypt.Aes.MULT_2_[t[0]] ^
435 goog.crypt.Aes.MULT_3_[t[1]] ^ t[2] ^ t[3]);
436 s[1][c] = (t[0] ^ goog.crypt.Aes.MULT_2_[t[1]] ^
437 goog.crypt.Aes.MULT_3_[t[2]] ^ t[3]);
438 s[2][c] = (t[0] ^ t[1] ^ goog.crypt.Aes.MULT_2_[t[2]] ^
439 goog.crypt.Aes.MULT_3_[t[3]]);
440 s[3][c] = (goog.crypt.Aes.MULT_3_[t[0]] ^ t[1] ^ t[2] ^
441 goog.crypt.Aes.MULT_2_[t[3]]);
442 }
443};

446/**
447 * AES's InvMixColumns procedure.
448 * @private
449 */
450goog.crypt.Aes.prototype.invMixColumns_ = function() {
451 var s = this.state_;
452 var t = this.temp_[0];
453
454 for (var c = 0; c < 4; c++) {
455 t[0] = s[0][c];
456 t[1] = s[1][c];
457 t[2] = s[2][c];
458 t[3] = s[3][c];
459
460 s[0][c] = (
461 goog.crypt.Aes.MULT_E_[t[0]] ^ goog.crypt.Aes.MULT_B_[t[1]] ^
462 goog.crypt.Aes.MULT_D_[t[2]] ^ goog.crypt.Aes.MULT_9_[t[3]]);
463
464 s[1][c] = (
465 goog.crypt.Aes.MULT_9_[t[0]] ^ goog.crypt.Aes.MULT_E_[t[1]] ^
466 goog.crypt.Aes.MULT_B_[t[2]] ^ goog.crypt.Aes.MULT_D_[t[3]]);
467
468 s[2][c] = (
469 goog.crypt.Aes.MULT_D_[t[0]] ^ goog.crypt.Aes.MULT_9_[t[1]] ^
470 goog.crypt.Aes.MULT_E_[t[2]] ^ goog.crypt.Aes.MULT_B_[t[3]]);
471
472 s[3][c] = (
473 goog.crypt.Aes.MULT_B_[t[0]] ^ goog.crypt.Aes.MULT_D_[t[1]] ^
474 goog.crypt.Aes.MULT_9_[t[2]] ^ goog.crypt.Aes.MULT_E_[t[3]]);
475 }
476};

479/**
480 * AES's KeyExpansion procedure. Create the key schedule from the initial key.
481 * @private
482 */
483goog.crypt.Aes.prototype.keyExpansion_ = function() {
484 this.keySchedule_ = new Array(goog.crypt.Aes.BLOCK_SIZE_ * (
485 this.numberOfRounds_ + 1));
486
487 for (var rowNum = 0; rowNum < this.keyLength_; rowNum++) {
488 this.keySchedule_[rowNum] = [
489 this.key_[4 * rowNum],
490 this.key_[4 * rowNum + 1],
491 this.key_[4 * rowNum + 2],
492 this.key_[4 * rowNum + 3]
493 ];
494 }
495
496 var temp = new Array(4);
497
498 for (var rowNum = this.keyLength_;
499 rowNum < (goog.crypt.Aes.BLOCK_SIZE_ * (this.numberOfRounds_ + 1));
500 rowNum++) {
501 temp[0] = this.keySchedule_[rowNum - 1][0];
502 temp[1] = this.keySchedule_[rowNum - 1][1];
503 temp[2] = this.keySchedule_[rowNum - 1][2];
504 temp[3] = this.keySchedule_[rowNum - 1][3];
505
506 if (rowNum % this.keyLength_ == 0) {
507 this.rotWord_(temp);
508 this.subWord_(temp);
509
510 temp[0] ^= goog.crypt.Aes.RCON_[rowNum / this.keyLength_][0];
511 temp[1] ^= goog.crypt.Aes.RCON_[rowNum / this.keyLength_][1];
512 temp[2] ^= goog.crypt.Aes.RCON_[rowNum / this.keyLength_][2];
513 temp[3] ^= goog.crypt.Aes.RCON_[rowNum / this.keyLength_][3];
514 } else if (this.keyLength_ > 6 && rowNum % this.keyLength_ == 4) {
515 this.subWord_(temp);
516 }
517
518 this.keySchedule_[rowNum] = new Array(4);
519 this.keySchedule_[rowNum][0] =
520 this.keySchedule_[rowNum - this.keyLength_][0] ^ temp[0];
521 this.keySchedule_[rowNum][1] =
522 this.keySchedule_[rowNum - this.keyLength_][1] ^ temp[1];
523 this.keySchedule_[rowNum][2] =
524 this.keySchedule_[rowNum - this.keyLength_][2] ^ temp[2];
525 this.keySchedule_[rowNum][3] =
526 this.keySchedule_[rowNum - this.keyLength_][3] ^ temp[3];
527 }
528};

531/**
532 * AES's SubWord procedure.
533 * @param {!Array<number>} w Bytes to find the SBox substitution for.
534 * @return {!Array<number>} The substituted bytes.
535 * @private
536 */
537goog.crypt.Aes.prototype.subWord_ = function(w) {
538 w[0] = goog.crypt.Aes.SBOX_[w[0]];
539 w[1] = goog.crypt.Aes.SBOX_[w[1]];
540 w[2] = goog.crypt.Aes.SBOX_[w[2]];
541 w[3] = goog.crypt.Aes.SBOX_[w[3]];
542
543 return w;
544};

547/**
548 * AES's RotWord procedure.
549 * @param {!Array<number>} w Array of bytes to rotate.
550 * @return {!Array<number>} The rotated bytes.
551 * @private
552 */
553goog.crypt.Aes.prototype.rotWord_ = function(w) {
554 var temp = w[0];
555
556 w[0] = w[1];
557 w[1] = w[2];
558 w[2] = w[3];
559 w[3] = temp;
560
561 return w;
562};
