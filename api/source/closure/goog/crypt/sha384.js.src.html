32/**
33 * Constructs a SHA-384 cryptographic hash.
34 *
35 * @constructor
36 * @extends {goog.crypt.Sha2_64bit}
37 * @final
38 * @struct

40goog.crypt.Sha384 = function() {
41 goog.crypt.Sha384.base(this, 'constructor', 6 /* numHashBlocks */,
42 goog.crypt.Sha384.INIT_HASH_BLOCK_);
