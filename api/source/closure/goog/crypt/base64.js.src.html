114/**
115 * Base64-encode an array of bytes.
116 *
117 * @param {Array<number>|Uint8Array} input An array of bytes (numbers with
118 * value in [0, 255]) to encode.
119 * @param {boolean=} opt_webSafe True indicates we should use the alternative
120 * alphabet, which does not require escaping for use in URLs.
121 * @return {string} The base64 encoded string.
122 */
123goog.crypt.base64.encodeByteArray = function(input, opt_webSafe) {
124 // Assert avoids runtime dependency on goog.isArrayLike, which helps reduce
125 // size of jscompiler output, and which yields slight performance increase.
126 goog.asserts.assert(goog.isArrayLike(input),
127 'encodeByteArray takes an array as a parameter');
128
129 goog.crypt.base64.init_();
130
131 var byteToCharMap = opt_webSafe ?
132 goog.crypt.base64.byteToCharMapWebSafe_ :
133 goog.crypt.base64.byteToCharMap_;
134
135 var output = [];
136
137 for (var i = 0; i < input.length; i += 3) {
138 var byte1 = input[i];
139 var haveByte2 = i + 1 < input.length;
140 var byte2 = haveByte2 ? input[i + 1] : 0;
141 var haveByte3 = i + 2 < input.length;
142 var byte3 = haveByte3 ? input[i + 2] : 0;
143
144 var outByte1 = byte1 >> 2;
145 var outByte2 = ((byte1 & 0x03) << 4) | (byte2 >> 4);
146 var outByte3 = ((byte2 & 0x0F) << 2) | (byte3 >> 6);
147 var outByte4 = byte3 & 0x3F;
148
149 if (!haveByte3) {
150 outByte4 = 64;
151
152 if (!haveByte2) {
153 outByte3 = 64;
154 }
155 }
156
157 output.push(byteToCharMap[outByte1],
158 byteToCharMap[outByte2],
159 byteToCharMap[outByte3],
160 byteToCharMap[outByte4]);
161 }
162
163 return output.join('');
164};

167/**
168 * Base64-encode a string.
169 *
170 * @param {string} input A string to encode.
171 * @param {boolean=} opt_webSafe True indicates we should use the alternative
172 * alphabet, which does not require escaping for use in URLs.
173 * @return {string} The base64 encoded string.
174 */
175goog.crypt.base64.encodeString = function(input, opt_webSafe) {
176 // Shortcut for browsers that implement
177 // a native base64 encoder in the form of "btoa/atob"
178 if (goog.crypt.base64.HAS_NATIVE_ENCODE_ && !opt_webSafe) {
179 return goog.global.btoa(input);
180 }
181 return goog.crypt.base64.encodeByteArray(
182 goog.crypt.stringToByteArray(input), opt_webSafe);
183};

186/**
187 * Base64-decode a string.
188 *
189 * @param {string} input Input to decode. Any whitespace is ignored, and the
190 * input maybe encoded with either supported alphabet (or a mix thereof).
191 * @param {boolean=} opt_webSafe True indicates we should use the alternative
192 * alphabet, which does not require escaping for use in URLs. Note that
193 * passing false may also still allow webSafe input decoding, when the
194 * fallback decoder is used on browsers without native support.
195 * @return {string} string representing the decoded value.
196 */
197goog.crypt.base64.decodeString = function(input, opt_webSafe) {
198 // Shortcut for browsers that implement
199 // a native base64 encoder in the form of "btoa/atob"
200 if (goog.crypt.base64.HAS_NATIVE_DECODE_ && !opt_webSafe) {
201 return goog.global.atob(input);
202 }
203 var output = '';
204 function pushByte(b) { output += String.fromCharCode(b); }
205
206 goog.crypt.base64.decodeStringInternal_(input, pushByte);
207
208 return output;
209};

212/**
213 * Base64-decode a string to an Array of numbers.
214 *
215 * In base-64 decoding, groups of four characters are converted into three
216 * bytes. If the encoder did not apply padding, the input length may not
217 * be a multiple of 4.
218 *
219 * In this case, the last group will have fewer than 4 characters, and
220 * padding will be inferred. If the group has one or two characters, it decodes
221 * to one byte. If the group has three characters, it decodes to two bytes.
222 *
223 * @param {string} input Input to decode. Any whitespace is ignored, and the
224 * input maybe encoded with either supported alphabet (or a mix thereof).
225 * @param {boolean=} opt_ignored Unused parameter, retained for compatibility.
226 * @return {!Array<number>} bytes representing the decoded value.
227 */
228goog.crypt.base64.decodeStringToByteArray = function(input, opt_ignored) {
229 var output = [];
230 function pushByte(b) { output.push(b); }
231
232 goog.crypt.base64.decodeStringInternal_(input, pushByte);
233
234 return output;
235};

238/**
239 * Base64-decode a string to a Uint8Array.
240 *
241 * Note that Uint8Array is not supported on older browsers, e.g. IE < 10.
242 * @see http://caniuse.com/uint8array
243 *
244 * In base-64 decoding, groups of four characters are converted into three
245 * bytes. If the encoder did not apply padding, the input length may not
246 * be a multiple of 4.
247 *
248 * In this case, the last group will have fewer than 4 characters, and
249 * padding will be inferred. If the group has one or two characters, it decodes
250 * to one byte. If the group has three characters, it decodes to two bytes.
251 *
252 * @param {string} input Input to decode. Any whitespace is ignored, and the
253 * input maybe encoded with either supported alphabet (or a mix thereof).
254 * @return {!Uint8Array} bytes representing the decoded value.
255 */
256goog.crypt.base64.decodeStringToUint8Array = function(input) {
257 goog.asserts.assert(
258 !goog.userAgent.IE || goog.userAgent.isVersionOrHigher('10'),
259 'Browser does not support typed arrays');
260 var output = new Uint8Array(Math.ceil(input.length * 3 / 4));
261 var outLen = 0;
262 function pushByte(b) { output[outLen++] = b; }
263
264 goog.crypt.base64.decodeStringInternal_(input, pushByte);
265
266 return output.subarray(0, outLen);
267};

270/**
271 * @param {string} input Input to decode.
272 * @param {function(number):void} pushByte result accumulator.
273 * @private
274 */
275goog.crypt.base64.decodeStringInternal_ = function(input, pushByte) {
276 goog.crypt.base64.init_();
277
278 var nextCharIndex = 0;
279 /** @return {?number} The next 6-bit value, or null for end-of-input. */
280 function getByte() {
281 while (nextCharIndex < input.length) {
282 var ch = input.charAt(nextCharIndex++);
283 var b = goog.crypt.base64.charToByteMap_[ch];
284 if (b != null) {
285 return b; // Common case: decoded the char.
286 }
287 if (!goog.string.isEmptyOrWhitespace(ch)) {
288 throw Error('Unknown base64 encoding at char: ' + ch);
289 }
290 // We encountered whitespace: loop around to the next input char.
291 }
292 return null; // No more input remaining.
293 }
294
295 while (true) {
296 var byte1 = getByte();
297 var byte2 = getByte();
298 var byte3 = getByte();
299 var byte4 = getByte();
300
301 // The common case is that all four bytes are present, so if we have byte4
302 // we can skip over the truncated input special case handling.
303 if (byte4 == null) {
304 if (byte1 == null) {
305 return; // Terminal case: no input left to decode.
306 }
307 // Here we know an intermediate number of bytes are missing, so apply the
308 // inferred padding rules per the public API documentation. i.e: 1 byte
309 // missing should yield 2 bytes of output, but 2 or 3 missing bytes yield
310 // a single byte of output. (Recall that 64 corresponds the padding char).
311 byte4 = 64;
312 byte3 = byte3 != null ? byte3 : 64;
313 byte2 = byte2 != null ? byte2 : 0;
314 }
315
316 var outByte1 = (byte1 << 2) | (byte2 >> 4);
317 pushByte(outByte1);
318
319 if (byte3 != 64) {
320 var outByte2 = ((byte2 << 4) & 0xF0) | (byte3 >> 2);
321 pushByte(outByte2);
322
323 if (byte4 != 64) {
324 var outByte3 = ((byte3 << 6) & 0xC0) | byte4;
325 pushByte(outByte3);
326 }
327 }
328 }
329};

332/**
333 * Lazy static initialization function. Called before
334 * accessing any of the static map variables.
335 * @private
336 */
337goog.crypt.base64.init_ = function() {
338 if (!goog.crypt.base64.byteToCharMap_) {
339 goog.crypt.base64.byteToCharMap_ = {};
340 goog.crypt.base64.charToByteMap_ = {};
341 goog.crypt.base64.byteToCharMapWebSafe_ = {};
342
343 // We want quick mappings back and forth, so we precompute two maps.
344 for (var i = 0; i < goog.crypt.base64.ENCODED_VALS.length; i++) {
345 goog.crypt.base64.byteToCharMap_[i] =
346 goog.crypt.base64.ENCODED_VALS.charAt(i);
347 goog.crypt.base64.charToByteMap_[goog.crypt.base64.byteToCharMap_[i]] = i;
348 goog.crypt.base64.byteToCharMapWebSafe_[i] =
349 goog.crypt.base64.ENCODED_VALS_WEBSAFE.charAt(i);
350
351 // Be forgiving when decoding and correctly decode both encodings.
352 if (i >= goog.crypt.base64.ENCODED_VALS_BASE.length) {
353 goog.crypt.base64.charToByteMap_[
354 goog.crypt.base64.ENCODED_VALS_WEBSAFE.charAt(i)] = i;
355 }
356 }
357 }
