99/**
100 * Converts number of bytes to string representation. Binary conversion.
101 * Default is to return the additional 'B' suffix, e.g. '10.5KB' to minimize
102 * confusion with counts that are scaled by powers of 1000.
103 * @param {number} val Value to be converted.
104 * @param {number=} opt_decimals The number of decimals to use. Defaults to 2.
105 * @param {boolean=} opt_suffix If true, include trailing 'B' in returned
106 * string. Default is true.
107 * @param {boolean=} opt_useSeparator If true, number and scale will be
108 * separated by a no break space. Default is false.
109 * @return {string} String representation of number of bytes.
110 */
111goog.format.numBytesToString = function(val, opt_decimals, opt_suffix,
112 opt_useSeparator) {
113 var suffix = '';
114 if (!goog.isDef(opt_suffix) || opt_suffix) {
115 suffix = 'B';
116 }
117 return goog.format.numericValueToString_(
118 val, goog.format.NUMERIC_SCALES_BINARY_, opt_decimals, suffix,
119 opt_useSeparator);
120};

123/**
124 * Converts a string to numeric value, taking into account the units.
125 * @param {string} stringValue String to be converted to numeric value.
126 * @param {Object} conversion Dictionary of conversion scales.
127 * @return {number} Numeric value for string. If it cannot be converted,
128 * returns NaN.
129 * @private
130 */
131goog.format.stringToNumericValue_ = function(stringValue, conversion) {
132 var match = stringValue.match(goog.format.SCALED_NUMERIC_RE_);
133 if (!match) {
134 return NaN;
135 }
136 var val = match[1] * conversion[match[2]];
137 return val;
138};

141/**
142 * Converts a numeric value to string, using specified conversion
143 * scales.
144 * @param {number} val Value to be converted.
145 * @param {Object} conversion Dictionary of scaling factors.
146 * @param {number=} opt_decimals The number of decimals to use. Default is 2.
147 * @param {string=} opt_suffix Optional suffix to append.
148 * @param {boolean=} opt_useSeparator If true, number and scale will be
149 * separated by a space. Default is false.
150 * @return {string} The human readable form of the byte size.
151 * @private
152 */
153goog.format.numericValueToString_ = function(val, conversion,
154 opt_decimals, opt_suffix, opt_useSeparator) {
155 var prefixes = goog.format.NUMERIC_SCALE_PREFIXES_;
156 var orig_val = val;
157 var symbol = '';
158 var separator = '';
159 var scale = 1;
160 if (val < 0) {
161 val = -val;
162 }
163 for (var i = 0; i < prefixes.length; i++) {
164 var unit = prefixes[i];
165 scale = conversion[unit];
166 if (val >= scale || (scale <= 1 && val > 0.1 * scale)) {
167 // Treat values less than 1 differently, allowing 0.5 to be "0.5" rather
168 // than "500m"
169 symbol = unit;
170 break;
171 }
172 }
173 if (!symbol) {
174 scale = 1;
175 } else {
176 if (opt_suffix) {
177 symbol += opt_suffix;
178 }
179 if (opt_useSeparator) {
180 separator = ' ';
181 }
182 }
183 var ex = Math.pow(10, goog.isDef(opt_decimals) ? opt_decimals : 2);
184 return Math.round(orig_val / scale * ex) / ex + separator + symbol;
185};

188/**
189 * Regular expression for detecting scaling units, such as K, M, G, etc. for
190 * converting a string representation to a numeric value.
191 *
192 * Also allow 'k' to be aliased to 'K'. These could be used for SI (powers
193 * of 1000) or Binary (powers of 1024) conversions.
194 *
195 * Also allow final 'B' to be interpreted as byte-count, implicitly triggering
196 * binary conversion (e.g., '10.2MB').
197 *
198 * @type {RegExp}
199 * @private
200 */
201goog.format.SCALED_NUMERIC_RE_ = /^([-]?\d+\.?\d*)([K,M,G,T,P,k,m,u,n]?)[B]?$/;

260/**
261 * Returns true if and only if given character should be treated as a breaking
262 * space. All ASCII control characters, the main Unicode range of spacing
263 * characters (U+2000 to U+200B inclusive except for U+2007), and several other
264 * Unicode space characters are treated as breaking spaces.
265 * @param {number} charCode The character code under consideration.
266 * @return {boolean} True if the character is a breaking space.
267 * @private
268 */
269goog.format.isTreatedAsBreakingSpace_ = function(charCode) {
270 return (charCode <= goog.format.WbrToken_.SPACE) ||
271 (charCode >= 0x1000 &&
272 ((charCode >= 0x2000 && charCode <= 0x2006) ||
273 (charCode >= 0x2008 && charCode <= 0x200B) ||
274 charCode == 0x1680 ||
275 charCode == 0x180E ||
276 charCode == 0x2028 ||
277 charCode == 0x2029 ||
278 charCode == 0x205f ||
279 charCode == 0x3000));
280};

290goog.format.isInvisibleFormattingCharacter_ = function(charCode) {

292 return (charCode >= 0x200C && charCode <= 0x200F) ||
293 (charCode >= 0x202A && charCode <= 0x202E);

297/**
298 * Inserts word breaks into an HTML string at a given interval. The counter is
299 * reset if a space or a character which behaves like a space is encountered,
300 * but it isn't incremented if an invisible formatting character is encountered.
301 * WBRs aren't inserted into HTML tags or entities. Entities count towards the
302 * character count, HTML tags do not.
303 *
304 * With common strings aliased, objects allocations are constant based on the
305 * length of the string: N + 3. This guarantee does not hold if the string
306 * contains an element >= U+0300 and hasGraphemeBreak is non-trivial.
307 *
308 * @param {string} str HTML to insert word breaks into.
309 * @param {function(number, number, boolean): boolean} hasGraphemeBreak A
310 * function determining if there is a grapheme break between two characters,
311 * in the same signature as goog.i18n.GraphemeBreak.hasGraphemeBreak.
312 * @param {number=} opt_maxlen Maximum length after which to ensure
313 * there is a break. Default is 10 characters.
314 * @return {string} The string including word breaks.
315 * @private
316 */
317goog.format.insertWordBreaksGeneric_ = function(str, hasGraphemeBreak,
318 opt_maxlen) {
319 var maxlen = opt_maxlen || 10;
320 if (maxlen > str.length) return str;
321
322 var rv = [];
323 var n = 0; // The length of the current token
324
325 // This will contain the ampersand or less-than character if one of the
326 // two has been seen; otherwise, the value is zero.
327 var nestingCharCode = 0;
328
329 // First character position from input string that has not been outputted.
330 var lastDumpPosition = 0;
331
332 var charCode = 0;
333 for (var i = 0; i < str.length; i++) {
334 // Using charCodeAt versus charAt avoids allocating new string objects.
335 var lastCharCode = charCode;
336 charCode = str.charCodeAt(i);
337
338 // Don't add a WBR before characters that might be grapheme extending.
339 var isPotentiallyGraphemeExtending =
340 charCode >= goog.format.FIRST_GRAPHEME_EXTEND_ &&
341 !hasGraphemeBreak(lastCharCode, charCode, true);
342
343 // Don't add a WBR at the end of a word. For the purposes of determining
344 // work breaks, all ASCII control characters and some commonly encountered
345 // Unicode spacing characters are treated as breaking spaces.
346 if (n >= maxlen &&
347 !goog.format.isTreatedAsBreakingSpace_(charCode) &&
348 !isPotentiallyGraphemeExtending) {
349 // Flush everything seen so far, and append a word break.
350 rv.push(str.substring(lastDumpPosition, i), goog.format.WORD_BREAK_HTML);
351 lastDumpPosition = i;
352 n = 0;
353 }
354
355 if (!nestingCharCode) {
356 // Not currently within an HTML tag or entity
357
358 if (charCode == goog.format.WbrToken_.LT ||
359 charCode == goog.format.WbrToken_.AMP) {
360
361 // Entering an HTML Entity '&' or open tag '<'
362 nestingCharCode = charCode;
363 } else if (goog.format.isTreatedAsBreakingSpace_(charCode)) {
364
365 // A space or control character -- reset the token length
366 n = 0;
367 } else if (!goog.format.isInvisibleFormattingCharacter_(charCode)) {
368
369 // A normal flow character - increment. For grapheme extending
370 // characters, this is not *technically* a new character. However,
371 // since the grapheme break detector might be overly conservative,
372 // we have to continue incrementing, or else we won't even be able
373 // to add breaks when we get to things like punctuation. For the
374 // case where we have a full grapheme break detector, it is okay if
375 // we occasionally break slightly early.
376 n++;
377 }
378 } else if (charCode == goog.format.WbrToken_.GT &&
379 nestingCharCode == goog.format.WbrToken_.LT) {
380
381 // Leaving an HTML tag, treat the tag as zero-length
382 nestingCharCode = 0;
383 } else if (charCode == goog.format.WbrToken_.SEMI_COLON &&
384 nestingCharCode == goog.format.WbrToken_.AMP) {
385
386 // Leaving an HTML entity, treat it as length one
387 nestingCharCode = 0;
388 n++;
389 }
390 }
391
392 // Take care of anything we haven't flushed so far.
393 rv.push(str.substr(lastDumpPosition));
394
395 return rv.join('');
396};

399/**
400 * Inserts word breaks into an HTML string at a given interval.
401 *
402 * This method is as aggressive as possible, using a full table of Unicode
403 * characters where it is legal to insert word breaks; however, this table
404 * comes at a 2.5k pre-gzip (~1k post-gzip) size cost. Consider using
405 * insertWordBreaksBasic to minimize the size impact.
406 *
407 * @param {string} str HTML to insert word breaks into.
408 * @param {number=} opt_maxlen Maximum length after which to ensure there is a
409 * break. Default is 10 characters.
410 * @return {string} The string including word breaks.
411 */
412goog.format.insertWordBreaks = function(str, opt_maxlen) {
413 return goog.format.insertWordBreaksGeneric_(str,
414 goog.i18n.GraphemeBreak.hasGraphemeBreak, opt_maxlen);
415};

418/**
419 * Determines conservatively if a character has a Grapheme break.
420 *
421 * Conforms to a similar signature as goog.i18n.GraphemeBreak, but is overly
422 * conservative, returning true only for characters in common scripts that
423 * are simple to account for.
424 *
425 * @param {number} lastCharCode The previous character code. Ignored.
426 * @param {number} charCode The character code under consideration. It must be
427 * at least \u0300 as a precondition -- this case is covered by
428 * insertWordBreaksGeneric_.
429 * @param {boolean=} opt_extended Ignored, to conform with the interface.
430 * @return {boolean} Whether it is one of the recognized subsets of characters
431 * with a grapheme break.
432 * @private
433 */
434goog.format.conservativelyHasGraphemeBreak_ = function(
435 lastCharCode, charCode, opt_extended) {
436 // Return false for everything except the most common Cyrillic characters.
437 // Don't worry about Latin characters, because insertWordBreaksGeneric_
438 // itself already handles those.
439 // TODO(gboyer): Also account for Greek, Armenian, and Georgian if it is
440 // simple to do so.
441 return charCode >= 0x400 && charCode < 0x523;
442};

447/**
448 * Inserts word breaks into an HTML string at a given interval.
449 *
450 * This method is less aggressive than insertWordBreaks, only inserting
451 * breaks next to punctuation and between Latin or Cyrillic characters.
452 * However, this is good enough for the common case of URLs. It also
453 * works for all Latin and Cyrillic languages, plus CJK has no need for word
454 * breaks. When this method is used, goog.i18n.GraphemeBreak may be dead
455 * code eliminated.
456 *
457 * @param {string} str HTML to insert word breaks into.
458 * @param {number=} opt_maxlen Maximum length after which to ensure there is a
459 * break. Default is 10 characters.
460 * @return {string} The string including word breaks.
461 */
462goog.format.insertWordBreaksBasic = function(str, opt_maxlen) {
463 return goog.format.insertWordBreaksGeneric_(str,
464 goog.format.conservativelyHasGraphemeBreak_, opt_maxlen);
465};

477/**
478 * Constant for the WBR replacement used by insertWordBreaks. Safari requires
479 * <wbr></wbr>, Opera needs the &shy; entity, though this will give a visible
480 * hyphen at breaks. IE8 uses a zero width space.
481 * Other browsers just use <wbr>.
482 * @type {string}
483 */
484goog.format.WORD_BREAK_HTML =
485 goog.userAgent.WEBKIT ?
486 '<wbr></wbr>' : goog.userAgent.OPERA ?
487 '&shy;' : goog.format.IS_IE8_OR_ABOVE_ ?
488 '&#8203;' : '<wbr>';
