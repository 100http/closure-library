46/**
47 * A UI for editing tweak settings / clicking tweak actions.
48 * @param {!goog.tweak.Registry} registry The registry to render.
49 * @param {goog.dom.DomHelper=} opt_domHelper The DomHelper to render with.
50 * @constructor
51 * @final
52 */
53goog.tweak.TweakUi = function(registry, opt_domHelper) {
54 /**
55 * The registry to create a UI from.
56 * @type {!goog.tweak.Registry}
57 * @private
58 */
59 this.registry_ = registry;
60
61 /**
62 * The element to display when the UI is visible.
63 * @type {goog.tweak.EntriesPanel|undefined}
64 * @private
65 */
66 this.entriesPanel_;
67
68 /**
69 * The DomHelper to render with.
70 * @type {!goog.dom.DomHelper}
71 * @private
72 */
73 this.domHelper_ = opt_domHelper || goog.dom.getDomHelper();
74
75 // Listen for newly registered entries (happens with lazy-loaded modules).
76 registry.addOnRegisterListener(goog.bind(this.onNewRegisteredEntry_, this));
77};

166 * @param {goog.dom.DomHelper=} opt_domHelper The DomHelper to render with.

202/**
203 * Compares the given entries. Orders alphabetically and groups buttons and
204 * expandable groups.
205 * @param {goog.tweak.BaseEntry} a The first entry to compare.
206 * @param {goog.tweak.BaseEntry} b The second entry to compare.
207 * @return {number} Refer to goog.array.defaultCompare.
208 * @private
209 */
210goog.tweak.TweakUi.entryCompare_ = function(a, b) {
211 return (
212 goog.array.defaultCompare(a instanceof goog.tweak.NamespaceEntry_,
213 b instanceof goog.tweak.NamespaceEntry_) ||
214 goog.array.defaultCompare(a instanceof goog.tweak.BooleanGroup,
215 b instanceof goog.tweak.BooleanGroup) ||
216 goog.array.defaultCompare(a instanceof goog.tweak.ButtonAction,
217 b instanceof goog.tweak.ButtonAction) ||
218 goog.array.defaultCompare(a.label, b.label) ||
219 goog.array.defaultCompare(a.getId(), b.getId()));
220};

224 * @param {!goog.tweak.BaseEntry} entry The entry.
225 * @return {boolean} Returns whether the given entry contains sub-entries.
226 * @private

228goog.tweak.TweakUi.isGroupEntry_ = function(entry) {
229 return entry instanceof goog.tweak.NamespaceEntry_ ||
230 entry instanceof goog.tweak.BooleanGroup;

234/**
235 * Returns the list of entries from the given boolean group.
236 * @param {!goog.tweak.BooleanGroup} group The group to get the entries from.
237 * @return {!Array<!goog.tweak.BaseEntry>} The sorted entries.
238 * @private
239 */
240goog.tweak.TweakUi.extractBooleanGroupEntries_ = function(group) {
241 var ret = goog.object.getValues(group.getChildEntries());
242 ret.sort(goog.tweak.TweakUi.entryCompare_);
243 return ret;

247/**
248 * @param {!goog.tweak.BaseEntry} entry The entry.
249 * @return {string} Returns the namespace for the entry, or '' if it is not
250 * namespaced.
251 * @private
252 */
253goog.tweak.TweakUi.extractNamespace_ = function(entry) {
254 var namespaceMatch = /.+(?=\.)/.exec(entry.getId());
255 return namespaceMatch ? namespaceMatch[0] : '';
256};

259/**
260 * @param {!goog.tweak.BaseEntry} entry The entry.
261 * @return {string} Returns the part of the label after the last period, unless
262 * the label has been explicly set (it is different from the ID).
263 * @private
264 */
265goog.tweak.TweakUi.getNamespacedLabel_ = function(entry) {
266 var label = entry.label;
267 if (label == entry.getId()) {
268 label = label.substr(label.lastIndexOf('.') + 1);
269 }
270 return label;

289 var queryString = this.registry_.makeUrlQuery();

300 * Installs the required CSS styles.
301 * @private
302 */
303goog.tweak.TweakUi.prototype.installStyles_ = function() {
304 // Use an marker to install the styles only once per document.
305 // Styles are injected via JS instead of in a separate style sheet so that
306 // they are automatically excluded when tweaks are stripped out.
307 var doc = this.domHelper_.getDocument();
308 if (!(goog.tweak.TweakUi.STYLE_SHEET_INSTALLED_MARKER_ in doc)) {
309 goog.style.installStyles(

311 doc[goog.tweak.TweakUi.STYLE_SHEET_INSTALLED_MARKER_] = true;

316/**
317 * Creates the element to display when the UI is visible.
318 * @return {!Element} The root element.
319 */
320goog.tweak.TweakUi.prototype.render = function() {
321 this.installStyles_();
322 var dh = this.domHelper_;
323 // The submit button
324 var submitButton = dh.createDom(
325 goog.dom.TagName.BUTTON, {style: 'font-weight:bold'}, 'Apply Tweaks');
326 submitButton.onclick = goog.bind(this.restartWithAppliedTweaks_, this);
327
328 var rootPanel = new goog.tweak.EntriesPanel([], dh);
329 var rootPanelDiv = rootPanel.render(submitButton);
330 rootPanelDiv.className += ' ' + goog.tweak.TweakUi.ROOT_PANEL_CLASS_;
331 this.entriesPanel_ = rootPanel;
332
333 var entries = this.registry_.extractEntries(true /* excludeChildEntries */,
334 false /* excludeNonSettings */);
335 for (var i = 0, entry; entry = entries[i]; i++) {
336 this.insertEntry_(entry);
337 }
338
339 return rootPanelDiv;
340};

355/**
356 * Updates the UI with the given entry.
357 * @param {!goog.tweak.BaseEntry} entry The newly registered entry.
358 * @private
359 */
360goog.tweak.TweakUi.prototype.insertEntry_ = function(entry) {
361 var panel = this.entriesPanel_;
362 var namespace = goog.tweak.TweakUi.extractNamespace_(entry);
363
364 if (namespace) {
365 // Find the NamespaceEntry that the entry belongs to.
366 var namespaceEntryId = goog.tweak.NamespaceEntry_.ID_PREFIX + namespace;
367 var nsPanel = panel.childPanels[namespaceEntryId];
368 if (nsPanel) {
369 panel = nsPanel;
370 } else {
371 entry = new goog.tweak.NamespaceEntry_(namespace, [entry]);
372 }
373 }
374 if (entry instanceof goog.tweak.BooleanInGroupSetting) {
375 var group = entry.getGroup();
376 // BooleanGroup entries are always registered before their
377 // BooleanInGroupSettings.
378 panel = panel.childPanels[group.getId()];
379 }
380 goog.asserts.assert(panel, 'Missing panel for entry %s', entry.getId());
381 panel.insertEntry(entry);
382};

460/**
461 * Creates and returns the expanded element.
462 * The markup looks like:
463 * <div>
464 * <a>Show Descriptions</a>
465 * <div>
466 * ...
467 * {endElement}
468 * </div>
469 * </div>
470 * @param {Element|DocumentFragment=} opt_endElement Element to insert after all
471 * tweak entries.
472 * @return {!Element} The root element for the panel.
473 */
474goog.tweak.EntriesPanel.prototype.render = function(opt_endElement) {
475 var dh = this.domHelper_;
476 var entries = this.entries_;
477 var ret = dh.createDom(goog.dom.TagName.DIV);
478
479 var showAllDescriptionsLink = dh.createDom(goog.dom.TagName.A, {
480 href: 'javascript:;',
481 onclick: goog.bind(this.toggleAllDescriptions, this)
482 }, 'Toggle all Descriptions');
483 ret.appendChild(showAllDescriptionsLink);
484
485 // Add all of the entries.
486 var mainPanel = dh.createElement(goog.dom.TagName.DIV);
487 this.mainPanel_ = mainPanel;
488 for (var i = 0, entry; entry = entries[i]; i++) {
489 mainPanel.appendChild(this.createEntryElem_(entry));
490 }
491
492 if (opt_endElement) {
493 mainPanel.appendChild(opt_endElement);
494 }
495 ret.appendChild(mainPanel);
496 this.rootElem_ = ret;
497 return /** @type {!Element} */ (ret);
498};

501/**
502 * Inserts the given entry into the panel.
503 * @param {!goog.tweak.BaseEntry} entry The entry to insert.
504 */
505goog.tweak.EntriesPanel.prototype.insertEntry = function(entry) {
506 var insertIndex = -goog.array.binarySearch(this.entries_, entry,
507 goog.tweak.TweakUi.entryCompare_) - 1;
508 goog.asserts.assert(insertIndex >= 0, 'insertEntry failed for %s',
509 entry.getId());
510 goog.array.insertAt(this.entries_, entry, insertIndex);
511 this.mainPanel_.insertBefore(
512 this.createEntryElem_(entry),
513 // IE doesn't like 'undefined' here.
514 this.mainPanel_.childNodes[insertIndex] || null);
515};

518/**
519 * Creates and returns a form element for the given entry.
520 * @param {!goog.tweak.BaseEntry} entry The entry.
521 * @return {!Element} The root DOM element for the entry.
522 * @private
523 */
524goog.tweak.EntriesPanel.prototype.createEntryElem_ = function(entry) {
525 var dh = this.domHelper_;
526 var isGroupEntry = goog.tweak.TweakUi.isGroupEntry_(entry);
527 var classes = isGroupEntry ? goog.tweak.TweakUi.ENTRY_GROUP_CSS_CLASSES_ :
528 goog.tweak.TweakUi.ENTRY_CSS_CLASSES_;
529 // Containers should not use label tags or else all descendent inputs will be
530 // connected on desktop browsers.
531 var containerNodeName = isGroupEntry ?
532 goog.dom.TagName.SPAN : goog.dom.TagName.LABEL;
533 var ret = dh.createDom(goog.dom.TagName.DIV, classes,
534 dh.createDom(containerNodeName, {
535 // Make the hover text the description.
536 title: entry.description,
537 style: 'color:' + (entry.isRestartRequired() ? '' : 'blue')
538 }, this.createTweakEntryDom_(entry)),
539 // Add the expandable help question mark.
540 this.createHelpElem_(entry));
541 return ret;
542};

569/**
570 * Creates and returns a help element for the given entry.
571 * @param {goog.tweak.BaseEntry} entry The entry.
572 * @return {!Element} The root element of the created DOM.
573 * @private
574 */
575goog.tweak.EntriesPanel.prototype.createHelpElem_ = function(entry) {
576 // The markup looks like:
577 // <span onclick=...><b>?</b><span>{description}</span></span>
578 var ret = this.domHelper_.createElement(goog.dom.TagName.SPAN);
579 goog.dom.safe.setInnerHtml(ret, goog.html.SafeHtml.concat(
580 goog.html.SafeHtml.create('b',
581 {'style': goog.string.Const.from('padding:0 1em 0 .5em')}, '?'),
582 goog.html.SafeHtml.create('span',
583 {'style': goog.string.Const.from('display:none;color:#666')})));
584 ret.onclick = this.boundHelpOnClickHandler_;
585 // IE<9 doesn't support lastElementChild.
586 var descriptionElem = /** @type {!Element} */ (ret.lastChild);
587 if (entry.isRestartRequired()) {
588 goog.dom.setTextContent(descriptionElem, entry.description);
589 } else {
590 goog.dom.safe.setInnerHtml(descriptionElem, goog.html.SafeHtml.concat(
591 goog.html.SafeHtml.htmlEscape(entry.description),
592 goog.html.SafeHtml.create('span',
593 {'style': goog.string.Const.from('color: blue')},
594 '(no restart required)')));
595 }
596 return ret;
597};

615/**
616 * Creates the DOM element to control the given enum setting.
617 * @param {!goog.tweak.StringSetting|!goog.tweak.NumericSetting} tweak The
618 * setting.
619 * @param {string} label The label for the entry.
620 * @param {!Function} onchangeFunc onchange event handler.
621 * @return {!DocumentFragment} The DOM element.
622 * @private
623 */
624goog.tweak.EntriesPanel.prototype.createComboBoxDom_ =
625 function(tweak, label, onchangeFunc) {
626 // The markup looks like:
627 // Label: <select><option></option></select>
628 var dh = this.domHelper_;
629 var ret = dh.getDocument().createDocumentFragment();
630 ret.appendChild(dh.createTextNode(label + ': '));
631 var selectElem = dh.createElement(goog.dom.TagName.SELECT);
632 var values = tweak.getValidValues();
633 for (var i = 0, il = values.length; i < il; ++i) {
634 var optionElem = dh.createElement(goog.dom.TagName.OPTION);
635 optionElem.text = String(values[i]);
636 // Setting the option tag's value is required for selectElem.value to work
637 // properly.
638 optionElem.value = String(values[i]);
639 selectElem.appendChild(optionElem);
640 }
641 ret.appendChild(selectElem);
642
643 // Set the value and add a callback.
644 selectElem.value = tweak.getNewValue();
645 selectElem.onchange = onchangeFunc;
646 tweak.addCallback(function() {
647 selectElem.value = tweak.getNewValue();
648 });
649 return ret;
650};

677 checkbox.checked = tweak.getNewValue();
678 });

683/**
684 * Creates the DOM for a BooleanGroup or NamespaceEntry.
685 * @param {!goog.tweak.BooleanGroup|!goog.tweak.NamespaceEntry_} entry The
686 * entry.
687 * @param {string} label The label for the entry.
688 * @param {!Array<goog.tweak.BaseEntry>} childEntries The child entries.
689 * @return {!DocumentFragment} The DOM element.
690 * @private
691 */
692goog.tweak.EntriesPanel.prototype.createSubPanelDom_ = function(entry, label,
693 childEntries) {
694 var dh = this.domHelper_;
695 var toggleLink = dh.createDom(goog.dom.TagName.A, {href: 'javascript:;'},
696 label + ' \xBB');
697 var toggleLink2 = dh.createDom(goog.dom.TagName.A, {href: 'javascript:;'},
698 '\xAB ' + label);
699 toggleLink2.style.marginRight = '10px';
700
701 var innerUi = new goog.tweak.EntriesPanel(childEntries, dh);
702 this.childPanels[entry.getId()] = innerUi;
703
704 var elem = innerUi.render();
705 // Move the toggle descriptions link into the legend.
706 var descriptionsLink = elem.firstChild;
707 var childrenElem = dh.createDom(goog.dom.TagName.FIELDSET,
708 goog.getCssName('goog-inline-block'),
709 dh.createDom(goog.dom.TagName.LEGEND, null,
710 toggleLink2, descriptionsLink),
711 elem);
712
713 new goog.ui.Zippy(toggleLink, childrenElem, false /* expanded */,
714 toggleLink2);
715

766/**
767 * Creates the DOM element to control the given entry.
768 * @param {!goog.tweak.BaseEntry} entry The entry.
769 * @return {!Element|!DocumentFragment} The DOM element.
770 * @private
771 */
772goog.tweak.EntriesPanel.prototype.createTweakEntryDom_ = function(entry) {
773 var label = goog.tweak.TweakUi.getNamespacedLabel_(entry);
774 if (entry instanceof goog.tweak.BooleanSetting) {
775 return this.createBooleanSettingDom_(entry, label);
776 } else if (entry instanceof goog.tweak.BooleanGroup) {
777 var childEntries = goog.tweak.TweakUi.extractBooleanGroupEntries_(entry);
778 return this.createSubPanelDom_(entry, label, childEntries);
779 } else if (entry instanceof goog.tweak.StringSetting) {
780 /** @this {Element} */
781 var setValueFunc = function() {
782 entry.setValue(this.value);
783 };
784 return entry.getValidValues() ?
785 this.createComboBoxDom_(entry, label, setValueFunc) :
786 this.createTextBoxDom_(entry, label, setValueFunc);
787 } else if (entry instanceof goog.tweak.NumericSetting) {
788 setValueFunc = function() {
789 // Reset the value if it's not a number.
790 if (isNaN(this.value)) {
791 this.value = entry.getNewValue();
792 } else {
793 entry.setValue(+this.value);
794 }
795 };
796 return entry.getValidValues() ?
797 this.createComboBoxDom_(entry, label, setValueFunc) :
798 this.createTextBoxDom_(entry, label, setValueFunc);
799 } else if (entry instanceof goog.tweak.NamespaceEntry_) {
800 return this.createSubPanelDom_(entry, entry.label, entry.entries);
801 }
802 goog.asserts.assertInstanceof(entry, goog.tweak.ButtonAction,
803 'invalid entry: %s', entry);
804 return this.createButtonActionDom_(
805 /** @type {!goog.tweak.ButtonAction} */ (entry), label);
806};
