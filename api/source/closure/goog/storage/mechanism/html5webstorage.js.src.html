31/**
32 * Provides a storage mechanism that uses HTML5 Web storage.
33 *
34 * @param {Storage} storage The Web storage object.
35 * @constructor
36 * @struct
37 * @extends {goog.storage.mechanism.IterableMechanism}
38 */
39goog.storage.mechanism.HTML5WebStorage = function(storage) {
40 goog.storage.mechanism.HTML5WebStorage.base(this, 'constructor');
41
42 /**
43 * The web storage object (window.localStorage or window.sessionStorage).
44 * @private {Storage}
45 */
46 this.storage_ = storage;

52/**
53 * The key used to check if the storage instance is available.
54 * @private {string}
55 * @const
56 */
57goog.storage.mechanism.HTML5WebStorage.STORAGE_AVAILABLE_KEY_ = '__sak';

60/**
61 * Determines whether or not the mechanism is available.
62 * It works only if the provided web storage object exists and is enabled.
63 *
64 * @return {boolean} True if the mechanism is available.
65 */
66goog.storage.mechanism.HTML5WebStorage.prototype.isAvailable = function() {
67 if (!this.storage_) {
68 return false;
69 }
70 /** @preserveTry */
71 try {
72 // setItem will throw an exception if we cannot access WebStorage (e.g.,
73 // Safari in private mode).
74 this.storage_.setItem(
75 goog.storage.mechanism.HTML5WebStorage.STORAGE_AVAILABLE_KEY_, '1');
76 this.storage_.removeItem(
77 goog.storage.mechanism.HTML5WebStorage.STORAGE_AVAILABLE_KEY_);
78 return true;
79 } catch (e) {
80 return false;
81 }
82};

85/** @override */
86goog.storage.mechanism.HTML5WebStorage.prototype.set = function(key, value) {
87 /** @preserveTry */
88 try {
89 // May throw an exception if storage quota is exceeded.
90 this.storage_.setItem(key, value);
91 } catch (e) {
92 // In Safari Private mode, conforming to the W3C spec, invoking
93 // Storage.prototype.setItem will allways throw a QUOTA_EXCEEDED_ERR
94 // exception. Since it's impossible to verify if we're in private browsing
95 // mode, we throw a different exception if the storage is empty.
96 if (this.storage_.length == 0) {
97 throw goog.storage.mechanism.ErrorCode.STORAGE_DISABLED;
98 } else {
99 throw goog.storage.mechanism.ErrorCode.QUOTA_EXCEEDED;
100 }
101 }
102};

105/** @override */
106goog.storage.mechanism.HTML5WebStorage.prototype.get = function(key) {
107 // According to W3C specs, values can be of any type. Since we only save
108 // strings, any other type is a storage error. If we returned nulls for
109 // such keys, i.e., treated them as non-existent, this would lead to a
110 // paradox where a key exists, but it does not when it is retrieved.
111 // http://www.w3.org/TR/2009/WD-webstorage-20091029/#the-storage-interface
112 var value = this.storage_.getItem(key);
113 if (!goog.isString(value) && !goog.isNull(value)) {
114 throw goog.storage.mechanism.ErrorCode.INVALID_VALUE;
115 }
116 return value;
117};

132/** @override */
133goog.storage.mechanism.HTML5WebStorage.prototype.__iterator__ = function(
134 opt_keys) {
135 var i = 0;
136 var storage = this.storage_;
137 var newIter = new goog.iter.Iterator();
138 newIter.next = function() {
139 if (i >= storage.length) {
140 throw goog.iter.StopIteration;
141 }
142 var key = goog.asserts.assertString(storage.key(i++));
143 if (opt_keys) {
144 return key;
145 }
146 var value = storage.getItem(key);
147 // The value must exist and be a string, otherwise it is a storage error.
148 if (!goog.isString(value)) {
149 throw goog.storage.mechanism.ErrorCode.INVALID_VALUE;
150 }
151 return value;
152 };
153 return newIter;
154};
