23goog.provide('goog.storage.mechanism.iterableMechanismTester');

25goog.require('goog.iter.Iterator');
26goog.require('goog.storage.mechanism.IterableMechanism');
27goog.require('goog.testing.asserts');
28goog.setTestOnly('iterableMechanismTester');

31var mechanism = null;
32
33
34function testCount() {
35 if (!mechanism) {
36 return;
37 }
38 assertEquals(0, mechanism.getCount());
39 mechanism.set('first', 'one');
40 assertEquals(1, mechanism.getCount());
41 mechanism.set('second', 'two');
42 assertEquals(2, mechanism.getCount());
43 mechanism.set('first', 'three');
44 assertEquals(2, mechanism.getCount());
45}

48function testIteratorBasics() {
49 if (!mechanism) {
50 return;
51 }
52 mechanism.set('first', 'one');
53 assertEquals('first', mechanism.__iterator__(true).next());
54 assertEquals('one', mechanism.__iterator__(false).next());
55 var iterator = mechanism.__iterator__();
56 assertEquals('one', iterator.next());
57 assertEquals(goog.iter.StopIteration,
58 assertThrows(iterator.next));
59}

61
62function testIteratorWithTwoValues() {
63 if (!mechanism) {
64 return;
65 }
66 mechanism.set('first', 'one');
67 mechanism.set('second', 'two');
68 assertSameElements(['one', 'two'], goog.iter.toArray(mechanism));
69 assertSameElements(['first', 'second'],
70 goog.iter.toArray(mechanism.__iterator__(true)));
71}

74function testClear() {
75 if (!mechanism) {
76 return;
77 }
78 mechanism.set('first', 'one');
79 mechanism.set('second', 'two');
80 mechanism.clear();
81 assertNull(mechanism.get('first'));
82 assertNull(mechanism.get('second'));
83 assertEquals(0, mechanism.getCount());
84 assertEquals(goog.iter.StopIteration,
85 assertThrows(mechanism.__iterator__(true).next));
86 assertEquals(goog.iter.StopIteration,
87 assertThrows(mechanism.__iterator__(false).next));
88}

91function testClearClear() {
92 if (!mechanism) {
93 return;

95 mechanism.clear();
96 mechanism.clear();

101function testIteratorWithWeirdKeys() {
102 if (!mechanism) {
103 return;
104 }
105 mechanism.set(' ', 'space');
106 mechanism.set('=+!@#$%^&*()-_\\|;:\'",./<>?[]{}~`', 'control');
107 mechanism.set(
108 '\u4e00\u4e8c\u4e09\u56db\u4e94\u516d\u4e03\u516b\u4e5d\u5341', 'ten');
109 assertEquals(3, mechanism.getCount());
110 assertSameElements([
111 ' ',
112 '=+!@#$%^&*()-_\\|;:\'",./<>?[]{}~`',
113 '\u4e00\u4e8c\u4e09\u56db\u4e94\u516d\u4e03\u516b\u4e5d\u5341'
114 ], goog.iter.toArray(mechanism.__iterator__(true)));
115 mechanism.clear();
116 assertEquals(0, mechanism.getCount());
117}
