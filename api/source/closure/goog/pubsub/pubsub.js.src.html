28/**
29 * Topic-based publish/subscribe channel. Maintains a map of topics to
30 * subscriptions. When a message is published to a topic, all functions
31 * subscribed to that topic are invoked in the order they were added.
32 * Uncaught errors abort publishing.
33 *
34 * Topics may be identified by any nonempty string, <strong>except</strong>
35 * strings corresponding to native Object properties, e.g. "constructor",
36 * "toString", "hasOwnProperty", etc.
37 *
38 * @constructor
39 * @extends {goog.Disposable}
40 */
41goog.pubsub.PubSub = function() {
42 goog.pubsub.PubSub.base(this, 'constructor');
43
44 /**
45 * The next available subscription key. Internally, this is an index into the
46 * sparse array of subscriptions.
47 *
48 * @private {number}
49 */
50 this.key_ = 1;
51
52 /**
53 * Array of subscription keys pending removal once publishing is done.
54 *
55 * @private {!Array<number>}
56 * @const
57 */
58 this.pendingKeys_ = [];
59
60 /**
61 * Lock to prevent the removal of subscriptions during publishing. Incremented
62 * at the beginning of {@link #publish}, and decremented at the end.
63 *
64 * @private {number}
65 */
66 this.publishDepth_ = 0;
67
68 /**
69 * Sparse array of subscriptions. Each subscription is represented by a tuple
70 * comprising a topic identifier, a function, and an optional context object.
71 * Each tuple occupies three consecutive positions in the array, with the
72 * topic identifier at index n, the function at index (n + 1), the context
73 * object at index (n + 2), the next topic at index (n + 3), etc. (This
74 * representation minimizes the number of object allocations and has been
75 * shown to be faster than an array of objects with three key-value pairs or
76 * three parallel arrays, especially on IE.) Once a subscription is removed
77 * via {@link #unsubscribe} or {@link #unsubscribeByKey}, the three
78 * corresponding array elements are deleted, and never reused. This means the
79 * total number of subscriptions during the lifetime of the pubsub channel is
80 * limited by the maximum length of a JavaScript array to (2^32 - 1) / 3 =
81 * 1,431,655,765 subscriptions, which should suffice for most applications.
82 *
83 * @private {!Array<?>}
84 * @const
85 */
86 this.subscriptions_ = [];
87
88 /**
89 * Map of topics to arrays of subscription keys.
90 *
91 * @private {!Object<!Array<number>>}
92 */
93 this.topics_ = {};
94};
95goog.inherits(goog.pubsub.PubSub, goog.Disposable);

98/**
99 * Subscribes a function to a topic. The function is invoked as a method on
100 * the given {@code opt_context} object, or in the global scope if no context
101 * is specified. Subscribing the same function to the same topic multiple
102 * times will result in multiple function invocations while publishing.
103 * Returns a subscription key that can be used to unsubscribe the function from
104 * the topic via {@link #unsubscribeByKey}.
105 *
106 * @param {string} topic Topic to subscribe to.
107 * @param {Function} fn Function to be invoked when a message is published to
108 * the given topic.
109 * @param {Object=} opt_context Object in whose context the function is to be
110 * called (the global scope if none).
111 * @return {number} Subscription key.
112 */
113goog.pubsub.PubSub.prototype.subscribe = function(topic, fn, opt_context) {
114 var keys = this.topics_[topic];
115 if (!keys) {
116 // First subscription to this topic; initialize subscription key array.
117 keys = this.topics_[topic] = [];
118 }
119
120 // Push the tuple representing the subscription onto the subscription array.
121 var key = this.key_;
122 this.subscriptions_[key] = topic;
123 this.subscriptions_[key + 1] = fn;
124 this.subscriptions_[key + 2] = opt_context;
125 this.key_ = key + 3;
126
127 // Push the subscription key onto the list of subscriptions for the topic.
128 keys.push(key);
129
130 // Return the subscription key.
131 return key;
132};

135/**
136 * Subscribes a single-use function to a topic. The function is invoked as a
137 * method on the given {@code opt_context} object, or in the global scope if
138 * no context is specified, and is then unsubscribed. Returns a subscription
139 * key that can be used to unsubscribe the function from the topic via
140 * {@link #unsubscribeByKey}.
141 *
142 * @param {string} topic Topic to subscribe to.
143 * @param {Function} fn Function to be invoked once and then unsubscribed when
144 * a message is published to the given topic.
145 * @param {Object=} opt_context Object in whose context the function is to be
146 * called (the global scope if none).
147 * @return {number} Subscription key.
148 */
149goog.pubsub.PubSub.prototype.subscribeOnce = function(topic, fn, opt_context) {
150 // Behold the power of lexical closures!
151 var key = this.subscribe(topic, function(var_args) {
152 fn.apply(opt_context, arguments);
153 this.unsubscribeByKey(key);
154 }, this);
155 return key;
156};

159/**
160 * Unsubscribes a function from a topic. Only deletes the first match found.
161 * Returns a Boolean indicating whether a subscription was removed.
162 *
163 * @param {string} topic Topic to unsubscribe from.
164 * @param {Function} fn Function to unsubscribe.
165 * @param {Object=} opt_context Object in whose context the function was to be
166 * called (the global scope if none).
167 * @return {boolean} Whether a matching subscription was removed.
168 */
169goog.pubsub.PubSub.prototype.unsubscribe = function(topic, fn, opt_context) {
170 var keys = this.topics_[topic];
171 if (keys) {
172 // Find the subscription key for the given combination of topic, function,
173 // and context object.
174 var subscriptions = this.subscriptions_;
175 var key = goog.array.find(keys, function(k) {
176 return subscriptions[k + 1] == fn && subscriptions[k + 2] == opt_context;
177 });
178 // Zero is not a valid key.
179 if (key) {
180 return this.unsubscribeByKey(/** @type {number} */ (key));
181 }
182 }
183
184 return false;
185};

188/**
189 * Removes a subscription based on the key returned by {@link #subscribe}.
190 * No-op if no matching subscription is found. Returns a Boolean indicating
191 * whether a subscription was removed.
192 *
193 * @param {number} key Subscription key.
194 * @return {boolean} Whether a matching subscription was removed.
195 */
196goog.pubsub.PubSub.prototype.unsubscribeByKey = function(key) {
197 if (this.publishDepth_ != 0) {
198 // Defer removal until after publishing is complete.
199 this.pendingKeys_.push(key);
200 return false;
201 }
202
203 var topic = this.subscriptions_[key];
204 if (topic) {
205 // Subscription tuple found.
206 var keys = this.topics_[topic];
207 if (keys) {
208 goog.array.remove(keys, key);
209 }
210 delete this.subscriptions_[key];
211 delete this.subscriptions_[key + 1];
212 delete this.subscriptions_[key + 2];
213 }
214
215 return !!topic;
216};

219/**
220 * Publishes a message to a topic. Calls functions subscribed to the topic in
221 * the order in which they were added, passing all arguments along. If any of
222 * the functions throws an uncaught error, publishing is aborted.
223 *
224 * @param {string} topic Topic to publish to.
225 * @param {...*} var_args Arguments that are applied to each subscription
226 * function.
227 * @return {boolean} Whether any subscriptions were called.
228 */
229goog.pubsub.PubSub.prototype.publish = function(topic, var_args) {
230 var keys = this.topics_[topic];
231 if (keys) {
232 // We must lock subscriptions and remove them at the end, so we don't
233 // adversely affect the performance of the common case by cloning the key
234 // array.
235 this.publishDepth_++;
236
237 // Copy var_args to a new array so they can be passed to subscribers.
238 // Note that we can't use Array.slice or goog.array.toArray for this for
239 // performance reasons. Using those with the arguments object will cause
240 // deoptimization.
241 var args = new Array(arguments.length - 1);
242 for (var i = 1, len = arguments.length; i < len; i++) {
243 args[i - 1] = arguments[i];
244 }
245
246 // For each key in the list of subscription keys for the topic, apply the
247 // function to the arguments in the appropriate context. The length of the
248 // array mush be fixed during the iteration, since subscribers may add new
249 // subscribers during publishing.
250 for (var i = 0, len = keys.length; i < len; i++) {
251 var key = keys[i];
252 this.subscriptions_[key + 1].apply(this.subscriptions_[key + 2], args);
253 }
254
255 // Unlock subscriptions.
256 this.publishDepth_--;
257
258 if (this.pendingKeys_.length > 0 && this.publishDepth_ == 0) {
259 var pendingKey;
260 while ((pendingKey = this.pendingKeys_.pop())) {
261 this.unsubscribeByKey(pendingKey);
262 }
263 }
264
265 // At least one subscriber was called.
266 return i != 0;
267 }
268
269 // No subscribers were found.
270 return false;
271};

274/**
275 * Clears the subscription list for a topic, or all topics if unspecified.
276 * @param {string=} opt_topic Topic to clear (all topics if unspecified).
277 */
278goog.pubsub.PubSub.prototype.clear = function(opt_topic) {
279 if (opt_topic) {
280 var keys = this.topics_[opt_topic];
281 if (keys) {
282 goog.array.forEach(keys, this.unsubscribeByKey, this);
283 delete this.topics_[opt_topic];
284 }
285 } else {
286 this.subscriptions_.length = 0;
287 this.topics_ = {};
288 // We don't reset key_ on purpose, because we want subscription keys to be
289 // unique throughout the lifetime of the application. Reusing subscription
290 // keys could lead to subtle errors in client code.
291 }
292};

295/**
296 * Returns the number of subscriptions to the given topic (or all topics if
297 * unspecified).
298 * @param {string=} opt_topic The topic (all topics if unspecified).
299 * @return {number} Number of subscriptions to the topic.
300 */
301goog.pubsub.PubSub.prototype.getCount = function(opt_topic) {
302 if (opt_topic) {
303 var keys = this.topics_[opt_topic];
304 return keys ? keys.length : 0;
305 }
306
307 var count = 0;
308 for (var topic in this.topics_) {
309 count += this.getCount(topic);
310 }
311
312 return count;
313};
