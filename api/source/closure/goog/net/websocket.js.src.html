15/**
16 * @fileoverview Definition of the WebSocket class. A WebSocket provides a
17 * bi-directional, full-duplex communications channel, over a single TCP socket.
18 *
19 * See http://dev.w3.org/html5/websockets/
20 * for the full HTML5 WebSocket API.
21 *
22 * Typical usage will look like this:
23 *
24 * var ws = new goog.net.WebSocket();
25 *
26 * var handler = new goog.events.EventHandler();
27 * handler.listen(ws, goog.net.WebSocket.EventType.OPENED, onOpen);
28 * handler.listen(ws, goog.net.WebSocket.EventType.MESSAGE, onMessage);
29 *
30 * try {
31 * ws.open('ws://127.0.0.1:4200');
32 * } catch (e) {
33 * ...
34 * }
35 *
36 */

53/**
54 * Class encapsulating the logic for using a WebSocket.
55 *
56 * @param {boolean=} opt_autoReconnect True if the web socket should
57 * automatically reconnect or not. This is true by default.
58 * @param {function(number):number=} opt_getNextReconnect A function for
59 * obtaining the time until the next reconnect attempt. Given the reconnect
60 * attempt count (which is a positive integer), the function should return a
61 * positive integer representing the milliseconds to the next reconnect
62 * attempt. The default function used is an exponential back-off. Note that
63 * this function is never called if auto reconnect is disabled.
64 * @constructor
65 * @extends {goog.events.EventTarget}
66 */
67goog.net.WebSocket = function(opt_autoReconnect, opt_getNextReconnect) {
68 goog.net.WebSocket.base(this, 'constructor');
69
70 /**
71 * True if the web socket should automatically reconnect or not.
72 * @type {boolean}
73 * @private
74 */
75 this.autoReconnect_ = goog.isDef(opt_autoReconnect) ?
76 opt_autoReconnect : true;
77
78 /**
79 * A function for obtaining the time until the next reconnect attempt.
80 * Given the reconnect attempt count (which is a positive integer), the
81 * function should return a positive integer representing the milliseconds to
82 * the next reconnect attempt.
83 * @type {function(number):number}
84 * @private
85 */
86 this.getNextReconnect_ = opt_getNextReconnect ||
87 goog.net.WebSocket.EXPONENTIAL_BACKOFF_;
88
89 /**
90 * The time, in milliseconds, that must elapse before the next attempt to
91 * reconnect.
92 * @type {number}
93 * @private
94 */
95 this.nextReconnect_ = this.getNextReconnect_(this.reconnectAttempt_);
96};
97goog.inherits(goog.net.WebSocket, goog.events.EventTarget);

201/**
202 * The maximum amount of time between reconnect attempts for the exponential
203 * back-off in milliseconds.
204 * @type {number}
205 * @private
206 */
207goog.net.WebSocket.EXPONENTIAL_BACKOFF_CEILING_ = 60 * 1000;

209
210/**
211 * Computes the next reconnect time given the number of reconnect attempts since
212 * the last successful connection.
213 *
214 * @param {number} attempt The number of reconnect attempts since the last
215 * connection.
216 * @return {number} The time, in milliseconds, until the next reconnect attempt.
217 * @const
218 * @private
219 */
220goog.net.WebSocket.EXPONENTIAL_BACKOFF_ = function(attempt) {
221 var time = Math.pow(2, attempt) * 1000;
222 return Math.min(time, goog.net.WebSocket.EXPONENTIAL_BACKOFF_CEILING_);
223};

248/**
249 * Creates and opens the actual WebSocket. Only call this after attaching the
250 * appropriate listeners to this object. If listeners aren't registered, then
251 * the {@code goog.net.WebSocket.EventType.OPENED} event might be missed.
252 *
253 * @param {string} url The URL to which to connect.
254 * @param {string=} opt_protocol The subprotocol to use. The connection will
255 * only be established if the server reports that it has selected this
256 * subprotocol. The subprotocol name must all be a non-empty ASCII string
257 * with no control characters and no spaces in them (i.e. only characters
258 * in the range U+0021 to U+007E).
259 */
260goog.net.WebSocket.prototype.open = function(url, opt_protocol) {
261 // Sanity check. This works only in modern browsers.
262 goog.asserts.assert(goog.global['WebSocket'],
263 'This browser does not support WebSocket');
264
265 // Don't do anything if the web socket is already open.
266 goog.asserts.assert(!this.isOpen(), 'The WebSocket is already open');
267
268 // Clear any pending attempts to reconnect.
269 this.clearReconnectTimer_();
270
271 // Construct the web socket.
272 this.url_ = url;
273 this.protocol_ = opt_protocol;
274
275 // This check has to be made otherwise you get protocol mismatch exceptions
276 // for passing undefined, null, '', or [].
277 if (this.protocol_) {
278 goog.log.info(this.logger_, 'Opening the WebSocket on ' + this.url_ +
279 ' with protocol ' + this.protocol_);
280 this.webSocket_ = new WebSocket(this.url_, this.protocol_);
281 } else {
282 goog.log.info(this.logger_, 'Opening the WebSocket on ' + this.url_);
283 this.webSocket_ = new WebSocket(this.url_);
284 }
285
286 // Register the event handlers. Note that it is not possible for these
287 // callbacks to be missed because it is registered after the web socket is
288 // instantiated. Because of the synchronous nature of JavaScript, this code
289 // will execute before the browser creates the resource and makes any calls
290 // to these callbacks.
291 this.webSocket_.onopen = goog.bind(this.onOpen_, this);
292 this.webSocket_.onclose = goog.bind(this.onClose_, this);
293 this.webSocket_.onmessage = goog.bind(this.onMessage_, this);
294 this.webSocket_.onerror = goog.bind(this.onError_, this);
295};

298/**
299 * Closes the web socket connection.
300 */
301goog.net.WebSocket.prototype.close = function() {
302
303 // Clear any pending attempts to reconnect.
304 this.clearReconnectTimer_();
305
306 // Attempt to close only if the web socket was created.
307 if (this.webSocket_) {
308 goog.log.info(this.logger_, 'Closing the WebSocket.');
309
310 // Close is expected here since it was a direct call. Close is considered
311 // unexpected when opening the connection fails or there is some other form
312 // of connection loss after being connected.
313 this.closeExpected_ = true;
314 this.webSocket_.close();
315 this.webSocket_ = null;
316 }
317};

371/**
372 * Called when the web socket has closed.
373 *
374 * @param {!Event} event The close event.
375 * @private
376 */
377goog.net.WebSocket.prototype.onClose_ = function(event) {
378 goog.log.info(this.logger_, 'The WebSocket on ' + this.url_ + ' closed.');
379
380 // Firing this event allows handlers to query the URL.
381 this.dispatchEvent(goog.net.WebSocket.EventType.CLOSED);
382
383 // Always clear out the web socket on a close event.
384 this.webSocket_ = null;
385
386 // See if this is an expected call to onClose_.
387 if (this.closeExpected_) {
388 goog.log.info(this.logger_, 'The WebSocket closed normally.');
389 // Only clear out the URL if this is a normal close.
390 this.url_ = null;
391 this.protocol_ = undefined;
392 } else {
393 // Unexpected, so try to reconnect.
394 goog.log.error(this.logger_, 'The WebSocket disconnected unexpectedly: ' +
395 event.data);
396
397 // Only try to reconnect if it is enabled.
398 if (this.autoReconnect_) {
399 // Log the reconnect attempt.
400 var seconds = Math.floor(this.nextReconnect_ / 1000);
401 goog.log.info(this.logger_,
402 'Seconds until next reconnect attempt: ' + seconds);
403
404 // Actually schedule the timer.
405 this.reconnectTimer_ = goog.Timer.callOnce(
406 goog.bind(this.open, this, this.url_, this.protocol_),
407 this.nextReconnect_, this);
408
409 // Set the next reconnect interval.
410 this.reconnectAttempt_++;
411 this.nextReconnect_ = this.getNextReconnect_(this.reconnectAttempt_);
412 }
413 }
414 this.closeExpected_ = false;
415};
