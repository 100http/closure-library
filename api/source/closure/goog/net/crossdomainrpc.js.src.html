15/**
16 * @fileoverview Cross domain RPC library using the <a
17 * href="http://go/xd2_design" target="_top">XD2 approach</a>.
18 *
19 * <h5>Protocol</h5>
20 * Client sends a request across domain via a form submission. Server
21 * receives these parameters: "xdpe:request-id", "xdpe:dummy-uri" ("xdpe" for
22 * "cross domain parameter to echo back") and other user parameters prefixed
23 * with "xdp" (for "cross domain parameter"). Headers are passed as parameters
24 * prefixed with "xdh" (for "cross domain header"). Only strings are supported
25 * for parameters and headers. A GET method is mapped to a form GET. All
26 * other methods are mapped to a POST. Server is expected to produce a
27 * HTML response such as the following:
28 * <pre>
29 * &lt;body&gt;
30 * &lt;script type="text/javascript"
31 * src="path-to-crossdomainrpc.js"&gt;&lt;/script&gt;
32 * var currentDirectory = location.href.substring(
33 * 0, location.href.lastIndexOf('/')
34 * );
35 *
36 * // echo all parameters prefixed with "xdpe:"
37 * var echo = {};
38 * echo[goog.net.CrossDomainRpc.PARAM_ECHO_REQUEST_ID] =
39 * &lt;value of parameter "xdpe:request-id"&gt;;
40 * echo[goog.net.CrossDomainRpc.PARAM_ECHO_DUMMY_URI] =
41 * &lt;value of parameter "xdpe:dummy-uri"&gt;;
42 *
43 * goog.net.CrossDomainRpc.sendResponse(
44 * '({"result":"&lt;responseInJSON"})',
45 * true, // is JSON
46 * echo, // parameters to echo back
47 * status, // response status code
48 * headers // response headers
49 * );
50 * &lt;/script&gt;
51 * &lt;/body&gt;
52 * </pre>
53 *
54 * <h5>Server Side</h5>
55 * For an example of the server side, refer to the following files:
56 * <ul>
57 * <li>http://go/xdservletfilter.java</li>
58 * <li>http://go/xdservletrequest.java</li>
59 * <li>http://go/xdservletresponse.java</li>
60 * </ul>
61 *
62 * <h5>System Requirements</h5>
63 * Tested on IE6, IE7, Firefox 2.0 and Safari nightly r23841.
64 *
65 */

152/**
153 * Checks to see if we are executing inside a response iframe. This is the
154 * case when this page is used as a dummy resource to gain caller's domain.
155 * @return {*} True if we are executing inside a response iframe; false
156 * otherwise.
157 * @private
158 */
159goog.net.CrossDomainRpc.isInResponseIframe_ = function() {
160 return window.location && (window.location.hash.indexOf(
161 goog.net.CrossDomainRpc.RESPONSE_MARKER_) == 1 ||
162 window.location.search.indexOf(
163 goog.net.CrossDomainRpc.RESPONSE_MARKER_) == 1);
164};
165
166
167/**
168 * Stops execution of the rest of the page if this page is loaded inside a
169 * response iframe.
170 */
171if (goog.net.CrossDomainRpc.isInResponseIframe_()) {
172 if (goog.userAgent.IE) {
173 document.execCommand('Stop');
174 } else if (goog.userAgent.GECKO) {
175 window.stop();
176 } else {
177 throw Error('stopped');
178 }
179}

205/**
206 * Sends a request across domain.
207 * @param {string} uri Uri to make request to.
208 * @param {Function=} opt_continuation Continuation function to be called
209 * when request is completed. Takes one argument of an event object
210 * whose target has the following properties: "status" is the HTTP
211 * response status code, "responseText" is the response text,
212 * and "headers" is an object with all response headers. The event
213 * target's getResponseJson() method returns a JavaScript object evaluated
214 * from the JSON response or undefined if response is not JSON.
215 * @param {string=} opt_method Method of request. Default is POST.
216 * @param {Object=} opt_params Parameters. Each property is turned into a
217 * request parameter.
218 * @param {Object=} opt_headers Map of headers of the request.
219 */
220goog.net.CrossDomainRpc.send =
221 function(uri, opt_continuation, opt_method, opt_params, opt_headers) {
222 var xdrpc = new goog.net.CrossDomainRpc();
223 if (opt_continuation) {
224 goog.events.listen(xdrpc, goog.net.EventType.COMPLETE, opt_continuation);
225 }
226 goog.events.listen(xdrpc, goog.net.EventType.READY, xdrpc.reset);
227 xdrpc.sendRequest(uri, opt_method, opt_params, opt_headers);
228};

278/**
279 * Finds a dummy resource that can be used by response to gain domain of
280 * requester's page.
281 * @return {string} URI of the resource to use.
282 * @private
283 */
284goog.net.CrossDomainRpc.getDummyResourceUri_ = function() {
285 if (goog.net.CrossDomainRpc.dummyResourceUri_) {
286 return goog.net.CrossDomainRpc.dummyResourceUri_;
287 }
288
289 // find a style sheet if not on IE, which will attempt to save style sheet
290 if (goog.userAgent.GECKO) {
291 var links = document.getElementsByTagName(goog.dom.TagName.LINK);
292 for (var i = 0; i < links.length; i++) {
293 var link = links[i];
294 // find a link which is on the same domain as this page
295 // cannot use one with '?' or '#' in its URL as it will confuse
296 // goog.net.CrossDomainRpc.getFramePayload_()
297 if (link.rel == 'stylesheet' &&
298 goog.Uri.haveSameDomain(link.href, window.location.href) &&
299 link.href.indexOf('?') < 0) {
300 return goog.net.CrossDomainRpc.removeHash_(link.href);
301 }
302 }
303 }
304
305 var images = document.getElementsByTagName(goog.dom.TagName.IMG);
306 for (var i = 0; i < images.length; i++) {
307 var image = images[i];
308 // find a link which is on the same domain as this page
309 // cannot use one with '?' or '#' in its URL as it will confuse
310 // goog.net.CrossDomainRpc.getFramePayload_()
311 if (goog.Uri.haveSameDomain(image.src, window.location.href) &&
312 image.src.indexOf('?') < 0) {
313 return goog.net.CrossDomainRpc.removeHash_(image.src);
314 }
315 }
316
317 if (!goog.net.CrossDomainRpc.useFallBackDummyResource_) {
318 throw Error(
319 'No suitable dummy resource specified or detected for this page');
320 }
321
322 if (goog.userAgent.IE) {
323 // use this page as the dummy resource; remove hash from URL if any
324 return goog.net.CrossDomainRpc.removeHash_(window.location.href);
325 } else {
326 /**
327 * Try to use "http://<this-domain>/robots.txt" which may exist. Even if
328 * it does not, an error page is returned and is a good dummy resource to
329 * use on Firefox and Safari. An existing resource is faster because it
330 * is cached.
331 */
332 var locationHref = window.location.href;
333 var rootSlash = locationHref.indexOf('/', locationHref.indexOf('//') + 2);
334 var rootHref = locationHref.substring(0, rootSlash);
335 return rootHref + '/robots.txt';
336 }
337};

408/**
409 * Sends a request across domain.
410 * @param {string} uri Uri to make request to.
411 * @param {string=} opt_method Method of request. Default is POST.
412 * @param {Object=} opt_params Parameters. Each property is turned into a
413 * request parameter.
414 * @param {Object=} opt_headers Map of headers of the request.
415 */
416goog.net.CrossDomainRpc.prototype.sendRequest =
417 function(uri, opt_method, opt_params, opt_headers) {
418 // create request frame
419 var requestFrame = this.requestFrame_ = /** @type {!HTMLIFrameElement} */ (
420 document.createElement(goog.dom.TagName.IFRAME));
421 var requestId = goog.net.CrossDomainRpc.nextRequestId_++;
422 requestFrame.id = goog.net.CrossDomainRpc.REQUEST_MARKER_ + '-' + requestId;
423 if (!goog.net.CrossDomainRpc.debugMode_) {
424 requestFrame.style.position = 'absolute';
425 requestFrame.style.top = '-5000px';
426 requestFrame.style.left = '-5000px';
427 }
428 document.body.appendChild(requestFrame);
429
430 // build inputs
431 var inputs = [];
432
433 // add request id
434 inputs.push(goog.net.CrossDomainRpc.createInputHtml_(
435 goog.net.CrossDomainRpc.PARAM_ECHO_REQUEST_ID, requestId));
436
437 // add dummy resource uri
438 var dummyUri = goog.net.CrossDomainRpc.getDummyResourceUri_();
439 goog.log.fine(goog.net.CrossDomainRpc.logger_,
440 'dummyUri: ' + dummyUri);
441 inputs.push(goog.net.CrossDomainRpc.createInputHtml_(
442 goog.net.CrossDomainRpc.PARAM_ECHO_DUMMY_URI, dummyUri));
443
444 // add parameters
445 if (opt_params) {
446 for (var name in opt_params) {
447 var value = opt_params[name];
448 inputs.push(goog.net.CrossDomainRpc.createInputHtml_(
449 goog.net.CrossDomainRpc.PARAM + name, value));
450 }
451 }
452
453 // add headers
454 if (opt_headers) {
455 for (var name in opt_headers) {
456 var value = opt_headers[name];
457 inputs.push(goog.net.CrossDomainRpc.createInputHtml_(
458 goog.net.CrossDomainRpc.HEADER + name, value));
459 }
460 }
461
462 var requestFrameContent = '<body><form method="' +
463 (opt_method == 'GET' ? 'GET' : 'POST') + '" action="' +
464 uri + '">' + inputs.join('') + '</form></body>';
465 var requestFrameContentHtml = goog.html.legacyconversions.safeHtmlFromString(
466 requestFrameContent);
467 var requestFrameDoc = goog.dom.getFrameContentDocument(requestFrame);
468 requestFrameDoc.open();
469 goog.dom.safe.documentWrite(requestFrameDoc, requestFrameContentHtml);
470 requestFrameDoc.close();
471
472 requestFrameDoc.forms[0].submit();
473 requestFrameDoc = null;
474
475 this.loadListenerKey_ = goog.events.listen(
476 requestFrame, goog.events.EventType.LOAD, function() {
477 goog.log.fine(goog.net.CrossDomainRpc.logger_, 'response ready');
478 this.responseReady_ = true;
479 }, false, this);
480
481 this.receiveResponse_();
482};

514/**
515 * Detects response inside request frame
516 * @param {number} responseDetectorHandle Handle of detector.
517 * @private
518 */
519goog.net.CrossDomainRpc.prototype.detectResponse_ =
520 function(responseDetectorHandle) {
521 var requestFrameWindow = this.requestFrame_.contentWindow;
522 var grandChildrenLength = requestFrameWindow.frames.length;
523 var responseInfoFrame = null;
524 if (grandChildrenLength > 0 &&
525 goog.net.CrossDomainRpc.isResponseInfoFrame_(responseInfoFrame =
526 requestFrameWindow.frames[grandChildrenLength - 1])) {
527 goog.log.fine(goog.net.CrossDomainRpc.logger_,
528 'xd response ready');
529
530 var responseInfoPayload = goog.net.CrossDomainRpc.getFramePayload_(
531 responseInfoFrame).substring(1);
532 var params = new goog.Uri.QueryData(responseInfoPayload);
533
534 var chunks = [];
535 var numChunks = Number(params.get('n'));
536 goog.log.fine(goog.net.CrossDomainRpc.logger_,
537 'xd response number of chunks: ' + numChunks);
538 for (var i = 0; i < numChunks; i++) {
539 var responseFrame = requestFrameWindow.frames[i];
540 if (!responseFrame || !responseFrame.location ||
541 !responseFrame.location.href) {
542 // On Safari 3.0, it is sometimes the case that the
543 // iframe exists but doesn't have a same domain href yet.
544 goog.log.fine(goog.net.CrossDomainRpc.logger_,
545 'xd response iframe not ready');
546 return;
547 }
548 var responseChunkPayload =
549 goog.net.CrossDomainRpc.getFramePayload_(responseFrame);
550 // go past "chunk="
551 var chunkIndex = responseChunkPayload.indexOf(
552 goog.net.CrossDomainRpc.PARAM_CHUNK_) +
553 goog.net.CrossDomainRpc.PARAM_CHUNK_.length + 1;
554 var chunk = responseChunkPayload.substring(chunkIndex);
555 chunks.push(chunk);
556 }
557
558 window.clearInterval(responseDetectorHandle);
559
560 var responseData = chunks.join('');
561 // Payload is not encoded to begin with on IE. Decode in other cases only.
562 if (!goog.userAgent.IE) {
563 responseData = decodeURIComponent(responseData);
564 }
565
566 this.status = Number(params.get('status'));
567 this.responseText = responseData;
568 this.responseTextIsJson_ = params.get('isDataJson') == 'true';
569 this.responseHeaders = goog.json.unsafeParse(
570 /** @type {string} */ (params.get('headers')));
571
572 this.dispatchEvent(goog.net.EventType.READY);
573 this.dispatchEvent(goog.net.EventType.COMPLETE);
574 } else {
575 if (this.responseReady_) {
576 /* The response has come back. But the first response iframe has not
577 * been created yet. If this lasts long enough, it is an error.
578 */
579 this.timeWaitedAfterResponseReady_ +=
580 goog.net.CrossDomainRpc.RESPONSE_POLLING_PERIOD_;
581 if (this.timeWaitedAfterResponseReady_ >
582 goog.net.CrossDomainRpc.SEND_RESPONSE_TIME_OUT_) {
583 goog.log.fine(goog.net.CrossDomainRpc.logger_,
584 'xd response timed out');
585 window.clearInterval(responseDetectorHandle);
586
587 this.status = goog.net.HttpStatus.INTERNAL_SERVER_ERROR;
588 this.responseText = 'response timed out';
589
590 this.dispatchEvent(goog.net.EventType.READY);
591 this.dispatchEvent(goog.net.EventType.ERROR);
592 this.dispatchEvent(goog.net.EventType.COMPLETE);
593 }
594 }
595 }
596};

599/**
600 * Checks whether a frame is response info frame.
601 * @param {Object} frame Frame to check.
602 * @return {boolean} True if frame is a response info frame; false otherwise.
603 * @private
604 */
605goog.net.CrossDomainRpc.isResponseInfoFrame_ = function(frame) {
606 /** @preserveTry */
607 try {
608 return goog.net.CrossDomainRpc.getFramePayload_(frame).indexOf(
609 goog.net.CrossDomainRpc.RESPONSE_INFO_MARKER_) == 1;
610 } catch (e) {
611 // frame not ready for same-domain access yet
612 return false;
613 }
614};

617/**
618 * Returns the payload of a frame (value after # or ? on the URL). This value
619 * is URL encoded except IE, where the value is not encoded to begin with.
620 * @param {Object} frame Frame.
621 * @return {string} Payload of that frame.
622 * @private
623 */
624goog.net.CrossDomainRpc.getFramePayload_ = function(frame) {
625 var href = frame.location.href;
626 var question = href.indexOf('?');
627 var hash = href.indexOf('#');
628 // On IE, beucase the URL is not encoded, we can have a case where ?
629 // is the delimiter before payload and # in payload or # as the delimiter
630 // and ? in payload. So here we treat whoever is the first as the delimiter.
631 var delimiter = question < 0 ? hash :
632 hash < 0 ? question : Math.min(question, hash);
633 return href.substring(delimiter);
634};

722 * call to pick up by creating a series of iframes pointed to the dummy URI
723 * with a payload (value after either ? or #) carrying a chunk of response
724 * data and a response info iframe that tells the grandparent (requester) the
725 * readiness of response.
726 * @param {string} data Response data (string or JSON string).
727 * @param {boolean} isDataJson true if data is a JSON string; false if just a
728 * string.
729 * @param {Object} echo Parameters to echo back
730 * "xdpe:request-id": Server that produces the response needs to
731 * copy it here to support multiple current XD requests on the same page.
732 * "xdpe:dummy-uri": URI to a dummy resource that response
733 * iframes point to to gain the domain of the client. This can be an
734 * image (IE) or a CSS file (FF) found on the requester's page.
735 * Server should copy value from request parameter "xdpe:dummy-uri".
736 * @param {number} status HTTP response status code.
737 * @param {string} headers Response headers in JSON format.
738 */
739goog.net.CrossDomainRpc.sendResponse =
740 function(data, isDataJson, echo, status, headers) {
741 var dummyUri = echo[goog.net.CrossDomainRpc.PARAM_ECHO_DUMMY_URI];
742
743 // since the dummy-uri can be specified by the user, verify that it doesn't
744 // use any other protocols. (Specifically we don't want users to use a
745 // dummy-uri beginning with "javascript:").
746 if (!goog.string.caseInsensitiveStartsWith(dummyUri, 'http://') &&
747 !goog.string.caseInsensitiveStartsWith(dummyUri, 'https://')) {
748 dummyUri = 'http://' + dummyUri;
749 }
750
751 // usable chunk size is max less dummy URI less chunk prefix length
752 // TODO(user): Figure out why we need to do "- 1" below
753 var chunkSize = goog.net.CrossDomainRpc.MAX_CHUNK_SIZE_ - dummyUri.length -
754 1 - // payload delimiter ('#' or '?')
755 goog.net.CrossDomainRpc.CHUNK_PREFIX_.length - 1;
756
757 /*
758 * Here we used to do URI encoding of data before we divide it into chunks
759 * and decode on the receiving end. We don't do this any more on IE for the
760 * following reasons.
761 *
762 * 1) On IE, calling decodeURIComponent on a relatively large string is
763 * extremely slow (~22s for 160KB). So even a moderate amount of data
764 * makes this library pretty much useless. Fortunately, we can actually
765 * put unencoded data on IE's URL and get it back reliably. So we are
766 * completely skipping encoding and decoding on IE. When we call
767 * getFrameHash_ to get it back, the value is still intact(*) and unencoded.
768 * 2) On Firefox, we have to call decodeURIComponent because location.hash
769 * does decoding by itself. Fortunately, decodeURIComponent is not slow
770 * on Firefox.
771 * 3) Safari automatically encodes everything you put on URL and it does not
772 * automatically decode when you access it via location.hash or
773 * location.href. So we encode it here and decode it in detectResponse_().
774 *
775 * Note(*): IE actually does encode only space to %20 and decodes that
776 * automatically when you do location.href or location.hash.
777 */
778 if (!goog.userAgent.IE) {
779 data = encodeURIComponent(data);
780 }
781
782 var numChunksToSend = Math.ceil(data.length / chunkSize);
783 if (numChunksToSend == 0) {
784 goog.net.CrossDomainRpc.createResponseInfo_(
785 dummyUri, numChunksToSend, isDataJson, status, headers);
786 } else {
787 var numChunksSent = 0;
788 var checkToCreateResponseInfo_ = function() {
789 if (++numChunksSent == numChunksToSend) {
790 goog.net.CrossDomainRpc.createResponseInfo_(
791 dummyUri, numChunksToSend, isDataJson, status, headers);
792 }
793 };
794
795 for (var i = 0; i < numChunksToSend; i++) {
796 var chunkStart = i * chunkSize;
797 var chunkEnd = chunkStart + chunkSize;
798 var chunk = chunkEnd > data.length ?
799 data.substring(chunkStart) :
800 data.substring(chunkStart, chunkEnd);
801
802 var responseFrame = document.createElement(goog.dom.TagName.IFRAME);
803 responseFrame.src = dummyUri +
804 goog.net.CrossDomainRpc.getPayloadDelimiter_(dummyUri) +
805 goog.net.CrossDomainRpc.CHUNK_PREFIX_ + chunk;
806 document.body.appendChild(responseFrame);
807
808 // We used to call the function below when handling load event of
809 // responseFrame. But that event does not fire on IE when current
810 // page is used as the dummy resource (because its loading is stopped?).
811 // It also does not fire sometimes on Firefox. So now we call it
812 // directly.
813 checkToCreateResponseInfo_();
814 }
815 }
816};

822 * @param {number} numChunks Total number of chunks.

840/**

854/**
855 * Removes all parameters (after ? or #) from URI.
856 * @param {string} uri URI to remove parameters from.
857 * @return {string} URI with all parameters removed.
858 * @private
859 */
860goog.net.CrossDomainRpc.removeUriParams_ = function(uri) {
861 // remove everything after question mark
862 var question = uri.indexOf('?');
863 if (question > 0) {
864 uri = uri.substring(0, question);
865 }
866
867 // remove everything after hash mark
868 var hash = uri.indexOf('#');
869 if (hash > 0) {
870 uri = uri.substring(0, hash);
871 }

873 return uri;
