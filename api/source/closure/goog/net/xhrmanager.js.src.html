15/**
16 * @fileoverview Manages a pool of XhrIo's. This handles all the details of
17 * dealing with the XhrPool and provides a simple interface for sending requests
18 * and managing events.
19 *
20 * This class supports queueing & prioritization of requests (XhrIoPool
21 * handles this) and retrying of requests.
22 *
23 * The events fired by the XhrManager are an aggregation of the events of
24 * each of its XhrIo objects (with some filtering, i.e., ERROR only called
25 * when there are no more retries left). For this reason, all send requests have
26 * to have an id, so that the user of this object can know which event is for
27 * which request.
28 *
29 */
30

50 * A manager of an XhrIoPool.
51 * @param {number=} opt_maxRetries Max. number of retries (Default: 1).
52 * @param {goog.structs.Map=} opt_headers Map of default headers to add to every
53 * request.
54 * @param {number=} opt_minCount Min. number of objects (Default: 1).
55 * @param {number=} opt_maxCount Max. number of objects (Default: 10).

57 * attempt (Default: 0ms).
58 * @constructor
59 * @extends {goog.events.EventTarget}
60 */
61goog.net.XhrManager = function(
62 opt_maxRetries,
63 opt_headers,
64 opt_minCount,
65 opt_maxCount,
66 opt_timeoutInterval) {
67 goog.net.XhrManager.base(this, 'constructor');
68
69 /**
70 * Maximum number of retries for a given request
71 * @type {number}
72 * @private
73 */
74 this.maxRetries_ = goog.isDef(opt_maxRetries) ? opt_maxRetries : 1;
75
76 /**
77 * Timeout interval for an attempt of a given request.
78 * @type {number}
79 * @private
80 */
81 this.timeoutInterval_ =
82 goog.isDef(opt_timeoutInterval) ? Math.max(0, opt_timeoutInterval) : 0;
83
84 /**
85 * The pool of XhrIo's to use.
86 * @type {goog.net.XhrIoPool}
87 * @private
88 */
89 this.xhrPool_ = new goog.net.XhrIoPool(
90 opt_headers, opt_minCount, opt_maxCount);

92 /**
93 * Map of ID's to requests.
94 * @type {goog.structs.Map<string, !goog.net.XhrManager.Request>}
95 * @private
96 */
97 this.requests_ = new goog.structs.Map();
98
99 /**
100 * The event handler.
101 * @type {goog.events.EventHandler<!goog.net.XhrManager>}

164/**
165 * Registers the given request to be sent. Throws an error if a request
166 * already exists with the given ID.
167 * NOTE: It is not sent immediately. It is queued and will be sent when an
168 * XhrIo object becomes available, taking into account the request's
169 * priority.
170 * @param {string} id The id of the request.
171 * @param {string} url Uri to make the request too.
172 * @param {string=} opt_method Send method, default: GET.
173 * @param {ArrayBuffer|ArrayBufferView|Blob|Document|FormData|string=}
174 * opt_content Post data.
175 * @param {Object|goog.structs.Map=} opt_headers Map of headers to add to the
176 * request.
177 * @param {number=} opt_priority The priority of the request. A smaller value
178 * means a higher priority.
179 * @param {Function=} opt_callback Callback function for when request is
180 * complete. The only param is the event object from the COMPLETE event.
181 * @param {number=} opt_maxRetries The maximum number of times the request
182 * should be retried.
183 * @param {goog.net.XhrIo.ResponseType=} opt_responseType The response type of
184 * this request; defaults to goog.net.XhrIo.ResponseType.DEFAULT.
185 * @return {!goog.net.XhrManager.Request} The queued request object.
186 */
187goog.net.XhrManager.prototype.send = function(
188 id,
189 url,
190 opt_method,
191 opt_content,
192 opt_headers,
193 opt_priority,
194 opt_callback,
195 opt_maxRetries,
196 opt_responseType) {
197 var requests = this.requests_;
198 // Check if there is already a request with the given id.
199 if (requests.get(id)) {
200 throw Error(goog.net.XhrManager.ERROR_ID_IN_USE_);
201 }
202
203 // Make the Request object.
204 var request = new goog.net.XhrManager.Request(
205 url,
206 goog.bind(this.handleEvent_, this, id),
207 opt_method,
208 opt_content,
209 opt_headers,
210 opt_callback,
211 goog.isDef(opt_maxRetries) ? opt_maxRetries : this.maxRetries_,
212 opt_responseType);
213 this.requests_.set(id, request);
214
215 // Setup the callback for the pool.
216 var callback = goog.bind(this.handleAvailableXhr_, this, id);
217 this.xhrPool_.getObject(callback, opt_priority);
218
219 return request;
220};

223/**
224 * Aborts the request associated with id.
225 * @param {string} id The id of the request to abort.
226 * @param {boolean=} opt_force If true, remove the id now so it can be reused.
227 * No events are fired and the callback is not called when forced.
228 */
229goog.net.XhrManager.prototype.abort = function(id, opt_force) {
230 var request = this.requests_.get(id);
231 if (request) {
232 var xhrIo = request.xhrIo;
233 request.setAborted(true);
234 if (opt_force) {
235 if (xhrIo) {
236 // We remove listeners to make sure nothing gets called if a new request
237 // with the same id is made.
238 this.removeXhrListener_(xhrIo, request.getXhrEventCallback());
239 goog.events.listenOnce(
240 xhrIo,
241 goog.net.EventType.READY,
242 function() { this.xhrPool_.releaseObject(xhrIo); },
243 false,
244 this);
245 }
246 this.requests_.remove(id);
247 }
248 if (xhrIo) {
249 xhrIo.abort();
250 }
251 }
252};

255/**
256 * Handles when an XhrIo object becomes available. Sets up the events, fires
257 * the READY event, and starts the process to send the request.
258 * @param {string} id The id of the request the XhrIo is for.
259 * @param {goog.net.XhrIo} xhrIo The available XhrIo object.
260 * @private
261 */
262goog.net.XhrManager.prototype.handleAvailableXhr_ = function(id, xhrIo) {
263 var request = this.requests_.get(id);
264 // Make sure the request doesn't already have an XhrIo attached. This can
265 // happen if a forced abort occurs before an XhrIo is available, and a new
266 // request with the same id is made.
267 if (request && !request.xhrIo) {
268 this.addXhrListener_(xhrIo, request.getXhrEventCallback());
269
270 // Set properties for the XhrIo.
271 xhrIo.setTimeoutInterval(this.timeoutInterval_);
272 xhrIo.setResponseType(request.getResponseType());
273
274 // Add a reference to the XhrIo object to the request.
275 request.xhrIo = xhrIo;
276
277 // Notify the listeners.
278 this.dispatchEvent(new goog.net.XhrManager.Event(
279 goog.net.EventType.READY, this, id, xhrIo));
280
281 // Send the request.
282 this.retry_(id, xhrIo);
283
284 // If the request was aborted before it got an XhrIo object, abort it now.
285 if (request.getAborted()) {
286 xhrIo.abort();
287 }
288 } else {
289 // If the request has an XhrIo object already, or no request exists, just
290 // return the XhrIo back to the pool.
291 this.xhrPool_.releaseObject(xhrIo);
292 }
293};

296/**
297 * Handles all events fired by the XhrIo object for a given request.
298 * @param {string} id The id of the request.
299 * @param {goog.events.Event} e The event.
300 * @return {Object} The return value from the handler, if any.
301 * @private
302 */
303goog.net.XhrManager.prototype.handleEvent_ = function(id, e) {
304 var xhrIo = /** @type {goog.net.XhrIo} */(e.target);
305 switch (e.type) {
306 case goog.net.EventType.READY:
307 this.retry_(id, xhrIo);
308 break;
309
310 case goog.net.EventType.COMPLETE:
311 return this.handleComplete_(id, xhrIo, e);
312
313 case goog.net.EventType.SUCCESS:
314 this.handleSuccess_(id, xhrIo);
315 break;
316
317 // A timeout is handled like an error.
318 case goog.net.EventType.TIMEOUT:
319 case goog.net.EventType.ERROR:
320 this.handleError_(id, xhrIo);
321 break;
322
323 case goog.net.EventType.ABORT:
324 this.handleAbort_(id, xhrIo);
325 break;
326 }
327 return null;
328};

331/**
332 * Attempts to retry the given request. If the request has already attempted
333 * the maximum number of retries, then it removes the request and releases
334 * the XhrIo object back into the pool.
335 * @param {string} id The id of the request.
336 * @param {goog.net.XhrIo} xhrIo The XhrIo object.
337 * @private
338 */
339goog.net.XhrManager.prototype.retry_ = function(id, xhrIo) {
340 var request = this.requests_.get(id);
341
342 // If the request has not completed and it is below its max. retries.
343 if (request && !request.getCompleted() && !request.hasReachedMaxRetries()) {
344 request.increaseAttemptCount();
345 xhrIo.send(request.getUrl(), request.getMethod(), request.getContent(),
346 request.getHeaders());
347 } else {
348 if (request) {
349 // Remove the events on the XhrIo objects.
350 this.removeXhrListener_(xhrIo, request.getXhrEventCallback());
351
352 // Remove the request.
353 this.requests_.remove(id);
354 }
355 // Release the XhrIo object back into the pool.
356 this.xhrPool_.releaseObject(xhrIo);
357 }
358};

361/**
362 * Handles the complete of a request. Dispatches the COMPLETE event and sets the
363 * the request as completed if the request has succeeded, or is done retrying.
364 * @param {string} id The id of the request.
365 * @param {goog.net.XhrIo} xhrIo The XhrIo object.
366 * @param {goog.events.Event} e The original event.
367 * @return {Object} The return value from the callback, if any.
368 * @private
369 */
370goog.net.XhrManager.prototype.handleComplete_ = function(id, xhrIo, e) {
371 // Only if the request is done processing should a COMPLETE event be fired.
372 var request = this.requests_.get(id);
373 if (xhrIo.getLastErrorCode() == goog.net.ErrorCode.ABORT ||
374 xhrIo.isSuccess() || request.hasReachedMaxRetries()) {
375 this.dispatchEvent(new goog.net.XhrManager.Event(
376 goog.net.EventType.COMPLETE, this, id, xhrIo));
377
378 // If the request exists, we mark it as completed and call the callback
379 if (request) {
380 request.setCompleted(true);
381 // Call the complete callback as if it was set as a COMPLETE event on the
382 // XhrIo directly.
383 if (request.getCompleteCallback()) {
384 return request.getCompleteCallback().call(xhrIo, e);
385 }
386 }
387 }
388 return null;
389};

410 * @param {string} id The id of the request.
411 * @param {goog.net.XhrIo} xhrIo The XhrIo object.

414goog.net.XhrManager.prototype.handleSuccess_ = function(id, xhrIo) {
415 // Fire event.
416 // NOTE: We don't release the XhrIo object from the pool here.
417 // It is released in the retry method, when we know it is back in the

424/**
425 * Handles the error of a request. If the request has not reach its maximum
426 * number of retries, then it lets the request retry naturally (will let the
427 * request hit the READY state). Else, it dispatches the ERROR event.
428 * @param {string} id The id of the request.
429 * @param {goog.net.XhrIo} xhrIo The XhrIo object.
430 * @private
431 */
432goog.net.XhrManager.prototype.handleError_ = function(id, xhrIo) {
433 var request = this.requests_.get(id);
434
435 // If the maximum number of retries has been reached.
436 if (request.hasReachedMaxRetries()) {
437 // Fire event.
438 // NOTE: We don't release the XhrIo object from the pool here.
439 // It is released in the retry method, when we know it is back in the
440 // ready state.
441 this.dispatchEvent(new goog.net.XhrManager.Event(
442 goog.net.EventType.ERROR, this, id, xhrIo));
443 }
444};

526/**
527 * An encapsulation of everything needed to make a Xhr request.
528 * NOTE: This is used internal to the XhrManager.
529 *
530 * @param {string} url Uri to make the request too.
531 * @param {Function} xhrEventCallback Callback attached to the events of the
532 * XhrIo object of the request.
533 * @param {string=} opt_method Send method, default: GET.
534 * @param {ArrayBuffer|ArrayBufferView|Blob|Document|FormData|string=}
535 * opt_content Post data.
536 * @param {Object|goog.structs.Map=} opt_headers Map of headers to add to the
537 * request.
538 * @param {Function=} opt_callback Callback function for when request is
539 * complete. NOTE: Only 1 callback supported across all events.
540 * @param {number=} opt_maxRetries The maximum number of times the request
541 * should be retried (Default: 1).
542 * @param {goog.net.XhrIo.ResponseType=} opt_responseType The response type of
543 * this request; defaults to goog.net.XhrIo.ResponseType.DEFAULT.
544 *
545 * @constructor
546 * @final
547 */
548goog.net.XhrManager.Request = function(url, xhrEventCallback, opt_method,
549 opt_content, opt_headers, opt_callback, opt_maxRetries, opt_responseType) {
550 /**
551 * Uri to make the request too.
552 * @type {string}
553 * @private
554 */
555 this.url_ = url;
556
557 /**
558 * Send method.
559 * @type {string}
560 * @private
561 */
562 this.method_ = opt_method || 'GET';
563
564 /**
565 * Post data.
566 * @type {ArrayBuffer|ArrayBufferView|Blob|Document|FormData|string|undefined}
567 * @private
568 */
569 this.content_ = opt_content;
570
571 /**
572 * Map of headers
573 * @type {Object|goog.structs.Map|null}
574 * @private
575 */
576 this.headers_ = opt_headers || null;
577
578 /**
579 * The maximum number of times the request should be retried.
580 * @type {number}
581 * @private
582 */
583 this.maxRetries_ = goog.isDef(opt_maxRetries) ? opt_maxRetries : 1;
584
585 /**
586 * The number of attempts so far.
587 * @type {number}
588 * @private
589 */
590 this.attemptCount_ = 0;
591
592 /**
593 * Whether the request has been completed.
594 * @type {boolean}
595 * @private
596 */
597 this.completed_ = false;
598
599 /**
600 * Whether the request has been aborted.
601 * @type {boolean}
602 * @private
603 */
604 this.aborted_ = false;
605
606 /**
607 * Callback attached to the events of the XhrIo object.
608 * @type {Function}
609 * @private
610 */
611 this.xhrEventCallback_ = xhrEventCallback;
612
613 /**
614 * Callback function called when request is complete.

616 * @private
617 */
618 this.completeCallback_ = opt_callback;

620 /**
621 * A response type to set on this.xhrIo when it's populated.
622 * @type {!goog.net.XhrIo.ResponseType}
623 * @private
624 */
625 this.responseType_ = opt_responseType || goog.net.XhrIo.ResponseType.DEFAULT;
626
627 /**
628 * The XhrIo instance handling this request. Set in handleAvailableXhr.
629 * @type {goog.net.XhrIo}
630 */
631 this.xhrIo = null;
632
633};
