15/**
16 * @fileoverview Provides the namesspace for client-side communication
17 * between pages originating from different domains (it works also
18 * with pages from the same domain, but doing that is kinda
19 * pointless).
20 *
21 * The only publicly visible class is goog.net.xpc.CrossPageChannel.
22 *
23 * Note: The preferred name for the main class would have been
24 * CrossDomainChannel. But as there already is a class named like
25 * that (which serves a different purpose) in the maps codebase,
26 * CrossPageChannel was chosen to avoid confusion.
27 *
28 * CrossPageChannel abstracts the underlying transport mechanism to
29 * provide a common interface in all browsers.
30 *

119 * The transport type to use. Possible values are entries from

122 */
123 TRANSPORT: 'tp',

125 * Local relay URI identifier (IframeRelayTransport-specific).
126 * The URI (can't contain a fragment identifier) used by the peer to

132 * The URI (can't contain a fragment identifier) used to relay data
133 * to the peer.
134 */
135 PEER_RELAY_URI: 'pru',
136 /**
137 * Local poll URI identifier (IframePollingTransport-specific).
138 * The URI (can't contain a fragment identifier)which is polled
139 * to receive data from the peer.
140 */
141 LOCAL_POLL_URI: 'lpu',
142 /**
143 * Local poll URI identifier (IframePollingTransport-specific).
144 * The URI (can't contain a fragment identifier) used to send data
145 * to the peer.
146 */
147 PEER_POLL_URI: 'ppu',
148 /**
149 * The hostname of the peer window, including protocol, domain, and port
150 * (if specified). Used for security sensitive applications that make
151 * use of NativeMessagingTransport (i.e. most applications).
152 */
153 PEER_HOSTNAME: 'ph',
154 /**
155 * Usually both frames using a connection initially send a SETUP message to
156 * each other, and each responds with a SETUP_ACK. A frame marks itself
157 * connected when it receives that SETUP_ACK. If this parameter is true
158 * however, the channel it is passed to will not send a SETUP, but rather will
159 * wait for one from its peer and mark itself connected when that arrives.
160 * Peer iframes created using such a channel will send SETUP however, and will
161 * wait for SETUP_ACK before marking themselves connected. The goal is to
162 * cope with a situation where the availability of the URL for the peer frame
163 * cannot be relied on, eg when the application is offline. Without this
164 * setting, the primary frame will attempt to send its SETUP message every
165 * 100ms, forever. This floods the javascript console with uncatchable
166 * security warnings, and fruitlessly burns CPU. There is one scenario this
167 * mode will not support, and that is reconnection by the outer frame, ie the
168 * creation of a new channel object to connect to a peer iframe which was
169 * already communicating with a previous channel object of the same name. If
170 * that behavior is needed, this mode should not be used. Reconnection by
171 * inner frames is supported in this mode however.
172 */
173 ONE_SIDED_HANDSHAKE: 'osh',
174 /**
175 * The frame role (inner or outer). Used to explicitly indicate the role for
176 * each peer whenever the role cannot be reliably determined (e.g. the two
177 * peer windows are not parent/child frames). If unspecified, the role will
178 * be dynamically determined, assuming a parent/child frame setup.
179 */
180 ROLE: 'role',
181 /**
182 * Which version of the native transport startup protocol should be used, the
183 * default being '2'. Version 1 had various timing vulnerabilities, which
184 * had to be compensated for by introducing delays, and is deprecated. V1
185 * and V2 are broadly compatible, although the more robust timing and lack
186 * of delays is not gained unless both sides are using V2. The only
187 * unsupported case of cross-protocol interoperation is where a connection
188 * starts out with V2 at both ends, and one of the ends reconnects as a V1.
189 * All other initial startup and reconnection scenarios are supported.
190 */
191 NATIVE_TRANSPORT_PROTOCOL_VERSION: 'nativeProtocolVersion',
192 /**
193 * Whether the direct transport runs in synchronous mode. The default is to
194 * emulate the other transports and run asyncronously but there are some
195 * circumstances where syncronous calls are required. If this property is
196 * set to true, the transport will send the messages synchronously.
197 */
198 DIRECT_TRANSPORT_SYNC_MODE: 'directSyncMode'
199};

270/**
271 * Returns a random string.
272 * @param {number} length How many characters the string shall contain.
273 * @param {string=} opt_characters The characters used.
274 * @return {string} The random string.
275 */
276goog.net.xpc.getRandomString = function(length, opt_characters) {
277 var chars = opt_characters || goog.net.xpc.randomStringCharacters_;
278 var charsLength = chars.length;
279 var s = '';
280 while (length-- > 0) {
281 s += chars.charAt(Math.floor(Math.random() * charsLength));
282 }
283 return s;
284};
