46/**
47 * A direct window to window method transport.
48 *
49 * If the windows are in the same security context, this transport calls
50 * directly into the other window without using any additional mechanism. This
51 * is mainly used in scenarios where you want to optionally use a cross domain
52 * transport in cross security context situations, or optionally use a direct
53 * transport in same security context situations.
54 *
55 * Note: Global properties are exported by using this transport. One to
56 * communicate with the other window by, currently crosswindowmessaging.channel,
57 * and by using goog.getUid on window, currently closure_uid_[0-9]+.
58 *
59 * @param {!goog.net.xpc.CrossPageChannel} channel The channel this
60 * transport belongs to.
61 * @param {goog.dom.DomHelper=} opt_domHelper The dom helper to use for
62 * finding the correct window/document. If omitted, uses the current
63 * document.
64 * @constructor
65 * @extends {Transport}
66 */
67goog.net.xpc.DirectTransport = function(channel, opt_domHelper) {
68 goog.net.xpc.DirectTransport.base(this, 'constructor', opt_domHelper);
69
70 /**
71 * The channel this transport belongs to.
72 * @private {!goog.net.xpc.CrossPageChannel}
73 */
74 this.channel_ = channel;
75
76 /** @private {!EventHandler<!goog.net.xpc.DirectTransport>} */
77 this.eventHandler_ = new EventHandler(this);
78 this.registerDisposable(this.eventHandler_);
79
80 /**
81 * Timer for connection reattempts.
82 * @private {!Timer}
83 */
84 this.maybeAttemptToConnectTimer_ = new Timer(
85 DirectTransport.CONNECTION_ATTEMPT_INTERVAL_MS_,
86 this.getWindow());
87 this.registerDisposable(this.maybeAttemptToConnectTimer_);
88
89 /**
90 * Fires once we've received our SETUP_ACK message.
91 * @private {!Deferred}
92 */
93 this.setupAckReceived_ = new Deferred();
94
95 /**
96 * Fires once we've sent our SETUP_ACK message.
97 * @private {!Deferred}
98 */
99 this.setupAckSent_ = new Deferred();
100
101 /**
102 * Fires once we're marked connected.
103 * @private {!Deferred}
104 */
105 this.connected_ = new Deferred();
106
107 /**
108 * The unique ID of this side of the connection. Used to determine when a peer
109 * is reloaded.
110 * @private {string}
111 */
112 this.endpointId_ = goog.net.xpc.getRandomString(10);
113
114 /**
115 * The unique ID of the peer. If we get a message from a peer with an ID we
116 * don't expect, we reset the connection.
117 * @private {?string}
118 */
119 this.peerEndpointId_ = null;
120
121 /**
122 * The map of sending messages.
123 * @private {Object}
124 */
125 this.asyncSendsMap_ = {};
126
127 /**
128 * The original channel name.
129 * @private {string}
130 */
131 this.originalChannelName_ = this.channel_.name;
132
133 // We reconfigure the channel name to include the role so that we can
134 // communicate in the same window between the different roles on the
135 // same channel.
136 this.channel_.updateChannelNameAndCatalog(
137 DirectTransport.getRoledChannelName_(this.channel_.name,
138 this.channel_.getRole()));
139
140 /**
141 * Flag indicating if this instance of the transport has been initialized.
142 * @private {boolean}
143 */
144 this.initialized_ = false;
145
146 // We don't want to mark ourselves connected until we have sent whatever
147 // message will cause our counterpart in the other frame to also declare
148 // itself connected, if there is such a message. Otherwise we risk a user
149 // message being sent in advance of that message, and it being discarded.
150
151 // Two sided handshake:
152 // SETUP_ACK has to have been received, and sent.
153 this.connected_.awaitDeferred(this.setupAckReceived_);
154 this.connected_.awaitDeferred(this.setupAckSent_);
155
156 this.connected_.addCallback(this.notifyConnected_, this);
157 this.connected_.callback(true);
158
159 this.eventHandler_.
160 listen(this.maybeAttemptToConnectTimer_, Timer.TICK,
161 this.maybeAttemptToConnect_);
162
163 goog.log.info(
164 goog.net.xpc.logger,
165 'DirectTransport created. role=' + this.channel_.getRole());
166};

230/**
231 * Initializes this transport. Registers a method for 'message'-events in the
232 * global scope.
233 * @param {!Window} listenWindow The window to listen to events on.
234 * @private
235 */
236DirectTransport.initialize_ = function(listenWindow) {
237 var uid = goog.getUid(listenWindow);
238 var value = DirectTransport.activeCount_[uid] || 0;
239 if (value == 0) {
240 // Set up a handler on the window to proxy messages to class.
241 var globalProxy = goog.getObjectByName(
242 DirectTransport.GLOBAL_TRANPORT_PATH_,
243 listenWindow);
244 if (globalProxy == null) {
245 goog.exportSymbol(
246 DirectTransport.GLOBAL_TRANPORT_PATH_,
247 DirectTransport.messageReceivedHandler_,
248 listenWindow);
249 }
250 }
251 DirectTransport.activeCount_[uid]++;
252};

266/**
267 * @param {!Object} literal The literal unrenamed message.
268 * @return {boolean} Whether the message was successfully delivered to a
269 * channel.
270 * @private
271 */
272DirectTransport.messageReceivedHandler_ = function(literal) {
273 var msg = DirectTransport.Message_.fromLiteral(literal);
274
275 var channelName = msg.channelName;
276 var service = msg.service;
277 var payload = msg.payload;
278
279 goog.log.fine(goog.net.xpc.logger,
280 'messageReceived: channel=' + channelName +
281 ', service=' + service + ', payload=' + payload);
282
283 // Attempt to deliver message to the channel. Keep in mind that it may not
284 // exist for several reasons, including but not limited to:
285 // - a malformed message
286 // - the channel simply has not been created
287 // - channel was created in a different namespace
288 // - message was sent to the wrong window
289 // - channel has become stale (e.g. caching iframes and back clicks)
290 var channel = goog.net.xpc.channels[channelName];
291 if (channel) {
292 channel.xpcDeliver(service, payload);
293 return true;
294 }
295
296 var transportMessageType = DirectTransport.parseTransportPayload_(payload)[0];
297
298 // Check if there are any stale channel names that can be updated.
299 for (var staleChannelName in goog.net.xpc.channels) {
300 var staleChannel = goog.net.xpc.channels[staleChannelName];
301 if (staleChannel.getRole() == CrossPageChannelRole.INNER &&
302 !staleChannel.isConnected() &&
303 service == goog.net.xpc.TRANSPORT_SERVICE_ &&
304 transportMessageType == goog.net.xpc.SETUP) {
305 // Inner peer received SETUP message but channel names did not match.
306 // Start using the channel name sent from outer peer. The channel name
307 // of the inner peer can easily become out of date, as iframe's and their
308 // JS state get cached in many browsers upon page reload or history
309 // navigation (particularly Firefox 1.5+).
310 staleChannel.updateChannelNameAndCatalog(channelName);
311 staleChannel.xpcDeliver(service, payload);
312 return true;
313 }
314 }
315
316 // Failed to find a channel to deliver this message to, so simply ignore it.
317 goog.log.info(goog.net.xpc.logger, 'channel name mismatch; message ignored.');
318 return false;
319};

330/**
331 * Handles transport service messages.
332 * @param {string} payload The message content.
333 * @override
334 */
335DirectTransport.prototype.transportServiceHandler = function(payload) {
336 var transportParts = DirectTransport.parseTransportPayload_(payload);
337 var transportMessageType = transportParts[0];
338 var peerEndpointId = transportParts[1];
339 switch (transportMessageType) {
340 case goog.net.xpc.SETUP_ACK_:
341 if (!this.setupAckReceived_.hasFired()) {
342 this.setupAckReceived_.callback(true);
343 }
344 break;
345 case goog.net.xpc.SETUP:
346 this.sendSetupAckMessage_();
347 if ((this.peerEndpointId_ != null) &&
348 (this.peerEndpointId_ != peerEndpointId)) {
349 // Send a new SETUP message since the peer has been replaced.
350 goog.log.info(goog.net.xpc.logger,
351 'Sending SETUP and changing peer ID to: ' + peerEndpointId);
352 this.sendSetupMessage_();
353 }
354 this.peerEndpointId_ = peerEndpointId;
355 break;
356 }
357};

360/**
361 * Sends a SETUP transport service message.
362 * @private
363 */
364DirectTransport.prototype.sendSetupMessage_ = function() {
365 // Although we could send real objects, since some other transports are
366 // limited to strings we also keep this requirement.
367 var payload = goog.net.xpc.SETUP;
368 payload += DirectTransport.MESSAGE_DELIMITER_;
369 payload += this.endpointId_;
370 this.send(goog.net.xpc.TRANSPORT_SERVICE_, payload);
371};

399/**
400 * Connects to other peer. In the case of the outer peer, the setup messages are
401 * likely sent before the inner peer is ready to receive them. Therefore, this
402 * function will continue trying to send the SETUP message until the inner peer
403 * responds. In the case of the inner peer, it will occasionally have its
404 * channel name fall out of sync with the outer peer, particularly during
405 * soft-reloads and history navigations.
406 * @private
407 */
408DirectTransport.prototype.maybeAttemptToConnect_ = function() {
409 var outerRole = this.channel_.getRole() == CrossPageChannelRole.OUTER;
410 if (this.channel_.isConnected()) {
411 this.maybeAttemptToConnectTimer_.stop();
412 return;
413 }
414 this.maybeAttemptToConnectTimer_.start();
415 this.sendSetupMessage_();
416};

419/**
420 * Prepares to send a message.
421 * @param {string} service The name of the service the message is to be
422 * delivered to.
423 * @param {string} payload The message content.
424 * @override
425 */
426DirectTransport.prototype.send = function(service, payload) {
427 if (!this.channel_.getPeerWindowObject()) {
428 goog.log.fine(goog.net.xpc.logger, 'send(): window not ready');
429 return;
430 }
431 var channelName = DirectTransport.getRoledChannelName_(
432 this.originalChannelName_,
433 this.getPeerRole_());
434
435 var message = new DirectTransport.Message_(
436 channelName,
437 service,
438 payload);
439
440 if (this.channel_.getConfig()[CfgFields.DIRECT_TRANSPORT_SYNC_MODE]) {
441 this.executeScheduledSend_(message);
442 } else {
443 // Note: goog.async.nextTick doesn't support cancelling or disposal so
444 // leaving as 0ms timer, though this may have performance implications.
445 this.asyncSendsMap_[goog.getUid(message)] =
446 Timer.callOnce(goog.bind(this.executeScheduledSend_, this, message), 0);
447 }
448};

451/**
452 * Sends the message.
453 * @param {!DirectTransport.Message_} message The message to send.
454 * @private
455 */
456DirectTransport.prototype.executeScheduledSend_ = function(message) {
457 var messageId = goog.getUid(message);
458 if (this.asyncSendsMap_[messageId]) {
459 delete this.asyncSendsMap_[messageId];
460 }
461
462 /** @preserveTry */
463 try {
464 var peerProxy = goog.getObjectByName(
465 DirectTransport.GLOBAL_TRANPORT_PATH_,
466 this.channel_.getPeerWindowObject());
467 } catch (error) {
468 goog.log.warning(
469 goog.net.xpc.logger,
470 'Can\'t access other window, ignoring.',
471 error);
472 return;
473 }
474
475 if (goog.isNull(peerProxy)) {
476 goog.log.warning(
477 goog.net.xpc.logger,
478 'Peer window had no global function.');
479 return;
480 }
481
482 /** @preserveTry */
483 try {
484 peerProxy(message.toLiteral());
485 goog.log.info(
486 goog.net.xpc.logger,
487 'send(): channelName=' + message.channelName +
488 ' service=' + message.service +
489 ' payload=' + message.payload);
490 } catch (error) {
491 goog.log.warning(
492 goog.net.xpc.logger,
493 'Error performing call, ignoring.',
494 error);
495 }
496};

525/** @override */
526DirectTransport.prototype.disposeInternal = function() {
527 if (this.initialized_) {
528 var listenWindow = this.getWindow();
529 var uid = goog.getUid(listenWindow);
530 var value = --DirectTransport.activeCount_[uid];
531 if (value == 1) {
532 goog.exportSymbol(
533 DirectTransport.GLOBAL_TRANPORT_PATH_,
534 null,
535 listenWindow);
536 }
537 }
538
539 if (this.asyncSendsMap_) {
540 goog.object.forEach(this.asyncSendsMap_, function(timerId) {
541 Timer.clear(timerId);
542 });
543 this.asyncSendsMap_ = null;
544 }
545
546 // Deferred's aren't disposables.
547 if (this.setupAckReceived_) {
548 this.setupAckReceived_.cancel();
549 delete this.setupAckReceived_;
550 }
551 if (this.setupAckSent_) {
552 this.setupAckSent_.cancel();
553 delete this.setupAckSent_;
554 }
555 if (this.connected_) {
556 this.connected_.cancel();
557 delete this.connected_;
558 }
559
560 DirectTransport.base(this, 'disposeInternal');
561};

564/**
565 * Parses a transport service payload message.
566 * @param {string} payload The payload.
567 * @return {!Array<?string>} An array with the message type as the first member
568 * and the endpoint id as the second, if one was sent, or null otherwise.
569 * @private
570 */
571DirectTransport.parseTransportPayload_ = function(payload) {
572 var transportParts = /** @type {!Array<?string>} */ (payload.split(
573 DirectTransport.MESSAGE_DELIMITER_));
574 transportParts[1] = transportParts[1] || null; // Usually endpointId.
575 return transportParts;
576};
