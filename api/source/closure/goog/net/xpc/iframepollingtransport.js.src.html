38/**
39 * Iframe polling transport. Uses hidden iframes to transfer data
40 * in the fragment identifier of the URL. The peer polls the iframe's location
41 * for changes.
42 * Unfortunately, in Safari this screws up the history, because Safari doesn't
43 * allow to call location.replace() on a window containing a document from a
44 * different domain (last version tested: 2.0.4).
45 *
46 * @param {goog.net.xpc.CrossPageChannel} channel The channel this
47 * transport belongs to.
48 * @param {goog.dom.DomHelper=} opt_domHelper The dom helper to use for finding
49 * the correct window.
50 * @constructor
51 * @extends {goog.net.xpc.Transport}
52 * @final
53 */
54goog.net.xpc.IframePollingTransport = function(channel, opt_domHelper) {
55 goog.net.xpc.IframePollingTransport.base(this, 'constructor', opt_domHelper);
56
57 /**
58 * The channel this transport belongs to.
59 * @type {goog.net.xpc.CrossPageChannel}
60 * @private
61 */
62 this.channel_ = channel;
63
64 /**
65 * The URI used to send messages.
66 * @type {string}
67 * @private
68 */
69 this.sendUri_ =
70 this.channel_.getConfig()[goog.net.xpc.CfgFields.PEER_POLL_URI];
71
72 /**
73 * The URI which is polled for incoming messages.
74 * @type {string}
75 * @private
76 */
77 this.rcvUri_ =
78 this.channel_.getConfig()[goog.net.xpc.CfgFields.LOCAL_POLL_URI];
79
80 /**
81 * The queue to hold messages which can't be sent immediately.
82 * @type {Array<string>}
83 * @private
84 */
85 this.sendQueue_ = [];
86};
87goog.inherits(goog.net.xpc.IframePollingTransport, goog.net.xpc.Transport);

238/**
239 * Safely retrieves the frames from the peer window. If an error is thrown
240 * (e.g. the window is closing) an empty frame object is returned.
241 * @return {!Object<!Window>} The frames from the peer window.
242 * @private
243 */
244goog.net.xpc.IframePollingTransport.prototype.getPeerFrames_ = function() {
245 try {
246 if (this.isChannelAvailable()) {
247 return this.channel_.getPeerWindowObject().frames || {};
248 }
249 } catch (e) {
250 // An error may be thrown if the window is closing.
251 goog.log.fine(goog.net.xpc.logger, 'error retrieving peer frames');
252 }
253 return {};
254};

269/**
270 * Connects this transport.
271 * @override
272 */
273goog.net.xpc.IframePollingTransport.prototype.connect = function() {
274 if (!this.isChannelAvailable()) {
275 // When the channel is unavailable there is no peer to poll so stop trying
276 // to connect.
277 return;
278 }
279
280 goog.log.fine(goog.net.xpc.logger, 'transport connect called');
281 if (!this.initialized_) {
282 goog.log.fine(goog.net.xpc.logger, 'initializing...');
283 this.constructSenderFrames_();
284 this.initialized_ = true;
285 }
286 this.checkForeignFramesReady_();
287};

290/**
291 * Creates the iframes which are used to send messages (and acknowledgements)
292 * to the peer. Sender iframes contain a document from a different origin and
293 * therefore their content can't be accessed.
294 * @private
295 */
296goog.net.xpc.IframePollingTransport.prototype.constructSenderFrames_ =
297 function() {
298 var name = this.getMsgFrameName_();
299 this.msgIframeElm_ = this.constructSenderFrame_(name);
300 this.msgWinObj_ = this.getWindow().frames[name];
301
302 name = this.getAckFrameName_();
303 this.ackIframeElm_ = this.constructSenderFrame_(name);
304 this.ackWinObj_ = this.getWindow().frames[name];
305};

314goog.net.xpc.IframePollingTransport.prototype.constructSenderFrame_ =

329/**
330 * The protocol for reconnecting is for the inner frame to change channel
331 * names, and then communicate the new channel name to the outer peer.
332 * The outer peer looks in a predefined location for the channel name
333 * upate. It is important to use a completely new channel name, as this
334 * will ensure that all messaging iframes are not in the bfcache.
335 * Otherwise, Safari may pollute the history when modifying the location
336 * of bfcached iframes.
337 * @private
338 */
339goog.net.xpc.IframePollingTransport.prototype.maybeInnerPeerReconnect_ =
340 function() {
341 // Reconnection has been found to not function on some browsers (eg IE7), so
342 // it's important that the mechanism only be triggered as a last resort. As
343 // such, we poll a number of times to find the outer iframe before triggering
344 // it.
345 if (this.reconnectFrame_ || this.pollsBeforeReconnect_-- > 0) {
346 return;
347 }
348
349 goog.log.log(goog.net.xpc.logger, goog.log.Level.FINEST,
350 'Inner peer reconnect triggered.');
351 this.channel_.updateChannelNameAndCatalog(goog.net.xpc.getRandomString(10));
352 goog.log.log(goog.net.xpc.logger, goog.log.Level.FINEST,
353 'switching channels: ' + this.channel_.name);
354 this.deconstructSenderFrames_();
355 this.initialized_ = false;
356 // Communicate new channel name to outer peer.
357 this.reconnectFrame_ = this.constructSenderFrame_(
358 goog.net.xpc.IframePollingTransport.IFRAME_PREFIX +
359 '_reconnect_' + this.channel_.name);
360};

363/**
364 * Scans inner peer for a reconnect message, which will be used to update
365 * the outer peer's channel name. If a reconnect message is found, the
366 * sender frames will be cleaned up to make way for the new sender frames.
367 * Only called by the outer peer.
368 * @private
369 */
370goog.net.xpc.IframePollingTransport.prototype.outerPeerReconnect_ = function() {
371 goog.log.log(goog.net.xpc.logger, goog.log.Level.FINEST,
372 'outerPeerReconnect called');
373 var frames = this.getPeerFrames_();
374 var length = frames.length;
375 for (var i = 0; i < length; i++) {
376 var frameName;
377 try {
378 if (frames[i] && frames[i].name) {
379 frameName = frames[i].name;
380 }
381 } catch (e) {
382 // Do nothing.
383 }
384 if (!frameName) {
385 continue;
386 }
387 var message = frameName.split('_');
388 if (message.length == 3 &&
389 message[0] == goog.net.xpc.IframePollingTransport.IFRAME_PREFIX &&
390 message[1] == 'reconnect') {
391 // This is a legitimate reconnect message from the peer. Start using
392 // the peer provided channel name, and start a connection over from
393 // scratch.
394 this.channel_.name = message[2];
395 this.deconstructSenderFrames_();
396 this.initialized_ = false;
397 break;
398 }
399 }
400};

403/**
404 * Cleans up the existing sender frames owned by this peer. Only called by
405 * the outer peer.
406 * @private
407 */
408goog.net.xpc.IframePollingTransport.prototype.deconstructSenderFrames_ =

411 'deconstructSenderFrames called');
412 if (this.msgIframeElm_) {

425/**
426 * Checks if the frames in the peer's page are ready. These contain a
427 * document from the own domain and are the ones messages are received through.
428 * @private
429 */
430goog.net.xpc.IframePollingTransport.prototype.checkForeignFramesReady_ =
431 function() {
432 // check if the connected iframe ready
433 if (!(this.isRcvFrameReady_(this.getMsgFrameName_()) &&
434 this.isRcvFrameReady_(this.getAckFrameName_()))) {
435 goog.log.log(goog.net.xpc.logger, goog.log.Level.FINEST,
436 'foreign frames not (yet) present');
437
438 if (this.channel_.getRole() == goog.net.xpc.CrossPageChannelRole.INNER) {
439 // The outer peer might need a short time to get its frames ready, as
440 // CrossPageChannel prevents them from getting created until the inner
441 // peer's frame has thrown its loaded event. This method is a noop for
442 // the first few times it's called, and then allows the reconnection
443 // sequence to begin.
444 this.maybeInnerPeerReconnect_();
445 } else if (this.channel_.getRole() ==
446 goog.net.xpc.CrossPageChannelRole.OUTER) {
447 // The inner peer is either not loaded yet, or the receiving
448 // frames are simply missing. Since we cannot discern the two cases, we
449 // should scan for a reconnect message from the inner peer.
450 this.outerPeerReconnect_();
451 }
452
453 // start a timer to check again
454 this.getWindow().setTimeout(goog.bind(this.connect, this), 100);
455 } else {
456 goog.log.fine(goog.net.xpc.logger, 'foreign frames present');
457
458 // Create receivers.
459 this.msgReceiver_ = new goog.net.xpc.IframePollingTransport.Receiver(
460 this,
461 this.getPeerFrame_(this.getMsgFrameName_()),
462 goog.bind(this.processIncomingMsg, this));
463 this.ackReceiver_ = new goog.net.xpc.IframePollingTransport.Receiver(
464 this,
465 this.getPeerFrame_(this.getAckFrameName_()),
466 goog.bind(this.processIncomingAck, this));
467
468 this.checkLocalFramesPresent_();
469 }
470};

473/**
474 * Checks if the receiving frame is ready.
475 * @param {string} frameName Which receiving frame to check.
476 * @return {boolean} Whether the receiving frame is ready.
477 * @private
478 */
479goog.net.xpc.IframePollingTransport.prototype.isRcvFrameReady_ =
480 function(frameName) {
481 goog.log.log(goog.net.xpc.logger, goog.log.Level.FINEST,
482 'checking for receive frame: ' + frameName);
483 /** @preserveTry */
484 try {
485 var winObj = this.getPeerFrame_(frameName);
486 if (!winObj || winObj.location.href.indexOf(this.rcvUri_) != 0) {
487 return false;
488 }
489 } catch (e) {
490 return false;
491 }
492 return true;
493};

496/**
497 * Checks if the iframes created in the own document are ready.
498 * @private
499 */
500goog.net.xpc.IframePollingTransport.prototype.checkLocalFramesPresent_ =
501 function() {
502
503 // Are the sender frames ready?
504 // These contain a document from the peer's domain, therefore we can only
505 // check if the frame itself is present.
506 var frames = this.getPeerFrames_();
507 if (!(frames[this.getAckFrameName_()] &&
508 frames[this.getMsgFrameName_()])) {
509 // start a timer to check again
510 if (!this.checkLocalFramesPresentCb_) {
511 this.checkLocalFramesPresentCb_ = goog.bind(
512 this.checkLocalFramesPresent_, this);
513 }
514 this.getWindow().setTimeout(this.checkLocalFramesPresentCb_, 100);
515 goog.log.fine(goog.net.xpc.logger, 'local frames not (yet) present');
516 } else {
517 // Create senders.
518 this.msgSender_ = new goog.net.xpc.IframePollingTransport.Sender(
519 this.sendUri_, this.msgWinObj_);
520 this.ackSender_ = new goog.net.xpc.IframePollingTransport.Sender(
521 this.sendUri_, this.ackWinObj_);
522
523 goog.log.fine(goog.net.xpc.logger, 'local frames ready');
524
525 this.getWindow().setTimeout(goog.bind(function() {
526 this.msgSender_.send(goog.net.xpc.SETUP);
527 this.sentConnectionSetup_ = true;
528 this.waitForAck_ = true;
529 goog.log.fine(goog.net.xpc.logger, 'SETUP sent');
530 }, this), 100);
531 }
532};

535/**
536 * Check if connection is ready.
537 * @private
538 */
539goog.net.xpc.IframePollingTransport.prototype.checkIfConnected_ = function() {
540 if (this.sentConnectionSetupAck_ && this.rcvdConnectionSetupAck_) {
541 this.channel_.notifyConnected();
542
543 if (this.deliveryQueue_) {
544 goog.log.fine(goog.net.xpc.logger, 'delivering queued messages ' +
545 '(' + this.deliveryQueue_.length + ')');
546
547 for (var i = 0, m; i < this.deliveryQueue_.length; i++) {
548 m = this.deliveryQueue_[i];
549 this.channel_.xpcDeliver(m.service, m.payload);
550 }
551 delete this.deliveryQueue_;
552 }
553 } else {
554 goog.log.log(goog.net.xpc.logger, goog.log.Level.FINEST,
555 'checking if connected: ' +
556 'ack sent:' + this.sentConnectionSetupAck_ +
557 ', ack rcvd: ' + this.rcvdConnectionSetupAck_);
558 }
559};

562/**
563 * Processes an incoming message.
564 * @param {string} raw The complete received string.
565 */
566goog.net.xpc.IframePollingTransport.prototype.processIncomingMsg =
567 function(raw) {
568 goog.log.log(goog.net.xpc.logger, goog.log.Level.FINEST,
569 'msg received: ' + raw);
570
571 if (raw == goog.net.xpc.SETUP) {
572 if (!this.ackSender_) {
573 // Got SETUP msg, but we can't send an ack.
574 return;
575 }
576
577 this.ackSender_.send(goog.net.xpc.SETUP_ACK_);
578 goog.log.log(goog.net.xpc.logger, goog.log.Level.FINEST, 'SETUP_ACK sent');
579
580 this.sentConnectionSetupAck_ = true;
581 this.checkIfConnected_();
582
583 } else if (this.channel_.isConnected() || this.sentConnectionSetupAck_) {
584
585 var pos = raw.indexOf('|');
586 var head = raw.substring(0, pos);
587 var frame = raw.substring(pos + 1);
588
589 // check if it is a framed message
590 pos = head.indexOf(',');
591 if (pos == -1) {
592 var seq = head;
593 // send acknowledgement
594 this.ackSender_.send('ACK:' + seq);
595 this.deliverPayload_(frame);
596 } else {
597 var seq = head.substring(0, pos);
598 // send acknowledgement
599 this.ackSender_.send('ACK:' + seq);
600
601 var partInfo = head.substring(pos + 1).split('/');
602 var part0 = parseInt(partInfo[0], 10);
603 var part1 = parseInt(partInfo[1], 10);
604 // create an array to accumulate the parts if this is the
605 // first frame of a message
606 if (part0 == 1) {
607 this.parts_ = [];
608 }
609 this.parts_.push(frame);
610 // deliver the message if this was the last frame of a message
611 if (part0 == part1) {
612 this.deliverPayload_(this.parts_.join(''));
613 delete this.parts_;
614 }
615 }
616 } else {
617 goog.log.warning(goog.net.xpc.logger,
618 'received msg, but channel is not connected');
619 }
620};

623/**
624 * Process an incoming acknowdedgement.
625 * @param {string} msgStr The incoming ack string to process.
626 */
627goog.net.xpc.IframePollingTransport.prototype.processIncomingAck =
628 function(msgStr) {
629 goog.log.log(goog.net.xpc.logger, goog.log.Level.FINEST,
630 'ack received: ' + msgStr);
631
632 if (msgStr == goog.net.xpc.SETUP_ACK_) {
633 this.waitForAck_ = false;
634 this.rcvdConnectionSetupAck_ = true;
635 // send the next frame
636 this.checkIfConnected_();
637
638 } else if (this.channel_.isConnected()) {
639 if (!this.waitForAck_) {
640 goog.log.warning(goog.net.xpc.logger, 'got unexpected ack');
641 return;
642 }
643
644 var seq = parseInt(msgStr.split(':')[1], 10);
645 if (seq == this.sequence_) {
646 this.waitForAck_ = false;
647 this.sendNextFrame_();
648 } else {
649 goog.log.warning(goog.net.xpc.logger, 'got ack with wrong sequence');
650 }
651 } else {
652 goog.log.warning(goog.net.xpc.logger,
653 'received ack, but channel not connected');
654 }
655};

658/**
659 * Sends a frame (message part).
660 * @private
661 */
662goog.net.xpc.IframePollingTransport.prototype.sendNextFrame_ = function() {
663 // do nothing if we are waiting for an acknowledgement or the
664 // queue is emtpy
665 if (this.waitForAck_ || !this.sendQueue_.length) {
666 return;
667 }
668
669 var s = this.sendQueue_.shift();
670 ++this.sequence_;
671 this.msgSender_.send(this.sequence_ + s);
672 goog.log.log(goog.net.xpc.logger, goog.log.Level.FINEST,
673 'msg sent: ' + this.sequence_ + s);
674
675
676 this.waitForAck_ = true;
677};

680/**
681 * Delivers a message.
682 * @param {string} s The complete message string ("<service_name>:<payload>").
683 * @private
684 */
685goog.net.xpc.IframePollingTransport.prototype.deliverPayload_ = function(s) {
686 // determine the service name and the payload
687 var pos = s.indexOf(':');
688 var service = s.substr(0, pos);
689 var payload = s.substring(pos + 1);
690
691 // deliver the message
692 if (!this.channel_.isConnected()) {
693 // as valid messages can come in before a SETUP_ACK has
694 // been received (because subchannels for msgs and acks are independent),
695 // delay delivery of early messages until after 'connect'-event
696 (this.deliveryQueue_ || (this.deliveryQueue_ = [])).
697 push({service: service, payload: payload});
698 goog.log.log(goog.net.xpc.logger, goog.log.Level.FINEST,
699 'queued delivery');
700 } else {
701 this.channel_.xpcDeliver(service, payload);
702 }
703};

709/**
710 * Maximal frame length.
711 * @type {number}
712 * @private
713 */
714goog.net.xpc.IframePollingTransport.prototype.MAX_FRAME_LENGTH_ = 3800;
715
716
717/**
718 * Sends a message. Splits it in multiple frames if too long (exceeds IE's
719 * URL-length maximum.
720 * Wireformat: <seq>[,<frame_no>/<#frames>]|<frame_content>
721 *
722 * @param {string} service Name of service this the message has to be delivered.
723 * @param {string} payload The message content.
724 * @override
725 */
726goog.net.xpc.IframePollingTransport.prototype.send =
727 function(service, payload) {
728 var frame = service + ':' + payload;
729 // put in queue
730 if (!goog.userAgent.IE || payload.length <= this.MAX_FRAME_LENGTH_) {
731 this.sendQueue_.push('|' + frame);
732 }
733 else {
734 var l = payload.length;
735 var num = Math.ceil(l / this.MAX_FRAME_LENGTH_); // number of frames
736 var pos = 0;
737 var i = 1;
738 while (pos < l) {
739 this.sendQueue_.push(',' + i + '/' + num + '|' +
740 frame.substr(pos, this.MAX_FRAME_LENGTH_));
741 i++;
742 pos += this.MAX_FRAME_LENGTH_;
743 }
744 }
745 this.sendNextFrame_();

799/**
800 * Polls all receivers.
801 * @private
802 */
803goog.net.xpc.IframePollingTransport.receive_ = function() {
804 var receivers = goog.net.xpc.IframePollingTransport.receivers_;
805 var receiver;
806 var rcvd = false;
807
808 /** @preserveTry */
809 try {
810 for (var i = 0; receiver = receivers[i]; i++) {
811 rcvd = rcvd || receiver.receive();
812 }
813 } catch (e) {
814 goog.log.info(goog.net.xpc.logger, 'receive_() failed: ' + e);
815
816 // Notify the channel that the transport had an error.
817 receiver.transport_.channel_.notifyTransportError();
818
819 // notifyTransportError() closes the channel and disposes the transport.
820 // If there are no other channels present, this.receivers_ will now be empty
821 // and there is no need to keep polling.
822 if (!receivers.length) {
823 return;
824 }
825 }
826
827 var now = goog.now();
828 if (rcvd) {
829 goog.net.xpc.IframePollingTransport.lastActivity_ = now;
830 }
831
832 // Schedule next check.
833 var t = now - goog.net.xpc.IframePollingTransport.lastActivity_ <
834 goog.net.xpc.IframePollingTransport.TIME_SHORT_POLL_AFTER_ACTIVITY_ ?
835 goog.net.xpc.IframePollingTransport.TIME_POLL_SHORT_ :
836 goog.net.xpc.IframePollingTransport.TIME_POLL_LONG_;
837 goog.net.xpc.IframePollingTransport.rcvTimer_ = window.setTimeout(
838 goog.net.xpc.IframePollingTransport.receiveCb_, t);
839};

916/**
917 * Sends a message-part (frame) to the peer.
918 * The message-part is encoded and put in the fragment identifier
919 * of the URL used for sending (and belongs to the origin/domain of the peer).
920 * @param {string} payload The message to send.

922goog.net.xpc.IframePollingTransport.Sender.prototype.send = function(payload) {
923 this.cycle_ = ++this.cycle_ % 2;

925 var url =
926 this.sanitizedSendUri_ + '#' + this.cycle_ + encodeURIComponent(payload);

928 // TODO(user) Find out if try/catch is still needed

937 } catch (e) {
938 goog.log.error(goog.net.xpc.logger, 'sending failed', e);
939 }

942 // where we need to capture responses quickly.
943 goog.net.xpc.IframePollingTransport.startRcvTimer_();

978/**
979 * Polls the location of the receiver-frame for changes.
980 * @return {boolean} Whether a change has been detected.
981 */
982goog.net.xpc.IframePollingTransport.Receiver.prototype.receive = function() {
983 var loc = this.rcvFrame_.location.href;
984
985 if (loc != this.currentLoc_) {
986 this.currentLoc_ = loc;
987 var payload = loc.split('#')[1];
988 if (payload) {
989 payload = payload.substr(1); // discard first character (cycle)
990 this.cb_(decodeURIComponent(payload));

992 return true;
