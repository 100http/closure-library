54/**
55 * A communication channel between two documents from different domains.
56 * Provides asynchronous messaging.
57 *
58 * @param {Object} cfg Channel configuration object.
59 * @param {goog.dom.DomHelper=} opt_domHelper The optional dom helper to
60 * use for looking up elements in the dom.
61 * @constructor
62 * @extends {goog.messaging.AbstractChannel}
63 */
64goog.net.xpc.CrossPageChannel = function(cfg, opt_domHelper) {
65 goog.net.xpc.CrossPageChannel.base(this, 'constructor');
66
67 for (var i = 0, uriField; uriField = goog.net.xpc.UriCfgFields[i]; i++) {
68 if (uriField in cfg && !/^https?:\/\//.test(cfg[uriField])) {
69 throw Error('URI ' + cfg[uriField] + ' is invalid for field ' + uriField);
70 }
71 }
72
73 /**
74 * The configuration for this channel.
75 * @type {Object}
76 * @private
77 */
78 this.cfg_ = cfg;
79
80 /**
81 * The name of the channel. Please use
82 * <code>updateChannelNameAndCatalog</code> to change this from the transports
83 * vs changing the property directly.
84 * @type {string}
85 */
86 this.name = this.cfg_[goog.net.xpc.CfgFields.CHANNEL_NAME] ||
87 goog.net.xpc.getRandomString(10);
88
89 /**
90 * The dom helper to use for accessing the dom.
91 * @type {goog.dom.DomHelper}
92 * @private
93 */
94 this.domHelper_ = opt_domHelper || goog.dom.getDomHelper();
95
96 /**
97 * Collects deferred function calls which will be made once the connection
98 * has been fully set up.
99 * @type {!Array<function()>}
100 * @private
101 */
102 this.deferredDeliveries_ = [];
103
104 /**
105 * An event handler used to listen for load events on peer iframes.
106 * @type {!goog.events.EventHandler<!goog.net.xpc.CrossPageChannel>}
107 * @private
108 */
109 this.peerLoadHandler_ = new goog.events.EventHandler(this);
110
111 // If LOCAL_POLL_URI or PEER_POLL_URI is not available, try using
112 // robots.txt from that host.
113 cfg[goog.net.xpc.CfgFields.LOCAL_POLL_URI] =
114 cfg[goog.net.xpc.CfgFields.LOCAL_POLL_URI] ||
115 goog.uri.utils.getHost(this.domHelper_.getWindow().location.href) +
116 '/robots.txt';
117 // PEER_URI is sometimes undefined in tests.
118 cfg[goog.net.xpc.CfgFields.PEER_POLL_URI] =
119 cfg[goog.net.xpc.CfgFields.PEER_POLL_URI] ||
120 goog.uri.utils.getHost(cfg[goog.net.xpc.CfgFields.PEER_URI] || '') +
121 '/robots.txt';
122
123 goog.net.xpc.channels[this.name] = this;
124
125 if (!goog.events.getListener(window, goog.events.EventType.UNLOAD,
126 goog.net.xpc.CrossPageChannel.disposeAll_)) {
127 // Set listener to dispose all registered channels on page unload.
128 goog.events.listenOnce(window, goog.events.EventType.UNLOAD,
129 goog.net.xpc.CrossPageChannel.disposeAll_);
130 }
131
132 goog.log.info(goog.net.xpc.logger, 'CrossPageChannel created: ' + this.name);
133};
134goog.inherits(goog.net.xpc.CrossPageChannel, goog.messaging.AbstractChannel);

260/**
261 * Determines whether the peer window is available (e.g. not closed).
262 *
263 * @return {boolean} Whether the peer window is available.
264 * @package
265 */
266goog.net.xpc.CrossPageChannel.prototype.isPeerAvailable = function() {
267 // NOTE(user): This check is not reliable in IE, where a document in an
268 // iframe does not get unloaded when removing the iframe element from the DOM.
269 // TODO(user): Find something that works in IE as well.
270 // NOTE(user): "!this.peerWindowObject_.closed" evaluates to 'false' in IE9
271 // sometimes even though typeof(this.peerWindowObject_.closed) is boolean and
272 // this.peerWindowObject_.closed evaluates to 'false'. Casting it to a Boolean
273 // results in sane evaluation. When this happens, it's in the inner iframe
274 // when querying its parent's 'closed' status. Note that this is a different
275 // case than mibuerge@'s note above.
276 try {
277 return !!this.peerWindowObject_ && !Boolean(this.peerWindowObject_.closed);
278 } catch (e) {
279 // If the window is closing, an error may be thrown.
280 return false;
281 }
282};

285/**
286 * Determine which transport type to use for this channel / useragent.
287 * @return {goog.net.xpc.TransportTypes|undefined} The best transport type.
288 * @private
289 */
290goog.net.xpc.CrossPageChannel.prototype.determineTransportType_ = function() {
291 var transportType;
292 if (goog.isFunction(document.postMessage) ||
293 goog.isFunction(window.postMessage) ||
294 // IE8 supports window.postMessage, but
295 // typeof window.postMessage returns "object"
296 (goog.userAgent.IE && window.postMessage)) {
297 transportType = goog.net.xpc.TransportTypes.NATIVE_MESSAGING;
298 } else if (goog.userAgent.GECKO) {
299 transportType = goog.net.xpc.TransportTypes.FRAME_ELEMENT_METHOD;
300 } else if (goog.userAgent.IE &&
301 this.cfg_[goog.net.xpc.CfgFields.PEER_RELAY_URI]) {
302 transportType = goog.net.xpc.TransportTypes.IFRAME_RELAY;
303 } else if (goog.userAgent.IE && goog.net.xpc.NixTransport.isNixSupported()) {
304 transportType = goog.net.xpc.TransportTypes.NIX;
305 } else {
306 transportType = goog.net.xpc.TransportTypes.IFRAME_POLLING;
307 }
308 return transportType;
309};

312/**
313 * Creates the transport for this channel. Chooses from the available
314 * transport based on the user agent and the configuration.
315 * @private
316 */
317goog.net.xpc.CrossPageChannel.prototype.createTransport_ = function() {
318 // return, if the transport has already been created
319 if (this.transport_) {
320 return;
321 }
322
323 // TODO(user): Use goog.scope.
324 var CfgFields = goog.net.xpc.CfgFields;
325
326 if (!this.cfg_[CfgFields.TRANSPORT]) {
327 this.cfg_[CfgFields.TRANSPORT] =
328 this.determineTransportType_();
329 }
330
331 switch (this.cfg_[CfgFields.TRANSPORT]) {
332 case goog.net.xpc.TransportTypes.NATIVE_MESSAGING:
333 var protocolVersion = this.cfg_[
334 CfgFields.NATIVE_TRANSPORT_PROTOCOL_VERSION] || 2;
335 this.transport_ = new goog.net.xpc.NativeMessagingTransport(
336 this,
337 this.cfg_[CfgFields.PEER_HOSTNAME],
338 this.domHelper_,
339 !!this.cfg_[CfgFields.ONE_SIDED_HANDSHAKE],
340 protocolVersion);
341 break;
342 case goog.net.xpc.TransportTypes.NIX:
343 this.transport_ = new goog.net.xpc.NixTransport(this, this.domHelper_);
344 break;
345 case goog.net.xpc.TransportTypes.FRAME_ELEMENT_METHOD:
346 this.transport_ =
347 new goog.net.xpc.FrameElementMethodTransport(this, this.domHelper_);
348 break;
349 case goog.net.xpc.TransportTypes.IFRAME_RELAY:
350 this.transport_ =
351 new goog.net.xpc.IframeRelayTransport(this, this.domHelper_);
352 break;
353 case goog.net.xpc.TransportTypes.IFRAME_POLLING:
354 this.transport_ =
355 new goog.net.xpc.IframePollingTransport(this, this.domHelper_);
356 break;
357 case goog.net.xpc.TransportTypes.DIRECT:
358 if (this.peerWindowObject_ &&
359 goog.net.xpc.DirectTransport.isSupported(/** @type {!Window} */ (
360 this.peerWindowObject_))) {
361 this.transport_ =
362 new goog.net.xpc.DirectTransport(this, this.domHelper_);
363 } else {
364 goog.log.info(
365 goog.net.xpc.logger,
366 'DirectTransport not supported for this window, peer window in' +
367 ' different security context or not set yet.');
368 }
369 break;
370 }
371
372 if (this.transport_) {
373 goog.log.info(goog.net.xpc.logger,
374 'Transport created: ' + this.transport_.getName());
375 } else {
376 throw Error('CrossPageChannel: No suitable transport found!');
377 }
378};

399/**
400 * @return {!Object} Configuration-object to be used by the peer to
401 * initialize the channel.
402 */
403goog.net.xpc.CrossPageChannel.prototype.getPeerConfiguration = function() {
404 var peerCfg = {};
405 peerCfg[goog.net.xpc.CfgFields.CHANNEL_NAME] = this.name;
406 peerCfg[goog.net.xpc.CfgFields.TRANSPORT] =
407 this.cfg_[goog.net.xpc.CfgFields.TRANSPORT];
408 peerCfg[goog.net.xpc.CfgFields.ONE_SIDED_HANDSHAKE] =
409 this.cfg_[goog.net.xpc.CfgFields.ONE_SIDED_HANDSHAKE];
410
411 if (this.cfg_[goog.net.xpc.CfgFields.LOCAL_RELAY_URI]) {
412 peerCfg[goog.net.xpc.CfgFields.PEER_RELAY_URI] =
413 this.cfg_[goog.net.xpc.CfgFields.LOCAL_RELAY_URI];
414 }
415 if (this.cfg_[goog.net.xpc.CfgFields.LOCAL_POLL_URI]) {
416 peerCfg[goog.net.xpc.CfgFields.PEER_POLL_URI] =
417 this.cfg_[goog.net.xpc.CfgFields.LOCAL_POLL_URI];
418 }
419 if (this.cfg_[goog.net.xpc.CfgFields.PEER_POLL_URI]) {
420 peerCfg[goog.net.xpc.CfgFields.LOCAL_POLL_URI] =
421 this.cfg_[goog.net.xpc.CfgFields.PEER_POLL_URI];
422 }
423 var role = this.cfg_[goog.net.xpc.CfgFields.ROLE];
424 if (role) {
425 peerCfg[goog.net.xpc.CfgFields.ROLE] =
426 role == goog.net.xpc.CrossPageChannelRole.INNER ?
427 goog.net.xpc.CrossPageChannelRole.OUTER :
428 goog.net.xpc.CrossPageChannelRole.INNER;
429 }
430
431 return peerCfg;
432};

435/**
436 * Creates the iframe containing the peer page in a specified parent element.
437 * This method does not connect the channel, connect() still has to be called
438 * separately.
439 *
440 * @param {!Element} parentElm The container element the iframe is appended to.
441 * @param {Function=} opt_configureIframeCb If present, this function gets
442 * called with the iframe element as parameter to allow setting properties
443 * on it before it gets added to the DOM. If absent, the iframe's width and
444 * height are set to '100%'.
445 * @param {boolean=} opt_addCfgParam Whether to add the peer configuration as
446 * URL parameter (default: true).
447 * @return {!HTMLIFrameElement} The iframe element.
448 */
449goog.net.xpc.CrossPageChannel.prototype.createPeerIframe = function(
450 parentElm, opt_configureIframeCb, opt_addCfgParam) {
451 goog.log.info(goog.net.xpc.logger, 'createPeerIframe()');
452
453 var iframeId = this.cfg_[goog.net.xpc.CfgFields.IFRAME_ID];
454 if (!iframeId) {
455 // Create a randomized ID for the iframe element to avoid
456 // bfcache-related issues.
457 iframeId = this.cfg_[goog.net.xpc.CfgFields.IFRAME_ID] =
458 'xpcpeer' + goog.net.xpc.getRandomString(4);
459 }
460
461 // TODO(user) Opera creates a history-entry when creating an iframe
462 // programmatically as follows. Find a way which avoids this.
463
464 var iframeElm = goog.dom.getDomHelper(parentElm).createElement(
465 goog.dom.TagName.IFRAME);
466 iframeElm.id = iframeElm.name = iframeId;
467 if (opt_configureIframeCb) {
468 opt_configureIframeCb(iframeElm);
469 } else {
470 iframeElm.style.width = iframeElm.style.height = '100%';
471 }
472
473 this.cleanUpIncompleteConnection_();
474 this.peerWindowDeferred_ =
475 new goog.async.Deferred(undefined, this);
476 var peerUri = this.getPeerUri(opt_addCfgParam);
477 this.peerLoadHandler_.listenOnceWithScope(iframeElm, 'load',
478 this.peerWindowDeferred_.callback, false, this.peerWindowDeferred_);
479
480 if (goog.userAgent.GECKO || goog.userAgent.WEBKIT) {
481 // Appending the iframe in a timeout to avoid a weird fastback issue, which
482 // is present in Safari and Gecko.
483 window.setTimeout(
484 goog.bind(function() {
485 parentElm.appendChild(iframeElm);
486 iframeElm.src = peerUri.toString();
487 goog.log.info(goog.net.xpc.logger,
488 'peer iframe created (' + iframeId + ')');
489 }, this), 1);
490 } else {
491 iframeElm.src = peerUri.toString();
492 parentElm.appendChild(iframeElm);
493 goog.log.info(goog.net.xpc.logger,
494 'peer iframe created (' + iframeId + ')');
495 }
496
497 return /** @type {!HTMLIFrameElement} */ (iframeElm);
498};

501/**
502 * Clean up after any incomplete attempt to establish and connect to a peer
503 * iframe.
504 * @private
505 */
506goog.net.xpc.CrossPageChannel.prototype.cleanUpIncompleteConnection_ =
507 function() {
508 if (this.peerWindowDeferred_) {

511 }
512 this.deferredDeliveries_.length = 0;

517/**
518 * Returns the peer URI, with an optional URL parameter for configuring the peer
519 * window.
520 *
521 * @param {boolean=} opt_addCfgParam Whether to add the peer configuration as
522 * URL parameter (default: true).
523 * @return {!goog.Uri} The peer URI.
524 */
525goog.net.xpc.CrossPageChannel.prototype.getPeerUri = function(opt_addCfgParam) {
526 var peerUri = this.cfg_[goog.net.xpc.CfgFields.PEER_URI];
527 if (goog.isString(peerUri)) {
528 peerUri = this.cfg_[goog.net.xpc.CfgFields.PEER_URI] =
529 new goog.Uri(peerUri);
530 }
531
532 // Add the channel configuration used by the peer as URL parameter.
533 if (opt_addCfgParam !== false) {
534 peerUri.setParameterValue('xpc',
535 goog.json.serialize(
536 this.getPeerConfiguration()));
537 }
538
539 return peerUri;

543/**
544 * Initiates connecting the channel. When this method is called, all the
545 * information needed to connect the channel has to be available.
546 *
547 * @override
548 * @param {Function=} opt_connectCb The function to be called when the
549 * channel has been connected and is ready to be used.
550 */
551goog.net.xpc.CrossPageChannel.prototype.connect = function(opt_connectCb) {
552 this.connectCb_ = opt_connectCb || goog.nullFunction;
553
554 // If this channel was previously closed, transition back to the NOT_CONNECTED
555 // state to ensure that the connection can proceed (xpcDeliver blocks
556 // transport messages while the connection state is CLOSED).
557 if (this.state_ == goog.net.xpc.ChannelStates.CLOSED) {
558 this.state_ = goog.net.xpc.ChannelStates.NOT_CONNECTED;
559 }
560
561 // If we know of a peer window whose creation has been requested but is not
562 // complete, peerWindowDeferred_ will be non-null, and we should block on it.
563 if (this.peerWindowDeferred_) {
564 this.peerWindowDeferred_.addCallback(this.continueConnection_);
565 } else {
566 this.continueConnection_();
567 }
568};

571/**
572 * Continues the connection process once we're as sure as we can be that the
573 * peer iframe has been created.
574 * @private
575 */
576goog.net.xpc.CrossPageChannel.prototype.continueConnection_ = function() {
577 goog.log.info(goog.net.xpc.logger, 'continueConnection_()');
578 this.peerWindowDeferred_ = null;
579 if (this.cfg_[goog.net.xpc.CfgFields.IFRAME_ID]) {
580 this.iframeElement_ = this.domHelper_.getElement(
581 this.cfg_[goog.net.xpc.CfgFields.IFRAME_ID]);
582 }
583 if (this.iframeElement_) {
584 var winObj = this.iframeElement_.contentWindow;
585 // accessing the window using contentWindow doesn't work in safari
586 if (!winObj) {
587 winObj = window.frames[this.cfg_[goog.net.xpc.CfgFields.IFRAME_ID]];
588 }
589 this.setPeerWindowObject(winObj);
590 }
591
592 // if the peer window object has not been set at this point, we assume
593 // being in an iframe and the channel is meant to be to the containing page
594 if (!this.peerWindowObject_) {
595 // throw an error if we are in the top window (== not in an iframe)
596 if (window == window.top) {
597 throw Error(
598 "CrossPageChannel: Can't connect, peer window-object not set.");
599 } else {
600 this.setPeerWindowObject(window.parent);
601 }
602 }
603
604 this.createTransport_();
605
606 this.transport_.connect();
607
608 // Now we run any deferred deliveries collected while connection was deferred.
609 while (this.deferredDeliveries_.length > 0) {
610 this.deferredDeliveries_.shift()();
611 }
612};

615/**
616 * Closes the channel.
617 */
618goog.net.xpc.CrossPageChannel.prototype.close = function() {
619 this.cleanUpIncompleteConnection_();
620 this.state_ = goog.net.xpc.ChannelStates.CLOSED;
621 goog.dispose(this.transport_);
622 this.transport_ = null;
623 this.connectCb_ = null;
624 goog.dispose(this.connectionDelay_);
625 this.connectionDelay_ = null;
626 goog.log.info(goog.net.xpc.logger, 'Channel "' + this.name + '" closed');
627};

630/**
631 * Package-private.
632 * Called by the transport when the channel is connected.
633 * @param {number=} opt_delay Delay this number of milliseconds before calling
634 * the connection callback. Usage is discouraged, but can be used to paper
635 * over timing vulnerabilities when there is no alternative.
636 */
637goog.net.xpc.CrossPageChannel.prototype.notifyConnected = function(opt_delay) {
638 if (this.isConnected() ||
639 (this.connectionDelay_ && this.connectionDelay_.isActive())) {
640 return;
641 }
642 this.state_ = goog.net.xpc.ChannelStates.CONNECTED;
643 goog.log.info(goog.net.xpc.logger, 'Channel "' + this.name + '" connected');
644 goog.dispose(this.connectionDelay_);
645 if (goog.isDef(opt_delay)) {
646 this.connectionDelay_ =
647 new goog.async.Delay(this.connectCb_, opt_delay);
648 this.connectionDelay_.start();
649 } else {
650 this.connectionDelay_ = null;
651 this.connectCb_();
652 }
653};

666/** @override */
667goog.net.xpc.CrossPageChannel.prototype.send = function(serviceName, payload) {
668 if (!this.isConnected()) {
669 goog.log.error(goog.net.xpc.logger, 'Can\'t send. Channel not connected.');
670 return;
671 }
672 // Check if the peer is still around.
673 if (!this.isPeerAvailable()) {
674 goog.log.error(goog.net.xpc.logger, 'Peer has disappeared.');
675 this.close();
676 return;
677 }
678 if (goog.isObject(payload)) {
679 payload = goog.json.serialize(payload);
680 }
681
682 // Partially URL-encode the service name because some characters (: and |) are
683 // used as delimiters for some transports, and we want to allow those
684 // characters in service names.
685 this.transport_.send(this.escapeServiceName_(serviceName), payload);
686};

689/**
690 * Delivers messages to the appropriate service-handler. Named xpcDeliver to
691 * avoid name conflict with {@code deliver} function in superclass
692 * goog.messaging.AbstractChannel.
693 *
694 * @param {string} serviceName The name of the port.
695 * @param {string} payload The payload.
696 * @param {string=} opt_origin An optional origin for the message, where the
697 * underlying transport makes that available. If this is specified, and
698 * the PEER_HOSTNAME parameter was provided, they must match or the message
699 * will be rejected.
700 * @package
701 */
702goog.net.xpc.CrossPageChannel.prototype.xpcDeliver = function(
703 serviceName, payload, opt_origin) {
704
705 // This check covers the very rare (but producable) case where the inner frame
706 // becomes ready and sends its setup message while the outer frame is
707 // deferring its connect method waiting for the inner frame to be ready. The
708 // resulting deferral ensures the message will not be processed until the
709 // channel is fully configured.
710 if (this.peerWindowDeferred_) {
711 this.deferredDeliveries_.push(
712 goog.bind(this.xpcDeliver, this, serviceName, payload, opt_origin));
713 return;
714 }
715
716 // Check whether the origin of the message is as expected.
717 if (!this.isMessageOriginAcceptable_(opt_origin)) {
718 goog.log.warning(goog.net.xpc.logger,
719 'Message received from unapproved origin "' +
720 opt_origin + '" - rejected.');
721 return;
722 }
723
724 // If there is another channel still open, the native transport's global
725 // postMessage listener will still be active. This will mean that messages
726 // being sent to the now-closed channel will still be received and delivered,
727 // such as transport service traffic from its previous correspondent in the
728 // other frame. Ensure these messages don't cause exceptions.
729 // Example: http://b/12419303
730 if (this.isDisposed() || this.state_ == goog.net.xpc.ChannelStates.CLOSED) {
731 goog.log.warning(goog.net.xpc.logger,
732 'CrossPageChannel::xpcDeliver(): Channel closed.');
733 } else if (!serviceName ||
734 serviceName == goog.net.xpc.TRANSPORT_SERVICE_) {
735 this.transport_.transportServiceHandler(payload);
736 } else {
737 // only deliver messages if connected
738 if (this.isConnected()) {
739 this.deliver(this.unescapeServiceName_(serviceName), payload);
740 } else {
741 goog.log.info(goog.net.xpc.logger,
742 'CrossPageChannel::xpcDeliver(): Not connected.');
743 }
744 }
745};

748/**
749 * Escape the user-provided service name for sending across the channel. This
750 * URL-encodes certain special characters so they don't conflict with delimiters
751 * used by some of the transports, and adds a special prefix if the name
752 * conflicts with the reserved transport service name.
753 *
754 * This is the opposite of {@link #unescapeServiceName_}.
755 *
756 * @param {string} name The name of the service to escape.
757 * @return {string} The escaped service name.
758 * @private
759 */
760goog.net.xpc.CrossPageChannel.prototype.escapeServiceName_ = function(name) {
761 if (goog.net.xpc.CrossPageChannel.TRANSPORT_SERVICE_ESCAPE_RE_.test(name)) {
762 name = '%' + name;
763 }
764 return name.replace(/[%:|]/g, encodeURIComponent);
765};

768/**
769 * Unescape the escaped service name that was sent across the channel. This is
770 * the opposite of {@link #escapeServiceName_}.
771 *
772 * @param {string} name The name of the service to unescape.
773 * @return {string} The unescaped service name.
774 * @private
775 */
776goog.net.xpc.CrossPageChannel.prototype.unescapeServiceName_ = function(name) {
777 name = name.replace(/%[0-9a-f]{2}/gi, decodeURIComponent);
778 if (goog.net.xpc.CrossPageChannel.TRANSPORT_SERVICE_UNESCAPE_RE_.test(name)) {
779 return name.substring(1);
780 } else {
781 return name;
782 }

816/**
817 * Returns whether an incoming message with the given origin is acceptable.
818 * If an incoming request comes with a specified (non-empty) origin, and the
819 * PEER_HOSTNAME config parameter has also been provided, the two must match,
820 * or the message is unacceptable.
821 * @param {string=} opt_origin The origin associated with the incoming message.
822 * @return {boolean} Whether the message is acceptable.
823 * @private
824 */
825goog.net.xpc.CrossPageChannel.prototype.isMessageOriginAcceptable_ = function(
826 opt_origin) {
827 var peerHostname = this.cfg_[goog.net.xpc.CfgFields.PEER_HOSTNAME];
828 return goog.string.isEmptyOrWhitespace(goog.string.makeSafe(opt_origin)) ||
829 goog.string.isEmptyOrWhitespace(goog.string.makeSafe(peerHostname)) ||
830 opt_origin == this.cfg_[goog.net.xpc.CfgFields.PEER_HOSTNAME];
831};
