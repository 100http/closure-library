15/**
16 * @fileoverview Wrapper class for handling XmlHttpRequests.
17 *
18 * One off requests can be sent through goog.net.XhrIo.send() or an
19 * instance can be created to send multiple requests. Each request uses its
20 * own XmlHttpRequest object and handles clearing of the event callback to
21 * ensure no leaks.
22 *
23 * XhrIo is event based, it dispatches events when a request finishes, fails or
24 * succeeds or when the ready-state changes. The ready-state or timeout event
25 * fires first, followed by a generic completed event. Then the abort, error,
26 * or success event is fired as appropriate. Lastly, the ready event will fire
27 * to indicate that the object may be used to make another request.
28 *
29 * The error event may also be called before completed and
30 * ready-state-change if the XmlHttpRequest.open() or .send() methods throw.
31 *
32 * This class does not support multiple requests, queuing, or prioritization.
33 *
34 * Tested = IE6, FF1.5, Safari, Opera 8.5
35 *
36 * TODO(user): Error cases aren't playing nicely in Safari.
37 *
38 */
39

287/**
288 * Static send that creates a short lived instance of XhrIo to send the
289 * request.
290 * @see goog.net.XhrIo.cleanup
291 * @param {string|goog.Uri} url Uri to make request to.
292 * @param {?function(this:goog.net.XhrIo, ?)=} opt_callback Callback function
293 * for when request is complete.
294 * @param {string=} opt_method Send method, default: GET.
295 * @param {ArrayBuffer|ArrayBufferView|Blob|Document|FormData|string=}
296 * opt_content Body data.
297 * @param {Object|goog.structs.Map=} opt_headers Map of headers to add to the
298 * request.
299 * @param {number=} opt_timeoutInterval Number of milliseconds after which an
300 * incomplete request will be aborted; 0 means no timeout is set.
301 * @param {boolean=} opt_withCredentials Whether to send credentials with the
302 * request. Default to false. See {@link goog.net.XhrIo#setWithCredentials}.
303 * @return {!goog.net.XhrIo} The sent XhrIo.
304 */
305goog.net.XhrIo.send = function(url, opt_callback, opt_method, opt_content,
306 opt_headers, opt_timeoutInterval,
307 opt_withCredentials) {
308 var x = new goog.net.XhrIo();
309 goog.net.XhrIo.sendInstances_.push(x);
310 if (opt_callback) {
311 x.listen(goog.net.EventType.COMPLETE, opt_callback);
312 }
313 x.listenOnce(goog.net.EventType.READY, x.cleanupSend_);
314 if (opt_timeoutInterval) {
315 x.setTimeoutInterval(opt_timeoutInterval);
316 }
317 if (opt_withCredentials) {
318 x.setWithCredentials(opt_withCredentials);
319 }
320 x.send(url, opt_method, opt_content, opt_headers);
321 return x;
322};

325/**
326 * Disposes all non-disposed instances of goog.net.XhrIo created by
327 * {@link goog.net.XhrIo.send}.
328 * {@link goog.net.XhrIo.send} cleans up the goog.net.XhrIo instance
329 * it creates when the request completes or fails. However, if
330 * the request never completes, then the goog.net.XhrIo is not disposed.
331 * This can occur if the window is unloaded before the request completes.
332 * We could have {@link goog.net.XhrIo.send} return the goog.net.XhrIo
333 * it creates and make the client of {@link goog.net.XhrIo.send} be
334 * responsible for disposing it in this case. However, this makes things
335 * significantly more complicated for the client, and the whole point
336 * of {@link goog.net.XhrIo.send} is that it's simple and easy to use.
337 * Clients of {@link goog.net.XhrIo.send} should call
338 * {@link goog.net.XhrIo.cleanup} when doing final
339 * cleanup on window unload.
340 */
341goog.net.XhrIo.cleanup = function() {
342 var instances = goog.net.XhrIo.sendInstances_;
343 while (instances.length) {
344 instances.pop().dispose();
345 }
346};

444/**
445 * Instance send that actually uses XMLHttpRequest to make a server call.
446 * @param {string|goog.Uri} url Uri to make request to.
447 * @param {string=} opt_method Send method, default: GET.
448 * @param {ArrayBuffer|ArrayBufferView|Blob|Document|FormData|string=}
449 * opt_content Body data.
450 * @param {Object|goog.structs.Map=} opt_headers Map of headers to add to the
451 * request.
452 */
453goog.net.XhrIo.prototype.send = function(url, opt_method, opt_content,
454 opt_headers) {
455 if (this.xhr_) {
456 throw Error('[goog.net.XhrIo] Object is active with another request=' +
457 this.lastUri_ + '; newUri=' + url);
458 }
459
460 var method = opt_method ? opt_method.toUpperCase() : 'GET';
461
462 this.lastUri_ = url;
463 this.lastError_ = '';
464 this.lastErrorCode_ = goog.net.ErrorCode.NO_ERROR;
465 this.lastMethod_ = method;
466 this.errorDispatched_ = false;
467 this.active_ = true;
468
469 // Use the factory to create the XHR object and options
470 this.xhr_ = this.createXhr();
471 this.xhrOptions_ = this.xmlHttpFactory_ ?
472 this.xmlHttpFactory_.getOptions() : goog.net.XmlHttp.getOptions();
473
474 // Set up the onreadystatechange callback
475 this.xhr_.onreadystatechange = goog.bind(this.onReadyStateChange_, this);
476
477 /**
478 * Try to open the XMLHttpRequest (always async), if an error occurs here it
479 * is generally permission denied
480 * @preserveTry
481 */
482 try {
483 goog.log.fine(this.logger_, this.formatMsg_('Opening Xhr'));
484 this.inOpen_ = true;
485 this.xhr_.open(method, String(url), true); // Always async!
486 this.inOpen_ = false;
487 } catch (err) {
488 goog.log.fine(this.logger_,
489 this.formatMsg_('Error opening Xhr: ' + err.message));
490 this.error_(goog.net.ErrorCode.EXCEPTION, err);
491 return;
492 }
493
494 // We can't use null since this won't allow requests with form data to have a
495 // content length specified which will cause some proxies to return a 411
496 // error.
497 var content = opt_content || '';
498
499 var headers = this.headers.clone();
500
501 // Add headers specific to this request
502 if (opt_headers) {
503 goog.structs.forEach(opt_headers, function(value, key) {
504 headers.set(key, value);
505 });
506 }
507
508 // Find whether a content type header is set, ignoring case.
509 // HTTP header names are case-insensitive. See:
510 // http://www.w3.org/Protocols/rfc2616/rfc2616-sec4.html#sec4.2
511 var contentTypeKey = goog.array.find(headers.getKeys(),
512 goog.net.XhrIo.isContentTypeHeader_);
513
514 var contentIsFormData = (goog.global['FormData'] &&
515 (content instanceof goog.global['FormData']));
516 if (goog.array.contains(goog.net.XhrIo.METHODS_WITH_FORM_DATA, method) &&
517 !contentTypeKey && !contentIsFormData) {
518 // For requests typically with form data, default to the url-encoded form
519 // content type unless this is a FormData request. For FormData,
520 // the browser will automatically add a multipart/form-data content type
521 // with an appropriate multipart boundary.
522 headers.set(goog.net.XhrIo.CONTENT_TYPE_HEADER,
523 goog.net.XhrIo.FORM_CONTENT_TYPE);
524 }
525
526 // Add the headers to the Xhr object
527 headers.forEach(function(value, key) {
528 this.xhr_.setRequestHeader(key, value);
529 }, this);
530
531 if (this.responseType_) {
532 this.xhr_.responseType = this.responseType_;
533 }
534
535 if (goog.object.containsKey(this.xhr_, 'withCredentials')) {
536 this.xhr_.withCredentials = this.withCredentials_;
537 }
538
539 /**
540 * Try to send the request, or other wise report an error (404 not found).
541 * @preserveTry
542 */
543 try {
544 this.cleanUpTimeoutTimer_(); // Paranoid, should never be running.
545 if (this.timeoutInterval_ > 0) {
546 this.useXhr2Timeout_ = goog.net.XhrIo.shouldUseXhr2Timeout_(this.xhr_);
547 goog.log.fine(this.logger_, this.formatMsg_('Will abort after ' +
548 this.timeoutInterval_ + 'ms if incomplete, xhr2 ' +
549 this.useXhr2Timeout_));
550 if (this.useXhr2Timeout_) {
551 this.xhr_[goog.net.XhrIo.XHR2_TIMEOUT_] = this.timeoutInterval_;
552 this.xhr_[goog.net.XhrIo.XHR2_ON_TIMEOUT_] =
553 goog.bind(this.timeout_, this);
554 } else {
555 this.timeoutId_ = goog.Timer.callOnce(this.timeout_,
556 this.timeoutInterval_, this);
557 }
558 }
559 goog.log.fine(this.logger_, this.formatMsg_('Sending request'));
560 this.inSend_ = true;
561 this.xhr_.send(content);
562 this.inSend_ = false;
563
564 } catch (err) {
565 goog.log.fine(this.logger_, this.formatMsg_('Send error: ' + err.message));
566 this.error_(goog.net.ErrorCode.EXCEPTION, err);
567 }
568};

571/**
572 * Determines if the argument is an XMLHttpRequest that supports the level 2
573 * timeout value and event.
574 *
575 * Currently, FF 21.0 OS X has the fields but won't actually call the timeout
576 * handler. Perhaps the confusion in the bug referenced below hasn't
577 * entirely been resolved.
578 *
579 * @see http://www.w3.org/TR/XMLHttpRequest/#the-timeout-attribute
580 * @see https://bugzilla.mozilla.org/show_bug.cgi?id=525816
581 *
582 * @param {!goog.net.XhrLike.OrNative} xhr The request.
583 * @return {boolean} True if the request supports level 2 timeout.
584 * @private
585 */
586goog.net.XhrIo.shouldUseXhr2Timeout_ = function(xhr) {
587 return goog.userAgent.IE &&
588 goog.userAgent.isVersionOrHigher(9) &&
589 goog.isNumber(xhr[goog.net.XhrIo.XHR2_TIMEOUT_]) &&
590 goog.isDef(xhr[goog.net.XhrIo.XHR2_ON_TIMEOUT_]);
591};

617/**
618 * The request didn't complete after {@link goog.net.XhrIo#timeoutInterval_}
619 * milliseconds; raises a {@link goog.net.EventType.TIMEOUT} event and aborts
620 * the request.
621 * @private
622 */
623goog.net.XhrIo.prototype.timeout_ = function() {
624 if (typeof goog == 'undefined') {
625 // If goog is undefined then the callback has occurred as the application
626 // is unloading and will error. Thus we let it silently fail.
627 } else if (this.xhr_) {
628 this.lastError_ = 'Timed out after ' + this.timeoutInterval_ +
629 'ms, aborting';
630 this.lastErrorCode_ = goog.net.ErrorCode.TIMEOUT;
631 goog.log.fine(this.logger_, this.formatMsg_(this.lastError_));
632 this.dispatchEvent(goog.net.EventType.TIMEOUT);
633 this.abort(goog.net.ErrorCode.TIMEOUT);
634 }
635};

638/**
639 * Something errorred, so inactivate, fire error callback and clean up
640 * @param {goog.net.ErrorCode} errorCode The error code.
641 * @param {Error} err The error object.
642 * @private
643 */
644goog.net.XhrIo.prototype.error_ = function(errorCode, err) {
645 this.active_ = false;
646 if (this.xhr_) {
647 this.inAbort_ = true;
648 this.xhr_.abort(); // Ensures XHR isn't hung (FF)
649 this.inAbort_ = false;
650 }
651 this.lastError_ = err;
652 this.lastErrorCode_ = errorCode;
653 this.dispatchErrors_();
654 this.cleanUpXhr_();
655};

658/**
659 * Dispatches COMPLETE and ERROR in case of an error. This ensures that we do
660 * not dispatch multiple error events.
661 * @private
662 */
663goog.net.XhrIo.prototype.dispatchErrors_ = function() {
664 if (!this.errorDispatched_) {
665 this.errorDispatched_ = true;
666 this.dispatchEvent(goog.net.EventType.COMPLETE);
667 this.dispatchEvent(goog.net.EventType.ERROR);
668 }

672/**
673 * Abort the current XMLHttpRequest
674 * @param {goog.net.ErrorCode=} opt_failureCode Optional error code to use -
675 * defaults to ABORT.
676 */
677goog.net.XhrIo.prototype.abort = function(opt_failureCode) {
678 if (this.xhr_ && this.active_) {
679 goog.log.fine(this.logger_, this.formatMsg_('Aborting'));
680 this.active_ = false;
681 this.inAbort_ = true;
682 this.xhr_.abort();
683 this.inAbort_ = false;
684 this.lastErrorCode_ = opt_failureCode || goog.net.ErrorCode.ABORT;
685 this.dispatchEvent(goog.net.EventType.COMPLETE);
686 this.dispatchEvent(goog.net.EventType.ABORT);
687 this.cleanUpXhr_();
688 }
689};

692/**
693 * Nullifies all callbacks to reduce risks of leaks.
694 * @override
695 * @protected
696 */
697goog.net.XhrIo.prototype.disposeInternal = function() {
698 if (this.xhr_) {
699 // We explicitly do not call xhr_.abort() unless active_ is still true.
700 // This is to avoid unnecessarily aborting a successful request when
701 // dispose() is called in a callback triggered by a complete response, but
702 // in which browser cleanup has not yet finished.
703 // (See http://b/issue?id=1684217.)
704 if (this.active_) {
705 this.active_ = false;
706 this.inAbort_ = true;
707 this.xhr_.abort();
708 this.inAbort_ = false;
709 }
710 this.cleanUpXhr_(true);
711 }
712
713 goog.net.XhrIo.base(this, 'disposeInternal');

717/**
718 * Internal handler for the XHR object's readystatechange event. This method
719 * checks the status and the readystate and fires the correct callbacks.
720 * If the request has ended, the handlers are cleaned up and the XHR object is
721 * nullified.
722 * @private
723 */
724goog.net.XhrIo.prototype.onReadyStateChange_ = function() {
725 if (this.isDisposed()) {
726 // This method is the target of an untracked goog.Timer.callOnce().
727 return;
728 }
729 if (!this.inOpen_ && !this.inSend_ && !this.inAbort_) {
730 // Were not being called from within a call to this.xhr_.send
731 // this.xhr_.abort, or this.xhr_.open, so this is an entry point
732 this.onReadyStateChangeEntryPoint_();
733 } else {
734 this.onReadyStateChangeHelper_();
735 }
736};

752/**
753 * Helper for {@link #onReadyStateChange_}. This is used so that
754 * entry point calls to {@link #onReadyStateChange_} can be routed through
755 * {@link #onReadyStateChangeEntryPoint_}.
756 * @private
757 */
758goog.net.XhrIo.prototype.onReadyStateChangeHelper_ = function() {
759 if (!this.active_) {
760 // can get called inside abort call
761 return;
762 }
763
764 if (typeof goog == 'undefined') {
765 // NOTE(user): If goog is undefined then the callback has occurred as the
766 // application is unloading and will error. Thus we let it silently fail.
767
768 } else if (
769 this.xhrOptions_[goog.net.XmlHttp.OptionType.LOCAL_REQUEST_ERROR] &&
770 this.getReadyState() == goog.net.XmlHttp.ReadyState.COMPLETE &&
771 this.getStatus() == 2) {
772 // NOTE(user): In IE if send() errors on a *local* request the readystate
773 // is still changed to COMPLETE. We need to ignore it and allow the
774 // try/catch around send() to pick up the error.
775 goog.log.fine(this.logger_, this.formatMsg_(
776 'Local request error detected and ignored'));
777
778 } else {
779
780 // In IE when the response has been cached we sometimes get the callback
781 // from inside the send call and this usually breaks code that assumes that
782 // XhrIo is asynchronous. If that is the case we delay the callback
783 // using a timer.
784 if (this.inSend_ &&
785 this.getReadyState() == goog.net.XmlHttp.ReadyState.COMPLETE) {
786 goog.Timer.callOnce(this.onReadyStateChange_, 0, this);
787 return;
788 }
789
790 this.dispatchEvent(goog.net.EventType.READY_STATE_CHANGE);
791
792 // readyState indicates the transfer has finished
793 if (this.isComplete()) {
794 goog.log.fine(this.logger_, this.formatMsg_('Request complete'));
795
796 this.active_ = false;
797
798 try {
799 // Call the specific callbacks for success or failure. Only call the
800 // success if the status is 200 (HTTP_OK) or 304 (HTTP_CACHED)
801 if (this.isSuccess()) {
802 this.dispatchEvent(goog.net.EventType.COMPLETE);
803 this.dispatchEvent(goog.net.EventType.SUCCESS);
804 } else {
805 this.lastErrorCode_ = goog.net.ErrorCode.HTTP_ERROR;
806 this.lastError_ =
807 this.getStatusText() + ' [' + this.getStatus() + ']';
808 this.dispatchErrors_();
809 }
810 } finally {
811 this.cleanUpXhr_();
812 }
813 }
814 }
815};

818/**
819 * Remove the listener to protect against leaks, and nullify the XMLHttpRequest
820 * object.
821 * @param {boolean=} opt_fromDispose If this is from the dispose (don't want to
822 * fire any events).
823 * @private
824 */
825goog.net.XhrIo.prototype.cleanUpXhr_ = function(opt_fromDispose) {
826 if (this.xhr_) {
827 // Cancel any pending timeout event handler.
828 this.cleanUpTimeoutTimer_();
829
830 // Save reference so we can mark it as closed after the READY event. The
831 // READY event may trigger another request, thus we must nullify this.xhr_
832 var xhr = this.xhr_;
833 var clearedOnReadyStateChange =
834 this.xhrOptions_[goog.net.XmlHttp.OptionType.USE_NULL_FUNCTION] ?
835 goog.nullFunction : null;
836 this.xhr_ = null;
837 this.xhrOptions_ = null;
838
839 if (!opt_fromDispose) {
840 this.dispatchEvent(goog.net.EventType.READY);
841 }
842
843 try {
844 // NOTE(user): Not nullifying in FireFox can still leak if the callbacks
845 // are defined in the same scope as the instance of XhrIo. But, IE doesn't
846 // allow you to set the onreadystatechange to NULL so nullFunction is
847 // used.
848 xhr.onreadystatechange = clearedOnReadyStateChange;
849 } catch (e) {
850 // This seems to occur with a Gears HTTP request. Delayed the setting of
851 // this onreadystatechange until after READY is sent out and catching the
852 // error to see if we can track down the problem.
853 goog.log.error(this.logger_,
854 'Problem encountered resetting onreadystatechange: ' + e.message);
855 }
856 }
857};

1068/**
1069 * Get the response as the type specificed by {@link #setResponseType}. At time
1070 * of writing, this is only directly supported in very recent versions of WebKit
1071 * (10.0.612.1 dev and later). If the field is not supported directly, we will
1072 * try to emulate it.
1073 *
1074 * Emulating the response means following the rules laid out at
1075 * http://www.w3.org/TR/XMLHttpRequest/#the-response-attribute
1076 *
1077 * On browsers with no support for this (Chrome < 10, Firefox < 4, etc), only
1078 * response types of DEFAULT or TEXT may be used, and the response returned will
1079 * be the text response.
1080 *
1081 * On browsers with Mozilla's draft support for array buffers (Firefox 4, 5),
1082 * only response types of DEFAULT, TEXT, and ARRAY_BUFFER may be used, and the
1083 * response returned will be either the text response or the Mozilla
1084 * implementation of the array buffer response.
1085 *
1086 * On browsers will full support, any valid response type supported by the
1087 * browser may be used, and the response provided by the browser will be
1088 * returned.
1089 *
1090 * @return {*} The response.
1091 */
1092goog.net.XhrIo.prototype.getResponse = function() {
1093 /** @preserveTry */
1094 try {
1095 if (!this.xhr_) {
1096 return null;
1097 }
1098 if ('response' in this.xhr_) {
1099 return this.xhr_.response;
1100 }
1101 switch (this.responseType_) {
1102 case goog.net.XhrIo.ResponseType.DEFAULT:
1103 case goog.net.XhrIo.ResponseType.TEXT:
1104 return this.xhr_.responseText;
1105 // DOCUMENT and BLOB don't need to be handled here because they are
1106 // introduced in the same spec that adds the .response field, and would
1107 // have been caught above.
1108 // ARRAY_BUFFER needs an implementation for Firefox 4, where it was
1109 // implemented using a draft spec rather than the final spec.
1110 case goog.net.XhrIo.ResponseType.ARRAY_BUFFER:
1111 if ('mozResponseArrayBuffer' in this.xhr_) {
1112 return this.xhr_.mozResponseArrayBuffer;
1113 }
1114 }
1115 // Fell through to a response type that is not supported on this browser.
1116 goog.log.error(this.logger_,
1117 'Response type ' + this.responseType_ + ' is not ' +
1118 'supported on this browser');
1119 return null;
1120 } catch (e) {
1121 goog.log.fine(this.logger_, 'Can not get response: ' + e.message);
1122 return null;
1123 }

1152/**
1153 * Returns all response headers as a key-value map.
1154 * Multiple values for the same header key can be combined into one,
1155 * separated by a comma and a space.
1156 * Note that the native getResponseHeader method for retrieving a single header
1157 * does a case insensitive match on the header name. This method does not
1158 * include any case normalization logic, it will just return a key-value
1159 * representation of the headers.
1160 * See: http://www.w3.org/TR/XMLHttpRequest/#the-getresponseheader()-method
1161 * @return {!Object<string, string>} An object with the header keys as keys
1162 * and header values as values.
1163 */
1164goog.net.XhrIo.prototype.getResponseHeaders = function() {
1165 var headersObject = {};
1166 var headersArray = this.getAllResponseHeaders().split('\r\n');
1167 for (var i = 0; i < headersArray.length; i++) {
1168 if (goog.string.isEmptyOrWhitespace(headersArray[i])) {
1169 continue;
1170 }
1171 var keyValue = goog.string.splitLimit(headersArray[i], ': ', 2);
1172 if (headersObject[keyValue[0]]) {
1173 headersObject[keyValue[0]] += ', ' + keyValue[1];
1174 } else {
1175 headersObject[keyValue[0]] = keyValue[1];
1176 }
1177 }
1178 return headersObject;
1179};
