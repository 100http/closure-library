17 * Implements 4x4 matrices and their related functions which are
18 * compatible with WebGL. The API is structured to avoid unnecessary memory
19 * allocations. The last parameter will typically be the output vector and
20 * an object can be both an input and output parameter to all methods except
21 * where noted. Matrix operations follow the mathematical form when multiplying
22 * vectors as follows: resultVec = matrix * vec.
23 *
24 */

628 *

637goog.vec.Matrix4.multMat = function(mat0, mat1, resultMat) {
638 var a00 = mat0[0], a10 = mat0[1], a20 = mat0[2], a30 = mat0[3];
639 var a01 = mat0[4], a11 = mat0[5], a21 = mat0[6], a31 = mat0[7];

643 var b00 = mat1[0], b10 = mat1[1], b20 = mat1[2], b30 = mat1[3];
644 var b01 = mat1[4], b11 = mat1[5], b21 = mat1[6], b31 = mat1[7];

751/**
752 * Computes the inverse of mat storing the result into resultMat. If the
753 * inverse is defined, this function returns true, false otherwise.
754 *
755 * @param {goog.vec.ArrayType} mat The matrix to invert.
756 * @param {goog.vec.ArrayType} resultMat The matrix to receive
757 * the result (may be mat).
758 * @return {boolean} True if the inverse is defined. If false is returned,
759 * resultMat is not modified.
760 */
761goog.vec.Matrix4.invert = function(mat, resultMat) {
762 var m00 = mat[0], m10 = mat[1], m20 = mat[2], m30 = mat[3];
763 var m01 = mat[4], m11 = mat[5], m21 = mat[6], m31 = mat[7];
764 var m02 = mat[8], m12 = mat[9], m22 = mat[10], m32 = mat[11];
765 var m03 = mat[12], m13 = mat[13], m23 = mat[14], m33 = mat[15];
766
767 var a0 = m00 * m11 - m10 * m01;
768 var a1 = m00 * m21 - m20 * m01;
769 var a2 = m00 * m31 - m30 * m01;
770 var a3 = m10 * m21 - m20 * m11;
771 var a4 = m10 * m31 - m30 * m11;
772 var a5 = m20 * m31 - m30 * m21;
773 var b0 = m02 * m13 - m12 * m03;
774 var b1 = m02 * m23 - m22 * m03;
775 var b2 = m02 * m33 - m32 * m03;
776 var b3 = m12 * m23 - m22 * m13;
777 var b4 = m12 * m33 - m32 * m13;
778 var b5 = m22 * m33 - m32 * m23;
779
780 var det = a0 * b5 - a1 * b4 + a2 * b3 + a3 * b2 - a4 * b1 + a5 * b0;
781 if (det == 0) {
782 return false;
783 }
784
785 var idet = 1.0 / det;
786 resultMat[0] = (m11 * b5 - m21 * b4 + m31 * b3) * idet;
787 resultMat[1] = (-m10 * b5 + m20 * b4 - m30 * b3) * idet;
788 resultMat[2] = (m13 * a5 - m23 * a4 + m33 * a3) * idet;
789 resultMat[3] = (-m12 * a5 + m22 * a4 - m32 * a3) * idet;
790 resultMat[4] = (-m01 * b5 + m21 * b2 - m31 * b1) * idet;
791 resultMat[5] = (m00 * b5 - m20 * b2 + m30 * b1) * idet;
792 resultMat[6] = (-m03 * a5 + m23 * a2 - m33 * a1) * idet;
793 resultMat[7] = (m02 * a5 - m22 * a2 + m32 * a1) * idet;
794 resultMat[8] = (m01 * b4 - m11 * b2 + m31 * b0) * idet;
795 resultMat[9] = (-m00 * b4 + m10 * b2 - m30 * b0) * idet;
796 resultMat[10] = (m03 * a4 - m13 * a2 + m33 * a0) * idet;
797 resultMat[11] = (-m02 * a4 + m12 * a2 - m32 * a0) * idet;
798 resultMat[12] = (-m01 * b3 + m11 * b1 - m21 * b0) * idet;
799 resultMat[13] = (m00 * b3 - m10 * b1 + m20 * b0) * idet;
800 resultMat[14] = (-m03 * a3 + m13 * a1 - m23 * a0) * idet;
801 resultMat[15] = (m02 * a3 - m12 * a1 + m22 * a0) * idet;
802 return true;
803};

1122/**
1123 * Updates a matrix representing the modelview matrix of a camera so that
1124 * the camera is 'looking at' the given center point.
1125 * @param {goog.vec.ArrayType} viewMatrix The matrix.
1126 * @param {goog.vec.ArrayType} eyePt The position of the eye point
1127 * (camera origin).
1128 * @param {goog.vec.ArrayType} centerPt The point to aim the camera
1129 * at.
1130 * @param {goog.vec.ArrayType} worldUpVec The vector that
1131 * identifies the up direction for the camera.
1132 */
1133goog.vec.Matrix4.lookAt = function(
1134 viewMatrix, eyePt, centerPt, worldUpVec) {
1135 // Compute the direction vector from the eye point to the center point and
1136 // normalize.
1137 var fwdVec = goog.vec.Matrix4.tmpVec4_[0];
1138 goog.vec.Vec3.subtract(centerPt, eyePt, fwdVec);
1139 goog.vec.Vec3.normalize(fwdVec, fwdVec);
1140 fwdVec[3] = 0;
1141
1142 // Compute the side vector from the forward vector and the input up vector.
1143 var sideVec = goog.vec.Matrix4.tmpVec4_[1];
1144 goog.vec.Vec3.cross(fwdVec, worldUpVec, sideVec);
1145 goog.vec.Vec3.normalize(sideVec, sideVec);
1146 sideVec[3] = 0;
1147
1148 // Now the up vector to form the orthonormal basis.
1149 var upVec = goog.vec.Matrix4.tmpVec4_[2];
1150 goog.vec.Vec3.cross(sideVec, fwdVec, upVec);
1151 goog.vec.Vec3.normalize(upVec, upVec);
1152 upVec[3] = 0;
1153
1154 // Update the view matrix with the new orthonormal basis and position the
1155 // camera at the given eye point.
1156 goog.vec.Vec3.negate(fwdVec, fwdVec);
1157 goog.vec.Matrix4.setRow(viewMatrix, 0, sideVec);
1158 goog.vec.Matrix4.setRow(viewMatrix, 1, upVec);
1159 goog.vec.Matrix4.setRow(viewMatrix, 2, fwdVec);
1160 goog.vec.Matrix4.setRowValues(viewMatrix, 3, 0, 0, 0, 1);
1161 goog.vec.Matrix4.applyTranslate(
1162 viewMatrix, -eyePt[0], -eyePt[1], -eyePt[2]);
1163};

1171 * @param {goog.vec.ArrayType} viewMatrix The matrix.
