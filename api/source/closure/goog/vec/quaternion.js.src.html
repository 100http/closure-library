252/**
253 * Computes the inverse of the quaternion in quat, storing the result into
254 * resultQuat.
255 *
256 * If the quaternion is already normalized, goog.vec.Quaternion.conjugate
257 * is faster than this function and produces the same result.
258 *
259 * @param {!goog.vec.Quaternion.AnyType} quat The quaternion to invert.
260 * @param {!goog.vec.Quaternion.AnyType} resultQuat The quaternion to receive
261 * the result.
262 * @return {!goog.vec.Quaternion.AnyType} Return resultQuat so that
263 * operations can be chained together.
264 */
265goog.vec.Quaternion.invert = function(quat, resultQuat) {
266 var a0 = quat[0], a1 = quat[1], a2 = quat[2], a3 = quat[3];
267 var dot = a0 * a0 + a1 * a1 + a2 * a2 + a3 * a3;
268 var invDot = dot ? 1.0 / dot : 0;
269
270 resultQuat[0] = -a0 * invDot;
271 resultQuat[1] = -a1 * invDot;
272 resultQuat[2] = -a2 * invDot;
273 resultQuat[3] = a3 * invDot;
274 return resultQuat;
275};

300/**
301 * Concatenates the two quaternions storing the result into resultQuat.
302 *
303 * @param {!goog.vec.Quaternion.AnyType} quat0 The first quaternion.
304 * @param {!goog.vec.Quaternion.AnyType} quat1 The second quaternion.
305 * @param {!goog.vec.Quaternion.AnyType} resultQuat The quaternion to
306 * receive the result.
307 * @return {!goog.vec.Quaternion.AnyType} Return resultQuat so that
308 * operations can be chained together.
309 */
310goog.vec.Quaternion.concat = function(quat0, quat1, resultQuat) {
311 var x0 = quat0[0], y0 = quat0[1], z0 = quat0[2], w0 = quat0[3];
312 var x1 = quat1[0], y1 = quat1[1], z1 = quat1[2], w1 = quat1[3];
313 resultQuat[0] = w0 * x1 + x0 * w1 + y0 * z1 - z0 * y1;
314 resultQuat[1] = w0 * y1 - x0 * z1 + y0 * w1 + z0 * x1;
315 resultQuat[2] = w0 * z1 + x0 * y1 - y0 * x1 + z0 * w1;
316 resultQuat[3] = w0 * w1 - x0 * x1 - y0 * y1 - z0 * z1;
317 return resultQuat;
318};

337/**
338 * Generates a unit quaternion from the given angle-axis rotation pair.
339 * The rotation axis is not required to be a unit vector, but should
340 * have non-zero length. The angle should be specified in radians.
341 *
342 * @param {number} angle The angle (in radians) to rotate about the axis.
343 * @param {!goog.vec.Quaternion.AnyType} axis Unit vector specifying the
344 * axis of rotation.
345 * @param {!goog.vec.Quaternion.AnyType} quat Unit quaternion to store the
346 * result.
347 * @return {!goog.vec.Quaternion.AnyType} Return quat so that
348 * operations can be chained together.
349 */
350goog.vec.Quaternion.fromAngleAxis = function(angle, axis, quat) {
351 // Normalize the axis of rotation.
352 goog.vec.Vec3.normalize(axis, axis);
353
354 var halfAngle = 0.5 * angle;
355 var sin = Math.sin(halfAngle);
356 goog.vec.Quaternion.setFromValues(
357 quat, sin * axis[0], sin * axis[1], sin * axis[2], Math.cos(halfAngle));
358
359 // Normalize the resulting quaternion.
360 goog.vec.Quaternion.normalize(quat, quat);
361 return quat;
362};

365/**
366 * Generates an angle-axis rotation pair from a unit quaternion.
367 * The quaternion is assumed to be of unit length. The calculated
368 * values are returned via the passed 'axis' object and the 'angle'
369 * number returned by the function itself. The returned rotation axis
370 * is a non-zero length unit vector, and the returned angle is in
371 * radians in the range of [-PI, +PI].
372 *
373 * @param {!goog.vec.Quaternion.AnyType} quat Unit quaternion to convert.
374 * @param {!goog.vec.Quaternion.AnyType} axis Vector to store the returned
375 * rotation axis.
376 * @return {number} angle Angle (in radians) to rotate about 'axis'.
377 * The range of the returned angle is [-PI, +PI].
378 */
379goog.vec.Quaternion.toAngleAxis = function(quat, axis) {
380 var angle = 2 * Math.acos(quat[3]);
381 var magnitude = Math.min(Math.max(1 - quat[3] * quat[3], 0), 1);
382 if (magnitude < goog.vec.EPSILON) {
383 // This is nearly an identity rotation, so just use a fixed +X axis.
384 goog.vec.Vec3.setFromValues(axis, 1, 0, 0);
385 } else {
386 // Compute the proper rotation axis.
387 goog.vec.Vec3.setFromValues(axis, quat[0], quat[1], quat[2]);
388 // Make sure the rotation axis is of unit length.
389 goog.vec.Vec3.normalize(axis, axis);
390 }
391 // Adjust the range of the returned angle to [-PI, +PI].
392 if (angle > Math.PI) {
393 angle -= 2 * Math.PI;
394 }
395 return angle;
396};

399/**
400 * Generates the quaternion from the given 3x3 rotation matrix.
401 *
402 * Perf: http://jsperf.com/conversion-of-3x3-matrix-to-quaternion
403 * http://jsperf.com/goog-vec-fromrotationmatrix3-a
404 *
405 * @param {!goog.vec.AnyType} matrix The source matrix.
406 * @param {!goog.vec.Quaternion.AnyType} quat The resulting quaternion.
407 * @return {!goog.vec.Quaternion.AnyType} Return quat so that
408 * operations can be chained together.
409 */
410goog.vec.Quaternion.fromRotationMatrix3 = function(matrix, quat) {
411 // Algorithm in Ken Shoemake's article in 1987 SIGGRAPH course notes
412 // article "Quaternion Calculus and Fast Animation".
413 var fTrace = matrix[0] + matrix[4] + matrix[8];
414 var fRoot;
415
416 if (fTrace > 0.0) {
417 // |w| > 1/2, may as well choose w > 1/2
418 fRoot = Math.sqrt(fTrace + 1.0); // 2w
419 quat[3] = 0.5 * fRoot;
420 fRoot = 0.5 / fRoot; // 1 / (4w)
421 quat[0] = (matrix[5] - matrix[7]) * fRoot;
422 quat[1] = (matrix[6] - matrix[2]) * fRoot;
423 quat[2] = (matrix[1] - matrix[3]) * fRoot;
424 } else {
425 // |w| <= 1/2
426 var i = 0;
427 if (matrix[4] > matrix[0])
428 i = 1;
429 if (matrix[8] > matrix[i * 3 + i])
430 i = 2;
431 var j = (i + 1) % 3;
432 var k = (i + 2) % 3;
433
434 fRoot = Math.sqrt(matrix[i * 3 + i] -
435 matrix[j * 3 + j] -
436 matrix[k * 3 + k] + 1.0);
437 quat[i] = 0.5 * fRoot;
438 fRoot = 0.5 / fRoot;
439 quat[3] = (matrix[j * 3 + k] - matrix[k * 3 + j]) * fRoot;
440 quat[j] = (matrix[j * 3 + i] + matrix[i * 3 + j]) * fRoot;
441 quat[k] = (matrix[k * 3 + i] + matrix[i * 3 + k]) * fRoot;
442
443 // Flip all signs if w is negative.
444 if (quat[3] < 0) {
445 quat[0] = -quat[0];
446 quat[1] = -quat[1];
447 quat[2] = -quat[2];
448 quat[3] = -quat[3];
449 }
450 }
451 return quat;
452};

455/**
456 * Generates the quaternion from the given 4x4 rotation matrix.
457 *
458 * Perf: http://jsperf.com/goog-vec-fromrotationmatrix4
459 *
460 * Implementation is the same as fromRotationMatrix3 but using indices from
461 * the top left 3x3 in a 4x4 matrix.
462 *
463 * @param {!goog.vec.AnyType} matrix The source matrix.
464 * @param {!goog.vec.Quaternion.AnyType} quat The resulting quaternion.
465 * @return {!goog.vec.Quaternion.AnyType} Return quat so that
466 * operations can be chained together.
467 */
468goog.vec.Quaternion.fromRotationMatrix4 = function(matrix, quat) {
469 var fTrace = matrix[0] + matrix[5] + matrix[10];
470 var fRoot;
471
472 if (fTrace > 0.0) {
473 // |w| > 1/2, may as well choose w > 1/2
474 fRoot = Math.sqrt(fTrace + 1.0); // 2w
475 quat[3] = 0.5 * fRoot;
476 fRoot = 0.5 / fRoot; // 1 / (4w)
477 quat[0] = (matrix[6] - matrix[9]) * fRoot;
478 quat[1] = (matrix[8] - matrix[2]) * fRoot;
479 quat[2] = (matrix[1] - matrix[4]) * fRoot;
480 } else {
481 // |w| <= 1/2
482 var i = 0;
483 if (matrix[8] > matrix[0])
484 i = 1;
485 if (matrix[10] > matrix[i * 4 + i])
486 i = 2;
487 var j = (i + 1) % 3;
488 var k = (i + 2) % 3;
489
490 fRoot = Math.sqrt(matrix[i * 4 + i] -
491 matrix[j * 4 + j] -
492 matrix[k * 4 + k] + 1.0);
493 quat[i] = 0.5 * fRoot;
494 fRoot = 0.5 / fRoot;
495 quat[3] = (matrix[j * 4 + k] - matrix[k * 4 + j]) * fRoot;
496 quat[j] = (matrix[j * 4 + i] + matrix[i * 4 + j]) * fRoot;
497 quat[k] = (matrix[k * 4 + i] + matrix[i * 4 + k]) * fRoot;
498
499 // Flip all signs if w is negative.
500 if (quat[3] < 0) {
501 quat[0] = -quat[0];
502 quat[1] = -quat[1];
503 quat[2] = -quat[2];
504 quat[3] = -quat[3];
505 }
506 }
507 return quat;
508};

511/**
512 * Generates the 3x3 rotation matrix from the given quaternion.
513 *
514 * @param {!goog.vec.Quaternion.AnyType} quat The source quaternion.
515 * @param {!goog.vec.AnyType} matrix The resulting matrix.
516 * @return {!goog.vec.AnyType} Return resulting matrix so that
517 * operations can be chained together.
518 */
519goog.vec.Quaternion.toRotationMatrix3 = function(quat, matrix) {
520 var x = quat[0], y = quat[1], z = quat[2], w = quat[3];
521 var x2 = 2 * x, y2 = 2 * y, z2 = 2 * z;
522 var wx = x2 * w;
523 var wy = y2 * w;
524 var wz = z2 * w;
525 var xx = x2 * x;
526 var xy = y2 * x;
527 var xz = z2 * x;
528 var yy = y2 * y;
529 var yz = z2 * y;
530 var zz = z2 * z;
531
532 matrix[0] = 1 - (yy + zz);
533 matrix[1] = xy + wz;
534 matrix[2] = xz - wy;
535 matrix[3] = xy - wz;
536 matrix[4] = 1 - (xx + zz);
537 matrix[5] = yz + wx;
538 matrix[6] = xz + wy;
539 matrix[7] = yz - wx;
540 matrix[8] = 1 - (xx + yy);
541 return matrix;
542};

545/**
546 * Generates the 4x4 rotation matrix from the given quaternion.
547 *
548 * @param {!goog.vec.Quaternion.AnyType} quat The source quaternion.
549 * @param {!goog.vec.AnyType} matrix The resulting matrix.
550 * @return {!goog.vec.AnyType} Return resulting matrix so that
551 * operations can be chained together.
552 */
553goog.vec.Quaternion.toRotationMatrix4 = function(quat, matrix) {
554 var x = quat[0], y = quat[1], z = quat[2], w = quat[3];
555 var x2 = 2 * x, y2 = 2 * y, z2 = 2 * z;
556 var wx = x2 * w;
557 var wy = y2 * w;
558 var wz = z2 * w;
559 var xx = x2 * x;
560 var xy = y2 * x;
561 var xz = z2 * x;
562 var yy = y2 * y;
563 var yz = z2 * y;
564 var zz = z2 * z;
565
566 matrix[0] = 1 - (yy + zz);
567 matrix[1] = xy + wz;
568 matrix[2] = xz - wy;
569 matrix[3] = 0;
570 matrix[4] = xy - wz;
571 matrix[5] = 1 - (xx + zz);
572 matrix[6] = yz + wx;
573 matrix[7] = 0;
574 matrix[8] = xz + wy;
575 matrix[9] = yz - wx;
576 matrix[10] = 1 - (xx + yy);
577 matrix[11] = 0;
578 matrix[12] = 0;
579 matrix[13] = 0;
580 matrix[14] = 0;
581 matrix[15] = 1;
582 return matrix;
583};

586/**
587 * Rotates a quaternion by the given angle about the X axis.
588 *
589 * @param {!goog.vec.Quaternion.AnyType} quat The quaternion.
590 * @param {number} angle The angle in radians.
591 * @param {!goog.vec.Quaternion.AnyType} resultQuat The quaternion to
592 * receive the result.
593 * @return {!goog.vec.Quaternion.AnyType} Return resultQuat so that
594 * operations can be chained together.
595 */
596goog.vec.Quaternion.rotateX = function(quat, angle, resultQuat) {
597 angle *= 0.5;
598 var ax = quat[0], ay = quat[1], az = quat[2], aw = quat[3];
599 var bx = Math.sin(angle), bw = Math.cos(angle);
600
601 resultQuat[0] = ax * bw + aw * bx;
602 resultQuat[1] = ay * bw + az * bx;
603 resultQuat[2] = az * bw - ay * bx;
604 resultQuat[3] = aw * bw - ax * bx;
605 return resultQuat;
606};

609/**
610 * Rotates a quaternion by the given angle about the Y axis.
611 *
612 * @param {!goog.vec.Quaternion.AnyType} quat The quaternion.
613 * @param {number} angle The angle in radians.
614 * @param {!goog.vec.Quaternion.AnyType} resultQuat The quaternion to
615 * receive the result.
616 * @return {!goog.vec.Quaternion.AnyType} Return resultQuat so that
617 * operations can be chained together.
618 */
619goog.vec.Quaternion.rotateY = function(quat, angle, resultQuat) {
620 angle *= 0.5;
621 var ax = quat[0], ay = quat[1], az = quat[2], aw = quat[3];
622 var by = Math.sin(angle), bw = Math.cos(angle);
623
624 resultQuat[0] = ax * bw - az * by;
625 resultQuat[1] = ay * bw + aw * by;
626 resultQuat[2] = az * bw + ax * by;
627 resultQuat[3] = aw * bw - ay * by;
628 return resultQuat;
629};

655/**
656 * Transforms a vec with a quaternion. Works on both vec3s and vec4s.
657 *
658 * @param {!goog.vec.AnyType} vec The vec to transform.
659 * @param {!goog.vec.Quaternion.AnyType} quat The quaternion.
660 * @param {!goog.vec.AnyType} resultVec The vec to receive the result.
661 * @return {!goog.vec.AnyType} Return resultVec so that operations can be
662 * chained together. Note that the caller is responsible for type-casting.
663 */
664goog.vec.Quaternion.transformVec = function(vec, quat, resultVec) {
665 var x = vec[0], y = vec[1], z = vec[2];
666 var qx = quat[0], qy = quat[1], qz = quat[2], qw = quat[3];
667 // Calculate quat * vec.
668 var ix = qw * x + qy * z - qz * y;
669 var iy = qw * y + qz * x - qx * z;
670 var iz = qw * z + qx * y - qy * x;
671 var iw = -qx * x - qy * y - qz * z;
672 // Calculate result * inverse quat.
673 resultVec[0] = ix * qw + iw * -qx + iy * -qz - iz * -qy;
674 resultVec[1] = iy * qw + iw * -qy + iz * -qx - ix * -qz;
675 resultVec[2] = iz * qw + iw * -qz + ix * -qy - iy * -qx;
676 return resultVec;
677};

680/**
681 * Computes the spherical linear interpolated value from the given quaternions
682 * q0 and q1 according to the coefficient t. The resulting quaternion is stored
683 * in resultQuat.
684 *
685 * @param {!goog.vec.Quaternion.AnyType} q0 The first quaternion.
686 * @param {!goog.vec.Quaternion.AnyType} q1 The second quaternion.
687 * @param {number} t The interpolating coefficient.
688 * @param {!goog.vec.Quaternion.AnyType} resultQuat The quaternion to
689 * receive the result.
690 * @return {!goog.vec.Quaternion.AnyType} Return resultQuat so that
691 * operations can be chained together.
692 */
693goog.vec.Quaternion.slerp = function(q0, q1, t, resultQuat) {
694 // Compute the dot product between q0 and q1 (cos of the angle between q0 and
695 // q1). If it's outside the interval [-1,1], then the arccos is not defined.
696 // The usual reason for this is that q0 and q1 are colinear. In this case
697 // the angle between the two is zero, so just return q1.
698 var cosVal = goog.vec.Quaternion.dot(q0, q1);
699 if (cosVal > 1 || cosVal < -1) {
700 goog.vec.Vec4.setFromArray(resultQuat, q1);
701 return resultQuat;
702 }
703
704 // Quaternions are a double cover on the space of rotations. That is, q and -q
705 // represent the same rotation. Thus we have two possibilities when
706 // interpolating between q0 and q1: going the short way or the long way. We
707 // prefer the short way since that is the likely expectation from users.
708 var factor = 1;
709 if (cosVal < 0) {
710 factor = -1;
711 cosVal = -cosVal;
712 }
713
714 // Compute the angle between q0 and q1. If it's very small, then just return
715 // q1 to avoid a very large denominator below.
716 var angle = Math.acos(cosVal);
717 if (angle <= goog.vec.EPSILON) {
718 goog.vec.Vec4.setFromArray(resultQuat, q1);
719 return resultQuat;
720 }
721
722 // Compute the coefficients and interpolate.
723 var invSinVal = 1 / Math.sin(angle);
724 var c0 = Math.sin((1 - t) * angle) * invSinVal;
725 var c1 = factor * Math.sin(t * angle) * invSinVal;
726
727 resultQuat[0] = q0[0] * c0 + q1[0] * c1;
728 resultQuat[1] = q0[1] * c0 + q1[1] * c1;
729 resultQuat[2] = q0[2] * c0 + q1[2] * c1;
730 resultQuat[3] = q0[3] * c0 + q1[3] * c1;
731 return resultQuat;
732};
