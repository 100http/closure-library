24/**
25 * @fileoverview Provides functions for operating on 3x3 double (64bit)
26 * matrices. The matrices are stored in column-major order.
27 *
28 * The last parameter will typically be the output object and an object
29 * can be both an input and output parameter to all methods except where
30 * noted.
31 *
32 * See the README for notes about the design and structure of the API
33 * (especially related to performance).
34 *
35 */

55/**
56 * Initializes the matrix from the set of values. Note the values supplied are
57 * in column major order.
58 *
59 * @param {goog.vec.mat3d.Type} mat The matrix to receive the
60 * values.
61 * @param {number} v00 The values at (0, 0).
62 * @param {number} v10 The values at (1, 0).
63 * @param {number} v20 The values at (2, 0).
64 * @param {number} v01 The values at (0, 1).
65 * @param {number} v11 The values at (1, 1).
66 * @param {number} v21 The values at (2, 1).
67 * @param {number} v02 The values at (0, 2).
68 * @param {number} v12 The values at (1, 2).
69 * @param {number} v22 The values at (2, 2).
70 * @return {!goog.vec.mat3d.Type} return mat so that operations can be
71 * chained together.
72 */
73goog.vec.mat3d.setFromValues = function(
74 mat, v00, v10, v20, v01, v11, v21, v02, v12, v22) {
75 mat[0] = v00;
76 mat[1] = v10;
77 mat[2] = v20;
78 mat[3] = v01;
79 mat[4] = v11;
80 mat[5] = v21;
81 mat[6] = v02;
82 mat[7] = v12;
83 mat[8] = v22;
84 return mat;
85};

513/**
514 * Multiplies the two matrices mat0 and mat1 using matrix multiplication,
515 * storing the result into resultMat.
516 *
517 * @param {goog.vec.mat3d.Type} mat0 The first (left hand) matrix.
518 * @param {goog.vec.mat3d.Type} mat1 The second (right hand) matrix.
519 * @param {goog.vec.mat3d.Type} resultMat The matrix to receive
520 * the results (may be either mat0 or mat1).
521 * @return {!goog.vec.mat3d.Type} return resultMat so that operations can be
522 * chained together.
523 */
524goog.vec.mat3d.multMat = function(mat0, mat1, resultMat) {
525 var a00 = mat0[0], a10 = mat0[1], a20 = mat0[2];
526 var a01 = mat0[3], a11 = mat0[4], a21 = mat0[5];
527 var a02 = mat0[6], a12 = mat0[7], a22 = mat0[8];
528
529 var b00 = mat1[0], b10 = mat1[1], b20 = mat1[2];
530 var b01 = mat1[3], b11 = mat1[4], b21 = mat1[5];
531 var b02 = mat1[6], b12 = mat1[7], b22 = mat1[8];
532
533 resultMat[0] = a00 * b00 + a01 * b10 + a02 * b20;
534 resultMat[1] = a10 * b00 + a11 * b10 + a12 * b20;
535 resultMat[2] = a20 * b00 + a21 * b10 + a22 * b20;
536 resultMat[3] = a00 * b01 + a01 * b11 + a02 * b21;
537 resultMat[4] = a10 * b01 + a11 * b11 + a12 * b21;
538 resultMat[5] = a20 * b01 + a21 * b11 + a22 * b21;
539 resultMat[6] = a00 * b02 + a01 * b12 + a02 * b22;
540 resultMat[7] = a10 * b02 + a11 * b12 + a12 * b22;
541 resultMat[8] = a20 * b02 + a21 * b12 + a22 * b22;
542 return resultMat;
543};

546/**
547 * Transposes the given matrix mat storing the result into resultMat.
548 *
549 * @param {goog.vec.mat3d.Type} mat The matrix to transpose.
550 * @param {goog.vec.mat3d.Type} resultMat The matrix to receive
551 * the results (may be mat).
552 * @return {!goog.vec.mat3d.Type} return resultMat so that operations can be
553 * chained together.
554 */
555goog.vec.mat3d.transpose = function(mat, resultMat) {
556 if (resultMat == mat) {
557 var a10 = mat[1], a20 = mat[2], a21 = mat[5];
558 resultMat[1] = mat[3];
559 resultMat[2] = mat[6];
560 resultMat[3] = a10;
561 resultMat[5] = mat[7];
562 resultMat[6] = a20;
563 resultMat[7] = a21;
564 } else {
565 resultMat[0] = mat[0];
566 resultMat[1] = mat[3];
567 resultMat[2] = mat[6];
568 resultMat[3] = mat[1];
569 resultMat[4] = mat[4];
570 resultMat[5] = mat[7];
571 resultMat[6] = mat[2];
572 resultMat[7] = mat[5];
573 resultMat[8] = mat[8];
574 }
575 return resultMat;
576};

579/**
580 * Computes the inverse of mat0 storing the result into resultMat. If the
581 * inverse is defined, this function returns true, false otherwise.
582 *
583 * @param {goog.vec.mat3d.Type} mat0 The matrix to invert.
584 * @param {goog.vec.mat3d.Type} resultMat The matrix to receive
585 * the result (may be mat0).
586 * @return {boolean} True if the inverse is defined. If false is returned,
587 * resultMat is not modified.
588 */
589goog.vec.mat3d.invert = function(mat0, resultMat) {
590 var a00 = mat0[0], a10 = mat0[1], a20 = mat0[2];
591 var a01 = mat0[3], a11 = mat0[4], a21 = mat0[5];
592 var a02 = mat0[6], a12 = mat0[7], a22 = mat0[8];
593
594 var t00 = a11 * a22 - a12 * a21;
595 var t10 = a12 * a20 - a10 * a22;
596 var t20 = a10 * a21 - a11 * a20;
597 var det = a00 * t00 + a01 * t10 + a02 * t20;
598 if (det == 0) {
599 return false;
600 }
601
602 var idet = 1 / det;
603 resultMat[0] = t00 * idet;
604 resultMat[3] = (a02 * a21 - a01 * a22) * idet;
605 resultMat[6] = (a01 * a12 - a02 * a11) * idet;
606
607 resultMat[1] = t10 * idet;
608 resultMat[4] = (a00 * a22 - a02 * a20) * idet;
609 resultMat[7] = (a02 * a10 - a00 * a12) * idet;
610
611 resultMat[2] = t20 * idet;
612 resultMat[5] = (a01 * a20 - a00 * a21) * idet;
613 resultMat[8] = (a00 * a11 - a01 * a10) * idet;
614 return true;
615};

618/**
619 * Returns true if the components of mat0 are equal to the components of mat1.
620 *
621 * @param {goog.vec.mat3d.Type} mat0 The first matrix.
622 * @param {goog.vec.mat3d.Type} mat1 The second matrix.
623 * @return {boolean} True if the the two matrices are equivalent.

704 * angle about the axis defined by the vector (ax, ay, az).
705 *
706 * @param {goog.vec.mat3d.Type} mat The matrix.

814/**
815 * Rotate the given matrix by angle about the x,y,z axis. Equivalent to:
816 * goog.vec.mat3d.multMat(
817 * mat,
818 * goog.vec.mat3d.makeRotate(goog.vec.mat3d.create(), angle, x, y, z),
819 * mat);
820 *
821 * @param {goog.vec.mat3d.Type} mat The matrix.
822 * @param {number} angle The angle in radians.
823 * @param {number} x The x component of the rotation axis.
824 * @param {number} y The y component of the rotation axis.
825 * @param {number} z The z component of the rotation axis.
826 * @return {!goog.vec.mat3d.Type} return mat so that operations can be
827 * chained.
828 */
829goog.vec.mat3d.rotate = function(mat, angle, x, y, z) {
830 var m00 = mat[0], m10 = mat[1], m20 = mat[2];
831 var m01 = mat[3], m11 = mat[4], m21 = mat[5];
832 var m02 = mat[6], m12 = mat[7], m22 = mat[8];
833
834 var cosAngle = Math.cos(angle);
835 var sinAngle = Math.sin(angle);
836 var diffCosAngle = 1 - cosAngle;
837 var r00 = x * x * diffCosAngle + cosAngle;
838 var r10 = x * y * diffCosAngle + z * sinAngle;
839 var r20 = x * z * diffCosAngle - y * sinAngle;
840
841 var r01 = x * y * diffCosAngle - z * sinAngle;
842 var r11 = y * y * diffCosAngle + cosAngle;
843 var r21 = y * z * diffCosAngle + x * sinAngle;
844
845 var r02 = x * z * diffCosAngle + y * sinAngle;
846 var r12 = y * z * diffCosAngle - x * sinAngle;
847 var r22 = z * z * diffCosAngle + cosAngle;
848
849 mat[0] = m00 * r00 + m01 * r10 + m02 * r20;
850 mat[1] = m10 * r00 + m11 * r10 + m12 * r20;
851 mat[2] = m20 * r00 + m21 * r10 + m22 * r20;
852 mat[3] = m00 * r01 + m01 * r11 + m02 * r21;
853 mat[4] = m10 * r01 + m11 * r11 + m12 * r21;
854 mat[5] = m20 * r01 + m21 * r11 + m22 * r21;
855 mat[6] = m00 * r02 + m01 * r12 + m02 * r22;
856 mat[7] = m10 * r02 + m11 * r12 + m12 * r22;
857 mat[8] = m20 * r02 + m21 * r12 + m22 * r22;
858
859 return mat;
860};

953/**
954 * Makes the given 3x3 matrix a rotation matrix given Euler angles using
955 * the ZXZ convention.
956 * Given the euler angles [theta1, theta2, theta3], the rotation is defined as
957 * rotation = rotation_z(theta1) * rotation_x(theta2) * rotation_z(theta3),
958 * with theta1 in [0, 2 * pi], theta2 in [0, pi] and theta3 in [0, 2 * pi].
959 * rotation_x(theta) means rotation around the X axis of theta radians.
960 *
961 * @param {goog.vec.mat3d.Type} mat The matrix.
962 * @param {number} theta1 The angle of rotation around the Z axis in radians.
963 * @param {number} theta2 The angle of rotation around the X axis in radians.
964 * @param {number} theta3 The angle of rotation around the Z axis in radians.
965 * @return {!goog.vec.mat3d.Type} return mat so that operations can be
966 * chained.
967 */
968goog.vec.mat3d.makeEulerZXZ = function(mat, theta1, theta2, theta3) {
969 var c1 = Math.cos(theta1);
970 var s1 = Math.sin(theta1);
971
972 var c2 = Math.cos(theta2);
973 var s2 = Math.sin(theta2);
974
975 var c3 = Math.cos(theta3);
976 var s3 = Math.sin(theta3);
977
978 mat[0] = c1 * c3 - c2 * s1 * s3;
979 mat[1] = c2 * c1 * s3 + c3 * s1;
980 mat[2] = s3 * s2;
981
982 mat[3] = -c1 * s3 - c3 * c2 * s1;
983 mat[4] = c1 * c2 * c3 - s1 * s3;
984 mat[5] = c3 * s2;
985
986 mat[6] = s2 * s1;
987 mat[7] = -c1 * s2;
988 mat[8] = c2;
989
990 return mat;
991};

994/**
995 * Decomposes a rotation matrix into Euler angles using the ZXZ convention so
996 * that rotation = rotation_z(theta1) * rotation_x(theta2) * rotation_z(theta3),
997 * with theta1 in [0, 2 * pi], theta2 in [0, pi] and theta3 in [0, 2 * pi].
998 * rotation_x(theta) means rotation around the X axis of theta radians.
999 *
1000 * @param {goog.vec.mat3d.Type} mat The matrix.
1001 * @param {goog.vec.vec3d.Type} euler The ZXZ Euler angles in
1002 * radians as [theta1, theta2, theta3].
1003 * @param {boolean=} opt_theta2IsNegative Whether theta2 is in [-pi, 0] instead
1004 * of the default [0, pi].
1005 * @return {!goog.vec.vec3d.Type} return euler so that operations can be
1006 * chained together.
1007 */
1008goog.vec.mat3d.toEulerZXZ = function(mat, euler, opt_theta2IsNegative) {
1009 // There is an ambiguity in the sign of sinTheta2 because of the sqrt.
1010 var sinTheta2 = Math.sqrt(mat[2] * mat[2] + mat[5] * mat[5]);
1011
1012 // By default we explicitely constrain theta2 to be in [0, pi],
1013 // so sinTheta2 is always positive. We can change the behavior and specify
1014 // theta2 to be negative in [-pi, 0] with opt_Theta2IsNegative.
1015 var signTheta2 = opt_theta2IsNegative ? -1 : 1;
1016
1017 if (sinTheta2 > goog.vec.EPSILON) {
1018 euler[2] = Math.atan2(mat[2] * signTheta2, mat[5] * signTheta2);
1019 euler[1] = Math.atan2(sinTheta2 * signTheta2, mat[8]);
1020 euler[0] = Math.atan2(mat[6] * signTheta2, -mat[7] * signTheta2);
1021 } else {
1022 // There is also an arbitrary choice for theta1 = 0 or theta2 = 0 here.
1023 // We assume theta1 = 0 as some applications do not allow the camera to roll
1024 // (i.e. have theta1 != 0).
1025 euler[0] = 0;
1026 euler[1] = Math.atan2(sinTheta2 * signTheta2, mat[8]);
1027 euler[2] = Math.atan2(mat[1], mat[0]);
1028 }
1029
1030 // Atan2 outputs angles in [-pi, pi] so we bring them back to [0, 2 * pi].
1031 euler[0] = (euler[0] + Math.PI * 2) % (Math.PI * 2);
1032 euler[2] = (euler[2] + Math.PI * 2) % (Math.PI * 2);
1033 // For theta2 we want the angle to be in [0, pi] or [-pi, 0] depending on
1034 // signTheta2.
1035 euler[1] = ((euler[1] * signTheta2 + Math.PI * 2) % (Math.PI * 2)) *
1036 signTheta2;
1037
1038 return euler;
1039};
