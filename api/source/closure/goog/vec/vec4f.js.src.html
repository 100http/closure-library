248/**
249 * Normalizes the given vector storing the result into resultVec.
250 *
251 * @param {goog.vec.vec4f.Type} vec0 The vector to normalize.
252 * @param {goog.vec.vec4f.Type} resultVec The vector to
253 * receive the result. May be vec0.
254 * @return {!goog.vec.vec4f.Type} Return resultVec so that operations can be
255 * chained together.
256 */
257goog.vec.vec4f.normalize = function(vec0, resultVec) {
258 var x = vec0[0], y = vec0[1], z = vec0[2], w = vec0[3];
259 var ilen = 1 / Math.sqrt(x * x + y * y + z * z + w * w);
260 resultVec[0] = x * ilen;
261 resultVec[1] = y * ilen;
262 resultVec[2] = z * ilen;
263 resultVec[3] = w * ilen;
264 return resultVec;
265};

280/**
281 * Linearly interpolate from v0 to v1 according to f. The value of f should be
282 * in the range [0..1] otherwise the results are undefined.
283 *
284 * @param {goog.vec.vec4f.Type} v0 The first vector.
285 * @param {goog.vec.vec4f.Type} v1 The second vector.
286 * @param {number} f The interpolation factor.
287 * @param {goog.vec.vec4f.Type} resultVec The vector to receive the
288 * results (may be v0 or v1).
289 * @return {!goog.vec.vec4f.Type} Return resultVec so that operations can be
290 * chained together.
291 */
292goog.vec.vec4f.lerp = function(v0, v1, f, resultVec) {
293 var x = v0[0], y = v0[1], z = v0[2], w = v0[3];
294 resultVec[0] = (v1[0] - x) * f + x;
295 resultVec[1] = (v1[1] - y) * f + y;
296 resultVec[2] = (v1[2] - z) * f + z;
297 resultVec[3] = (v1[3] - w) * f + w;
298 return resultVec;
299};

302/**
303 * Compares the components of vec0 with the components of another vector or
304 * scalar, storing the larger values in resultVec.
305 *
306 * @param {goog.vec.vec4f.Type} vec0 The source vector.
307 * @param {goog.vec.vec4f.Type|number} limit The limit vector or scalar.
308 * @param {goog.vec.vec4f.Type} resultVec The vector to receive the
309 * results (may be vec0 or limit).
310 * @return {!goog.vec.vec4f.Type} Return resultVec so that operations can be
311 * chained together.

313goog.vec.vec4f.max = function(vec0, limit, resultVec) {
314 if (goog.isNumber(limit)) {
315 resultVec[0] = Math.max(vec0[0], limit);
316 resultVec[1] = Math.max(vec0[1], limit);
317 resultVec[2] = Math.max(vec0[2], limit);
318 resultVec[3] = Math.max(vec0[3], limit);
319 } else {
320 resultVec[0] = Math.max(vec0[0], limit[0]);
321 resultVec[1] = Math.max(vec0[1], limit[1]);
322 resultVec[2] = Math.max(vec0[2], limit[2]);
323 resultVec[3] = Math.max(vec0[3], limit[3]);
324 }
325 return resultVec;
