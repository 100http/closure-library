31/**
32 * Support class for spell checker components. Provides basic functionality
33 * such as word lookup and caching.
34 *
35 * @param {Function=} opt_lookupFunction Function to use for word lookup. Must
36 * accept an array of words, an object reference and a callback function as
37 * parameters. It must also call the callback function (as a method on the
38 * object), once ready, with an array containing the original words, their
39 * spelling status and optionally an array of suggestions.
40 * @param {string=} opt_language Content language.
41 * @constructor
42 * @extends {goog.events.EventTarget}
43 * @final
44 */
45goog.spell.SpellCheck = function(opt_lookupFunction, opt_language) {
46 goog.events.EventTarget.call(this);
47
48 /**
49 * Function used to lookup spelling of words.
50 * @type {Function}
51 * @private
52 */
53 this.lookupFunction_ = opt_lookupFunction || null;
54
55 /**
56 * Cache for words not yet checked with lookup function.
57 * @type {goog.structs.Set}
58 * @private
59 */
60 this.unknownWords_ = new goog.structs.Set();
61
62 this.setLanguage(opt_language);
63};

239/**
240 * Checks spelling for a block of text.
241 *
242 * @param {string} text Block of text to spell check.
243 */
244goog.spell.SpellCheck.prototype.checkBlock = function(text) {
245 var words = text.split(goog.spell.SpellCheck.WORD_BOUNDARY_REGEX);
246
247 var len = words.length;
248 for (var word, i = 0; i < len; i++) {
249 word = words[i];
250 this.checkWord_(word);
251 }
252
253 if (!this.queueTimer_ && !this.lookupInProgress_ &&
254 this.unknownWords_.getCount()) {
255 this.processPending_();
256 }
257 else if (this.unknownWords_.getCount() == 0) {
258 this.dispatchEvent(goog.spell.SpellCheck.EventType.READY);
259 }
260};

263/**
264 * Checks spelling for a single word. Returns the status of the supplied word,
265 * or UNKNOWN if it's not cached. If it's not cached the word is added to a
266 * queue and checked with the verification implementation with a short delay.
267 *
268 * @param {string} word Word to check spelling of.
269 * @return {goog.spell.SpellCheck.WordStatus} The status of the supplied word,
270 * or UNKNOWN if it's not cached.
271 */
272goog.spell.SpellCheck.prototype.checkWord = function(word) {
273 var status = this.checkWord_(word);
274
275 if (status == goog.spell.SpellCheck.WordStatus.UNKNOWN &&
276 !this.queueTimer_ && !this.lookupInProgress_) {
277 this.queueTimer_ = goog.Timer.callOnce(this.processPending_,
278 goog.spell.SpellCheck.LOOKUP_DELAY_, this);
279 }
280
281 return status;
282};

285/**
286 * Checks spelling for a single word. Returns the status of the supplied word,
287 * or UNKNOWN if it's not cached.
288 *
289 * @param {string} word Word to check spelling of.
290 * @return {goog.spell.SpellCheck.WordStatus} The status of the supplied word,
291 * or UNKNOWN if it's not cached.
292 * @private
293 */
294goog.spell.SpellCheck.prototype.checkWord_ = function(word) {
295 if (!word) {
296 return goog.spell.SpellCheck.WordStatus.INVALID;
297 }
298
299 var cacheEntry = this.cache_[word];
300 if (!cacheEntry) {
301 this.unknownWords_.add(word);
302 return goog.spell.SpellCheck.WordStatus.UNKNOWN;
303 }
304
305 return cacheEntry[goog.spell.SpellCheck.CacheIndex.STATUS];
306};

309/**
310 * Processes pending words unless a lookup operation has already been queued or
311 * is in progress.
312 *
313 * @throws {Error}

315goog.spell.SpellCheck.prototype.processPending = function() {
316 if (this.unknownWords_.getCount()) {
317 if (!this.queueTimer_ && !this.lookupInProgress_) {

319 }
320 } else {
321 this.dispatchEvent(goog.spell.SpellCheck.EventType.READY);
322 }

326/**
327 * Processes pending words using the verification callback.
328 *
329 * @throws {Error}
330 * @private
331 */
332goog.spell.SpellCheck.prototype.processPending_ = function() {
333 if (!this.lookupFunction_) {
334 throw Error('No lookup function provided for spell checker.');
335 }
336
337 if (this.unknownWords_.getCount()) {
338 this.lookupInProgress_ = true;
339 var func = this.lookupFunction_;
340 func(this.unknownWords_.getValues(), this, this.lookupCallback_);
341 } else {
342 this.dispatchEvent(goog.spell.SpellCheck.EventType.READY);
343 }
344
345 this.queueTimer_ = 0;
346};

349/**
350 * Callback for lookup function.
351 *
352 * @param {Array<Array<?>>} data Data array. Each word is represented by an
353 * array containing the word, the status and optionally an array of
354 * suggestions. Passing null indicates that the operation failed.
355 * @private
356 *
357 * Example:
358 * obj.lookupCallback_([
359 * ['word', VALID],
360 * ['wrod', INVALID, ['word', 'wood', 'rod']]
361 * ]);
362 */
363goog.spell.SpellCheck.prototype.lookupCallback_ = function(data) {
364
365 // Lookup function failed; abort then dispatch error event.
366 if (data == null) {
367 if (this.queueTimer_) {
368 goog.Timer.clear(this.queueTimer_);
369 this.queueTimer_ = 0;
370 }
371 this.lookupInProgress_ = false;
372
373 this.dispatchEvent(goog.spell.SpellCheck.EventType.ERROR);
374 return;
375 }
376
377 for (var a, i = 0; a = data[i]; i++) {
378 this.setWordStatus_(a[0], a[1], a[2]);
379 }
380 this.lookupInProgress_ = false;
381
382 // Fire ready event if all pending words have been processed.
383 if (this.unknownWords_.getCount() == 0) {
384 this.dispatchEvent(goog.spell.SpellCheck.EventType.READY);
385
386 // Process pending
387 } else if (!this.queueTimer_) {
388 this.queueTimer_ = goog.Timer.callOnce(this.processPending_,
389 goog.spell.SpellCheck.LOOKUP_DELAY_, this);
390 }
391};

416 * @param {Array<string>=} opt_suggestions Suggestions.
417 * @private

419goog.spell.SpellCheck.prototype.setWordStatus_ =
420 function(word, status, opt_suggestions) {
421 var suggestions = opt_suggestions || [];
422 this.cache_[word] = [status, suggestions];
423 this.unknownWords_.remove(word);

425 this.dispatchEvent(
426 new goog.spell.SpellCheck.WordChangedEvent(this, word, status));

430/**
431 * Returns suggestions for the given word.
432 *
433 * @param {string} word Word to get suggestions for.
434 * @return {Array<string>} An array of suggestions for the given word.
435 */
436goog.spell.SpellCheck.prototype.getSuggestions = function(word) {
437 var cacheEntry = this.cache_[word];
438
439 if (!cacheEntry) {
440 this.checkWord(word);
441 return [];
442 }
443
444 return cacheEntry[goog.spell.SpellCheck.CacheIndex.STATUS] ==
445 goog.spell.SpellCheck.WordStatus.INVALID ?
446 cacheEntry[goog.spell.SpellCheck.CacheIndex.SUGGESTIONS] : [];
447};
