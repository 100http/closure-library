31/**
32 * @define {boolean} Whether to require all Soy templates to be "strict html".
33 * Soy templates that use strict autoescaping forbid noAutoescape along with
34 * many dangerous directives, and return a runtime type SanitizedContent that
35 * marks them as safe.
36 *
37 * If this flag is enabled, Soy templates will fail to render if a template
38 * returns plain text -- indicating it is a non-strict template.
39 */
40goog.define('goog.soy.REQUIRE_STRICT_AUTOESCAPE', false);

43/**
44 * Sets the processed template as the innerHTML of an element. It is recommended
45 * to use this helper function instead of directly setting innerHTML in your
46 * hand-written code, so that it will be easier to audit the code for cross-site
47 * scripting vulnerabilities.
48 *
49 * @param {?Element} element The element whose content we are rendering into.
50 * @param {!goog.soy.data.SanitizedContent} templateResult The processed
51 * template of kind HTML or TEXT (which will be escaped).
52 * @template ARG_TYPES
53 */
54goog.soy.renderHtml = function(element, templateResult) {
55 element.innerHTML = goog.soy.ensureTemplateOutputHtml_(templateResult);
56};

59/**
60 * Renders a Soy template and then set the output string as
61 * the innerHTML of an element. It is recommended to use this helper function
62 * instead of directly setting innerHTML in your hand-written code, so that it
63 * will be easier to audit the code for cross-site scripting vulnerabilities.
64 *
65 * @param {Element} element The element whose content we are rendering into.
66 * @param {null|function(ARG_TYPES, null=, Object<string, *>=):*} template
67 * The Soy template defining the element's content.
68 * @param {ARG_TYPES=} opt_templateData The data for the template.
69 * @param {Object=} opt_injectedData The injected data for the template.
70 * @template ARG_TYPES
71 */
72goog.soy.renderElement = function(element, template, opt_templateData,
73 opt_injectedData) {
74 // Soy template parameter is only nullable for historical reasons.
75 goog.asserts.assert(template, 'Soy template may not be null.');
76 element.innerHTML = goog.soy.ensureTemplateOutputHtml_(template(
77 opt_templateData || goog.soy.defaultTemplateData_, undefined,
78 opt_injectedData));
79};

82/**
83 * Renders a Soy template into a single node or a document
84 * fragment. If the rendered HTML string represents a single node, then that
85 * node is returned (note that this is *not* a fragment, despite them name of
86 * the method). Otherwise a document fragment is returned containing the
87 * rendered nodes.
88 *
89 * @param {null|function(ARG_TYPES, null=, Object<string, *>=):*} template
90 * The Soy template defining the element's content.
91 * @param {ARG_TYPES=} opt_templateData The data for the template.
92 * @param {Object=} opt_injectedData The injected data for the template.
93 * @param {goog.dom.DomHelper=} opt_domHelper The DOM helper used to
94 * create DOM nodes; defaults to {@code goog.dom.getDomHelper}.
95 * @return {!Node} The resulting node or document fragment.
96 * @template ARG_TYPES
97 */
98goog.soy.renderAsFragment = function(template, opt_templateData,
99 opt_injectedData, opt_domHelper) {
100 // Soy template parameter is only nullable for historical reasons.
101 goog.asserts.assert(template, 'Soy template may not be null.');
102 var dom = opt_domHelper || goog.dom.getDomHelper();
103 var html = goog.soy.ensureTemplateOutputHtml_(
104 template(opt_templateData || goog.soy.defaultTemplateData_,
105 undefined, opt_injectedData));
106 goog.soy.assertFirstTagValid_(html);
107 return dom.htmlToDocumentFragment(html);
108};

113 * HTML string represents a single node, then that node is returned. Otherwise,

116 * @param {null|function(ARG_TYPES, null=, Object<string, *>=):*} template

126goog.soy.renderAsElement = function(template, opt_templateData,

128 // Soy template parameter is only nullable for historical reasons.
129 goog.asserts.assert(template, 'Soy template may not be null.');
130 return goog.soy.convertToElement_(template(
131 opt_templateData || goog.soy.defaultTemplateData_,
132 undefined, opt_injectedData), opt_domHelper);

136/**
137 * Converts a processed Soy template into a single node. If the rendered
138 * HTML string represents a single node, then that node is returned. Otherwise,
139 * a DIV element is returned containing the rendered nodes.
140 *
141 * @param {!goog.soy.data.SanitizedContent} templateResult The processed
142 * template of kind HTML or TEXT (which will be escaped).
143 * @param {?goog.dom.DomHelper=} opt_domHelper The DOM helper used to

145 * @return {!Element} Rendered template contents, wrapped in a parent DIV
146 * element if necessary.

153/**
154 * Non-strict version of {@code goog.soy.convertToElement}.
155 *
156 * @param {*} templateResult The processed template.
157 * @param {?goog.dom.DomHelper=} opt_domHelper The DOM helper used to
158 * create DOM nodes; defaults to {@code goog.dom.getDomHelper}.
159 * @return {!Element} Rendered template contents, wrapped in a parent DIV
160 * element if necessary.
161 * @private
162 */
163goog.soy.convertToElement_ = function(templateResult, opt_domHelper) {
164 var dom = opt_domHelper || goog.dom.getDomHelper();
165 var wrapper = dom.createElement(goog.dom.TagName.DIV);
166 var html = goog.soy.ensureTemplateOutputHtml_(templateResult);
167 goog.soy.assertFirstTagValid_(html);
168 wrapper.innerHTML = html;
169
170 // If the template renders as a single element, return it.
171 if (wrapper.childNodes.length == 1) {
172 var firstChild = wrapper.firstChild;
173 if (firstChild.nodeType == goog.dom.NodeType.ELEMENT) {
174 return /** @type {!Element} */ (firstChild);
175 }
176 }
177
178 // Otherwise, return the wrapper DIV.
179 return wrapper;
180};

183/**
184 * Ensures the result is "safe" to insert as HTML.
185 *
186 * Note if the template has non-strict autoescape, the guarantees here are very
187 * weak. It is recommended applications switch to requiring strict
188 * autoescaping over time by tweaking goog.soy.REQUIRE_STRICT_AUTOESCAPE.
189 *
190 * In the case the argument is a SanitizedContent object, it either must
191 * already be of kind HTML, or if it is kind="text", the output will be HTML
192 * escaped.
193 *
194 * @param {*} templateResult The template result.
195 * @return {string} The assumed-safe HTML output string.
196 * @private
197 */
198goog.soy.ensureTemplateOutputHtml_ = function(templateResult) {
199 // Allow strings as long as strict autoescaping is not mandated. Note we
200 // allow everything that isn't an object, because some non-escaping templates
201 // end up returning non-strings if their only print statement is a
202 // non-escaped argument, plus some unit tests spoof templates.
203 // TODO(gboyer): Track down and fix these cases.
204 if (!goog.soy.REQUIRE_STRICT_AUTOESCAPE && !goog.isObject(templateResult)) {
205 return String(templateResult);
206 }
207
208 // Allow SanitizedContent of kind HTML.
209 if (templateResult instanceof goog.soy.data.SanitizedContent) {
210 templateResult = /** @type {!goog.soy.data.SanitizedContent} */ (
211 templateResult);
212 var ContentKind = goog.soy.data.SanitizedContentKind;
213 if (templateResult.contentKind === ContentKind.HTML) {
214 return goog.asserts.assertString(templateResult.getContent());
215 }
216 if (templateResult.contentKind === ContentKind.TEXT) {
217 // Allow text to be rendered, as long as we escape it. Other content
218 // kinds will fail, since we don't know what to do with them.
219 // TODO(gboyer): Perhaps also include URI in this case.
220 return goog.string.htmlEscape(templateResult.getContent());
221 }
222 }
223
224 goog.asserts.fail('Soy template output is unsafe for use as HTML: ' +
225 templateResult);
226
227 // In production, return a safe string, rather than failing hard.
228 return 'zSoyz';
229};

232/**
233 * Checks that the rendered HTML does not start with an invalid tag that would
234 * likely cause unexpected output from renderAsElement or renderAsFragment.
235 * See {@link http://www.w3.org/TR/html5/semantics.html#semantics} for reference
236 * as to which HTML elements can be parents of each other.
237 * @param {string} html The output of a template.
238 * @private
239 */
240goog.soy.assertFirstTagValid_ = function(html) {
241 if (goog.asserts.ENABLE_ASSERTS) {
242 var matches = html.match(goog.soy.INVALID_TAG_TO_RENDER_);
243 goog.asserts.assert(!matches, 'This template starts with a %s, which ' +
244 'cannot be a child of a <div>, as required by soy internals. ' +
245 'Consider using goog.soy.renderElement instead.\nTemplate output: %s',
246 matches && matches[0], html);
247 }
248};

251/**
252 * A pattern to find templates that cannot be rendered by renderAsElement or
253 * renderAsFragment, as these elements cannot exist as the child of a <div>.
254 * @type {!RegExp}
255 * @private
256 */
257goog.soy.INVALID_TAG_TO_RENDER_ =
258 /^<(body|caption|col|colgroup|head|html|tr|td|th|tbody|thead|tfoot)>/i;
