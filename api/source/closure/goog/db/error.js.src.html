30/**
31 * A database error. Since the stack trace can be unhelpful in an asynchronous
32 * context, the error provides a message about where it was produced.
33 *
34 * @param {number|!DOMError} error The DOMError instance returned by the
35 * browser for Chrome22+, or an error code for previous versions.
36 * @param {string} context A description of where the error occured.
37 * @param {string=} opt_message Additional message.
38 * @constructor
39 * @extends {goog.debug.Error}
40 * @final
41 */
42goog.db.Error = function(error, context, opt_message) {
43 var errorCode = null;
44 var internalError = null;
45 if (goog.isNumber(error)) {
46 errorCode = error;
47 internalError = {name: goog.db.Error.getName(errorCode)};
48 } else {
49 internalError = error;
50 errorCode = goog.db.Error.getCode(error.name);
51 }
52
53 /**
54 * The code for this error.
55 *
56 * @type {number}
57 */
58 this.code = errorCode;
59
60 /**
61 * The DOMException as returned by the browser.
62 *
63 * @type {!DOMError}
64 * @private
65 */
66 this.error_ = /** @type {!DOMError} */ (internalError);
67
68 var msg = 'Error ' + context + ': ' + this.getName();
69 if (opt_message) {
70 msg += ', ' + opt_message;
71 }
72 goog.db.Error.base(this, 'constructor', msg);
73};
74goog.inherits(goog.db.Error, goog.debug.Error);

177/**
178 * Translates an error code into a more useful message.
179 *
180 * @param {number} code Error code.
181 * @return {string} A debug message.
182 */
183goog.db.Error.getMessage = function(code) {
184 switch (code) {
185 case goog.db.Error.ErrorCode.UNKNOWN_ERR:
186 return 'Unknown error';
187 case goog.db.Error.ErrorCode.NON_TRANSIENT_ERR:
188 return 'Invalid operation';
189 case goog.db.Error.ErrorCode.NOT_FOUND_ERR:
190 return 'Required database object not found';
191 case goog.db.Error.ErrorCode.CONSTRAINT_ERR:
192 return 'Constraint unsatisfied';
193 case goog.db.Error.ErrorCode.DATA_ERR:
194 return 'Invalid data';
195 case goog.db.Error.ErrorCode.NOT_ALLOWED_ERR:
196 return 'Operation disallowed';
197 case goog.db.Error.ErrorCode.TRANSACTION_INACTIVE_ERR:
198 return 'Transaction not active';
199 case goog.db.Error.ErrorCode.ABORT_ERR:
200 return 'Request aborted';
201 case goog.db.Error.ErrorCode.READ_ONLY_ERR:
202 return 'Modifying operation not allowed in a read-only transaction';
203 case goog.db.Error.ErrorCode.TIMEOUT_ERR:
204 return 'Transaction timed out';
205 case goog.db.Error.ErrorCode.QUOTA_ERR:
206 return 'Database storage space quota exceeded';
207 case goog.db.Error.ErrorCode.INVALID_ACCESS_ERR:
208 return 'Invalid operation';
209 case goog.db.Error.ErrorCode.INVALID_STATE_ERR:
210 return 'Invalid state';
211 default:
212 return 'Unrecognized exception with code ' + code;
213 }
214};

315/**
316 * Constructs an goog.db.Error instance from an IDBRequest. This abstraction is
317 * necessary to provide backwards compatibility with Chrome21.
318 *
319 * @param {!IDBRequest} request The request that failed.
320 * @param {string} message The error message to add to err if it's wrapped.
321 * @return {!goog.db.Error} The error that caused the failure.
322 */
323goog.db.Error.fromRequest = function(request, message) {
324 if ('error' in request) {
325 // Chrome 21 and before.
326 return new goog.db.Error(request.error, message);
327 } else if ('name' in request) {
328 // Chrome 22+.
329 var errorName = goog.db.Error.getName(request.errorCode);
330 return new goog.db.Error(
331 /**@type {!DOMError} */ ({name: errorName}), message);
332 } else {
333 return new goog.db.Error(/** @type {!DOMError} */ (
334 {name: goog.db.Error.ErrorName.UNKNOWN_ERR}), message);
335 }
336};

339/**
340 * Constructs an goog.db.Error instance from an DOMException. This abstraction
341 * is necessary to provide backwards compatibility with Chrome21.
342 *
343 * @param {!IDBDatabaseException} ex The exception that was thrown.
344 * @param {string} message The error message to add to err if it's wrapped.
345 * @return {!goog.db.Error} The error that caused the failure.
346 * @suppress {invalidCasts} The cast from IDBDatabaseException to DOMError
347 * is invalid and will not compile.
348 */
349goog.db.Error.fromException = function(ex, message) {
350 if ('name' in ex) {
351 // Chrome 22+.
352 var errorMessage = message + ': ' + ex.message;
353 return new goog.db.Error(/** @type {!DOMError} */ (ex), errorMessage);
354 } else if ('code' in ex) {
355 // Chrome 21 and before.
356 var errorName = goog.db.Error.getName(ex.code);
357 var errorMessage = message + ': ' + ex.message;
358 return new goog.db.Error(
359 /** @type {!DOMError} */ ({name: errorName}), errorMessage);
360 } else {
361 return new goog.db.Error(/** @type {!DOMError} */ (
362 {name: goog.db.Error.ErrorName.UNKNOWN_ERR}), message);
363 }
