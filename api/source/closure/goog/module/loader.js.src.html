37/**
38 * The dynamic loading functionality is defined as a class. The class
39 * will be used as singleton. There is, however, a two step
40 * initialization procedure because parameters need to be passed to
41 * the goog.module.Loader instance.
42 *
43 * @constructor
44 * @final
45 */
46goog.module.Loader = function() {
47 /**
48 * Map of module name/array of {symbol name, callback} pairs that are pending
49 * to be loaded.
50 * @type {Object}
51 * @private
52 */
53 this.pending_ = {};
54
55 /**
56 * Provides associative access to each module and the symbols of each module
57 * that have aready been loaded (one lookup for the module, another lookup
58 * on the module for the symbol).
59 * @type {Object}
60 * @private
61 */
62 this.modules_ = {};
63
64 /**
65 * Map of module name to module url. Used to avoid fetching the same URL
66 * twice by keeping track of in-flight URLs.
67 * Note: this allows two modules to be bundled into the same file.
68 * @type {Object}
69 * @private
70 */
71 this.pendingModuleUrls_ = {};
72
73 /**
74 * The base url to load modules from. This property will be set in init().
75 * @type {?string}
76 * @private
77 */
78 this.urlBase_ = null;
79
80 /**
81 * Array of modules that have been requested before init() was called.
82 * If require() is called before init() was called, the required
83 * modules can obviously not yet be loaded, because their URL is
84 * unknown. The modules that are requested before init() are
85 * therefore stored in this array, and they are loaded at init()
86 * time.
87 * @type {Array<string>}
88 * @private
89 */
90 this.pendingBeforeInit_ = [];
91};

146/**
147 * Produces a function that delegates all its arguments to a
148 * dynamically loaded function. This is used to export dynamically
149 * loaded functions.
150 *
151 * @param {string} module The module to load from.
152 * @param {number|string} symbol The ID of the symbol to load from the module.
153 * This symbol must resolve to a function.
154 * @return {!Function} A function that forwards all its arguments to
155 * the dynamically loaded function specified by module and symbol.
156 */
157goog.module.Loader.loaderCall = function(module, symbol) {
158 return function() {
159 var args = arguments;
160 goog.module.Loader.require(module, symbol, function(f) {
161 f.apply(null, args);
162 });
163 };

188/**
189 * Loads the module by evaluating the javascript text in the current
190 * scope. Uncompiled, base identifiers are visible in the global scope;
191 * when compiled they are visible in the closure of the anonymous
192 * namespace. Notice that this cannot be replaced by the global eval,
193 * because the global eval isn't in the scope of the anonymous
194 * namespace function that the jscompiled code lives in.
195 *
196 * @param {string} t_ The javascript text to evaluate. IMPORTANT: The
197 * name of the identifier is chosen so that it isn't compiled and
198 * hence cannot shadow compiled identifiers in the surrounding scope.
199 * @private
200 */
201goog.module.Loader.loaderEval_ = function(t_) {
202 eval(t_);

206/**
207 * Initializes the Loader to be fully functional. Also executes load
208 * requests that were received before initialization. Must be called
209 * exactly once, with the URL of the base library. Module URLs are
210 * derived from the URL of the base library by inserting the module
211 * name, preceded by a period, before the .js prefix of the base URL.
212 *
213 * @param {string} baseUrl The URL of the base library.
214 * @param {Function=} opt_urlFunction Function that creates the URL for the
215 * module file. It will be passed the base URL for module files and the
216 * module name and should return the fully-formed URL to the module file to
217 * load.
218 */
219goog.module.Loader.prototype.init = function(baseUrl, opt_urlFunction) {
220 // For the use by the module wrappers, loaderEval_ is exported to
221 // the page. Note that, despite the name, this is not part of the
222 // API, so it is here and not in api_app.js. Cf. BUILD. Note this is
223 // done before the first load requests are sent.
224 goog.exportSymbol(goog.module.Loader.LOAD_CALLBACK,
225 goog.module.Loader.loaderEval_);
226
227 this.urlBase_ = baseUrl.replace(/\.js$/, '');
228 if (opt_urlFunction) {
229 this.getModuleUrl_ = opt_urlFunction;
230 }
231
232 goog.array.forEach(this.pendingBeforeInit_, function(module) {
233 this.load_(module);
234 }, this);
235 goog.array.clear(this.pendingBeforeInit_);
236};

239/**
240 * Requests the loading of a symbol from a module. When the module is
241 * loaded, the requested symbol will be passed as argument to the
242 * function callback.
243 *
244 * @param {string} module The name of the module. Usually, the value
245 * is defined as a constant whose name starts with MOD_.
246 * @param {number|string} symbol The ID of the symbol. Usually, the value is
247 * defined as a constant whose name starts with SYM_.
248 * @param {Function} callback This function will be called with the
249 * resolved symbol as the argument once the module is loaded.
250 */
251goog.module.Loader.prototype.require = function(module, symbol, callback) {
252 var pending = this.pending_;
253 var modules = this.modules_;
254 if (modules[module]) {
255 // already loaded
256 callback(modules[module][symbol]);
257 } else if (pending[module]) {
258 // loading is pending from another require of the same module
259 pending[module].push([symbol, callback]);
260 } else {
261 // not loaded, and not requested
262 pending[module] = [[symbol, callback]]; // Yes, really [[ ]].
263 // Defer loading to initialization if Loader is not yet
264 // initialized, otherwise load the module.
265 if (goog.isString(this.urlBase_)) {
266 this.load_(module);
267 } else {
268 this.pendingBeforeInit_.push(module);
269 }
270 }
271};

274/**
275 * Registers a symbol in a loaded module. When called without symbol,
276 * registers the module to be fully loaded and executes all callbacks
277 * from pending require() callbacks for this module.
278 *
279 * @param {string} module The name of the module. Cf. parameter module
280 * of method require().
281 * @param {number|string=} opt_symbol The symbol being defined, or nothing when
282 * all symbols of the module are defined. Cf. parameter symbol of method
283 * require().
284 * @param {Object=} opt_object The object bound to the symbol, or nothing when
285 * all symbols of the module are defined.
286 */
287goog.module.Loader.prototype.provide = function(
288 module, opt_symbol, opt_object) {
289 var modules = this.modules_;
290 var pending = this.pending_;
291 if (!modules[module]) {
292 modules[module] = {};
293 }
294 if (opt_object) {
295 // When an object is provided, just register it.
296 modules[module][opt_symbol] = opt_object;
297 } else if (pending[module]) {
298 // When no object is provided, and there are pending require()
299 // callbacks for this module, execute them.
300 for (var i = 0; i < pending[module].length; ++i) {
301 var symbol = pending[module][i][0];
302 var callback = pending[module][i][1];
303 callback(modules[module][symbol]);
304 }
305 delete pending[module];
306 delete this.pendingModuleUrls_[module];
307 }
308};

311/**
312 * Starts to load a module. Assumes that init() was called.
313 *
314 * @param {string} module The name of the module.
315 * @private
316 */
317goog.module.Loader.prototype.load_ = function(module) {
318 // NOTE(user): If the module request happens inside a click handler
319 // (presumably inside any user event handler, but the onload event
320 // handler is fine), IE will load the script but not execute
321 // it. Thus we break out of the current flow of control before we do
322 // the load. For the record, for IE it would have been enough to
323 // just defer the assignment to src. Safari doesn't execute the
324 // script if the assignment to src happens *after* the script
325 // element is inserted into the DOM.
326 goog.Timer.callOnce(function() {
327 // The module might have been registered in the interim (if fetched as part
328 // of another module fetch because they share the same url)
329 if (this.modules_[module]) {
330 return;
331 }
332
333 goog.asserts.assertString(this.urlBase_);
334 var url = this.getModuleUrl_(this.urlBase_, module);
335
336 // Check if specified URL is already in flight
337 var urlInFlight = goog.object.containsValue(this.pendingModuleUrls_, url);
338 this.pendingModuleUrls_[module] = url;
339 if (urlInFlight) {
340 return;
341 }
342
343 var s = goog.dom.createDom(goog.dom.TagName.SCRIPT,
344 {'type': 'text/javascript', 'src': url});
345 document.body.appendChild(s);
346 }, 0, this);
347};
