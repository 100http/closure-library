40/**
41 * The ModuleManager keeps track of all modules in the environment.
42 * Since modules may not have their code loaded, we must keep track of them.
43 * @constructor
44 * @extends {goog.Disposable}
45 * @struct
46 */
47goog.module.ModuleManager = function() {
48 goog.module.ModuleManager.base(this, 'constructor');
49
50 /**
51 * A mapping from module id to ModuleInfo object.
52 * @private {Object<string, !goog.module.ModuleInfo>}
53 */
54 this.moduleInfoMap_ = {};
55
56 // TODO (malteubl): Switch this to a reentrant design.
57 /**
58 * The ids of the currently loading modules. If batch mode is disabled, then
59 * this array will never contain more than one element at a time.
60 * @type {Array<string>}
61 * @private
62 */
63 this.loadingModuleIds_ = [];
64
65 /**
66 * The requested ids of the currently loading modules. This does not include
67 * module dependencies that may also be loading.
68 * @type {Array<string>}
69 * @private
70 */
71 this.requestedLoadingModuleIds_ = [];
72
73 // TODO(user): Make these and other arrays that are used as sets be
74 // actual sets.
75 /**
76 * All module ids that have ever been requested. In concurrent loading these
77 * are the ones to subtract from future requests.
78 * @type {!Array<string>}
79 * @private
80 */
81 this.requestedModuleIds_ = [];
82
83 /**
84 * A queue of the ids of requested but not-yet-loaded modules. The zero
85 * position is the front of the queue. This is a 2-D array to group modules
86 * together with other modules that should be batch loaded with them, if
87 * batch loading is enabled.
88 * @type {Array<Array<string>>}
89 * @private
90 */
91 this.requestedModuleIdsQueue_ = [];
92
93 /**
94 * The ids of the currently loading modules which have been initiated by user
95 * actions.
96 * @type {Array<string>}
97 * @private
98 */
99 this.userInitiatedLoadingModuleIds_ = [];
100
101 /**
102 * A map of callback types to the functions to call for the specified
103 * callback type.
104 * @type {Object<goog.module.ModuleManager.CallbackType, Array<Function>>}
105 * @private
106 */
107 this.callbackMap_ = {};
108
109 /**
110 * Module info for the base module (the one that contains the module
111 * manager code), which we set as the loading module so one can
112 * register initialization callbacks in the base module.
113 *
114 * The base module is considered loaded when #setAllModuleInfo is called or
115 * #setModuleContext is called, whichever comes first.
116 *
117 * @type {goog.module.ModuleInfo}
118 * @private
119 */
120 this.baseModuleInfo_ = new goog.module.ModuleInfo([], '');
121
122 /**
123 * The module that is currently loading, or null if not loading anything.
124 * @type {goog.module.ModuleInfo}
125 * @private
126 */
127 this.currentlyLoadingModule_ = this.baseModuleInfo_;
128
129 /**
130 * The id of the last requested initial module. When it loaded
131 * the deferred in {@code this.initialModulesLoaded_} resolves.
132 * @private {?string}
133 */
134 this.lastInitialModuleId_ = null;
135
136 /**
137 * Deferred for when all initial modules have loaded. We currently block

141 * requests even sooner.
142 * @private {!goog.async.Deferred}

144 this.initialModulesLoaded_ = new goog.async.Deferred();
145
146 /**
147 * A logger.
148 * @private {goog.log.Logger}
149 */
150 this.logger_ = goog.log.getLogger('goog.module.ModuleManager');
151
152 /**
153 * Whether the batch mode (i.e. the loading of multiple modules with just one
154 * request) has been enabled.
155 * @private {boolean}

157 this.batchModeEnabled_ = false;
158
159 /**
160 * Whether the module requests may be sent out of order.
161 * @private {boolean}
162 */
163 this.concurrentLoadingEnabled_ = false;
164
165 /**
166 * A loader for the modules that implements loadModules(ids, moduleInfoMap,
167 * opt_successFn, opt_errorFn, opt_timeoutFn, opt_forceReload) method.
168 * @private {goog.module.AbstractModuleLoader}

170 this.loader_ = null;
171
172 // TODO(user): Remove tracer.
173 /**
174 * Tracer that measures how long it takes to load a module.
175 * @private {?number}
176 */
177 this.loadTracer_ = null;
178
179 /**
180 * The number of consecutive failures that have happened upon module load
181 * requests.
182 * @private {number}
183 */
184 this.consecutiveFailures_ = 0;
185
186 /**
187 * Determines if the module manager was just active before the processing of
188 * the last data.
189 * @private {boolean}
190 */
191 this.lastActive_ = false;
192
193 /**
194 * Determines if the module manager was just user active before the processing

197 * @private {boolean}

199 this.userLastActive_ = false;
200
201 /**
202 * The module context needed for module initialization.
203 * @private {Object}
204 */
205 this.moduleContext_ = null;

296/**
297 * Sets the module info for all modules. Should only be called once. Also
298 * marks modules that are currently being loaded.
299 *
300 * @param {string=} opt_info A string representation of the module dependency
301 * graph, in the form: module1:dep1,dep2/module2:dep1,dep2 etc.
302 * Where depX is the base-36 encoded position of the dep in the module list.
303 * @param {Array<string>=} opt_loadingModuleIds A list of moduleIds that
304 * are currently being loaded.
305 */
306goog.module.ModuleManager.prototype.setAllModuleInfoString = function(
307 opt_info, opt_loadingModuleIds) {
308 if (!goog.isString(opt_info)) {
309 // The call to this method is generated in two steps, the argument is added
310 // after some of the compilation passes. This means that the initial code
311 // doesn't have any arguments and causes compiler errors. We make it
312 // optional to satisfy this constraint.
313 return;
314 }
315
316 var modules = opt_info.split('/');
317 var moduleIds = [];
318
319 // Split the string into the infoMap of id->deps
320 for (var i = 0; i < modules.length; i++) {
321 var parts = modules[i].split(':');
322 var id = parts[0];
323 var deps;
324 if (parts[1]) {
325 deps = parts[1].split(',');
326 for (var j = 0; j < deps.length; j++) {
327 var index = parseInt(deps[j], 36);
328 goog.asserts.assert(
329 moduleIds[index], 'No module @ %s, dep of %s @ %s', index, id, i);
330 deps[j] = moduleIds[index];
331 }
332 } else {
333 deps = [];
334 }
335 moduleIds.push(id);
336 this.moduleInfoMap_[id] = new goog.module.ModuleInfo(deps, id);
337 }
338 if (opt_loadingModuleIds && opt_loadingModuleIds.length) {
339 goog.array.extend(this.loadingModuleIds_, opt_loadingModuleIds);
340 // The last module in the list of initial modules. When it has loaded all
341 // initial modules have loaded.
342 this.lastInitialModuleId_ = /** @type {?string} */ (
343 goog.array.peek(opt_loadingModuleIds));
344 } else {
345 if (!this.initialModulesLoaded_.hasFired()) {
346 this.initialModulesLoaded_.callback();
347 }
348 }
349 this.maybeFinishBaseLoad_();
350};

434/**
435 * Dispatches an ACTIVE or IDLE event if necessary.
436 * @private
437 */
438goog.module.ModuleManager.prototype.dispatchActiveIdleChangeIfNeeded_ =
439 function() {
440 var lastActive = this.lastActive_;
441 var active = this.isActive();
442 if (active != lastActive) {
443 this.executeCallbacks_(active ?
444 goog.module.ModuleManager.CallbackType.ACTIVE :
445 goog.module.ModuleManager.CallbackType.IDLE);
446
447 // Flip the last active value.
448 this.lastActive_ = active;
449 }
450
451 // Check if the module manager is user active i.e., there are user initiated
452 // modules being loaded or queued up to be loaded.
453 var userLastActive = this.userLastActive_;
454 var userActive = this.isUserActive();
455 if (userActive != userLastActive) {
456 this.executeCallbacks_(userActive ?
457 goog.module.ModuleManager.CallbackType.USER_ACTIVE :
458 goog.module.ModuleManager.CallbackType.USER_IDLE);
459
460 // Flip the last user active value.
461 this.userLastActive_ = userActive;
462 }

466/**
467 * Preloads a module after a short delay.
468 *
469 * @param {string} id The id of the module to preload.
470 * @param {number=} opt_timeout The number of ms to wait before adding the
471 * module id to the loading queue (defaults to 0 ms). Note that the module
472 * will be loaded asynchronously regardless of the value of this parameter.
473 * @return {!goog.async.Deferred} A deferred object.
474 */
475goog.module.ModuleManager.prototype.preloadModule = function(
476 id, opt_timeout) {
477 var d = new goog.async.Deferred();
478 window.setTimeout(
479 goog.bind(this.addLoadModule_, this, id, d),
480 opt_timeout || 0);
481 return d;
482};

485/**
486 * Prefetches a JavaScript module and its dependencies, which means that the
487 * module will be downloaded, but not evaluated. To complete the module load,
488 * the caller should also call load or execOnLoad after prefetching the module.
489 *
490 * @param {string} id The id of the module to prefetch.
491 */
492goog.module.ModuleManager.prototype.prefetchModule = function(id) {
493 var moduleInfo = this.getModuleInfo(id);
494 if (moduleInfo.isLoaded() || this.isModuleLoading(id)) {
495 throw Error('Module load already requested: ' + id);
496 } else if (this.batchModeEnabled_) {
497 throw Error('Modules prefetching is not supported in batch mode');
498 } else {
499 var idWithDeps = this.getNotYetLoadedTransitiveDepIds_(id);
500 for (var i = 0; i < idWithDeps.length; i++) {
501 this.loader_.prefetchModule(idWithDeps[i],
502 this.moduleInfoMap_[idWithDeps[i]]);
503 }
504 }
505};

508/**
509 * Loads a single module for use with a given deferred.
510 *
511 * @param {string} id The id of the module to load.
512 * @param {goog.async.Deferred} d A deferred object.
513 * @private
514 */
515goog.module.ModuleManager.prototype.addLoadModule_ = function(id, d) {
516 var moduleInfo = this.getModuleInfo(id);
517 if (moduleInfo.isLoaded()) {
518 d.callback(this.moduleContext_);
519 return;
520 }
521
522 this.registerModuleLoadCallbacks_(id, moduleInfo, false, d);
523 if (!this.isModuleLoading(id)) {
524 this.loadModulesOrEnqueue_([id]);
525 }
526};

529/**
530 * Loads a list of modules or, if some other module is currently being loaded,
531 * appends the ids to the queue of requested module ids. Registers callbacks a
532 * module that is currently loading and returns a fired deferred for a module
533 * that is already loaded.
534 *
535 * @param {Array<string>} ids The id of the module to load.
536 * @param {boolean=} opt_userInitiated If the load is a result of a user action.
537 * @return {!Object<string, !goog.async.Deferred>} A mapping from id (String)
538 * to deferred objects that will callback or errback when the load for that
539 * id is finished.
540 * @private
541 */
542goog.module.ModuleManager.prototype.loadModulesOrEnqueueIfNotLoadedOrLoading_ =
543 function(ids, opt_userInitiated) {
544 var uniqueIds = [];
545 goog.array.removeDuplicates(ids, uniqueIds);
546 var idsToLoad = [];
547 var deferredMap = {};
548 for (var i = 0; i < uniqueIds.length; i++) {
549 var id = uniqueIds[i];
550 var moduleInfo = this.getModuleInfo(id);
551 if (!moduleInfo) {
552 throw new Error('Unknown module: ' + id);
553 }
554 var d = new goog.async.Deferred();
555 deferredMap[id] = d;
556 if (moduleInfo.isLoaded()) {
557 d.callback(this.moduleContext_);
558 } else {
559 this.registerModuleLoadCallbacks_(id, moduleInfo, !!opt_userInitiated, d);
560 if (!this.isModuleLoading(id)) {
561 idsToLoad.push(id);
562 }
563 }
564 }
565
566 // If there are ids to load, load them, otherwise, they are all loading or
567 // loaded.
568 if (idsToLoad.length > 0) {
569 this.loadModulesOrEnqueue_(idsToLoad);
570 }
571 return deferredMap;
572};

575/**
576 * Registers the callbacks and handles logic if it is a user initiated module
577 * load.
578 *
579 * @param {string} id The id of the module to possibly load.
580 * @param {!goog.module.ModuleInfo} moduleInfo The module identifier for the
581 * given id.
582 * @param {boolean} userInitiated If the load was user initiated.
583 * @param {goog.async.Deferred} d A deferred object.
584 * @private
585 */
586goog.module.ModuleManager.prototype.registerModuleLoadCallbacks_ =
587 function(id, moduleInfo, userInitiated, d) {
588 moduleInfo.registerCallback(d.callback, d);
589 moduleInfo.registerErrback(function(err) { d.errback(Error(err)); });
590 // If it's already loading, we don't have to do anything besides handle
591 // if it was user initiated
592 if (this.isModuleLoading(id)) {
593 if (userInitiated) {
594 goog.log.info(this.logger_,
595 'User initiated module already loading: ' + id);
596 this.addUserInitiatedLoadingModule_(id);
597 this.dispatchActiveIdleChangeIfNeeded_();
598 }
599 } else {
600 if (userInitiated) {
601 goog.log.info(this.logger_, 'User initiated module load: ' + id);
602 this.addUserInitiatedLoadingModule_(id);
603 } else {
604 goog.log.info(this.logger_, 'Initiating module load: ' + id);
605 }
606 }
607};

610/**
611 * Initiates loading of a list of modules or, if a module is currently being
612 * loaded, appends the modules to the queue of requested module ids.
613 *
614 * The caller should verify that the requested modules are not already loaded or
615 * loading. {@link #loadModulesOrEnqueueIfNotLoadedOrLoading_} is a more lenient
616 * alternative to this method.
617 *
618 * @param {Array<string>} ids The ids of the modules to load.
619 * @private
620 */
621goog.module.ModuleManager.prototype.loadModulesOrEnqueue_ = function(ids) {
622 // With concurrent loading we always just send off the request.
623 if (this.concurrentLoadingEnabled_) {
624 // For now we wait for initial modules to have downloaded as this puts the
625 // loader in a good state for calculating the needed deps of additional
626 // loads.
627 // TODO(user): Make this wait unnecessary.
628 this.initialModulesLoaded_.addCallback(
629 goog.bind(this.loadModules_, this, ids));
630 } else {
631 if (goog.array.isEmpty(this.loadingModuleIds_)) {
632 this.loadModules_(ids);
633 } else {
634 this.requestedModuleIdsQueue_.push(ids);
635 this.dispatchActiveIdleChangeIfNeeded_();
636 }
637 }
638};

641/**
642 * Gets the amount of delay to wait before sending a request for more modules.
643 * If a certain module request fails, we backoff a little bit and try again.
644 * @return {number} Delay, in ms.
645 * @private
646 */
647goog.module.ModuleManager.prototype.getBackOff_ = function() {
648 // 5 seconds after one error, 20 seconds after 2.
649 return Math.pow(this.consecutiveFailures_, 2) * 5000;
650};

653/**
654 * Loads a list of modules and any of their not-yet-loaded prerequisites.
655 * If batch mode is enabled, the prerequisites will be loaded together with the
656 * requested modules and all requested modules will be loaded at the same time.
657 *
658 * The caller should verify that the requested modules are not already loaded
659 * and that no modules are currently loading before calling this method.
660 *
661 * @param {Array<string>} ids The ids of the modules to load.
662 * @param {boolean=} opt_isRetry If the load is a retry of a previous load
663 * attempt.
664 * @param {boolean=} opt_forceReload Whether to bypass cache while loading the
665 * module.
666 * @private
667 */
668goog.module.ModuleManager.prototype.loadModules_ = function(
669 ids, opt_isRetry, opt_forceReload) {
670 if (!opt_isRetry) {
671 this.consecutiveFailures_ = 0;
672 }
673
674 // Not all modules may be loaded immediately if batch mode is not enabled.
675 var idsToLoadImmediately = this.processModulesForLoad_(ids);
676
677 goog.log.info(this.logger_, 'Loading module(s): ' + idsToLoadImmediately);
678 this.loadingModuleIds_ = idsToLoadImmediately;
679
680 if (this.batchModeEnabled_) {
681 this.requestedLoadingModuleIds_ = ids;
682 } else {
683 // If batch mode is disabled, we treat each dependency load as a separate
684 // load.
685 this.requestedLoadingModuleIds_ = goog.array.clone(idsToLoadImmediately);
686 }
687
688 // Dispatch an active/idle change if needed.
689 this.dispatchActiveIdleChangeIfNeeded_();
690
691 if (goog.array.isEmpty(idsToLoadImmediately)) {
692 // All requested modules and deps have been either loaded already or have
693 // already been requested.
694 return;
695 }
696
697 this.requestedModuleIds_.push.apply(this.requestedModuleIds_,
698 idsToLoadImmediately);
699
700 var loadFn = goog.bind(this.loader_.loadModules, this.loader_,
701 goog.array.clone(idsToLoadImmediately),
702 this.moduleInfoMap_,
703 null,
704 goog.bind(this.handleLoadError_, this, this.requestedLoadingModuleIds_,
705 idsToLoadImmediately),
706 goog.bind(this.handleLoadTimeout_, this),
707 !!opt_forceReload);
708
709 var delay = this.getBackOff_();
710 if (delay) {
711 window.setTimeout(loadFn, delay);
712 } else {
713 loadFn();
714 }
715};

718/**
719 * Processes a list of module ids for loading. Checks if any of the modules are
720 * already loaded and then gets transitive deps. Queues any necessary modules
721 * if batch mode is not enabled. Returns the list of ids that should be loaded.
722 *
723 * @param {Array<string>} ids The ids that need to be loaded.
724 * @return {!Array<string>} The ids to load, including dependencies.
725 * @throws {Error} If the module is already loaded.
726 * @private
727 */
728goog.module.ModuleManager.prototype.processModulesForLoad_ = function(ids) {
729 for (var i = 0; i < ids.length; i++) {
730 var moduleInfo = this.moduleInfoMap_[ids[i]];
731 if (moduleInfo.isLoaded()) {
732 throw Error('Module already loaded: ' + ids[i]);
733 }
734 }
735
736 // Build a list of the ids of this module and any of its not-yet-loaded
737 // prerequisite modules in dependency order.
738 var idsWithDeps = [];
739 for (var i = 0; i < ids.length; i++) {
740 idsWithDeps = idsWithDeps.concat(
741 this.getNotYetLoadedTransitiveDepIds_(ids[i]));
742 }
743 goog.array.removeDuplicates(idsWithDeps);
744
745 if (!this.batchModeEnabled_ && idsWithDeps.length > 1) {
746 var idToLoad = idsWithDeps.shift();
747 goog.log.info(this.logger_,
748 'Must load ' + idToLoad + ' module before ' + ids);
749
750 // Insert the requested module id and any other not-yet-loaded prereqs
751 // that it has at the front of the queue.
752 var queuedModules = goog.array.map(idsWithDeps, function(id) {
753 return [id];
754 });
755 this.requestedModuleIdsQueue_ = queuedModules.concat(

764/**
765 * Builds a list of the ids of the not-yet-loaded modules that a particular
766 * module transitively depends on, including itself.
767 *
768 * @param {string} id The id of a not-yet-loaded module.
769 * @return {!Array<string>} An array of module ids in dependency order that's
770 * guaranteed to end with the provided module id.
771 * @private
772 */
773goog.module.ModuleManager.prototype.getNotYetLoadedTransitiveDepIds_ =
774 function(id) {
775 // NOTE(user): We want the earliest occurrance of a module, not the first
776 // dependency we find. Therefore we strip duplicates at the end rather than
777 // during. See the tests for concrete examples.
778 var ids = [];
779 if (!goog.array.contains(this.requestedModuleIds_, id)) {
780 ids.push(id);
781 }
782 var depIds = goog.array.clone(this.getModuleInfo(id).getDependencies());
783 while (depIds.length) {
784 var depId = depIds.pop();
785 if (!this.getModuleInfo(depId).isLoaded() &&
786 !goog.array.contains(this.requestedModuleIds_, depId)) {
787 ids.unshift(depId);
788 // We need to process direct dependencies first.
789 Array.prototype.unshift.apply(depIds,
790 this.getModuleInfo(depId).getDependencies());
791 }
792 }
793 goog.array.removeDuplicates(ids);
794 return ids;

882/**
883 * Requests that a function be called once a particular module is loaded.
884 * Client code can use this method to safely call into modules that may not yet
885 * be loaded. For consistency, this method always calls the function
886 * asynchronously -- even if the module is already loaded. Initiates loading of
887 * the module if necessary, unless opt_noLoad is true.
888 *
889 * @param {string} moduleId A module id.
890 * @param {Function} fn Function to execute when the module has loaded.
891 * @param {Object=} opt_handler Optional handler under whose scope to execute
892 * the callback.
893 * @param {boolean=} opt_noLoad TRUE iff not to initiate loading of the module.
894 * @param {boolean=} opt_userInitiated TRUE iff the loading of the module was
895 * user initiated.
896 * @param {boolean=} opt_preferSynchronous TRUE iff the function should be
897 * executed synchronously if the module has already been loaded.
898 * @return {!goog.module.ModuleLoadCallback} A callback wrapper that exposes
899 * an abort and execute method.
900 */
901goog.module.ModuleManager.prototype.execOnLoad = function(
902 moduleId, fn, opt_handler, opt_noLoad,
903 opt_userInitiated, opt_preferSynchronous) {
904 var moduleInfo = this.moduleInfoMap_[moduleId];
905 var callbackWrapper;
906
907 if (moduleInfo.isLoaded()) {
908 goog.log.info(this.logger_, moduleId + ' module already loaded');
909 // Call async so that code paths don't change between loaded and unloaded
910 // cases.
911 callbackWrapper = new goog.module.ModuleLoadCallback(fn, opt_handler);
912 if (opt_preferSynchronous) {
913 callbackWrapper.execute(this.moduleContext_);
914 } else {
915 window.setTimeout(
916 goog.bind(callbackWrapper.execute, callbackWrapper), 0);
917 }
918 } else if (this.isModuleLoading(moduleId)) {
919 goog.log.info(this.logger_, moduleId + ' module already loading');
920 callbackWrapper = moduleInfo.registerCallback(fn, opt_handler);
921 if (opt_userInitiated) {
922 goog.log.info(this.logger_,
923 'User initiated module already loading: ' + moduleId);
924 this.addUserInitiatedLoadingModule_(moduleId);
925 this.dispatchActiveIdleChangeIfNeeded_();
926 }
927 } else {
928 goog.log.info(this.logger_,
929 'Registering callback for module: ' + moduleId);
930 callbackWrapper = moduleInfo.registerCallback(fn, opt_handler);
931 if (!opt_noLoad) {
932 if (opt_userInitiated) {
933 goog.log.info(this.logger_, 'User initiated module load: ' + moduleId);
934 this.addUserInitiatedLoadingModule_(moduleId);
935 }
936 goog.log.info(this.logger_, 'Initiating module load: ' + moduleId);
937 this.loadModulesOrEnqueue_([moduleId]);
938 }
939 }
940 return callbackWrapper;

1113/**
1114 * Handles a module load failure.
1115 *
1116 * @param {!Array<string>} requestedLoadingModuleIds Modules ids that were
1117 * requested in failed request. Does not included calculated dependencies.
1118 * @param {!Array<string>} requestedModuleIdsWithDeps All module ids requested
1119 * in the failed request including all dependencies.
1120 * @param {?number} status The error status.
1121 * @private
1122 */
1123goog.module.ModuleManager.prototype.handleLoadError_ = function(
1124 requestedLoadingModuleIds, requestedModuleIdsWithDeps, status) {
1125 this.consecutiveFailures_++;
1126 // Module manager was not designed to be reentrant. Reinstate the instance
1127 // var with actual value when request failed (Other requests may have
1128 // started already.)
1129 this.requestedLoadingModuleIds_ = requestedLoadingModuleIds;
1130 // Pretend we never requested the failed modules.
1131 goog.array.forEach(requestedModuleIdsWithDeps,
1132 goog.partial(goog.array.remove, this.requestedModuleIds_), this);
1133
1134 if (status == 401) {
1135 // The user is not logged in. They've cleared their cookies or logged out
1136 // from another window.
1137 goog.log.info(this.logger_, 'Module loading unauthorized');
1138 this.dispatchModuleLoadFailed_(
1139 goog.module.ModuleManager.FailureType.UNAUTHORIZED);
1140 // Drop any additional module requests.
1141 this.requestedModuleIdsQueue_.length = 0;
1142 } else if (status == 410) {
1143 // The requested module js is old and not available.
1144 this.requeueBatchOrDispatchFailure_(
1145 goog.module.ModuleManager.FailureType.OLD_CODE_GONE);
1146 this.loadNextModules_();
1147 } else if (this.consecutiveFailures_ >= 3) {
1148 goog.log.info(this.logger_, 'Aborting after failure to load: ' +
1149 this.loadingModuleIds_);
1150 this.requeueBatchOrDispatchFailure_(
1151 goog.module.ModuleManager.FailureType.CONSECUTIVE_FAILURES);
1152 this.loadNextModules_();
1153 } else {
1154 goog.log.info(this.logger_, 'Retrying after failure to load: ' +
1155 this.loadingModuleIds_);
1156 var forceReload =
1157 status == goog.module.ModuleManager.CORRUPT_RESPONSE_STATUS_CODE;
1158 this.loadModules_(this.requestedLoadingModuleIds_, true, forceReload);
1159 }
1160};

1203/**
1204 * Handles when a module load failed.
1205 * @param {goog.module.ModuleManager.FailureType} cause The reason for the
1206 * failure.
1207 * @private
1208 */
1209goog.module.ModuleManager.prototype.dispatchModuleLoadFailed_ = function(
1210 cause) {
1211 var failedIds = this.requestedLoadingModuleIds_;
1212 this.loadingModuleIds_.length = 0;
1213 // If any pending modules depend on the id that failed,
1214 // they need to be removed from the queue.
1215 var idsToCancel = [];
1216 for (var i = 0; i < this.requestedModuleIdsQueue_.length; i++) {
1217 var dependentModules = goog.array.filter(
1218 this.requestedModuleIdsQueue_[i],
1219 /**
1220 * Returns true if the requestedId has dependencies on the modules that
1221 * just failed to load.
1222 * @param {string} requestedId The module to check for dependencies.
1223 * @return {boolean} True if the module depends on failed modules.
1224 */
1225 function(requestedId) {
1226 var requestedDeps = this.getNotYetLoadedTransitiveDepIds_(
1227 requestedId);
1228 return goog.array.some(failedIds, function(id) {
1229 return goog.array.contains(requestedDeps, id);
1230 });
1231 }, this);
1232 goog.array.extend(idsToCancel, dependentModules);
1233 }
1234
1235 // Also insert the ids that failed to load as ids to cancel.
1236 for (var i = 0; i < failedIds.length; i++) {
1237 goog.array.insert(idsToCancel, failedIds[i]);
1238 }
1239
1240 // Remove ids to cancel from the queues.
1241 for (var i = 0; i < idsToCancel.length; i++) {
1242 for (var j = 0; j < this.requestedModuleIdsQueue_.length; j++) {
1243 goog.array.remove(this.requestedModuleIdsQueue_[j], idsToCancel[i]);
1244 }
1245 goog.array.remove(this.userInitiatedLoadingModuleIds_, idsToCancel[i]);
1246 }
1247
1248 // Call the functions for error notification.
1249 var errorCallbacks = this.callbackMap_[
1250 goog.module.ModuleManager.CallbackType.ERROR];
1251 if (errorCallbacks) {
1252 for (var i = 0; i < errorCallbacks.length; i++) {
1253 var callback = errorCallbacks[i];
1254 for (var j = 0; j < idsToCancel.length; j++) {
1255 callback(goog.module.ModuleManager.CallbackType.ERROR, idsToCancel[j],
1256 cause);
1257 }
1258 }
1259 }
1260
1261 // Call the errbacks on the module info.
1262 for (var i = 0; i < failedIds.length; i++) {
1263 if (this.moduleInfoMap_[failedIds[i]]) {
1264 this.moduleInfoMap_[failedIds[i]].onError(cause);
1265 }
1266 }
1267
1268 // Clear the requested loading module ids.
1269 this.requestedLoadingModuleIds_.length = 0;
1270
1271 this.dispatchActiveIdleChangeIfNeeded_();
1272};
